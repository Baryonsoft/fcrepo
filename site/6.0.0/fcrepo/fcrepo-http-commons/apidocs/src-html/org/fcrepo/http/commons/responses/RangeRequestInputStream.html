018package org.fcrepo.http.commons.responses;
019
020import java.io.IOException;
021import java.io.InputStream;
022
023import org.apache.commons.io.input.BoundedInputStream;
024
025/**
026 * An {@link InputStream} that skips bytes and only returns the data up to a certain limit
027 *
028 * @author awoods
029 * @author ajs6f
030 */
031public class RangeRequestInputStream extends BoundedInputStream {
032
033    /**
034     * @param in the underlying input stream, or <code>null</code> if
035     *           this instance is to be created without an underlying stream.
036     * @param skip the number of bytes to skip at the beginning of the stream
037     * @param length the number of bytes from the inputstream to read
038     * @throws IOException if IO exception occurred
039     */
040    public RangeRequestInputStream(final InputStream in, final long skip, final long length) throws IOException {
041        super(in, length);
042        in.skip(skip);
043    }
044}
