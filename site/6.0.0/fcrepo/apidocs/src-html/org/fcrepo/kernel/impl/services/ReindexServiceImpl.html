018package org.fcrepo.kernel.impl.services;
019
020import org.fcrepo.kernel.api.Transaction;
021import org.fcrepo.kernel.api.TransactionManager;
022import org.fcrepo.kernel.api.identifiers.FedoraId;
023import org.fcrepo.kernel.api.operations.ReindexResourceOperationFactory;
024import org.fcrepo.kernel.api.services.ReindexService;
025import org.fcrepo.persistence.api.PersistentStorageSessionManager;
026import org.springframework.stereotype.Component;
027
028import javax.inject.Inject;
029
030/**
031 * Implementation of {@link org.fcrepo.kernel.api.services.ReindexService}
032 *
033 * @author dbernstein
034 */
035@Component
036public class ReindexServiceImpl extends AbstractService implements ReindexService {
037
038    @Inject
039    private TransactionManager transactionManager;
040
041    @Inject
042    private PersistentStorageSessionManager persistentStorageSessionManager;
043
044    @Inject
045    private ReindexResourceOperationFactory resourceOperationFactory;
046
047    @Override
048    public void reindexByFedoraId(final Transaction transaction, final String principal, final FedoraId fedoraId) {
049        final var tx = transactionManager.get(transaction.getId());
050        final var psession = persistentStorageSessionManager.getSession(transaction);
051        final var operation = resourceOperationFactory.create(transaction, fedoraId)
052                .userPrincipal(principal).build();
053        tx.lockResource(fedoraId);
054
055        psession.persist(operation);
056    }
057}
