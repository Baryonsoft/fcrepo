019package org.fcrepo.kernel.impl.services;
020
021import io.micrometer.core.instrument.Metrics;
022import io.micrometer.core.instrument.Timer;
023import org.fcrepo.common.metrics.MetricsHelper;
024import org.fcrepo.kernel.api.RdfStream;
025import org.fcrepo.kernel.api.Transaction;
026import org.fcrepo.kernel.api.identifiers.FedoraId;
027import org.fcrepo.kernel.api.models.FedoraResource;
028import org.fcrepo.kernel.api.services.ReferenceService;
029import org.springframework.beans.factory.annotation.Autowired;
030import org.springframework.beans.factory.annotation.Qualifier;
031import org.springframework.stereotype.Component;
032
033/**
034 * ReferenceService wrapper for collecting metrics
035 *
036 * @author pwinckles
037 */
038@Component("referenceService")
039public class ReferenceServiceMetrics implements ReferenceService {
040
041    private static final String METRIC_NAME = "fcrepo.db";
042    private static final String DB = "db";
043    private static final String REFERENCE = "reference";
044    private static final String OPERATION = "operation";
045
046    private static final Timer getInboundReferences = Metrics.timer(METRIC_NAME,
047            DB, REFERENCE, OPERATION, "getInboundReferences");
048    private static final Timer deleteAllReferencesTimer = Metrics.timer(METRIC_NAME,
049            DB, REFERENCE, OPERATION, "deleteAllReferences");
050    private static final Timer updateReferencesTimer = Metrics.timer(METRIC_NAME,
051            DB, REFERENCE, OPERATION, "updateReferences");
052    private static final Timer commitTransactionTimer = Metrics.timer(METRIC_NAME,
053            DB, REFERENCE, OPERATION, "commitTransaction");
054    private static final Timer rollbackTransactionTimer = Metrics.timer(METRIC_NAME,
055            DB, REFERENCE, OPERATION, "rollbackTransaction");
056    private static final Timer resetTimer = Metrics.timer(METRIC_NAME,
057            DB, REFERENCE, OPERATION, "reset");
058
059    @Autowired
060    @Qualifier("referenceServiceImpl")
061    private ReferenceService referenceServiceImpl;
062
063    @Override
064    public RdfStream getInboundReferences(final Transaction tx, final FedoraResource resource) {
065        return MetricsHelper.time(getInboundReferences, () -> {
066            return referenceServiceImpl.getInboundReferences(tx, resource);
067        });
068    }
069
070    @Override
071    public void deleteAllReferences(final Transaction tx, final FedoraId resourceId) {
072        deleteAllReferencesTimer.record(() -> {
073            referenceServiceImpl.deleteAllReferences(tx, resourceId);
074        });
075    }
076
077    @Override
078    public void updateReferences(final Transaction tx,
079                                 final FedoraId resourceId,
080                                 final String userPrincipal,
081                                 final RdfStream rdfStream) {
082        updateReferencesTimer.record(() -> {
083            referenceServiceImpl.updateReferences(tx, resourceId, userPrincipal, rdfStream);
084        });
085    }
086
087    @Override
088    public void commitTransaction(final Transaction tx) {
089        commitTransactionTimer.record(() -> {
090            referenceServiceImpl.commitTransaction(tx);
091        });
092    }
093
094    @Override
095    public void rollbackTransaction(final Transaction tx) {
096        rollbackTransactionTimer.record(() -> {
097            referenceServiceImpl.rollbackTransaction(tx);
098        });
099    }
100
101    @Override
102    public void reset() {
103        resetTimer.record(() -> {
104            referenceServiceImpl.reset();
105        });
106    }
107
108}
