18  package org.fcrepo.kernel.impl.services;
19  
20  import com.google.common.annotations.VisibleForTesting;
21  
22  import org.fcrepo.kernel.api.RdfLexicon;
23  import org.fcrepo.kernel.api.Transaction;
24  import org.fcrepo.kernel.api.exception.RepositoryRuntimeException;
25  import org.fcrepo.kernel.api.identifiers.FedoraId;
26  import org.fcrepo.kernel.api.operations.VersionResourceOperationFactory;
27  import org.fcrepo.kernel.api.services.VersionService;
28  import org.fcrepo.persistence.api.PersistentStorageSessionManager;
29  import org.fcrepo.persistence.api.exceptions.PersistentStorageException;
30  import org.springframework.stereotype.Component;
31  
32  import javax.inject.Inject;
33  
34  /**
35   * Implementation of {@link VersionService}
36   *
37   * @author dbernstein
38   */
39  @Component
40  public class VersionServiceImpl extends AbstractService implements VersionService {
41  
42      @Inject
43      private PersistentStorageSessionManager psManager;
44  
45      @Inject
46      private VersionResourceOperationFactory versionOperationFactory;
47  
48      @Override
49      public void createVersion(final Transaction transaction, final FedoraId fedoraId, final String userPrincipal) {
50          final var session = psManager.getSession(transaction);
51          final var operation = versionOperationFactory.createBuilder(transaction, fedoraId)
52                  .userPrincipal(userPrincipal)
53                  .build();
54  
55          lockArchivalGroupResource(transaction, session, fedoraId);
56          final var headers = session.getHeaders(fedoraId, null);
57          if (RdfLexicon.FEDORA_NON_RDF_SOURCE_DESCRIPTION_URI.equals(headers.getInteractionModel())) {
58              transaction.lockResource(fedoraId.asBaseId());
59          }
60          if (RdfLexicon.NON_RDF_SOURCE.toString().equals(headers.getInteractionModel())) {
61              transaction.lockResource(fedoraId.asDescription());
62          }
63          transaction.lockResource(fedoraId);
64  
65          try {
66              session.persist(operation);
67              recordEvent(transaction, fedoraId, operation);
68          } catch (final PersistentStorageException e) {
69              throw new RepositoryRuntimeException(String.format("Failed to create new version of %s",
70                      fedoraId.getResourceId()), e);
71          }
72      }
73  
74      @VisibleForTesting
75      void setPsManager(final PersistentStorageSessionManager psManager) {
76          this.psManager = psManager;
77      }
78  
79      @VisibleForTesting
80      void setVersionOperationFactory(final VersionResourceOperationFactory versionOperationFactory) {
81          this.versionOperationFactory = versionOperationFactory;
82      }
83  
84  }
