018package org.fcrepo.kernel.impl.operations;
019
020import org.fcrepo.kernel.api.Transaction;
021import org.fcrepo.kernel.api.identifiers.FedoraId;
022
023/**
024 * Build a reference operation.
025 * @author whikloj
026 */
027public class ReferenceOperationBuilder extends AbstractResourceOperationBuilder {
028
029    /**
030     * Constructor.
031     *
032     * @param transaction the transaction
033     * @param rescId the resource identifier.
034     */
035    public ReferenceOperationBuilder(final Transaction transaction, final FedoraId rescId) {
036        super(transaction, rescId);
037    }
038
039    @Override
040    public ReferenceOperation build() {
041        final var operation = new ReferenceOperation(transaction, rescId);
042        operation.setUserPrincipal(userPrincipal);
043        return operation;
044    }
045
046}
