19  package org.fcrepo.kernel.modeshape.utils;
20  
21  /**
22   * @author Daniel Bernstein
23   * @since Apr 27, 2017
24   */
25  public class BNodeSkolemizationUtil {
26  
27      private static final String FCREPO_BNODE_SKOLEMIZE_TO_HASH_FLAG = "fcrepo.bnode.hash-uri";
28  
29      /**
30       * prevents instantiation
31       */
32      private BNodeSkolemizationUtil() {
33      }
34  
35      /**
36       * Returns true if blank nodes should follow the new strategy of skolemizing blank nodes (bnodes) to hash URIs
37       * (rather than using /.well-known/genid/ - c.f. https://jira.duraspace.org/browse/FCREPO-2431).
38       * 
39       * @return true if fcrepo.bnode.hash-uri system property is set to "true"
40       */
41      public static boolean isSkolemizeToHashURIs() {
42          return Boolean.valueOf(System.getProperty(FCREPO_BNODE_SKOLEMIZE_TO_HASH_FLAG, "false"));
43      }
44  
45      /**
46       * Sets the fcrepo.bnode.hash-uri system property.
47       * @param flag to toggle this feature
48       */
49      public static void setSkolemizeToHashURIs(final boolean flag) {
50          System.setProperty(FCREPO_BNODE_SKOLEMIZE_TO_HASH_FLAG, Boolean.toString(flag));
51      }
52  }
