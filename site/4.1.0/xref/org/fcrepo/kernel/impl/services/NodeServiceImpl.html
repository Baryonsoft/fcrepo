39  /**
40   * Service for managing access to Fedora 'nodes' (either datastreams or objects, we don't care.)
41   *
42   * @author Chris Beer
43   * @author ajs6f
44   * @since May 9, 2013
45   */
46  @Component
47  public class NodeServiceImpl extends AbstractService implements NodeService {
48  
49      private static final Logger LOGGER = getLogger(NodeServiceImpl.class);
50  
51      /* (non-Javadoc)
52       * @see org.fcrepo.kernel.services.Service#exists(javax.jcr.Session, java.lang.String)
53       */
54      @Override
55      public boolean exists(final Session session, final String path) {
56          try {
57              validatePath(session, path);
58              return session.nodeExists(path);
59          } catch (final RepositoryException e) {
60              throw new RepositoryRuntimeException(e);
61          }
62      }
63  
64      /**
65       * Retrieve an existing Fedora resource at the given path
66       *
67       * @param session a JCR session
68       * @param path a JCR path
69       * @return Fedora resource at the given path
70       * @throws RepositoryException
71       */
72      @Override
73      public FedoraResource find(final Session session, final String path) {
74          try {
75              return new FedoraResourceImpl(session.getNode(path));
76          } catch (final RepositoryException e) {
77              throw new RepositoryRuntimeException(e);
78          }
79      }
80  
81      /**
82       * Copy an existing object from the source path to the destination path
83       *
84       * @param session
85       * @param source
86       * @param destination
87       * @throws RepositoryException
88       */
89      @Override
90      public void copyObject(final Session session, final String source, final String destination) {
91          try {
92              session.getWorkspace().copy(source, destination);
93          } catch (final RepositoryException e) {
94              throw new RepositoryRuntimeException(e);
95          }
96      }
97  
98      /**
99       * Move an existing object from the source path to the destination path
100      *
101      * @param session
102      * @param source
103      * @param destination
104      * @throws RepositoryException
105      */
106     @Override
107     public void moveObject(final Session session, final String source, final String destination) {
108         try {
109             final FedoraResource srcResource = find(session, source);
110             final Node sourceNode = srcResource.getNode();
111             final String name = sourceNode.getName();
112             final Node parent = sourceNode.getDepth() > 0 ? sourceNode.getParent() : null;
113 
114             session.getWorkspace().move(source, destination);
115 
116             if (parent != null) {
117                 createTombstone(parent, name);
118             }
119 
120         } catch (final RepositoryException e) {
121             throw new RepositoryRuntimeException(e);
122         }
123     }
124 
125     private static void createTombstone(final Node parent, final String path) throws RepositoryException {
126         final FedoraResourceImpl fedoraResource = new FedoraResourceImpl(parent);
127         final Node n  = fedoraResource.findOrCreateChild(parent, path, FEDORA_TOMBSTONE);
128         LOGGER.info("Created tombstone at {} ", n.getPath());
129     }

142         }

151     @Override
152     public void registerNodeTypes(final Session session, final InputStream cndStream) throws IOException {
153         try {
154             final NodeTypeManager nodeTypeManager = (NodeTypeManager) session.getWorkspace().getNodeTypeManager();

158         }
159     }
