16  package org.modeshape.connector.filesystem;
17  
18  import org.modeshape.jcr.cache.document.DocumentTranslator;
19  
20  import java.io.File;
21  import java.nio.file.Path;
22  
23  /**
24   * An extension of JsonSidecarExtraPropertyStore that stores the properties in a
25   * separate configured directory than the filesystem federation itself.  Because
26   * the class we're extending is package protected this class is in the
27   * org.modeshape.connector.filesystem package.
28   *
29   * @author Mike Durbin
30   */
31  public class ExternalJsonSidecarExtraPropertyStore extends JsonSidecarExtraPropertyStore {
32  
33      private FileSystemConnector connector;
34  
35      private File propertyStoreRoot;
36  
37      /**
38       * Default constructor.
39       * @param connector the FileSystemConnector for which this class will store properties.
40       * @param propertyStoreRoot the root of a filesystem into which properties will be
41       *                          serialized.
42       */
43      public ExternalJsonSidecarExtraPropertyStore(final FileSystemConnector connector,
44                                                   final DocumentTranslator translator,
45                                                   final File propertyStoreRoot) {
46          super(connector, translator);
47          this.connector = connector;
48          this.propertyStoreRoot = propertyStoreRoot;
49      }
50  
51      @Override
52      protected File sidecarFile(final String id) {
53          final File file;
54          if (connector.isRoot(id)) {
55              file = new File(propertyStoreRoot, "federation-root.modeshape.json");
56          } else {
57              final Path propertyFileInFederation = super.sidecarFile(id).getAbsoluteFile().toPath();
58              final Path relativePath
59                      = connector.fileFor("/").getAbsoluteFile().toPath().relativize(propertyFileInFederation);
60              file = propertyStoreRoot.toPath().resolve(relativePath).toFile();
61          }
62          if (!file.getParentFile().exists() && !file.getParentFile().mkdirs()) {
63              throw new RuntimeException("Unable to create directories " + file.getParentFile() + ".");
64          }
65          return file;
66      }
67  }
