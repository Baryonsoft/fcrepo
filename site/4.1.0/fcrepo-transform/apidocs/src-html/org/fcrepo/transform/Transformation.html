016package org.fcrepo.transform;
017
018import com.google.common.base.Function;
019import org.fcrepo.kernel.utils.iterators.RdfStream;
020
021import java.io.InputStream;
022
023/**
024 * Generic interface for transforming a resource's properties
025 * to an implementation-defined type
026 *
027 * @author cbeer
028 */
029public interface Transformation<T> extends Function<RdfStream, T> {
030
031    /**
032     * Execute a transform on an rdf stream
033     * @param stream
034     * @return transformed output
035     */
036    @Override
037    T apply(final RdfStream stream);
038
039    /**
040     * Get the Query the transformation is using
041     * @return query used by the transformation as an input stream
042     */
043    InputStream getQuery();
044
045    /**
046     * @return a new Transform of this type, for use as a factory
047     */
048    Transformation<T> newTransform(InputStream query);
049
050}
