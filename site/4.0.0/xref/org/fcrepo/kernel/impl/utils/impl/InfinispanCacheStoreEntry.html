45  public class InfinispanCacheStoreEntry extends LocalBinaryStoreEntry {
46      private static final Logger LOGGER = getLogger(InfinispanCacheStoreEntry.class);
47  
48      private static final GetClusterExecutor EXECUTOR_FACTORY = new GetClusterExecutor();
49      /**
50       *
51       * @param store
52       * @param property
53       */
54      public InfinispanCacheStoreEntry(final InfinispanBinaryStore store, final Property property) {
55          super(store, property);
56      }
57  
58      @Override
59      public Collection<FixityResult> checkFixity(final String algorithm) {
60          final BinaryKey key = binaryKey();
61          final ImmutableSet.Builder<FixityResult> fixityResults = new ImmutableSet.Builder<>();
62  
63          if (store().hasBinary(key)) {
64              final String dataKey = InfinispanUtils.dataKeyFrom((InfinispanBinaryStore)store(), key);
65              final ChunkBinaryMetadata metadata = InfinispanUtils.getMetadata((InfinispanBinaryStore)store(), key);
66  
67              final DistributedFixityCheck task = new DistributedFixityCheck(dataKey, algorithm, metadata.getChunkSize(),
68                      metadata.getLength());
69  
70              final List<Future<Collection<FixityResult>>> futures
71                  = clusterExecutor().submitEverywhere(task, dataKey + "-0");
72  
73              while (!futures.isEmpty()) {
74                  final Iterator<Future<Collection<FixityResult>>> iterator = futures.iterator();
75                  while (iterator.hasNext()) {
76                      final Future<Collection<FixityResult>> future = iterator.next();
77                      try {
78                          final Collection<FixityResult> result = future.get(100, MILLISECONDS);
79                          iterator.remove();
80  
81                          fixityResults.addAll(result);
82                      } catch (final TimeoutException e) {
83                          LOGGER.trace("Going to retry cluster transform after timeout", e);
84                      } catch (InterruptedException | ExecutionException e) {
85                          throw propagate(e);
86                      }
87                  }
88              }
89          }
90          return fixityResults.build();
91      }
92  
93      private DistributedExecutorService clusterExecutor() {
94          return EXECUTOR_FACTORY.apply((InfinispanBinaryStore)store());
95      }
96  }
