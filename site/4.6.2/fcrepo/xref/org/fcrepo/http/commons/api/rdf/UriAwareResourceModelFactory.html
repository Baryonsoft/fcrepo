18  package org.fcrepo.http.commons.api.rdf;
19  
20  import javax.ws.rs.core.UriInfo;
21  
22  import com.hp.hpl.jena.rdf.model.Resource;
23  import org.fcrepo.kernel.api.models.FedoraResource;
24  import org.fcrepo.kernel.api.identifiers.IdentifierConverter;
25  
26  import com.hp.hpl.jena.rdf.model.Model;
27  
28  /**
29   * Helper to generate an RDF model for a FedoraResourceImpl that (likely) creates
30   * relations from our resource to other HTTP components
31   *
32   * @author awoods
33   */
34  public interface UriAwareResourceModelFactory {
35  
36      /**
37       * Given a resource, the UriInfo and a way to generate graph subjects,
38       * create a model with triples to inject into an RDF response for the
39       * resource (e.g. to add HATEOAS links)
40       *
41       * @param resource the resource
42       * @param uriInfo the uri info
43       * @param idTranslator the id translator
44       * @return model containing triples for the given resource
45       */
46      Model createModelForResource(final FedoraResource resource,
47              final UriInfo uriInfo, final IdentifierConverter<Resource,FedoraResource> idTranslator);
48  }
