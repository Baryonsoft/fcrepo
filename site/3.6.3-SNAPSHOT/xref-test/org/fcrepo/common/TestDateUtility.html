87  				"1970-01-01T00:00:00.00Z", "1970-01-01T00:00:00.0Z",
88  				"1970-01-01T00:00:00Z", "1970-01-01Z",
89  				"1970-01-01T00:00:00.000", "1970-01-01T00:00:00.00",
90  				"1970-01-01T00:00:00.0", "1970-01-01T00:00:00", "1970-01-01",
91  				"Thu, 01 Jan 1970 00:00:00 GMT" };
92  		for (String element : dates) {
93  			assertEquals(EPOCH, DateUtility.parseDateLoose(element));
94  		}
95  
96  		String[] badDates = { "", "ABCD-EF-GHTIJ:KL:MN.OPQZ", "1234", "1",
97  				"1970-01", "1970-1-1", "12345-01-01T00:00:00.000Z",
98  				"12345-01-01T00:00:00." };
99  		for (String element : badDates) {
100 			assertNull(element + " not null",
101 					DateUtility.parseDateLoose(element));
102 		}
103 
104 		assertEquals(ONE_CE, DateUtility.parseDateLoose(ONE_CE_XSD_DT));
105 		assertEquals(ONE_BCE, DateUtility.parseDateLoose(ONE_BCE_XSD_DT));
106 		assertEquals(TWO_BCE, DateUtility.parseDateLoose(TWO_BCE_XSD_DT));

109 	@Test
110 	public void testMillis() throws Exception {
111 		// canonical form of 200 milliseconds after Epoch
112 		String a = "1970-01-01T00:00:00.2Z";
113 		// also 200 ms after Epoch
114 		String b = "1970-01-01T00:00:00.200Z";
115 		Date aDate = DateUtility.parseDateStrict(a);
116 		Date bDate = DateUtility.parseDateStrict(b);
117 		assertEquals(200, aDate.getTime());
118 		assertEquals(200, bDate.getTime());
119 		assertEquals(a, DateUtility.convertDateToXSDString(aDate));
120 		assertEquals(b, DateUtility.convertDateToString(bDate));
121 
122 		// canonical form of 20 milliseconds after Epoch
123 		String c = "1970-01-01T00:00:00.02Z";
124 		// also 20 ms after Epoch
125 		String d = "1970-01-01T00:00:00.020Z";
126 		Date cDate = DateUtility.parseDateStrict(c);
127 		Date dDate = DateUtility.parseDateStrict(d);
128 		assertEquals(20, cDate.getTime());
129 		assertEquals(20, dDate.getTime());
130 		assertEquals(c, DateUtility.convertDateToXSDString(cDate));
131 		assertEquals(d, DateUtility.convertDateToString(dDate));
132 
133 		// canonical form of 2 milliseconds after Epoch
134 		String e = "1970-01-01T00:00:00.002Z";
135 		Date eDate = DateUtility.parseDateStrict(e);
136 		assertEquals(2, eDate.getTime());
137 		assertEquals(e, DateUtility.convertDateToXSDString(eDate));
138 		assertEquals(e, DateUtility.convertDateToString(eDate));
139 
140 		// variations of Epoch
141 		String f = "1970-01-01T00:00:00.0Z";
142 		String g = "1970-01-01T00:00:00.00Z";
143 		Date fDate = DateUtility.parseDateStrict(f);
144 		Date gDate = DateUtility.parseDateStrict(g);
145 		assertEquals(0, fDate.getTime());
146 		assertEquals(0, gDate.getTime());
147 		assertEquals(EPOCH_XSD_DT, DateUtility.convertDateToXSDString(fDate));
148 		assertEquals(EPOCH_XSD_DT, DateUtility.convertDateToXSDString(gDate));
149 		assertEquals(EPOCH_DT, DateUtility.convertDateToString(fDate));
150 		assertEquals(EPOCH_DT, DateUtility.convertDateToString(gDate));
151 
152 		// negative dates
153 		String n = "-1234-01-01T00:00:00.2Z";
154 		String o = "-1234-01-01T00:00:00.200Z";
155 		Date nDate = DateUtility.parseDateStrict(n);
156 		Date oDate = DateUtility.parseDateStrict(o);
157 		assertEquals(nDate.getTime(), oDate.getTime());
158 		assertEquals(n, DateUtility.convertDateToXSDString(nDate));
159 	}
