28  /**
29   * Tests the METS deserializer and serializer by opening a METS file (supplied
30   * at command-line), deserializing it, re-serializing it, and sending it to
31   * STDOUT.
32   * 
33   * @author Chris Wilper
34   */
35  public class METSInOutTest
36          implements Constants {
37  
38      public static void main(String args[]) {
39          FileInputStream in = null;
40          // set system properties for testing purposes          
41          System.setProperty("fedoraServerHost", "localhost");
42          System.setProperty("fedoraServerPort", "80");
43  
44          try {
45              if (args.length < 1) {
46                  throw new IOException("At least one parameter needed.");
47              }
48              in = new FileInputStream(new File(args[0]));
49          } catch (IOException ioe) {
50              System.out.println("Error: " + ioe.getMessage());
51              System.out
52                      .println("Give the path to an existing METS file, and optionally, the level of validation to perform on the re-serialized version.");
53              System.exit(0);
54          }
55          try {
56              METSFedoraExt1_1DODeserializer deser =
57                      new METSFedoraExt1_1DODeserializer();
58              METSFedoraExt1_1DOSerializer ser =
59                      new METSFedoraExt1_1DOSerializer();
60              HashMap<String, DODeserializer> desers = new HashMap<String, DODeserializer>();
61              HashMap<String, DOSerializer> sers = new HashMap<String, DOSerializer>();
62              desers.put(METS_EXT1_1.uri, deser);
63              sers.put(METS_EXT1_1.uri, ser);
64              DOTranslatorImpl trans = new DOTranslatorImpl(sers, desers);
65              DigitalObject obj = new BasicDigitalObject();
66              System.out.println("Deserializing...");
67              trans.deserialize(in,
68                                obj,
69                                METS_EXT1_1.uri,
70                                "UTF-8",
71                                DOTranslationUtility.DESERIALIZE_INSTANCE);
72              System.out.println("Done.");
73              ByteArrayOutputStream out = new ByteArrayOutputStream();
74              System.out.println("Re-serializing for STORAGE...");
75              trans.serialize(obj,
76                              out,
77                              METS_EXT1_1.uri,
78                              "UTF-8",
79                              DOTranslationUtility.SERIALIZE_STORAGE_INTERNAL);
80              System.out.println("Done.");
81              if (args.length > 1) {
82                  ByteArrayInputStream newIn =
83                          new ByteArrayInputStream(out.toByteArray());
84                  HashMap<String, String> xmlSchemaMap = new HashMap<String, String>();
85                  // LOOK!  These path values should work if test is run from
86                  // the FEDORA HOME directory.  Adjust accordingly for test environment.
87                  xmlSchemaMap.put(METS_EXT1_1.uri,
88                                   "dist/server/xsd/mets-fedora-ext1-1.xsd");
89                  HashMap<String, String> ruleSchemaMap = new HashMap<String, String>();
90                  ruleSchemaMap.put(METS_EXT1_1.uri,
91                                    "dist/server/schematron/metsExtRules1-0.xml");
92                  DOValidatorImpl v =
93                          new DOValidatorImpl(null,
94                                              xmlSchemaMap,
95                                              "dist/server/schematron/preprocessor.xslt",
96                                              ruleSchemaMap);
97                  if (args[1].equals("1")) {
98                      v.validate(newIn,
99                                 METS_EXT1_1.uri,
100                                DOValidatorImpl.VALIDATE_XML_SCHEMA,
101                                "ingest");
102                     System.out.println("XML Schema validation: PASSED!");
103                 } else {
104                     if (args[1].equals("2")) {
105                         v.validate(newIn,
106                                    METS_EXT1_1.uri,
107                                    DOValidatorImpl.VALIDATE_SCHEMATRON,
108                                    "ingest");
109                         System.out.println("Schematron validation: PASSED!");
110                     } else {
111                         System.out.println("Unrecognized validation level, '"
112                                 + args[1] + "'");
113                     }
114                 }
115             } else {
116                 System.out.println("Here it is:");
117                 System.out.println(out.toString("UTF-8"));
118             }
119 
120         } catch (Exception e) {
121             System.out.println("Error: (" + e.getClass().getName() + "):"
122                     + e.getMessage());
123             e.printStackTrace();
124         }
125     }
