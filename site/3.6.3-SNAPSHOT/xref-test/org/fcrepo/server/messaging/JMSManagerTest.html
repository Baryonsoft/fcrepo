330     /**
331      * Waits for a message and checks to see if it is valid.
332      */
333     private void checkMessage(String destination, DestinationType type, String messageText) throws Exception {
334         long startTime = System.currentTimeMillis();
335 
336         while (true) { // Wait for the message
337             if (messageCount > 0) {
338                 assertNotNull(currentMessage);
339                 if(currentMessage instanceof TextMessage) {
340                     assertEquals(messageText, ((TextMessage)currentMessage).getText());
341                 }
342 
343                 Destination messageDestination = currentMessage.getJMSDestination();
344                 if(type.equals(DestinationType.Topic)) {
345                     if (messageDestination instanceof Topic) {
346                         String topic = ((Topic) messageDestination).getTopicName();
347                         assertEquals(topic, destination);
348                     } else {
349                         fail("Destination type for message should have been Topic");
350                     }
351                 } else {
352                     if (messageDestination instanceof Queue) {
353                         String queue = ((Queue) messageDestination).getQueueName();
354                         assertEquals(queue, destination);
355                     } else {
356                         fail("Destination type for message should have been Queue");
357                     }
358                 }
359                 break;
360             } else { // Check for timeout
361                 long currentTime = System.currentTimeMillis();
362                 if (currentTime > (startTime + timeout)) {
363                     fail("Timeout reached waiting for message.");
364                     break;
365                 } else {
366                     try {
367                         Thread.sleep(100);
368                     } catch (Exception e) {
369                         throw new RuntimeException(e);
370                     }
371                 }
372             }
373         }
374 
375         messageCount = 0;
376         currentMessage = null;
377     }

379     /**
380      * Waits for a message to make sure none come through.
381      */
382     private void checkNoMessage() {
383         long startTime = System.currentTimeMillis();
384 
385         while (true) { // Wait for the notification message
386             if (messageCount > 0) {
387                 fail("No messages should be received during this test.");
388                 break;
389             } else { // Check for timeout
390                 long currentTime = System.currentTimeMillis();
391                 if (currentTime > (startTime + timeout)) {
392                     break;
393                 } else {
394                     try {
395                         Thread.sleep(100);
396                     } catch (Exception e) {
397                         throw new RuntimeException(e);
398                     }
399                 }
400             }
401         }
402 
403         messageCount = 0;
404         currentMessage = null;
405     }

416     public class JmsBroker extends Thread {
417 
418         private final String connectorUrl;
419         private boolean stop = false;
420 
421         public JmsBroker(String connectorUrl) {
422             this.connectorUrl = connectorUrl;
423         }
424 
425         @Override
426         public void run() {
427             BrokerService broker = new BrokerService();
428             try {
429                 broker.addConnector(connectorUrl);
430                 broker.start();
431             } catch (Exception e) {
432                 System.err.println("Exception encountered starting "
433                         + "broker with connectorUrl: " + connectorUrl
434                         + ". Exception message: " + e.getMessage());
435             }
436 
437             while (!stop) {
438                 // Run the broker
439             }
440 
441             try {
442                 broker.stop();
443             } catch (Exception e) {
444                 System.err.println("Exception encountered stopping "
445                         + "broker with connectorUrl: " + connectorUrl
446                         + ". Exception message: " + e.getMessage());
447             }
448         }
449 
450         public void exit() {
451             stop = true;
452         }
