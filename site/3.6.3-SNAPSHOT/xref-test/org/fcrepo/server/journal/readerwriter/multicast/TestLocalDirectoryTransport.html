115     @Test
116     public void testOperations() throws JournalException, IOException,
117             XMLStreamException, ParseException {
118         XMLEventFactory factory = XMLEventFactory.newInstance();
119         QName name1a = new QName("junkElement1a");
120         QName name1b = new QName("junkElement1b");
121         QName name2 = new QName("junkElement2");
122         SimpleDateFormat parser =
123                 new SimpleDateFormat("yyyy-MM-dd'T'HH:mm:ss.SSSZ");
124 
125         Transport transport =
126                 new LocalDirectoryTransport(parameters, CRUCIAL, parent);
127 
128         // open creates the temp file
129         parent.setCurrentDate(parser.parse("2007-03-05T16:49:21.392-0500"));
130         transport
131                 .openFile("firstSillyHash", "fileOne", parent.getCurrentDate());
132         File tempfile1 = new File(journalDirectory, "_fileOne");
133         assertFileExists(tempfile1);
134 
135         // write to the file
136         transport.getWriter().add(factory
137                 .createStartElement(name1a, null, null));
138         transport.getWriter().add(factory.createEndElement(name1a, null));
139         transport.getWriter().add(factory
140                 .createStartElement(name1b, null, null));
141         transport.getWriter().add(factory.createEndElement(name1b, null));
142 
143         // closing renames the file
144         transport.closeFile();
145         File file1 = new File(journalDirectory, "fileOne");
146         assertFileExists(file1);
147         assertFileDoesNotExist(tempfile1);
148 
149         // open creates another temp file
150         transport.openFile("secondSillyHash", "fileTwo", parent
151                 .getCurrentDate());
152         File tempfile2 = new File(journalDirectory, "_fileTwo");
153         assertFileExists(tempfile2);
154 
155         // write to the file
156         transport.getWriter()
157                 .add(factory.createStartElement(name2, null, null));
158         transport.getWriter().add(factory.createEndElement(name2, null));
159 
160         // closing renames the file
161         transport.closeFile();
162         File file2 = new File(journalDirectory, "fileTwo");
163         assertFileExists(file2);
164         assertFileDoesNotExist(tempfile2);
165 
166         // shut it down
167         transport.shutdown();
168 
169         // did we write what was expected?
170         assertFileContents(EXPECTED_JOURNAL_1_CONTENTS, file1);
171         assertFileContents(EXPECTED_JOURNAL_2_CONTENTS, file2);
172     }
