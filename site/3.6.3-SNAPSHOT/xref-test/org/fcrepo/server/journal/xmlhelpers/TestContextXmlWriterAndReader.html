49      public void testBasicWriteAndRead() throws Exception {

52          context1.setNoOp(true);
53          context1.setEnvironmentAttributes(createMap("envAttr", "envValue"));
54          context1.setSubjectAttributes(createMap(new String[][] {

59                  "recoveryAttribute",

63          contextWriter.writeContext(context1, xmlWriter);

65          String xmlString = xmlStringWriter.toString();
66          XMLEventReader xmlReader = createXmlReaderAndPosition(xmlString);

68          ContextXmlReader contextReader = new ContextXmlReader();

99      private XMLEventReader createXmlReaderAndPosition(String xmlString)
100             throws FactoryConfigurationError, XMLStreamException {
101         XMLInputFactory factory = XMLInputFactory.newInstance();
102         XMLEventReader xmlReader =
103                 factory.createXMLEventReader(new StringReader(xmlString));
104         advanceToContext(xmlReader);
105         return xmlReader;
106     }
107 
108     private XMLEventWriter createXmlWriter() throws FactoryConfigurationError,
109             XMLStreamException {
110         XMLOutputFactory factory = XMLOutputFactory.newInstance();
111         return new IndentingXMLEventWriter(factory
112                 .createXMLEventWriter(xmlStringWriter));
113     }

134     private void advanceToContext(XMLEventReader reader)
135             throws XMLStreamException {
136         XMLEvent event = reader.peek();
137         if (event.isStartDocument()) {
138             reader.nextEvent();
139         } else {
140             fail("Document does not begin with a 'StartDocument' event.");
141         }
142     }

144     private void assertContextsAreEqual(JournalEntryContext context1,
145                                         JournalEntryContext context2) {
146         assertEquals(context1.getPassword(), context2.getPassword());
147         assertEquals(context1.getNoOp(), context2.getNoOp());
148         assertEquals(context1.now(), context2.now());
149         assertEqualMultiMaps(context1.getEnvironmentAttributes(), context2
150                 .getEnvironmentAttributes());
151         assertEqualMultiMaps(context1.getSubjectAttributes(), context2
152                 .getSubjectAttributes());
153         assertEqualMultiMaps(context1.getActionAttributes(), context2
154                 .getActionAttributes());
155         assertEqualMultiMaps(context1.getResourceAttributes(), context2
156                 .getResourceAttributes());
157         assertEqualMultiMaps(context1.getRecoveryAttributes(), context2
158                 .getRecoveryAttributes());
159     }

161     private void assertEqualMultiMaps(MultiValueMap map1, MultiValueMap map2) {
162         Iterator names1 = map1.names();
163         Iterator names2 = map2.names();
164         while (names1.hasNext() && names2.hasNext()) {
165             String name1 = (String) names1.next();
166             String name2 = (String) names2.next();
167             assertEquals(name1, name2);
168             String[] values1 = map1.getStringArray(name1);
169             String[] values2 = map1.getStringArray(name2);
170             assertEqualSets(new HashSet<String>(Arrays.asList(values1)),
171                             new HashSet<String>(Arrays.asList(values2)));
172         }
173     }
174 
175     private void assertEqualSets(Set<String> set1, Set<String> set2) {
176         assertEquals(set1, set2);
177     }
