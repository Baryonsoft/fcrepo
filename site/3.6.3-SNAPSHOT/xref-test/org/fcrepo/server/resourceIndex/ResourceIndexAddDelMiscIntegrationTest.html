29      /**
30       * Add, then delete an object with the RI at level 0. This test ensures that
31       * adds and deletes at level 0 don't do anything.
32       */
33      @Test
34      public void testAddDelObjLv0() throws Exception {
35          Set<DigitalObject> objects = getTestObjects(1, 0);
36  
37          // add at level 0
38          initRI(0);
39          addAll(objects, true);
40  
41          assertEquals("Did not get expected triples after add",
42                       Collections.EMPTY_SET,
43                       getActualTriples());
44  
45          // add at level 1
46          initRI(1);
47          addAll(objects, true);
48          Set<Triple> expected = getExpectedTriples(1, objects);
49  
50          // delete at level 0
51          initRI(0);
52          deleteAll(objects, true);
53  
54          assertTrue("Did not get expected triples after delete",
55                     sameTriples(expected, getActualTriples(), true));
56      }
