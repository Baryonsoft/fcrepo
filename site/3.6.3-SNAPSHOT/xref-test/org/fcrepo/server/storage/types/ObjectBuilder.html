97      private static void addDatastream(DigitalObject obj,
98                                        String id,
99                                        Datastream ds) {
100         int size = 0;
101         for (Datastream d : obj.datastreams(id)) {
102             size++;
103         }
104         ds.DatastreamID = id;
105         ds.DSState = "A";
106         ds.DSVersionable = true;
107         ds.DSVersionID = id + "." + size;
108         ds.DSLabel = "ds label";
109         ds.DSCreateDT = new Date();
110         obj.addDatastreamVersion(ds, true);
111     }

125     public static DigitalObject getTestObject(String pid,
126                                               URIReference[] models,
127                                               String state,
128                                               String ownerId,
129                                               String label,
130                                               Date createDate,
131                                               Date lastModDate) {
132         DigitalObject obj = new BasicDigitalObject();
133         obj.setPid(pid);
134 
135         StringBuilder rdf = new StringBuilder();
136         rdf.append("<rdf:RDF xmlns:rdf=\"http://www.w3.org/1999/02/22-rdf-syntax-ns#\" "
137                    + "xmlns:fedora-model=\"info:fedora/fedora-system:def/model#\">\n"
138                    + "<rdf:Description rdf:about=\"");
139         rdf.append(PID.getInstance(pid).toURI() + "\">\n");
140         for (URIReference model : models) {
141             rdf.append("<fedora-model:hasModel rdf:resource=\""
142                        + model.getURI().toString()
143                        + "\"></fedora-model:hasModel>\n");
144         }
145         rdf.append("</rdf:Description></rdf:RDF>");
146 
147         addXDatastream(obj, "RELS-EXT", rdf.toString());
148 
149         obj.setState(state);
150         obj.setOwnerId(ownerId);
151         obj.setLabel(label);
152         obj.setCreateDate(createDate);
153         obj.setLastModDate(lastModDate);
154         return obj;
155     }

157     /**
158      * Make a deep copy of the given digital object.
159      */
160     public static DigitalObject deepCopy(DigitalObject obj) throws Exception {
161 
162         // make sure DOTranslationUtility doesn't die
163         if (System.getProperty("fedoraServerHost") == null
164             || System.getProperty("fedoraServerPort") == null) {
165             System.setProperty("fedoraServerHost", "localhost");
166             System.setProperty("fedoraServerPort", "8080");
167         }
168 
169         String charEncoding = "UTF-8";
170         int transContext = DOTranslationUtility.SERIALIZE_STORAGE_INTERNAL;
171 
172         ByteArrayOutputStream out = new ByteArrayOutputStream();
173         FOXML1_1DOSerializer ser = new FOXML1_1DOSerializer();
174         ser.serialize(obj, out, charEncoding, transContext);
175 
176         FOXML1_1DODeserializer deser = new FOXML1_1DODeserializer();
177         ByteArrayInputStream in = new ByteArrayInputStream(out.toByteArray());
178         DigitalObject objCopy = new BasicDigitalObject();
179         deser.deserialize(in, objCopy, charEncoding, transContext);
180 
181         // make sure dates of any to-be-added new components differ
182         try {
183             Thread.sleep(100);
184         } catch (Exception e) {
185         }
186 
187         return objCopy;
188     }

235     /**
236      * Sets any un-set dates in the object to the given date.
237      */
238     public static void setDates(DigitalObject obj, Date date) {
239         if (obj.getCreateDate() == null) {
240             obj.setCreateDate(date);
241         }
242         Iterator<String> dsIds = obj.datastreamIdIterator();
243         while (dsIds.hasNext()) {
244             String dsid = dsIds.next();
245             for (Datastream ds : obj.datastreams(dsid)) {
246                 if (ds.DSCreateDT == null) {
247                     ds.DSCreateDT = new Date();
248                 }
249             }
250         }
251     }
