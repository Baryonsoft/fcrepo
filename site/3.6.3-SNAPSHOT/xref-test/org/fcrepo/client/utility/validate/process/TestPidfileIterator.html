42      /**
43       * Confirm that we iterate over the lines, ignoring comment lines and blank
44       * or empty lines.
45       */
46      @Test
47      public void simpleExerciseWithPidsBlanksAndComments() throws IOException {
48          String[] lines =
49                  new String[] {"#comment", " #comment start with space",
50                          FIRST_PID, SECOND_PID, "", THIRD_PID, " "};
51  
52          File dummyFile = null;
53          try {
54              dummyFile =
55                      File.createTempFile("TestValidatorProcessParameter",
56                                          "dummyFile");
57              dummyFile.deleteOnExit();
58              loadPidFile(dummyFile, lines);
59  
60              PidfileIterator iterator = new PidfileIterator(dummyFile);
61  
62              assertTrue("has first line", iterator.hasNext());
63              assertEquals("first line", FIRST_PID, iterator.next());
64              assertTrue("has second line", iterator.hasNext());
65              assertEquals("second line", SECOND_PID, iterator.next());
66              assertTrue("has third line", iterator.hasNext());
67              assertEquals("third line", THIRD_PID, iterator.next());
68              assertFalse("no fourth line", iterator.hasNext());
69          } finally {
70              if (dummyFile != null || dummyFile.exists()) {
71                  dummyFile.delete();
72              }
73          }
74      }
75  
76      /**
77       * Fill the file with these lines.
78       */
79      private void loadPidFile(File file, String[] lines) throws IOException {
80          Writer writer = null;
81          try {
82              writer = new FileWriter(file);
83              for (String line : lines) {
84                  writer.write(line + '\n');
85              }
86          } finally {
87              if (writer != null) {
88                  try {
89                      writer.close();
90                  } catch (IOException e) {
91                      System.err.println("failed to close writer on file '"
92                              + file + "'");
93                  }
94              }
95          }
96      }
