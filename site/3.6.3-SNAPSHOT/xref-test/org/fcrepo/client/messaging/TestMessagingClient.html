248     private void sendMessage(String jndiName) throws Exception {
249         JMSManager jmsManager = new JMSManager(properties);
250         TextMessage message = jmsManager.createTextMessage(jndiName, messageText);
251         message.setStringProperty(propertyName, propertyValue);
252         jmsManager.send(jndiName, message);
253         jmsManager.stop(jndiName);
254         jmsManager.close();

257     /**
258      * Waits for a message and checks to see if it is valid.
259      */
260     private void checkMessage(String clientId, String destination) throws JMSException {
261         long startTime = System.currentTimeMillis();
262 
263         while (true) { // Wait for the message
264             if (messageCount > 0) {
265                 assertNotNull(currentMessage);
266                 if(currentMessage instanceof TextMessage) {
267                     assertEquals(messageText, ((TextMessage)currentMessage).getText());
268                 } else {
269                     fail("Text Message expected.");
270                 }
271 
272                 assertEquals(clientId, currentClientId);
273 
274                 Destination messageDestination = currentMessage.getJMSDestination();
275                 if (messageDestination instanceof Topic) {
276                     String topic = ((Topic) messageDestination).getTopicName();
277                     assertEquals(topic, destination);
278                 } else if (messageDestination instanceof Queue) {
279                     String queue = ((Queue) messageDestination).getQueueName();
280                     assertEquals(queue, destination);
281                 }
282 
283                 String propertyTest =
284                     currentMessage.getStringProperty(propertyName);
285                 assertEquals(propertyValue, propertyTest);
286                 break;
287             } else { // Check for timeout
288                 long currentTime = System.currentTimeMillis();
289                 if (currentTime > (startTime + messageTimeout)) {
290                     fail("Timeout reached waiting for message.");
291                     break;
292                 } else {
293                     try {
294                         Thread.sleep(100);
295                     } catch (Exception e) {
296                         throw new RuntimeException(e);
297                     }
298                 }
299             }
300         }
301 
302         messageCount = 0;
303         currentMessage = null;
304         currentClientId = null;
305     }

307     /**
308      * Waits for a message to make sure none come through.
309      */
310     private void checkNoMessages() {
311         long startTime = System.currentTimeMillis();
312 
313         while (true) { // Wait for the message
314             if (messageCount > 0) {
315                 fail("No messagess should be received during this test.");
316                 break;
317             } else { // Check for timeout
318                 long currentTime = System.currentTimeMillis();
319                 if (currentTime > (startTime + messageTimeout)) {
320                     break;
321                 } else {
322                     try {
323                         Thread.sleep(100);
324                     } catch (Exception e) {
325                         throw new RuntimeException(e);
326                     }
327                 }
328             }
329         }
330 
331         messageCount = 0;
332         currentMessage = null;
333         currentClientId = null;
334     }
