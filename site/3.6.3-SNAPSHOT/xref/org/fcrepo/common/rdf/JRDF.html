16  /**
17   * Utility methods for working with JRDF.
18   * 
19   * @author Chris Wilper
20   */
21  public abstract class JRDF {
22  
23      /**
24       * Tells whether the given nodes are equivalent.
25       * <p>
26       * Nodes are equivalent if they are both resources or both literals and they
27       * match according to the rules of those types.
28       * 
29       * @param n1
30       *        first node.
31       * @param n2
32       *        second node.
33       * @return true if equivalent, false otherwise.
34       */
35      public static boolean sameNode(Node n1, Node n2) {
36          if (n1 instanceof URIReference && n2 instanceof URIReference) {
37              return sameResource((URIReference) n1, (URIReference) n2);
38          } else if (n1 instanceof Literal && n2 instanceof Literal) {
39              return sameLiteral((Literal) n1, (Literal) n2);
40          } else {
41              return false;
42          }
43      }
44  
45      /**
46       * Tells whether the given resources are equivalent.
47       * <p>
48       * Two resources are equivalent if their URIs match.
49       * 
50       * @param u1
51       *        first resource.
52       * @param u2
53       *        second resource.
54       * @return true if equivalent, false otherwise.
55       */
56      public static boolean sameResource(URIReference u1, URIReference u2) {
57          return sameResource(u1, u2.getURI().toString());
58      }
59  
60      /**
61       * Tells whether the given resources are equivalent, with one given as a URI
62       * string.
63       * 
64       * @param u1
65       *        first resource.
66       * @param u2
67       *        second resource, given as a URI string.
68       * @return true if equivalent, false otherwise.
69       */
70      public static boolean sameResource(URIReference u1, String u2) {
71          return u1.getURI().toString().equals(u2);
72      }
73  
74      /**
75       * Tells whether the given literals are equivalent.
76       * <p>
77       * Two literals are equivalent if they have the same lexical value, language
78       * (which may be unspecified), and datatype (which may be unspecified).
79       * 
80       * @param l1
81       *        first literal.
82       * @param l2
83       *        second literal.
84       * @return true if equivalent, false otherwise.
85       */
86      public static boolean sameLiteral(Literal l1, Literal l2) {
87          URI type = l2.getDatatypeURI();
88          return sameLiteral(l1, l2.getLexicalForm(), type, l2.getLanguage());
89      }
90  
91      /**
92       * Tells whether the given literals are equivalent, with one given as a set
93       * of simple values.
94       * 
95       * @param l1
96       *        first literal.
97       * @param l2
98       *        second literal's lexical value.
99       * @param type
100      *        second literal's datatype URI string, if applicable.
101      * @param lang
102      *        second literal's language tag string, if applicable.
103      * @return true if equivalent, false otherwise.
104      */
105     public static boolean sameLiteral(Literal l1,
106                                       String l2,
107                                       URI type,
108                                       String lang) {
109         if (l1.getLexicalForm().equals(l2) && eq(l1.getLanguage(), lang)) {
110             if (l1.getDatatypeURI() == null) {
111                 return type == null;
112             } else {
113                 return type != null
114                         && type.equals(l1.getDatatypeURI());
115             }
116         } else {
117             return false;
118         }
119     }

121     /**
122      * Tells whether the given subjects are equivalent.
123      * <p>
124      * Two subjects are equivalent if they are both resources and they match
125      * according to the rules for resources.
126      * 
127      * @param s1
128      *        first subject.
129      * @param s2
130      *        second subject.
131      * @return true if equivalent, false otherwise.
132      */
133     public static boolean sameSubject(SubjectNode s1, SubjectNode s2) {
134         if (s1 instanceof URIReference && s2 instanceof URIReference) {
135             return sameResource((URIReference) s1, (URIReference) s2);
136         } else {
137             return false;
138         }
139     }

141     /**
142      * Tells whether the given subjects are equivalent, with one given as a URI
143      * string.
144      * 
145      * @param s1
146      *        first subject.
147      * @param s2
148      *        second subject, given as a URI string.
149      * @return true if equivalent, false otherwise.
150      */
151     public static boolean sameSubject(SubjectNode s1, String s2) {
152         if (s1 instanceof URIReference) {
153             return sameResource((URIReference) s1, s2);
154         } else {
155             return false;
156         }
157     }

159     /**
160      * Tells whether the given predicates are equivalent.
161      * <p>
162      * Two predicates are equivalent if they match according to the rules for
163      * resources.
164      * 
165      * @param p1
166      *        first predicate.
167      * @param p2
168      *        second predicate.
169      * @return true if equivalent, false otherwise.
170      */
171     public static boolean samePredicate(PredicateNode p1, PredicateNode p2) {
172         return sameResource((URIReference) p1, (URIReference) p2);
173     }
174 
175     /**
176      * Tells whether the given predicates are equivalent, with one given as a
177      * URI string.
178      * 
179      * @param p1
180      *        first predicate.
181      * @param p2
182      *        second predicate, given as a URI string.
183      * @return true if equivalent, false otherwise.
184      */
185     public static boolean samePredicate(PredicateNode p1, String p2) {
186         return sameResource((URIReference) p1, p2);
187     }

189     /**
190      * Tells whether the given objects are equivalent.
191      * <p>
192      * Two objects are equivalent if they match according to the rules for
193      * nodes.
194      * 
195      * @param o1
196      *        first object.
197      * @param o2
198      *        second object.
199      * @return true if equivalent, false otherwise.
200      */
201     public static boolean sameObject(ObjectNode o1, ObjectNode o2) {
202         return sameNode(o1, o2);
203     }
204 
205     /**
206      * Tells whether the given objects are equivalent, with one given as a set
207      * of simple values.
208      * 
209      * @param o1
210      *        first node.
211      * @param o2
212      *        second node URI (if isLiteral is false) or lexical value (if
213      *        isLiteral is true)
214      * @param isLiteral
215      *        whether the second node is a literal.
216      * @param type
217      *        second literal's datatype URI string, if applicable.
218      * @param lang
219      *        second literal's language tag string, if applicable.
220      * @return true if equivalent, false otherwise.
221      */
222     public static boolean sameObject(ObjectNode o1,
223                                      String o2,
224                                      boolean isLiteral,
225                                      URI type,
226                                      String lang) {
227         if (o1 instanceof URIReference) {
228             return sameResource((URIReference) o1, o2);
229         } else if (o1 instanceof Literal || isLiteral) {
230             return sameLiteral((Literal) o1, o2, type, lang);
231         } else {
232             return false;
233         }
234     }

236     // test for equality, accounting for null values
237     private static boolean eq(Object a, Object b) {
238         if (a == null) {
239             return b == null;
240         } else {
241             return b != null && a.equals(b);
242         }
243     }
