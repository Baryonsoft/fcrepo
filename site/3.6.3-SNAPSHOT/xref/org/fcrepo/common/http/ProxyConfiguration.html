12  /**
13   * Configures the proxy settings of the web client. Can either be created using
14   * the system settings or with a custom configuration.
15   *
16   * @version $Id$
17   */
18  public class ProxyConfiguration {
19  
20      private static final Logger logger =
21          LoggerFactory.getLogger(ProxyConfiguration.class);
22  
23      private final String proxyHost;
24  
25      private final int proxyPort;
26  
27      private final String proxyUser;
28  
29      private final String proxyPassword;
30  
31      private final Pattern nonProxyPattern;
32  
33      /**
34       * Default constructor.
35       * Takes the system provided proxy settings.
36       */
37      public ProxyConfiguration() {
38          this.proxyHost = System.getProperty("http.proxyHost");
39          this.proxyPort =
40                  Integer.parseInt(System.getProperty("http.proxyPort", "80"));
41          this.proxyUser = System.getProperty("http.proxyUser");
42          this.proxyPassword = System.getProperty("http.proxyPassword");
43          this.nonProxyPattern = createNonProxyPattern(System.getProperty("http.nonProxyHosts"));
44      }
45  
46      /**
47       * Enables the creation of a proxy configuration using
48       * custom values.
49       *
50       * @param proxyHost the host name of the proxy
51       * @param proxyPort the port of the proxy
52       * @param proxyUser the username for the proxy
53       * @param proxyPassword the password for the proxy
54       * @param nonProxyHosts
55       */
56      public ProxyConfiguration(String proxyHost,
57                                int proxyPort,
58                                String proxyUser,
59                                String proxyPassword,
60                                String nonProxyHosts) {
61          this.proxyHost = proxyHost;
62          this.proxyPort = proxyPort;
63          this.proxyUser = proxyUser;
64          this.proxyPassword = proxyPassword;
65          this.nonProxyPattern = createNonProxyPattern(nonProxyHosts);
66      }
67  
68      /**
69       * Mimics <a href=
70       * "http://java.sun.com/javase/6/docs/technotes/guides/net/proxies.html">
71       * the java nonproxyhost syntax</a> and turns the nonproxyhost expression
72       * into a regex pattern. Basic idea taken from the HttpImpl class of the
73       * Liferay Portal software.
74       *
75       * @param nonProxyHosts
76       *        the String that contains the http.nonProxyHosts value. If this
77       *        value is empty or invalid, null is returned. If the value is
78       *        valid, a pattern is retured.
79       * @return the Pattern or null.
80       */
81      protected Pattern createNonProxyPattern(String nonProxyHosts) {
82          if (nonProxyHosts == null || nonProxyHosts.equals("")) return null;
83  
84          // "*.fedora-commons.org" -> ".*?\.fedora-commons\.org"
85          nonProxyHosts = nonProxyHosts.replaceAll("\\.", "\\\\.").replaceAll("\\*", ".*?");
86  
87          // a|b|*.c -> (a)|(b)|(.*?\.c)
88          nonProxyHosts = "(" + nonProxyHosts.replaceAll("\\|", ")|(") + ")";
89  
90          try {
91              return Pattern.compile(nonProxyHosts);
92  
93              //we don't want to bring down the whole server by misusing the nonProxy pattern
94              //therefore the error is logged and the web client moves on.
95          } catch (Exception e) {
96              logger.error("Creating the nonProxyHosts pattern failed for http.nonProxyHosts="
97                              + nonProxyHosts, e);
98              return null;
99          }
100     }

134     /**
135      * Checks whether a proxy has been configured and the given host is not in
136      * the nonProxyHost list or the nonProxyList is empty.
137      *
138      * @param host
139      *        the host to be matched
140      * @return true if the host satifies the above stated condition, otherwise
141      *         false.
142      */
143     public boolean isHostProxyable(String host) {
144         return getProxyHost() != null
145                 && getProxyPort() > 0
146                 && (nonProxyPattern == null || !nonProxyPattern.matcher(host)
147                         .matches());
148     }
149 
150     /**
151      * Checks whether the proxy credentials are valid.
152      * Username and password must be non-null and non-empty to qualify.
153      *
154      * @return true if the credentials are valid, false otherwise
155      */
156     public boolean hasValidCredentials() {
157         return getProxyUser() != null && !getProxyUser().equals("")
158                 && getProxyPassword() != null && !getProxyPassword().equals("");
159     }
