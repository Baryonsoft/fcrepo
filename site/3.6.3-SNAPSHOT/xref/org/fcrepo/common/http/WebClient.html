64      private void configureConnectionManager(){
65          logger.debug("User-Agent is '" + wconfig.getUserAgent() + "'");
66          logger.debug("Max total connections is " + wconfig.getMaxTotalConn());
67          logger.debug("Max connections per host is " + wconfig.getMaxConnPerHost());
68          logger.debug("Connection timeout is " + wconfig.getTimeoutSecs());
69          logger.debug("Socket Connection timeout is " + wconfig.getSockTimeoutSecs());
70          logger.debug("Follow redirects? " + wconfig.getFollowRedirects());
71          logger.debug("Max number of redirects to follow is " + wconfig.getMaxRedirects());
72  
73          m_cManager = new MultiThreadedHttpConnectionManager();
74          m_cManager.getParams().setDefaultMaxConnectionsPerHost(wconfig.getMaxConnPerHost());
75          m_cManager.getParams().setMaxTotalConnections(wconfig.getMaxTotalConn());
76          m_cManager.getParams().setConnectionTimeout(wconfig.getTimeoutSecs() * 1000);
77          m_cManager.getParams().setSoTimeout(wconfig.getSockTimeoutSecs() * 1000);
78      }

84      public HttpClient getHttpClient(String hostOrURL,
85                                      UsernamePasswordCredentials creds)
86              throws IOException, ConnectTimeoutException {
87  
88          String host = null;
89  
90          if (hostOrURL != null) {
91              if (hostOrURL.indexOf("/") != -1) {
92                  URL url = new URL(hostOrURL);
93                  host = url.getHost();
94              } else {
95                  host = hostOrURL;
96              }
97          }
98  
99          HttpClient client = new HttpClient(m_cManager);
100         if (host != null && creds != null) {
101             client.getState().setCredentials(new AuthScope(host,
102                                                            AuthScope.ANY_PORT),
103                                              creds);
104             client.getParams().setAuthenticationPreemptive(true);
105         }
106 
107         if (proxy.isHostProxyable(host)) {
108             client.getHostConfiguration().setProxy(proxy.getProxyHost(),
109                                                    proxy.getProxyPort());
110             if (proxy.hasValidCredentials()) {
111                 client.getState().setProxyCredentials(new AuthScope(proxy.getProxyHost(),
112                                                            proxy.getProxyPort(),
113                                                            null),
114                                                       new UsernamePasswordCredentials(proxy.getProxyUser(),
115                                                                              proxy.getProxyPassword()));
116             }
117         }
118         return client;
119     }

137     /**
138      * Get an HTTP resource with the response as an InputStream, given a URL. If
139      * FOLLOW_REDIRECTS is true, up to MAX_REDIRECTS redirects will be followed.
140      * Note that if credentials are provided, for security reasons they will
141      * only be provided to the FIRST url in a chain of redirects. Note that if
142      * the HTTP response has no body, the InputStream will be empty. The success
143      * of a request can be checked with getResponseCode(). Usually you'll want
144      * to see a 200. See http://www.w3.org/Protocols/rfc2616/rfc2616-sec10.html
145      * for other codes.
146      *
147      * @param url
148      *        A URL that we want to do an HTTP GET upon
149      * @param failIfNotOK
150      *        boolean value indicating if an exception should be thrown if we do
151      *        NOT receive an HTTP 200 response (OK)
152      * @return HttpInputStream the HTTP response
153      * @throws IOException
154      */
155     public HttpInputStream get(String url,
156                                boolean failIfNotOK,
157                                UsernamePasswordCredentials creds)
158             throws IOException {
159 
160         HttpClient client;
161         GetMethod getMethod = new GetMethod(url);
162 
163         if (wconfig.getUserAgent() != null) {
164             getMethod.setRequestHeader("User-Agent", wconfig.getUserAgent());
165         }
166         if (creds != null && creds.getUserName() != null
167                 && creds.getUserName().length() > 0) {
168             client = getHttpClient(url, creds);
169             getMethod.setDoAuthentication(true);
170         } else {
171             client = getHttpClient(url);
172         }
173 
174         HttpInputStream in = new HttpInputStream(client, getMethod, url);
175         int status = in.getStatusCode();
176         if (failIfNotOK) {
177             if (status != 200) {
178                 //if (followRedirects && in.getStatusCode() == 302){
179                 if (wconfig.getFollowRedirects() && 300 <= status && status <= 399) {
180                     int count = 1;
181                     while (300 <= status && status <= 399
182                             && count <= wconfig.getMaxRedirects()) {
183                         if (in.getResponseHeader("location") == null) {
184                             throw new IOException("Redirect HTTP response provided no location header.");
185                         }
186                         url = in.getResponseHeader("location").getValue();
187                         in.close();
188                         getMethod = new GetMethod(url);
189                         if (wconfig.getUserAgent() != null) {
190                             getMethod
191                                     .setRequestHeader("User-Agent", wconfig.getUserAgent());
192                         }
193                         in = new HttpInputStream(client, getMethod, url);
194                         status = in.getStatusCode();
195                         count++;
196                     }
197                     if (300 <= status && status <= 399) {
198                         in.close();
199                         throw new IOException("Too many redirects");
200                     } else if (status != 200) {
201                         in.close();
202                         throw new IOException("Request failed ["
203                                 + in.getStatusCode() + " " + in.getStatusText()
204                                 + "]");
205                     }
206                     // redirect was successful!
207                 } else {
208                     try {
209                         throw new IOException("Request failed ["
210                                 + in.getStatusCode() + " " + in.getStatusText()
211                                 + "]");
212                     } finally {
213                         try {
214                             in.close();
215                         } catch (Exception e) {
216                             logger.error("Can't close InputStream: "
217                                     + e.getMessage());
218                         }
219                     }
220                 }
221             }
222         }
223         return in;
224     }

231     public String getResponseAsString(String url,
232                                       boolean failIfNotOK,
233                                       UsernamePasswordCredentials creds)
234             throws IOException {
235 
236         InputStream in = get(url, failIfNotOK, creds);
237 
238         // Convert the response into a String.
239         try {
240             BufferedReader reader =
241                     new BufferedReader(new InputStreamReader(in));
242             StringBuffer buffer = new StringBuffer();
243             String line = reader.readLine();
244             while (line != null) {
245                 buffer.append(line + "\n");
246                 line = reader.readLine();
247             }
248             return buffer.toString();
249         } finally {
250             try {
251                 in.close();
252             } catch (Exception e) {
253                 logger.error("Can't close InputStream: " + e.getMessage());
254             }
255         }
256     }
