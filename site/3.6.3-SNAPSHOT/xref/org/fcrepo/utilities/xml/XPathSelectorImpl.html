65      @Override
66      public Boolean selectBoolean(Node node, String xpath, Boolean defaultValue) {
67          if (defaultValue == null || Boolean.TRUE.equals(defaultValue)) {
68              // Using QName.BOOLEAN will always return false if it is not found
69              // therefore we must try and look it up as a string
70              String tmp = selectString(node, xpath, null);
71              if (tmp == null) {
72                  return defaultValue;
73              }
74              return Boolean.parseBoolean(tmp);
75          } else {
76              // The defaultValue is false so we can always just return what
77              // we take from the XPath expression
78              return (Boolean) selectObject(node, xpath, XPathConstants.BOOLEAN);
79          }
80      }

87      @Override
88      public String selectString(Node node, String xpath, String defaultValue) {
89          if ("".equals(defaultValue)) {
90              // By default the XPath engine will return an empty string
91              // if it is unable to find the requested path
92              return (String) selectObject(node, xpath, XPathConstants.STRING);
93          }
94  
95          Node n = selectNode(node, xpath);
96          if (n == null) {
97              return defaultValue;
98          }
99  
100         // FIXME: Can we avoid running the xpath twice?
101         //        The local expression cache helps, but anyway...
102         return (String) selectObject(node, xpath, XPathConstants.STRING);
103     }

120     private Object selectObject(Node dom, String xpath, QName returnType) {
121         Object retval = null;
122 
123         try {
124             // Get the compiled xpath from the cache or compile and
125             // cache it if we don't have it
126             XPathExpression exp = cache.get(xpath);
127             if (exp == null) {
128                 synchronized (xpathCompiler) {
129                     if (nsContext != null) {
130                         xpathCompiler.setNamespaceContext(nsContext);
131                     }
132                     exp = xpathCompiler.compile(xpath);
133                 }
134                 cache.put(xpath, exp);
135             }
136 
137             retval = exp.evaluate(dom, returnType);
138         } catch (NullPointerException e) {
139             log.debug(String.format(
140                     "NullPointerException when extracting XPath '%s' on " +
141                     "element type %s. Returning null",
142                     xpath, returnType.getLocalPart()), e);
143         } catch (XPathExpressionException e) {
144             log.warn(String.format(
145                     "Error in XPath expression '%s' when selecting %s: %s",
146                     xpath, returnType.getLocalPart(), e.getMessage()), e);
147         }
148 
149         return retval;
150     }
