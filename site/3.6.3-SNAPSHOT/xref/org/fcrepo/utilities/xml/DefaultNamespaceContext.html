7   /**
8    * Default implementation of a {@link NamespaceContext}
9    */
10  public class DefaultNamespaceContext implements NamespaceContext {
11  
12      /* URI -> prefixes map */
13      private Map<String, Collection<String>> namespace;
14  
15      /* prefix -> URI map */
16      private Map<String, String> prefixes;

18      private String defaultNamespaceURI;

21      /**
22       * Constructs a NamespaceContext with no default namespace.
23       * <p/>
24       * Beware that the default namespace can only be set during
25       * construction.
26       */
27      public DefaultNamespaceContext() {
28          this(null);
29      }

31      /**
32       * Constructs a NamespaceContext with a default namespace.
33       * <p/>
34       *
35       * @param defaultNamespaceURI , the default namespace for this context.
36       */
37      public DefaultNamespaceContext(String defaultNamespaceURI) {
38          namespace = new HashMap<String, Collection<String>>();
39          prefixes = new HashMap<String, String>();
40          this.defaultNamespaceURI = defaultNamespaceURI;
41  
42          namespace.put(XMLConstants.XMLNS_ATTRIBUTE_NS_URI,
43                        Arrays.asList(XMLConstants.XMLNS_ATTRIBUTE));
44  
45          prefixes.put(XMLConstants.XMLNS_ATTRIBUTE,
46                       XMLConstants.XMLNS_ATTRIBUTE_NS_URI);
47  
48          namespace.put(XMLConstants.XML_NS_URI,
49                        Arrays.asList(XMLConstants.XML_NS_PREFIX));
50  
51          prefixes.put(XMLConstants.XML_NS_PREFIX,
52                       XMLConstants.XML_NS_URI);
53      }
54  
55      public DefaultNamespaceContext(String defaultNamespaceURI,
56                                     String... nsContext) {
57          this(defaultNamespaceURI);
58  
59          if (nsContext.length % 2 != 0) {
60              throw new IllegalArgumentException("Odd number of arguments. " +
61                                                 "Prefix/URI pairs must match up");
62          }
63  
64          for (int i = 0; i < nsContext.length; i += 2) {
65              setNameSpace(nsContext[i + 1], nsContext[i]);
66          }
67      }

69      /**
70       * Set or add a namespace to the context and associated it with a prefix.
71       * <p/>
72       * A given prefix can only be associated with one namespace in the context.
73       * A namespace can have multiple prefixes.
74       * <p/>
75       * The prefixes: {@code xml}, and {@code xmlns} are reserved and
76       * predefined in any context.
77       *
78       * @param namespaceURL the namespace uri
79       * @param prefix       the prifix to registere with the uri
80       * @throws IllegalArgumentException thrown when trying to assign a
81       *                                  namespace to a reserved prefix
82       */
83      public void setNameSpace(String namespaceURL, String prefix)
84              throws IllegalArgumentException {
85          Collection<String> s = namespace.get(namespaceURL);
86  
87          if (s == null) {
88              s = new HashSet<String>();
89          }
90  
91          s.add(prefix);
92          namespace.put(namespaceURL, s);
93          this.prefixes.put(prefix, namespaceURL);
94      }

96      @Override
97      public String getNamespaceURI(String prefix) {
98          if (prefix == null) {
99              throw new IllegalArgumentException();
100         }
101 
102         if (prefix.equals(XMLConstants.DEFAULT_NS_PREFIX)
103             && this.defaultNamespaceURI != null) {
104             return this.defaultNamespaceURI;
105         }
106 
107 
108         if (!this.prefixes.containsKey(prefix)) {
109             return XMLConstants.NULL_NS_URI;
110         }
111 
112         return this.prefixes.get(prefix);
113 
114     }

116     @Override
117     public String getPrefix(String namespaceURI) {
118         if (namespaceURI == null) {
119             throw new IllegalArgumentException();
120         }
121 
122         if (namespaceURI.equals(defaultNamespaceURI)) {
123             return XMLConstants.DEFAULT_NS_PREFIX;
124         }
125 
126         Collection<String> s = namespace.get(namespaceURI);
127         if (s != null && !s.isEmpty()) {
128             return s.iterator().next();
129         } else {
130             return null;
131         }
132     }

134     @Override
135     public Iterator getPrefixes(String namespaceURI) {
136 
137         if (namespaceURI == null) {
138             throw new IllegalArgumentException();
139         }
140 
141         if (namespaceURI.equals(XMLConstants.XML_NS_URI)) {
142             return new NonModifiableIterator(
143                     Arrays.asList(XMLConstants.XML_NS_PREFIX).iterator());
144         }
145         if (namespaceURI.equals(XMLConstants.XMLNS_ATTRIBUTE_NS_URI)) {
146             return new NonModifiableIterator(
147                     Arrays.asList(XMLConstants.XMLNS_ATTRIBUTE).iterator());
148         }
149         if (namespaceURI.equals(defaultNamespaceURI)) {
150             return new NonModifiableIterator(
151                     Arrays.asList(XMLConstants.DEFAULT_NS_PREFIX).iterator());
152         }
153 
154         Collection<String> s = namespace.get(namespaceURI);
155         if (s != null && !namespace.isEmpty()) {
156             return new NonModifiableIterator(s.iterator());
157         } else {
158             return new NonModifiableIterator(new HashSet().iterator());
159         }
160     }

162     /**
163      * This Iterator wraps any Iterator and makes the remove() method
164      * unsupported.<br>
165      *
166      * @author Hans Lund, State and University Library, Aarhus Denamrk.
167      * @version $Id: DefaultNamespaceContext.java,v 1.5 2007/10/04 13:28:21 te Exp $
168      * @see javax.xml.namespace.NamespaceContext#getPrefixes(String)
169      */
170     static class NonModifiableIterator implements Iterator {
171 
172         Iterator wrapped;
173 
174         NonModifiableIterator(Iterator iter) {
175             wrapped = iter;
176         }
177 
178         /**
179          * Returns <tt>true</tt> if the iteration has more elements. (In other
180          * words, returns <tt>true</tt> if <tt>next</tt> would return an element
181          * rather than throwing an exception.)
182          *
183          * @return <tt>true</tt> if the iterator has more elements.
184          */
185         public boolean hasNext() {
186             return wrapped.hasNext();
187         }
188 
189         /**
190          * Returns the next element in the iteration.  Calling this method
191          * repeatedly until the {@link #hasNext()} method returns false will
192          * return each element in the underlying collection exactly once.
193          *
194          * @return the next element in the iteration.
195          * @throws java.util.NoSuchElementException
196          *          iteration has no more elements.
197          */
198         public Object next() {
199             return wrapped.next();
200         }
201 
202         /**
203          * This method is not supported on this Iterator.
204          * <p/>
205          * Allways throws UnsupportedOperationException {@link javax.xml.namespace.NamespaceContext#getPrefixes(String)}
206          *
207          * @throws UnsupportedOperationException if the <tt>remove</tt>
208          *                                       operation is not supported by this Iterator.
209          */
210         public void remove() {
211             throw new UnsupportedOperationException("Conform to XML API please");
212         }
