11  /**
12   * Logger configuration utility methods.
13   * <p>
14   * This class provides convenience methods for configuring Logback when
15   * the default configuration files (logback.xml or logback-test.xml) are not
16   * expected to be found in the classpath of the running application.
17   * <p>
18   * Note:
19   * <ul>
20   *   <li> This class intentionally has no compile or run-time dependencies on
21   *        Logback.  If Logback is not in the classpath, calling these methods
22   *        will have no effect on the application's logging behavior.</li>
23   *   <li> If the <code>logback.configurationFile</code> system property is
24   *        already defined, calling these methods will have no effect.</li>
25   * </ul>
26   *
27   * @author Chris Wilper
28   */
29  public abstract class LogConfig {
30  
31      private static final String CONFIG_FILE_PROPERTY = "logback.configurationFile";
32  
33      /**
34       * Initializes logging with the given file.
35       *
36       * @param configFile the logback.xml file to use.
37       */
38      public static void initFromFile(File configFile) {
39          if (!alreadyConfigured()) {
40              System.setProperty(CONFIG_FILE_PROPERTY, configFile.getPath());
41          }
42      }
43  
44      /**
45       * Initializes logging to the console (standard error) with minimal logging.
46       * <p>
47       * For all categories, only WARN and ERROR messages will be printed,
48       * and the messages will be of the form "LEVEL time (classname) message"
49       */
50      public static void initMinimal() {
51          if (!alreadyConfigured()) {
52              initFromFile(createTempConfigFile("WARN"));
53          }
54      }
55  
56      private static boolean alreadyConfigured() {
57          return System.getProperty(CONFIG_FILE_PROPERTY) != null;
58      }
59  
60      private static File createTempConfigFile(String level) {
61          PrintWriter writer = null;
62          try {
63              File tempFile = File.createTempFile("fedora-logback-minimal", null);
64              tempFile.deleteOnExit();
65              writer = new PrintWriter(new OutputStreamWriter(
66                      new FileOutputStream(tempFile)));
67              writer.println("<?xml version=\"1.0\" encoding=\"UTF-8\"?>");
68              writer.println("<configuration>");
69              writer.println("  <appender name=\"STDERR\" class=\"ch.qos.logback.core.ConsoleAppender\">");
70              writer.println("    <Target>System.err</Target>");
71              writer.println("    <layout class=\"ch.qos.logback.classic.PatternLayout\">");
72              writer.println("      <Pattern>%p %d{HH:mm:ss.SSS} (%c{0}\\\\) %m%n</Pattern>");
73              writer.println("    </layout>");
74              writer.println("  </appender>");
75              writer.println("  <root additivity=\"false\" level=\"" + level + "\">");
76              writer.println("    <appender-ref ref=\"STDERR\"/>");
77              writer.println("  </root>");
78              writer.println("</configuration>");
79              return tempFile;
80          } catch (IOException e) {
81              throw new FaultException("Error creating temporary log config file", e);
82          } finally {
83              if (writer != null) {
84                  writer.close();
85              }
86          }
87      }
