69  	/**
70  	 * Unpacks the contents of the FEDORA_HOME directory from the Distribution.
71  	 * 
72  	 * @throws InstallationFailedException
73  	 */
74  	private void unpack() throws InstallationFailedException {
75  		System.out.println("Preparing FEDORA_HOME...");
76  
77  		if (!_installDir.exists() && !_installDir.mkdirs()) {
78  			throw new InstallationFailedException(
79  					"Unable to create FEDORA_HOME: "
80  							+ _installDir.getAbsolutePath());
81  		}
82  		if (!_installDir.isDirectory()) {
83  			throw new InstallationFailedException(_installDir.getAbsolutePath()
84  					+ " is not a directory");
85  		}
86  		try {
87  			Zip.unzip(_dist.get(Distribution.FEDORA_HOME), _installDir);
88  			setScriptsExecutable(new File(_installDir, "client"
89  					+ File.separator + "bin"));
90  
91  			File serverDir = new File(_installDir, "server");
92  			if (_clientOnlyInstall) {
93  				FileUtils.delete(serverDir);
94  			} else {
95  				setScriptsExecutable(new File(serverDir, "bin"));
96  			}
97  		} catch (IOException e) {
98  			throw new InstallationFailedException(e.getMessage(), e);
99  		}
100 	}

159 			dbPoolName = "localPostgreSQLPool";

164 		props.put("module.org.fcrepo.server.storage.DOManager:storagePool",

166 		props.put("module.org.fcrepo.server.search.FieldSearch:connectionPool",

168 		props.put(
169 				"module.org.fcrepo.server.storage.ConnectionPoolManager:poolNames",

171 		props.put(

180 				_opts.getValue(InstallOptions.DATABASE_USERNAME));
181 		props.put("datastore." + dbPoolName + ":dbPassword",
182 				_opts.getValue(InstallOptions.DATABASE_PASSWORD));
183 		props.put("datastore." + dbPoolName + ":jdbcDriverClass",
184 				_opts.getValue(InstallOptions.DATABASE_DRIVERCLASS));
185

187 			props.put(
188 					"module.org.fcrepo.server.security.Authorization:ENFORCE-MODE",
189 					"enforce-policies");
190 		} else {
191 			props.put(
192 					"module.org.fcrepo.server.security.Authorization:ENFORCE-MODE",
193 					"permit-all-requests");
194 		}
195 
196 		if (_opts.getBooleanValue(InstallOptions.RI_ENABLED, true)) {
197 			props.put(
198 					"module.org.fcrepo.server.resourceIndex.ResourceIndex:level",
199 					String.valueOf(ResourceIndex.INDEX_LEVEL_ON));

241 				config.getModuleConfigurations().remove(mConfig);
242 			}
243 
244 			config.serialize(new FileOutputStream(fcfg));
245 		} catch (IOException e) {
246 			throw new InstallationFailedException(e.getMessage(), e);
247 		}

250 	private void configureAkubra() throws InstallationFailedException {
251 	    if (usingAkubra()) {
252     		// Rewrite server/config/akubra-llstore.xml replacing the
253     		// /tmp/[object|datastream]Store constructor-arg values
254     		// with $FEDORA_HOME/data/[object|datastream]Store
255     		BufferedReader reader = null;
256     		PrintWriter writer = null;
257     		try {
258     			File file = new File(_installDir,
259     					"server/config/spring/akubra-llstore.xml");
260     			reader = new BufferedReader(new InputStreamReader(
261     					new FileInputStream(file), "UTF-8"));
262     
263     			File dataDir = new File(_installDir, "data");
264     			String oPath = dataDir.getPath() + File.separator + "objectStore";
265     			String dPath = dataDir.getPath() + File.separator
266     					+ "datastreamStore";
267     			StringBuilder xml = new StringBuilder();
268     
269     			String line = reader.readLine();
270     			while (line != null) {
271     				if (line.indexOf("/tmp/objectStore") != -1) {
272     					line = "    <constructor-arg value=\"" + oPath + "\"/>";
273     				} else if (line.indexOf("/tmp/datastreamStore") != -1) {
274     					line = "    <constructor-arg value=\"" + dPath + "\"/>";
275     				}
276     				xml.append(line + "\n");
277     				line = reader.readLine();
278     			}
279     			reader.close();
280     
281     			writer = new PrintWriter(new OutputStreamWriter(
282     					new FileOutputStream(file), "UTF-8"));
283     			writer.print(xml.toString());
284     			writer.close();
285     		} catch (IOException e) {
286     			IOUtils.closeQuietly(reader);
287     			IOUtils.closeQuietly(writer);
288     			throw new InstallationFailedException(e.getClass().getName() + ":"
289     					+ e.getMessage());
290     		}
291 	    } else {
292 	        // remove the stub akubra configuration
293             File file = new File(_installDir,
294                     "server/config/spring/akubra-llstore.xml");
295             file.delete();
296             if (file.exists()){
297                 throw new InstallationFailedException("Could not remove a conflicting config: akubra-llstore.xml");
298             }
299 	    }
300 	}

302 	private void configureSpringProperties() throws InstallationFailedException {
303 		Properties springProps = new Properties();
304 
305 		/* Set up ssl configuration */
306 		springProps.put("fedora.port",
307 				_opts.getValue(InstallOptions.TOMCAT_HTTP_PORT, "8080"));
308 		if (_opts.getBooleanValue(InstallOptions.SSL_AVAILABLE, false)) {
309 			springProps.put("fedora.port.secure",
310 					_opts.getValue(InstallOptions.TOMCAT_SSL_PORT, "8443"));
311 		} else {
312 			springProps.put("fedora.port.secure",
313 					_opts.getValue(InstallOptions.TOMCAT_HTTP_PORT, "8080"));
314 		}
315 
316 		springProps
317 				.put("security.ssl.api.access",
318 						_opts.getBooleanValue(InstallOptions.APIA_SSL_REQUIRED,
319 								false) ? "REQUIRES_SECURE_CHANNEL"
320 								: "ANY_CHANNEL");
321 		springProps
322 				.put("security.ssl.api.management",
323 						_opts.getBooleanValue(InstallOptions.APIM_SSL_REQUIRED,
324 								false) ? "REQUIRES_SECURE_CHANNEL"
325 								: "ANY_CHANNEL");
326 		springProps.put("security.ssl.api.default", "ANY_CHANNEL");
327 
328 		springProps.put("security.fesl.authN.jaas.apia.enabled",
329 				_opts.getValue(InstallOptions.APIA_AUTH_REQUIRED, "false"));
330 
331 		springProps.put("security.fesl.authZ.enabled",
332 				_opts.getValue(InstallOptions.FESL_AUTHZ_ENABLED, "false"));
333 
334 		/* Set up authN, authZ filter configuration */
335 		StringBuilder filters = new StringBuilder();
336         if (_opts.getBooleanValue(InstallOptions.FESL_AUTHN_ENABLED, true)) {
337 			filters.append("AuthFilterJAAS");
338 		} else {
339 			filters.append("SetupFilter,XmlUserfileFilter,EnforceAuthnFilter,FinalizeFilter");
340 		}
341 
342 		if (_opts.getBooleanValue(InstallOptions.FESL_AUTHZ_ENABLED, false)) {
343 			filters.append(",PEPFilter");
344 		}
345 
346 		springProps.put("security.auth.filters", filters.toString());
347 
348 		FileOutputStream out = null;
349 		try {
350 			out = new FileOutputStream(new File(_installDir,
351 					"server/config/spring/web/web.properties"));
352 			springProps.store(out, "Spring override properties");
353 		} catch (IOException e) {
354 			throw new InstallationFailedException(e.getMessage(), e);
355 		} finally {
356 			IOUtils.closeQuietly(out);
357 		}
358 	}

361 	 * This is an ugly workaround for the fact that spring security namespace
362 	 * config does not support property substitution into lists of beans. It is
363 	 * also ugly because
364 	 */
365 	private void configureSpringAuth() throws InstallationFailedException {
366 		String PATTERN = "${security.auth.filters}";
367 		String PATTERN_APIA = "${security.auth.filters.apia}";
368 		String PATTERN_REST = "${security.auth.filters.rest}";
369 
370 		boolean fesl_authn_enabled = _opts.getBooleanValue(
371 InstallOptions.FESL_AUTHN_ENABLED, true);
372 		boolean apia_auth_required = _opts.getBooleanValue(
373 				InstallOptions.APIA_AUTH_REQUIRED, false);
374 		boolean upstream_auth_enabled = _opts.getBooleanValue(
375 				InstallOptions.UPSTREAM_AUTH_ENABLED, false);
376 
377 		StringBuilder filters = new StringBuilder();
378 		StringBuilder filters_apia = new StringBuilder();
379 		StringBuilder filters_rest = new StringBuilder();
380 
381 		boolean needsbugFix = false;
382 
383 		if (fesl_authn_enabled) {
384 			filters.append("AuthFilterJAAS");
385 			filters_apia.append("AuthFilterJAAS");
386 			filters_rest.append("AuthFilterJAAS");
387 		} else if (upstream_auth_enabled) {
388 			// use upstream auth filter and nothing else
389 			String upstreamAuthFilter = "UpstreamAuthFilter";
390 			filters.append(upstreamAuthFilter);
391 			filters_apia.append(upstreamAuthFilter);
392 			filters_rest.append(upstreamAuthFilter);
393 		}
394 		else {
395 			filters.append("SetupFilter,XmlUserfileFilter,EnforceAuthnFilter,FinalizeFilter");
396 			if (apia_auth_required) {
397 				filters_apia
398 						.append("SetupFilter,XmlUserfileFilter,EnforceAuthnFilter,FinalizeFilter");
399 				filters_rest
400 						.append("SetupFilter,XmlUserfileFilter,EnforceAuthnFilter,FinalizeFilter");
401 			} else {
402 				filters_apia.append("");
403 				filters_rest
404 						.append("SetupFilter,XmlUserfileFilter,RestApiAuthnFilter,FinalizeFilter");
405 			}
406 
407 			needsbugFix = true;
408 		}
409 
410 		if (_opts.getBooleanValue(InstallOptions.FESL_AUTHZ_ENABLED, false)) {
411 			filters.append(",PEPFilter");
412 			filters_apia.append(",PEPFilter");
413 			filters_rest.append(",PEPFilter");
414 			copyFESLConfigs();
415 		}
416 
417 		FileInputStream springConfig = null;
418 		PrintWriter writer = null;
419 		try {
420 			File xmlFile = new File(_installDir,
421 					"server/config/spring/web/security.xml");
422 			springConfig = new FileInputStream(xmlFile);
423 			String content = IOUtils.toString(springConfig)
424 					.replace(PATTERN, filters)
425 					.replace(PATTERN_APIA, filters_apia)
426 					.replace(PATTERN_REST, filters_rest);
427 
428 			if (!needsbugFix) {
429 				/* Delete classic authN bugfix when not applicable */
430 				content = content.replaceFirst("(?s)<!-- BUG.+?/BUG -->", "");
431 			}
432 
433 			springConfig.close();
434 
435 			writer = new PrintWriter(new OutputStreamWriter(
436 					new FileOutputStream(xmlFile), "UTF-8"));
437 			writer.print(content);
438 			writer.close();
439 		} catch (Exception e) {
440 			IOUtils.closeQuietly(springConfig);
441 			IOUtils.closeQuietly(writer);
442 			throw new InstallationFailedException(e.getMessage(), e);
443 		}
444 	}

446 	private void copyFESLConfigs() throws InstallationFailedException {
447 		File feslWebDir = new File(_installDir, "server/config/spring/fesl/web");
448 		File feslModuleDir = new File(_installDir,
449 				"server/config/spring/fesl/module");
450 		File webDir = new File(_installDir, "server/config/spring/web");
451 		File moduleDir = new File(_installDir, "server/config/spring");
452 
453 		for (File beanDef : feslWebDir.listFiles()) {
454 			if (beanDef.isFile()) {
455 				FileReader reader = null;
456 				FileWriter writer = null;
457 				try {
458 					File copy = new File(webDir, beanDef.getName());
459 					reader = new FileReader(beanDef);
460 					writer = new FileWriter(copy);
461 					IOUtils.copy(reader, writer);
462 					writer.flush();
463 				} catch (Exception e) {
464 					throw new InstallationFailedException(e.getMessage(), e);
465 				} finally {
466 					IOUtils.closeQuietly(writer);
467 					IOUtils.closeQuietly(reader);
468 				}
469 			}
470 		}
471 		for (File beanDef : feslModuleDir.listFiles()) {
472 			if (beanDef.isFile()) {
473 				FileReader reader = null;
474 				FileWriter writer = null;
475 				try {
476 					File copy = new File(moduleDir, beanDef.getName());
477 					reader = new FileReader(beanDef);
478 					writer = new FileWriter(copy);
479 					IOUtils.copy(reader, writer);
480 					writer.flush();
481 				} catch (Exception e) {
482 					throw new InstallationFailedException(e.getMessage(), e);
483 				} finally {
484 					IOUtils.closeQuietly(writer);
485 					IOUtils.closeQuietly(reader);
486 				}
487 			}
488 		}
489 	}

493 		if (_opts.getBooleanValue(InstallOptions.TEST_SPRING_CONFIGS, false)) {
494 			FileInputStream springConfig = null;
495 			PrintWriter writer = null;
496 			try {
497 				File springDir = new File(_installDir, "server/config/spring");
498 				for (File file : springDir.listFiles()) {
499 					if (file.isFile()) {

502 						content = content.replaceAll(
503 								"(?s)<!-- TESTONLY(.+?)/TESTONLY -->", "$1");
504 						springConfig.close();
505 
506 						writer = new PrintWriter(new OutputStreamWriter(
507 								new FileOutputStream(file), "UTF-8"));
508 						writer.print(content);
509 						writer.close();
510 					}
511 				}
512 			} catch (Exception e) {
513 				IOUtils.closeQuietly(springConfig);
514 				IOUtils.closeQuietly(writer);
515 				throw new InstallationFailedException(e.getMessage(), e);
516 			}
517 		}

539 	private void configureBeSecurity() throws InstallationFailedException {
540 		System.out.println("\tInstalling beSecurity");
541 		File beSecurity = new File(_installDir, "/server/config/beSecurity.xml");
542 		boolean apiaAuth = _opts.getBooleanValue(
543 				InstallOptions.APIA_AUTH_REQUIRED, false);
544 		boolean apiaSSL = _opts.getBooleanValue(
545 				InstallOptions.APIA_SSL_REQUIRED, false);
546 		// boolean apimSSL =
547 		// _opts.getBooleanValue(InstallOptions.APIM_SSL_REQUIRED, false);
548 
549 		String[] ipList;
550 		String host = _opts.getValue(InstallOptions.FEDORA_SERVERHOST);
551 		if (host != null && host.length() != 0
552 				&& !(host.equals("localhost") || host.equals("127.0.01"))) {
553 			ipList = new String[] { "127.0.0.1", getHost() };
554 		} else {
555 			ipList = new String[] { "127.0.0.1" };
556 		}
557 
558 		PrintWriter pwriter;
559 		try {
560 			pwriter = new PrintWriter(new FileOutputStream(beSecurity));
561 		} catch (FileNotFoundException e) {
562 			throw new InstallationFailedException(e.getMessage(), e);
563 		}
564 		BESecurityConfig becfg = new BESecurityConfig();
565 
566 		becfg.setDefaultConfig(new DefaultRoleConfig());
567 		becfg.setInternalBasicAuth(new Boolean(apiaAuth));
568 		becfg.setInternalIPList(ipList);
569 		becfg.setInternalPassword("changeme");
570 		becfg.setInternalSSL(new Boolean(apiaSSL));
571 		becfg.setInternalUsername("fedoraIntCallUser");
572 		becfg.write(true, true, pwriter);
573 		pwriter.close();
574 	}

576 	private String getHost() throws InstallationFailedException {
