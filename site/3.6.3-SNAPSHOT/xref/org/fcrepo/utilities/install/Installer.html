32      public Installer(Distribution dist, InstallOptions opts) {
33          _dist = dist;

36          installDir = new File(fedoraHome, "install" + File.separator);

39      /**
40       * Install the distribution based on the options.
41       */
42      public void install() throws InstallationFailedException {
43          installDir.mkdirs();
44  
45          // Write out the install options used to a properties file in the install directory
46          try {
47              OutputStream out =
48                      new FileOutputStream(new File(installDir,
49                                                    "install.properties"));
50              _opts.dump(out);
51              out.close();
52          } catch (Exception e) {
53              throw new InstallationFailedException(e.getMessage(), e);
54          }
55          new FedoraHome(_dist, _opts).install();
56  
57          if (!_opts.getValue(InstallOptions.INSTALL_TYPE)
58                  .equals(InstallOptions.INSTALL_CLIENT)) {
59              Container container = ContainerFactory.getContainer(_dist, _opts);
60              container.install();
61              container.deploy(buildWAR());
62              if (_opts.getBooleanValue(InstallOptions.DEPLOY_LOCAL_SERVICES,
63                                        true)) {
64                  deployLocalService(container, Distribution.FOP_WAR);
65                  deployLocalService(container, Distribution.IMAGEMANIP_WAR);
66                  deployLocalService(container, Distribution.SAXON_WAR);
67                  deployLocalService(container, Distribution.DEMO_WAR);
68              }
69  
70              Database database = new Database(_dist, _opts);
71              database.install();
72          }
73  
74          System.out.println("Installation complete.");
75          if (!_opts.getValue(InstallOptions.INSTALL_TYPE)
76                  .equals(InstallOptions.INSTALL_CLIENT)
77                  && _opts.getValue(InstallOptions.SERVLET_ENGINE)
78                          .equals(InstallOptions.OTHER)) {
79              System.out
80                      .println("\n"
81                              + "----------------------------------------------------------------------\n"
82                              + "The Fedora Installer cannot automatically deploy the Web ARchives to  \n"
83                              + "the selected servlet container. You must deploy the WAR files         \n"
84                              + "manually. You can find fedora.war plus several sample back-end        \n"
85                              + "services and a demonstration object package in:                       \n"
86                              + "\t" + fedoraHome.getAbsolutePath()
87                              + File.separator + "install");
88          }
89          System.out
90                  .println("\n"
91                          + "----------------------------------------------------------------------\n"
92                          + "Before starting Fedora, please ensure that any required environment\n"
93                          + "variables are correctly defined\n"
94                          + "\t(e.g. FEDORA_HOME, JAVA_HOME, JAVA_OPTS, CATALINA_HOME).\n"
95                          + "For more information, please consult the Installation & Configuration\n"
96                          + "Guide in the online documentation.\n"
97                          + "----------------------------------------------------------------------\n");
98      }

100     private File buildWAR() throws InstallationFailedException {
101         String fedoraWarName = _opts.getValue(InstallOptions.FEDORA_APP_SERVER_CONTEXT) ;
102         System.out.println("Preparing " + fedoraWarName + ".war...");
103         try {
104             File outputFile = new File(installDir, fedoraWarName + ".war");
105             String driver = _opts.getValue(InstallOptions.DATABASE_DRIVER);
106             if (driver.equals(InstallOptions.INCLUDED)) {
107                 FileUtils.copy(_dist.get(Distribution.FEDORA_WAR),
108                         new FileOutputStream(outputFile));
109             } else {
110                 addLibrary(_dist.get(Distribution.FEDORA_WAR),
111                         driver, outputFile);
112             }
113             return outputFile;
114         } catch (IOException e) {
115             throw new InstallationFailedException(e.getMessage(), e);
116 		}
117     }
118     
119     private void addLibrary(InputStream inputStream, String libraryPath,
120             File outputFile) throws IOException {
121         // unzip, add file, re-zip, and remove staging dir
122         File stagingDir = new File(installDir, "fedorawar");
123         stagingDir.mkdirs();
124         Zip.unzip(inputStream, stagingDir);
125         File sourceFile = new File(libraryPath);
126         File destFile = new File(stagingDir, "WEB-INF/lib/" + sourceFile.getName());
127         FileUtils.copy(sourceFile, destFile);
128         Zip.zip(outputFile, stagingDir.listFiles());
129         FileUtils.delete(stagingDir);
130     }

132     private void deployLocalService(Container container, String filename)
133             throws InstallationFailedException {
134         try {
135             File war = new File(installDir, filename);
136             if (!FileUtils.copy(_dist.get(filename), new FileOutputStream(war))) {
137                 throw new InstallationFailedException("Copy to "
138                         + war.getAbsolutePath() + " failed.");
139             }
140             container.deploy(war);
141         } catch (IOException e) {
142             throw new InstallationFailedException(e.getMessage(), e);
143         }
144     }

146     /**
147      * Command-line entry point.
148      */
149     public static void main(String[] args) {
150         LogConfig.initMinimal();
151         try {
152             Distribution dist = new ClassLoaderDistribution();
153             InstallOptions opts = null;
154 
155             if (args.length == 0) {
156                 opts = new InstallOptions(dist);
157             } else {
158                 Map<String, String> props = new HashMap<String, String>();
159                 for (String file : args) {
160                     props.putAll(FileUtils.loadMap(new File(file)));
161                 }
162                 opts = new InstallOptions(dist, props);
163             }
164 
165             // set fedora.home
166             System.setProperty("fedora.home", opts
167                     .getValue(InstallOptions.FEDORA_HOME));
168             new Installer(dist, opts).install();
169 
170         } catch (Exception e) {
171             printException(e);
172             System.exit(1);
173         }
174     }

176     /**
177      * Print a message appropriate for the given exception in as human-readable
178      * way as possible.
179      */
180     private static void printException(Exception e) {
181 
182         if (e instanceof InstallationCancelledException) {
183             System.out.println("Installation cancelled.");
184             return;
185         }
186 
187         boolean recognized = false;
188         String msg = "ERROR: ";
189         if (e instanceof InstallationFailedException) {
190             msg += "Installation failed: " + e.getMessage();
191             recognized = true;
192         } else if (e instanceof OptionValidationException) {
193             OptionValidationException ove = (OptionValidationException) e;
194             msg +=
195                     "Bad value for '" + ove.getOptionId() + "': "
196                             + e.getMessage();
197             recognized = true;
198         }
199 
200         if (recognized) {
201             System.err.println(msg);
202             if (e.getCause() != null) {
203                 System.err.println("Caused by: ");
204                 e.getCause().printStackTrace(System.err);
205             }
206         } else {
207             System.err.println(msg + "Unexpected error; installation aborted.");
208             e.printStackTrace();
209         }
210     }
