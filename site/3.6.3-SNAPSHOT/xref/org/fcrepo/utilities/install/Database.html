53      /**
54       * Fedora 2.2 renamed the 'do' table to 'dobj' (because 'do' is reserved in
55       * Postgresql, which is supported as of Fedora 2.2). 
56       *
57       * @throws InstallationFailedException
58       */
59      private void updateDOTable() throws InstallationFailedException {
60          if (_db.equals(InstallOptions.INCLUDED)) {
61              return; // no need to update embedded
62          }
63          try {
64              Connection conn = getConnection();
65              DatabaseMetaData dmd = conn.getMetaData();
66              ResultSet rs = dmd.getTables(null, null, "%", null);
67              while (rs.next()) {
68                  if (rs.getString("TABLE_NAME").equals("do")) {
69                      Statement stmt = conn.createStatement();
70                      stmt.execute("ALTER TABLE do RENAME TO dobj");
71                      System.out.println("Renamed table 'do' to 'dobj'.");
72                      stmt.close();
73                  }
74              }
75              rs.close();
76          } catch (Exception e) {
77              throw new InstallationFailedException(e.getMessage(), e);
78          }
79      }

81      protected File getDriver() throws IOException {
82          File driver = null;
83          if (_opts.getValue(InstallOptions.DATABASE_DRIVER)
84                  .equals(InstallOptions.INCLUDED)) {
85              InputStream is;
86              boolean success = false;
87              // INCLUDED driver with INCLUDED database, uses embedded driver.
88              if (_db.equals(InstallOptions.INCLUDED)) {
89                  is = _dist.get(Distribution.JDBC_DERBY);
90                  driver =
91                          new File(System.getProperty("java.io.tmpdir"),
92                                   Distribution.JDBC_DERBY);
93                  success = FileUtils.copy(is, new FileOutputStream(driver));
94              } // INCLUDED driver with DERBY database, uses network driver.
95              else if (_db.equals(InstallOptions.DERBY)) {
96                  is = _dist.get(Distribution.JDBC_DERBY_NETWORK);
97                  driver =
98                          new File(System.getProperty("java.io.tmpdir"),
99                                   Distribution.JDBC_DERBY_NETWORK);
100                 success = FileUtils.copy(is, new FileOutputStream(driver));
101             } else if (_db.equals(InstallOptions.MYSQL)) {
102                 is = _dist.get(Distribution.JDBC_MYSQL);
103                 driver =
104                         new File(System.getProperty("java.io.tmpdir"),
105                                  Distribution.JDBC_MYSQL);
106                 success = FileUtils.copy(is, new FileOutputStream(driver));
107             } else if (_db.equals(InstallOptions.POSTGRESQL)) {
108                 is = _dist.get(Distribution.JDBC_POSTGRESQL);
109                 driver =
110                         new File(System.getProperty("java.io.tmpdir"),
111                                  Distribution.JDBC_POSTGRESQL);
112                 success = FileUtils.copy(is, new FileOutputStream(driver));
113             }
114             if (!success) {
115                 throw new IOException("Extraction of included JDBC driver failed.");
116             }
117         } else {
118             driver = new File(_opts.getValue(InstallOptions.DATABASE_DRIVER));
119         }
120         return driver;
121     }

140     /**
141      * Determines whether or not the database has a table named "do".
142      *
143      * @return true if the database contains a table with the name "do".
144      * @throws Exception
145      */
146     protected boolean usesDOTable() throws Exception {
147         Connection conn = getConnection();
148         DatabaseMetaData dmd = conn.getMetaData();
149 
150         // check if we need to update old table
151         ResultSet rs = dmd.getTables(null, null, "%", null);
152         while (rs.next()) {
153             if (rs.getString("TABLE_NAME").equals("do")) {
154                 rs.close();
155                 return true;
156             }
157         }
158         rs.close();
159         return false;
160     }

162     private Connection getConnection() throws Exception {
163         if (_conn == null) {
164             DriverShim.loadAndRegister(getDriver(), _opts
165                     .getValue(InstallOptions.DATABASE_DRIVERCLASS));
166             _conn =
167                     DriverManager.getConnection(_opts
168                             .getValue(InstallOptions.DATABASE_JDBCURL), _opts
169                             .getValue(InstallOptions.DATABASE_USERNAME), _opts
170                             .getValue(InstallOptions.DATABASE_PASSWORD));
171         }
172         return _conn;
173     }
