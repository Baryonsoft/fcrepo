21  /**
22   * This class models the server.xml that ships with Fedora's bundled Tomcat.
23   * 
24   * @author Edwin Shin
25   *
26   */
27  public class TomcatServerXML
28          extends XMLDocument {
29  
30      private static final String KEYSTORE_LOCATION =
31              Tomcat.CONF + "/" + Tomcat.KEYSTORE;
32  
33      private static final String KEYSTORE_PASSWORD_DEFAULT = "changeit";
34  
35      private static final String KEYSTORE_TYPE_DEFAULT = "JKS";
36  
37      private static final String URI_ENCODING = "UTF-8";
38  
39      private static final String HTTP_CONNECTOR_XPATH = "/Server/Service[@name='Catalina']/Connector[not(@scheme='https' or contains(@protocol, 'AJP'))]";
40  
41      private static final String HTTPS_CONNECTOR_XPATH = "/Server/Service[@name='Catalina']/Connector[@scheme='https' and not(contains(@protocol, 'AJP'))]";
42  
43      private static final String AJP_CONNECTOR_XPATH = "/Server/Service[@name='Catalina']/Connector[contains(@protocol, 'AJP')]";
44  
45      private final InstallOptions options;
46  
47      public TomcatServerXML(File serverXML, InstallOptions installOptions)
48              throws FileNotFoundException, DocumentException {
49          this(new FileInputStream(serverXML), installOptions);
50      }
51  
52      public TomcatServerXML(InputStream serverXML, InstallOptions installOptions)
53              throws FileNotFoundException, DocumentException {
54          super(serverXML);
55          options = installOptions;
56      }

58      public void update() throws InstallationFailedException {
59          setHTTPPort();
60          setShutdownPort();
61          setSSLPort();
62          setURIEncoding();
63      }

65      /**
66       * Sets the http port and Fedora default http connector options.
67       * 
68       * @see http://tomcat.apache.org/tomcat-6.0-doc/config/http.html
69       * @throws InstallationFailedException
70       */
71      public void setHTTPPort() throws InstallationFailedException {
72          // Note this very significant assumption: this xpath will select exactly one connector
73          Element httpConnector =
74                  (Element) getDocument()
75                          .selectSingleNode(HTTP_CONNECTOR_XPATH);
76  
77          if (httpConnector == null) {
78              throw new InstallationFailedException("Unable to set server.xml HTTP Port. XPath for Connector element failed.");
79          }
80  
81          httpConnector.addAttribute("port", options
82                  .getValue(InstallOptions.TOMCAT_HTTP_PORT));
83          httpConnector.addAttribute("enableLookups", "true"); // supports client dns/fqdn in xacml authz policies
84          httpConnector.addAttribute("acceptCount", "100");
85          httpConnector.addAttribute("maxThreads", "150");
86          httpConnector.addAttribute("minSpareThreads", "25");
87          httpConnector.addAttribute("maxSpareThreads", "75");
88      }
89  
90      public void setShutdownPort() throws InstallationFailedException {
91          Element server =
92                  (Element) getDocument()
93                          .selectSingleNode("/Server[@shutdown and @port]");
94  
95          if (server == null) {
96              throw new InstallationFailedException("Unable to set server.xml shutdown port. XPath for Server element failed.");
97          }
98  
99          server.addAttribute("port", options
100                 .getValue(InstallOptions.TOMCAT_SHUTDOWN_PORT));
101     }

103     /**
104      * Sets the port and keystore information on the SSL connector if it already
105      * exists; creates a new SSL connector, otherwise. Also sets the
106      * redirectPort on the non-SSL connector to match.
107      *
108      * @throws InstallationFailedException
109      */
110     public void setSSLPort() throws InstallationFailedException {
111         Element httpsConnector =
112                 (Element) getDocument()
113                         .selectSingleNode(HTTPS_CONNECTOR_XPATH);
114         if (options.getBooleanValue(InstallOptions.SSL_AVAILABLE, true)) {
115             if (httpsConnector == null) {
116                 Element service =
117                         (Element) getDocument()
118                                 .selectSingleNode("/Server/Service[@name='Catalina']");
119                 httpsConnector = service.addElement("Connector");
120                 httpsConnector.addAttribute("minSpareThreads", "25");
121                 httpsConnector.addAttribute("maxSpareThreads", "75");
122                 httpsConnector.addAttribute("acceptCount", "100");
123                 httpsConnector.addAttribute("scheme", "https");
124                 httpsConnector.addAttribute("secure", "true");
125                 httpsConnector.addAttribute("SSLEnabled", "true");
126             }
127             httpsConnector.addAttribute("port", options
128                     .getValue(InstallOptions.TOMCAT_SSL_PORT));
129             httpsConnector.addAttribute("enableLookups", "true"); // supports client dns/fqdn in xacml authz policies
130 
131             String keystore = options.getValue(InstallOptions.KEYSTORE_FILE);
132             if (keystore.equals(InstallOptions.INCLUDED)) {
133                 keystore = KEYSTORE_LOCATION;
134             }
135 
136             addAttribute(httpsConnector,
137                          "keystoreFile",
138                          keystore,
139                          InstallOptions.DEFAULT);
140             addAttribute(httpsConnector,
141                          "keystorePass",
142                          options.getValue(InstallOptions.KEYSTORE_PASSWORD),
143                          KEYSTORE_PASSWORD_DEFAULT);
144             addAttribute(httpsConnector,
145                          "keystoreType",
146                          options.getValue(InstallOptions.KEYSTORE_TYPE),
147                          KEYSTORE_TYPE_DEFAULT);
148 
149             // The redirectPort for the non-SSL connector should match the port on
150             // the SSL connector, per:
151             // http://tomcat.apache.org/tomcat-6.0-doc/ssl-howto.html
152             Element httpConnector =
153                     (Element) getDocument()
154                             .selectSingleNode(HTTP_CONNECTOR_XPATH);
155             if (httpConnector != null) {
156                 httpConnector.addAttribute("redirectPort", options
157                         .getValue(InstallOptions.TOMCAT_SSL_PORT));
158             } else {
159                 throw new InstallationFailedException("Unable to set server.xml SSL Port. XPath for Connector element failed.");
160             }
161         } else if (httpsConnector != null) {
162             httpsConnector.getParent().remove(httpsConnector);
163         }
164     }

166     public void setURIEncoding() throws InstallationFailedException {
167         // http connector
168         // Note this very significant assumption: this xpath will select exactly one connector
169         Element httpConnector =
170                 (Element) getDocument()
171                         .selectSingleNode(HTTP_CONNECTOR_XPATH);
172         httpConnector.addAttribute("URIEncoding", URI_ENCODING);
173 
174         // https connector
175         httpConnector =
176             (Element) getDocument()
177                 .selectSingleNode(HTTPS_CONNECTOR_XPATH);
178         if (httpConnector != null ) {
179             httpConnector.addAttribute("URIEncoding", URI_ENCODING);
180         }
181 
182         // AJP connector
183         httpConnector =
184             (Element) getDocument()
185                 .selectSingleNode(AJP_CONNECTOR_XPATH);
186         if (httpConnector != null ) {
187             httpConnector.addAttribute("URIEncoding", URI_ENCODING);
188         }
189     }

192     /**
193      * Adds the attribute to the element if the attributeValue is not equal to
194      * defaultValue. If attributeValue is null or equals defaultValue, remove
195      * the attribute from the element if it is present.
196      *
197      * @param element
198      * @param attributeName
199      * @param attributeValue
200      * @param defaultValue
201      */
202     private void addAttribute(Element element,
203                               String attributeName,
204                               String attributeValue,
205                               String defaultValue) {
206         if (attributeValue == null || attributeValue.equals(defaultValue)) {
207             Attribute attribute =
208                     (Attribute) element.selectSingleNode(attributeName);
209             if (attribute != null) {
210                 element.remove(attribute);
211             }
212         } else {
213             element.addAttribute(attributeName, attributeValue);
214         }
215     }
