24  /**
25   * Allows one to load a JDBC driver at runtime. java.sql.DriverManager will
26   * refuse to use a driver not loaded by the system ClassLoader. The workaround
27   * for this is to use a shim class that implements java.sql.Driver. This shim
28   * class will do nothing but call the methods of an instance of a JDBC driver
29   * that is loaded dynamically. This works because DriverShim is loaded by the
30   * system class loader, and DriverManager doesn't care that it invokes a class
31   * that wasn't. Note that we must perform the registration on the instance
32   * ourselves. See the utility method, loadAndRegister and the command-line test
33   * below. Adapted from http://www.kfu.com/~nsayer/Java/dyn-jdbc.html
34   */
35  public class DriverShim
36          implements Driver {
37  
38      /** The JDBC driver we're wrapping. */
39      private final Driver _driver;
40  
41      /**
42       * Constructs a DriverShim over the given driver in order to make it look
43       * like it came from this classloader.
44       */
45      public DriverShim(Driver d) {
46          _driver = d;
47      }
48  
49      /**
50       * Loads the driver from the given jar file and registers it with the driver
51       * manager.
52       */
53      public static final void loadAndRegister(File driverJarFile,
54                                               String driverClassName)
55              throws Exception {
56          loadAndRegister(new URL("jar:" + driverJarFile.toURI() + "!/"),
57                          driverClassName);
58      }
59  
60      public static final void loadAndRegister(URL driverURL,
61                                               String driverClassName)
62              throws Exception {
63          URLClassLoader urlCL = new URLClassLoader(new URL[] {driverURL});
64          Driver driver =
65                  (Driver) Class.forName(driverClassName, true, urlCL)
66                          .newInstance();
67          DriverManager.registerDriver(new DriverShim(driver));
68      }
69  
70      //
71      // Driver implementation
72      //
73  
74      /**
75       * {@inheritDoc}
76       */
77      public boolean acceptsURL(String u) throws SQLException {
78          return _driver.acceptsURL(u);
79      }
80  
81      /**
82       * {@inheritDoc}
83       */
84      public Connection connect(String u, Properties p) throws SQLException {
85          return _driver.connect(u, p);
86      }

107         return _driver.getPropertyInfo(u, p);
108     }

110     /**
111      * {@inheritDoc}
112      */
113     public boolean jdbcCompliant() {
114         return _driver.jdbcCompliant();
115     }

118         throw new UnsupportedOperationException(
119                 "Java 1.7 Driver methods are not supported");
120     }

122     /**
123      * Command-line test. Dynamically loads the driver given at the command
124      * line, opens a connection, and get the row count of the given table to
125      * prove it worked. Usage: java -cp . DriverShim someDriver.jar
126      * org.SomeDriver jdbcURL name pwd tbl
127      */
128     public static void main(String[] args) throws Exception {
129         if (args.length == 6) {
130             DriverShim.loadAndRegister(new File(args[0]), args[1]);
131             Connection conn =
132                     DriverManager.getConnection(args[2], args[3], args[4]);
133             String tbl = args[5];
134             PreparedStatement st = conn.prepareStatement("SELECT COUNT(*) FROM ?");
135             st.setString(1, tbl);
136             ResultSet rs = st.executeQuery();
137             rs.next();
138             System.out.println("The " + tbl + " table has " + rs.getInt(1)
139                     + " rows.");
140             rs.close();
141             st.close();
142             conn.close();
143         } else {
144             System.out.println("Usage: java -cp . DriverShim someDriver.jar"
145                     + " org.SomeDriver jdbcURL name pwd tbl");
146             System.exit(1);
147         }
148     }
