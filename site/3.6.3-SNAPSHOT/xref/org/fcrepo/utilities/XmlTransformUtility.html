17      /**
18       * Convenience method to get a new instance of a TransformerFactory.
19       * If the {@link #TransformerFactory} is an instance of
20       * net.sf.saxon.TransformerFactoryImpl, the attribute
21       * {@link #FeatureKeys.VERSION_WARNING} will be set to false in order to
22       * suppress the warning about using an XSLT1 stylesheet with an XSLT2
23       * processor.
24       *
25       * @return a new instance of TransformerFactory
26       */
27      public static TransformerFactory getTransformerFactory() {
28          TransformerFactory factory = TransformerFactory.newInstance();
29          if (factory.getClass().getName().equals("net.sf.saxon.TransformerFactoryImpl")) {
30              factory.setAttribute("http://saxon.sf.net/feature/version-warning", Boolean.FALSE);
31          }
32          return factory;
33      }
