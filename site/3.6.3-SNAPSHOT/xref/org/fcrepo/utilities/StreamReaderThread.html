5   package org.fcrepo.utilities;
6   
7   import java.io.InputStream;
8   import java.io.InputStreamReader;
9   
10  /**
11   * @author Edwin Shin
12   */
13  public class StreamReaderThread
14          extends Thread {
15  
16      StringBuffer mOut;
17  
18      InputStreamReader mIn;
19  
20      public StreamReaderThread(InputStream in, StringBuffer out) {
21          mOut = out;
22          mIn = new InputStreamReader(in);
23      }
24  
25      @Override
26      public void run() {
27          int ch;
28          try {
29              while (-1 != (ch = mIn.read())) {
30                  mOut.append((char) ch);
31              }
32          } catch (Exception e) {
33              mOut.append("\nRead error:" + e.getMessage());
34          }
35      }
36  }
