48      private static int getCount(FedoraClient client, String path, String query) {
49          String response = null;
50          try {
51              path =  path + URLEncoder.encode(query, "UTF-8");
52          } catch (UnsupportedEncodingException e) {
53              e.printStackTrace();
54              fail("Encoding error while querying resource index. See stack trace");
55          }
56          try {
57              response = client.getResponseAsString(path,
58                                                    true,
59                                                    true).trim();
60          } catch (IOException e) {
61              e.printStackTrace();
62              fail("Error while querying resource index (is it enabled?).  "
63                      + "See stack trace");
64          }
65          int count = 0;
66          try {
67              count = Integer.parseInt(response);
68          } catch (NumberFormatException e) {
69              fail("Expected numeric plaintext response body from RI query, but "
70                      + "got the following: " + response);
71          }
72          return count;
73      }
74      
75      public static int getSPARQLCount(FedoraClient client, String query) {
76          return getCount(client, RISEARCH_SPARQL_COUNT, query);
77      }
78  
79      public static int getSPOCount(FedoraClient client, String query) {
80          return getCount(client, RISEARCH_SPO_COUNT, query);
81      }
