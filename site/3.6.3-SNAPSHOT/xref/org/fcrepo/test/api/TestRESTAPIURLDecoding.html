9   import static org.apache.commons.httpclient.HttpStatus.SC_OK;

11  public class TestRESTAPIURLDecoding
12          extends FedoraServerTestCase {
13  
14      private static final String pid = "test:SomeObject";
15      private static final String pidWithEscapedColon1 = "test%3ASomeObject";
16      private static final String pidWithEscapedColon2 = "test%3aSomeObject";
17  
18      private static TestRESTAPI rest = new TestRESTAPI();
19  
20      private static String getFOXML(String pid) {
21          // Test minimal FOXML object
22          StringBuilder sb = new StringBuilder();
23          sb = new StringBuilder();
24          sb.append("<?xml version=\"1.0\" encoding=\"UTF-8\"?>");
25          sb.append("<foxml:digitalObject VERSION=\"1.1\" PID=\"" + pid + "\" ");
26          sb.append("  xmlns:foxml=\"info:fedora/fedora-system:def/foxml#\" ");
27          sb.append("  xmlns:xsi=\"http://www.w3.org/2001/XMLSchema-instance\" ");
28          sb.append("  xsi:schemaLocation=\"info:fedora/fedora-system:def/foxml# ");
29          sb.append("  http://www.fedora.info/definitions/1/0/foxml1-1.xsd\">");
30          sb.append("  <foxml:objectProperties>");
31          sb.append("    <foxml:property NAME=\"info:fedora/fedora-system:def/model#state\" VALUE=\"A\"/>");
32          sb.append("  </foxml:objectProperties>");
33          sb.append("</foxml:digitalObject>");
34          return sb.toString();
35      }
36      
37      @Override
38      public void setUp() throws Exception {
39          rest.url = String.format("/objects/%s", pid);
40          rest.post(getFOXML(pid), true);
41      }
42      
43      @Override
44      public void tearDown() throws Exception {
45          rest.url = String.format("/objects/%s", pid);
46          rest.delete(true);
47      }
48  
49      public void testGetObjectProfile() throws Exception {
50          getAndExpect("/objects/" + pid, SC_OK);
51          getAndExpect("/objects/" + pidWithEscapedColon1, SC_OK);
52          getAndExpect("/objects/" + pidWithEscapedColon2, SC_OK);
53      }
54      
55      private void getAndExpect(String url, int statusCode) throws Exception {
56          rest.url = url;
57          assertEquals(SC_OK, rest.get(true).getStatusCode());
58      }
