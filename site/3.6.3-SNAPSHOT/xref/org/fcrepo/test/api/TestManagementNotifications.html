438     /**
439      * Waits for a notification message and checks to see if the message
440      * body includes the includedText.
441      *
442      * @param methodName - the text that should be found in the message body
443      */
444     private void checkNotification(String pid, String methodName) throws Exception {
445         //messageNumber++;
446 
447         TextMessage message = messages.poll(messageTimeout, TimeUnit.MILLISECONDS);
448         if (message == null) {
449             fail("Timeout reached waiting for notification " +
450                  "on message regarding: " + methodName);
451         }
452 
453             String failureText = "Notification <<" + message.getText() +
454                                  ">> did not include text: " + methodName;
455             assertTrue(failureText, message.getText().contains(methodName));
456 
457             failureText = "Notification <<" + message.getStringProperty("methodName") +
458                           ">> did not include methodName property with " +
459                           "value: " + methodName;
460             assertTrue(failureText,
461                        methodName.equals(message.getStringProperty("methodName")));
462 
463             failureText = "Notification did not include pid property with " +
464                           "value: " + pid;
465             assertTrue(failureText,
466                        pid.equals(message.getStringProperty("pid")));
467 
468         //currentMessage = null;
469     }
