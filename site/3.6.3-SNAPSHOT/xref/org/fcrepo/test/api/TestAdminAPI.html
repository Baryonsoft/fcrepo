88      private boolean getAuthAccess() {
89          if (authorizeAccess == null) {
90              String property = System.getProperty(authAccessProperty);
91              if (property.equals("true")) {

96                  assertTrue("Failed to determine whether to perform authorization on Access requests from: " +

99                          "Failed to determine whether to perform authorization on Access requests from: " +
100                         authAccessProperty);
101             }
102 
103         }
104         return authorizeAccess;

248     private void testFileInput(File objectsListFile) throws Exception {
249 
250         // count objects we know already have Managed content DC
251         // (ingest creates some of these, ending in _M)
252         // used later in checking results
253         Set<String> objects = getDemoObjects();
254         int managedObjects = 0;
255         for (String pid : objects) {
256             if (pid.endsWith("_M"))
257                 managedObjects++;
258         }
259 
260         // modify datastreams, based on file input - DC
261         url = this.modifyDatastreamControlGroupUrl("file:///" + objectsListFile.getAbsolutePath(), "DC", "M", false, false, false);
262         HttpResponse res = get(true);
263 
264         assertEquals(SC_OK, res.getStatusCode());
265         String modified = res.getResponseBodyString();
266 
267         // object and datastream count message expected
268         String logExpected = "Updated " + (objects.size() - managedObjects) + " objects and " + (objects.size() - managedObjects) + " datastreams";
269         assertTrue("Wrong number of objects/datastreams updated.  Expected " + logExpected + "\n" + "Log file shows:" + "\n" + modified, modified.contains(logExpected));
270 
271         // do again, this time we expect no modifications (already modified, so should be ingored)
272         res = get(true);
273 
274         modified = res.getResponseBodyString();
275         assertEquals(SC_OK, res.getStatusCode());
276 
277         // object and datastream count message expected
278         logExpected = "Updated " + 0 + " objects and " + 0 + " datastreams";
279         assertTrue("Wrong number of objects/datastreams updated", modified.contains(logExpected));
280 
281         // do again modifying DC and RELS-EXT
282         // DC is already M, so won't result in modifications
283         // not all objects have RELS-EXT
284         // so we check that object and datastream count is greater than zero but less than number of objects
285         // FIXME: could iterate all objects before/after and do more exact tests
286         url = this.modifyDatastreamControlGroupUrl("file:///" + objectsListFile.getAbsolutePath(), "DC,RELS-EXT", "M", false, false, false);
287         res = get(true);
288         modified = res.getResponseBodyString();
289         assertEquals(SC_OK, res.getStatusCode());
290 
291         int[] counts = getCounts(modified);
292         int objectCount = counts[0];
293         int datastreamCount = counts[1];
294 
295         if (objectCount <= 0 || objectCount > (objects.size() - managedObjects - 1) || datastreamCount <= 0 || datastreamCount > (objects.size() - managedObjects - 1))
296             fail("Incorrect number of objects and datastreams modified: objects " + objectCount + " datastreams " + datastreamCount);
297 
298     }

301     /**
302      * Parses results, returns object count, datastream count
303      * @param response
304      * @return
305      */
306     private int[] getCounts(String response) {
307         int objectCountPos = response.lastIndexOf("Updated ");
308         int datastreamCountPos = response.lastIndexOf(" objects and ");
309         int datastreamEndCountPos = response.lastIndexOf(" datastreams");
310 
311         int objectCount = Integer.parseInt(response.substring(objectCountPos + "Updated ".length(), datastreamCountPos));
312         int datastreamCount = Integer.parseInt(response.substring(datastreamCountPos + " objects and ".length(), datastreamEndCountPos));
313 
314         int[] res = {objectCount, datastreamCount};
315 
316         return res;
317     }

319     private String modifyDatastreamControlGroupUrl(String pid, String dsID, String controlGroup, boolean addXMLHeader, boolean reformat, boolean setMIMETypeCharset) {
320 
321         String ret;
322         try {
323             ret = "/management/control" +
324                 "?action=modifyDatastreamControlGroup" +
325                 "&pid=" + URLEncoder.encode(pid, "UTF-8") +
326                 "&dsID=" + URLEncoder.encode(dsID, "UTF-8") +
327                 "&controlGroup=" + controlGroup +
328                 "&addXMLHeader=" + addXMLHeader +
329                 "&reformat=" + reformat +
330                 "&setMIMETypeCharset=" + setMIMETypeCharset;
331         } catch (UnsupportedEncodingException e) {
332             // should never happen
333             throw new RuntimeException(e);
334         }
335 
336         return ret;
337     }

342     private HttpClient getClient(boolean auth) {
343         HttpClient client = new HttpClient();
344         client.getParams().setAuthenticationPreemptive(true);
345         if (auth) {
346             client

404     private HttpResponse getOrDelete(String method, boolean authenticate)
405             throws Exception {
406         if (url == null || url.length() == 0) {
407             throw new IllegalArgumentException("url must be a non-empty value");
408         } else if (!(url.startsWith("http://") || url.startsWith("https://"))) {
409             url = getBaseURL() + url;
410         }
411         HttpMethod httpMethod = null;
412         try {
413             if (method.equals("GET")) {
414                 httpMethod = new GetMethod(url);
415             } else if (method.equals("DELETE")) {
416                 httpMethod = new DeleteMethod(url);
417             } else {
418                 throw new IllegalArgumentException("method must be one of GET or DELETE.");
419             }
420             httpMethod.setDoAuthentication(authenticate);
421             httpMethod.getParams().setParameter("Connection", "Keep-Alive");
422             getClient(authenticate).executeMethod(httpMethod);
423             return new HttpResponse(httpMethod);
424         } finally {
425             if (httpMethod != null) {
426                 httpMethod.releaseConnection();
427             }
428         }
429     }

431     private HttpResponse putOrPost(String method,
432                                    Object requestContent,
433                                    boolean authenticate) throws Exception {
434 
435 
436         if (url == null || url.length() == 0) {
437             throw new IllegalArgumentException("url must be a non-empty value");
438         } else if (!(url.startsWith("http://") || url.startsWith("https://"))) {
439             url = getBaseURL() + url;
440         }
441 
442 
443         EntityEnclosingMethod httpMethod = null;
444         try {
445             if (method.equals("PUT")) {
446                 httpMethod = new PutMethod(url);
447             } else if (method.equals("POST")) {
448                 httpMethod = new PostMethod(url);
449             } else {
450                 throw new IllegalArgumentException("method must be one of PUT or POST.");
451             }
452             httpMethod.setDoAuthentication(authenticate);
453             httpMethod.getParams().setParameter("Connection", "Keep-Alive");
454             if (requestContent != null) {
455                 httpMethod.setContentChunked(chunked);
456                 if (requestContent instanceof String) {
457                     httpMethod
458                             .setRequestEntity(new StringRequestEntity((String) requestContent,
459                                                                       "text/xml",
460                                                                       "utf-8"));
461                 } else if (requestContent instanceof File) {
462                     Part[] parts =
463                             {
464                                     new StringPart("param_name", "value"),
465                                     new FilePart(((File) requestContent)
466                                             .getName(), (File) requestContent)};
467                     httpMethod
468                             .setRequestEntity(new MultipartRequestEntity(parts,
469                                                                          httpMethod
470                                                                                  .getParams()));
471                 } else {
472                     throw new IllegalArgumentException("requestContent must be a String or File");
473                 }
474             }
475             getClient(authenticate).executeMethod(httpMethod);
476             return new HttpResponse(httpMethod);
477         } finally {
478             if (httpMethod != null) {
479                 httpMethod.releaseConnection();
480             }
481         }
482     }

493     class HttpResponse {
494 
495         private final int statusCode;
496 
497         private final byte[] responseBody;
498 
499         private final Header[] responseHeaders;
500 
501         private final Header[] responseFooters;
502 
503         HttpResponse(int status, byte[] body, Header[] headers, Header[] footers) {
504             statusCode = status;
505             responseBody = body;
506             responseHeaders = headers;
507             responseFooters = footers;
508         }
509 
510         HttpResponse(HttpMethod method)
511                 throws IOException {
512             statusCode = method.getStatusCode();
513             //responseBody = method.getResponseBody();
514             InputStream is = method.getResponseBodyAsStream();
515             responseBody = IOUtils.toByteArray(is);
516             IOUtils.closeQuietly(is);
517             responseHeaders = method.getResponseHeaders();
518             responseFooters = method.getResponseFooters();
519         }
520 
521         public int getStatusCode() {
522             return statusCode;
523         }
524 
525         public byte[] getResponseBody() {
526             return responseBody;
527         }
528 
529         public String getResponseBodyString() {
530             try {
531                 return new String(responseBody, "UTF-8");
532             } catch (UnsupportedEncodingException wontHappen) {
533                 throw new Error(wontHappen);
534             }
535         }
536 
537         public Header[] getResponseHeaders() {
538             return responseHeaders;
539         }
540 
541         public Header[] getResponseFooters() {
542             return responseFooters;
543         }
544 
545         public Header getResponseHeader(String headerName) {
546             for (Header header : responseHeaders) {
547                 if (header.getName().equalsIgnoreCase(headerName)) {
548                     return header;
549                 }
550             }
551             return null;
552         }
