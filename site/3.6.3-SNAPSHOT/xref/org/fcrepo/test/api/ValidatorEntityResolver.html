14  public class ValidatorEntityResolver extends DefaultHandler implements EntityResolver {
15      public static String W3C = "http://www.w3.org/2001/03/";
16      public static String FEDORA = "http://www.fedora.info/definitions/1/0/";
17      public static String FEDORA_COMMONS = "http://www.fedora-commons.org/definitions/1/0/";
18      public static String DC = "http://dublincore.org/schemas/xmls/";
19      public static String OAI = "http://www.openarchives.org/OAI/2.0/";
20      private static final String SCHEMA_RESOURCE_PATH =
21      System.getProperty("fcrepo-integrationtest-core.classes") != null ?
22          System.getProperty("fcrepo-integrationtest-core.classes")
23                          + "schema" : "src/test/resources/schema"; 
24      @Override
25      public InputSource resolveEntity(String publicId, String systemId)
26              throws SAXException, IOException {
27          String fname = systemId.substring(systemId.lastIndexOf('/') + 1);
28          String sname = systemId.substring(0,systemId.lastIndexOf('/') + 1);
29          if (W3C.equals(sname)) return inputsource("w3c/" + fname, publicId, systemId);
30          if (FEDORA.equals(sname)) return inputsource("fedora/" + fname, publicId, systemId);
31          if (FEDORA_COMMONS.equals(sname)) return inputsource("fedora/" + fname, publicId, systemId);
32          if (DC.equals(sname)) return inputsource("dc/" + fname, publicId, systemId);
33          if (OAI.equals(sname)) return inputsource("oai/" + fname, publicId, systemId);
34          System.out.println("FAILED to resolve p: " + publicId + " s: " + systemId);
35          return null;
36      }
37      private InputSource inputsource(String path, String publicId, String systemId)
38               throws IOException {
39          FileInputStream fis = new FileInputStream(SCHEMA_RESOURCE_PATH + '/' + path);
40          InputSource result =  new InputSource(fis);
41          result.setEncoding("UTF-8");
42          result.setPublicId(publicId);
43          result.setSystemId(systemId);
44          return result;
45      }
