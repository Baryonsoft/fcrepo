40      @Test
41      public void testFCREPO238() throws Exception {
42          String pid =
43                  apim.ingest(TypeUtility.convertBytesToDataHandler(IOUtils.toByteArray(new FileInputStream(RESOURCEBASE
44                                      + "/" + FOXMLPATH))),
45                              FOXML1_1.uri,
46                              "testing fcrepo 238");
47          org.fcrepo.server.types.gen.ObjectProfile profile = apia.getObjectProfile(pid, null);
48          Date objDate = dateFormat.parse(profile.getObjLastModDate());
49          for (org.fcrepo.server.types.gen.DatastreamDef dd : apia.listDatastreams(pid, null)) {
50              org.fcrepo.server.types.gen.Datastream ds = apim.getDatastream(pid, dd.getID(), null);
51              Date dsDate = dateFormat.parse(ds.getCreateDate());
52              System.out.print("object:" + dateFormat.format(objDate) + ", ");
53              System.out.println("datastream: " + dateFormat.format(dsDate));
54              Assert.assertTrue("object last modificaton date is before datastream's create date. check FCREPO-238",
55                                objDate.compareTo(dsDate) > -1);
56          }
57          apim.purgeObject(pid, "removing testobject", true);
58      }
