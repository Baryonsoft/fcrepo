23      /* Remove any system methods */
24      public static ObjectMethodsDef[] filterMethods(ObjectMethodsDef[] initial) {
25          ArrayList<ObjectMethodsDef> desiredDefs =
26                  new ArrayList<ObjectMethodsDef>();
27  
28          for (ObjectMethodsDef def : initial) {
29              if (!def.getServiceDefinitionPID().startsWith("fedora-system:")
30                      && def != null) {
31                  desiredDefs.add(def);
32              }
33          }
34  
35          return desiredDefs.toArray(new ObjectMethodsDef[0]);
36      }
37  
38      /* Get a given dissemination as a string */
39      public static String getDissemination(FedoraClient client,
40                                            String pid,
41                                            String sDef,
42                                            String method) throws Exception {
43          return new String(TypeUtility.convertDataHandlerToBytes(client
44                  .getAPIAMTOM().getDissemination(pid, sDef, method, null, null)
45                  .getStream()), "UTF-8");
46  
47      }

63          dir = new File(base + "test-objects/foxml" + specificPath);
64  
65          FedoraClient client = FedoraTestCase.getFedoraClient();
66  
67          Ingest.multiFromDirectory(dir,
68                                    FOXML1_1.uri,
69                                    client.getAPIAMTOM(),
