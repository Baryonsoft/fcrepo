74      public void testListMetadataFormats() throws Exception {
75          String request = "/oai?verb=ListMetadataFormats";
76          Document result = getXMLQueryResult(request);
77          assertXpathEvaluatesTo("oai_dc",
78                                 "/oai:OAI-PMH/oai:ListMetadataFormats/oai:metadataFormat/oai:metadataPrefix",
79                                 result);
80      }

82      public void testListRecords() throws Exception {
83          FedoraAPIMMTOM apim = client.getAPIMMTOM();
84          FileInputStream in =
85                  new FileInputStream(FEDORA_HOME
86                                      + "/client/demo/foxml/local-server-demos/simple-document-demo/obj_demo_31.xml");
87  
88          ByteArrayOutputStream out = new ByteArrayOutputStream();
89          StreamUtility.pipeStream(in, out, 4096);
90  
91          apim.ingest(TypeUtility.convertBytesToDataHandler(out.toByteArray()), FOXML1_1.uri, "for testing");
92  
93          String request = "/oai?verb=ListRecords&metadataPrefix=oai_dc";
94          Document result = getXMLQueryResult(request);
95          assertXpathExists("/oai:OAI-PMH/oai:ListRecords/oai:record", result);
96  
97          request = "/oai?verb=ListRecords&metadataPrefix=oai_dc&from=2000-01-01";
98          result = getXMLQueryResult(request);
99          assertXpathExists("/oai:OAI-PMH/oai:ListRecords/oai:record", result);
100 
101         apim.purgeObject("demo:31", "for testing", false);
102     }
