9   /**
10   * A token that can be used to retrieve the remaining portion of an 
11   * incomplete list response.
12   * 
13   * @author Chris Wilper
14   * @see <a
15   *      href="http://www.openarchives.org/OAI/openarchivesprotocol.html#FlowControl">
16   *      http://www.openarchives.org/OAI/openarchivesprotocol.html#FlowControl</a>
17   */
18  public interface ResumptionToken {
19  
20      /**
21       * Get the value of the token. A null value indicates that the associated
22       * list is complete.
23       */
24      public abstract String getValue();
25  
26      /**
27       * Get the expiration date of the token. A null value indicates an unknown
28       * or unprovided expiration date.
29       */
30      public abstract Date getExpirationDate();
31  
32      /**
33       * Get the size of the list. A negative value indicates an unknown or
34       * unprovided list size.
35       */
36      public abstract long getCompleteListSize();
37  
38      /**
39       * Get the position in the list that this record starts at. A negative value
40       * indicates an unknown or unprovided position.
41       */
42      public abstract long getCursor();
43  
44  }
