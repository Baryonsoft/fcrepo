36       * Similar to above, this is "actor" in soap1_1 and "role" in 1_2. Soap 1.1

48      public static void throwFault(ServerException se) throws SoapFault {
49          throw getFault(se);
50      }
51  
52      public static SoapFault getFault(ServerException se) {
53          String[] details = se.getDetails();
54          String detailString = "";
55          if (details.length > 0) {
56              StringBuffer buf = new StringBuffer();
57              for (String element : details) {
58                  buf.append("<detail>");
59                  buf.append(se.getClass().getName());
60                  buf.append(": ");
61                  buf.append(element);
62                  buf.append("</detail>\n");
63              }
64              detailString = buf.toString();
65          }
66          SoapFault fault = new SoapFault(detailString, se, SoapFault.FAULT_CODE_CLIENT);
67          return fault;
68      }
69  
70      public static SoapFault getFault(AuthzException e) {
71          String reason = "";
72          if (e instanceof AuthzOperationalException) {
73              reason = AuthzOperationalException.BRIEF_DESC;
74          } else if (e instanceof AuthzDeniedException) {
75              reason = AuthzDeniedException.BRIEF_DESC;
76          } else if (e instanceof AuthzPermittedException) {
77              reason = AuthzPermittedException.BRIEF_DESC;
78          }
79          SoapFault fault = new SoapFault(reason, e, new QName("Authz"));
80          return fault;
81      }
82  
83      public static SoapFault getFault(Throwable th) {
84          if (th instanceof ServerException) {
85              if (th instanceof AuthzException) {
86                  return getFault((AuthzException) th);
87              } else {
88                  return getFault((ServerException) th);
89              }
90          } else {
91              if (th instanceof FaultException){
92                  return new SoapFault(th.toString(), th, SoapFault.FAULT_CODE_CLIENT);
93              }
94              else {
95                  return new SoapFault(th.toString(), new Exception("Uncaught exception from Fedora Server",
96                                                                    th), SoapFault.FAULT_CODE_CLIENT);
97              }
98          }
99      }
