66      public Rebuild(Rebuilder rebuilder, Map<String, String> options,
67              Server server)
68              throws Exception {
69          // set these here so DOTranslationUtility doesn't try to get a Server
70          // instance
71          System.setProperty("fedoraServerHost", server
72                  .getParameter("fedoraServerHost"));
73          System.setProperty("fedoraServerPort", server
74                  .getParameter("fedoraServerPort"));
75          System.setProperty("fedoraAppServerContext", server
76                  .getParameter("fedoraAppServerContext"));
77          boolean serverIsRunning = ServerUtility.pingServer("http", null, null);
78          if (serverIsRunning && rebuilder.shouldStopServer()) {
79              throw new Exception("The Fedora server appears to be running."
80                      + "  It must be stopped before the rebuilder can run.");
81          }
82          m_options = options;
83          m_rebuilder = rebuilder;
84          if (options != null) {
85              try {
86                  // ensure rebuilds are possible before trying anything,
87                  // as rebuilder.start() may be destructive!
88                  Module mod = server.getBean(llstoreInterface, Module.class);
89                  Class<?> clazz = mod.getClass();
90                  boolean isListable = false;
91                  for (Class<?> iface : clazz.getInterfaces()) {
92                      if (iface.getName().equals(listableInterface)) {
93                          isListable = true;
94                      }
95                  }
96                  if (!isListable) {
97                      throw new Exception("ERROR: Rebuilds are not supported" +
98                              " by " + clazz.getName() +
99                              " because it does not implement the" +
100                             " org.fcrepo.server.storage.lowlevel.IListable" +
101                             " interface.");
102                 }
103 
104             } finally {
105             }
106         } else {
107             logger.warn("Null options for " + getClass().getName());
108         }
109     }

111     public void run() {
112         try {
113             if (m_options != null) {
114                 System.err.println();
115                 System.err.println("Rebuilding...");
116                 try {
117                     // looks good, so init the rebuilder
118                     m_rebuilder.start(m_options);
119 
120                     // add each object in llstore
121                     ILowlevelStorage llstore =
122                             (ILowlevelStorage) getServer().getModule(
123                                     llstoreInterface);
124                     if (llstore == null) {
125                         logger.error("No module/bean definition for " +
126                                 llstoreInterface);
127                     } else {
128                         logger.info("Loaded bean/module " + llstoreInterface +
129                                 " with impl " + llstore.getClass().getName());
130                     }
131                     Iterator<String> pids = ((IListable) llstore).listObjects();
132                     int total = 0;
133                     int errors = 0;
134                     DODeserializer deser = new FOXML1_1DODeserializer();
135 
136                     while (pids.hasNext()) {
137                         total++;
138                         String pid = pids.next();
139                         System.out.println("Adding object #" + total + ": " +
140                                 pid);
141                         if (!addObject(m_rebuilder, llstore, deser, pid)) {
142                             errors++;
143                         }
144                     }
145                     if (errors == 0) {
146                         System.out.println("SUCCESS: " + total +
147                                 " objects rebuilt.");
148                     } else {
149                         System.out.println("WARNING: " + errors + " of " +
150                                 total +
151                                 " objects failed to rebuild due to errors.");
152                     }
153                 } finally {
154                     m_rebuilder.finish();
155                     if (server != null) {
156                         server.shutdown(null);
157                         server = null;
158                     }
159                     System.err.print("Finished.");
160                     System.err.println();
161                 }
162                 return;
163             }
164         } catch (Exception e) {
165             System.err.println("Rebuild failed:");
166             System.err.println(e.toString());
167             e.printStackTrace(System.err);
168         }
169     }

171     private boolean addObject(Rebuilder rebuilder, ILowlevelStorage llstore,
172             DODeserializer deser, String pid) {
173         InputStream in = null;
174         try {
175             in = llstore.retrieveObject(pid);
176             DigitalObject obj = new BasicDigitalObject();
177             deser.deserialize(in, obj, "UTF-8",
178                     DOTranslationUtility.SERIALIZE_STORAGE_INTERNAL);
179             rebuilder.addObject(obj);
180             return true;
181         } catch (Exception e) {
182             System.out.println("WARNING: Skipped " + pid +
183                     " due to exception: ");
184             e.printStackTrace();
185             return false;
186         } finally {
187             if (in != null) {
188                 try {
189                     in.close();
190                 } catch (IOException e) {
191                 }
192             }
193         }
194     }

213     private static Map<String, String> getOptions(Map<String, String> descs)
214             throws IOException {
215         Map<String, String> options = new HashMap<String, String>();
216         if (descs != null) {
217             Iterator<String> iter = descs.keySet().iterator();
218             while (iter.hasNext()) {
219                 String name = iter.next();
220                 String desc = descs.get(name);
221                 options.put(name, getOptionValue(name, desc));
222             }
223         }
224         int c = 1;
225 
226         if (System.getProperty("rebuilder") == null) {
227             if (options.size() > 0) {
228                 c =
229                         getChoice("Start rebuilding with the above options?",
230                                 new String[] {"Yes",
231                                         "No, let me re-enter the options.",
232                                         "No, exit."});
233 
234                 if (c == 0) {
235                     return options;
236                 }
237                 if (c == 1) {
238                     System.err.println();
239                     return getOptions(descs);
240                 }
241             } else {
242                 c =
243                         getChoice("No options to set. Start rebuilding?",
244                                 new String[] {"Yes", "No, exit."});
245                 if (c == 0) {
246                     return options;
247                 }
248             }
249         } else {
250             return options;
251         }
252         return null;
253     }

267     private static Rebuilder getRebuilder() throws Exception {
268         Server server = getServer();
269         String[] rebuilders = server.getBeanNamesForType(Rebuilder.class);
270         String[] labels = new String[rebuilders.length + 1];
271         int i = 0;
272         for (i = 0; i < rebuilders.length; i++) {
273             Rebuilder r = server.getBean(rebuilders[i], Rebuilder.class);
274 
275             labels[i] = r.getAction();
276         }
277         labels[i] = "Exit";
278         int choiceNum = i;
279         System.out.println("Getting rebuilder... " +
280                 System.getProperty("rebuilder"));
281         if (System.getProperty("rebuilder") == null) {
282             choiceNum = getChoice("What do you want to do?", labels);
283         } else {
284             for (int j = 0; j < rebuilders.length; j++) {
285                 if (rebuilders[j].equals(System.getProperty("rebuilder"))) {
286                     choiceNum = j;
287                 }
288             }
289         }
290         if (choiceNum == i) {
291             return null;
292         } else {
293             return server.getBean(rebuilders[choiceNum], Rebuilder.class);
294         }
295     }

297     private static int getChoice(String title, String[] labels)
298             throws IOException {
299         boolean validChoice = false;
300         int choiceIndex = -1;
301         System.err.println(title);
302         System.err.println();
303         for (int i = 1; i <= labels.length; i++) {
304             System.err.println("  " + i + ") " + labels[i - 1]);
305         }
306         System.err.println();
307         while (!validChoice) {
308             System.err.print("Enter (1-" + labels.length + ") --> ");
309             BufferedReader in =
310                     new BufferedReader(new InputStreamReader(System.in));
311             String line = in.readLine();
312             try {
313                 int choiceNum = Integer.parseInt(line);
314                 if (choiceNum > 0 && choiceNum <= labels.length) {
315                     choiceIndex = choiceNum - 1;
316                     validChoice = true;
317                 }
318             } catch (NumberFormatException nfe) {
319             }
320         }
321         return choiceIndex;
322     }

324     private static ServerConfiguration getServerConfig(File serverDir,
325             String profile) throws IOException {
326         ServerConfigurationParser parser =
327                 new ServerConfigurationParser(new FileInputStream(new File(
328                         serverDir, "config/fedora.fcfg")));
329         ServerConfiguration serverConfig = parser.parse();
330         // set all the values according to the profile, if specified
331         if (profile != null) {
332             int c = setValuesForProfile(serverConfig, profile);
333 
334             c +=
335                     setValuesForProfile(serverConfig.getModuleConfigurations(),
336                             profile);
337             c +=
338                     setValuesForProfile(serverConfig
339                             .getDatastoreConfigurations(), profile);
340             if (c == 0) {
341                 throw new IOException("Unrecognized server-profile: " + profile);
342             }
343         }
344         return serverConfig;
345     }

347     private static int
348             setValuesForProfile(Configuration config, String profile) {
349         int c = 0;
350         Iterator<Parameter> iter =
351                 config.getParameters(Parameter.class).iterator();
352         while (iter.hasNext()) {
353             Parameter param = iter.next();
354             String profileValue = param.getProfileValues().get(profile);
355             if (profileValue != null) {
356                 param.setValue(profileValue);
357                 c++;
358             }
359         }
360         return c;
361     }
362 
363     private static int setValuesForProfile(
364             List<? extends Configuration> configs, String profile) {
365         Iterator<? extends Configuration> iter = configs.iterator();
366         int c = 0;
367         while (iter.hasNext()) {
368             c += setValuesForProfile(iter.next(), profile);
369         }
370         return c;
371     }

402     public static void main(String[] args) {
403         String profile = null;
404         if (args.length > 0) {
405             profile = args[0];
406         }
407         if (args.length > 1) {
408             fail("Too many arguments", true, true);
409         }
410         try {
411             File fedoraHomeDir = new File(Constants.FEDORA_HOME);
412 
413             // Configure logging from file
414             System.setProperty("fedora.home", Constants.FEDORA_HOME);
415             System.setProperty("logfile.extension", "-rebuild.log");
416             LogConfig.initFromFile(new File(fedoraHomeDir,
417                     "server/config/logback.xml"));
418 
419             // Replace java.util.logging's default handlers with one that
420             // redirects everything to SLF4J
421             java.util.logging.Logger rootLogger =
422                     java.util.logging.LogManager.getLogManager().getLogger("");
423             java.util.logging.Handler[] handlers = rootLogger.getHandlers();
424             for (int i = 0; i < handlers.length; i++) {
425                 rootLogger.removeHandler(handlers[i]);
426             }
427             SLF4JBridgeHandler.install();
428 
429             File serverDir = new File(fedoraHomeDir, "server");
430             ServerConfiguration serverConfig =
431                     getServerConfig(serverDir, profile);
432             System.err.println();
433             System.err.println("                       Fedora Rebuild Utility");
434             System.err
435                     .println("                     ..........................");
436             System.err.println();
437             System.err
438                     .println("WARNING: Live rebuilds are not currently supported.");
439             System.err
440                     .println("         Make sure your server is stopped before continuing.");
441             System.err.println();
442             System.err.println("Server directory is " + serverDir.toString());
443             if (profile != null) {
444                 System.err.print("Server profile is " + profile);
445             }
446             System.err.println();
447             System.err
448                     .println("---------------------------------------------------------------------");
449             System.err.println();
450             Rebuilder rebuilder = getRebuilder();
451             Map<String, String> options =
452                     getUserInput(rebuilder, serverDir, serverConfig);
453             new Rebuild(rebuilder, options, getServer()).run();
454             return;
455         } catch (Throwable th) {
456             String msg = th.getMessage();
457             if (msg == null) {
458                 msg = th.getClass().getName();
459             }
460             fail(msg, false, false);
461             th.printStackTrace();
462             try {
463                 getServer().shutdown(null);
464             } catch (Throwable t) {
465                 System.err.println("Server shutdown error: " + t.toString());
466             }
467         }
468     }
