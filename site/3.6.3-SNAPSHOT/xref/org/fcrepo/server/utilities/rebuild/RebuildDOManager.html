34       * @param moduleParameters

36       * @param role
37       * @throws ModuleInitializationException

39      public RebuildDOManager(Map moduleParameters, Server server, String role)
40              throws ModuleInitializationException {
41          super(moduleParameters, server, role);
42      }
43  
44      @Override
45      protected void initRetainPID() {
46          // retainPIDs (optional, default=demo,test)
47          // when m_retainPIDS is set to null, that means "all"
48          m_retainPIDs = null;
49      }

51      @Override
52      public void postInitModule() throws ModuleInitializationException {
53          // does not use management module
54  //        m_management =
55  //                (Management) getServer()
56  //                        .getModule("org.fcrepo.server.management.Management");
57  
58          // get ref to contentmanager module
59          m_contentManager =
60                  (ExternalContentManager) getServer()
61                          .getModule("org.fcrepo.server.storage.ExternalContentManager");
62          if (m_contentManager == null) {
63              throw new ModuleInitializationException("ExternalContentManager not loaded.",
64                                                      getRole());
65          }
66          // get ref to fieldsearch module
67          m_fieldSearch =
68                  (FieldSearch) getServer()
69                          .getModule("org.fcrepo.server.search.FieldSearch");
70          // get ref to pidgenerator
71          m_pidGenerator =
72                  (PIDGenerator) getServer()
73                          .getModule("org.fcrepo.server.management.PIDGenerator");
74          // note: permanent and temporary storage handles are lazily instantiated
75  
76          // get ref to translator and derive storageFormat default if not given
77          m_translator =
78                  (DOTranslator) getServer()
79                          .getModule("org.fcrepo.server.storage.translation.DOTranslator");
80          //        // get ref to replicator
81          //        m_replicator=(DOReplicator) getServer().
82          //                getModule("org.fcrepo.server.storage.replication.DOReplicator");
83          // get ref to digital object validator
84          m_validator =
85                  (DOValidator) getServer()
86                          .getModule("org.fcrepo.server.validation.DOValidator");
87          if (m_validator == null) {
88              throw new ModuleInitializationException("DOValidator not loaded.",
89                                                      getRole());
90          }
91          // will not use ref to ResourceIndex (ok if it's not loaded)
92  //        m_resourceIndex =
93  //                (ResourceIndex) getServer()
94  //                        .getModule("org.fcrepo.server.resourceIndex.ResourceIndex");
95  
96          // now get the connectionpool
97          ConnectionPoolManager cpm =
98                  (ConnectionPoolManager) getServer()
99                          .getModule("org.fcrepo.server.storage.ConnectionPoolManager");
100         if (cpm == null) {
101             throw new ModuleInitializationException("ConnectionPoolManager not loaded.",
102                                                     getRole());
103         }
104         try {
105             if (m_storagePool == null) {
106                 m_connectionPool = cpm.getPool();
107             } else {
108                 m_connectionPool = cpm.getPool(m_storagePool);
109             }
110         } catch (ConnectionPoolNotFoundException cpnfe) {
111             String storagePool = (m_storagePool == null)?"[null]":m_storagePool;
112             throw new ModuleInitializationException("Couldn't get required "
113                     + "connection pool " + storagePool + " ...wasn't found", getRole());
114         }
115         try {
116             String dbSpec =
117                     "org/fcrepo/server/storage/resources/DefaultDOManager.dbspec";
118             InputStream specIn =
119                     this.getClass().getClassLoader()
120                             .getResourceAsStream(dbSpec);
121             if (specIn == null) {
122                 throw new IOException("Cannot find required " + "resource: "
123                         + dbSpec);
124             }
125             SQLUtility.createNonExistingTables(m_connectionPool, specIn);
126         } catch (Exception e) {
127             throw new ModuleInitializationException("Error checking for and "
128                     + "creating non-existing tables", getRole(), e);
129         }
130 
131         // get ref to lowlevelstorage module
132         m_permanentStore =
133                 (ILowlevelStorage) getServer()
134                         .getModule("org.fcrepo.server.storage.lowlevel.ILowlevelStorage");
135         if (m_permanentStore == null) {
136             throw new ModuleInitializationException("LowlevelStorage not loaded",
137                                                     getRole());
138         }
139 
140     }
