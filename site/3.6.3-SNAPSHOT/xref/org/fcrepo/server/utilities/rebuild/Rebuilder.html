15  /**
16   * Interface for a class that rebuilds some aspect of the repository.
17   *
18   * <p>It is expected that clients of this interface will first call init, then
19   * start, then addObject (possibly a series of times), then finish.
20   *
21   * @author Chris Wilper
22   */
23  public interface Rebuilder {
24  
25      /**
26       * Get a short phrase describing what the user can do with this rebuilder.
27       */
28      public String getAction();
29  
30      /**
31       * Initialize the rebuilder, given the server configuration.
32       *
33       * @returns a map of option names to plaintext descriptions.
34       */
35      public void setServerConfiguration(ServerConfiguration serverConfig);
36  
37      public void setServerDir(File serverBaseDir);
38  
39      public void init();
40  
41      public Map<String, String> getOptions()
42              throws Exception;
43  
44      /**
45       * Returns true is the server _must_ be shut down for this rebuilder to
46       * safely operate.
47       */
48      public boolean shouldStopServer();
49  
50      /**
51       * Validate the provided options and perform any necessary startup tasks.
52       */
53      public void start(Map<String, String> options) throws Exception;
54  
55      /**
56       * Add the data of interest for the given object.
57       */
58      public void addObject(DigitalObject object) throws Exception;
59  
60      /**
61       * Free up any system resources associated with rebuilding.
62       */
63      public void finish() throws Exception;
64  
65  }
