80      /**
81       * Get the baseURL of the Fedora server from the host and port configured.
82       * It will look like http://localhost:8080/fedora
83       */
84      public static String getBaseURL(String protocol) {
85          String port;
86          if (protocol.equals("http")) {
87              port = CONFIG.getParameter(FEDORA_SERVER_PORT,Parameter.class).getValue();
88          } else if (protocol.equals("https")) {
89              port = CONFIG.getParameter(FEDORA_REDIRECT_PORT,Parameter.class).getValue();
90          } else {
91              throw new RuntimeException("Unrecogonized protocol: " + protocol);
92          }
93          return protocol + "://"
94                  + CONFIG.getParameter(FEDORA_SERVER_HOST,Parameter.class).getValue() + ":"
95                  + port + "/" + CONFIG.getParameter(FEDORA_SERVER_CONTEXT,Parameter.class).getValue();
96      }

109     /**
110      * Hits the given Fedora Server URL and returns the response as a String.
111      * Throws an IOException if the response code is not 200(OK).
112      */
113     private static String getServerResponse(String protocol,
114                                             String user,
115                                             String pass,
116                                             String path) throws IOException {
117         String url = getBaseURL(protocol) + path;
118         logger.info("Getting URL: " + url);
119         UsernamePasswordCredentials creds =
120                 new UsernamePasswordCredentials(user, pass);
121         WebClientConfiguration webconfig = new WebClientConfiguration();
122         initWebClientConfig(webconfig);
123         return new WebClient(webconfig).getResponseAsString(url, true, creds);
124     }

142     /**
143      * Tell whether the given URL appears to be referring to somewhere within
144      * the Fedora webapp.
145      */
146     public static boolean isURLFedoraServer(String url) {
147 
148         // scheme must be http or https
149         URI uri = URI.create(url);
150         String scheme = uri.getScheme();
151         if (!scheme.equals("http") && !scheme.equals("https")) {
152             return false;
153         }
154 
155         // host must be configured hostname or localhost
156         String fHost = CONFIG.getParameter(FEDORA_SERVER_HOST,Parameter.class).getValue();
157         String host = uri.getHost();
158         if (!host.equals(fHost) && !host.equals("localhost")) {
159             return false;
160         }
161 
162         // path must begin with configured webapp context
163         String path = uri.getPath();
164         String fedoraContext = CONFIG.getParameter(
165                 FEDORA_SERVER_CONTEXT,Parameter.class).getValue();
166         if (!path.startsWith("/" + fedoraContext + "/")) {
167             return false;
168         }
169 
170         // port specification must match http or https port as appropriate
171         String httpPort = CONFIG.getParameter(FEDORA_SERVER_PORT,Parameter.class).getValue();
172         String httpsPort = CONFIG.getParameter(FEDORA_REDIRECT_PORT,Parameter.class).getValue();
173         if (uri.getPort() == -1) {
174             // unspecified, so fedoraPort must be 80 (http), or 443 (https)
175             if (scheme.equals("http")) {
176                 return httpPort.equals("80");
177             } else {
178                 return httpsPort.equals("443");
179             }
180         } else {
181             // specified, so must match appropriate http or https port
182             String port = "" + uri.getPort();
183             if (scheme.equals("http")) {
184                 return port.equals(httpPort);
185             } else {
186                 return port.equals(httpsPort);
187             }
188         }
189 
190     }

211     /**
212      * Initializes the web client http connection settings.
213      */
214     private static void initWebClientConfig(WebClientConfiguration wconf) {
215 
216         if (CONFIG.getParameter("httpClientTimeoutSecs") != null)
217             wconf.setTimeoutSecs(Integer.parseInt(CONFIG.getParameter("httpClientTimeoutSecs")));
218 
219         if (CONFIG.getParameter("httpClientSocketTimeoutSecs") != null)
220             wconf.setSockTimeoutSecs(Integer.parseInt(CONFIG.getParameter("httpClientSocketTimeoutSecs")));
221 
222         if (CONFIG.getParameter("httpClientMaxConnectionsPerHost") != null)
223             wconf.setMaxConnPerHost(Integer.parseInt(CONFIG.getParameter("httpClientMaxConnectionsPerHost")));
224 
225         if (CONFIG.getParameter("httpClientMaxTotalConnections") != null)
226             wconf.setMaxTotalConn(Integer.parseInt(CONFIG.getParameter("httpClientMaxTotalConnections")));
227 
228         if (CONFIG.getParameter("httpClientFollowRedirects") != null)
229             wconf.setFollowRedirects(Boolean.parseBoolean(CONFIG.getParameter("httpClientFollowRedirects")));
230 
231         if (CONFIG.getParameter("httpClientMaxFollowRedirects") != null)
232             wconf.setMaxRedirects(Integer.parseInt(CONFIG.getParameter("httpClientMaxFollowRedirects")));
233 
234         if (CONFIG.getParameter("httpClientUserAgent") != null)
235             wconf.setUserAgent(CONFIG.getParameter("httpClientUserAgent"));
236     }

239      * Command-line entry point to reload policies. Takes 3 args: protocol user
240      * pass
241      */
242     public static void main(String[] args) {
243 
244         if (args.length < 1) {
245             System.out.println("Parameters:  method arg1 arg2 arg3 etc");
246             System.out.println("");
247             System.out.println("Methods:");
248             System.out.println("    reloadpolicies");
249             System.out.println("    migratedatastreamcontrolgroup");
250             System.exit(0);
251         }
252 
253         String method = args[0].toLowerCase();
254 
255         if (method.equals("reloadpolicies")) {
256             if (args.length == 4) {
257             try {
258                     reloadPolicies(args[1], args[2], args[3]);
259                 System.out.println("SUCCESS: Policies have been reloaded");
260                 System.exit(0);
261             } catch (Throwable th) {
262                 th.printStackTrace();
263                 System.err
264                         .println("ERROR: Reloading policies failed; see above");
265                 System.exit(1);
266             }
267         } else {
268             System.err.println("ERROR: Three arguments required: "
269                     + "http|https username password");
270             System.exit(1);
271         }
272 
273         } else if (method.equals("migratedatastreamcontrolgroup")) {
274 
275             // too many args
276             if (args.length > 10) {
277                 System.err.println("ERROR: too many arguments provided");
278                 System.exit(1);
279     }
280 
281             // too few
282             if (args.length < 7) {
283 
284                 System.err.println("ERROR: insufficient arguments provided.  Arguments are: ");
285                 System.err.println("    protocol [http|https]"); // 1; 0 is method
286                 System.err.println("    user"); // 2
287                 System.err.println("    password"); // 3
288                 System.err.println("    pid - either"); // 4
289                 System.err.println("        a single pid, eg demo:object");
290                 System.err.println("        list of pids separated by commas, eg demo:object1,demo:object2");
291                 System.err.println("        name of file containing pids, eg file:///path/to/file");
292                 System.err.println("    dsid - either"); // 5
293                 System.err.println("        a single datastream id, eg DC");
294                 System.err.println("        list of ids separated by commas, eg DC,RELS-EXT");
295                 System.err.println("    controlGroup - target control group (note only M is implemented)"); // 6
296                 System.err.println("    addXMLHeader - add an XML header to the datastream [true|false, default false]"); // 7
297                 System.err.println("    reformat - reformat the XML [true|false, default false]"); // 8
298                 System.err.println("    setMIMETypeCharset - add charset=UTF-8 to the MIMEType [true|false, default false]"); // 9
299                 System.exit(1);
300             }
301 
302 
303             try {
304                 // optional args
305                 boolean addXMLHeader = getArgBoolean(args, 7, false);
306                 boolean reformat = getArgBoolean(args, 8, false);
307                 boolean setMIMETypeCharset = getArgBoolean(args, 9, false);;
308 
309                 InputStream is = modifyDatastreamControlGroup(args[1], args[2], args[3], args[4], args[5], args[6], addXMLHeader, reformat, setMIMETypeCharset);
310 
311                 IOUtils.copy(is, System.out);
312                 is.close();
313 
314                 System.out.println("SUCCESS: Datastreams modified");
315                 System.exit(0);
316 
317             } catch (Throwable th) {
318                 th.printStackTrace();
319                 System.err
320                 .println("ERROR: migrating datastream control group failed; see above");
321                 System.exit(1);
322             }
323 
324 
325         } else {
326             System.err.println("ERROR: unrecognised method " + method);
327             System.exit(1);
328         }

330     /**
331      * Get boolean argument from list of arguments at position; use defaultValue if not present
332      * @param args
333      * @param position
334      * @param defaultValue
335      * @return
336      */
337     private static boolean getArgBoolean(String[] args, int position, boolean defaultValue) {
338         if (args.length > position) {
339             String lowerArg = args[position].toLowerCase();
340             if (lowerArg.equals("true") || lowerArg.equals("yes")) {
341                 return true;
342             } else if (lowerArg.equals("false") || lowerArg.equals("no")) {
343                 return false;
344             } else {
345                 throw new IllegalArgumentException(args[position] + " not a valid value.  Specify true or false");
346             }
347         } else {
348             return defaultValue;
349         }
350     }
