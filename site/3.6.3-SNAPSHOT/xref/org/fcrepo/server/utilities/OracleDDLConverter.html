7   import java.util.ArrayList;
8   import java.util.Iterator;
9   import java.util.List;
10  
11  /**
12   * A DDLConverter that works with Oracle.
13   *
14   * @author Chris Wilper
15   */
16  public class OracleDDLConverter
17          implements DDLConverter {
18  
19      public OracleDDLConverter() {
20      }
21  
22      public List<String> getDDL(TableSpec spec) {
23          ArrayList<String> l = new ArrayList<String>();
24          StringBuffer out = new StringBuffer();
25          StringBuffer end = new StringBuffer();
26          out.append("CREATE TABLE " + spec.getName() + " (\n");
27          Iterator<ColumnSpec> csi = spec.columnSpecIterator();
28          int csNum = 0;
29          while (csi.hasNext()) {
30              if (csNum > 0) {
31                  out.append(",\n");
32              }
33              csNum++;
34              ColumnSpec cs = csi.next();
35              out.append("  ");
36              out.append(cs.getName());
37              out.append(' ');
38              if (cs.getType().toLowerCase().indexOf("int(") == 0) {
39                  // if precision was specified for int, use oracle's default int precision
40                  out.append("int");
41              } else {
42                  if (cs.getType().toLowerCase().indexOf("smallint(") == 0) {
43                      out.append("smallint");
44                  } else {
45                      if (cs.getType().toLowerCase().equals("bigint")) {
46                          out.append("NUMBER(20,0)");
47                      } else if (cs.getType().toLowerCase().equals("text")) {
48                          out.append("CLOB");
49                      } else {
50                          out.append(cs.getType());
51                      }
52                  }
53              }
54              if (cs.isAutoIncremented()) {
55                  // oracle doesn't support auto-increment in a CREATE TABLE
56                  // ... but it can be done by creating the table,
57                  // creating a sequence, then creating a trigger that
58                  // inserts the sequence's next value for that column
59                  // upon insert.
60                  StringBuffer createSeq = new StringBuffer();
61                  createSeq.append("CREATE SEQUENCE ");
62                  createSeq.append(spec.getName());
63                  createSeq.append("_S");
64                  createSeq.append(csNum);
65                  createSeq.append("\n");
66                  createSeq.append("  START WITH 1\n");
67                  createSeq.append("  INCREMENT BY 1\n");
68                  createSeq.append("  NOMAXVALUE");
69                  l.add(createSeq.toString());
70                  StringBuffer createTrig = new StringBuffer();
71                  createTrig.append("CREATE TRIGGER ");
72                  createTrig.append(spec.getName());
73                  createTrig.append("_T");
74                  createTrig.append(csNum);
75                  createTrig.append("\n");
76                  createTrig.append("  BEFORE INSERT ON ");
77                  createTrig.append(spec.getName());
78                  createTrig.append("\n  FOR EACH ROW");
79                  createTrig.append("\n  BEGIN");
80                  createTrig.append("\n    SELECT ");
81                  createTrig.append(spec.getName());
82                  createTrig.append("_S");
83                  createTrig.append(csNum);
84                  createTrig.append(".NEXTVAL INTO :NEW.");
85                  createTrig.append(cs.getName());
86                  createTrig.append(" FROM DUAL;");
87                  createTrig.append("\n  END;");
88                  l.add(createTrig.toString());
89              }
90              if (cs.getDefaultValue() != null) {
91                  out.append(" DEFAULT '");
92                  out.append(cs.getDefaultValue());
93                  out.append("'");
94              }
95              if (cs.isNotNull()) {
96                  out.append(" NOT NULL");
97              }
98              if (cs.isUnique()) {
99                  if (!end.toString().equals("")) {
100                     end.append(",\n");
101                 }
102                 end.append("  UNIQUE ");
103                 end.append(" (");
104                 end.append(cs.getName());
105                 end.append(")");
106             }
107             if (cs.getIndexName() != null) {
108                 l.add("CREATE INDEX " + spec.getName() + "_" + cs.getName()
109                       + " ON " + spec.getName() + " (" + cs.getName() + ")");
110             }
111             if (cs.getForeignTableName() != null) {
112                 if (!end.toString().equals("")) {
113                     end.append(",\n");
114                 }
115                 end.append("  FOREIGN KEY ");
116                 end.append(cs.getName());
117                 end.append(" (");
118                 end.append(cs.getName());
119                 end.append(") REFERENCES ");
120                 end.append(cs.getForeignTableName());
121                 end.append(" (");
122                 end.append(cs.getForeignColumnName());
123                 end.append(")");
124                 if (cs.getOnDeleteAction() != null) {
125                     end.append(" ON DELETE ");
126                     end.append(cs.getOnDeleteAction());
127                 }
128             }
129         }
130         if (spec.getPrimaryColumnName() != null) {
131             out.append(",\n  PRIMARY KEY (");
132             out.append(spec.getPrimaryColumnName());
133             out.append(")");
134         }
135         if (!end.toString().equals("")) {
136             out.append(",\n");
137             out.append(end);
138         }
139         out.append("\n");
140         out.append(")");
141         l.add(0, out.toString());
142         return l;
143     }
