5   package org.fcrepo.server.utilities;
6   
7   import java.sql.Connection;
8   import java.sql.SQLException;
9   import java.sql.Statement;
10  
11  import java.util.Iterator;
12  
13  /**
14   * A ConnectionWrapper that creates tables on the target database given a
15   * TableSpec.
16   * 
17   * @author Chris Wilper
18   */
19  public class TableCreatingConnection
20          extends ConnectionWrapper {
21  
22      private final DDLConverter m_converter;
23  
24      /**
25       * Constructs a TableCreatingConnection.
26       * 
27       * @param wrapped
28       *        The wrapped connection.
29       * @param converter
30       *        A converter that can translate from a TableSpec to DB-specific
31       *        DDL.
32       */
33      public TableCreatingConnection(Connection wrapped, DDLConverter converter) {
34          super(wrapped);
35          m_converter = converter;
36      }
37  
38      /**
39       * Get the DDLConverter this TableCreatingConnection works with.
40       * 
41       * @return The converter.
42       */
43      public DDLConverter getDDLConverter() {
44          return m_converter;
45      }
46  
47      /**
48       * Creates a table in the target database.
49       * <p>
50       * </p>
51       * This method may execute more than one update command and it ignores the
52       * transaction state of the connection.
53       * 
54       * @param spec
55       *        A description of the table to be created.
56       */
57      public void createTable(TableSpec spec) throws SQLException {
58          Statement s = createStatement();
59          Iterator<String> iter = m_converter.getDDL(spec).iterator();
60          while (iter.hasNext()) {
61              String updateSQL = iter.next();
62              s.executeUpdate(updateSQL);
63          }
64      }
65  
66  }
