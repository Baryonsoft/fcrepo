5   package org.fcrepo.server.journal.helpers;

7   import java.io.BufferedInputStream;
8   import java.io.BufferedOutputStream;
9   import java.io.File;
10  import java.io.FileInputStream;
11  import java.io.FileOutputStream;
12  import java.io.IOException;
13  
14  /**
15   * Provides a workaround to the fact that
16   * {@link java.io.File.renameTo(java.io.File)} doesn't work across NFS file
17   * systems.
18   * <p>
19   * This code is taken from a workaround provided on the Sun Developer Network
20   * Bug Database (http://bugs.sun.com/bugdatabase/view_bug.do?bug_id=4073756), by
21   * mailto:morgan.sziraki@cartesian.co.uk
22   * </p>
23   * <p>
24   * The code is modified to protect against a situation where
25   * <ul>
26   * <li>We need to copy the file across NFS mounted filesystems</li>
27   * <li>Another process is waiting for the file to be renamed</li>
28   * </ul>
29   * If we do a simple copy, we leave open the possibility that the second process
30   * will see the new file created before we have a chance to copy the contents.
31   * </p>
32   * <p>
33   * To avoid this, we create the new file with a modified filename (prefixed by
34   * an underscore '_'). When the copying is complete, we rename the file to the
35   * desired name. This rename is within a directory, and therefore does not
36   * extend across NFS filesystem boundaries, so it should work!
37   * </p>
38   * <p>
39   * Modify again to make sure that neither the destination file nor the temp file
40   * exist - throw an exception if they do. We can't rename on top of an existing
41   * file.
42   * </p>
43   * 
44   * @author Jim Blake
45   */
46  public class FileMovingUtil {
47  
48      private FileMovingUtil() {
49          // No need to instantiate, since the only public method is static.
50      }
51  
52      /**
53       * <p>
54       * Move a File
55       * </p>
56       * <p>
57       * The renameTo method does not allow action across NFS mounted filesystems.
58       * This method is the workaround.
59       * </p>
60       * 
61       * @param fromFile
62       *        The existing File
63       * @param toFile
64       *        The new File
65       * @throws IOException
66       *         if any problems occur
67       */
68      public final static void move(File fromFile, File toFile)
69              throws IOException {
70          // try the simple way first.
71          if (fromFile.renameTo(toFile)) {
72              return;
73          }
74  
75          // copy to the temp file, then rename to the desired name.
76          checkThatFileDoesntExist(toFile);
77          File tempFile = createTempFile(toFile);
78          checkThatFileDoesntExist(tempFile);
79          copy(fromFile, tempFile);
80          tempFile.renameTo(toFile);
81          // delete the old one
82          if (!fromFile.delete()) {
83              throw new IOException("Failed to delete '" + fromFile.getParent()
84                      + "'");
85          }
86      }
87  
88      /**
89       * If we're trying to rename to a file that already exists, we'll throw an
90       * exception. Make sure that the same thing happens if the rename is
91       * accomplished by copying.
92       * 
93       * @throws IOException
94       *         if the file exists.
95       */
96      private static void checkThatFileDoesntExist(File file) throws IOException {
97          if (file.exists()) {
98              throw new IOException("File '" + file.getPath()
99                      + "' already exists.");
100         }
101 
102     }
103 
104     /**
105      * Create a temporary File object. Prefix the name of the base file with an
106      * underscore.
107      */
108     private static File createTempFile(File baseFile) {
109         File parentDirectory = baseFile.getParentFile();
110         String filename = baseFile.getName();
111         return new File(parentDirectory, '_' + filename);
112     }
113 
114     /**
115      * Copy a File
116      * 
117      * @param fromFile
118      *        The existing File
119      * @param toFile
120      *        The new File
121      * @throws IOException
122      *         if any problems are encountered
123      */
124     private final static void copy(File fromFile, File toFile)
125             throws IOException {
126         FileInputStream in = new FileInputStream(fromFile);
127         FileOutputStream out = new FileOutputStream(toFile);
128         BufferedInputStream inBuffer = new BufferedInputStream(in);
129         BufferedOutputStream outBuffer = new BufferedOutputStream(out);
130 
131         int theByte = 0;
132 
133         while ((theByte = inBuffer.read()) > -1) {
134             outBuffer.write(theByte);
135         }
136 
137         outBuffer.close();
138         inBuffer.close();
139         out.close();
140         in.close();
141 
142         // cleanupif files are not the same length
143         if (fromFile.length() != toFile.length()) {
144             toFile.delete();
145 
146             throw new IOException("Copy failed: source file length="
147                     + fromFile.length() + ", target file length="
148                     + toFile.length());
149         }
150     }
