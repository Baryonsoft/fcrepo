5   package org.fcrepo.server.journal.helpers;
6   
7   import java.io.IOException;
8   import java.io.OutputStream;
9   
10  import java.util.regex.Pattern;
11  
12  import org.fcrepo.utilities.Base64;
13  
14  
15  /**
16   * Wraps an OutputStream with a Base64 decoder, so when you "write" to the
17   * stream, you write Strings of Base64-encoded characters, but the OutputStream
18   * receives decoded bytes.
19   * <p>
20   * Base64 encoding is defined in Internet RFC 3548, found at
21   * http://tools.ietf.org/html/rfc3548 (among other places).
22   *
23   * @author Jim Blake
24   */
25  public class DecodingBase64OutputStream {
26  
27      private final Pattern pattern = Pattern.compile("[^A-Za-z0-9+/=]*");
28  
29      private final OutputStream stream;
30  
31      private String residual = "";
32  
33      private boolean open = true;
34  
35      /**
36       * @param stream
37       *        the destination for the decoded bytes.
38       */
39      public DecodingBase64OutputStream(OutputStream stream) {
40          this.stream = stream;
41      }
42  
43      /**
44       * Add Base64-encoded characters to be decoded. This is not a trivial
45       * operation for two reasons: any characters that are not valid for
46       * Base64-encoding must be ignored, and we can only decode groups of 4
47       * characters. So, when data is received, we remove any invalid characters
48       * and then strip off any trailing characters that don't fit in the
49       * 4-character groups. Those trailing characters will be prefixed to the
50       * next set of data, and hopefully we will have none left over when the
51       * writer is closed.
52       *
53       * @throws IllegalStateException
54       *         if called after close().
55       * @throws IOException
56       *         from the inner OutputStream.
57       */
58      public void write(String data) throws IOException {
59          if (!open) {
60              throw new IllegalStateException("Stream has already been closed.");
61          }
62  
63          String buffer = pattern.matcher(residual + data).replaceAll("");
64          int usableLength = buffer.length() - buffer.length() % 4;
65          stream.write(Base64.decode((buffer.substring(0, usableLength))));
66          residual = buffer.substring(usableLength);
67      }
68  
69      /**
70       * Close the writer. If there are any residual characters at this point, the
71       * data stream was not a valid Base64 encoding.
72       *
73       * @throws IOException
74       *         from the inner OutputStream.
75       */
76      public void close() throws IOException {
77          if (open) {
78              if (residual.length() > 0) {
79                  throw new IOException("Base64 error - data is not properly"
80                          + "padded to 4-character groups.");
81              }
82              stream.close();
83              open = false;
84          }
85      }
86  
87  }
