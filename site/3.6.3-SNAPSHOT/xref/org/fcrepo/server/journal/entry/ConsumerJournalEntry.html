5   package org.fcrepo.server.journal.entry;

7   import org.fcrepo.server.errors.ServerException;
8   import org.fcrepo.server.journal.JournalException;
9   import org.fcrepo.server.journal.recoverylog.JournalRecoveryLog;
10  import org.fcrepo.server.management.ManagementDelegate;
11  
12  
13  /**
14   * The JournalEntry to use when consuming a journal file.
15   * <p>
16   * Before invoking a method, write the entry to the recovery logger. After invoking
17   * the method, log a completion message.
18   * 
19   * @author Jim Blake
20   */
21  public class ConsumerJournalEntry
22          extends JournalEntry {
23  
24      private String identifier = "no identifier";
25  
26      public ConsumerJournalEntry(String methodName, JournalEntryContext context) {
27          super(methodName, context);
28      }
29  
30      public void invokeMethod(ManagementDelegate delegate,
31                               JournalRecoveryLog recoveryLog)
32              throws ServerException, JournalException {
33          recoveryLog.log(this);
34          super.getMethod().invoke(delegate);
35          recoveryLog.log("Call complete:" + super.getMethodName());
36      }
37  
38      public void setIdentifier(String identifier) {
39          this.identifier = identifier;
40      }
41  
42      public String getIdentifier() {
43          return identifier;
44      }
45  
46      @Override
47      public String toString() {
48          return "ConsumerJournalEntry[identifier=" + identifier
49                  + ", methodName=" + getMethodName() + ", context="
50                  + getContext() + "]";
51      }
52  
53  }
