9   import org.fcrepo.server.journal.JournalException;
10  import org.fcrepo.server.journal.JournalOperatingMode;
11  import org.fcrepo.server.journal.JournalWriter;
12  import org.fcrepo.server.management.ManagementDelegate;

15  /**
16   * The {@link JournalEntry} to use when creating a journal file.
17   * <p>
18   * When invoking the management method, take a moment to write to the journal
19   * before returning.
20   * 
21   * @author Jim Blake
22   */
23  public class CreatorJournalEntry
24          extends JournalEntry {
25  
26      /**
27       * Don't store the Context that was given; store a writable version of it.
28       */
29      public CreatorJournalEntry(String methodName, Context context) {
30          super(methodName, new JournalEntryContext(context));
31      }
32  
33      /**
34       * Process the management method:
35       * <ul>
36       * <li>Check the operating mode - if we are in
37       * {@link JournalOperatingMode#READ_ONLY Read-Only} mode, this check will
38       * throw an exception.</li>
39       * <li>Prepare the writer in case we need to initialize a new file with a
40       * repository hash.</li>
41       * <li>Invoke the method on the ManagementDelegate.</li>
42       * <li>Write the full journal entry, including any context changes from the
43       * Management method.</li>
44       * </ul>
45       * These operations occur within a synchronized block. We must be sure that
46       * any pending operations are complete before we get the repository hash, so
47       * we are confident that the hash accurately reflects the state of the
48       * repository. Since all API-M operations go through this synchronized
49       * block, we can be confident that the previous one had completed before the
50       * current one started.
51       * <p>
52       * There might be a way to enforce the synchronization at a lower level,
53       * thus increasing throughput, but we haven't explored it yet.
54       */
55      public Object invokeMethod(ManagementDelegate delegate, JournalWriter writer)
56              throws ServerException, JournalException {
57          synchronized (JournalWriter.SYNCHRONIZER) {
58              JournalOperatingMode.enforceCurrentMode();
59              writer.prepareToWriteJournalEntry();
60              Object result = super.getMethod().invoke(delegate);
61              writer.writeJournalEntry(this);
62              return result;
63          }
64      }
65  
66      /**
67       * A convenience method that invokes the management method and then closes
68       * the JournalEntry, thereby cleaning up any temp files.
69       */
70      public Object invokeAndClose(ManagementDelegate delegate,
71                                   JournalWriter writer) throws ServerException,
72              JournalException {
73          Object result = invokeMethod(delegate, writer);
74          close();
75          return result;
76      }
77  
78  }
