7   
8   import java.io.BufferedWriter;
9   import java.io.IOException;
10  import java.io.Writer;
11  
12  import java.rmi.RemoteException;
13  
14  import org.fcrepo.server.journal.JournalException;
15  
16  
17  /**
18   * <p>
19   * <b>Title:</b> RmiTransportWriter.java
20   * </p>
21   * <p>
22   * <b>Description:</b> A writer that passes characters on to an
23   * RmiTransportSink. You should wrap this in a {@link BufferedWriter} to reduce
24   * network traffic to something manageable.
25   * </p>
26   *
27   * @author jblake
28   * @version $Id: RmiTransportWriter.java,v 1.3 2007/06/01 17:21:32 jblake Exp $
29   */
30  public class RmiTransportWriter
31          extends Writer {
32  
33      private final RmiJournalReceiverInterface receiver;
34  
35      private final String repositoryHash;
36  
37      private long itemIndex;
38  
39      private boolean m_closed = false;
40  
41      /**
42       * When the writer is created, open the connection to the receiver.
43       */
44      public RmiTransportWriter(RmiJournalReceiverInterface receiver,
45                                String repositoryHash,
46                                String filename)
47              throws JournalException {
48          this.receiver = receiver;
49          this.repositoryHash = repositoryHash;
50  
51          try {
52              receiver.openFile(repositoryHash, filename);
53          } catch (RemoteException e) {
54              throw new JournalException(e);
55          }
56      }
57  
58      /**
59       * Each time characters are written, send them to the receiver. Keep track
60       * of how many writes, so we can create a proper item hash.
61       */
62      @Override
63      public void write(char[] cbuf, int off, int len) throws IOException {
64          try {
65              String indexedHash =
66                      RmiJournalReceiverHelper.figureIndexedHash(repositoryHash,
67                                                                 itemIndex);
68              receiver.writeText(indexedHash, new String(cbuf, off, len));
69              itemIndex++;
70          } catch (JournalException e) {
71              throw new IOException(e);
72          }
73      }
74  
75      /**
76       * Time to close the file? Tell the receiver.
77       */
78      @Override
79      public void close() throws IOException {
80          try {
81              if (!m_closed) {
82                  receiver.closeFile();
83                  m_closed= true;
84              }
85          } catch (JournalException e) {
86              throw new IOException(e);
87          }
88      }
89  
90      /**
91       * Flush has no effect. Everything is written as it comes in.
92       */
93      @Override
94      public void flush() throws IOException {
95      }
96  
97  }
