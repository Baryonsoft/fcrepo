28  /**
29   * A rudimentary implementation of JournalReader that just reads all entries
30   * from a single Journal file.
31   * <p>
32   * Useful only for System tests.
33   * 
34   * @author Jim Blake
35   */
36  public class SingleFileJournalReader
37          extends JournalReader
38          implements SingleFileJournalConstants {
39  
40      private final File journalFile;
41  
42      private final XMLEventReader reader;
43  
44      private boolean open = true;
45  
46      private boolean advancedPastHeader = false;
47  
48      /**
49       * Get the name of the journal file from the server parameters, wrap it in
50       * an XMLEventReader, and advance past the document header to the first
51       * JournalEntry.
52       * 
53       * @throws JournalException
54       */
55      public SingleFileJournalReader(Map<String, String> parameters,
56                                     String role,
57                                     JournalRecoveryLog recoveryLog,
58                                     ServerInterface server)
59              throws ModuleInitializationException, JournalException {
60          super(parameters, role, recoveryLog, server);
61          recoveryLog.log("Using a SingleFileJournalReader");
62  
63          if (!parameters.containsKey(PARAMETER_JOURNAL_FILENAME)) {
64              throw new ModuleInitializationException("Parameter '"
65                      + PARAMETER_JOURNAL_FILENAME + "' not set.", role);
66          }
67  
68          String filename = parameters.get(PARAMETER_JOURNAL_FILENAME);
69          journalFile = new File(filename);
70  
71          if (!journalFile.exists()) {
72              throw new ModuleInitializationException("Journal file '"
73                      + journalFile.getPath() + "' does not exist.", role);
74          }
75          if (!journalFile.isFile()) {
76              throw new ModuleInitializationException("Journal file '"
77                      + journalFile.getPath() + "' is not a file.", role);
78          }
79          if (!journalFile.canRead()) {
80              throw new ModuleInitializationException("Journal file '"
81                      + journalFile.getPath() + "' is not readable.", role);
82          }
83  
84          try {
85              XMLInputFactory factory = XMLInputFactory.newInstance();
86              reader = factory.createXMLEventReader(new FileReader(journalFile));
87  
88          } catch (FileNotFoundException e) {
89              throw new ModuleInitializationException("Problem dumping file",
90                                                      role);
91          } catch (XMLStreamException e) {
92              throw new ModuleInitializationException("Error opening XML Event reader on Journal file '"
93                                                              + journalFile
94                                                                      .getPath()
95                                                              + "'",
96                                                      role,
97                                                      e);
98          }
99  
100     }

102     /**
103      * Advance past the document header to the first JournalEntry.
104      */
105     private void advanceIntoFile() throws XMLStreamException, JournalException {
106         XMLEvent event = reader.nextEvent();
107         if (!event.isStartDocument()) {
108             throw new JournalException("Expecting XML document header, but event was '"
109                     + event + "'");
110         }
111 
112         event = reader.nextTag();
113         if (!isStartTagEvent(event, QNAME_TAG_JOURNAL)) {
114             throw new JournalException("Expecting FedoraJournal start tag, but event was '"
115                     + event + "'");
116         }
117 
118         String hash =
119                 getOptionalAttributeValue(event.asStartElement(),
120                                           QNAME_ATTR_REPOSITORY_HASH);
121         checkRepositoryHash(hash);
122     }

124     /**
125      * Advance past any white space, and then see whether we have any more
126      * JournalEntry tags. If we don't, just return null.
127      */
128     @Override
129     public synchronized ConsumerJournalEntry readJournalEntry()
130             throws JournalException, XMLStreamException {
131         if (!open) {
132             return null;
133         }
134 
135         if (!advancedPastHeader) {
136             advanceIntoFile();
137             advancedPastHeader = true;
138         }
139 
140         XMLEvent next = reader.peek();
141 
142         // advance past any whitespace events.
143         while (next.isCharacters() && next.asCharacters().isWhiteSpace()) {
144             reader.nextEvent();
145             next = reader.peek();
146         }
147         if (isStartTagEvent(next, QNAME_TAG_JOURNAL_ENTRY)) {
148             String identifier = peekAtJournalEntryIdentifier();
149             ConsumerJournalEntry journalEntry = super.readJournalEntry(reader);
150             journalEntry.setIdentifier(identifier);
151             return journalEntry;
152         } else if (isEndTagEvent(next, QNAME_TAG_JOURNAL)) {
153             return null;
154         } else {
155             throw getNotNextMemberOrEndOfGroupException(QNAME_TAG_JOURNAL,
156                                                         QNAME_TAG_JOURNAL_ENTRY,
157                                                         next);
158         }
159     }

161     /**
162      * Create an identifier string for the Journal Entry, so we can easily
163      * connect the entries in the Recovery Logger with those in the Journal. Call
164      * this before calling
165      * {@link JournalReader#readJournalEntry(XMLEventReader)}, because the
166      * reader is positioned at the beginning of the JournalEntry, so a peek()
167      * will give us the start tag, with the info we need.
168      */
169     private String peekAtJournalEntryIdentifier() throws XMLStreamException {
170         XMLEvent event = reader.peek();
171 
172         String timeString = "unknown";
173         if (event.isStartElement()) {
174             StartElement start = event.asStartElement();
175             Attribute timeStamp =
176                     start.getAttributeByName(QNAME_ATTR_TIMESTAMP);
177             if (timeStamp != null) {
178                 timeString = timeStamp.getValue();
179             }
180         }
181 
182         return "entry='" + timeString + "'";
183     }

185     /**
186      * On the first call, Just close the reader.
187      */
188     @Override
189     public synchronized void shutdown() throws JournalException {
190         try {
191             if (open) {
192                 reader.close();
193                 open = false;
194             }
195         } catch (XMLStreamException e) {
196             throw new JournalException(e);
197         }
198     }

200     @Override
201     public String toString() {
202         return super.toString() + ", journalFile='" + journalFile.getPath()
203                 + "'";
204     }
