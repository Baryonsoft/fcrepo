72      /**
73       * Inquires upon all object Datastreams to obtain datastreams contained by a
74       * digital object. This returns a set of datastream locations that represent
75       * all possible datastreams available in the object.
76       * <p/>
77       * GET /objects/{pid}/datastreams ? asOfDateTime format
78       *
79       * @param pid
80       * @param dateTime
81       * @param format
82       */
83      @GET
84      public Response listDatastreams(@PathParam(RestParam.PID) String pid,
85                                      @QueryParam(RestParam.AS_OF_DATE_TIME) String dateTime,
86                                      @QueryParam(RestParam.FORMAT) @DefaultValue(HTML) String format,
87                                      @QueryParam(RestParam.FLASH) @DefaultValue("false") boolean flash,
88                                      @QueryParam(RestParam.PROFILES) @DefaultValue("false") boolean profiles,
89                                      @QueryParam(RestParam.DS_STATE) String dsState,
90                                      @QueryParam(RestParam.VALIDATE_CHECKSUM) @DefaultValue("false") boolean validateChecksum
91                                      ) {
92  
93          try {
94              Date asOfDateTime = DateUtility.parseDateOrNull(dateTime);
95              Context context = getContext();
96              MediaType mime = RestHelper.getContentType(format);
97  
98              String output;
99  
100             if (profiles){
101                 mime=MediaType.TEXT_XML_TYPE;
102                 final Datastream[] datastreams = m_management.getDatastreams(context, pid, asOfDateTime, dsState);
103                 output=getSerializer(context).datastreamProfilesToXML(pid, datastreams, asOfDateTime, validateChecksum);
104             } else {
105                 mime = RestHelper.getContentType(format);
106                 DatastreamDef[] dsDefs =
107                         m_access.listDatastreams(context, pid, asOfDateTime);
108                 output = getSerializer(context).dataStreamsToXML(pid, asOfDateTime, dsDefs);
109                 if (TEXT_HTML.isCompatible(mime)) {
110                     final CharArrayWriter writer = new CharArrayWriter();
111                     transform(output, "access/listDatastreams.xslt", writer);
112                     output = writer.toString();
113                 }
114             }
115 
116             return Response.ok(output, mime).build();
117         } catch (Exception ex) {
118             return handleException(ex, flash);
119         }
120     }

397     protected Response addOrUpdateDatastream(boolean posted,
398                                              String pid,
399                                              String dsID,
400                                              MediaType mediaType,
401                                              String mimeType,
402                                              String controlGroup,
403                                              String dsLocation,
404                                              List<String> altIDList,
405                                              String dsLabel,
406                                              Boolean versionable,
407                                              String dsState,
408                                              String formatURI,
409                                              String checksumType,
410                                              String checksum,
411                                              String logMessage,
412                                              boolean ignoreContent,
413                                              DateTimeParam lastModifiedDate,
414                                              boolean flash) {
415 
416         try {
417             String[] altIDs = {};
418 
419             if (altIDList != null && altIDList.size() > 0) {
420                 altIDs = altIDList.toArray(new String[altIDList.size()]);
421             }
422 
423             Context context = getContext();
424 
425             Datastream existingDS =
426                     m_management.getDatastream(context, pid, dsID, null);
427             if (!posted && versionable == null && existingDS != null){
428                 versionable = existingDS.DSVersionable;
429             }
430             Date requestModDate = null;
431             if (lastModifiedDate != null) {
432                 requestModDate = lastModifiedDate.getValue();
433             }
434 
435             // If a datastream is set to Deleted state, it must be set to
436             // another state before any other changes can be made
437             if (existingDS != null && existingDS.DSState.equals("D")
438                     && dsState != null) {
439                 if (dsState.equals("A") || dsState.equals("I")) {
440                     m_management.setDatastreamState(context,
441                                                    pid,
442                                                    dsID,
443                                                    dsState,
444                                                    logMessage);
445                     existingDS.DSState = dsState;
446                 }
447             }
448 
449             InputStream is = null;
450 
451             // Determine if datastream content is included in the request
452             if (!ignoreContent) {
453                 RestUtil restUtil = new RestUtil();
454                 RequestContent content =
455                         restUtil.getRequestContent(m_servletRequest, m_headers);
456 
457                 if (content != null && content.getContentStream() != null) {
458                     is = content.getContentStream();
459                     // Give preference to the passed in mimeType
460                     if (mimeType == null && content.getMimeType() != null) {
461                         mimeType = content.getMimeType();
462                     }
463                 }
464             }
465 
466             // Make sure that there is a mime type value
467             if (mimeType == null && mediaType != null) {
468                 mimeType = mediaType.toString();
469             } else if (mimeType == null && mediaType == null
470                     && existingDS != null) {
471                 mimeType = existingDS.DSMIME;
472             }
473 
474             // set default control group based on mimeType
475             if (dsLocation == null
476                     && TEXT_XML.isCompatible(MediaType.valueOf(mimeType))
477                     && controlGroup == null) {
478                 controlGroup = "X";
479             }
480 
481             if (existingDS == null) {
482                 if (posted) {
483                     if ((dsLocation == null || dsLocation.equals(""))
484                             && ("X".equals(controlGroup) || "M"
485                                     .equals(controlGroup))) {
486                         dsLocation = m_management.putTempStream(context, is);
487                     }
488                     dsID =
489                             m_management.addDatastream(context,
490                                                       pid,
491                                                       dsID,
492                                                       altIDs,
493                                                       dsLabel,
494                                                       versionable,
495                                                       mimeType,
496                                                       formatURI,
497                                                       dsLocation,
498                                                       controlGroup,
499                                                       dsState,
500                                                       checksumType,
501                                                       checksum,
502                                                       logMessage);
503                 } else {
504                     return Response.status(Response.Status.NOT_FOUND).build();
505                 }
506             } else {
507                 if ("X".equals(existingDS.DSControlGrp)) {
508                     // Inline XML can only be modified by value. If there is no stream,
509                     // but there is a dsLocation attempt to retrieve the content.
510                     if (is == null && dsLocation != null
511                             && !dsLocation.equals("")) {
512                         try {
513                             WebClientConfiguration webconfig = m_server.getWebClientConfig();
514                             WebClient webClient = new WebClient(webconfig);
515                             is = webClient.get(dsLocation, true);
516                         } catch (IOException ioe) {
517                             throw new Exception("Could not retrive content from "
518                                     + dsLocation
519                                     + " due to error: "
520                                     + ioe.getMessage());
521                         }
522                     }
523                     m_management.modifyDatastreamByValue(context,
524                                                         pid,
525                                                         dsID,
526                                                         altIDs,
527                                                         dsLabel,
528                                                         mimeType,
529                                                         formatURI,
530                                                         is,
531                                                         checksumType,
532                                                         checksum,
533                                                         logMessage,
534                                                         requestModDate);
535                 } else {
536                     // Managed content can only be modified by reference.
537                     // If there is no dsLocation, but there is a content stream,
538                     // store the stream in a temporary location.
539                     if (dsLocation == null
540                             && ("M".equals(existingDS.DSControlGrp))) {
541                         if (is != null) {
542                             dsLocation = m_management.putTempStream(context, is);
543                         } else {
544                             dsLocation = null;
545                         }
546                     }
547 
548                     m_management.modifyDatastreamByReference(context,
549                                                             pid,
550                                                             dsID,
551                                                             altIDs,
552                                                             dsLabel,
553                                                             mimeType,
554                                                             formatURI,
555                                                             dsLocation,
556                                                             checksumType,
557                                                             checksum,
558                                                             logMessage,
559                                                             requestModDate);
560                 }
561 
562                 if (dsState != null) {
563                     if (dsState.equals("A") || dsState.equals("D")
564                             || dsState.equals("I")) {
565                         if (!dsState.equals(existingDS.DSState)) {
566                             m_management.setDatastreamState(context,
567                                                            pid,
568                                                            dsID,
569                                                            dsState,
570                                                            logMessage);
571                         }
572                     }
573                 }
574 
575                 if (versionable != existingDS.DSVersionable) {
576                     m_management.setDatastreamVersionable(context,
577                                                          pid,
578                                                          dsID,
579                                                          versionable,
580                                                          logMessage);
581                 }
582             }
583 
584             ResponseBuilder builder;
585             if (posted) {
586                 builder =
587                         Response.created(m_uriInfo.getRequestUri()
588                                 .resolve(URLEncoder.encode(dsID, DEFAULT_ENC)));
589             } else { // put
590                 builder = Response.ok();
591             }
592             builder.header("Content-Type", MediaType.TEXT_XML);
593             Datastream dsProfile =
594                     m_management.getDatastream(context, pid, dsID, null);
595             String xml =
596                     getSerializer(context).datastreamProfileToXML(pid,
597                                                                   dsID,
598                                                                   dsProfile,
599                                                                   null,
600                                                                   false);
601             builder.entity(xml);
602             return builder.build();
603         } catch (Exception ex) {
604             return handleException(ex, flash);
605         }
606     }
