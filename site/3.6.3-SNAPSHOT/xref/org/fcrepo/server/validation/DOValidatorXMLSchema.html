34  /**
35   * XML Schema validation for Digital Objects.
36   *
37   * @author Sandy Payette
38   */
39  public class DOValidatorXMLSchema
40          implements Constants, EntityResolver {
41  
42      private static final Logger logger =
43              LoggerFactory.getLogger(DOValidatorXMLSchema.class);
44  
45      /** Constants used for JAXP 1.2 */
46      private static final String JAXP_SCHEMA_LANGUAGE =
47              "http://java.sun.com/xml/jaxp/properties/schemaLanguage";
48  
49      private URI schemaURI = null;
50  
51      public DOValidatorXMLSchema(String schemaPath)
52              throws GeneralException {
53          try {
54              schemaURI = (new File(schemaPath)).toURI();
55          } catch (Exception e) {
56              logger.error("Error constructing validator", e);
57              throw new GeneralException(e.getMessage());
58          }
59      }
60  
61      public void validate(File objectAsFile) throws ObjectValidityException,
62              GeneralException {
63          try {
64              validate(new InputSource(new FileInputStream(objectAsFile)));
65          } catch (IOException e) {
66              String msg =
67                      "DOValidatorXMLSchema returned error.\n"
68                              + "The underlying exception was a "
69                              + e.getClass().getName() + ".\n"
70                              + "The message was " + "\"" + e.getMessage() + "\"";
71              throw new GeneralException(msg);
72          }
73      }
74  
75      public void validate(InputStream objectAsStream)
76              throws ObjectValidityException, GeneralException {
77          validate(new InputSource(objectAsStream));
78      }
79  
80      private void validate(InputSource objectAsSource)
81              throws ObjectValidityException, GeneralException {
82          InputSource doXML = objectAsSource;
83          try {
84              // XMLSchema validation via SAX parser
85              SAXParserFactory spf = SAXParserFactory.newInstance();
86              spf.setNamespaceAware(true);
87              spf.setValidating(true);
88              SAXParser sp = spf.newSAXParser();
89              sp.setProperty(JAXP_SCHEMA_LANGUAGE, XML_XSD.uri);
90  
91              // JAXP property for schema location
92              sp
93                      .setProperty("http://java.sun.com/xml/jaxp/properties/schemaSource",
94                                   schemaURI.toString());
95  
96              XMLReader xmlreader = sp.getXMLReader();
97              xmlreader.setErrorHandler(new DOValidatorXMLErrorHandler());
98              xmlreader.setEntityResolver(this);
99              xmlreader.parse(doXML);
100         } catch (ParserConfigurationException e) {
101             String msg =
102                     "DOValidatorXMLSchema returned parser error.\n"
103                             + "The underlying exception was a "
104                             + e.getClass().getName() + ".\n"
105                             + "The message was " + "\"" + e.getMessage() + "\"";
106             throw new GeneralException(msg, e);
107         } catch (SAXException e) {
108             String msg =
109                     "DOValidatorXMLSchema returned validation exception.\n"
110                             + "The underlying exception was a "
111                             + e.getClass().getName() + ".\n"
112                             + "The message was " + "\"" + e.getMessage() + "\"";
113             Validation validation = new Validation("unknown");
114             List<String> problems = new ArrayList<String>();
115             problems.add(msg);
116             validation.setObjectProblems(problems);
117             throw new ObjectValidityException(msg, validation, e);
118         } catch (Exception e) {
119             String msg =
120                     "DOValidatorXMLSchema returned error.\n"
121                             + "The underlying error was a "
122                             + e.getClass().getName() + ".\n"
123                             + "The message was " + "\"" + e.getMessage() + "\"";
124             throw new GeneralException(msg, e);
125         }
126     }

128     /**
129      * Resolve the entity if it's referring to a local schema. Otherwise, return
130      * an empty InputSource. This behavior is required in order to ensure that
131      * Xerces never attempts to load external schemas specified with
132      * xsi:schemaLocation. It is not enough that we specify
133      * processContents="skip" in our own schema.
134      */
135     public InputSource resolveEntity(String publicId, String systemId) {
136         if (systemId != null && systemId.startsWith("file:")) {
137             return null;
138         } else {
139             return new InputSource();
140         }
141     }
