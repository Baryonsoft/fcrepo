35  /**
36   * Schematron validation for fedora objects encoded in schematron schema for
37   * Fedora. The schematron schema (metsExtRules1-0.xml) expresses a set of rules
38   * using XPATH that enable us to check for things that are either not expressed
39   * in the METS XML schema, or that cannot be expressed with XML Schema language.
40   * Generally we will look for things that are requirements of Fedora objects,
41   * which are not requirements for METS objects in general.
42   *
43   * @author Sandy Payette
44   * @version $Id$
45   */
46  public class DOValidatorSchematron {

62       * preprocessor that is provided by the calling class. This will allow the
63       * DOValidator module to pass in the preprocessor that is configured with
64       * the Fedora repository.

104     /**
105      * Run the Schematron validation on a Fedora object.
106      *
107      * @param objectSource
108      *        the Fedora object as an StreamSource
109      * @throws ServerException
110      */
111     public void validate(StreamSource objectSource) throws ServerException {
112         DOValidatorSchematronResult result = null;
113         try {
114             // Create a transformer that uses the validating stylesheet.
115             // Run the Schematron validation of the Fedora object and
116             // output results in DOM format.
117             TransformerFactory tfactory = XmlTransformUtility.getTransformerFactory();
118             Transformer vtransformer =
119                     tfactory.newTransformer(validatingStyleSheet);
120             DOMResult validationResult = new DOMResult();
121             vtransformer.transform(objectSource, validationResult);
122             result = new DOValidatorSchematronResult(validationResult);
123         } catch (Exception e) {
124         	Validation validation = new Validation("unknown");
125         	List<String> probs = new ArrayList<String>();
126         	probs.add("Schematron validation failed:" + e.getMessage());
127         	validation.setObjectProblems(probs);
128             logger.error("Schematron validation failed", e);
129             throw new ObjectValidityException(e.getMessage(), validation);
130         }
131 
132         if (!result.isValid()) {
133             String msg = null;
134             try {
135                 msg = result.getXMLResult();
136             } catch (Exception e) {
137                 logger.warn("Error getting XML result of schematron validation failure", e);
138             }
139         	Validation validation = new Validation("unknown");
140         	List<String> probs = new ArrayList<String>();
141         	if (msg != null) {
142         		probs.add(msg);
143         	} else {
144         		probs.add("Unknown schematron error.  Error getting XML results of schematron validation");
145         	}
146             throw new ObjectValidityException(msg, validation);
147         }
148     }

150     /**
151      * Run setup to prepare for Schematron validation. This entails dynamically
152      * creating the validating stylesheet using the preprocessor and the schema.
153      *
154      * @param preprocessorPath
155      *        the location of the Schematron preprocessor
156      * @param fedoraschemaPath
157      *        the URL of the Schematron schema
158      * @param phase
159      *        the phase in the fedora object lifecycle to which validation
160      *        should pertain. (Currently options are "ingest" and "store")
161      * @return StreamSource
162      * @throws ObjectValidityException
163      */
164     private StreamSource setUp(String preprocessorPath,
165                                String fedoraschemaPath,
166                                String phase) throws ObjectValidityException {
167         String key = fedoraschemaPath + "#" + phase;
168         ByteArrayOutputStream out =
169                 generatedStyleSheets.get(key);
170         if (out == null) {
171             rulesSource = fileToStreamSource(fedoraschemaPath);
172             preprocessorSource = fileToStreamSource(preprocessorPath);
173             out =
174                     createValidatingStyleSheet(rulesSource,
175                                                preprocessorSource,
176                                                phase);
177             generatedStyleSheets.put(key, out);
178         }
179         return new StreamSource(new ByteArrayInputStream(out.toByteArray()));
180     }

182     /**
183      * Create the validating stylesheet which will be used to perform the actual
184      * Schematron validation. The validating stylesheet is created dynamically
185      * using the preprocessor stylesheet and the Schematron schema for Fedora.
186      * The phase is key. The stylesheet is created for the appropriate phase as
187      * specified in the Schematron rules schema. Valid work flow phases are
188      * currently "ingest" and "store." Different schematron rules apply to
189      * different phases of the object lifecycle. Some rules are applied when an
190      * object is first being ingested into the repository. Other rules apply
191      * before the object is put into permanent storage.
192      *
193      * @param rulesSource
194      *        the location of the rules
195      * @param preprocessorSource
196      *        the location of the Schematron preprocessor
197      * @param phase
198      *        the phase in the fedora object lifecycle to which validation
199      *        should pertain. (Currently options are "ingest" and "store"
200      * @return A ByteArrayOutputStream containing the stylesheet
201      * @throws ObjectValidityException
202      */
203     private ByteArrayOutputStream createValidatingStyleSheet(StreamSource rulesSource,
204                                                              StreamSource preprocessorSource,
205                                                              String phase)
206             throws ObjectValidityException {
207         ByteArrayOutputStream out = new ByteArrayOutputStream();
208         try {
209             // Create a transformer for that uses the Schematron preprocessor stylesheet.
210             // Transform the Schematron schema (rules) into a validating stylesheet.
211             TransformerFactory tfactory = XmlTransformUtility.getTransformerFactory();
212             Transformer ptransformer =
213                     tfactory.newTransformer(preprocessorSource);
214             ptransformer.setParameter("phase", phase);
215             ptransformer.transform(rulesSource, new StreamResult(out));
216         } catch (TransformerException e) {
217             logger.error("Schematron validation failed", e);
218             throw new ObjectValidityException(e.getMessage());
219         }
220         return out;
221     }

231     static public StreamSource fileToStreamSource(File file) {
232         String path = file.getAbsolutePath();
233         String fSep = System.getProperty("file.separator");
234         if (fSep != null && fSep.length() == 1) {
235             path = path.replace(fSep.charAt(0), '/');
236         }
237         if (path.length() > 0 && path.charAt(0) != '/') {
238             path = '/' + path;
239         }
240         try {
241             String url = new URL("file", null, path).toString();
242             return new StreamSource(url);
243         } catch (java.net.MalformedURLException e) {
244             /*
245              * According to the spec this could only happen if the file protocol
246              * were not recognized.
247              */
248             throw new Error("unexpected MalformedURLException");
249         }
250     }
