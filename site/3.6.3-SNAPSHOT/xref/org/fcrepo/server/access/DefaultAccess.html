137     @Override

139         // get ref to DOManager
140         m_manager =
141                 (DOManager) getServer()
142                         .getModule("org.fcrepo.server.storage.DOManager");
143         if (m_manager == null) {
144             throw new ModuleInitializationException("Can't get a DOManager "
145                                                     + "from Server.getModule", getRole());
146         }
147         // get ref to DynamicAccess module
148         m_dynamicAccess =
149                 (DynamicAccessModule) getServer()
150                         .getModule("org.fcrepo.server.access.DynamicAccess");
151 
152         // get ref to ExternalContentManager
153         m_externalContentManager =
154                 (ExternalContentManager) getServer()
155                         .getModule("org.fcrepo.server.storage.ExternalContentManager");
156 
157         // get ref to OAIProvider, for repositoryDomainName param for oai info
158         Module oaiProvider = getServer().getModule("org.fcrepo.oai.OAIProvider");
159         if (oaiProvider == null) {
160             throw new ModuleInitializationException("DefaultAccess module requires that the server "
161                                                     + "has an OAIProvider module configured so that it can get the repositoryDomainName parameter.",
162                                                     getRole());
163         }
164         m_repositoryDomainName =
165                 oaiProvider.getParameter("repositoryDomainName");
166         if (m_repositoryDomainName == null) {
167             throw new ModuleInitializationException("DefaultAccess module requires that the OAIProvider "
168                                                     + "module has the repositoryDomainName parameter specified.",
169                                                     getRole());
170         }
171 
172         m_authorizationModule =
173                 getServer().getBean("org.fcrepo.server.security.Authorization", Authorization.class);
174         if (m_authorizationModule == null) {
175             throw new ModuleInitializationException("Can't get an Authorization module (in default access) from Server.getModule",
176                                                     getRole());
177         }

187     /**
188      * <p>
189      * Disseminates the content produced by executing the specified method of
190      * the associated deployment object of the specified digital object.
191      * </p>
192      *
193      * @param context
194      *        The context of this request.
195      * @param PID
196      *        The persistent identifier of the digital object.
197      * @param sDefPID
198      *        The persistent identifier of the Service Definition object.
199      * @param methodName
200      *        The name of the method to be executed.
201      * @param userParms
202      *        An array of user-supplied method parameters consisting of
203      *        name/value pairs.
204      * @param asOfDateTime
205      *        The versioning datetime stamp.
206      * @return A MIME-typed stream containing the result of the dissemination.
207      * @throws ServerException
208      *         If any type of error occurred fulfilling the request.
209      */
210     @Override
211     public MIMETypedStream getDissemination(Context context,
212                                             String PID,
213                                             String sDefPID,
214                                             String methodName,
215                                             Property[] userParms,
216                                             Date asOfDateTime)
217             throws ServerException {
218         PID = Server.getPID(PID).toString();
219         sDefPID = Server.getPID(sDefPID).toString();
220         long initStartTime = new Date().getTime();
221         long startTime = new Date().getTime();
222         long stopTime;
223         long interval;
224         ServiceDeploymentReader deploymentReader = null;
225 
226         DOReader reader =
227                 m_manager.getReader(asOfDateTime == null, context, PID);
228         String authzAux_objState = reader.GetObjectState();
229 
230         // DYNAMIC!! If service deployment is defined as dynamic, then
231         // perform the dissemination via the DynamicAccess module.
232         if (m_dynamicAccess.isDynamicService(context, PID, sDefPID)) {
233             m_authorizationModule.enforceGetDissemination(context,
234                                                           PID,
235                                                           sDefPID,
236                                                           methodName,
237                                                           asOfDateTime,
238                                                           authzAux_objState,
239                                                           "A",
240                                                           "fedora-system:4",
241                                                           "A",
242                                                           "A");
243             MIMETypedStream retVal =
244                     m_dynamicAccess.getDissemination(context,
245                                                      PID,
246                                                      sDefPID,
247                                                      methodName,
248                                                      userParms,
249                                                      asOfDateTime);
250             stopTime = new Date().getTime();
251             interval = stopTime - startTime;
252             logger.debug("Roundtrip DynamicDisseminator: " + interval
253                          + " milliseconds.");
254             return retVal;
255         }
256 
257         /*
258          * Find the service deployment that is contractor for a model this
259          * object has, and deploys the requested service. If object<->model
260          * mappings are ever stored in the registry, this may be simplified.
261          */
262         String serviceDeploymentPID = null;
263         for (String cModelURI: reader.getContentModels()){
264             String cModelPID = cModelURI.substring("info:fedora/".length());
265 
266 /*
267         for (RelationshipTuple rel : reader.getRelationships(MODEL.HAS_MODEL,
268                                                              null)) {
269             String cModelPID = rel.getObjectPID();
270 */
271             String foundDeploymentPID =
272                     m_manager.lookupDeploymentForCModel(cModelPID, sDefPID);
273 
274             if (foundDeploymentPID != null) {
275                 if (serviceDeploymentPID != null
276                     && !foundDeploymentPID.equals(serviceDeploymentPID)) {
277                     throw new DisseminationException("More than one deployment ("
278                                                      + foundDeploymentPID
279                                                      + ", "
280                                                      + serviceDeploymentPID
281                                                      + ") found for service "
282                                                      + sDefPID
283                                                      + " in model "
284                                                      + cModelPID);
285 
286                 }
287 
288                 serviceDeploymentPID = foundDeploymentPID;
289             } else {
290                 logger.debug("No deployment for (" + cModelPID + ", " + sDefPID
291                              + ")");
292             }
293         }
294 
295         if (serviceDeploymentPID != null) {
296             deploymentReader =
297                     m_manager.getServiceDeploymentReader(false,
298                                                          context,
299                                                          serviceDeploymentPID);
300         }
301 
302         ServiceDefinitionReader sDefReader =
303                 m_manager.getServiceDefinitionReader(asOfDateTime == null,
304                                                      context,
305                                                      sDefPID);
306 
307         String authzAux_sdefState = sDefReader.GetObjectState();
308 
309         String authzAux_dissState = "unknown";
310 
311         /*
312          * if reader is null, it means that no suitable deployments have been
313          * found. This can happen if (a), the object does not have any models
314          * that have that service, or (b) the object has a suitable model, but
315          * no implementation of that service has been deployed. We do a bit of
316          * checking here to determine which case this represents, as the error
317          * message could be very useful.
318          */
319         if (deploymentReader == null) {
320 
321             boolean suitableModelFound = false;
322             String cModelPID = null;
323             String message = null;
324 
325 /*
326             models: for (RelationshipTuple rel : reader
327                     .getRelationships(MODEL.HAS_MODEL, null)) {
328                 cModelPID = rel.getObjectPID();
329 */
330             models: for (String cm:reader.getContentModels()){
331                 cModelPID = cm.substring(12);
332 
333                 /* Skip over system models */
334                 if (Models.contains("info:fedora/" + cModelPID)) {
335                     continue;
336                 }
337 
338                 /* Open up each model and peek at its sDefs for a match */
339                 for (RelationshipTuple r : m_manager.getReader(false,
340                                                                context,
341                                                                cModelPID)
342                         .getRelationships(MODEL.HAS_SERVICE, null)) {
343                     if (sDefPID.equals(r.getObjectPID())) {
344                         suitableModelFound = true;
345                         break models;
346                     }
347                 }
348             }
349 
350             if (suitableModelFound) {
351                 message =
352                         "Unable to find deployment for service " + sDefPID
353                         + " on " + reader.GetObjectPID() + " in model "
354                         + cModelPID;
355             } else {
356                 message =
357                         reader.GetObjectPID()
358                         + " does not have a model with service "
359                         + sDefPID;
360             }
361             throw new DisseminatorNotFoundException(message);
362         }
363         stopTime = new Date().getTime();
364         interval = stopTime - startTime;
365         logger.debug("Roundtrip Looping Diss: " + interval + " milliseconds.");
366 
367         // Check deployment object state
368         String authzAux_sDepState = deploymentReader.GetObjectState();
369         String authzAux_sDepPID = deploymentReader.GetObjectPID();
370 
371         m_authorizationModule.enforceGetDissemination(context,
372                                                       PID,
373                                                       sDefPID,
374                                                       methodName,
375                                                       asOfDateTime,
376                                                       authzAux_objState,
377                                                       authzAux_sdefState,
378                                                       authzAux_sDepPID,
379                                                       authzAux_sDepState,
380                                                       authzAux_dissState);
381 
382         // Get method parms
383         Hashtable<String, String> h_userParms = new Hashtable<String, String>();
384         MIMETypedStream dissemination = null;
385         MethodParmDef[] defaultMethodParms = null;
386 
387         startTime = new Date().getTime();
388         // Put any user-supplied method parameters into hash table
389         if (userParms != null) {
390             for (Property element : userParms) {
391                 h_userParms.put(element.name, element.value);
392             }
393         }
394         // Validate user-supplied parameters
395         validateUserParms(context,
396                           PID,
397                           sDefPID,
398                           deploymentReader,
399                           methodName,
400                           h_userParms,
401                           asOfDateTime);
402 
403         stopTime = new Date().getTime();
404         interval = stopTime - startTime;
405         logger.debug("Roundtrip Get/Validate User Parms: " + interval
406                      + " milliseconds.");
407 
408         startTime = new Date().getTime();
409         // SDP: GET INFO FROM DEPLOYMENT READER:
410         // Add any default method parameters to validated user parm list
411         defaultMethodParms =
412                 deploymentReader
413                         .getServiceMethodParms(methodName, asOfDateTime);
414         for (int i = 0; i < defaultMethodParms.length; i++) {
415             if (!defaultMethodParms[i].parmType
416                     .equals(MethodParmDef.DATASTREAM_INPUT)) {
417                 if (!h_userParms.containsKey(defaultMethodParms[i].parmName)) {
418                     logger.debug("addedDefaultName: "
419                                  + defaultMethodParms[i].parmName);
420                     String pdv = defaultMethodParms[i].parmDefaultValue;
421                     try {
422                         // here we make sure the PID is decoded so that encoding
423                         // later won't doubly-encode it
424                         if (pdv.equalsIgnoreCase("$pid")) {
425                             pdv = URLDecoder.decode(PID, "UTF-8");
426                         } else if (pdv.equalsIgnoreCase("$objuri")) {
427                             pdv =
428                                     "info:fedora/"
429                                     + URLDecoder.decode(PID, "UTF-8");
430                         }
431                     } catch (UnsupportedEncodingException uee) {
432                     }
433                     logger.debug("addedDefaultValue: " + pdv);
434                     h_userParms.put(defaultMethodParms[i].parmName, pdv);
435                 }
436             }
437         }
438 
439         stopTime = new Date().getTime();
440         interval = stopTime - startTime;
441         logger.debug("Roundtrip Get Deployment Parms: " + interval
442                      + " milliseconds.");
443 
444         startTime = new Date().getTime();
445         DisseminationBindingInfo[] dissBindInfo;
446         dissBindInfo =
447                 getDisseminationBindingInfo(context,
448                                             reader,
449                                             deploymentReader,
450                                             methodName,
451                                             asOfDateTime);
452 
453         // Assemble and execute the dissemination request from the binding info.
454         DisseminationService dissService = new DisseminationService(getServer());
455         dissemination =
456                 dissService.assembleDissemination(context,
457                                                   PID,
458                                                   h_userParms,
459                                                   dissBindInfo,
460                                                   authzAux_sDepPID,
461                                                   deploymentReader,
462                                                   methodName);
463 
464         stopTime = new Date().getTime();
465         interval = stopTime - startTime;
466         logger.debug("Roundtrip Assemble Dissemination: " + interval
467                      + " milliseconds.");
468 
469         stopTime = new Date().getTime();
470         interval = stopTime - initStartTime;
471         logger.debug("Roundtrip GetDissemination: " + interval + " milliseconds.");
472         return dissemination;
473     }

475     private DisseminationBindingInfo[] getDisseminationBindingInfo(Context context,
476                                                                    DOReader dObj,
477                                                                    ServiceDeploymentReader bmReader,
478                                                                    String methodName,
479                                                                    Date versDateTime)
480             throws MethodNotFoundException, ServerException {
481 
482         // The sDep reader provides information about the service and params.
483         MethodParmDef[] methodParms =
484                 bmReader.getServiceMethodParms(methodName, versDateTime);
485         // Find the operation bindings for the method in question
486         MethodDefOperationBind[] opBindings =
487                 bmReader.getServiceMethodBindings(versDateTime);
488 
489         String addressLocation = null;
490         String operationLocation = null;
491         String protocolType = null;
492         boolean foundMethod = false;
493         for (MethodDefOperationBind element : opBindings) {
494             if (element.methodName.equals(methodName)) {
495                 foundMethod = true;
496                 addressLocation = element.serviceBindingAddress;
497                 operationLocation = element.operationLocation;
498                 protocolType = element.protocolType;
499             }
500         }
501         if (!foundMethod) {
502             throw new MethodNotFoundException("Method " + methodName
503                                               + " was not found in " + bmReader.GetObjectPID()
504                                               + "'s operation " + " binding.");
505         }
506 
507         DeploymentDSBindSpec dsBindSpec =
508                 bmReader.getServiceDSInputSpec(versDateTime);
509         DeploymentDSBindRule[] dsBindRules =
510                 dsBindSpec.dsBindRules == null ? new DeploymentDSBindRule[0]
511                                                : dsBindSpec.dsBindRules;
512 
513         // Results will be returned in this list, one item per *existing*
514         // datastream. If a datastream mentioned in the dsBindRules is not
515         // present in the object, it will not be present in this list.
516         // If the datastream is *really* required in order to invoke the
517         // dissemination method in question, rest assured it will fail later.
518         List<DisseminationBindingInfo> bindingInfoList =
519                 new ArrayList<DisseminationBindingInfo>();
520 
521         for (int i = 0; i < dsBindRules.length; i++) {
522             DeploymentDSBindRule dsBindRule = dsBindRules[i];
523             String dsPid =
524                     dsBindRule.pid == null ? dObj.GetObjectPID()
525                                            : dsBindRule.pid;
526             String dsId = dsBindRule.bindingKeyName;
527 
528             DOReader reader = m_manager.getReader(false, context, dsPid);
529             Datastream ds = reader.GetDatastream(dsId, versDateTime);
530 
531             if (ds != null) {
532                 DisseminationBindingInfo bindingInfo =
533                         new DisseminationBindingInfo();
534                 bindingInfo.DSBindKey = dsId;
535                 bindingInfo.dsLocation = ds.DSLocation;
536                 bindingInfo.dsControlGroupType = ds.DSControlGrp;
537                 bindingInfo.dsID = ds.DatastreamID;
538                 bindingInfo.dsVersionID = ds.DSVersionID;
539                 bindingInfo.dsState = ds.DSState;
540                 bindingInfo.dsCreateDT = ds.DSCreateDT;
541                 // these will be the same for all elements of the array
542                 bindingInfo.methodParms = methodParms;
543                 bindingInfo.AddressLocation = addressLocation;
544                 bindingInfo.OperationLocation = operationLocation;
545                 bindingInfo.ProtocolType = protocolType;
546                 bindingInfoList.add(bindingInfo);
547             }
548         }
549         return bindingInfoList.toArray(new DisseminationBindingInfo[0]);
550     }

552     @Override
553     public ObjectMethodsDef[] listMethods(Context context,
554                                           String PID,
555                                           Date asOfDateTime)
556             throws ServerException {
557         long startTime = new Date().getTime();
558         PID = Server.getPID(PID).toString();
559         m_authorizationModule.enforceListMethods(context, PID, asOfDateTime);
560         DOReader reader =
561                 m_manager.getReader(Server.USE_DEFINITIVE_STORE, context, PID);
562 
563         ObjectMethodsDef[] methodDefs = reader.listMethods(asOfDateTime);
564         long stopTime = new Date().getTime();
565         long interval = stopTime - startTime;
566         logger.debug("Roundtrip listMethods: " + interval + " milliseconds.");
567 
568         // DYNAMIC!! Grab any dynamic method definitions and merge them with
569         // the statically bound method definitions
570         ObjectMethodsDef[] dynamicMethodDefs =
571                 //m_dynamicAccess.getObjectMethods(context, PID, asOfDateTime);
572                 m_dynamicAccess.listMethods(context, PID, asOfDateTime);
573         ArrayList<ObjectMethodsDef> methodList =
574                 new ArrayList<ObjectMethodsDef>();
575         for (ObjectMethodsDef element : methodDefs) {
576             methodList.add(element);
577         }
578         for (ObjectMethodsDef element : dynamicMethodDefs) {
579             methodList.add(element);
580         }
581         return methodList.toArray(new ObjectMethodsDef[0]);
582     }

819     private String convertToCSV(String list) {
820         // make sure values in the list are comma delimited
821         if (list == null) {
822             return "*";
823         }
824         String original = list.trim();
825         Pattern spaces = Pattern.compile(" ++");
826         Matcher m = spaces.matcher(original);
827         String interim = m.replaceAll(",");
828         Pattern multcommas = Pattern.compile(",++");
829         Matcher m2 = multcommas.matcher(interim);
830         String csv = m2.replaceAll(",");
831         return csv;
832     }

834     /**
835      * <p>
836      * Validates user-supplied method parameters against values in the
837      * corresponding Service Definition object. The method will validate for:
838      * </p>
839      * <ol>
840      * <li>Valid name - each name must match a valid method parameter name</li>
841      * <li>DefaultValue - any specified parameters with valid default values
842      * will have the default value substituted if the user-supplied value is
843      * null</li>
844      * <li>Required name - each required method parameter name must be present
845      * </ol>
846      *
847      * @param context
848      *        The context of this request.
849      * @param PID
850      *        The persistent identifier of the digital object.
851      * @param sDefPID
852      *        The persistent identifier of the Service Definition object.
853      * @param methodName
854      *        The name of the method.
855      * @param h_userParms
856      *        A hashtable of user-supplied method parameter name/value pairs.
857      * @param versDateTime
858      *        The version datetime stamp of the digital object.
859      * @throws ServerException
860      *         If any type of error occurred fulfilling the request.
861      */
862     private void validateUserParms(Context context,
863                                    String PID,
864                                    String sDefPID,
865                                    ServiceDeploymentReader sdepreader,
866                                    String methodName,
867                                    Hashtable<String, String> h_userParms,
868                                    Date versDateTime) throws ServerException {
869         PID = Server.getPID(PID).toString();
870         sDefPID = Server.getPID(sDefPID).toString();
871         MethodParmDef[] methodParms = null;
872         MethodParmDef methodParm = null;
873         StringBuffer sb = new StringBuffer();
874         Hashtable<String, MethodParmDef> h_validParms =
875                 new Hashtable<String, MethodParmDef>();
876         boolean isValid = true;
877 
878         if (sdepreader != null) // this code will be used for the CMDA example
879         {
880             MethodDef[] methods = sdepreader.getServiceMethods(versDateTime);
881             // Filter out parms that are internal to the mechanism and not part
882             // of the abstract method definition.  We just want user parms.
883             for (MethodDef element : methods) {
884                 if (element.methodName.equalsIgnoreCase(methodName)) {
885                     ArrayList<MethodParmDef> filteredParms =
886                             new ArrayList<MethodParmDef>();
887                     MethodParmDef[] parms = element.methodParms;
888                     for (MethodParmDef element2 : parms) {
889                         if (element2.parmType
890                                 .equalsIgnoreCase(MethodParmDef.USER_INPUT)) {
891                             filteredParms.add(element2);
892                         }
893                     }
894                     methodParms = filteredParms.toArray(new MethodParmDef[0]);
895                 }
896             }
897         } else {
898             String message =
899                     "[DefaultAccess] Old-style disseminators are no longer supported ";
900             throw new DisseminatorNotFoundException(message);
901             //        reader = m_manager.getReader(Server.GLOBAL_CHOICE, context, PID);
902             //        methodParms = reader.getObjectMethodParms(sDefPID, methodName, versDateTime);
903         }
904 
905         // Put valid method parameters and their attributes into hashtable
906         if (methodParms != null) {
907             for (int i = 0; i < methodParms.length; i++) {
908                 methodParm = methodParms[i];
909                 h_validParms.put(methodParm.parmName, methodParm);
910                 if (logger.isDebugEnabled()) {
911                     logger.debug("methodParms[" + i + "]: "
912                                  + methodParms[i].parmName + "\nlabel: "
913                                  + methodParms[i].parmLabel + "\ndefault: "
914                                  + methodParms[i].parmDefaultValue + "\nrequired: "
915                                  + methodParms[i].parmRequired + "\ntype: "
916                                  + methodParms[i].parmType);
917                     for (String element : methodParms[i].parmDomainValues) {
918                         logger.debug("domainValue: " + element);
919                     }
920                 }
921             }
922         }
923 
924         if (!h_validParms.isEmpty()) {
925             // Iterate over valid parmameters to check for any missing required parms.
926             Enumeration<String> e = h_validParms.keys();
927             while (e.hasMoreElements()) {
928                 String validName = e.nextElement();
929                 MethodParmDef mp = h_validParms.get(validName);
930                 if (mp.parmRequired && h_userParms.get(validName) == null) {
931                     // This is a fatal error. A required method parameter does not
932                     // appear in the list of user supplied parameters.
933                     sb.append("The required parameter \"" + validName
934                               + "\" was not found in the "
935                               + "user-supplied parameter list.");
936                     throw new InvalidUserParmException("[Invalid User Parameters] "
937                                                        + sb.toString());
938                 }
939             }
940 
941             // Iterate over each user supplied parameter name
942             Enumeration<String> parmNames = h_userParms.keys();
943             while (parmNames.hasMoreElements()) {
944                 String parmName = parmNames.nextElement();
945                 methodParm = h_validParms.get(parmName);
946                 if (methodParm != null && methodParm.parmName != null) {
947                     // Method has one or more parameters defined
948                     // Check for default value if user-supplied value is null or empty
949                     String value = h_userParms.get(methodParm.parmName);
950                     if (value == null || value.equalsIgnoreCase("")) {
951                         // Value of user-supplied parameter is  null or empty
952                         if (methodParm.parmDefaultValue != null) {
953                             // Default value is specified for this parameter.
954                             // Substitute default value.
955                             h_userParms.put(methodParm.parmName,
956                                             methodParm.parmDefaultValue);
957                         } else {
958                             // This is a non-fatal error. There is no default specified
959                             // for this parameter and the user has supplied no value for
960                             // the parameter. The value of the empty string will be used
961                             // as the value of the parameter.
962                             logger.warn("The method parameter \""
963                                         + methodParm.parmName
964                                         + "\" has no default value and no "
965                                         + "value was specified by the user.  "
966                                         + "The value of the empty string has "
967                                         + "been assigned to this parameter.");
968                         }
969                     } else {
970                         // Value of user-supplied parameter contains a value.
971                         // Validate the supplied value against the parmDomainValues list.
972                         String[] parmDomainValues = methodParm.parmDomainValues;
973                         if (parmDomainValues.length > 0) {
974                             if (!parmDomainValues[0].equalsIgnoreCase("null")) {
975                                 boolean isValidValue = false;
976                                 String userValue =
977                                         h_userParms.get(methodParm.parmName);
978                                 for (String element : parmDomainValues) {
979                                     if (userValue.equalsIgnoreCase(element)
980                                         || element.equalsIgnoreCase("null")) {
981                                         isValidValue = true;
982                                     }
983                                 }
984                                 if (!isValidValue) {
985                                     for (int i = 0; i < parmDomainValues.length; i++) {
986                                         if (i == parmDomainValues.length - 1) {
987                                             sb.append(parmDomainValues[i]);
988                                         } else {
989                                             sb.append(parmDomainValues[i]
990                                                       + ", ");
991                                         }
992                                     }
993                                     sb
994                                             .append("The method parameter \""
995                                                     + methodParm.parmName
996                                                     + "\" with a value of \""
997                                                     + h_userParms
998                                                     .get(methodParm.parmName)
999                                                     + "\" is not allowed for the method \""
1000                                                     + methodName
1001                                                     + "\". Allowed values for this "
1002                                                     + "method include \""
1003                                                     + sb.toString() + "\".");
1004                                     isValid = false;
1005                                 }
1006                             }
1007                         }
1008                     }
1009                 } else {
1010                     // This is a fatal error. A user-supplied parameter name does
1011                     // not match any valid parameter names for this method.
1012                     sb.append("The method parameter \"" + parmName
1013                               + "\" is not valid for the method \"" + methodName
1014                               + "\".");
1015                     isValid = false;
1016                 }
1017             }
1018         } else {
1019             // There are no method parameters define for this method.
1020             if (!h_userParms.isEmpty()) {
1021                 // This is an error. There are no method parameters defined for
1022                 // this method and user parameters are specified in the
1023                 // dissemination request.
1024                 Enumeration<String> e = h_userParms.keys();
1025                 while (e.hasMoreElements()) {
1026                     sb.append("The method parameter \"" + e.nextElement()
1027                               + "\" is not valid for the method \"" + methodName
1028                               + "\"." + "The method \"" + methodName
1029                               + "\" defines no method parameters.");
1030                 }
1031                 throw new InvalidUserParmException("[Invalid User Parameters] "
1032                                                    + sb.toString());
1033             }
1034         }
1035         if (!isValid) {
1036             throw new InvalidUserParmException("[Invalid User Parameter] "
1037                                                + sb.toString());
1038         }
1039         return;
1040     }
