26  /**
27   * Dynamically provides complete and accurate WSDL files for Fedora APIs. This
28   * servlet directly includes the common XSD type definitions in each WSDL file
29   * and ensures that the binding address reflects the base URL of the Fedora
30   * instance, based on the request URI.
31   * 
32   * @author Chris Wilper
33   */
34  public class WSDLServlet
35          extends HttpServlet {
36  
37      private static final long serialVersionUID = 1L;
38  
39      /**
40       * This servlet's path, relative to the root of the webapp. IMPLEMENTATION
41       * NOTE: For the servlet to work properly, this should match the url-pattern
42       * specified for this servlet's mapping in web.xml. This code intentionally
43       * does not use request.getServletPath due to inconsistent behavior in
44       * various servlet containers.
45       */
46      private static final String _SERVLET_PATH = "/wsdl";
47  
48      /** Relative path to our XSD source file. */
49      private static final String _XSD_PATH = "xsd/fedora-types.xsd";
50  
51      /** Source WSDL file relative paths, mapped by name */
52      private static final Map _WSDL_PATHS = new HashMap();
53  
54      /** Paths to each service endpoint, relative to the root of the webapp. */
55      private static final Map _SERVICE_PATHS = new HashMap();
56  
57      /** $FEDORA_HOME/server */
58      private File _serverDir;
59  
60      static {
61          _WSDL_PATHS.put("API-A", "access/Fedora-API-A.wsdl");
62          _WSDL_PATHS.put("API-A-LITE", "access/Fedora-API-A-LITE.wsdl");
63          _WSDL_PATHS.put("API-M", "management/Fedora-API-M.wsdl");
64          _WSDL_PATHS.put("API-M-LITE", "management/Fedora-API-M-LITE.wsdl");
65  
66          _SERVICE_PATHS.put("API-A", "services/access");
67          _SERVICE_PATHS.put("API-A-LITE", "");
68          _SERVICE_PATHS.put("API-M", "services/management");
69          _SERVICE_PATHS.put("API-M-LITE", "management");
70      }
71  
72      /**
73       * Respond to an HTTP GET request. The single parameter, "api", indicates
74       * which WSDL file to provide. If no parameters are given, a simple HTML
75       * index is given instead.
76       */
77      @Override
78      public void doGet(HttpServletRequest request, HttpServletResponse response)
79              throws IOException, ServletException {
80  
81          String body;
82          String api = request.getParameter("api");
83  
84          if (api == null || api.length() == 0) {
85              body = getIndex();
86              response.setContentType("text/html; charset=UTF-8");
87          } else {
88              body =
89                      getWSDL(api.toUpperCase(), request.getRequestURL()
90                              .toString());
91              response.setContentType("text/xml; charset=UTF-8");
92          }
93  
94          PrintWriter writer = response.getWriter();
95          writer.print(body);
96          writer.flush();
97          writer.close();
98      }
99  
100     /**
101      * Get a simple HTML index pointing to each WSDL file provided by the
102      * servlet.
103      */
104     private String getIndex() {
105         StringBuffer out = new StringBuffer();
106         out.append("<html><body>WSDL Index<ul>\n");
107         Iterator names = _WSDL_PATHS.keySet().iterator();
108         while (names.hasNext()) {
109             String name = (String) names.next();
110             out.append("<li> <a href=\"?api=" + name + "\">" + name
111                     + "</a></li>\n");
112         }
113         out.append("</ul></body></html>");
114         return out.toString();
115     }
116 
117     /**
118      * Get the self-contained WSDL given the api name and request URL.
119      */
120     private String getWSDL(String api, String requestURL) throws IOException,
121             ServletException {
122 
123         String wsdlPath = (String) _WSDL_PATHS.get(api);
124         if (wsdlPath != null) {
125 
126             File schemaFile = new File(_serverDir, _XSD_PATH);
127             File sourceWSDL = new File(_serverDir, wsdlPath);
128 
129             String baseURL = getFedoraBaseURL(requestURL);
130             String svcPath = (String) _SERVICE_PATHS.get(api);
131 
132             RuntimeWSDL wsdl =
133                     new RuntimeWSDL(schemaFile, sourceWSDL, baseURL + "/"
134                             + svcPath);
135 
136             StringWriter stringWriter = new StringWriter();
137             PrintWriter out = new PrintWriter(stringWriter);
138 
139             wsdl.serialize(out);
140             out.flush();
141             out.close();
142 
143             return stringWriter.toString();
144 
145         } else {
146             throw new ServletException("No such api: '" + api + "'");
147         }
148     }
149 
150     /**
151      * Determine the base URL of the Fedora webapp given the request URL to this
152      * servlet.
153      */
154     private String getFedoraBaseURL(String requestURL) throws ServletException {
155         int i = requestURL.lastIndexOf(_SERVLET_PATH);
156         if (i != -1) {
157             return requestURL.substring(0, i);
158         } else {
159             throw new ServletException("Unable to determine Fedora baseURL "
160                     + "from request URL.  Request URL does not contain the "
161                     + "string '" + _SERVLET_PATH + "', as expected.");
162         }
163     }
164 
165     /**
166      * Initialize by setting serverDir based on FEDORA_HOME.
167      */
168     @Override
169     public void init() throws ServletException {
170 
171         String fedoraHome = Constants.FEDORA_HOME;
172 
173         if (fedoraHome == null || fedoraHome.length() == 0) {
174             throw new ServletException("FEDORA_HOME is not defined");
175         } else {
176             _serverDir = new File(new File(fedoraHome), "server");
177             if (!_serverDir.isDirectory()) {
178                 throw new ServletException("No such directory: "
179                         + _serverDir.getPath());
180             }
181         }
182 
183     }
