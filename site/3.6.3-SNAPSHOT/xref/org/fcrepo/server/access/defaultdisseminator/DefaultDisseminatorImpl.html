45  /**
46   * Implements the methods defined in the DefaultDisseminator interface. This is
47   * the default Service Deployment that implements the "contract" of the default

92      /**
93       * Returns an HTML rendering of the object profile which contains key
94       * metadata from the object, plus URLs for the object's Dissemination Index
95       * and Item Index. The data is returned as HTML in a presentation-oriented
96       * format. This is accomplished by doing an XSLT transform on the XML that
97       * is obtained from getObjectProfile in API-A.
98       *
99       * @return html packaged as a MIMETypedStream
100      * @throws ServerException
101      */
102     public MIMETypedStream viewObjectProfile() throws ServerException {
103         try {
104             ObjectProfile profile =
105                     m_access.getObjectProfile(context,
106                                               reader.GetObjectPID(),
107                                               asOfDateTime);
108             InputStream in = null;
109             try {
110                 in =
111                         new ByteArrayInputStream(new ObjectInfoAsXML(context)
112                                 .getObjectProfile(reposBaseURL,
113                                                   profile,
114                                                   asOfDateTime)
115                                 .getBytes("UTF-8"));
116             } catch (UnsupportedEncodingException uee) {
117                 throw new GeneralException("[DefaultDisseminatorImpl] An error has occurred. "
118                         + "The error was a \""
119                         + uee.getClass().getName()
120                         + "\"  . The "
121                         + "Reason was \""
122                         + uee.getMessage()
123                         + "\"  .");
124             }
125             //InputStream in = getObjectProfile().getStream();
126             ByteArrayOutputStream out = new ByteArrayOutputStream();
127             File xslFile =
128                     new File(reposHomeDir, "access/viewObjectProfile.xslt");
129             TransformerFactory factory =
130                     XmlTransformUtility.getTransformerFactory();
131             Templates template =
132                     factory.newTemplates(new StreamSource(xslFile));
133             Transformer transformer = template.newTransformer();
134             transformer.setParameter("fedora", context
135                     .getEnvironmentValue(Constants.FEDORA_APP_CONTEXT_NAME));
136             transformer.transform(new StreamSource(in), new StreamResult(out));
137             in = new ByteArrayInputStream(out.toByteArray());
138             return new MIMETypedStream("text/html", in, null,out.toByteArray().length);
139         } catch (TransformerException e) {
140             throw new DisseminationException("[DefaultDisseminatorImpl] had an error "
141                     + "in transforming xml for viewObjectProfile. "
142                     + "Underlying exception was: " + e.getMessage());
143         }
144     }

146     /**
147      * Returns an HTML rendering of the Dissemination Index for the object. The
148      * Dissemination Index is a list of method definitions that represent all
149      * disseminations possible on the object. The Dissemination Index is
150      * returned as HTML in a presentation-oriented format. This is accomplished
151      * by doing an XSLT transform on the XML that is obtained from listMethods
152      * in API-A.
153      *
154      * @return html packaged as a MIMETypedStream
155      * @throws ServerException
156      */
157     public MIMETypedStream viewMethodIndex() throws ServerException {
158         // sdp: the dissemination index is disabled for service definition and deployment objects
159         // so send back a message saying so.
160         if (reader.hasContentModel(Models.SERVICE_DEFINITION_3_0)
161                 || reader.hasContentModel(Models.SERVICE_DEPLOYMENT_3_0)) {
162             return noMethodIndexMsg();
163         }
164 
165         // get xml expression of method definitions
166         ObjectMethodsDef[] methods =
167                 m_access.listMethods(context,
168                                      reader.GetObjectPID(),
169                                      asOfDateTime);
170         InputStream in = null;
171         try {
172             in =
173                     new ByteArrayInputStream(new ObjectInfoAsXML(context)
174                             .getMethodIndex(reposBaseURL,
175                                             reader.GetObjectPID(),
176                                             methods,
177                                             asOfDateTime).getBytes("UTF-8"));
178         } catch (UnsupportedEncodingException uee) {
179             throw new GeneralException("[DefaultDisseminatorImpl] An error has occurred. "
180                     + "The error was a \""
181                     + uee.getClass().getName()
182                     + "\"  . The "
183                     + "Reason was \""
184                     + uee.getMessage()
185                     + "\"  .");
186         }
187 
188         // transform the method definitions xml to an html view
189         try {
190             ByteArrayOutputStream out = new ByteArrayOutputStream();
191             File xslFile =
192                     new File(reposHomeDir, "access/listMethods.xslt");
193             TransformerFactory factory =
194                     XmlTransformUtility.getTransformerFactory();
195             Templates template =
196                     factory.newTemplates(new StreamSource(xslFile));
197             Transformer transformer = template.newTransformer();
198             transformer.setParameter("fedora", context
199                     .getEnvironmentValue(Constants.FEDORA_APP_CONTEXT_NAME));
200             transformer.transform(new StreamSource(in), new StreamResult(out));
201             in = new ByteArrayInputStream(out.toByteArray());
202             return new MIMETypedStream("text/html", in, null,out.toByteArray().length);
203         } catch (TransformerException e) {
204             throw new DisseminationException("[DefaultDisseminatorImpl] had an error "
205                     + "in transforming xml for viewItemIndex. "
206                     + "Underlying exception was: " + e.getMessage());
207         }
208     }

364     public static MethodDef[] reflectMethods() {
365         ArrayList<MethodDef> methodList = new ArrayList<MethodDef>();
366         MethodDef method = null;
367 
368         method = new MethodDef();
369         method.methodName = "viewObjectProfile";
370         method.methodLabel = "View description of the object";
371         method.methodParms = new MethodParmDef[0];
372         methodList.add(method);
373 
374         method = new MethodDef();
375         method.methodName = "viewMethodIndex";
376         method.methodLabel =
377                 "View a list of dissemination methods in the object";
378         method.methodParms = new MethodParmDef[0];
379         methodList.add(method);
380 
381         method = new MethodDef();
382         method.methodName = "viewItemIndex";
383         method.methodLabel = "View a list of items in the object";
384         method.methodParms = new MethodParmDef[0];
385         methodList.add(method);

388         method.methodName = "viewDublinCore";
389         method.methodLabel = "View the Dublin Core record for the object";
