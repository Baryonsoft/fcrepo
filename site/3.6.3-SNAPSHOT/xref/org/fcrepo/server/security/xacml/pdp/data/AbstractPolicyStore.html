40      public static synchronized void addDocuments(AbstractPolicyStore policyStore, boolean reload) throws PolicyStoreException,
41      FileNotFoundException {
42  
43          if (policyStore.policiesLoaded)
44              return;
45  
46          File[] files = PopulatePolicyDatabase.getPolicyFiles();
47          if (files.length == 0) {
48              return;
49          }
50          PolicyUtils utils = new PolicyUtils();
51  
52          // don't fail if a single policy fails, instead continue and list failed policies when done
53          StringBuilder failedPolicies = new StringBuilder();
54  
55          for (File f : files) {
56              try {
57                  String policyID = utils.getPolicyName(f);
58  
59                  // TODO: name mangling only if Fedora policy store; use consts for ns from that
60                  if (policyStore instanceof FedoraPolicyStore) {
61  
62                      // get the policy ID - note that adding a policy with no name will generate a PID from
63                      // the policy ID, but using the default PID namespace; we want specific namespace for bootstrap policies hence doing this here
64                      // if XACML policy ID contains a pid separator, escape it
65                      if (policyID.contains(":")) {
66                          policyID = policyID.replace(":", "%3A");
67                      }
68                      policyID = FedoraPolicyStore.FESL_BOOTSTRAP_POLICY_NAMESPACE + ":" + policyID;
69                  }
70  
71                  if (policyStore.contains(policyID) && !reload) {
72                      if (LOGGER.isDebugEnabled()) {
73                          LOGGER.debug("Policy database already contains " + policyID + " (" + f.getName()+ ")"
74                                       + ". Skipping.");
75                      }
76                  } else {
77                      AbstractPolicyStore.policyNames.add(policyStore.addPolicy(f, policyID));
78                  }
79              } catch (MelcoePDPException e){
80                  LOGGER.warn("Failed to add bootstrap policy " + f.getName() + " - " + e.getMessage());
81                  failedPolicies.append(f.getName() + "\n");
82              }
83  
84          }
85          if (failedPolicies.length() != 0) {
86              throw new PolicyStoreException("Failed to load some bootstrap policies: " + failedPolicies.toString());
87          }
88          policyStore.policiesLoaded = true;
89      }
