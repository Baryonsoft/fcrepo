39      /**
40       * Used to get an attribute. If one of those values isn't being asked for,
41       * or if the types are wrong, then an empty bag is returned.
42       *
43       * @param attributeType
44       *        the datatype of the attributes to find, which must be time, date,
45       *        or dateTime for this module to resolve a value
46       * @param attributeId
47       *        the identifier of the attributes to find, which must be one of the
48       *        three ENVIRONMENT_* fields for this module to resolve a value
49       * @param issuer
50       *        the issuer of the attributes, or null if unspecified
51       * @param subjectCategory
52       *        the category of the attribute or null, which ignored since this
53       *        only handles non-subjects
54       * @param context
55       *        the representation of the request data
56       * @param designatorType
57       *        the type of designator, which must be ENVIRONMENT_TARGET for this
58       *        module to resolve a value
59       * @return the result of attribute retrieval, which will be a bag with a
60       *         single attribute, an empty bag, or an error
61       */
62      @Override
63      public EvaluationResult findAttribute(URI attributeType,
64                                            URI attributeId,
65                                            URI issuer,
66                                            URI subjectCategory,
67                                            EvaluationCtx context,
68                                            int designatorType) {
69  
70          String resourceId = context.getResourceId().encode();
71          if (logger.isDebugEnabled()) {
72              logger.debug("RIAttributeFinder: [" + attributeType.toString() + "] "
73                      + attributeId + ", rid=" + resourceId);
74          }
75  
76          if (resourceId == null || resourceId.equals("")) {
77              return new EvaluationResult(BagAttribute
78                      .createEmptyBag(attributeType));
79          }
80  
81          if (resourceId.equals("/FedoraRepository")) {
82              return new EvaluationResult(BagAttribute
83                                          .createEmptyBag(attributeType));
84          }
85  
86          // figure out which attribute we're looking for
87          String attrName = attributeId.toString();
88  
89          // we only know about registered attributes from config file
90          Map<String,Attribute> allowedAttributes = m_attributes.get(designatorType);
91  
92          if (allowedAttributes == null) {
93              if (logger.isDebugEnabled()) {
94                  logger.debug("Does not know about designatorType: "
95                          + designatorType);
96              }
97              return new EvaluationResult(BagAttribute
98                      .createEmptyBag(attributeType));
99          }
100 
101         Attribute attribute = allowedAttributes.get(attrName);
102 
103         if (attribute == null) {
104             if (logger.isDebugEnabled()) {
105                 logger.debug("Does not know about attribute: " + attrName);
106             }
107             return new EvaluationResult(BagAttribute
108                     .createEmptyBag(attributeType));
109         }
110 
111 
112         EvaluationResult result = null;
113         try {
114             result = getEvaluationResult(resourceId, attrName, attribute, attributeType);
115         } catch (Exception e) {
116             logger.error("Error finding attribute: " + e.getMessage(), e);
117             return new EvaluationResult(BagAttribute
118                     .createEmptyBag(attributeType));
119         }
120 
121         return result;
122     }

124     /**
125      *
126      * @param resourceID - the hierarchical XACML resource ID
127      * @param attribute - attribute to get - this is a URI that maps to a Fedora relationship name
128      * @param type
129      * @return
130      * @throws AttributeFinderException
131      */
132     private EvaluationResult getEvaluationResult(String resourceID,
133                                                  String attributeID,
134                                                  Attribute attribute,
135                                                  URI type)
136             throws AttributeFinderException {
137 
138         // split up the path of the hierarchical resource id
139         String resourceParts[] = resourceID.split("/");
140         Set<String> results;
141 
142         // either the last part is the pid, or the last-but one is the pid and the last is the datastream
143         // if we have a pid, we query on that, if we have a datastream we query on the datastream
144         String subject; // the full subject, ie pid or pid/ds
145         String pid;
146         if (resourceParts.length > 1) {
147             if (resourceParts[resourceParts.length - 1].contains(":")) { // ends with a pid, we have pid only
148                 subject = resourceParts[resourceParts.length - 1];
149                 pid = subject;
150             } else { // datastream
151                 pid = resourceParts[resourceParts.length - 2];
152                 subject = pid + "/" + resourceParts[resourceParts.length - 1];
153             }
154         } else {
155             // eg /FedoraRepository, not a valid path to PID or PID/DS
156             logger.debug("Resource ID not valid path to PID or datastream: " + resourceID);
157             return new EvaluationResult(BagAttribute.createEmptyBag(type));
158         }
159 
160         logger.debug("Getting attribute for resource " + subject);
161 
162         // the different types of RI attribute specification...
163         // if there is no "query" option for the attribute
164         String query = attribute.get("query");
165         if (query == null) {
166             // it's a simple relationship lookup
167             // see if a relationship is specified, otherwise default to the attribute name URI
168             String relationship = attribute.get("relationship");
169             if (relationship == null) {
170                 relationship = attributeID; // default to use attribute URI as relationship if none specified
171             }
172 
173             // see if we are querying based on the resource (object, datstream etc) or just on the object (pid)
174             String target = attribute.get("target");
175             String queryTarget;
176             if (target != null && target.equals("object")) {
177                 queryTarget = pid;
178             } else {
179                 queryTarget = subject;
180             }
181 
182             Map<String, Set<String>> relationships;
183 
184             try {
185                 logger.debug("Getting attribute using relationship " + relationship);
186                 relationships = m_relationshipResolver.getRelationships(queryTarget, relationship);
187             } catch (MelcoeXacmlException e) {
188                 throw new AttributeFinderException(e.getMessage(), e);
189             }
190 
191             if (relationships == null || relationships.isEmpty()) {
192                 return new EvaluationResult(BagAttribute.createEmptyBag(type));
193             }
194 
195             // there will only be results for one attribute, this will get all the values
196             results = relationships.get(relationship);
197 
198         } else {
199             // get the language and query output variable
200             String queryLang = attribute.get("queryLang"); // language
201             String variable =  attribute.get("value"); // query text
202             String resource =  attribute.get("resource"); // resource marker in query
203             String object = attribute.get("object"); // object/pid marker in query
204 
205             String subjectURI = "info:fedora/" + subject;
206             String pidURI  = "info:fedora/" + pid;
207 
208             // replace the resource marker in the query with the subject
209             if (resource != null) {
210                 query = query.replace(resource, subjectURI);
211             }
212             // and the pid/object marker
213             if (object != null) {
214                 query = query.replace(object, pidURI);
215             }
216 
217             // run it
218             try {
219                 logger.debug("Using a " + queryLang + " query to get attribute " + attribute);
220                 results = m_relationshipResolver.getAttributesFromQuery(query, queryLang, variable);
221             } catch (MelcoeXacmlException e) {
222                 throw new AttributeFinderException(e.getMessage(), e);
223             }
224         }
225 
226         Set<AttributeValue> bagValues = new HashSet<AttributeValue>();
227         logger.debug("Attribute values found: " + results.size());
228         for (String s : results) {
229             AttributeValue attributeValue = null;
230             try {
231                 attributeValue = m_attributeFactory.createValue(type, s);
232             } catch (Exception e) {
233                 logger.error("Error creating attribute: " + e.getMessage(), e);
234                 continue;
235             }
236 
237             bagValues.add(attributeValue);
238 
239             if (logger.isDebugEnabled()) {
240                 logger.debug("AttributeValue found: [" + type.toASCIIString()
241                              + "] " + s);
242             }
243 
244         }
245 
246         BagAttribute bag = new BagAttribute(type, bagValues);
247         return new EvaluationResult(bag);
248 
249     }
