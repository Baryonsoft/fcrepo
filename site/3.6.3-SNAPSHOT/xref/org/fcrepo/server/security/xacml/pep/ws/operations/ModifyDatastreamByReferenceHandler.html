42  /**
43   * @author nishen@melcoe.mq.edu.au
44   */
45  public class ModifyDatastreamByReferenceHandler
46          extends AbstractOperationHandler {
47  
48      private static final Logger logger =
49              LoggerFactory.getLogger(ModifyDatastreamByReferenceHandler.class);
50  
51      public ModifyDatastreamByReferenceHandler(ContextHandler contextHandler)
52              throws PEPException {
53          super(contextHandler);
54      }
55  
56      @Override
57      public RequestCtx handleResponse(SOAPMessageContext context)
58              throws OperationHandlerException {
59          return null;
60      }
61  
62      @Override
63      public RequestCtx handleRequest(SOAPMessageContext context)
64              throws OperationHandlerException {
65          logger.debug("ModifyDatastreamByReferenceHandler/handleRequest!");
66  
67          RequestCtx req = null;
68          Object oMap = null;
69  
70          String pid = null;
71          String dsID = null;
72          String mimeType = null;
73          String formatURI = null;
74          String dsLocation = null;
75          String checksumType = null;
76          String checksum = null;
77  
78          try {
79              oMap = getSOAPRequestObjects(context);
80              logger.debug("Retrieved SOAP Request Objects");
81          } catch (SoapFault af) {
82              logger.error("Error obtaining SOAP Request Objects", af);
83              throw new OperationHandlerException("Error obtaining SOAP Request Objects",
84                                                  af);
85          }
86  
87          try {
88              pid = (String) callGetter("getPid",oMap);
89              dsID = (String) callGetter("getDsID", oMap);
90              mimeType = (String) callGetter("getMIMEType", oMap);
91              formatURI = (String) callGetter("getFormatURI", oMap);
92              dsLocation = (String) callGetter("getDsLocation", oMap);
93              checksumType = (String) callGetter("getChecksumType", oMap);
94              checksum = (String) callGetter("getChecksum", oMap);
95          } catch (Exception e) {
96              logger.error("Error obtaining parameters", e);
97              throw new OperationHandlerException("Error obtaining parameters.",
98                                                  e);
99          }
100 
101         logger.debug("Extracted SOAP Request Objects");
102 
103         Map<URI, AttributeValue> actions = new HashMap<URI, AttributeValue>();
104         Map<URI, AttributeValue> resAttr;
105 
106         try {
107             resAttr = ResourceAttributes.getResources(pid);
108             if (dsID != null && !"".equals(dsID)) {
109                 resAttr.put(Constants.DATASTREAM.ID.getURI(),
110                             new StringAttribute(dsID));
111             }
112             if (mimeType != null && !"".equals(mimeType)) {
113                 resAttr.put(Constants.DATASTREAM.NEW_MIME_TYPE.getURI(),
114                             new StringAttribute(mimeType));
115             }
116             if (formatURI != null && !"".equals(formatURI)) {
117                 resAttr.put(Constants.DATASTREAM.NEW_FORMAT_URI.getURI(),
118                             new AnyURIAttribute(new URI(formatURI)));
119             }
120             if (dsLocation != null && !"".equals(dsLocation)) {
121                 resAttr.put(Constants.DATASTREAM.NEW_LOCATION.getURI(),
122                             new AnyURIAttribute(new URI(dsLocation)));
123             }
124             if (checksumType != null && !"".equals(checksumType)) {
125                 resAttr.put(Constants.DATASTREAM.NEW_CHECKSUM_TYPE.getURI(),
126                             new StringAttribute(checksumType));
127             }
128             if (checksum != null && !"".equals(checksum)) {
129                 resAttr.put(Constants.DATASTREAM.NEW_CHECKSUM.getURI(),
130                             new StringAttribute(checksum));
131             }
132 
133             actions
134                     .put(Constants.ACTION.ID.getURI(),
135                          Constants.ACTION.MODIFY_DATASTREAM_BY_REFERENCE
136                                  .getStringAttribute());
137             actions.put(Constants.ACTION.API.getURI(),
138                         Constants.ACTION.APIM
139                                 .getStringAttribute());
140             // modifying the FeSL policy datastream requires policy management permissions
141             if (dsID != null && dsID.equals(FedoraPolicyStore.FESL_POLICY_DATASTREAM)) {
142                 actions.put(Constants.ACTION.ID.getURI(),
143                             Constants.ACTION.MANAGE_POLICIES.getStringAttribute());
144 
145             }
146 
147 
148             req =
149                     getContextHandler().buildRequest(getSubjects(context),
150                                                      actions,
151                                                      resAttr,
152                                                      getEnvironment(context));
153 
154             LogUtil.statLog(getUser(context),
155                             Constants.ACTION.MODIFY_DATASTREAM_BY_REFERENCE
156                                     .uri,
157                             pid,
158                             dsID);
159         } catch (Exception e) {
160             logger.error(e.getMessage(), e);
161             throw new OperationHandlerException(e.getMessage(), e);
162         }
163 
164         return req;
165     }
