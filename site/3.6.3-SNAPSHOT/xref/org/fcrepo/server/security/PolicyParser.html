40  /**
41   * A validating parser for XACML policies.
42   * <p>
43   * This class also provides a commandline XACML validation utility.
44   * <p>
45   * NOTE: Although instances may be re-used, this class is not thread-safe.
46   * Use the <code>copy()</code> method to support concurrent parsing.
47   */
48  public class PolicyParser {
49      
50      private static final String W3C_XML_SCHEMA_NS_URI = "http://www.w3.org/2001/XMLSchema";
51      
52      // Neither of these factories are thread-safe, so access is synchronized in methods below
53      private static final SchemaFactory SCHEMA_FACTORY = SchemaFactory.newInstance(W3C_XML_SCHEMA_NS_URI);
54  
55      private static final DocumentBuilderFactory BUILDER_FACTORY = DocumentBuilderFactory.newInstance();
56      static {
57          BUILDER_FACTORY.setIgnoringComments(true);
58          BUILDER_FACTORY.setNamespaceAware(true);
59      }
60  
61      private final Schema m_schema;
62  
63      private final Validator m_validator;
64  
65      private final DocumentBuilder m_domParser;
66  
67      /**
68       * Creates an instance that will validate according to the given schema.
69       *
70       * @param schemaStream the XSD schema to use for schema validation
71       * @throws IOException if the schema can't be read
72       * @throws SAXException if the schema isn't valid
73       */
74      public PolicyParser(InputStream schemaStream)
75              throws IOException, SAXException {
76          this(getSchema(schemaStream));
77      }
78  
79      // actual constructor keeps schema (which is thread safe) for cheap copying
80      private PolicyParser(Schema schema)
81              throws SAXException {
82          m_schema = schema;
83          m_validator = schema.newValidator();
84          m_domParser = createDOMParser();
85      }
86      /**
87       * Schema Factory is not thread safe
88       * @return
89       */
90      private static Schema getSchema(InputStream schemaStream) throws SAXException {
91          Schema result;
92          synchronized(SCHEMA_FACTORY){
93              result = SCHEMA_FACTORY.newSchema(new StreamSource(schemaStream));
94          }
95          return result;
96      }

98      /**
99       * Gets a new instance that uses the same schema as this one.
100      *
101      * @return a copy of this instance
102      */
103     public PolicyParser copy() {
104         try {
105             return new PolicyParser(m_schema);
106         } catch (SAXException wontHappen) {
107             throw new FaultException(wontHappen);
108         }
109     }

111     /**
112      * Parses the given policy and optionally schema validates it.
113      *
114      * @param policyStream
115      *          the serialized XACML policy
116      * @param validate
117      *          whether to schema validate
118      * @return the parsed policy.
119      * @throws ValidationException
120      *           if the given xml is not a valid policy. This will occur if it
121      *           is not well-formed XML, its root element is not named
122      *           <code>Policy</code> or <code>PolicySet</code>, it triggers
123      *           a parse exception in the Sun libraries when constructing an
124      *           <code>AbstractPolicy</code> from the DOM, or (if validation
125      *           is true) it is not schema-valid.
126      */
127     public AbstractPolicy parse(InputStream policyStream,
128                                 boolean schemaValidate)
129             throws ValidationException {
130 
131         // Parse; die if not well-formed
132         Document doc = null;
133         try {
134             doc = m_domParser.parse(policyStream);
135         } catch (Exception e) {
136             throw new ValidationException("Policy invalid; malformed XML", e);
137         }
138 
139         if (schemaValidate) {
140             // XSD-validate; die if not schema-valid
141             try {
142                 m_validator.validate(new DOMSource(doc));
143             } catch (Exception e) {
144                 throw new ValidationException("Policy invalid; schema"
145                                               + " validation failed", e);
146             }
147         }
148 
149         // Construct AbstractPolicy from doc; die if root isn't "Policy[Set]"
150         Element root = doc.getDocumentElement();
151         String rootName = root.getTagName();
152         try {
153             if (rootName.equals("Policy")) {
154                 return Policy.getInstance(root);
155             } else if (rootName.equals("PolicySet")) {
156                 return PolicySet.getInstance(root);
157             } else {
158                 throw new ValidationException("Policy invalid; root element is "
159                                               + rootName + ", but should be "
160                                               + "Policy or PolicySet");
161             }
162         } catch (ParsingException e) {
163             throw new ValidationException("Policy invalid; failed parsing by "
164                                           + "Sun XACML implementation", e);
165         }
166     }
167     
168     private static DocumentBuilder createDOMParser() {
169         try {
170             DocumentBuilder builder;
171             synchronized(BUILDER_FACTORY) {
172                 builder = BUILDER_FACTORY.newDocumentBuilder();
173             }
174             builder.setErrorHandler(new ThrowAllErrorHandler());
175             return builder;
176         } catch (ParserConfigurationException e) {
177             throw new FaultException(e);
178         }
179     }

181     /**
182      * Command-line utility for validating XACML policies.
183      * <p>
184      * Accepts a single argument: the path to the policy instance to validate.
185      * <p>
186      * Also requires that the com.sun.xacml.PolicySchema system property points
187      * to the XACML schema.
188      */
189     public static void main(String[] args) {
190         if (args.length != 1) {
191             fail("One argument required: /path/to/xacml-policy-to-validate.xml");
192         }
193         final String schemaPathProperty = "com.sun.xacml.PolicySchema";
194         String schemaPath = System.getProperty(schemaPathProperty);
195         if (schemaPath == null) {
196             fail("System property " + schemaPathProperty + " (path to XACML "
197                  + "schema) must be set. (e.g. -D" + schemaPathProperty
198                  + "=/path/to/schema)");
199         }
200         try {
201             InputStream instance = getStream(args[0]);
202             PolicyParser parser = new PolicyParser(getStream(schemaPath));
203             parser.parse(instance, true);
204             System.out.println("Validation successful");
205             System.exit(0);
206         } catch (ValidationException e) {
207             if (e.getCause() != null && e.getCause() instanceof SAXParseException) {
208                 fail(e.getCause().getMessage());
209             } else {
210                 fail(e);
211             }
212         } catch (Exception e) {
213             fail(e);
214         }
215     }
216 
217     private static InputStream getStream(String path) {
218         try {
219             return new FileInputStream(path);
220         } catch (Exception e) {
221             fail("File not found: " + path);
222             return null;
223         }
224     }

226     private static void fail(String message) {
227         System.out.println("ERROR: " + message);
228         System.out.println("Validation failed");
229         System.exit(1);
230     }

232     private static void fail(Exception e) {
233         e.printStackTrace();
234         fail(e.getClass().getName() + ": See above for detail");
235     }
