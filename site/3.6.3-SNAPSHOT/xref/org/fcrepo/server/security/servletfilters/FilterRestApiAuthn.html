16  @Deprecated
17  public class FilterRestApiAuthn
18          extends FilterEnforceAuthn {
19  
20      private static final Logger logger =
21              LoggerFactory.getLogger(FilterRestApiAuthn.class);
22  
23      @Override
24      public boolean doThisSubclass(ExtendedHttpServletRequest request,
25                                    HttpServletResponse response)
26              throws Throwable {
27          if (logger.isDebugEnabled()) {
28              logger.debug(enter("doThisSubclass()"));
29          }
30  
31          boolean enforceAuthN = false;
32  
33          // Since API-A AuthN is off, leave AuthN off for all GET requests
34          // except those which are known to be part of API-M
35          if(request.getMethod().equals("GET")) {
36              String requestPath = request.getPathInfo();
37              if(requestPath != null) {
38                  // API-M.export() or API-M.getObjectXML()
39                  if (requestPath.endsWith("/export") ||
40                      requestPath.endsWith("/objectXML")) {
41                      enforceAuthN = true;
42                  }
43                  // API-M.getDatastream()
44                  if (requestPath.contains("/datastreams/") &&
45                      !requestPath.endsWith("/content")) {
46                      enforceAuthN = true;
47                  }
48                  if (requestPath.endsWith("/datastreams") &&
49                      Boolean.valueOf(request.getParameter("profiles"))) {
50                      enforceAuthN = true;
51                  }
52                  if (!enforceAuthN) logger.warn("No authn requested for {}", requestPath);
53              }
54          } else {
55              enforceAuthN = true;
56          }
57  
58          if(enforceAuthN) {
59              return super.doThisSubclass(request, response);
60          } else {
61              return false;
62          }
63      }
