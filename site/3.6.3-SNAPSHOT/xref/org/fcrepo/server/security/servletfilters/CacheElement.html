71      /**
72       * Populates this cache element with the given authenticated state
73       * and named values, then puts it in the valid state.
74       *
75       * Note: Prior to the call, the element must be in the invalid state.
76       * TODO: The predicates parameter is deprecated and should be removed.
77       *        For now, callers can avoid a warning by giving it as null.
78       */
79      public final void populate(Boolean authenticated,
80                                 Set predicates,
81                                 Map namedValues,
82                                 String errorMessage) {
83          String m = m_cacheabbrev + " populate() ";
84          logger.debug(m + ">");
85          try {
86              if (predicates != null) {
87                  logger.warn(m + " predicates are deprecated; will be ignored");
88              }
89              assertInvalid();
90              if (errorMessage != null) {
91                  logger.error(m + "errorMessage==" + errorMessage);
92                  throw new Exception(errorMessage);
93              } else {
94                  validate(authenticated, namedValues);
95                  // can't set expiration here -- don't have cache reference
96                  // can't set pwd here, don't have it
97              }
98          } catch (Throwable t) {
99              logger.error(m + "invalidating to be sure");
100             this.invalidate(errorMessage);
101         } finally {
102             logger.debug(m + "<");
103         }
104     }

106     /**
107      * If in the valid state and not expired:
108      *   If authenticated and given password is instance password, return true.
109      *   Else return false.
110      * If invalid or expired:
111      *   Re-initialize this element by setting it to invalid state and running
112      *     the underlying authN code.
113      *   If never authenticated or currently not valid, return m_authenticated
114      *   If authenticated, set thekkkk instance password to the given one and return true.
115      *   If not authenticated, return false.
116      */
117     public final synchronized Boolean authenticate(Cache cache, String pwd) {
118         // Original Comment:
119         // Synchronized so evaluation of cache item state will be sequential,
120         // non-interlaced.  This protects against overlapping calls resulting in
121         // redundant authenticator calls.
122         String m = m_cacheabbrev + " authenticate() ";
123         logger.debug(m + ">");
124         Boolean rc = null;
125         try {
126             logger.debug(m + "m_valid==" + m_valid);
127             if (m_valid && !CacheElement.isExpired(m_expiration)) {
128                 logger.debug(m + "valid and not expired, so use");
129                 if (!isAuthenticated()) {
130                     logger.debug(m + "auth==" + m_authenticated);
131                     rc = m_authenticated;
132                 } else {
133                     logger.debug(m + "already authd, request password==" + pwd);
134                     if (pwd == null) {
135                         logger.debug(m + "null request password");
136                         rc = Boolean.FALSE;
137                     } else if ("".equals(pwd)) {
138                         logger.debug(m + "zero-length request password");
139                         rc = Boolean.FALSE;
140                     } else {
141                         logger.debug(m + "stored password==" + m_password);
142                         rc = pwd.equals(m_password);
143                     }
144                 }
145             } else { // expired or invalid
146                 logger.debug(m + "expired or invalid, so try to repopulate");
147                 this.invalidate();
148                 CacheElementPopulator cePop = cache.getCacheElementPopulator();
149                 cePop.populateCacheElement(this, pwd);
150                 int duration = 0;
151                 String unit = null;
152                 m_password = null;
153                 if (m_authenticated == null || !m_valid) {
154                     duration = cache.getAuthExceptionTimeoutDuration();
155                     unit = cache.getAuthExceptionTimeoutUnit();
156                     logger.debug(m + "couldn't complete population");
157                 } else {
158                     logger.debug(m + "populate completed");
159                     if (isAuthenticated()) {
160                         m_password = pwd;
161                         duration = cache.getAuthSuccessTimeoutDuration();
162                         unit = cache.getAuthSuccessTimeoutUnit();
163                         logger.debug(m + "populate succeeded");
164                     } else {
165                         duration = cache.getAuthFailureTimeoutDuration();
166                         unit = cache.getAuthFailureTimeoutUnit();
167                         logger.debug(m + "populate failed");
168                     }
169                 }
170                 m_expiration = CacheElement.calcExpiration(duration, unit);
171                 rc = m_authenticated;
172             }
173         } catch (Throwable th) {
174             this.invalidate();
175             rc = m_authenticated;
176             logger.error(m + "invalidating to be sure");
177         } finally {
178             audit();
179             logger.debug(m + "< " + rc);
180         }
181         return rc;
182     }

310     private static final void checkCalcExpiration(int duration, int unit)
311             throws IllegalArgumentException {
312         if (duration < 0) {
313             throw new IllegalArgumentException("bad duration==" + duration);
314         }
315         switch (unit) {
316             case Calendar.MILLISECOND:
317             case Calendar.SECOND:
318             case Calendar.MINUTE:
319             case Calendar.HOUR:
320                 break;
321             default:
322                 throw new IllegalArgumentException("bad unit==" + unit);
323         }

338     private static final void auditNamedValues(String m, Map namedValues) {
339         if (logger.isDebugEnabled()) {
340             assert namedValues != null;
341             for (Iterator outer = namedValues.keySet().iterator(); outer
342                     .hasNext();) {
343                 Object name = outer.next();
344                 assert name instanceof String : "not a string, name==" + name;
345                 StringBuffer sb = new StringBuffer(m + name + "==");
346                 Object temp = namedValues.get(name);
347                 assert temp instanceof String || temp instanceof Set : "neither string nor set, temp=="
348                         + temp;
349                 if (temp instanceof String) {
350                     sb.append(temp.toString());
351                 } else if (temp instanceof Set) {
352                     Set values = (Set) temp;
353                     sb.append("(" + values.size() + ") {");
354                     String punct = "";
355                     for (Iterator it = values.iterator(); it.hasNext();) {
356                         temp = it.next();
357                         if (!(temp instanceof String)) {
358                             logger.error(m + "set member not string, ==" + temp);
359                         } else {
360                             String value = (String) temp;
361                             sb.append(punct + value);
362                             punct = ",";
363                         }
364                     }
365                     sb.append("}");
366                 }
367                 logger.debug(sb.toString());
368             }
369         }
370     }

372     private static final String pad(long i, String pad, boolean padLeft) {
373         String rc = "";
374         String st = Long.toString(i);
375         if (st.length() == pad.length()) {
376             rc = st;
377         } else if (st.length() > pad.length()) {
378             rc = st.substring(0, pad.length());
379         } else {
380             String padNeeded = pad.substring(0, pad.length() - st.length());
381             if (padLeft) {
382                 rc = padNeeded + st;
383             } else {
384                 rc = st + padNeeded;
385             }
386         }
387         return rc;
388     }

444     private static final String difference(Calendar earlier, Calendar later) {
445         long milliseconds = later.getTimeInMillis() - earlier.getTimeInMillis();
446 
447         long days = milliseconds / MILLIS_IN_DAY;
448         milliseconds = milliseconds % MILLIS_IN_DAY;
449         long hours = milliseconds / MILLIS_IN_HOUR;
450         milliseconds = milliseconds % MILLIS_IN_HOUR;
451         long minutes = milliseconds / MILLIS_IN_MINUTE;
452         milliseconds = milliseconds % MILLIS_IN_MINUTE;
453         long seconds = milliseconds / MILLIS_IN_SECOND;
454         milliseconds = milliseconds % MILLIS_IN_SECOND;
455         String rc = format(days, hours, minutes, seconds, milliseconds, null);
456         return rc;
457     }

459     private static final String compareForExpiration(Calendar first,
460                                                     Calendar second) {
461         String rc = null;
462         if (first.before(second)) {
463             rc = "expires in " + difference(first, second);
464         } else {
465             rc = "expired " + difference(second, first) + " ago";
466         }

470     private static final Calendar calcExpiration(int duration, int unit) {
471         String m = "- calcExpiration(int,int) ";
472         logger.debug(m + ">");
473         Calendar now = Calendar.getInstance();
474         Calendar rc = Calendar.getInstance();
475         try {
476             CacheElement.checkCalcExpiration(duration, unit);
477             if (duration > 0) {
478                 rc.add(unit, duration);
479                 logger.debug(m + CacheElement.compareForExpiration(now, rc));
480             } else {
481                 logger.debug(m + "timeout set to now (effectively, no caching)");
482             }
483         } finally {
484             if (logger.isDebugEnabled()) {
485                 logger.debug(m + "< " + format(rc));
486             }
487         }
488         return rc;
489     }

491     private static final int calcCalendarUnit(String unit) {
492         String m = "- calcCalendarUnit() ";
493         int rc = Calendar.SECOND;
494         if (!unit.endsWith("s")) {
495             unit += "s";
496         }
497         if ("milliseconds".equalsIgnoreCase(unit)) {
498             rc = Calendar.MILLISECOND;
499         } else if ("seconds".equalsIgnoreCase(unit)) {
500             rc = Calendar.SECOND;
501         } else if ("minutes".equalsIgnoreCase(unit)) {
502             rc = Calendar.MINUTE;
503         } else if ("hours".equalsIgnoreCase(unit)) {
504             rc = Calendar.HOUR;
505         } else {
506             String msg = "illegal Calendar unit: " + unit;
507             logger.error(m + "(" + msg + ")");
508             throw new IllegalArgumentException(msg);
509         }
510         return rc;
511     }

529     private static final boolean isExpired(Calendar now,
530                                           Calendar expiration,
531                                           boolean verbose) {
532         String m = "- isExpired() ";
533         if (verbose) {
534             logger.debug(m + ">");
535         }
536         boolean rc = CacheElement.s_expired_default;
537         try {
538             if (now == null) {
539                 String msg = "illegal parm now==" + now;
540                 logger.error(m + "(" + msg + ")");
541                 throw new IllegalArgumentException(msg);
542             }
543             if (expiration == null) {
544                 String msg = "illegal parm expiration==" + expiration;
545                 logger.error(m + "(" + msg + ")");
546                 throw new IllegalArgumentException(msg);
547             }
548             if (verbose) {
549                 logger.debug(m + "now==" + format(now));
550                 logger.debug(m + "exp==" + format(expiration));
551             }
552             rc = !now.before(expiration);
553         } catch (Throwable th) {
554             logger.error(m + "failed comparison");
555             rc = CacheElement.s_expired_default;
556         } finally {
557             if (verbose) {
558                 logger.debug(m + compareForExpiration(now, expiration));
559                 logger.debug(m + "< " + rc);
560             }
561         }
562         return rc;
563     }
