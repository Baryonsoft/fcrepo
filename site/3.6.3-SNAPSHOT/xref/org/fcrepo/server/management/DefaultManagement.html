143     public String ingest(Context context,
144                          InputStream serialization,
145                          String logMessage,
146                          String format,
147                          String encoding,
148                          String pid) throws ServerException {
149         DOWriter w = null;
150         String objPid = null;
151         try {
152             logger.debug("Entered ingest");
153             w = m_manager.getIngestWriter(Server.USE_DEFINITIVE_STORE,
154                                           context,
155                                           serialization,
156                                           format,
157                                           encoding,
158                                           pid);
159             objPid = w.GetObjectPID();
160 
161             m_authz.enforceIngest(context, objPid, format, encoding);
162 
163             // Only create an audit record if there is a log message to capture
164             if (logMessage != null && !logMessage.equals("")) {
165                 Date nowUTC = Server.getCurrentDate(context);
166                 addAuditRecord(context, w, "ingest", "", logMessage, nowUTC);
167             }
168 
169             w.commit(logMessage);
170             return objPid;
171         } finally {
172             // Logger completion
173             if (logger.isInfoEnabled()) {
174                 StringBuilder logMsg = new StringBuilder("Completed ingest(");
175                 logMsg.append("objectXML");
176                 logMsg.append(", format: ").append(format);
177                 logMsg.append(", encoding: ").append(encoding);
178                 logMsg.append(", pid	: ").append(objPid);
179                 logMsg.append(", logMessage: ").append(logMessage);
180                 logMsg.append(")");
181                 logger.info(logMsg.toString());
182             }
183 
184             finishModification(w, "ingest");
185         }
186     }

188     private void finishModification(DOWriter w, String method)
189             throws ServerException {
190         if (w != null) {
191             m_manager.releaseWriter(w);
192         }
193         if (logger.isDebugEnabled()) {
194             logger.debug("Exiting " + method);
195             Runtime r = Runtime.getRuntime();
196             logger.debug("Memory: " + r.freeMemory() + " bytes free of "
197                          + r.totalMemory() + " available.");
198         }

201     @Override
202     public Date modifyObject(Context context,
203                              String pid,
204                              String state,
205                              String label,
206                              String ownerId,
207                              String logMessage,
208                              Date lastModifiedDate) throws ServerException {
209         DOWriter w = null;
210         try {
211             logger.debug("Entered modifyObject");
212 
213             m_authz.enforceModifyObject(context,
214                                         pid,
215                                         state,
216                                         ownerId);
217 
218             checkObjectLabel(label);
219 
220             w = m_manager.getWriter(Server.USE_DEFINITIVE_STORE, context, pid);
221 
222             // if provided, check request lastModifiedDate against the object,
223             // rejecting the request if the object's mod date is more recent.
224             if (lastModifiedDate != null) {
225                 if (lastModifiedDate.before(w.getLastModDate())) {
226                     String objDate = DateUtility.convertDateToXSDString(w.getLastModDate());
227                     String reqDate = DateUtility.convertDateToXSDString(lastModifiedDate);
228                     String msg = String.format("%s lastModifiedDate (%s) " +
229                                                "is more recent than the " +
230                                                "request (%s)", pid, objDate, reqDate);
231                     throw new ObjectLockedException(msg);
232                 }
233             }
234 
235             if (state != null && !state.equals("")) {
236                 if (!state.equals("A") && !state.equals("D")
237                     && !state.equals("I")) {
238                     throw new InvalidStateException("The object state of \""
239                                                     + state
240                                                     + "\" is invalid. The allowed values for state are: "
241                                                     + " A (active), D (deleted), and I (inactive).");
242                 }
243                 w.setState(state);
244             }
245 
246             if (label != null) {
247                 w.setLabel(label);
248             }
249             if (ownerId != null) {
250                 w.setOwnerId(ownerId);
251             }
252 
253             // Update audit trail
254             Date nowUTC = Server.getCurrentDate(context);
255             addAuditRecord(context, w, "modifyObject", "", logMessage, nowUTC);
256 
257             w.commit(logMessage);
258             return w.getLastModDate();
259         } finally {
260             // Logger completion
261             if (logger.isInfoEnabled()) {
262                 StringBuilder logMsg =
263                         new StringBuilder("Completed modifyObject(");
264                 logMsg.append("pid: ").append(pid);
265                 logMsg.append(", state: ").append(state);
266                 logMsg.append(", label: ").append(label);
267                 logMsg.append(", ownderId: ").append(ownerId);
268                 logMsg.append(", logMessage: ").append(logMessage);
269                 logMsg.append(")");
270                 logger.info(logMsg.toString());
271             }
272 
273             finishModification(w, "modifyObject");
274         }

380     @Override
381     public String addDatastream(Context context,
382                                 String pid,
383                                 String dsID,
384                                 String[] altIDs,
385                                 String dsLabel,
386                                 boolean versionable,
387                                 String MIMEType,
388                                 String formatURI,
389                                 String dsLocation,
390                                 String controlGroup,
391                                 String dsState,
392                                 String checksumType,
393                                 String checksum,
394                                 String logMessage) throws ServerException {
395 
396         logger.debug("Entered addDatastream");
397 
398         // empty MIME types are allowed. assume they meant "" if they provide it
399         // as null.
400         if (MIMEType == null) {
401             MIMEType = "";
402         }
403 
404         // empty altIDs are allowed. assume they meant String[0] if they provide
405         // it as null.
406         if (altIDs == null) {
407             altIDs = new String[0];
408         }
409 
410         // If the datastream ID is not specified directly, see
411         // if we can get it from the RecoveryContext
412         if (dsID == null && context instanceof RecoveryContext) {
413             RecoveryContext rContext = (RecoveryContext) context;
414             dsID =
415                     rContext
416                             .getRecoveryValue(Constants.RECOVERY.DATASTREAM_ID.uri);
417             if (dsID != null) {
418                 logger.debug("Using new dsID from recovery context");
419             }
420         }
421 
422         // check for valid xml name for datastream ID
423         if (dsID != null) {
424             if (!XMLUtils.isWellFormedXMLName(dsID)) {
425                 throw new InvalidXMLNameException("Invalid syntax for datastream ID. "
426                                                   + "The datastream ID of \""
427                                                   + dsID
428                                                   + "\" is"
429                                                   + "not a valid XML Name");
430             }
431         }
432 
433         if (dsID != null
434             && (dsID.equals("AUDIT") || dsID.equals("FEDORA-AUDITTRAIL"))) {
435             throw new GeneralException("Creation of a datastream with an"
436                                        + " identifier of 'AUDIT' or 'FEDORA-AUDITTRAIL' is not permitted.");
437         }
438         DOWriter w = null;
439         try {
440             m_authz.enforceAddDatastream(context,
441                                          pid,
442                                          dsID,
443                                          altIDs,
444                                          MIMEType,
445                                          formatURI,
446                                          dsLocation,
447                                          controlGroup,
448                                          dsState,
449                                          checksumType,
450                                          checksum);
451 
452             checkDatastreamID(dsID);
453             checkDatastreamLabel(dsLabel);
454 
455             w = m_manager.getWriter(Server.USE_DEFINITIVE_STORE, context, pid);
456             Datastream ds;
457             if (controlGroup.equals("X")) {
458                 ds = new DatastreamXMLMetadata();
459                 ds.DSInfoType = ""; // field is now deprecated
460                 try {
461                     InputStream in;
462                     MIMETypedStream mimeTypedStream = null;
463                     if (dsLocation.startsWith(DatastreamManagedContent.UPLOADED_SCHEME)) {
464                         in = getTempStream(dsLocation);
465                     } else {
466                         ContentManagerParams params = new ContentManagerParams(dsLocation);
467                         params.setContext(context);
468                         mimeTypedStream = m_contentManager.getExternalContent(params);
469                         in = mimeTypedStream.getStream();
470                     }
471                     // set and validate the content
472                     DatastreamXMLMetadata dsm = (DatastreamXMLMetadata) ds;
473                     dsm.xmlContent = getEmbeddableXML(in);
474                     dsm.DSSize = dsm.xmlContent.length;
475                     ValidationUtility.validateReservedDatastream(PID.getInstance(pid),
476                                                                  dsID,
477                                                                  dsm);
478                     if (mimeTypedStream != null) {
479                         mimeTypedStream.close();
480                     }
481                 } catch (Exception e) {
482                     String extraInfo;
483                     if (e.getMessage() == null) {
484                         extraInfo = "";
485                     } else {
486                         extraInfo = " : " + e.getMessage();
487                     }
488                     throw new GeneralException("Error with " + dsLocation
489                                                + extraInfo);
490                 }
491             } else if (controlGroup.equals("M")) {
492                 ds = new DatastreamManagedContent();
493                 ds.DSInfoType = "DATA";
494                 ds.DSLocationType = Datastream.DS_LOCATION_TYPE_URL;
495             } else if (controlGroup.equals("R") || controlGroup.equals("E")) {
496                 ds = new DatastreamReferencedContent();
497                 ds.DSInfoType = "DATA";
498                 ds.DSLocationType = Datastream.DS_LOCATION_TYPE_URL;
499             } else {
500                 throw new GeneralException("Invalid control group: "
501                                            + controlGroup);
502             }
503             ds.isNew = true;
504             ds.DSControlGrp = controlGroup;
505             ds.DSVersionable = versionable;
506             if (!dsState.equals("A") && !dsState.equals("D")
507                 && !dsState.equals("I")) {
508                 throw new InvalidStateException("The datastream state of \""
509                                                 + dsState
510                                                 + "\" is invalid. The allowed values for state are: "
511                                                 + " A (active), D (deleted), and I (inactive).");
512             }
513             ds.DSState = dsState;
514             // set new datastream id if not provided...
515             if (dsID == null || dsID.length() == 0) {
516                 ds.DatastreamID = w.newDatastreamID();
517             } else {
518                 if (dsID.indexOf(" ") != -1) {
519                     throw new GeneralException("Datastream ids cannot contain spaces.");
520                 }
521                 if (dsID.indexOf("+") != -1) {
522                     throw new GeneralException("Datastream ids cannot contain plusses.");
523                 }
524                 if (dsID.indexOf(":") != -1) {
525                     throw new GeneralException("Datastream ids cannot contain colons.");
526                 }
527                 if (w.GetDatastream(dsID, null) != null) {
528                     throw new GeneralException("A datastream already exists with ID: "
529                                                + dsID);
530                 } else {
531                     ds.DatastreamID = dsID;
532                 }
533             }
534             // add version level attributes and
535             // create new ds version id ...
536             ds.DSVersionID = ds.DatastreamID + ".0";
537             ds.DSLabel = dsLabel;
538             ds.DSLocation = dsLocation;
539             if (dsLocation != null) {
540                 ValidationUtility.validateURL(dsLocation, ds.DSControlGrp);
541             }
542             ds.DSFormatURI = formatURI;
543             ds.DatastreamAltIDs = altIDs;
544             ds.DSMIME = MIMEType;
545             ds.DSChecksumType = Datastream.validateChecksumType(checksumType);
546 
547             // M reserved datastream validation (X done above)
548             if (controlGroup.equals("M")) {
549                 ValidationUtility.validateReservedDatastream(PID.getInstance(pid),
550                                                              dsID,
551                                                              ds);
552             }
553 
554             if (checksum != null && checksumType != null) {
555                 String check = ds.getChecksum();
556                 if (!checksum.equals(check)) {
557                     throw new ValidationException("Checksum Mismatch: " + check);
558                 }
559             }
560 
561             // Update audit trail
562             Date nowUTC = Server.getCurrentDate(context);
563             addAuditRecord(context,
564                            w,
565                            "addDatastream",
566                            ds.DatastreamID,
567                            logMessage,
568                            nowUTC);
569 
570             // Commit the updates
571             ds.DSCreateDT = nowUTC;
572             w.addDatastream(ds, true);
573             w.commit("Added a new datastream");
574 
575             return ds.DatastreamID;
576         } finally {
577             // Logger completion
578             if (logger.isInfoEnabled()) {
579                 StringBuilder logMsg =
580                         new StringBuilder("Completed addDatastream(");
581                 logMsg.append("pid: ").append(pid);
582                 logMsg.append(", dsID: ").append(dsID);
583                 appendAltIDs(logMsg, altIDs);
584                 logMsg.append(", dsLabel: ").append(dsLabel);
585                 logMsg.append(", versionable: ").append(versionable);
586                 logMsg.append(", MIMEType: ").append(MIMEType);
587                 logMsg.append(", formatURI: ").append(formatURI);
588                 logMsg.append(", dsLocation: ").append(dsLocation);
589                 logMsg.append(", controlGroup: ").append(controlGroup);
590                 logMsg.append(", dsState: ").append(dsState);
591                 logMsg.append(", checksumType: ").append(checksumType);
592                 logMsg.append(", checksum: ").append(checksum);
593                 logMsg.append(", logMessage: ").append(logMessage);
594                 logMsg.append(")");
595                 logger.info(logMsg.toString());
596             }
597 
598             finishModification(w, "addDatastream");
599         }
600     }

602     @Override
603     public Date modifyDatastreamByReference(Context context,
604                                             String pid,
605                                             String datastreamId,
606                                             String[] altIDs,
607                                             String dsLabel,
608                                             String mimeType,
609                                             String formatURI,
610                                             String dsLocation,
611                                             String checksumType,
612                                             String checksum,
613                                             String logMessage,
614                                             Date lastModifiedDate)
615             throws ServerException {
616 
617         // check for valid xml name for datastream ID
618         if (datastreamId != null) {
619             if (!XMLUtils.isWellFormedXMLName(datastreamId)) {
620                 throw new InvalidXMLNameException("Invalid syntax for "
621                                                   + "datastream ID. The datastream ID of \""
622                                                   + datastreamId + "\" is not a valid XML Name");
623             }
624         }
625 
626         if (datastreamId.equals("AUDIT")
627             || datastreamId.equals("FEDORA-AUDITTRAIL")) {
628             throw new GeneralException("Modification of the system-controlled AUDIT"
629                                        + " datastream is not permitted.");
630         }
631 
632         DOWriter w = null;
633         try {
634             logger.debug("Entered modifyDatastreamByReference");
635             m_authz
636                     .enforceModifyDatastreamByReference(context,
637                                                         pid,
638                                                         datastreamId,
639                                                         altIDs,
640                                                         mimeType,
641                                                         formatURI,
642                                                         dsLocation,
643                                                         checksumType,
644                                                         checksum);
645 
646             checkDatastreamLabel(dsLabel);
647             w = m_manager.getWriter(Server.USE_DEFINITIVE_STORE, context, pid);
648             org.fcrepo.server.storage.types.Datastream orig =
649                     w.GetDatastream(datastreamId, null);
650             if (orig == null) {
651                 throw new DatastreamNotFoundException("Object " + pid + " has no datastream "
652                                                       + datastreamId + " to modify");
653             }
654             // if provided, check request lastModifiedDate against the datastream,
655             // rejecting the request if the datastream's mod date is more recent.
656             if (lastModifiedDate != null) {
657                 if (lastModifiedDate.before(orig.DSCreateDT)) {
658                     String dsDate = DateUtility.convertDateToXSDString(w.getLastModDate());
659                     String reqDate = DateUtility.convertDateToXSDString(lastModifiedDate);
660                     String msg = String.format("%s/%s lastModifiedDate (%s) " +
661                                                "is more recent than the " +
662                                                "request (%s)", pid,
663                                                datastreamId, dsDate, reqDate);
664                     throw new DatastreamLockedException(msg);
665                 }
666             }
667 
668             Date nowUTC; // variable for ds modified date
669 
670             // some forbidden scenarios...
671             if (orig.DSControlGrp.equals("X")) {
672                 throw new GeneralException("Inline XML datastreams must be modified by value, not by reference.");
673             }
674             if (orig.DSState.equals("D")) {
675                 throw new GeneralException("Changing attributes on deleted datastreams is forbidden.");
676             }
677

1888     /**
1889      * Deletes expired uploaded files.
1890      * <p/>
1891      * This method is called for each upload. But we respect a minimim delay
1892      * between two purges. This delay is given by m_purgeDelayInMillis.
1893      */
1894     private void purgeUploadedFiles() {
1895         long currentTimeMillis = System.currentTimeMillis();
1896 
1897         // Do purge if purge delay is past before last purge
1898         // -------------------------------------------------
1899         long nextPurgeInMillis =
1900                 this.m_lastPurgeInMillis + this.m_purgeDelayInMillis;
1901         if (nextPurgeInMillis < currentTimeMillis) {
1902             this.m_lastPurgeInMillis = currentTimeMillis;
1903 
1904             // Compute limit file time to purged
1905             // ---------------------------------
1906             long minStartTime =
1907                     currentTimeMillis - (this.m_uploadStorageMinutes * 60000);
1908 
1909             // List files to purge and remove filename to map
1910             // This operation is synchronized to be thread-safe
1911             // ------------------------------------------------
1912             List<String> removeList = new ArrayList<String>();
1913             synchronized (this.m_uploadStartTime) {
1914                 for (Entry<String, Long> entry :
1915                         m_uploadStartTime.entrySet()) {
1916                     String filename = entry.getKey();
1917                     long startTime = entry.getValue().longValue();
1918                     if (startTime < minStartTime) {
1919                         removeList.add(filename);
1920                     }
1921                 }
1922                 for (String filename : removeList) {
1923                     this.m_uploadStartTime.remove(filename);
1924                 }
1925             }
1926 
1927             // Delete file to purged
1928             // This operation is out of synchronised block for performances
1929             // ------------------------------------------------------------
1930             for (int i = 0; i < removeList.size(); i++) {
1931                 String id = removeList.get(i);
1932 
1933                 File file = new File(this.m_tempDir, id);
1934                 if (file.exists()) {
1935                     if (file.delete()) {
1936                         logger.info("Removed uploaded file '" + id
1937                                     + "' because it expired.");
1938                     } else {
1939                         logger.warn("Could not remove expired uploaded file '"
1940                                     + id + "'. Check permissions in " + m_tempDir.getPath() + " directory.");
1941                     }
1942                 }
1943             }
