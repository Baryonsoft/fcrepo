43      /**
44       * {@inheritDoc}
45       */
46      public Set<Triple> getTriplesForObject(DOReader reader)
47              throws ResourceIndexException {
48  
49          Set<Triple> set = new HashSet<Triple>();
50  
51          addCommonTriples(reader, set);
52  
53          return set;
54      }

56      /**
57       * Add the common core and datastream triples for the given object.
58       */
59      private URIReference addCommonTriples(DOReader reader, Set<Triple> set)
60              throws ResourceIndexException {
61  
62          try {
63  
64              URIReference objURI = new SimpleURIReference(
65                      new URI(PID.toURI(reader.GetObjectPID())));
66  
67              addCoreObjectTriples(reader, objURI, set);
68  
69              Datastream[] datastreams = reader.GetDatastreams(null, null);
70              for (Datastream ds : datastreams) {
71                  addCoreDatastreamTriples(ds, objURI, set);
72                  if (ds.DatastreamID.equals("DC")) {
73                      addDCTriples(ds, objURI, set);
74                  }
75              }
76  
77              addRelationshipTriples(reader, objURI, set);
78  
79              return objURI;
80          } catch (ResourceIndexException e) {
81              throw e;
82          } catch (Exception e) {
83              throw new ResourceIndexException("Error generating triples", e);
84          }
85      }

87      /**
88       * For the given object, add the common core system metadata triples. This
89       * will include:
90       * <ul>
91       * <li> object <i>model:createdDate</i></li>
92       * <li> object <i>model:label</i></li>
93       * <li> object <i>model:owner</i></li>
94       * <li> object <i>model:state</i></li>
95       * <li> object <i>view:lastModifiedDate</i></li>
96       * </ul>
97       */
98      private void addCoreObjectTriples(DOReader r,
99                                        URIReference objURI,
100                                       Set<Triple> set) throws Exception {
101 
102         add(objURI, MODEL.CREATED_DATE, r.getCreateDate(), set);
103         add(objURI, MODEL.LABEL, r.GetObjectLabel(), set);
104         add(objURI, MODEL.OWNER, r.getOwnerId(), set);
105         add(objURI, MODEL.STATE, getStateResource(r.GetObjectState()), set);
106         add(objURI, VIEW.LAST_MODIFIED_DATE, r.getLastModDate(), set);
107     }
108 
109     /**
110      * For the given datastream, add the triples that are common for all
111      * datastreams. This will include:
112      * <ul>
113      * <li> object <i>view:disseminates</i> datastream</li>
114      * <li> datastream <i>view:disseminationType</i></li>
115      * <li> datastream <i>view:isVolatile</i></li>
116      * <li> datastream <i>view:lastModifiedDate</i></li>
117      * <li> datastream <i>view:mimeType</i></li>
118      * <li> datastream <i>model:state</i></li>
119      * </ul>
120      */
121     private void addCoreDatastreamTriples(Datastream ds,
122                                           URIReference objURI,
123                                           Set<Triple> set) throws Exception {
124 
125         URIReference dsURI = new SimpleURIReference(
126                 new URI(objURI.getURI().toString() + "/" + ds.DatastreamID));
127 
128         add(objURI, VIEW.DISSEMINATES, dsURI, set);
129 
130         URIReference dsDissType = new SimpleURIReference(
131                 new URI(FEDORA.uri + "*/" + ds.DatastreamID));
132 
133         add(dsURI, VIEW.DISSEMINATION_TYPE, dsDissType, set);
134 
135         boolean isVolatile =
136                 ds.DSControlGrp.equals("E") || ds.DSControlGrp.equals("R");
137 
138         add(dsURI, VIEW.IS_VOLATILE, isVolatile, set);
139         add(dsURI, VIEW.LAST_MODIFIED_DATE, ds.DSCreateDT, set);
140         add(dsURI, VIEW.MIME_TYPE, ds.DSMIME, set);
141         add(dsURI, MODEL.STATE, getStateResource(ds.DSState), set);
142 
143     }

145     /**
146      * Add a statement about the object for each predicate, value pair expressed
147      * in the DC datastream.
148      */
149     private void addDCTriples(Datastream ds,
150                               URIReference objURI,
151                               Set<Triple> set) throws Exception {
152         DCFields dc = new DCFields(ds.getContentStream());
153         Map<RDFName, List<DCField>> map = dc.getMap();
154         for (RDFName predicate : map.keySet()) {
155             for (DCField dcField : map.get(predicate)) {
156                 String lang = dcField.getLang();
157                 if (lang == null) {
158                     add(objURI, predicate, dcField.getValue(), set);
159                 } else {
160                     add(objURI, predicate, dcField.getValue(), lang, set);
161                 }
162             }
163         }
164     }
165 
166     /**
167      * Adds all triples given by reader.getRelationships(null, null).
168      * <p>
169      * This includes everything in RELS-EXT and RELS-INT as well as the implicit
170      * basic content model assertion, if any.
171      */
172     private void addRelationshipTriples(DOReader reader,
173                                         URIReference objURI,
174                                         Set<Triple> set)
175             throws Exception {
176         for (RelationshipTuple tuple : reader.getRelationships()) {
177             ObjectNode oNode;
178             if (tuple.isLiteral) {
179                 if (tuple.datatype != null) {
180                     oNode = new SimpleLiteral(tuple.object,
181                                               tuple.datatype);
182                 } else if (tuple.language != null){
183                     oNode = new SimpleLiteral(tuple.object, tuple.language);
184                 } else {
185                     oNode = new SimpleLiteral(tuple.object);
186                 }
187             } else {
188                 oNode = new SimpleURIReference(new URI(tuple.object));
189             }
190             set.add(new SimpleTriple(new SimpleURIReference(
191                                              new URI(tuple.subject)),
192                                      new SimpleURIReference(
193                                              new URI(tuple.predicate)),
194                                      oNode));
195         }
196     }
