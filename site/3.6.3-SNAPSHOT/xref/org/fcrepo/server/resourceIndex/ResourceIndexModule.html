72      /**
73       * Perform post-initialization of this module. ResourceIndexModule takes the
74       * following parameters:
75       * <ul>
76       * <li> level (required, integer between 0 and 1)<br/> The level of
77       * indexing that should be performed. Values correspond to
78       * <code>INDEX_LEVEL_OFF</code>, and <code>INDEX_LEVEL_ON</code>.
79       * </li>
80       * <li> datastore (required)<br/> The name of the datastore element that
81       * contains the Trippi Connector configuration.
82       * <li> connectionPool (optional, default is ConnectionPoolManager's
83       * default)<br/> Which connection pool to use for updating the database of
84       * method information. </li>
85       * <li> syncUpdates (optional, default is false)<br/> Whether to flush the
86       * triple buffer before returning from object modification operations.
87       * Specifying this as true will ensure that RI queries always reflect the
88       * latest triples. </li>
89       * <li> alias:xyz (optional, uri)<br/> Any parameter starting with "alias:"
90       * will be put into Trippi's alias map, and can be used for queries. For
91       * example, alias:xyz with a value of urn:example:long:uri:x:y:z: will make
92       * it possible to use "xyz:a" to mean "urn:example:long:uri:x:y:z:a" in
93       * queries. </li>
94       * </ul>
95       */
96      @Override
97      public void initModule() throws ModuleInitializationException {
98          int level = getRequiredInt("level", 0, 1);
99          if (level == 0) {
100             logger.info("Disabled in configuration ( level={} ), exiting", level);
101             return;
102         }
103         boolean syncUpdates = getBoolean("syncUpdates", false);
104         try {
105             TriplestoreConnector connector = null;
106             String ds = getParameter("datastore");
107             if (ds != null){
108                 logger.info("Configuring triplestore connector from fcfg datastore configuration {}", ds);
109                 connector =
110                     getConnector(getServer()
111                                  .getDatastoreConfig(ds));
112 
113             } else {
114                 logger.info("Configuring triplestore from bean " + TriplestoreConnector.class.getName());
115                 connector =
116                     getServer().getBean(TriplestoreConnector.class.getName(),TriplestoreConnector.class);
117             }
118             TripleGenerator generator =
119                 getServer().getBean(TripleGenerator.class.getName(), TripleGenerator.class);
120 
121             _ri = new ResourceIndexImpl(connector,
122                                         generator,
123                                         level,
124                                         syncUpdates);
125             setAliasMap(getAliases());
126 
127         } catch (Exception e) {
128             throw new ModuleInitializationException("Error initializing RI",
129                                                     getRole(),
130                                                     e);
131         }
132     }

134     private TriplestoreConnector getConnector(Parameterized datastore)
135             throws Exception {
136         if (datastore == null) {
137             throw new ModuleInitializationException("Specifed datastore "
138                     + "does not exist in fedora.fcfg", getRole());
139         }
140         Map<String, String> config = datastore.getParameters();
141         // make sure path, if specified and relative, is translated
142         // to an absolute path based on the value of FEDORA_HOME
143         String path = config.get("path");
144         if (path != null) {
145             config.put("path", datastore.getParameter("path", true));
146         }
147         String className = config.get("connectorClassName");
148         if (className == null) {
149             throw new ResourceIndexException("Required datastore parameter "
150                     + "is missing: connectorClassName");
151         }
152         getServer().getStatusFile().append(ServerState.STARTING,
153                                            "Initializing Triplestore");
154         return TriplestoreConnector.init(className, config);
155     }

157     private Map<String, String> getAliases() {
158         HashMap<String, String> map = new HashMap<String, String>();
159         Iterator<String> iter = parameterNames();
160         while (iter.hasNext()) {
161             String pName = iter.next();
162             String[] parts = pName.split(":");
163             if (parts.length == 2 && parts[0].equals("alias")) {
164                 map.put(parts[1], getParameter(pName));
165             }
166         }
167         map.put("fedora", Constants.FEDORA.uri);
168         map.put("dc", Constants.DC.uri);
169         map.put("fedora-model", Constants.MODEL.uri);
170         map.put("fedora-rels-ext", Constants.RELS_EXT.uri);
171         map.put("fedora-view", Constants.VIEW.uri);
172         map.put("rdf", Constants.RDF.uri);
173         map.put("mulgara", Constants.MULGARA.uri);
174         map.put("xml-schema", Constants.RDF_XSD.uri);
175         return map;
176     }

178     private int getRequiredInt(String name, int min, int max)
179             throws ModuleInitializationException {
180         try {
181             int value = Integer.parseInt(getRequired(name));
182             if (value < min || value > max) {
183                 throw new ModuleInitializationException(name
184                         + " parameter is out of range, expected [" + min + "-"
185                         + max + "]", getRole());
186             }
187             return value;
188         } catch (NumberFormatException e) {
189             throw new ModuleInitializationException(name
190                     + " parameter must be " + "an integer", getRole());
191         }
192     }
193 
194     private boolean getBoolean(String name, boolean defaultValue)
195             throws ModuleInitializationException {
196         String value = getParameter(name);
197         if (value == null) {
198             return defaultValue;
199         }
200         value = value.toLowerCase();
201         if (value.equals("true") || value.equals("yes") || value.equals("on")) {
202             return true;
203         } else if (value.equals("false") || value.equals("no")
204                 || value.equals("off")) {
205             return false;
206         } else {
207             throw new ModuleInitializationException(name + " parameter, if "
208                     + "specified, must be a boolean (true or false)", getRole());
209         }
210     }
211 
212     private String getRequired(String name)
213             throws ModuleInitializationException {
214         String value = getParameter(name);
215         if (value != null) {
216             return value;
217         } else {
218             throw new ModuleInitializationException(name + " parameter "
219                     + "is required", getRole());
220         }
221     }

225      *
226      * @throws ModuleShutdownException
227      *         if any error occurs while closing.

229     @Override
230     public void shutdownModule() throws ModuleShutdownException {
231         if (_ri != null) {
232             try {
233                 _ri.close();
234             } catch (TrippiException e) {
235                 throw new ModuleShutdownException("Error closing RI",
236                                                   getRole(),
237                                                   e);
238             }
239         }
240     }
