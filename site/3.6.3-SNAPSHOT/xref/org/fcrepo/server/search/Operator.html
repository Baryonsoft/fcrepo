5   package org.fcrepo.server.search;
6   
7   import org.fcrepo.server.errors.InvalidOperatorException;
8   
9   /**
10   * The {@link Operator}s that can be used in a {@link FieldSearchQuery}.
11   * 
12   * @author Jim Blake
13   */
14  public enum Operator {
15      EQUALS("=", "eq"), CONTAINS("~", "has"), GREATER_THAN(">", "gt"),
16      GREATER_OR_EQUAL(">=", "ge"), LESS_THAN("<", "lt"), LESS_OR_EQUAL("<=",
17              "le");
18  
19      private final String symbol;
20  
21      private final String abbreviation;
22  
23      private Operator(String symbol, String abbreviation) {
24          this.symbol = symbol;
25          this.abbreviation = abbreviation;
26      }
27  
28      public String getSymbol() {
29          return symbol;
30      }
31  
32      public String getAbbreviation() {
33          return abbreviation;
34      }
35  
36      public static Operator fromAbbreviation(String abbreviation)
37              throws InvalidOperatorException {
38          for (Operator operator : Operator.values()) {
39              if (operator.abbreviation.equals(abbreviation)) {
40                  return operator;
41              }
42          }
43          throw new InvalidOperatorException("Operator, '" + abbreviation
44                  + "' does not match one of eq, has, gt, ge, lt, or le.");
45  
46      }
47  
48      @Override
49      public String toString() {
50          return name() + "[" + symbol + ", " + abbreviation + "]";
51      }
52  
53  }
