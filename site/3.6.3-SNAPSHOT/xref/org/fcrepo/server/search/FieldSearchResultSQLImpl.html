91      /**
92       * Construct a FieldSearchResultSQLImpl object.
93       * <p />
94       * Upon construction, a connection is obtained from the connectionPool, and
95       * the query is executed. (The connection will be returned to the pool only
96       * after the last result has been obtained from the ResultSet, the session
97       * is expired, or some non-recoverable error has occurred)
98       * <p />
99       * Once the ResultSet is obtained, one result is requested of it (and
100      * remembered for use in step()); then the call returns.
101      *
102      * @param cPool
103      *        the connectionPool
104      * @param repoReader
105      *        the provider of object field information for results
106      * @param resultFields
107      *        which fields should be returned in results
108      * @param maxResults
109      *        how many results should be returned at one time. This should be
110      *        the smaller of a) the FieldSearchImpl's limit [the server limit]
111      *        and b) the requested limit [the client limit]
112      * @param query
113      *        the end-user query
114      */
115     protected FieldSearchResultSQLImpl(ConnectionPool cPool,
116                                        RepositoryReader repoReader,
117                                        String[] resultFields,
118                                        int maxResults,
119                                        int maxSeconds,
120                                        FieldSearchQuery query)
121             throws SQLException, QueryParseException {
122         m_cPool = cPool;
123         m_repoReader = repoReader;
124         m_resultFields = resultFields;
125         m_maxResults = maxResults;
126         m_maxSeconds = maxSeconds;
127         m_conn = m_cPool.getReadOnlyConnection();
128         boolean success = false;
129         try {
130             m_statement = m_conn.prepareStatement(logAndGetQueryText(query,
131                     m_resultFields));
132             m_resultSet =
133                     m_statement.executeQuery(); //2004.05.02 wdn5e
134             success = true;
135         } finally {
136             if (!success) {
137                 try {
138                     if (m_resultSet != null) {
139                         m_resultSet.close();
140                     }
141                     if (m_statement != null) {
142                         m_statement.close();
143                     }
144                     if (m_conn != null) {
145                         m_cPool.free(m_conn);
146                     }
147                 } catch (SQLException e) {
148                     logger.warn("SQL error during failed query cleanup", e);
149                 } finally {
150                     m_resultSet = null;
151                     m_statement = null;
152                 }
153             }
154         }
155     }

158     private String logAndGetQueryText(FieldSearchQuery query,
159                                       String[] resultFields) //2004.05.02 wdn5e

172                     delimiter = ", ";
173                 }
174             }
175             queryText.append(resultFieldsString);
176             queryText.append(" FROM doFields");
177             queryText.append(getWhereClause(query.getConditions()));
178             // disabled sorting: see bug 78

180             //            queryText.append(resultFieldsString);

182         }
183         String qt = queryText.toString();
184         logger.debug(qt);
185         return qt;

188     private String getWhereClause(String terms) throws QueryParseException {
189         StringBuffer whereClause = new StringBuffer();
190         if (!terms.equals("*") && !terms.equals("")) {
191             whereClause.append(" WHERE");
192             // formulate the where clause if the terms aren't * or ""
193             int usedCount = 0;
194             boolean needsEscape = false;
195             for (String column : FieldSearchSQLImpl.DB_COLUMN_NAMES) {
196                 // use only stringish columns in query
197                 boolean use = column.indexOf("Date") == -1;
198                 if (!use) {
199                     if (column.equals("dcDate")) {
200                         use = true;
201                     }
202                 }
203                 if (use) {
204                     if (usedCount > 0) {
205                         whereClause.append(" OR");
206                     }
207                     String qPart = toSql(column, terms);
208                     if (qPart.charAt(0) == ' ') {
209                         needsEscape = true;
210                     } else {
211                         whereClause.append(" ");
212                     }
213                     whereClause.append(qPart);
214                     usedCount++;
215                 }
216             }
217             if (needsEscape) {
218                 //    whereClause.append(" {escape '/'}");
219             }
220         }
221         return whereClause.toString();
222     }

224     private String getWhereClause(List conditions) throws QueryParseException {
225         StringBuffer whereClause = new StringBuffer();
226         boolean willJoin = false;
227         if (conditions.size() > 0) {
228             boolean needsEscape = false;
229             whereClause.append(" WHERE");
230             for (int i = 0; i < conditions.size(); i++) {
231                 Condition cond = (Condition) conditions.get(i);
232                 if (i > 0) {
233                     whereClause.append(" AND");
234                 }
235                 String op = cond.getOperator().getSymbol();
236                 String prop = cond.getProperty();
237                 if (prop.toLowerCase().endsWith("date")) {
238                     // deal with dates ... cDate mDate dcmDate date
239                     if (op.equals("~")) {
240                         if (prop.equals("date")) {
241                             // query for dcDate as string
242                             String sqlPart =
243                                     toSql("doFields.dcDate", cond.getValue());
244                             if (sqlPart.startsWith(" ")) {
245                                 needsEscape = true;
246                             } else {
247                                 whereClause.append(' ');
248                             }
249                             whereClause.append(sqlPart);
250                         } else {
251                             throw new QueryParseException("The ~ operator "
252                                     + "cannot be used with cDate, mDate, "
253                                     + "or dcmDate because they are not "
254                                     + "string-valued fields.");
255                         }
256                     } else { // =, <, <=, >, >=
257                         // property must be parsable as a date... if ok,
258                         // do (cDate, mDate, dcmDate)
259                         // or (date) <- dcDate from dcDates table
260                         Date dt;
261                         try {
262                             dt = DateUtility.parseDateStrict(cond.getValue());
263                         } catch (ParseException e) {
264                             throw new QueryParseException("When using "
265                                     + "equality or inequality operators "
266                                     + "with a date-based value, the date "
267                                     + "must be in yyyy-MM-DD[THH:mm:ss[.SSS][Z]] "
268                                     + "form.");
269                         }
270                         if (prop.equals("date")) {
271                             // do a left join on the dcDates table...dcDate
272                             // query will be of form:
273                             // select pid
274                             // from doFields
275                             // left join dcDates on doFields.pid=dcDates.pid
276                             // where...
277                             if (!willJoin) {
278                                 willJoin = true;
279                                 whereClause.insert(0, " LEFT JOIN dcDates "
280                                         + "ON doFields.pid=dcDates.pid");
281                             }
282                             whereClause.append(" dcDates.dcDate" + op
283                                     + dt.getTime());
284                         } else {
285                             whereClause.append(" doFields." + prop + op
286                                     + dt.getTime());
287                         }
288                     }
289                 } else {
290                     if (op.equals("=")) {
291                         if (isDCProp(prop)) {
292                             throw new QueryParseException("The = operator "
293                                     + "can only be used with dates and "
294                                     + "non-repeating fields.");
295                         } else {
296                             // do a real equals check... do a toSql but
297                             // reject it if it uses "LIKE"
298                             String sqlPart =
299                                     toSql("doFields." + prop, cond.getValue());
300                             if (sqlPart.indexOf("LIKE ") != -1) {
301                                 throw new QueryParseException("The = "
302                                         + "operator cannot be used with "
303                                         + "wildcards.");
304                             }
305                             if (sqlPart.startsWith(" ")) {
306                                 needsEscape = true;
307                             } else {
308                                 whereClause.append(' ');
309                             }
310                             whereClause.append(sqlPart);
311                         }
312                     } else if (op.equals("~")) {
313                         if (isDCProp(prop)) {
314                             // prepend dc and caps the first char first...
315                             prop =
316                                     "dc" + prop.substring(0, 1).toUpperCase()
317                                             + prop.substring(1);
318                         }
319                         // the field name is ok, so toSql it
320                         String sqlPart =
321                                 toSql("doFields." + prop, cond.getValue());
322                         if (sqlPart.startsWith(" ")) {
323                             needsEscape = true;
324                         } else {
325                             whereClause.append(' ');
326                         }
327                         whereClause.append(sqlPart);
328                     } else {
329                         throw new QueryParseException("Can't use >, >=, <, "
330                                 + "or <= operator on a string-based field.");
331                     }
332                 }
333             }
334             if (needsEscape) {
335                 //    whereClause.append(" {escape '/'}");
336             }
337         }
338         return whereClause.toString();
339     }

341     protected boolean isExpired() {
342         long passedSeconds =
343                 (System.currentTimeMillis() - m_startMillis) / 1000;
344         m_expired = passedSeconds > m_maxSeconds;
345         logger.debug("has fieldSearchResultSQL expired? "+m_expired +
346                   ", passed: "+passedSeconds);
347         if (m_expired) {
348             // clean up
349             try {
350                 if (m_resultSet != null) {
351                     m_resultSet.close();
352                 }
353                 if (m_statement != null) {
354                     m_statement.close();
355                 }
356                 if (m_conn != null) {
357                     m_cPool.free(m_conn);
358                 }
359             } catch (SQLException sqle) {
360             } finally {
361                 m_resultSet = null;
362                 m_statement = null;
363             }
364         }
365         return m_expired;
366     }

368     /**
369      * Update object with the next chunk of results. if getToken() is null after
370      * this call, the resultSet was exhausted.
371      */
372     protected void step() throws UnrecognizedFieldException,
373             ObjectIntegrityException, RepositoryConfigurationException,
374             StreamIOException, ServerException {
375         m_objectFields = new ArrayList();
376         int resultCount = 0;
377         // Run through resultSet, adding each result to m_objectFields
378         // for up to maxResults objects, or until the result set is
379         // empty, whichever comes first.
380         // Note: If this is the first chunk of results for the entire search,
381         // m_nextPID will be null, and will require the cursor to be advanced
382         // to the first result.  For all remaining chunks, the cursor will
383         // already be in the correct position (and m_nextPID will have been set)
384         // so a call to m_resultSet.next() is not initially necessary.
385         try {
386             while (resultCount < m_maxResults
387                     && (m_nextPID != null || m_resultSet.next())) {
388                 resultCount++;
389                 // add the current object's info to m_objectFields
390                 String pid;
391                 if (m_nextPID == null) {
392                     pid = m_resultSet.getString("pid");
393                 } else {
394                     pid = m_nextPID;
395                     m_nextPID = null;
396                 }
397                 m_objectFields.add(getObjectFields(pid));
398             }
399             // done with this block. now, are there more results?
400             if (resultCount == m_maxResults && m_resultSet.next()) {
401                 // yes, and we've now advanced the cursor so we must remember
402                 // the pid so the next chunk can use it
403                 m_nextPID = m_resultSet.getString("pid");
404                 // generate a token, make sure the cursor is set,
405                 // and make sure the expirationDate is set
406                 long now = System.currentTimeMillis();
407                 m_token = MD5Utility.getBase16Hash(hashCode() + "" + now);
408                 m_cursor = m_nextCursor;
409                 // keep m_nextCursor updated for next block
410                 m_nextCursor += resultCount;
411                 m_startMillis = now;
412                 Date dt = new Date();
413                 dt.setTime(m_startMillis + 1000 * m_maxSeconds);
414                 m_expirationDate = dt;
415             } else {
416                 // no, so make sure the token is null and clean up
417                 m_token = null;
418                 try {
419                     if (m_resultSet != null) {
420                         m_resultSet.close();
421                     }
422                     if (m_statement != null) {
423                         m_statement.close();
424                     }
425                     if (m_conn != null) {
426                         m_cPool.free(m_conn);
427                     }
428                 } catch (SQLException sqle2) {
429                     throw new StorageDeviceException("Error closing statement "
430                             + "or result set." + sqle2.getMessage());
431                 } finally {
432                     m_resultSet = null;
433                     m_statement = null;
434                 }
435             }
436         } catch (SQLException sqle) {
437             try {
438                 if (m_resultSet != null) {
439                     m_resultSet.close();
440                 }
441                 if (m_statement != null) {
442                     m_statement.close();
443                 }
444                 if (m_conn != null) {
445                     m_cPool.free(m_conn);
446                 }
447                 throw new StorageDeviceException("Error with sql database. "
448                         + sqle.getMessage());
449             } catch (SQLException sqle2) {
450                 throw new StorageDeviceException("Error closing statement "
451                         + "or result set." + sqle.getMessage()
452                         + sqle2.getMessage());
453             } finally {
454                 m_resultSet = null;
455                 m_statement = null;
456             }
457         }
458     }

460     /**
461      * For the given pid, get a reader on the object from the repository and
462      * return an ObjectFields object with resultFields fields populated.
463      *
464      * @param pid
465      *        the unique identifier of the object for which the information is
466      *        requested.
467      * @return ObjectFields populated with the requested fields
468      * @throws UnrecognizedFieldException
469      *         if a resultFields value isn't valid
470      * @throws ObjectIntegrityException
471      *         if the underlying digital object can't be parsed
472      * @throws RepositoryConfigurationException
473      *         if the sax parser can't be constructed
474      * @throws StreamIOException
475      *         if an error occurs while reading the serialized digital object
476      *         stream
477      * @throws ServerException
478      *         if any other kind of error occurs while reading the underlying
479      *         object
480      */
481     private ObjectFields getObjectFields(String pid)
482             throws UnrecognizedFieldException, ObjectIntegrityException,
483             RepositoryConfigurationException, StreamIOException,
484             ServerException {
485         DOReader r =
486                 m_repoReader.getReader(Server.USE_DEFINITIVE_STORE,
487                                        ReadOnlyContext.EMPTY,
488                                        pid);
489         ObjectFields f;
490         // If there's a DC record available, use SAX to parse the most
491         // recent version of it into f.
492         Datastream dcmd = null;
493         try {
494             dcmd = r.GetDatastream("DC", null);
495         } catch (ClassCastException cce) {
496             throw new ObjectIntegrityException("Object " + r.GetObjectPID()
497                     + " has a DC datastream, but it's not inline XML.");
498         }
499         if (dcmd != null) {
500             f = new ObjectFields(m_resultFields, dcmd.getContentStream());
501             // add dcmDate if wanted
502             for (String element : m_resultFields) {
503                 if (element.equals("dcmDate")) {
504                     f.setDCMDate(dcmd.DSCreateDT);
505                 }
506             }
507         } else {
508             f = new ObjectFields();
509         }
510         // add non-dc values from doReader for the others in m_resultFields[]
511         //        Disseminator[] disses=null;
512         for (String n : m_resultFields) {
513             if (n.equals("pid")) {
514                 f.setPid(pid);
515             }
516             if (n.equals("label")) {
517                 f.setLabel(r.GetObjectLabel());
518             }
519             if (n.equals("state")) {
520                 f.setState(r.GetObjectState());
521             }
522             if (n.equals("ownerId")) {
523                 f.setOwnerId(r.getOwnerId());
524             }
525             if (n.equals("cDate")) {
526                 f.setCDate(r.getCreateDate());
527             }
528             if (n.equals("mDate")) {
529                 f.setMDate(r.getLastModDate());
530             }
531         }
532         return f;
533     }

555     /**
556      * Return a condition suitable for a SQL WHERE clause, given a column name
557      * and a string with a possible pattern (using * and questionmark
558      * wildcards).
559      * <p>
560      * </p>
561      * If the string has any characters that need to be escaped, it will begin
562      * with a space, indicating to the caller that the entire WHERE clause
563      * should end with " {escape '/'}".
564      *
565      * @param name
566      *        the name of the field in the database
567      * @param in
568      *        the query string, where * and ? are treated as wildcards
569      * @return String a suitable string for use in a SQL WHERE clause, as
570      *         described above
571      */
572     private static String toSql(String name, String in) {
573         if (!name.endsWith("pid")) {
574             in = in.toLowerCase(); // if it's not a PID-type field,
575         }
576         // it's case insensitive
577         if (name.startsWith("dc") || name.startsWith("doFields.dc")) {
578             StringBuffer newIn = new StringBuffer();
579             if (!in.startsWith("*")) {
580                 newIn.append("* ");
581             }
582             newIn.append(in);
583             if (!in.endsWith("*")) {
584                 newIn.append(" *");
585             }
586             in = newIn.toString();
587         }
588         if (in.indexOf("\\") != -1) {
589             // has one or more escapes, un-escape and translate
590             StringBuffer out = new StringBuffer();
591             out.append("\'");
592             boolean needLike = false;
593             boolean needEscape = false;
594             boolean lastWasEscape = false;
595             for (int i = 0; i < in.length(); i++) {
596                 char c = in.charAt(i);
597                 if (!lastWasEscape && c == '\\') {
598                     lastWasEscape = true;
599                 } else {
600                     char nextChar = '!';
601                     boolean useNextChar = false;
602                     if (!lastWasEscape) {
603                         if (c == '?') {
604                             out.append('_');
605                             needLike = true;
606                         } else if (c == '*') {
607                             out.append('%');
608                             needLike = true;
609                         } else {
610                             nextChar = c;
611                             useNextChar = true;
612                         }
613                     } else {
614                         nextChar = c;
615                         useNextChar = true;
616                     }
617                     if (useNextChar) {
618                         if (nextChar == '\"') {
619                             out.append("\\\"");
620                             needEscape = true;
621                         } else if (nextChar == '\'') {
622                             out.append("''");
623                             needEscape = true;
624                         } else if (nextChar == '%') {
625                             out.append("\\%");
626                             needEscape = true;
627                         } else if (nextChar == '_') {
628                             out.append("\\_");
629                             needEscape = true;
630                         } else {
631                             out.append(nextChar);
632                         }
633                     }
634                     lastWasEscape = false;
635                 }
636             }
637             out.append("\'");
638             if (needLike) {
639                 out.insert(0, " LIKE ");
640             } else {
641                 // replace any \% and \_ in value string with % or _
642                 String fixedString =
643                         out.toString().replaceAll("\\\\%", "%")
644                                 .replaceAll("\\\\_", "_");
645                 out = new StringBuffer();
646                 out.append(fixedString);
647                 out.insert(0, " = ");
648             }
649             out.insert(0, name);
650             if (needEscape) {
651                 out.insert(0, ' ');
652             }
653             return out.toString();
654         } else {
655             // no escapes, just translate if needed
656             StringBuffer out = new StringBuffer();
657             out.append("\'");
658             boolean needLike = false;
659             boolean needEscape = false;
660             for (int i = 0; i < in.length(); i++) {
661                 char c = in.charAt(i);
662                 if (c == '?') {
663                     out.append('_');
664                     needLike = true;
665                 } else if (c == '*') {
666                     out.append('%');
667                     needLike = true;
668                 } else if (c == '\"') {
669                     out.append("\\\"");
670                     needEscape = true;
671                 } else if (c == '\'') {
672                     out.append("''");
673                     needEscape = true;
674                 } else if (c == '%') {
675                     out.append("\\%");
676                     needEscape = true;
677                 } else if (c == '_') {
678                     out.append("\\_");
679                     needEscape = true;
680                 } else {
681                     out.append(c);
682                 }
683             }
684             out.append("\'");
685             if (needLike) {
686                 out.insert(0, " LIKE ");
687             } else {
688                 // replace any \% and \_ in value string with % or _
689                 String fixedString =
690                         out.toString().replaceAll("\\\\%", "%")
691                                 .replaceAll("\\\\_", "_");
692                 out = new StringBuffer();
693                 out.append(fixedString);
694                 out.insert(0, " = ");
695             }
696             out.insert(0, name);
697             if (needEscape) {
698                 out.insert(0, ' ');
699             }
700             return out.toString();
701         }

704     /**
705      * Tell whether a field name, as given in the search request, is a dublin
706      * core field.
707      *
708      * @param the
709      *        field
710      * @return whether it's a dublin core field
711      */
712     private static final boolean isDCProp(String in) {//2004.05.18 wdn5e wasn't static final
713         if (in.equals("mDate") || in.equals("dcmDate")) {
714             return false;
715         }
