5   package org.fcrepo.server.search;
6   
7   import org.fcrepo.server.errors.ServerException;
8   import org.fcrepo.server.storage.DOReader;
9   
10  
11  /**
12   * Provides a simple field-based search service across all objects in the
13   * repository.
14   * 
15   * <p>Key object metadata and dublin core fields are searchable from via
16   * implementations of this interface.
17   * 
18   * <p>Key fields include:<dir> <i>pid, label, state, ownerId, cDate,
19   * mDate, dcmDate</i></dir>
20   * 
21   * <p>Dublin core fields include:<dir> <i>title, creator, subject, description,
22   * publisher, contributor, date, format, identifier, source, language,
23   * relation, coverage, rights</i></dir>
24   * 
25   * @author Chris Wilper
26   */
27  public interface FieldSearch {
28  
29      /**
30       * Update the search indexes with information from the provided DOReader.
31       * 
32       * @param reader
33       *        the DOReader containing all the field information for the object
34       * @throws ServerException
35       *         if anything went wrong
36       */
37      public void update(DOReader reader) throws ServerException;
38  
39      /**
40       * Remove an object from the search indexes.
41       * 
42       * @param pid
43       *        the unique id of the object whose info should be removed
44       * @return true if object successfully removed; false otherwise.
45       * @throws ServerException
46       *         if anything went wrong
47       */
48      public boolean delete(String pid) throws ServerException;
49  
50      /**
51       * Search across specific fields and return the desired fields.
52       * 
53       * @param resultFields
54       *        the desired fields
55       * @param maxResults
56       *        the maximum number of results the client wants
57       * @param query
58       *        the query
59       * @return FieldSearchResult the results
60       * @throws ServerException
61       *         if anything went wrong
62       */
63      public FieldSearchResult findObjects(String[] resultFields,
64                                           int maxResults,
65                                           FieldSearchQuery query)
66              throws ServerException;
67  
68      /**
69       * Resume an in-progress search across specific fields and return the
70       * desired fields.
71       * 
72       * @param sessionToken
73       *        the token of the session in which the remaining results can be
74       *        found
75       * @return FieldSearchResult the results
76       * @throws ServerException
77       *         if anything went wrong
78       */
79      public FieldSearchResult resumeFindObjects(String sessionToken)
80              throws ServerException;
81  
82  }
