11   * of JDBC connection pools for database access.
12   * 
13   * @author Ross Wayland
14   */
15  public interface ConnectionPoolManager {
16  
17      /**
18       * Gets the specified connection pool.
19       * 
20       * @param poolName
21       *        The name of the specified connection pool.
22       * @return The named connection pool.
23       * @throws ConnectionPoolNotFoundException
24       *         If the specified connection pool cannot be found.
25       */
26      public ConnectionPool getPool(String poolName)
27              throws ConnectionPoolNotFoundException;
28  
29      /**
30       * Gets the default Connection Pool. Overrides
31       * <code>getPool(String poolName)</code> to return the default connection

33       * 
34       * @return The default connection pool.
35       * @throws ConnectionPoolNotFoundException
36       *         If the default connection pool cannot be found.
37       */
38      public ConnectionPool getPool() throws ConnectionPoolNotFoundException;
