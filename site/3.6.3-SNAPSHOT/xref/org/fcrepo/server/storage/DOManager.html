16  /**
17   * A RepositoryReader that provides facilities for creating and modifying 
18   * objects within the repository, as well as a query facility.
19   * 
20   * @author Chris Wilper
21   */
22  public interface DOManager
23          extends RepositoryReader {
24  
25      /**
26       * Relinquishes control of a DOWriter back to the DOManager.
27       * 
28       * <p>When a DOManager provides a DOWriter, it creates a session lock. This
29       * is used to guarantee that there will never be concurrent changes to the 
30       * same object. To release the session lock, a DOWriter user calls this 
31       * method.
32       * 
33       * @param writer
34       *        an instance of a digital object writer.
35       * @throws ServerException
36       *         if an error occurs in obtaining a writer.
37       */
38      public abstract void releaseWriter(DOWriter writer) throws ServerException;
39  
40      /**
41       * Gets a DOWriter for an existing digital object.
42       * 
43       * @param context
44       *        The context of this request.
45       * @param pid
46       *        The PID of the object.
47       * @return A writer, or null if the pid didn't point to an accessible
48       *         object.
49       * @throws ServerException
50       *         If anything went wrong.
51       */
52      public abstract DOWriter getWriter(boolean cachedObjectRequired,
53                                         Context context,
54                                         String pid) throws ServerException;
55  
56      /**
57       * Creates a copy of the digital object given by the InputStream, with
58       * either a new PID or the PID indicated by the InputStream.
59       * 
60       * @param context
61       *        The context of this request.
62       * @param in
63       *        A serialization of the digital object.
64       * @param format
65       *        The format of the serialization.
66       * @param encoding
67       *        The character encoding.
68       * @param pid
69       *        Whether a new PID should be generated or the one indicated by the
70       *        InputStream should be used. If set to "new", a new pid will be generated.
71       * @return a writer.
72       * @throws ServerException
73       *         If anything went wrong.
74       */
75      public abstract DOWriter getIngestWriter(boolean cachedObjectRequired,
76                                               Context context,
77                                               InputStream in,
78                                               String format,
79                                               String encoding,
80                                               String pid)
81              throws ServerException;
82      
83      public boolean objectExists(String pid)
84              throws StorageDeviceException;
85  
86      public FieldSearchResult findObjects(Context context,
87                                           String[] resultFields,
88                                           int maxResults,
89                                           FieldSearchQuery query)
90              throws ServerException;
91  
92      public FieldSearchResult resumeFindObjects(Context context,
93                                                 String sessionToken)
94              throws ServerException;
95  
96      public String[] getNextPID(int numPIDs, String namespace)
97              throws ServerException;
98  
99      public String lookupDeploymentForCModel(String cModelPid, String sDefPid);
100 
101     /**
102      * Reserve a series of PIDs so that they are never used for subsequent PID
103      * generations.
104      */
105     public void reservePIDs(String[] pidList) throws ServerException;
106 
107     /**
108      * Get a "hash" of the repository. This value can be compared to a previous
109      * value to determine whether the content of the repository has changed. It
110      * is not necessary for this value to precisely reflect the state of the
111      * repository, but if the repository hasn't changed, subsequent calls should
112      * return the same value.
113      */
114     public String getRepositoryHash() throws ServerException;
115 
116 }
