150     @Override
151     public void initModule() throws ModuleInitializationException {
152         // pidNamespace (required, 1-17 chars, a-z, A-Z, 0-9 '-' '.')
153         m_pidNamespace = getParameter("pidNamespace");
154         if (m_pidNamespace == null) {
155             throw new ModuleInitializationException(
156                     "pidNamespace parameter must be specified.", getRole());
157         }
158         if (m_pidNamespace.length() > 17 || m_pidNamespace.length() < 1) {
159             throw new ModuleInitializationException(
160                     "pidNamespace parameter must be 1-17 chars long", getRole());
161         }
162         StringBuffer badChars = new StringBuffer();
163         for (int i = 0; i < m_pidNamespace.length(); i++) {
164             char c = m_pidNamespace.charAt(i);
165             boolean invalid = true;
166             if (c >= '0' && c <= '9') {
167                 invalid = false;
168             } else if (c >= 'a' && c <= 'z') {
169                 invalid = false;
170             } else if (c >= 'A' && c <= 'Z') {
171                 invalid = false;
172             } else if (c == '-') {
173                 invalid = false;
174             } else if (c == '.') {
175                 invalid = false;
176             }
177             if (invalid) {
178                 badChars.append(c);
179             }
180         }
181         if (badChars.toString().length() > 0) {
182             throw new ModuleInitializationException("pidNamespace contains " +
183                     "invalid character(s) '" + badChars.toString() + "'",
184                     getRole());
185         }
186         // storagePool (optional, default=ConnectionPoolManager's default pool)
187         m_storagePool = getParameter("storagePool");
188         if (m_storagePool == null) {
189             logger.debug("Parameter storagePool "
190                     + "not given, will defer to ConnectionPoolManager's "
191                     + "default pool.");
192         }
193         // internal storage format (required)
194         logger.debug("Server property format.storage= " + Server.STORAGE_FORMAT);
195         m_defaultStorageFormat = Server.STORAGE_FORMAT;
196         if (m_defaultStorageFormat == null) {
197             throw new ModuleInitializationException(
198                     "System property format.storage "
199                             + "not given, but it's required.", getRole());
200         }
201         // default export format (required)
202         m_defaultExportFormat = getParameter("defaultExportFormat");
203         if (m_defaultExportFormat == null) {
204             throw new ModuleInitializationException(
205                     "Parameter defaultExportFormat "
206                             + "not given, but it's required.", getRole());
207         }
208         // storageCharacterEncoding (optional, default=UTF-8)
209         m_storageCharacterEncoding = getParameter("storageCharacterEncoding");
210         if (m_storageCharacterEncoding == null) {
211             logger.debug("Parameter storage_character_encoding "
212                     + "not given, using UTF-8");
213             m_storageCharacterEncoding = "UTF-8";
214         }
215         initRetainPID();
216 
217         // readerCacheSize and readerCacheSeconds (optional, defaults = 20, 5)
218         String rcSize = getParameter("readerCacheSize");
219         if (rcSize == null) {
220             logger.debug("Parameter readerCacheSize not given, using 20");
221             rcSize = "20";
222         }
223         int readerCacheSize;
224         try {
225             readerCacheSize = Integer.parseInt(rcSize);
226             if (readerCacheSize < 0) {
227                 throw new Exception("Cannot be less than zero");
228             }
229         } catch (Exception e) {
230             throw new ModuleInitializationException(
231                     "Bad value for readerCacheSize parameter: " +
232                             e.getMessage(), getRole());
233         }
234 
235         String rcSeconds = getParameter("readerCacheSeconds");
236         if (rcSeconds == null) {
237             logger.debug("Parameter readerCacheSeconds not given, using 5");
238             rcSeconds = "5";
239         }
240         int readerCacheSeconds;
241         try {
242             readerCacheSeconds = Integer.parseInt(rcSeconds);
243             if (readerCacheSeconds < 1) {
244                 throw new Exception("Cannot be less than one");
245             }
246         } catch (Exception e) {
247             throw new ModuleInitializationException(
248                     "Bad value for readerCacheSeconds parameter: " +
249                             e.getMessage(), getRole());
250         }
251 
252         if (readerCacheSize > 0) {
253             m_readerCache =
254                     new DOReaderCache(readerCacheSize, readerCacheSeconds);
255         }
256 
257         // configuration of ingest validation
258         String ingestValidationLevel = getParameter("ingestValidationLevel");
259         if (ingestValidationLevel == null) {
260             logger.debug("Ingest validation level not specified, using default of all");
261             m_ingestValidationLevel = DOValidator.VALIDATE_ALL;
262         } else {
263             m_ingestValidationLevel = Integer.parseInt(ingestValidationLevel);
264             // check the values.  better to declare the levels as enums, but this
265             // would require DOValidator interface change
266             if (m_ingestValidationLevel < -1 || m_ingestValidationLevel > 2) {
267                 throw new ModuleInitializationException(
268                         "Bad value for ingestValidationLevel", getRole());
269             }
270         }
271     }

273     protected void initRetainPID() {
274         m_retainPIDs = new HashSet<String>();
275         String retainPIDs = getParameter("retainPIDs");
276         if (retainPIDs == null || retainPIDs.equals("*")) {
277             // when m_retainPIDS is set to null, that means "all"
278             m_retainPIDs = null;
279         } else {
280             // add to list (accept space and/or comma-separated)
281             String[] ns =
282                     retainPIDs.trim().replaceAll(" +", ",").replaceAll(",+",
283                             ",").split(",");
284             for (String element : ns) {
285                 if (element.length() > 0) {
286                     m_retainPIDs.add(element);
287                 }
288             }
289 
290             // fedora-system PIDs must be ingestable as-is
291             m_retainPIDs.add("fedora-system");
292         }

422     private void initializeCModelDeploymentCache() {
423         // Initialize Map containing links from Content Models to the Service
424         // Deployments.
425         m_cModelDeploymentMap = new ModelDeploymentMap();
426 
427         logger.debug("Initializing content model deployment map");
428 
429         Connection c = null;
430         PreparedStatement s = null;
431         ResultSet r = null;
432         try {
433             c = m_connectionPool.getReadOnlyConnection();
434             String query =
435                     "SELECT cModel, sDef, sDep, mDate "
436                             + " FROM modelDeploymentMap, doFields "
437                             + " WHERE doFields.pid = modelDeploymentMap.sDep";
438             s =
439                     c.prepareStatement(query, ResultSet.TYPE_FORWARD_ONLY,
440                             ResultSet.CONCUR_READ_ONLY);
441             ResultSet results = s.executeQuery();
442 
443             while (results.next()) {
444                 String cModel = results.getString(1);
445                 String sDef = results.getString(2);
446                 String sDep = results.getString(3);
447                 long lastMod = results.getLong(4);
448 
449                 m_cModelDeploymentMap.putDeployment(ServiceContext.getInstance(
450                         cModel, sDef), sDep, lastMod);
451             }
452 
453         } catch (SQLException e) {
454             throw new RuntimeException("Error loading cModel deployment cach",
455                     e);
456         } finally {
457             try {
458                 if (r != null) {
459                     r.close();
460                 }
461                 if (s != null) {
462                     s.close();
463                 }
464                 if (c != null) {
465                     m_connectionPool.free(c);
466                 }
467             } catch (SQLException e) {
468                 throw new RuntimeException(
469                         "Error loading cModel deployment cach", e);
470             }
471         }
472 
473     }

475     /**
476      * Update the registry and deployment cache to reflect the latest state of
477      * reality.
478      *
479      * @param obj
480      *        DOReader of a service deployment object
481      */
482     private synchronized void updateDeploymentMap(DigitalObject obj,
483             Connection c, boolean isPurge) throws SQLException {
484 
485         String sDep = obj.getPid();
486         Set<RelationshipTuple> sDefs =
487                 obj.getRelationships(Constants.MODEL.IS_DEPLOYMENT_OF, null);
488         Set<RelationshipTuple> models =
489                 obj.getRelationships(Constants.MODEL.IS_CONTRACTOR_OF, null);
490 
491         /* Read in the new deployment map from the object */
492         Set<ServiceContext> newContext = new HashSet<ServiceContext>();
493 
494         if (!isPurge) {
495             for (RelationshipTuple sDefTuple : sDefs) {
496                 String sDef = sDefTuple.getObjectPID();
497                 for (RelationshipTuple cModelTuple : models) {
498                     String cModel = cModelTuple.getObjectPID();
499                     newContext.add(ServiceContext.getInstance(cModel, sDef));
500                 }
501             }
502         }
503 
504         /* Read in the old deployment map from the cache */
505         Set<ServiceContext> oldContext =
506                 m_cModelDeploymentMap.getContextFor(sDep);
507 
508         /* Remove any obsolete deployments from the registry/cache */
509         for (ServiceContext o : oldContext) {
510             if (!newContext.contains(o)) {
511                 removeDeployment(o, obj, c);
512             }
513         }
514 
515         /* Add any new deployments from the registry/cache */
516         for (ServiceContext n : newContext) {
517             if (!oldContext.contains(n)) {
518                 addDeployment(n, obj, c);
519             } else {
520                 updateDeployment(n, obj, c);
521             }
522         }
523     }

525     private void addDeployment(ServiceContext context, DigitalObject sDep,
526             Connection c) throws SQLException {
527 
528         String query =
529                 "INSERT INTO modelDeploymentMap (cModel, sDef, sDep) VALUES (?, ?, ?)";
530         PreparedStatement s = c.prepareStatement(query);
531 
532         try {
533             s.setString(1, context.cModel);
534             s.setString(2, context.sDef);
535             s.setString(3, sDep.getPid());
536             s.executeUpdate();
537         } finally {
538             if (s != null) {
539                 s.close();
540             }
541         }
542 
543         m_cModelDeploymentMap.putDeployment(context, sDep.getPid(), sDep
544                 .getLastModDate().getTime());
545 
546     }

584     @Override
585     public void releaseWriter(DOWriter writer) {
586 
587         // If this is a new object, but object was not successfully committed
588         // need to backout object registration.
589         if (writer.isNew() && !writer.isCommitted()) {
590             try {
591                 unregisterObject(writer.getObject());
592             } catch (Exception e) {
593                 try {
594                     logger.warn("Error unregistering object: " +
595                             writer.GetObjectPID(), e);
596                 } catch (Exception e2) {
597                     logger.warn(
598                             "Error unregistering object; Unable to obtain PID from writer.",
599                             e2);
600                 }
601             }
602         }
603 
604         writer.invalidate();
605 
606         try {
607             releaseWriteLock(writer.GetObjectPID());
608         } catch (ServerException e) {
609             logger.warn("Error releasing object lock; Unable to obtain pid from writer.");
610         }
611     }

613     private void releaseWriteLock(String pid) {
614         synchronized (m_lockedPIDs) {
615             m_lockedPIDs.remove(pid);
616         }
617     }

619     private void getWriteLock(String pid) throws ObjectLockedException {
620         synchronized (m_lockedPIDs) {
621             if (m_lockedPIDs.contains(pid)) {
622                 throw new ObjectLockedException(pid + " is currently being " +
623                         "modified by another thread");
624             } else {
625                 m_lockedPIDs.add(pid);
626             }
627         }
628     }

666      * Gets a reader on an an existing digital object.
667      */
668     @Override
669     public DOReader getReader(boolean cachedObjectRequired, Context context,
670             String pid) throws ServerException {
671         long getReaderStartTime = System.currentTimeMillis();
672         String source = null;
673         try {
674             {
675                 DOReader reader = null;
676                 if (m_readerCache != null) {
677                     reader = m_readerCache.get(pid);
678                 }
679                 if (reader == null) {
680                     reader =
681                             new SimpleDOReader(context, this, m_translator,
682                                     m_defaultExportFormat,
683                                     m_defaultStorageFormat,
684                                     m_storageCharacterEncoding,
685                                     m_permanentStore.retrieveObject(pid));
686                     source = "filesystem";
687                     if (m_readerCache != null) {
688                         m_readerCache.put(reader);
689                     }
690                 } else {
691                     source = "memory";
692                 }
693                 return reader;
694             }
695         } finally {
696             if (logger.isDebugEnabled()) {
697                 long dur = System.currentTimeMillis() - getReaderStartTime;
698                 logger.debug("Got DOReader (source=" + source + ") for " + pid +
699                         " in " + dur + "ms.");
700             }
701         }
702     }

758      * Manages the INGEST process which includes validation of the ingest XML
759      * file, deserialization of the XML into a Digital Object instance, setting
760      * of properties on the object by the system (dates and states), PID
761      * validation or generation, object registry functions, getting a writer for
762      * the digital object, and ultimately writing the object to persistent
763      * storage via the writer.
764      *
765      * @param context
766      * @param in
767      *            the input stream that is the XML ingest file for a digital
768      *            object
769      * @param format
770      *        the format of the XML ingest file (e.g., FOXML, Fedora METS)
771      * @param encoding
772      *        the character encoding of the XML ingest file (e.g., UTF-8)
773      * @param pid
774      *            "new" if the system should generate a new PID for the object,
775      *            otherwise the value of the additional pid parameter for
776      *            ingests (may be null or any valid pid)
777      */
778     @Override
779     public synchronized DOWriter getIngestWriter(boolean cachedObjectRequired,
780             Context context, InputStream in, String format, String encoding,
781             String pid) throws ServerException {
782         logger.debug("Entered getIngestWriter");
783 
784         DOWriter w = null;
785         BasicDigitalObject obj = null;
786 
787         File tempFile = null;
788         if (cachedObjectRequired) {
789             throw new InvalidContextException(
790                     "A DOWriter is unavailable in a cached context.");
791         } else {
792             try {
793                 // CURRENT TIME:
794                 // Get the current time to use for created dates on object
795                 // and object components (if they are not already there).
796                 Date nowUTC = Server.getCurrentDate(context);
797 
798                 // TEMP STORAGE:
799                 // write ingest input stream to a temporary file
800                 tempFile = File.createTempFile("fedora-ingest-temp", ".xml");
801                 logger.debug("Creating temporary file for ingest: " +
802                         tempFile.toString());
803                 StreamUtility.pipeStream(in, new FileOutputStream(tempFile),
804                         4096);
805 
806                 // VALIDATION:
807                 // perform initial validation of the ingest submission file
808                 logger.debug("Validation (ingest phase)");
809                 m_validator.validate(tempFile, format, m_ingestValidationLevel,
810                         DOValidator.PHASE_INGEST);
811 
812                 // DESERIALIZE:
813                 // deserialize the ingest input stream into a digital object
814                 // instance
815                 obj = new BasicDigitalObject();
816                 obj.setNew(true);
817                 logger.debug("Deserializing from format: " + format);
818                 m_translator.deserialize(new FileInputStream(tempFile), obj,
819                         format, encoding,
820                         DOTranslationUtility.DESERIALIZE_INSTANCE);
821 
822                 // SET OBJECT PROPERTIES:
823                 logger.debug("Setting object/component states and create dates if unset");
824                 // set object state to "A" (Active) if not already set
825                 if (obj.getState() == null || obj.getState().equals("")) {
826                     obj.setState("A");
827                 }
828                 // set object create date to UTC if not already set
829                 if (obj.getCreateDate() == null ||
830                         obj.getCreateDate().equals("")) {
831                     obj.setCreateDate(nowUTC);
832                 }
833                 // set object last modified date to UTC
834                 obj.setLastModDate(nowUTC);
835 
836                 // SET DATASTREAM PROPERTIES...
837                 Iterator<String> dsIter = obj.datastreamIdIterator();
838                 while (dsIter.hasNext()) {
839                     for (Datastream ds : obj.datastreams(dsIter.next())) {
840                         // Set create date to UTC if not already set
841                         if (ds.DSCreateDT == null || ds.DSCreateDT.equals("")) {
842                             ds.DSCreateDT = nowUTC;
843                         }
844                         // Set state to "A" (Active) if not already set
845                         if (ds.DSState == null || ds.DSState.equals("")) {
846                             ds.DSState = "A";
847                         }
848                         ds.DSChecksumType =
849                                 Datastream
850                                         .validateChecksumType(ds.DSChecksumType);
851                     }
852                 }
853 
854                 // SET MIMETYPE AND FORMAT_URIS FOR LEGACY OBJECTS' DATASTREAMS
855                 if (FOXML1_0.uri.equals(format) ||
856                         FOXML1_0_LEGACY.equals(format) ||
857                         METS_EXT1_0.uri.equals(format) ||
858                         METS_EXT1_0_LEGACY.equals(format)) {
859                     DigitalObjectUtil.updateLegacyDatastreams(obj);
860                 }
861 
862                 // If the PID was supplied as additional parameter (see REST
863                 // API), make sure it doesn't conflict with the (optional) PID
864                 // of the digital object
865                 if (pid != null && pid.length() > 0 && !pid.equals("new")) {
866                     if (obj.getPid() != null && obj.getPid().length() > 0) {
867                         if (!pid.equals(obj.getPid())) {
868                             throw new GeneralException(
869                                     "The PID of the digital object and the PID provided as parameter are different. Digital object: " +
870                                             obj.getPid() + " parameter: " + pid);
871                         }
872                     } else {
873                         obj.setPid(pid);
874                     }
875 
876                 }
877                 // PID VALIDATION:
878                 // validate and normalized the provided pid, if any
879                 if (obj.getPid() != null && obj.getPid().length() > 0) {
880                     obj.setPid(Server.getPID(obj.getPid()).toString());
881                 }
882 
883                 // PID GENERATION:
884                 // have the system generate a PID if one was not provided
885                 if (obj.getPid() != null &&
886                         obj.getPid().indexOf(":") != -1 &&
887                         (m_retainPIDs == null || m_retainPIDs.contains(obj
888                                 .getPid().split(":")[0]))) {
889                     logger.debug("Stream contained PID with retainable namespace-id; will use PID from stream");
890                     try {
891                         m_pidGenerator.neverGeneratePID(obj.getPid());
892                     } catch (IOException e) {
893                         throw new GeneralException(
894                                 "Error calling pidGenerator.neverGeneratePID(): " +
895                                         e.getMessage());
896                     }
897                 } else {
898                     if (pid.equals("new")) {
899                         logger.debug("Client wants a new PID");
900                         // yes... so do that, then set it in the obj.
901                         String p = null;
902                         try {
903                             // If the context contains a recovery PID, use that.
904                             // Otherwise, generate a new PID as usual.
905                             if (context instanceof RecoveryContext) {
906                                 RecoveryContext rContext =
907                                         (RecoveryContext) context;
908                                 p =
909                                         rContext.getRecoveryValue(Constants.RECOVERY.PID.uri);
910                             }
911                             if (p == null) {
912                                 p =
913                                         m_pidGenerator.generatePID(
914                                                 m_pidNamespace).toString();
915                             } else {
916                                 logger.debug("Using new PID from recovery context");
917                                 m_pidGenerator.neverGeneratePID(p);
918                             }
919                         } catch (Exception e) {
920                             throw new GeneralException("Error generating PID",
921                                     e);
922                         }
923                         logger.info("Generated new PID: " + p);
924                         obj.setPid(p);
925                     } else {
926                         logger.debug("Client wants to use existing PID.");
927                     }
928                 }
929 
930                 logger.info("New object PID is {}", obj.getPid());
931 
932                 // CHECK REGISTRY:
933                 // ensure the object doesn't already exist
934                 if (objectExists(obj.getPid())) {
935                     throw new ObjectExistsException("The PID '" + obj.getPid() +
936                             "' already exists in the registry; the object can't be re-created.");
937                 }
938 
939                 // GET DIGITAL OBJECT WRITER:
940                 // get an object writer configured with the DEFAULT export
941                 // format
942                 logger.debug("Getting new writer with default export format: " +
943                         m_defaultExportFormat);
944                 logger.debug("Instantiating a SimpleDOWriter");
945                 w =
946                         new SimpleDOWriter(context, this, m_translator,
947                                 m_defaultExportFormat,
948                                 m_storageCharacterEncoding, obj);
949 
950                 // WRITE LOCK:
951                 // ensure no one else can modify the object now
952                 getWriteLock(obj.getPid());
953 
954                 // DEFAULT DATASTREAMS:
955                 populateDC(context, obj, w, nowUTC);
956 
957                 // DATASTREAM VALIDATION
958                 ValidationUtility.validateReservedDatastreams(w);
959 
960                 // REGISTRY:
961                 // at this point the object is valid, so make a record
962                 // of it in the digital object registry
963                 registerObject(obj);
964                 return w;
965             } catch (IOException e) {
966 
967                 if (w != null) {
968                     releaseWriteLock(obj.getPid());
969                 }
970 
971                 throw new GeneralException("Error reading/writing temporary "
972                         + "ingest file", e);
973             } catch (Exception e) {
974 
975                 if (w != null) {
976                     releaseWriteLock(obj.getPid());
977                 }
978 
979                 if (e instanceof ServerException) {
980                     ServerException se = (ServerException) e;
981                     throw se;
982                 }
983                 throw new GeneralException("Ingest failed: " +
984                         e.getClass().getName(), e);
985             } finally {
986                 if (tempFile != null) {
987                     logger.debug("Finally, removing temp file");
988                     try {
989                         tempFile.delete();
990                     } catch (Exception e) {
991                         // don't worry if it doesn't exist
992                     }
993                 }
994             }
995         }
996     }

1767     // translates simple wildcard string to sql-appropriate.
1768     // the first character is a " " if it needs an escape
1769     public static String toSql(String name, String in) {
1770         if (in.indexOf("\\") != -1) {
1771             // has one or more escapes, un-escape and translate
1772             StringBuffer out = new StringBuffer();
1773             out.append("\'");
1774             boolean needLike = false;
1775             boolean needEscape = false;
1776             boolean lastWasEscape = false;
1777             for (int i = 0; i < in.length(); i++) {
1778                 char c = in.charAt(i);
1779                 if (!lastWasEscape && c == '\\') {
1780                     lastWasEscape = true;
1781                 } else {
1782                     char nextChar = '!';
1783                     boolean useNextChar = false;
1784                     if (!lastWasEscape) {
1785                         if (c == '?') {
1786                             out.append('_');
1787                             needLike = true;
1788                         } else if (c == '*') {
1789                             out.append('%');
1790                             needLike = true;
1791                         } else {
1792                             nextChar = c;
1793                             useNextChar = true;
1794                         }
1795                     } else {
1796                         nextChar = c;
1797                         useNextChar = true;
1798                     }
1799                     if (useNextChar) {
1800                         if (nextChar == '\"') {
1801                             out.append("\\\"");
1802                             needEscape = true;
1803                         } else if (nextChar == '\'') {
1804                             out.append("\\\'");
1805                             needEscape = true;
1806                         } else if (nextChar == '%') {
1807                             out.append("\\%");
1808                             needEscape = true;
1809                         } else if (nextChar == '_') {
1810                             out.append("\\_");
1811                             needEscape = true;
1812                         } else {
1813                             out.append(nextChar);
1814                         }
1815                     }
1816                     lastWasEscape = false;
1817                 }
1818             }
1819             out.append("\'");
1820             if (needLike) {
1821                 out.insert(0, " LIKE ");
1822             } else {
1823                 out.insert(0, " = ");
1824             }
1825             out.insert(0, name);
1826             if (needEscape) {
1827                 out.insert(0, ' ');
1828             }
1829             return out.toString();
1830         } else {
1831             // no escapes, just translate if needed
1832             StringBuffer out = new StringBuffer();
1833             out.append("\'");
1834             boolean needLike = false;
1835             boolean needEscape = false;

1837                 char c = in.charAt(i);

1839                     out.append('_');

1854                     out.append("\\_");

1856                 } else {

1858                 }
1859             }
1860             out.append("\'");
1861             if (needLike) {
1862                 out.insert(0, " LIKE ");
1863             } else {
1864                 out.insert(0, " = ");
1865             }
1866             out.insert(0, name);
