30  /**
31   * A class for parsing the special XML format in Fedora for a Datastream Input
32   * Specification (DSInputSpec). A DSInputSpec exists within a Service Deployment
33   * Object, and is used to define the "data contract" between the service
34   * represented by the deployment and a data object.
35   * 
36   * @author Sandy Payette
37   * @version $Id$
38   */
39  public class DSInputSpecParser
40          extends DefaultHandler
41          implements Constants {

44       * URI-to-namespace prefix mapping info from SAX2 startPrefixMapping events.

52      private boolean inDSInputMIME = false;

71      /**
72       * Constructor allows this class to initiate the parsing
73       */
74      public DSInputSpecParser(String parentPID, InputStream in)
75              throws RepositoryConfigurationException, ObjectIntegrityException {
76          sDepPID = parentPID;
77          XMLReader xmlReader = null;
78          try {
79              SAXParserFactory saxfactory = SAXParserFactory.newInstance();
80              saxfactory.setValidating(false);
81              SAXParser parser = saxfactory.newSAXParser();
82              xmlReader = parser.getXMLReader();
83              xmlReader.setContentHandler(this);
84              xmlReader.setFeature("http://xml.org/sax/features/namespaces",
85                                   false);
86              xmlReader
87                      .setFeature("http://xml.org/sax/features/namespace-prefixes",
88                                  false);
89          } catch (Exception e) {
90              throw new RepositoryConfigurationException("Internal SAX error while "
91                      + "preparing for DSInputSpec datastream deserialization: "
92                      + e.getMessage());
93          }
94          try {
95              xmlReader.parse(new InputSource(in));
96          } catch (Exception e) {
97              throw new ObjectIntegrityException("Error parsing DSInputSpec datastream"
98                      + e.getClass().getName() + ": " + e.getMessage());
99          }
100     }

128     @Override
129     public void skippedEntity(String name) throws SAXException {
130         StringBuffer sb = new StringBuffer();
131         sb.append('&');
132         sb.append(name);
133         sb.append(';');
134         char[] text = new char[sb.length()];
135         sb.getChars(0, sb.length(), text, 0);
136         characters(text, 0, text.length);
137     }

139     @Override
140     public void characters(char ch[], int start, int length)
141             throws SAXException {
142         if (inDSInputLabel) {
143             dsInputRule.bindingLabel = new String(ch, start, length);
144         } else if (inDSInputInstructions) {
145             dsInputRule.bindingInstruction = new String(ch, start, length);
146         } else if (inDSInputMIME) {
147             StringTokenizer st =
148                     new StringTokenizer(new String(ch, start, length), " ");
149             String[] MIMETypes = new String[st.countTokens()];
150             for (int i = 0; i < MIMETypes.length; i++) {
151                 MIMETypes[i] = st.nextToken();
152             }
153             dsInputRule.bindingMIMETypes = MIMETypes;
154         }
155     }

157     @Override
158     public void startElement(String namespaceURI,
159                              String localName,
160                              String qName,
161                              Attributes attrs) throws SAXException {
162         if (namespaceURI.equalsIgnoreCase(BINDING_SPEC.uri)
163                 && localName.equalsIgnoreCase("DSInputSpec")) {
164             dsInputSpec.serviceDeploymentPID = sDepPID;
165             dsInputSpec.bindSpecLabel = attrs.getValue("label");
166         } else if (namespaceURI.equalsIgnoreCase(BINDING_SPEC.uri)
167                 && localName.equalsIgnoreCase("DSInput")) {
168             dsInputRule = new DeploymentDSBindRule();
169             
170             if (attrs.getValue("pid") != null) {
171                 dsInputRule.pid = attrs.getValue("pid");
172             }
173             
174             dsInputRule.bindingKeyName = attrs.getValue("wsdlMsgPartName");
175             dsInputRule.maxNumBindings =
176                     new Integer(attrs.getValue("DSMax")).intValue();
177             dsInputRule.minNumBindings =
178                     new Integer(attrs.getValue("DSMin")).intValue();
179             dsInputRule.ordinality =
180                     Boolean.parseBoolean(attrs.getValue("DSOrdinality"));
181         } else if (namespaceURI.equalsIgnoreCase(BINDING_SPEC.uri)
182                 && localName.equalsIgnoreCase("DSInputLabel")) {
183             inDSInputLabel = true;
184         } else if (namespaceURI.equalsIgnoreCase(BINDING_SPEC.uri)
185                 && localName.equalsIgnoreCase("DSInputInstruction")) {
186             inDSInputInstructions = true;
187         } else if (namespaceURI.equalsIgnoreCase(BINDING_SPEC.uri)
188                 && localName.equalsIgnoreCase("DSMIME")) {
189             inDSInputMIME = true;
190         }
191     }
192 
193     @Override
194     public void endElement(String namespaceURI, String localName, String qName)
195             throws SAXException {
196         if (namespaceURI.equalsIgnoreCase(BINDING_SPEC.uri)
197                 && localName.equalsIgnoreCase("DSInputSpec")) {
198         } else if (namespaceURI.equalsIgnoreCase(BINDING_SPEC.uri)
199                 && localName.equalsIgnoreCase("DSInput")) {
200             tmp_InputRules.add(dsInputRule);
201             dsInputRule = null;
202         } else if (namespaceURI.equalsIgnoreCase(BINDING_SPEC.uri)
203                 && localName.equalsIgnoreCase("DSInputLabel")) {
204             inDSInputLabel = false;
205         } else if (namespaceURI.equalsIgnoreCase(BINDING_SPEC.uri)
206                 && localName.equalsIgnoreCase("DSInputInstruction")) {
207             inDSInputInstructions = false;
208         } else if (namespaceURI.equalsIgnoreCase(BINDING_SPEC.uri)
209                 && localName.equalsIgnoreCase("DSMIME")) {
210             inDSInputMIME = false;
211         }
212     }
