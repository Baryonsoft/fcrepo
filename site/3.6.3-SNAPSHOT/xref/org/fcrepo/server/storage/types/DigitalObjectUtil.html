9   /**
10   * DigitalObject utility methods.

12   * @author Chris Wilper

14  public abstract class DigitalObjectUtil
15          implements Constants {

17      /**
18       * Upgrades a legacy (pre-Fedora-3.0) object by setting the correct MIME
19       * type and Format URI for all "reserved" datastreams.
20       * 
21       * @param obj the object to update.
22       */
23      @SuppressWarnings("deprecation")
24      public static void updateLegacyDatastreams(DigitalObject obj) {
25          final String xml = "text/xml";
26          final String rdf = "application/rdf+xml";
27          updateLegacyDatastream(obj, "DC", xml, OAI_DC2_0.uri);
28          updateLegacyDatastream(obj, "RELS-EXT", rdf, RELS_EXT1_0.uri);
29          updateLegacyDatastream(obj, "RELS-INT", rdf, RELS_INT1_0.uri);
30          updateLegacyDatastream(obj, "POLICY", xml, XACML_POLICY1_0.uri);
31          String fType = obj.getExtProperty(RDF.TYPE.uri);
32          if (MODEL.BDEF_OBJECT.looselyMatches(fType, false)) {
33              updateLegacyDatastream(obj,
34                                     "METHODMAP",
35                                     xml,
36                                     SDEF_METHOD_MAP1_0.uri);
37          } else if (MODEL.BMECH_OBJECT.looselyMatches(fType, false)) {
38              updateLegacyDatastream(obj,
39                                     "METHODMAP",
40                                     xml,
41                                     SDEP_METHOD_MAP1_1.uri);
42              updateLegacyDatastream(obj,
43                                     "DSINPUTSPEC",
44                                     xml,
45                                     DS_INPUT_SPEC1_1.uri);
46              updateLegacyDatastream(obj,
47                                     "WSDL",
48                                     xml,
49                                     WSDL.uri);
50          }
51      }
52      
53      private static void updateLegacyDatastream(DigitalObject obj,
54                                                 String dsId,
55                                                 String mimeType,
56                                                 String formatURI) {
57          for (Datastream ds: obj.datastreams(dsId)) {
58              ds.DSMIME = mimeType;
59              ds.DSFormatURI = formatURI;
60          }
61      }
