39  /**
40   * Serializes objects in the constructor-provider version of the METS Fedora
41   * Extension format.
42   *
43   * @author Sandy Payette
44   * @author Chris Wilper
45   */
46  @SuppressWarnings("deprecation")
47  public class METSFedoraExtDOSerializer
48          implements Constants, DOSerializer {
49  
50      /**
51       * The format this serializer will write if unspecified at construction.
52       * This defaults to the latest METS Fedora Extension format.
53       */
54      public static final XMLFormat DEFAULT_FORMAT = METS_EXT1_1;
55  
56      private static final Logger logger =
57              LoggerFactory.getLogger(METSFedoraExtDOSerializer.class);
58  
59      private static final char [] DS_INDENT = "              ".toCharArray();
60      
61      /** The format this serializer writes. */
62      private final XMLFormat m_format;
63  
64      /** The current translation context. */
65      private int m_transContext;
66  
67      /**
68       * Creates a serializer that writes the default METS Fedora Extension
69       * format.
70       */
71      public METSFedoraExtDOSerializer() {
72          m_format = DEFAULT_FORMAT;
73      }
74  
75      /**
76       * Creates a serializer that writes the given METS Fedora Extension format.
77       *
78       * @param format
79       *        the version-specific METS Fedora Extension format.
80       * @throws IllegalArgumentException
81       *         if format is not a known METS Fedora extension format.
82       */
83      public METSFedoraExtDOSerializer(XMLFormat format) {
84          if (format.equals(METS_EXT1_0) || format.equals(METS_EXT1_1)) {
85              m_format = format;
86          } else {
87              throw new IllegalArgumentException("Not a METS Fedora Extension "
88                      + "format: " + format.uri);
89          }
90      }
91  
92      //---
93      // DOSerializer implementation
94      //---
95  
96      /**
97       * {@inheritDoc}
98       */
99      public DOSerializer getInstance() {
100         return new METSFedoraExtDOSerializer(m_format);
101     }

103     /**
104      * {@inheritDoc}
105      */
106     public void serialize(DigitalObject obj,
107                           OutputStream out,
108                           String encoding,
109                           int transContext) throws ObjectIntegrityException,
110             StreamIOException, UnsupportedEncodingException {
111         if (logger.isDebugEnabled()) {
112             logger.debug("Serializing " + m_format.uri + " for transContext: "
113                 + transContext);
114         }
115         m_transContext = transContext;
116         OutputStreamWriter osWriter = new OutputStreamWriter(out, encoding);
117         PrintWriter writer = new PrintWriter(osWriter);
118         try {
119             appendXMLDeclaration(obj, encoding, writer);
120             appendRootElementStart(obj, writer);
121             appendHdr(obj, writer);
122             appendDescriptiveMD(obj, writer, encoding);
123             appendAuditRecordAdminMD(obj, writer);
124             appendOtherAdminMD(obj, writer, encoding);
125             appendFileSecs(obj, writer);
126             if (m_format.equals(METS_EXT1_0)) {
127                 appendStructMaps(obj, writer);
128                 appendDisseminators(obj, writer);
129             }
130             appendRootElementEnd(writer);
131         } finally {
132             writer.close();
133         }
134     }

140     private void appendXMLDeclaration(DigitalObject obj,
141                                       String encoding,
142                                       PrintWriter writer) {
143         writer.print("<?xml version=\"1.0\" encoding=\"");
144         writer.print(encoding);
145         writer.print("\" ?>\n");
146     }

148     private void appendRootElementStart(DigitalObject obj, PrintWriter writer)
149             throws ObjectIntegrityException {
150         writer.print("<");
151         writer.print(METS.METS.qName);
152         if (m_format.equals(METS_EXT1_1)) {
153             writer.print(" ");
154             writer.print(METS_EXT.EXT_VERSION.localName);
155             writer.print("=\"1.1\"");
156         }
157         writer.print(" ");
158         writer.print(METS.OBJID.localName);
159         writer.print("=\"");
160         writer.print(obj.getPid());
161         writer.print("\"");
162 
163         if (m_format.equals(METS_EXT1_0)
164                 && DOTranslationUtility.getTypeAttribute(obj) != null) {
165             writer.print(" ");
166             writer.print(METS.TYPE.localName);
167             writer.print("=\"");
168             writer.print(DOTranslationUtility.getTypeAttribute(obj).localName);
169             writer.print("\"");
170         }
171 
172         writer.print("\n");
173         String label = obj.getLabel();
174         if (label != null && label.length() > 0) {
175             writer.print(METS.LABEL.localName);
176             writer.print("=\"");
177             writer.print(StreamUtility.enc(label));
178             writer.print("\"\n");
179         }
180         writer.print("xmlns:");
181         writer.print(METS.prefix);
182         writer.print("=\"");
183         writer.print(METS.uri);
184         writer.print("\"\n");
185         if (m_format.equals(METS_EXT1_0)) {
186             writer.print("xmlns:");
187             writer.print(XLINK.prefix);
188             writer.print("=\"");
189             writer.print(OLD_XLINK.uri);
190             writer.print("\"\n");
191         } else {
192             writer.print("xmlns:");
193             writer.print(XLINK.prefix);
194             writer.print("=\"");
195             writer.print(XLINK.uri);
196             writer.print("\"\n");
197         }
198         writer.print("xmlns:");
199         writer.print(XSI.prefix);
200         writer.print("=\"");
201         writer.print(XSI.uri);
202         writer.print("\"\n");
203         writer.print(XSI.SCHEMA_LOCATION.qName);
204         writer.print("=\"");
205         writer.print(METS.uri);
206         writer.print(" ");
207         writer.print(m_format.xsdLocation);
208         writer.print("\">\n");
209     }

211     private void appendHdr(DigitalObject obj, PrintWriter writer) throws ObjectIntegrityException {
212         writer.print("<");
213         writer.print(METS.prefix);
214         writer.print(":metsHdr");
215         Date cDate = obj.getCreateDate();
216         if (cDate != null) {
217             writer.print(" CREATEDATE=\"");
218             writer.print(DateUtility.convertDateToString(cDate));
219             writer.print("\"");
220         }
221         Date mDate = obj.getLastModDate();
222         if (mDate != null) {
223             writer.print(" LASTMODDATE=\"");
224             writer.print(DateUtility.convertDateToString(mDate));
225             writer.print("\"");
226         }
227 
228         writer.print(" RECORDSTATUS=\"");
229         writer.print(DOTranslationUtility.getStateAttribute(obj));
230         writer.print("\"");
231 
232         writer.print(">\n");
233         // use agent to identify the owner of the digital object
234         String ownerId = obj.getOwnerId();
235         if (ownerId != null && !ownerId.equals("")) {
236             writer.print("<");
237             writer.print(METS.prefix);
238             writer.print(":agent");
239             writer.print(" ROLE=\"IPOWNER\">\n");
240             writer.print("<");
241             writer.print(METS.prefix);
242             writer.print(":name>");
243             writer.print(ownerId);
244             writer.print("</");
245             writer.print(METS.prefix);
246             writer.print(":name>\n");
247             writer.print("</");
248             writer.print(METS.prefix);
249             writer.print(":agent>\n");
250         }
251         writer.print("</");
252         writer.print(METS.prefix);
253         writer.print(":metsHdr>\n");
254     }

256     private void appendDescriptiveMD(DigitalObject obj,
257                                      PrintWriter writer,
258                                      String encoding)
259             throws ObjectIntegrityException, UnsupportedEncodingException,
260             StreamIOException {
261         Iterator<String> iter = obj.datastreamIdIterator();
262         while (iter.hasNext()) {
263             String id = iter.next();
264             Datastream firstDS = obj.datastreams(id).iterator().next();
265             if (firstDS.DSControlGrp.equals("X")
266                     && ((DatastreamXMLMetadata) firstDS).DSMDClass == DatastreamXMLMetadata.DESCRIPTIVE) {
267                 appendMDSec(obj,
268                             "dmdSecFedora",
269                             "descMD",
270                             obj.datastreams(id),
271                             writer,
272                             encoding);
273             }
274         }
275     }

277     private void appendMDSec(DigitalObject obj,
278                              String outerName,
279                              String innerName,
280                              Iterable<Datastream> XMLMetadata,
281                              PrintWriter writer,
282                              String encoding) throws ObjectIntegrityException,
283             UnsupportedEncodingException, StreamIOException {
284         DatastreamXMLMetadata first =
285                 (DatastreamXMLMetadata) DOTranslationUtility
286                         .setDatastreamDefaults(XMLMetadata
287                                 .iterator().next());
288         writer.print("<");
289         writer.print(METS.prefix);
290         writer.print(":");
291         writer.print(outerName);
292         writer.print(" ID=\"");
293         writer.print(first.DatastreamID);
294         writer.print("\" STATUS=\"");
295         writer.print(first.DSState);
296         writer.print("\" VERSIONABLE=\"");
297         writer.print(first.DSVersionable);
298         writer.print("\">\n");
299         for (Datastream d : XMLMetadata) {
300             DatastreamXMLMetadata ds =
301                     (DatastreamXMLMetadata) DOTranslationUtility
302                             .setDatastreamDefaults(d);
303 
304             writer.print("<");
305             writer.print(METS.prefix);
306             writer.print(":");
307             writer.print(innerName);
308             writer.print(" ID=\"");
309             writer.print(ds.DSVersionID);
310             writer.print("\"");
311             if (ds.DSCreateDT != null) {
312                 writer.print(" CREATED=\"");
313                 writer.print(DateUtility.convertDateToString(ds.DSCreateDT));
314                 writer.print("\"");
315             }
316             writer.print(">\n");
317 
318 
319 
320 
321             writer.print("<");
322             writer.print(METS.prefix);
323             writer.print(":mdWrap MIMETYPE=\"");
324             writer.print(StreamUtility.enc(ds.DSMIME));
325             writer.print("\" MDTYPE=\"");
326             String mdType = ds.DSInfoType;
327             if (!mdType.equals("MARC") && !mdType.equals("EAD")
328                     && !mdType.equals("DC") && !mdType.equals("NISOIMG")
329                     && !mdType.equals("LC-AV") && !mdType.equals("VRA")
330                     && !mdType.equals("TEIHDR") && !mdType.equals("DDI")
331                     && !mdType.equals("FGDC")) {
332                 writer.print("OTHER\" OTHERMDTYPE=\"");
333                 writer.print(StreamUtility.enc(mdType));
334             } else {
335                 writer.print(mdType);
336             }
337             writer.print("\" ");
338 
339             if (ds.DSLabel != null && !ds.DSLabel.equals("")) {
340                 writer.print(" LABEL=\"");
341                 writer.print(StreamUtility.enc(ds.DSLabel));
342                 writer.print("\"");
343             }
344 
345             if (ds.DSFormatURI != null && !ds.DSFormatURI.equals("")) {
346                 writer.print(" FORMAT_URI=\"");
347                 writer.print(StreamUtility.enc(ds.DSFormatURI));
348                 writer.print("\"");
349             }
350 
351             String altIds = DOTranslationUtility.oneString(ds.DatastreamAltIDs);
352             if (altIds != null && !altIds.equals("")) {
353                 writer.print(" ALT_IDS=\"");
354                 writer.print(StreamUtility.enc(altIds));
355                 writer.print("\"");
356             }
357 
358             // CHECKSUM and CHECKSUMTYPE are also optional
359             String csType = ds.DSChecksumType;
360             if (csType != null
361                     && csType.length() > 0
362                     && !csType.equals(Datastream.CHECKSUMTYPE_DISABLED)) {
363                 writer.print(" CHECKSUM=\"");
364                 writer.print(StreamUtility.enc(ds.DSChecksum));
365                 writer.print("\" CHECKSUMTYPE=\"");
366                 writer.print(StreamUtility.enc(csType));
367                 writer.print("\"");
368             }
369 
370             writer.print(">\n");
371             writer.print("<");
372             writer.print(METS.prefix);
373             writer.print(":xmlData>\n");
374 
375             // If WSDL or SERVICE-PROFILE datastream (in SDep)
376             // make sure that any embedded URLs are encoded
377             // appropriately for either EXPORT or STORE.
378             if (obj.hasContentModel(SERVICE_DEPLOYMENT_3_0)
379                     && ds.DatastreamID.equals("SERVICE-PROFILE")
380                     || ds.DatastreamID.equals("WSDL")) {
381                 writer.print(DOTranslationUtility
382                         .normalizeInlineXML(new String(ds.xmlContent, "UTF-8")
383                                 .trim(), m_transContext));
384             } else {
385                 DOTranslationUtility.appendXMLStream(ds.getContentStream(),
386                                                      writer,
387                                                      encoding);
388             }
389             writer.print("\n</");
390             writer.print(METS.prefix);
391             writer.print(":xmlData>");
392             writer.print("</");
393             writer.print(METS.prefix);
394             writer.print(":mdWrap>\n");
395             writer.print("</");
396             writer.print(METS.prefix);
397             writer.print(":");
398             writer.print(innerName);
399             writer.print(">\n");
400         }
401         writer.print("</");
402         writer.print(METS.prefix);
403         writer.print(":");
404         writer.print(outerName);
405         writer.print(">\n");
406     }

446     private void appendOtherAdminMD(DigitalObject obj,
447                                     PrintWriter writer,
448                                     String encoding)
449             throws ObjectIntegrityException, UnsupportedEncodingException,
450             StreamIOException {
451         Iterator<String> iter = obj.datastreamIdIterator();
452         while (iter.hasNext()) {
453             String id = iter.next();
454             Datastream firstDS =
455                     obj.datastreams(id).iterator().next();
456             // First, work with the first version to get the mdClass set to
457             // a proper value required in the METS XML Schema.
458             if (firstDS.DSControlGrp.equals("X")
459                     && ((DatastreamXMLMetadata) firstDS).DSMDClass != DatastreamXMLMetadata.DESCRIPTIVE) {
460                 DatastreamXMLMetadata md = (DatastreamXMLMetadata) firstDS;
461                 // Default mdClass to techMD when a valid one does not appear
462                 // (say because the object was born as FOXML)
463                 String mdClass = "techMD";
464                 if (md.DSMDClass == DatastreamXMLMetadata.TECHNICAL) {
465                     mdClass = "techMD";
466                 } else if (md.DSMDClass == DatastreamXMLMetadata.SOURCE) {
467                     mdClass = "sourceMD";
468                 } else if (md.DSMDClass == DatastreamXMLMetadata.RIGHTS) {
469                     mdClass = "rightsMD";
470                 } else if (md.DSMDClass == DatastreamXMLMetadata.DIGIPROV) {
471                     mdClass = "digiprovMD";
472                 }
473                 // Then, pass everything along to do the actual serialization
474                 appendMDSec(obj,
475                             "amdSec",
476                             mdClass,
477                             obj.datastreams(id),
478                             writer,
479                             encoding);
480             }
481         }
482     }

484     private void appendFileSecs(DigitalObject obj, PrintWriter writer)
485             throws ObjectIntegrityException, StreamIOException {
486         Iterator<String> iter = obj.datastreamIdIterator();
487         boolean didFileSec = false;
488         while (iter.hasNext()) {
489             Datastream ds =
490                     DOTranslationUtility.setDatastreamDefaults(obj
491                             .datastreams(iter.next()).iterator().next());
492             if (!ds.DSControlGrp.equals("X")) {
493                 if (!didFileSec) {
494                     didFileSec = true;
495                     writer.print("<");
496                     writer.print(METS.prefix);
497                     writer.print(":fileSec>\n");
498                     writer.print("<");
499                     writer.print(METS.prefix);
500                     writer.print(":fileGrp ID=\"DATASTREAMS\">\n");
501                 }
502                 writer.print("<");
503                 writer.print(METS.prefix);
504                 writer.print(":fileGrp ID=\"");
505                 writer.print(ds.DatastreamID);
506                 writer.print("\" STATUS=\"");
507                 writer.print(ds.DSState);
508                 writer.print("\" VERSIONABLE=\"");
509                 writer.print(ds.DSVersionable);
510                 writer.print("\">\n");
511                 Iterator<Datastream> contentIter =
512                         obj.datastreams(ds.DatastreamID).iterator();
513                 while (contentIter.hasNext()) {
514                     Datastream dsc =
515                             DOTranslationUtility
516                                     .setDatastreamDefaults(contentIter.next());
517 
518 
519                     writer.print("<");
520                     writer.print(METS.prefix);
521                     writer.print(":file ID=\"");
522                     writer.print(dsc.DSVersionID);
523                     writer.print("\"");
524                     if (dsc.DSCreateDT != null) {
525                         writer.print(" CREATED=\"");
526                         writer.print(DateUtility.convertDateToString(dsc.DSCreateDT));
527                         writer.print("\"");
528                     }
529                     writer.print(" MIMETYPE=\"");
530                     writer.print(StreamUtility.enc(dsc.DSMIME));
531                     writer.print("\"");
532                     if (dsc.DSSize != 0) {
533                         writer.print(" SIZE=\"" + dsc.DSSize + "\"");
534                     }
535                     if (dsc.DSFormatURI != null && !dsc.DSFormatURI.equals("")) {
536                         writer.print(" FORMAT_URI=\"");
537                         writer.print(StreamUtility.enc(dsc.DSFormatURI));
538                         writer.print("\"");
539                     }
540                     String altIds =
541                             DOTranslationUtility
542                                     .oneString(dsc.DatastreamAltIDs);
543                     if (altIds != null && !altIds.equals("")) {
544                         writer.print(" ALT_IDS=\"");
545                         writer.print(StreamUtility.enc(altIds));
546                         writer.print("\"");
547                     }
548                     String csType = dsc.DSChecksumType;
549                     if (csType != null
550                             && csType.length() > 0
551                             && !csType.equals(Datastream.CHECKSUMTYPE_DISABLED)) {
552                         writer.print(" CHECKSUM=\"");
553                         writer.print(StreamUtility.enc(dsc.DSChecksum));
554                         writer.print("\"");
555                         writer.print(" CHECKSUMTYPE=\"");
556                         writer.print(StreamUtility.enc(csType));
557                         writer.print("\"");
558                     }
559                     writer.print(" OWNERID=\"");
560                     writer.print(dsc.DSControlGrp);
561                     writer.print("\">\n");
562                     if (m_transContext == DOTranslationUtility.SERIALIZE_EXPORT_ARCHIVE
563                             && dsc.DSControlGrp.equalsIgnoreCase("M")) {
564                         serializeDatastreamContent(dsc, writer);
565                     } else {
566                         writer.print("<");
567                         writer.print(METS.prefix);
568                         writer.print(":FLocat");
569                         if (dsc.DSLabel != null && !dsc.DSLabel.equals("")) {
570                             writer.print(" ");
571                             writer.print(XLINK.prefix);
572                             writer.print(":title=\"");
573                             writer.print(StreamUtility.enc(dsc.DSLabel));
574                             writer.print("\"");
575                         }
576                         writer.print(" LOCTYPE=\"URL\" ");
577                         writer.print(XLINK.prefix);
578                         writer.print(":href=\"");
579                         writer.print(StreamUtility.enc(
580                                 DOTranslationUtility.normalizeDSLocationURLs(
581                                         obj.getPid(),
582                                         dsc,
583                                         m_transContext).DSLocation));
584                         writer.print("\"/>\n");
585                     }
586                     writer.print("</");
587                     writer.print(METS.prefix);
588                     writer.print(":file>\n");
589                 }
590                 writer.print("</");
591                 writer.print(METS.prefix);
592                 writer.print(":fileGrp>\n");
593             }
594         }
595         if (didFileSec) {
596             writer.print("</");
597             writer.print(METS.prefix);
598             writer.print(":fileGrp>\n");
599             writer.print("</");
600             writer.print(METS.prefix);
601             writer.print(":fileSec>\n");
602         }
603     }

605     protected void serializeDatastreamContent(Datastream dsc, PrintWriter writer)
606             throws StreamIOException {
607         writer.print("<");
608         writer.print(METS.prefix);
609         writer.print(":FContent> \n");
610         Reader encoded = new InputStreamReader(Base64.encodeToStream(dsc.getContentStream()));
611         char [] buffer = new char[80];
612         int len = 0;           
613         
614         try{
615             while ((len = encoded.read(buffer)) > -1){
616                 writer.write(DS_INDENT);
617                 writer.write(buffer,0,len);
618                 writer.write('\n');
619             }
620         } catch (IOException ioe) {
621            throw new StreamIOException(ioe.getMessage()); 
622         }
623         writer.print("</");
624         writer.print(METS.prefix);
625         writer.print(":FContent>\n");
626     }

628     private void appendStructMaps(DigitalObject obj, PrintWriter writer)
629             throws ObjectIntegrityException {
630         Iterator<String> dissIdIter = obj.disseminatorIdIterator();
631         while (dissIdIter.hasNext()) {
632             String did = dissIdIter.next();
633             Iterator<Disseminator> dissIter = obj.disseminators(did).iterator();
634             while (dissIter.hasNext()) {
635                 Disseminator diss =
636                         DOTranslationUtility.setDisseminatorDefaults(dissIter
637                                 .next());
638                 writer.print("<");
639                 writer.print(METS.prefix);
640                 writer.print(":structMap ID=\"");
641                 writer.print(diss.dsBindMapID);
642                 writer.print("\" TYPE=\"fedora:dsBindingMap\">\n");
643                 writer.print("<");
644                 writer.print(METS.prefix);
645                 writer.print(":div TYPE=\"");
646                 writer.print(diss.sDepID);
647                 writer.print("\"");
648                 if (diss.dsBindMap.dsBindMapLabel != null
649                         && !diss.dsBindMap.dsBindMapLabel.equals("")) {
650                     writer.print(" LABEL=\"");
651                     writer.print(StreamUtility.enc(diss.dsBindMap.dsBindMapLabel));
652                     writer.print("\"");
653                 }
654                 writer.print(">\n");
655                 DSBinding[] bindings = diss.dsBindMap.dsBindings;
656                 for (int i = 0; i < bindings.length; i++) {
657                     if (bindings[i].bindKeyName == null
658                             || bindings[i].bindKeyName.equals("")) {
659                         throw new ObjectIntegrityException("Object's disseminator"
660                                 + " binding map binding must have a binding key name.");
661                     }
662                     writer.print("<");
663                     writer.print(METS.prefix);
