40  /**
41   * Serializes objects in the constructor-provided version of FOXML.
42   *
43   * @author Sandy Payette
44   * @author Chris Wilper
45   */
46  @SuppressWarnings("deprecation")
47  public class FOXMLDOSerializer
48          implements DOSerializer, Constants {
49  
50      /**
51       * The format this serializer will write if unspecified at construction.
52       * This defaults to the latest FOXML format.
53       */
54      public static final XMLFormat DEFAULT_FORMAT = FOXML1_1;
55  
56      private static final Logger logger =
57              LoggerFactory.getLogger(FOXMLDOSerializer.class);
58  
59      /** The format this serializer writes. */
60      private final XMLFormat m_format;
61  
62      /** The current translation context. */
63      private int m_transContext;
64  
65      /**
66       * Creates a serializer that writes the default FOXML format.
67       */
68      public FOXMLDOSerializer() {
69          m_format = DEFAULT_FORMAT;
70      }
71  
72      /**
73       * Creates a serializer that writes the given FOXML format.
74       *
75       * @param format
76       *        the version-specific FOXML format.
77       * @throws IllegalArgumentException
78       *         if format is not a known FOXML format.
79       */
80      public FOXMLDOSerializer(XMLFormat format) {
81          if (format.equals(FOXML1_0) || format.equals(FOXML1_1)) {
82              m_format = format;
83          } else {
84              throw new IllegalArgumentException("Not a FOXML format: "
85                      + format.uri);
86          }
87      }
88

90      // DOSerializer implementation
91      //---
92  
93      /**
94       * {@inheritDoc}
95       */
96      public DOSerializer getInstance() {
97          return new FOXMLDOSerializer(m_format);
98      }

100     /**
101      * {@inheritDoc}
102      */
103     public void serialize(DigitalObject obj,
104                           OutputStream out,
105                           String encoding,
106                           int transContext) throws ObjectIntegrityException,
107             StreamIOException, UnsupportedEncodingException {
108         logger.debug("Serializing " + m_format.uri + " for transContext: "
109                 + transContext);
110         m_transContext = transContext;
111         OutputStreamWriter osWriter = new OutputStreamWriter(out, encoding);
112         PrintWriter writer = new PrintWriter(new BufferedWriter(osWriter));
113         try {
114             appendXMLDeclaration(obj, encoding, writer);
115             appendRootElementStart(obj, writer);
116             appendProperties(obj, writer, encoding);
117             appendAudit(obj, writer, encoding);
118             appendDatastreams(obj, writer, encoding);
119             if (m_format.equals(FOXML1_0)) {
120                 appendDisseminators(obj, writer);
121             }
122             appendRootElementEnd(writer);
123         } finally {
124             writer.close();
125         }
126     }

140     private void appendRootElementStart(DigitalObject obj, PrintWriter writer)
141             throws ObjectIntegrityException {
142         writer.print("<");
143         writer.print(FOXML.DIGITAL_OBJECT.qName);
144         if (m_format.equals(FOXML1_1)) {
145             writer.print(" ");
146             writer.print(FOXML.VERSION.localName);
147             writer.print("=\"1.1\"");
148         }
149         writer.print(" ");
150         writer.print(FOXML.PID.localName);
151         writer.print("=\"");
152         writer.print(obj.getPid());
153         writer.print("\"");
154         if (m_transContext == DOTranslationUtility.SERIALIZE_EXPORT_PUBLIC) {
155             writer.print(" ");
156             writer.print(FOXML.FEDORA_URI.localName);
157             writer.print("=\"info:fedora/");
158             writer.print(obj.getPid());
159             writer.print("\"");
160         }
161         writer.print("\nxmlns:");
162         writer.print(FOXML.prefix);
163         writer.print("=\"");
164         writer.print(FOXML.uri);
165         writer.print("\"\nxmlns:");
166         writer.print(XSI.prefix);
167         writer.print("=\"");
168         writer.print(XSI.uri);
169         writer.print("\"\n");
170         writer.print(XSI.SCHEMA_LOCATION.qName);
171         writer.print("=\"");
172         writer.print(FOXML.uri);
173         writer.print(" ");
174         writer.print(m_format.xsdLocation);
175         writer.print("\">\n");
176     }

178     private void appendProperties(DigitalObject obj,
179                                   PrintWriter writer,
180                                   String encoding)
181             throws ObjectIntegrityException {
182 
183         writer.print("<");
184         writer.print(FOXML.prefix);
185         writer.print(":objectProperties>\n");
186 
187         /*
188          * fType is eliminated in foxml 1.1+, so choose the best reasonable
189          * value for 1.0 serializations
190          */
191         if (m_format.equals(FOXML1_0)) {
192             RDFName ftype = DOTranslationUtility.getTypeAttribute(obj);
193             if (ftype != null) {
194                 appendProperty(RDF.TYPE.uri, ftype.uri, writer, false);
195             }
196         }
197 
198         appendProperty(MODEL.STATE.uri,
199                        DOTranslationUtility.getStateAttribute(obj),
200                        writer,
201                        false);
202         appendProperty(MODEL.LABEL.uri, obj.getLabel(), writer, false);
203         appendProperty(MODEL.OWNER.uri, obj.getOwnerId(), writer, false);
204         appendProperty(MODEL.CREATED_DATE.uri, obj.getCreateDate(), writer);
205         appendProperty(VIEW.LAST_MODIFIED_DATE.uri,
206                        obj.getLastModDate(),
207                        writer);
208 
209         Iterator<String> iter = obj.getExtProperties().keySet().iterator();
210         while (iter.hasNext()) {
211             String name = iter.next();
212             appendProperty(name, obj.getExtProperty(name), writer, true);
213         }
214         writer.print("</");
215         writer.print(FOXML.prefix);
216         writer.print(":objectProperties>\n");
217     }

219     private static void appendProperty(String uri,
220                                        String value,
221                                        PrintWriter writer,
222                                        boolean extProperty) {
223         if (value != null) {
224             writer.print("<");
225             writer.print(FOXML.prefix);
226             writer.print(':');
227             if (extProperty) {
228                 writer.print("ext");
229             }
230             writer.print("property NAME=\"");
231             writer.print(uri);
232             writer.print("\" VALUE=\"");
233             writer.print(StreamUtility.enc(value));
234             writer.print("\"/>\n");
235         }

249     private void appendDatastreams(DigitalObject obj,
250                                    PrintWriter writer,
251                                    String encoding)
252             throws ObjectIntegrityException, UnsupportedEncodingException,
253             StreamIOException {
254         Iterator<String> iter = obj.datastreamIdIterator();
255         while (iter.hasNext()) {
256             String dsid = iter.next();
257             boolean haveWrittenCommonAttributes = false;
258 
259             // AUDIT datastream is rebuilt from the latest in-memory audit trail
260             // which is a separate array list in the DigitalObject class.
261             // So, ignore it here.
262             if (dsid.equals("AUDIT") || dsid.equals("FEDORA-AUDITTRAIL")) {
263                 continue;
264             }
265             // Given a datastream ID, get all the datastream versions.
266             // Use the first version to pick up the attributes common to all versions.
267 
268             for (Datastream v : obj.datastreams(dsid)) {
269                 Datastream vds = DOTranslationUtility.setDatastreamDefaults(v);
270 
271                 // insert the ds attributes common to all versions, when necessary
272                 if (!haveWrittenCommonAttributes) {
273                     writer.print("<");
274                     writer.print(FOXML.prefix);
275                     writer.print(":datastream ID=\"");
276                     writer.print(vds.DatastreamID);
277                     writer.print("\"");
278                     if (m_transContext == DOTranslationUtility.SERIALIZE_EXPORT_PUBLIC) {
279                         writer.print(" FEDORA_URI=\"");
280                         writer.print("info:fedora/");
281                         writer.print(obj.getPid());
282                         writer.print("/");
283                         writer.print(vds.DatastreamID);
284                         writer.print("\"");
285                     }
286                     writer.print(" STATE=\"");
287                     writer.print(vds.DSState);
288                     writer.print("\"");
289                     writer.print(" CONTROL_GROUP=\"");
290                     writer.print(vds.DSControlGrp);
291                     writer.print("\"");
292                     writer.print(" VERSIONABLE=\"");
293                     writer.print(vds.DSVersionable);
294                     writer.print("\">\n");
295                     haveWrittenCommonAttributes = true;
296                 }
297 
298                 // insert the ds version elements
299                 writer.print("<");
300                 writer.print(FOXML.prefix);
301                 writer.print(":datastreamVersion ID=\"");
302                 writer.print(vds.DSVersionID);
303                 writer.print("\"");
304                 writer.print(" LABEL=\"");
305                 writer.print(StreamUtility.enc(vds.DSLabel));
306                 writer.print("\"");
307                 if (vds.DSCreateDT != null) {
308                     writer.print(" CREATED=\"");
309                     writer.print(DateUtility.convertDateToString(vds.DSCreateDT));
310                     writer.print("\"");
311                 }
312                 String altIds =
313                         DOTranslationUtility.oneString(vds.DatastreamAltIDs);
314                 if (altIds != null && !altIds.equals("")) {
315                     writer.print(" ALT_IDS=\"");
316                     writer.print(StreamUtility.enc(altIds));
317                     writer.print("\"");
318                 }
319                 writer.print(" MIMETYPE=\"");
320                 writer.print(StreamUtility.enc(vds.DSMIME));
321                 writer.print("\"");
322                 if (vds.DSFormatURI != null && !vds.DSFormatURI.equals("")) {
323                     writer.print(" FORMAT_URI=\"");
324                     writer.print(StreamUtility.enc(vds.DSFormatURI));
325                     writer.print("\"");
326                 }
327                 // include size if it's non-zero
328                 if (vds.DSSize != 0) {
329                     writer.print(" SIZE=\"");
330                     writer.print(vds.DSSize);
331                     writer.print("\"");
332                 }
333                 writer.print(">\n");
334 
335                 // include checksum if it has a value
336                 String csType = vds.getChecksumType();
337                 if (csType != null && csType.length() > 0
338                         && !csType.equals(Datastream.CHECKSUMTYPE_DISABLED)) {
339                     writer.print("<");
340                     writer.print(FOXML.prefix);
341                     writer.print(":contentDigest TYPE=\"");
342                     writer.print(csType);
343                     writer.print("\"");
344                     writer.print(" DIGEST=\"");
345                     writer.print(vds.getChecksum());
346                     writer.print("\"/>\n");
347                 }
348 
349                 // if E or R insert ds content location as URL
350                 if (vds.DSControlGrp.equalsIgnoreCase("E")
351                         || vds.DSControlGrp.equalsIgnoreCase("R")) {
352                     writer.print("<");
353                     writer.print(FOXML.prefix);
354                     writer.print(":contentLocation TYPE=\"");
355                     writer.print(Datastream.DS_LOCATION_TYPE_URL);
356                     writer.print("\" REF=\"");
357                     String urls = DOTranslationUtility.normalizeDSLocationURLs(
358                             obj.getPid(),
359                             vds,
360                             m_transContext).DSLocation;
361                     writer.print(StreamUtility.enc(urls));
362                     writer.print("\"/>\n");
363                     // if M insert ds content location as an internal identifier
364                 } else if (vds.DSControlGrp.equalsIgnoreCase("M")) {
365                     if (m_transContext == DOTranslationUtility.SERIALIZE_EXPORT_ARCHIVE) {
366                         writer.print("<");
367                         writer.print(FOXML.prefix);
368                         writer.print(":binaryContent> \n");
369                         String encoded = Base64.encodeToString(vds.getContentStream());
370                         writer.print(StringUtility.splitAndIndent(encoded,
371                                                                   14,
372                                                                   80));
373                         writer.print("</");
374                         writer.print(FOXML.prefix);
375                         writer.print(":binaryContent> \n");
376                     } else {
377                         writer.print("<");
378                         writer.print(FOXML.prefix);
379                         writer.print(":contentLocation TYPE=\"");
380                         writer.print(Datastream.DS_LOCATION_TYPE_INTERNAL);
381                         writer.print("\" REF=\"");
382                         String urls = DOTranslationUtility.normalizeDSLocationURLs(
383                                 obj.getPid(),
384                                 vds,
385                                 m_transContext).DSLocation;
386                         writer.print(StreamUtility.enc(urls));
387                         writer.print("\"/>\n");
388                     }
389                     // if X insert inline XML
390                 } else if (vds.DSControlGrp.equalsIgnoreCase("X")) {
391                     appendInlineXML(obj,
392                                     (DatastreamXMLMetadata) vds,
393                                     writer,
394                                     encoding);
395                 }
396                 writer.print("</");
397                 writer.print(FOXML.prefix);
398                 writer.print(":datastreamVersion>\n");
399             }
400             writer.print("</");
401             writer.print(FOXML.prefix);
402             writer.print(":datastream>\n");
403         }
404     }

406     private void appendAudit(DigitalObject obj,
407                              PrintWriter writer,
408                              String encoding) throws ObjectIntegrityException {
409 
410         if (obj.getAuditRecords().size() > 0) {
411             // Audit trail datastream re-created from audit records.
412             // There is only ONE version of the audit trail datastream!
413             writer.print("<");
414             writer.print(FOXML.prefix);
415             writer.print(":datastream ID=\"");
416             writer.print("AUDIT\"");
417             if (m_transContext == DOTranslationUtility.SERIALIZE_EXPORT_PUBLIC) {
418                 writer.print(" FEDORA_URI=\"info:fedora/");
419                 writer.print(obj.getPid());
420                 writer.print("/AUDIT\"");
421             }
422             writer.print(" STATE=\"A\" CONTROL_GROUP=\"X\" VERSIONABLE=\"false\">\n");
423             // insert the ds version-level elements
424             writer.print("<");
425             writer.print(FOXML.prefix);
426             writer.print(":datastreamVersion ID=\"AUDIT.0\" LABEL=\"");
427             writer.print("Audit Trail for this object\" CREATED=\"");
428             writer.print(DateUtility.convertDateToString(obj.getCreateDate()));
429             writer.print("\" MIMETYPE=\"text/xml\" FORMAT_URI=\"");
430             writer.print(AUDIT1_0.uri);
431             writer.print("\">\n");
432             writer.print("<");
433             writer.print(FOXML.prefix);
434             writer.print(":xmlContent>\n");
435             DOTranslationUtility.appendAuditTrail(obj, writer);
436             writer.print("</");
437             writer.print(FOXML.prefix);
438             writer.print(":xmlContent>\n");
439             writer.print("</");
440             writer.print(FOXML.prefix);
441             writer.print(":datastreamVersion>\n");
442             writer.print("</");
443             writer.print(FOXML.prefix);
444             writer.print(":datastream>\n");
445         }

448     private void appendInlineXML(DigitalObject obj,
449                                  DatastreamXMLMetadata ds,
450                                  PrintWriter writer,
451                                  String encoding)
452             throws ObjectIntegrityException, UnsupportedEncodingException,
453             StreamIOException {
454 
455         writer.print("<");
456         writer.print(FOXML.prefix);
457         writer.print(":xmlContent>\n");
458 
459         // Relative Repository URLs: If it's a WSDL or SERVICE-PROFILE datastream
460         // in a SDep object search for any embedded URLs that are relative to
461         // the local repository (like internal service URLs) and make sure they
462         // are converted appropriately for the translation context.
463         if (obj.hasContentModel(SERVICE_DEPLOYMENT_3_0)
464                 && (ds.DatastreamID.equals("SERVICE-PROFILE") || ds.DatastreamID
465                         .equals("WSDL"))) {
466             // FIXME! We need a more efficient way than to search
467             // the whole block of inline XML. We really only want to
468             // look at service URLs in the XML.
469             writer.print(DOTranslationUtility
470                     .normalizeInlineXML(new String(ds.xmlContent, "UTF-8")
471                             .trim(), m_transContext));
472         } else {
473             DOTranslationUtility.appendXMLStream(ds.getContentStream(),
474                                                  writer,
475                                                  encoding);
476         }
477         writer.print("\n</");
478         writer.print(FOXML.prefix);
479         writer.print(":xmlContent>\n");
480     }

482     private void appendDisseminators(DigitalObject obj, PrintWriter writer)
483             throws ObjectIntegrityException {
484 
485         Iterator<String> dissIdIter = obj.disseminatorIdIterator();
486         while (dissIdIter.hasNext()) {
487             String did = dissIdIter.next();
488             List<Disseminator> dissList = obj.disseminators(did);
489 
490             for (int i = 0; i < dissList.size(); i++) {
491                 Disseminator vdiss =
492                         DOTranslationUtility
493                                 .setDisseminatorDefaults(obj
494                                         .disseminators(did).get(i));
495                 // insert the disseminator elements common to all versions.
496                 if (i == 0) {
497                     writer.print("<");
498                     writer.print(FOXML.prefix);
499                     writer.print(":disseminator ID=\"");
500                     writer.print(did);
501                     writer.print("\" BDEF_CONTRACT_PID=\"");
502                     writer.print(vdiss.bDefID);
503                     writer.print("\" STATE=\"");
504                     writer.print(vdiss.dissState);
505                     writer.print("\" VERSIONABLE=\"");
506                     writer.print(vdiss.dissVersionable);
507                     writer.print("\">\n");
508                 }
509                 // insert the disseminator version-level elements
510                 writer.print("<");
511                 writer.print(FOXML.prefix);
512                 writer.print(":disseminatorVersion ID=\"");
513                 writer.print(vdiss.dissVersionID);
514                 writer.print("\"");
515                 if (vdiss.dissLabel != null && !vdiss.dissLabel.equals("")) {
516                     writer.print(" LABEL=\"");
517                     writer.print(StreamUtility.enc(vdiss.dissLabel));
518                     writer.print("\"");
519                 }
520                 writer.print(" BMECH_SERVICE_PID=\"");
521                 writer.print(vdiss.sDepID);
522                 writer.print("\"");
523                 if (vdiss.dissCreateDT != null) {
524                     writer.print(" CREATED=\"");
525                     writer.print(DateUtility.convertDateToString(vdiss.dissCreateDT));
526                     writer.print("\"");
527                 }
528                 writer.print(">\n");
529 
530                 // datastream bindings...
531                 DSBinding[] bindings = vdiss.dsBindMap.dsBindings;
532                 writer.print("<");
533                 writer.print(FOXML.prefix);
534                 writer.print(":serviceInputMap>\n");
535                 for (int j = 0; j < bindings.length; j++) {
536                     if (bindings[j].seqNo == null) {
537                         bindings[j].seqNo = "";
538                     }
539                     writer.print("<");
540                     writer.print(FOXML.prefix);
541                     writer.print(":datastreamBinding KEY=\"");
542                     writer.print(bindings[j].bindKeyName);
543                     writer.print("\" DATASTREAM_ID=\"");
544                     writer.print(bindings[j].datastreamID);
545                     writer.print("\"");
546                     if (bindings[j].bindLabel != null
547                             && !bindings[j].bindLabel.equals("")) {
548                         writer.print(" LABEL=\"");
549                         writer.print(StreamUtility.enc(bindings[j].bindLabel));
550                         writer.print("\"");
551                     }
552                     if (bindings[j].seqNo != null
553                             && !bindings[j].seqNo.equals("")) {
554                         writer.print(" ORDER=\"");
555                         writer.print(bindings[j].seqNo);
556                         writer.print("\"");
557                     }
558                     writer.print("/>\n");
559                 }
560                 writer.print("</");
561                 writer.print(FOXML.prefix);
562                 writer.print(":serviceInputMap>\n");
563                 writer.print("</");
564                 writer.print(FOXML.prefix);
565                 writer.print(":disseminatorVersion>\n");
566             }
567             writer.print("</");
568             writer.print(FOXML.prefix);
569             writer.print(":disseminator>\n");
570         }
571     }
