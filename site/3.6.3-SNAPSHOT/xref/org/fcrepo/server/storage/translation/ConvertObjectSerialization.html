33  /**
34   * Utility class to convert objects from one serialization format to another.

41  public class ConvertObjectSerialization {

43      private static final String ENCODING = "UTF-8";
44  
45      private static final DocumentBuilderFactory factory =
46              DocumentBuilderFactory.newInstance();
47      static {
48          factory.setNamespaceAware(true);

51      private static final OutputFormat fmt = new OutputFormat("XML", ENCODING, true);
52      static {
53          fmt.setIndent(2);
54          fmt.setLineWidth(80);
55          fmt.setPreserveSpace(false);

58      private final Date m_now = new Date();

70      public ConvertObjectSerialization(Class<DODeserializer> deserializer,
71                                        Class<DOSerializer> serializer,
72                                        boolean pretty,
73                                        String inExt,
74                                        String outExt) {
75          m_deserializer = getInstance(deserializer);
76          m_serializer = getInstance(serializer);
77          m_pretty = pretty;
78          m_inExt = inExt;
79          m_outExt = outExt;
80      }

82      private boolean convert(InputStream source, OutputStream destination) {
83          DigitalObject obj = new BasicDigitalObject();
84          try {
85              m_deserializer.deserialize(source,
86                                         obj,
87                                         ENCODING,
88                                         DOTranslationUtility.AS_IS);
89              setObjectDefaults(obj);
90  
91              if (m_pretty) {
92                  prettyPrint(obj, destination);
93              } else {
94                  m_serializer.serialize(obj,
95                                         destination,
96                                         ENCODING,
97                                         DOTranslationUtility.AS_IS);
98              }
99          } catch (Exception e) {
100             e.printStackTrace();
101             return false;
102         }
103         return true;
104     }

106     private void prettyPrint(DigitalObject obj, OutputStream destination)
107             throws Exception {
108         ByteArrayOutputStream outBuf = new ByteArrayOutputStream();
109         m_serializer.serialize(obj,
110                                outBuf,
111                                ENCODING,
112                                DOTranslationUtility.AS_IS);
113         InputStream inBuf = new ByteArrayInputStream(outBuf.toByteArray());
114         prettyPrint(inBuf, destination);
115     }

117     private static void prettyPrint(InputStream source,
118                                     OutputStream destination)
119             throws Exception {
120         XMLSerializer ser = new XMLSerializer(destination, fmt);
121         Document doc = factory.newDocumentBuilder().parse(source);
122         ser.serialize(doc);
123         destination.close();
124     }

126     /**
127      * Convert files from one format to the other.
128      * Hidden directories (directories starting with a ".") are skipped.
129      *
130      * @param source
131      * @param destination
132      * @return
133      */
134     public boolean convert(File source, File destination) {
135         boolean result = true;
136         if (source.isDirectory()) {
137             if (source.getName().startsWith(".")) {
138                 // skip "hidden" directories
139                 return result;
140             }
141             if (destination.exists()) {
142                 result = result && destination.isDirectory();
143             } else {
144                 result = result && destination.mkdirs();
145             }
146             File[] children = source.listFiles();
147             for (File element : children) {
148                 String inName = element.getName();
149                 String outName;
150                 if (element.isDirectory()) {
151                     outName = inName;
152                 } else {
153                     outName = inName.substring(0, inName.lastIndexOf('.') + 1)
154                             + m_outExt;
155                 }
156                 result = result && convert(new File(source, inName),
157                                            new File(destination, outName));
158             }
159             return result;
160         } else {
161             try {
162                 if (!source.getName().endsWith("." + m_inExt)) {
163                     return result;
164                 }
165                 InputStream in = new FileInputStream(source);
166                 if (!destination.getParentFile().exists()) {
167                     destination.getParentFile().mkdirs();
168                 }
169                 OutputStream out = new FileOutputStream(destination);
170                 result = result && convert(in, out);
171                 out.close();
172                 in.close();
173                 return result;
174             } catch (IOException e) {
175                 return false;
176             }
177         }
178     }

180     // - Sets dates to current date if unset
181     // - Sets datastream sizes to 0
182     private void setObjectDefaults(DigitalObject obj) {
183         if (obj.getCreateDate() == null) obj.setCreateDate(m_now);
184         if (obj.getLastModDate() == null) obj.setLastModDate(m_now);
185 
186         Iterator<String> dsIds = obj.datastreamIdIterator();
187         while (dsIds.hasNext()) {
188             String dsid = dsIds.next();
189             for (Datastream ds : obj.datastreams(dsid)) {
190                 ds.DSSize = 0;
191                 if (ds.DSCreateDT == null) {
192                     ds.DSCreateDT = m_now;
193                 }
194             }
195         }
196     }

198     /**
199      * Command-line utility to convert objects from one format to another.
200      *
201      * @param args command-line args.
202      */
203     public static void main(String[] args) throws ClassNotFoundException {
204         LogConfig.initMinimal();
205         if (args.length < 4 || args.length > 7) {
206             die("Expected 4 to 7 arguments", true);
207         }
208         File sourceDir = new File(args[0]);
209         if (!sourceDir.isDirectory()) {
210             die("Not a directory: " + sourceDir.getPath(), false);
211         }
212         File destDir = new File(args[1]);
213 
214         Class<DODeserializer> deserializer = (Class<DODeserializer>) Class.forName(args[2]);
215         Class<DOSerializer> serializer = (Class<DOSerializer>) Class.forName(args[3]);
216 
217         // So DOTranslationUtility works...
218         System.setProperty("fedora.hostname", "localhost");
219         System.setProperty("fedora.port", "8080");
220         System.setProperty("fedora.appServerContext", Constants.FEDORA_DEFAULT_APP_CONTEXT);
221 
222         boolean pretty = args.length > 4 && args[4].equals("true");
223 
224         String inExt = "xml";
225         if (args.length > 5) inExt = args[5];
226 
227         String outExt = "xml";
228         if (args.length > 6) outExt = args[6];
229 
230         ConvertObjectSerialization converter =
231                 new ConvertObjectSerialization(deserializer,
232                                                serializer,
233                                                pretty,
234                                                inExt,
235                                                outExt);
236         converter.convert(sourceDir, destDir);
237 
238         TripleIteratorFactory.defaultInstance().shutdown();
239     }

241     private static <T> T getInstance(Class<T> className) {
242         try {
243             return className.newInstance();
244         } catch (Exception e) {
245             e.printStackTrace();
246             die("Unable to instantiate: " + className, false);
247             return null; // unreachable
248         }
249     }

251     private static void die(String message, boolean showUsage) {
252         System.out.println("ERROR: " + message);
253         if (showUsage) {
254             System.out.println("Usage: ConvertObjectSerialization srcDir dstDir serClass deserClass");
255             System.out.println("                                  [pretty] [inExt] [outExt]");
256             System.out.println("Where: srcDir     : source directory");
257             System.out.println("       dstDir     : destination directory (created if necessary)");
258             System.out.println("       deserClass : DODeserializer class name");
259             System.out.println("       serClass   : DOSerializer class name");
260             System.out.println("       pretty     : if true and output is xml, it will be pretty-printed");
261             System.out.println("       inExt      : extension for input files, default is xml");
262             System.out.println("       outExt     : extension for output files, default is xml");
263         }
264         System.exit(1);
265     }
