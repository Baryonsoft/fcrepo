53  /**
54   * Utility methods for usage by digital object serializers and deserializers.
55   * This class provides methods for detecting various forms of relative
56   * repository URLs, which are URLs that point to the hostname and port of the
57   * local repository. Methods will detect these kinds of URLS in datastream
58   * location fields and in special cases of inline XML. Methods are available to
59   * convert these URLS back and forth from relative URL syntax, to Fedora's
60   * internal local URL syntax, and to absolute URL sytnax. This utility class
61   * defines different "translation contexts" and the format of these relative
62   * URLs will be set appropriately to the context. Currently defined translation
63   * contexts are: 0=Deserialize XML into java object appropriate for in-memory
64   * usage 1=Serialize java object to XML appropriate for "public" export
65   * (absolute URLs) 2=Serialize java object to XML appropriate for move/migrate
66   * to another repository 3=Serialize java object to XML appropriate for internal
67   * storage</b> </p> The public "normalize*" methods in this class should be
68   * called to make the right decisions about what conversions should occur for
69   * what contexts. Other utility methods set default values for datastreams and
70   * disseminators.
71   *
72   * @author Sandy Payette
73   * @version $Id$
74   */
75

83      /**
84       * DESERIALIZE_INSTANCE: Deserialize XML into a java object appropriate for
85       * in-memory usage. This will make the value of relative repository URLs
86       * appropriate for instantiations of the digital object in memory. For
87       * External (E) and Redirected (R) datastreams, any URLs that are relative
88       * to the local repository are converted to absolute URLs using the
89       * currently configured hostname:port of the repository. To do this, the
90       * dsLocation is searched for instances the Fedora local URL string
91       * ("http://local.fedora.server") which is the way Fedora internally keeps
92       * track of instances of relative repository URLs. For Managed Content (M)
93       * datastreams, the internal identifiers are instantiated as is. Also,
94       * certain reserved inline XML datastreams (WSDL and SERVICE_PROFILE) are
95       * searched for relative repository URLs and they are made absolute.
96       */
97      public static final int DESERIALIZE_INSTANCE = 0;
98  
99      /**
100      * SERIALIZE_EXPORT_PUBLIC: Serialize digital object to XML appropriate for
101      * "public" external use. This is context is appropriate when the exporting
102      * repository will continue to exist and will continue to support callback
103      * URLs for datastream content and disseminations. This gives a "public"
104      * export of an object in which all relative repository URLs AND internal
105      * identifiers are converted to absolute callback URLs. For External (E) and
106      * Redirected (R) datastreams, any URLs that are relative to the local
107      * repository are converted to absolute URLs using the currently configured
108      * hostname:port of the repository. For Managed Content (M) datastreams, the
109      * internal identifiers in dsLocation are converted to default dissemination
110      * URLs so they can serve as callbacks to the repository to obtain the
111      * internally managed content. Also, selected inline XML datastreams (i.e.,
112      * WSDL and SERVICE_PROFILE) are searched for relative repository URLs and
113      * they are made absolute.
114      */
115     public static final int SERIALIZE_EXPORT_PUBLIC = 1;
116 
117     /**
118      * SERIALIZE_EXPORT_MIGRATE: Serialize digital object to XML in a manner
119      * appropriate for migrating or moving objects from one repository to
120      * another. This context is appropriate when the local repository will NOT
121      * be available after objects have been migrated to a new repository. For
122      * External (E) and Redirected (R)datastreams, any URLs that are relative to
123      * the local repository will be expressed with the Fedora local URL syntax
124      * (which consists of the string "local.fedora.server" standing in place of
125      * the actual "hostname:port"). This enables a new repository to ingest the
126      * serialization and maintain the relative nature of the URLs (they will
127      * become relative to the *new* repository. Also, for Managed Content (M)
128      * datastreams, the internal identifiers in dsLocation are converted to
129      * default dissemination URLs. This enables the new repository to callback
130      * to the old repository to obtain the content bytestream to be stored in
131      * the new repository. Also, within selected inline XML datastreams (i.e.,
132      * WSDL and SERVICE_PROFILE) any URLs that are relative to the local
133      * repository will also be expressed with the Fedora local URL syntax.
134      */
135     public static final int SERIALIZE_EXPORT_MIGRATE = 2;
136 
137     /**
138      * SERIALIZE_STORAGE_INTERNAL: Serialize java object to XML appropriate for
139      * persistent storage in the repository, ensuring that any URLs that are
140      * relative to the local repository are stored with the Fedora local URL
141      * syntax. The Fedora local URL syntax consists of the string
142      * "local.fedora.server" standing in place of the actual "hostname:port" on
143      * the URL). Managed Content (M) datastreams are stored with internal
144      * identifiers in dsLocation. Also, within selected inline XML datastreams
145      * (i.e., WSDL and SERVICE_PROFILE) any URLs that are relative to the local
146      * repository will also be stored with the Fedora local URL syntax. Note
147      * that a view of the storage serialization can be obtained via the
148      * getObjectXML method of API-M.
149      */
150     public static final int SERIALIZE_STORAGE_INTERNAL = 3;
151 
152     /**
153      * SERIALIZE_EXPORT_ARCHIVE: Serialize digital object to XML in a manner
154      * appropriate for creating a stand alone archive of objects from a
155      * repository that will NOT be available after objects have been exported.
156      * For External (E) and Redirected (R)datastreams, any URLs that are
157      * relative to the local repository will be expressed with the Fedora local
158      * URL syntax (which consists of the string "local.fedora.server" standing
159      * in place of the actual "hostname:port"). This enables a new repository to
160      * ingest the serialization and maintain the relative nature of the URLs
161      * (they will become relative to the *new* repository. Also, for Managed
162      * Content (M) datastreams, the internal identifiers in dsLocation are
163      * converted to default dissemination URLs, and the contents of the URL's
164      * are included inline via base-64 encoding. This enables the new repository
165      * recreate the content bytestream to be stored in the new repository, when
166      * the original repository is no longer available. Also, within selected
167      * inline XML datastreams (i.e., WSDL and SERVICE_PROFILE) any URLs that are
168      * relative to the local repository will also be expressed with the Fedora
169      * local URL syntax.
170      */
171     public static final int SERIALIZE_EXPORT_ARCHIVE = 4;

229         // get host port from system properties (for testing without server instance)
230         String fedoraServerHost = System.getProperty("fedora.hostname");
231         String fedoraServerPort = System.getProperty("fedora.port");
232         String fedoraServerPortSSL = System.getProperty("fedoraRedirectPort");
233         String fedoraAppServerContext =
234                 System.getProperty("fedora.appServerContext");
235 
236         if (fedoraServerPort != null) {
237             if (fedoraServerPort.equals("80")) {
238                 m_serverOnPort80 = true;
239             }
240         }
241         if (fedoraServerPortSSL != null) {
242             if (fedoraServerPortSSL.equals("443")) {
243                 m_serverOnRedirectPort443 = true;
244             }
245         }
246 
247         // otherwise, get host port from the server instance if they are null
248         if (fedoraServerHost == null || fedoraServerPort == null
249                 || fedoraAppServerContext == null) {
250             // if fedoraServerHost or fedoraServerPort system properties
251             // are not defined, read them from server configuration
252             ServerConfiguration config = Server.getConfig();
253             fedoraServerHost =
254                     config.getParameter("fedoraServerHost",Parameter.class).getValue();
255             fedoraServerPort =
256                     config.getParameter("fedoraServerPort",Parameter.class).getValue();
257             fedoraAppServerContext =
258                     config.getParameter("fedoraAppServerContext",Parameter.class).getValue();
259             fedoraServerPortSSL =
260                     config.getParameter("fedoraRedirectPort",Parameter.class).getValue();
261             if (fedoraServerPort.equals("80")) {
262                 m_serverOnPort80 = true;
263             }
264             if (fedoraServerPortSSL.equals("443")) {
265                 m_serverOnRedirectPort443 = true;
266             }
267         }
268         // set the currently configured host:port of the repository
269         s_hostInfo = "http://" + fedoraServerHost;
270         if (!fedoraServerPort.equals("80") && !fedoraServerPort.equals("443")) {
271             s_hostInfo = s_hostInfo + ":" + fedoraServerPort;
272         }
273         s_hostInfo = s_hostInfo + "/";
274         s_hostContextInfo = s_hostInfo + fedoraAppServerContext + "/";
275 
276         // compile the pattern for public dissemination URLs at local server
277         s_localDissemUrlStart = s_hostInfo + fedoraAppServerContext + "/get/";
278 
279         s_concreteLocalUrl =
280             Pattern.compile("https?://(localhost|" + fedoraServerHost
281                             + "):" + fedoraServerPort + "/");
282 
283         s_concreteLocalUrlAppContext =
284             Pattern.compile("https?://(localhost|" + fedoraServerHost
285                             + "):" + fedoraServerPort + "/("
286                             + fedoraAppServerContext + "|fedora)/");
287 
288         s_concreteLocalUrlNoPort =
289             Pattern.compile("https?://(localhost|" + fedoraServerHost
290                             + ")/");
291 
292         s_concreteLocalUrlAppContextNoPort =
293             Pattern.compile("https?://(localhost|" + fedoraServerHost
294                             + ")/(" + fedoraAppServerContext + "|fedora)/");
295     }

297     /**
298      * Make URLs that are relative to the local Fedora repository ABSOLUTE URLs.
299      * First, see if any URLs are expressed in relative URL syntax (beginning
300      * with "fedora/get" or "fedora/search") and convert these to the special
301      * Fedora local URL syntax ("http://local.fedora.server/..."). Then look for
302      * all URLs that contain the special Fedora local URL syntax and replace
303      * instances of this string with the actual host:port configured for the
304      * repository. This ensures that all forms of relative repository URLs are
305      * converted to proper absolute URLs that reference the hostname:port of the
306      * local Fedora repository. Examples:
307      * "http://local.fedora.server/fedora/get/demo:1/DS1" is converted to
308      * "http://myrepo.com:8080/fedora/get/demo:1/DS1" "fedora/get/demo:1/DS1" is
309      * converted to "http://myrepo.com:8080/fedora/get/demo:1/DS1"
310      * "http://local.fedora.server/fedora/get/demo:1/sdef:1/getFoo?in="
311      * http://local.fedora.server/fedora/get/demo:2/DC" is converted to
312      * "http://myrepo.com:8080/fedora/get/demo:1/sdef:1/getFoo?in="
313      * http://myrepo.com:8080/fedora/get/demo:2/DC"
314      *
315      * @param m_xmlContent
316      * @return String with all relative repository URLs and Fedora local URLs
317      *         converted to absolute URL syntax.
318      */
319     public static String makeAbsoluteURLs(String input) {
320         String output = input;
321 
322         // First pass: convert fedora app context URLs via variable substitution
323         output =
324                 s_fedoraLocalAppContextPattern.matcher(output)
325                         .replaceAll(s_hostContextInfo);
326 
327         // Second pass: convert non-fedora-app-context URLs via variable substitution
328         output = s_fedoraLocalPattern.matcher(output).replaceAll(s_hostInfo);
329 
330         logger.debug("makeAbsoluteURLs: input=" + input + ", output=" + output);
331         return output;
332     }

334     /**
335      * Detect all forms of URLs that point to the local Fedora repository and
336      * make sure they are encoded in the special Fedora local URL syntax
337      * (http://local.fedora.server/..."). First, look for relative URLs that
338      * begin with "fedora/get" or "fedora/search" replaces instances of these
339      * string patterns with the special Fedora relative URL syntax. Then, look
340      * for absolute URLs that have a host:port equal to the host:port currently
341      * configured for the Fedora repository and replace host:port with the
342      * special string. The special Fedora relative URL string provides a
343      * consistent unique string be easily searched for and either converted back
344      * to an absolute URL or a relative URL to the repository. Examples:
345      * "http://myrepo.com:8080/fedora/get/demo:1/DS1" is converted to
346      * "http://local.fedora.server/fedora/get/demo:1/DS1"
347      * "https://myrepo.com:8443/fedora/get/demo:1/sdef:1/getFoo?in="
348      * http://myrepo.com:8080/fedora/get/demo:2/DC" is converted to
349      * "http://local.fedora.server/fedora/get/demo:1/sdef:1/getFoo?in="
350      * http://local.fedora.server/fedora/get/demo:2/DC"
351      * "http://myrepo.com:8080/saxon..." (internal service in sDep WSDL) is
352      * converted to "http://local.fedora.server/saxon..."
353      *
354      * @param input
355      * @return String with all forms of relative repository URLs converted to
356      *         the Fedora local URL syntax.
357      */
358     public static String makeFedoraLocalURLs(String input) {
359         String output = input;
360 
361         // Detect any absolute URLs that refer to the local repository
362         // and convert them to the Fedora LOCALIZATION URL syntax
363         // (i.e., "http://local.fedora.server/...")\
364 
365         if (m_serverOnPort80 || m_serverOnRedirectPort443) {
366             output = s_concreteLocalUrlAppContextNoPort.matcher(output)
367                         .replaceAll(s_fedoraLocalAppContextPattern.pattern());
368 
369             output = s_concreteLocalUrlNoPort.matcher(output)
370                         .replaceAll(s_fedoraLocalPattern.pattern());
371         } else {
372             output = s_concreteLocalUrlAppContext.matcher(output)
373                         .replaceAll(s_fedoraLocalAppContextPattern.pattern());
374 
375             output = s_concreteLocalUrl.matcher(output)
376                         .replaceAll(s_fedoraLocalPattern.pattern());
377         }
378 
379         logger.debug("makeFedoraLocalURLs: input=" + input + ", output=" + output);
380         return output;
381     }

404     /*
405      * Utility method to normalize the value of datastream location depending on
406      * the translation context. This is mainly to deal with External (E) and
407      * Redirected (R) datastream locations that are self-referential to the
408      * local repository (i.e., relative repository URLs) and with Managed
409      * Content (M) datastreams whose location is an internal identifier. @param
410      * PID The PID of the object that contains the datastream @param ds The
411      * datastream whose location is to be processed @param transContext Integer
412      * value indicating the serialization or deserialization context. Valid
413      * values are defined as constants in
414      * org.fcrepo.server.storage.translation.DOTranslationUtility:
415      * 0=DOTranslationUtility.DESERIALIZE_INSTANCE
416      * 1=DOTranslationUtility.SERIALIZE_EXPORT_PUBLIC
417      * 2=DOTranslationUtility.SERIALIZE_EXPORT_MIGRATE
418      * 3=DOTranslationUtility.SERIALIZE_STORAGE_INTERNAL
419      * 2=DOTranslationUtility.SERIALIZE_EXPORT_ARCHIVE @return
420      */
421     public static Datastream normalizeDSLocationURLs(String PID,
422                                                      Datastream origDS,
423                                                      int transContext) {
424         Datastream ds = origDS.copy();
425         if (transContext == AS_IS) {
426             return ds;
427         }
428         if (transContext == DOTranslationUtility.DESERIALIZE_INSTANCE) {
429             if (ds.DSControlGrp.equals("E") || ds.DSControlGrp.equals("R")) {
430                 // MAKE ABSOLUTE REPO URLs
431                 ds.DSLocation = makeAbsoluteURLs(ds.DSLocation);
432             }
433         } else if (transContext == DOTranslationUtility.SERIALIZE_EXPORT_PUBLIC) {
434             if (ds.DSControlGrp.equals("E") || ds.DSControlGrp.equals("R")) {
435                 // MAKE ABSOLUTE REPO URLs
436                 ds.DSLocation = makeAbsoluteURLs(ds.DSLocation);
437             } else if (ds.DSControlGrp.equals("M")) {
438                 //if (!ds.DSLocation.startsWith("http://localhost:8080/fedora-demo")) {
439 
440                 // MAKE DISSEMINATION URLs
441                 if (ds.DSCreateDT == null) {
442                     ds.DSLocation =
443                             s_localDissemUrlStart + PID + "/" + ds.DatastreamID;
444                 } else {
445                     ds.DSLocation =
446                             s_localDissemUrlStart
447                                     + PID
448                                     + "/"
449                                     + ds.DatastreamID
450                                     + "/"
451                                     + DateUtility
452                                             .convertDateToString(ds.DSCreateDT);
453                 }
454                 //}
455             }
456         } else if (transContext == DOTranslationUtility.SERIALIZE_EXPORT_MIGRATE) {
457             if (ds.DSControlGrp.equals("E") || ds.DSControlGrp.equals("R")) {
458                 // MAKE FEDORA LOCAL REPO URLs
459                 ds.DSLocation = makeFedoraLocalURLs(ds.DSLocation);
460             } else if (ds.DSControlGrp.equals("M")) {
461                 // MAKE DISSEMINATION URLs
462                 if (ds.DSCreateDT == null) {
463                     ds.DSLocation =
464                             s_localDissemUrlStart + PID + "/" + ds.DatastreamID;
465                 } else {
466                     ds.DSLocation =
467                             s_localDissemUrlStart
468                                     + PID
469                                     + "/"
470                                     + ds.DatastreamID
471                                     + "/"
472                                     + DateUtility
473                                             .convertDateToString(ds.DSCreateDT);
474                 }
475             }
476         } else if (transContext == DOTranslationUtility.SERIALIZE_STORAGE_INTERNAL) {
477             //String relativeLoc=ds.DSLocation;
478             if (ds.DSControlGrp.equals("E") || ds.DSControlGrp.equals("R")) {
479                 // MAKE FEDORA LOCAL REPO URLs
480                 ds.DSLocation = makeFedoraLocalURLs(ds.DSLocation);
481             } else if (ds.DSControlGrp.equals("M")) {
482                 // MAKE INTERNAL IDENTIFIERS (PID+DSID+DSVersionID)
483                 ds.DSLocation =
484                         PID + "+" + ds.DatastreamID + "+" + ds.DSVersionID;
485             }
486         } else if (transContext == DOTranslationUtility.SERIALIZE_EXPORT_ARCHIVE) {
487             if (ds.DSControlGrp.equals("E") || ds.DSControlGrp.equals("R")) {
488                 // MAKE FEDORA LOCAL REPO URLs
489                 ds.DSLocation = makeFedoraLocalURLs(ds.DSLocation);
490             } else if (ds.DSControlGrp.equals("M")) {
491                 // MAKE DISSEMINATION URLs
492                 if (ds.DSCreateDT == null) {
493                     ds.DSLocation =
494                             s_localDissemUrlStart + PID + "/" + ds.DatastreamID;
495                 } else {
496                     ds.DSLocation =
497                             s_localDissemUrlStart
498                                     + PID
499                                     + "/"
500                                     + ds.DatastreamID
501                                     + "/"
502                                     + DateUtility
503                                             .convertDateToString(ds.DSCreateDT);
504                 }
505             }
506         }
507 
508         // In any event, look for the deprecated getItem method of the default disseminator
509         // (i.e., "/fedora-system:3/getItem?itemID=") and convert to new API-A-LITE syntax.
510         if (ds.DSControlGrp.equals("E") || ds.DSControlGrp.equals("R")) {
511             ds.DSLocation = convertGetItemURLs(ds.DSLocation);
512         }
513 
514         return ds;
515     }

517     /**
518      * Utility method to normalize a chunk of inline XML depending on the
519      * translation context. This is mainly to deal with certain inline XML
520      * datastreams found in Service Deployment objects that may contain a
521      * service URL that references the host:port of the local Fedora server.
522      * This method will usually only ever be called to check WSDL and
523      * SERVICE_PROFILE inline XML datastream, but is of general utility for
524      * dealing with any datastreams that may contain URLs that reference the
525      * local Fedora server. However, it this method should be used sparingly,
526      * and only on inline XML datastreams where the impact of the conversions is
527      * well understood.
528      *
529      * @param xml
530      *        a chunk of XML that's contents of an inline XML datastream
531      * @param transContext
532      *        Integer value indicating the serialization or deserialization
533      *        context. Valid values are defined as constants in
534      *        org.fcrepo.server.storage.translation.DOTranslationUtility:
535      *        0=DOTranslationUtility.DESERIALIZE_INSTANCE
536      *        1=DOTranslationUtility.SERIALIZE_EXPORT_PUBLIC
537      *        2=DOTranslationUtility.SERIALIZE_EXPORT_MIGRATE
538      *        3=DOTranslationUtility.SERIALIZE_STORAGE_INTERNAL
539      *        4=DOTranslationUtility.SERIALIZE_EXPORT_ARCHIVE
540      * @return the inline XML contents with appropriate conversions.

636     /**
637      * Appends XML to a PrintWriter. Essentially, just appends all text content
638      * of the inputStream, trimming any leading and trailing whitespace. It does
639      * his in a streaming fashion, with resource consumption entirely comprised
640      * of fixed internal buffers.
641      *
642      * @param in
643      *        InputStreaming containing serialized XML.
644      * @param writer
645      *        PrintWriter to write XML content to.
646      * @param encoding
647      *        Character set encoding.
648      */
649     protected static void appendXMLStream(InputStream in,
650                                           PrintWriter writer,
651                                           String encoding)
652             throws ObjectIntegrityException, UnsupportedEncodingException,
653             StreamIOException {
654         if (in == null) {
655             throw new ObjectIntegrityException("Object's inline xml "
656                     + "stream cannot be null.");
657         }
658         try {
659             InputStreamReader chars =
660                     new InputStreamReader(in, Charset.forName(encoding));
661 
662             /* Content buffer */
663             char[] charBuf = new char[4096];
664 
665             /* Beginning/ending whitespace buffer */
666             char[] wsBuf = new char[4096];
667 
668             int len;
669             int start;
670             int end;
671             int wsLen = 0;
672             boolean atBeginning = true;
673             while ((len = chars.read(charBuf)) != -1) {
674                 start = 0;
675                 end = len - 1;
676 
677                 /* Strip out any leading whitespace */
678                 if (atBeginning) {
679                     while (start < len) {
680                         if (charBuf[start] > 0x20) break;
681                         start++;
682                     }
683                     if (start < len) atBeginning = false;
684                 }
685 
686                 /*
687                  * Hold aside any whitespace at the end of the current chunk. If
688                  * we make it to the next chunk, then append our whitespace to
689                  * the buffer. Using this methodology, we may "trim" at most
690                  * {buffer length} characters from the end.
691                  */
692 
693                 if (wsLen > 0) {
694                     /* Commit previous ending whitespace */
695                     writer.write(wsBuf, 0, wsLen);
696                     wsLen = 0;
697                 }
698 
699                 while (end > start) {
700                     /* Buffer current ending whitespace */
701                     if (charBuf[end] > 0x20) break;
702                     wsBuf[wsLen] = charBuf[end];
703                     wsLen++;
704                     end--;
705                 }
706 
707                 if (start < len) {
708                     writer.write(charBuf, start, end + 1 - start);
709                 }
710             }
711         } catch (UnsupportedEncodingException uee) {
712             throw uee;
713         } catch (IOException ioe) {
714             throw new StreamIOException("Error reading from inline xml datastream.");
715         } finally {
716             try {
717                 in.close();
718             } catch (IOException closeProb) {
719                 throw new StreamIOException("Error closing read stream.");
720             }
721         }
722     }
