39      protected SortTableModel makeModel() {
40          Vector data = new Vector();
41          for (int i = 0; i < 25; i++) {
42              Vector row = new Vector();
43              for (int j = 0; j < 5; j++) {
44                  row.add(new Integer((int) (Math.random() * 256)));
45              }
46              data.add(row);
47          }
48  
49          Vector names = new Vector();
50          names.add("One");
51          names.add("Two");
52          names.add("Three");
53          names.add("Four");
54          names.add("Five");
55  
56          return new DefaultSortTableModel(data, names);
57      }
