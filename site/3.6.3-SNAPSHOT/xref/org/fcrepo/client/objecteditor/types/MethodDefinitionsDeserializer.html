7   import java.io.IOException;
8   import java.io.InputStream;
9   
10  import java.util.ArrayList;
11  import java.util.List;
12  
13  import javax.xml.parsers.SAXParser;
14  import javax.xml.parsers.SAXParserFactory;
15  
16  import org.xml.sax.Attributes;
17  import org.xml.sax.SAXException;
18  import org.xml.sax.helpers.DefaultHandler;
19  
20  /**
21   * Turns xml for method defs into java object instances.
22   */
23  class MethodDefinitionsDeserializer
24          extends DefaultHandler {
25  
26      /** The deserialized MethodDefinition objects */
27      private final List m_result;
28  
29      // values gathered and built while parsing
30      private String m_methodName;
31  
32      private String m_methodLabel;
33  
34      private List m_methodParms;
35  
36      private String m_parmName;
37  
38      private String m_parmLabel;
39  
40      private String m_parmRequired;
41  
42      private String m_parmDefaultValue;
43  
44      private List m_parmValidValues;
45  
46      public MethodDefinitionsDeserializer(InputStream xml)
47              throws IOException {
48          m_result = new ArrayList();
49          try {
50              SAXParserFactory spf = SAXParserFactory.newInstance();
51              spf.setNamespaceAware(true);
52              SAXParser parser = spf.newSAXParser();
53              parser.parse(xml, this);
54          } catch (Exception e) {
55              if (e.getMessage() != null) {
56                  throw new IOException(e.getMessage());
57              } else {
58                  throw new IOException("Error parsing method definitions: "
59                          + e.getClass().getName());
60              }
61          }
62      }
63  
64      public List getResult() {
65          return m_result;
66      }
67  
68      @Override
69      public void startElement(String uri,
70                               String localName,
71                               String qName,
72                               Attributes a) {
73          if (localName.equals("Method")) {
74              m_methodName = null;
75              m_methodLabel = null;
76              m_methodParms = new ArrayList();
77              for (int i = 0; i < a.getLength(); i++) {
78                  String name = a.getLocalName(i);
79                  if (name.equals("operationName")) {
80                      m_methodName = a.getValue(i);
81                  } else if (name.equals("label")) {
82                      m_methodLabel = a.getValue(i);
83                  }
84              }
85          } else if (localName.equals("UserInputParm")) {
86              m_parmName = null;
87              m_parmLabel = null;
88              m_parmRequired = null;
89              m_parmDefaultValue = null;
90              m_parmValidValues = new ArrayList();
91              for (int i = 0; i < a.getLength(); i++) {
92                  String name = a.getLocalName(i);
93                  if (name.equals("parmName")) {
94                      m_parmName = a.getValue(i);
95                  } else if (name.equals("label")) {
96                      m_parmLabel = a.getValue(i);
97                  } else if (name.equals("required")) {
98                      m_parmRequired = a.getValue(i);
99                  } else if (name.equals("defaultValue")) {
100                     m_parmDefaultValue = a.getValue(i);
101                 }
102             }
103         } else if (localName.equals("ValidParm")) {
104             for (int i = 0; i < a.getLength(); i++) {
105                 String name = a.getLocalName(i);
106                 if (name.equals("value")) {
107                     m_parmValidValues.add(a.getValue(i));
108                 }
109             }
110         }
111     }
112 
113     @Override
114     public void endElement(String uri, String localName, String qName)
115             throws SAXException {
116         if (localName.equals("UserInputParm")) {
117             if (m_parmName == null || m_parmName.length() == 0) {
118                 throw new SAXException("UserInputParm must have a name "
119                         + "attribute with length>1.");
120             }
121             boolean parmRequired = false;
122             if (m_parmRequired != null) {
123                 if (m_parmRequired.equalsIgnoreCase("true")
124                         || m_parmRequired.equalsIgnoreCase("yes")
125                         || m_parmRequired.equals("1")) {
126                     parmRequired = true;
127                 }
128             }
129             m_methodParms.add(new ParameterDefinition(m_parmName,
130                                                       m_parmLabel,
131                                                       parmRequired,
132                                                       m_parmDefaultValue,
133                                                       m_parmValidValues));
134         } else if (localName.equals("Method")) {
135             if (m_methodName == null || m_methodName.length() == 0) {
136                 throw new SAXException("Method must have a name attribute "
137                         + "with length>1.");
138             }
139             m_result.add(new MethodDefinition(m_methodName,
140                                               m_methodLabel,
141                                               m_methodParms));
142         }
143     }
