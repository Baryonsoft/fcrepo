31      /**
32       * Aligns the first <code>rows</code> * <code>cols</code> components of
33       * <code>parent</code> in a grid. Each component is as big as the maximum
34       * preferred width and height of the components. The parent is made just big
35       * enough to fit them all.
36       *
37       * @param rows
38       *        number of rows
39       * @param cols
40       *        number of columns
41       * @param initialX
42       *        x location to start the grid at
43       * @param initialY
44       *        y location to start the grid at
45       * @param xPad
46       *        x padding between cells
47       * @param yPad
48       *        y padding between cells
49       */
50      public static void makeGrid(Container parent,
51                                  int rows,
52                                  int cols,
53                                  int initialX,
54                                  int initialY,
55                                  int xPad,
56                                  int yPad) {
57          SpringLayout layout;
58          try {
59              layout = (SpringLayout) parent.getLayout();
60          } catch (ClassCastException exc) {
61              System.err
62                      .println("The first argument to makeGrid must use SpringLayout.");
63              return;
64          }
65  
66          Spring xPadSpring = Spring.constant(xPad);
67          Spring yPadSpring = Spring.constant(yPad);
68          Spring initialXSpring = Spring.constant(initialX);
69          Spring initialYSpring = Spring.constant(initialY);
70          int max = rows * cols;
71  
72          //Calculate Springs that are the max of the width/height so that all
73          //cells have the same size.
74          Spring maxWidthSpring =
75                  layout.getConstraints(parent.getComponent(0)).getWidth();
76          Spring maxHeightSpring =
77                  layout.getConstraints(parent.getComponent(0)).getWidth();
78          for (int i = 1; i < max; i++) {
79              SpringLayout.Constraints cons =
80                      layout.getConstraints(parent.getComponent(i));
81  
82              maxWidthSpring = Spring.max(maxWidthSpring, cons.getWidth());
83              maxHeightSpring = Spring.max(maxHeightSpring, cons.getHeight());
84          }
85  
86          //Apply the new width/height Spring. This forces all the
87          //components to have the same size.
88          for (int i = 0; i < max; i++) {
89              SpringLayout.Constraints cons =
90                      layout.getConstraints(parent.getComponent(i));
91  
92              cons.setWidth(maxWidthSpring);
93              cons.setHeight(maxHeightSpring);
94          }
95  
96          //Then adjust the x/y constraints of all the cells so that they
97          //are aligned in a grid.
98          SpringLayout.Constraints lastCons = null;
99          SpringLayout.Constraints lastRowCons = null;
100         for (int i = 0; i < max; i++) {
101             SpringLayout.Constraints cons =
102                     layout.getConstraints(parent.getComponent(i));
103             if (i % cols == 0) { //start of new row
104                 lastRowCons = lastCons;
105                 cons.setX(initialXSpring);
106             } else { //x position depends on previous component
107                 cons.setX(Spring.sum(lastCons.getConstraint(SpringLayout.EAST),
108                                      xPadSpring));
109             }
110 
111             if (i / cols == 0) { //first row
112                 cons.setY(initialYSpring);
113             } else { //y position depends on previous row
114                 cons.setY(Spring.sum(lastRowCons
115                         .getConstraint(SpringLayout.SOUTH), yPadSpring));
116             }
117             lastCons = cons;
118         }
119 
120         //Set the parent's size.
121         SpringLayout.Constraints pCons = layout.getConstraints(parent);
122         pCons.setConstraint(SpringLayout.SOUTH, Spring.sum(Spring
123                 .constant(yPad), lastCons.getConstraint(SpringLayout.SOUTH)));
124         pCons.setConstraint(SpringLayout.EAST, Spring
125                 .sum(Spring.constant(xPad), lastCons
126                         .getConstraint(SpringLayout.EAST)));
127     }

129     /* Used by makeCompactGrid. */
130     private static SpringLayout.Constraints getConstraintsForCell(int row,
131                                                                   int col,
132                                                                   Container parent,
133                                                                   int cols) {
134         SpringLayout layout = (SpringLayout) parent.getLayout();
135         Component c = parent.getComponent(row * cols + col);
136         return layout.getConstraints(c);
137     }

139     /**
140      * Aligns the first <code>rows</code> * <code>cols</code> components of
141      * <code>parent</code> in a grid. Each component in a column is as wide as
142      * the maximum preferred width of the components in that column; height is
143      * similarly determined for each row. The parent is made just big enough to
144      * fit them all.
145      *
146      * @param rows
147      *        number of rows
148      * @param cols
149      *        number of columns
150      * @param initialX
151      *        x location to start the grid at
152      * @param initialY
153      *        y location to start the grid at
154      * @param xPad
155      *        x padding between cells
156      * @param yPad
157      *        y padding between cells
158      */
159     public static void makeCompactGrid(Container parent,
160                                        int rows,
161                                        int cols,
162                                        int initialX,
163                                        int initialY,
164                                        int xPad,
165                                        int yPad) {
166         SpringLayout layout;
167         try {
168             layout = (SpringLayout) parent.getLayout();
169         } catch (ClassCastException exc) {
170             System.err
171                     .println("The first argument to makeCompactGrid must use SpringLayout.");
172             return;
173         }
174 
175         //Align all cells in each column and make them the same width.
176         Spring x = Spring.constant(initialX);
177         for (int c = 0; c < cols; c++) {
178             Spring width = Spring.constant(0);
179             for (int r = 0; r < rows; r++) {
180                 width =
181                         Spring.max(width, getConstraintsForCell(r,
182                                                                 c,
183                                                                 parent,
184                                                                 cols)
185                                 .getWidth());
186             }
187             for (int r = 0; r < rows; r++) {
188                 SpringLayout.Constraints constraints =
189                         getConstraintsForCell(r, c, parent, cols);
190                 constraints.setX(x);
191                 constraints.setWidth(width);
192             }
193             x = Spring.sum(x, Spring.sum(width, Spring.constant(xPad)));
194         }
195 
196         //Align all cells in each row and make them the same height.
197         Spring y = Spring.constant(initialY);
198         for (int r = 0; r < rows; r++) {
199             Spring height = Spring.constant(0);
200             for (int c = 0; c < cols; c++) {
201                 height =
202                         Spring.max(height, getConstraintsForCell(r,
203                                                                  c,
204                                                                  parent,
205                                                                  cols)
206                                 .getHeight());
207             }
208             for (int c = 0; c < cols; c++) {
209                 SpringLayout.Constraints constraints =
210                         getConstraintsForCell(r, c, parent, cols);
211                 constraints.setY(y);
212                 constraints.setHeight(height);
213             }
214             y = Spring.sum(y, Spring.sum(height, Spring.constant(yPad)));
215         }
216 
217         //Set the parent's size.
218         SpringLayout.Constraints pCons = layout.getConstraints(parent);
219         pCons.setConstraint(SpringLayout.SOUTH, y);
220         pCons.setConstraint(SpringLayout.EAST, x);
221     }
