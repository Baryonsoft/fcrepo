92      @Override
93      public void setContent(InputStream data) throws IOException {
94          // get a string from the inputstream, assume it's UTF-8
95          String content;
96          if (m_xml) {
97              try {
98                  // use xerces to pretty print the xml to the editor
99                  OutputFormat fmt = new OutputFormat("XML", "UTF-8", true);
100                 fmt.setOmitXMLDeclaration(true);
101                 fmt.setIndent(2);
102                 fmt.setLineWidth(120);
103                 fmt.setPreserveSpace(false);
104                 ByteArrayOutputStream buf = new ByteArrayOutputStream();
105                 XMLSerializer ser = new XMLSerializer(buf, fmt);
106                 DocumentBuilderFactory factory =
107                         DocumentBuilderFactory.newInstance();
108                 factory.setNamespaceAware(true);
109                 DocumentBuilder builder = factory.newDocumentBuilder();
110                 Document doc = builder.parse(data);
111                 ser.serialize(doc);
112                 content = new String(buf.toByteArray(), "UTF-8");
113             } catch (Exception e) {
114                 throw new IOException("Error parsing as XML: " + e.getMessage());
115             }
116         } else {
117             StringBuffer out = new StringBuffer();
118             BufferedReader in = new BufferedReader(new InputStreamReader(data));
119             String thisLine;
120             while ((thisLine = in.readLine()) != null) {
121                 out.append(thisLine + "\n");
122             }
123             in.close();
124             content = out.toString();
125         }
126         m_editor.setText(content);
127         m_editor.setCaretPosition(0);
128         // if this is being called upon construction, set it to the original
129         // content.  else, let it be different.
130         if (m_origContent == null) {
131             m_origContent = content;
132         }
133     }

164     @Override
165     public InputStream getContent() throws IOException {
166         try {
167             if (m_xml) {
168                 // if it's xml, throw an exception if it's not well-formed
169                 DocumentBuilderFactory factory =
170                         DocumentBuilderFactory.newInstance();
171                 DocumentBuilder builder = factory.newDocumentBuilder();
172                 Document doc =
173                         builder.parse(new InputSource(new StringReader(m_editor
174                                 .getText())));
175             }
176             return new ByteArrayInputStream(m_editor.getText()
177                     .getBytes("UTF-8"));
178         } catch (Exception e) {
179             throw new IOException(e.getMessage());
180         }
181     }
