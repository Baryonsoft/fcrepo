35      public ConsoleCommandInvoker(ConsoleCommand command, Console console) {
36          m_command = command;
37          m_console = console;
38          setLayout(new BoxLayout(this, BoxLayout.Y_AXIS));
39          JLabel commandNameLabel = new JLabel("Command: " + m_command.getName());
40          JPanel jeez = new JPanel();
41          jeez.setLayout(new BorderLayout());
42          jeez.add(commandNameLabel, BorderLayout.WEST);
43          add(jeez);
44          Class types[] = command.getParameterTypes();
45          String names[] = command.getParameterNames();
46          m_inputPanels = new InputPanel[types.length];
47          for (int i = 0; i < types.length; i++) {
48              JPanel typeNameInputPanel = new JPanel();
49              typeNameInputPanel.setLayout(new BorderLayout());
50              typeNameInputPanel
51                      .add(new JLabel(names[i] + " ("
52                                   + command.getUnqualifiedName(types[i])
53                                   + ") : "),
54                           BorderLayout.WEST);
55              m_inputPanels[i] = InputPanelFactory.getPanel(types[i]);
56              typeNameInputPanel.add(m_inputPanels[i]);
57              add(typeNameInputPanel);
58          }
59          JLabel returnTypeLabel =
60                  new JLabel("Returns: "
61                          + m_command.getUnqualifiedName(m_command
62                                  .getReturnType()));
63          JPanel jeez2 = new JPanel();
64          jeez2.setLayout(new BorderLayout());
65          jeez2.add(returnTypeLabel, BorderLayout.WEST);
66          add(jeez2);
67      }

69      /**
70       * Invokes the console command with whatever parameters have been set thus
71       * far, sending any errors to the console.
72       */
73      public void invoke() {
74          try {
75              m_console.setBusy(true);
76              m_console.print("Invoking " + m_command.toString() + "\n");
77              Object[] parameters =
78                      new Object[m_command.getParameterTypes().length];
79              if (m_command.getParameterTypes().length > 0) {
80                  for (int i = 0; i < m_command.getParameterTypes().length; i++) {
81                      m_console.print(m_command.getParameterNames()[i]);
82                      m_console.print("=");
83                      Object paramValue = m_inputPanels[i].getValue();
84                      parameters[i] = paramValue;
85                      if (paramValue == null) {
86                          m_console.print("<null>");
87                      } else {
88                          m_console.print(stringify(paramValue));
89                      }
90                      m_console.print("\n");
91                  }
92              }
93              long startms = new Date().getTime();
94              Object returned =
95                      m_command.invoke(m_console.getInvocationTarget(m_command),
96                                       parameters);
97              long endms = new Date().getTime();
98              long totalms = endms - startms;
99              if (returned != null) {
100                 m_console.print("Returned: " + stringify(returned) + "\n");
101             } else {
102                 if (m_command.getReturnType() == null) {
103                     m_console.print("Returned.\n");
104                 } else {
105                     m_console.print("Returned: <null>\n");
106                 }
107             }
108             String duration;
109             if (totalms == 0) {
110                 duration = "< 0.001 seconds.";
111             } else {
112                 double secs = totalms / 1000.0;
113                 duration = secs + " seconds.";
114             }
115             m_console.print("Roundtrip time: ");
116             m_console.print(duration);
117             m_console.print("\n");
118         } catch (InvocationTargetException ite) {
119             m_console.print("ERROR ("
120                     + ite.getTargetException().getClass().getName() + ") : "
121                     + ite.getTargetException().getMessage() + "\n");
122         } catch (Throwable th) {
123             m_console.print("ERROR (" + th.getClass().getName() + ") : "
124                     + th.getMessage() + "\n");
125             StringWriter sw = new StringWriter();
126             th.printStackTrace(new PrintWriter(sw));
127             m_console.print(sw.toString());
128         } finally {
129             m_console.setBusy(false);
130         }
131     }

133     private String stringify(Object obj) {
134         if (obj == null) {
135             return "<null>";
136         }
137         String nm = obj.getClass().getName();
138         if (nm.startsWith("[L")) {
139             // print array
140             StringBuffer buf = new StringBuffer();
141             buf.append("{");
142             for (int i = 0; i < Array.getLength(obj); i++) {
143                 if (i > 0) {
144                     buf.append(",");
145                 }
146                 buf.append(Array.get(obj, i).toString());
147             }
148             buf.append("}");
149             return buf.toString();
150         }
151         return obj.toString();
152     }
