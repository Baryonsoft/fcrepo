158     public void init(String host, String port, String context, String username, String password) throws Exception {
159         this.host = host;
160         this.port = port;
161         this.username = username;
162         this.password = password;
163         this.context = context;
164 
165         String baseURL = "http://" + host + ":" + port + "/" + context;
166         FedoraClient fedoraClient = new FedoraClient(baseURL, username, password);
167         apim = fedoraClient.getAPIMMTOM();
168         apia = fedoraClient.getAPIAMTOM();
169 
170         PIDS = apim.getNextPID(new BigInteger(Integer.valueOf(iterations).toString()), "demo").toArray(new String[]{});
171         FOXML = new byte[iterations][];
172         for (int i = 0; i < iterations; i++) {
173             FOXML[i] = DEMO_FOXML_TEXT.replaceAll(pid, PIDS[i]).getBytes("UTF-8");
174         }
175     }

408     private HttpMethod getHttpMethod(String pid) {
409         String url = "http://" + host + ":" + port + "/" + context + "/get/" + pid + "/" + "MDS1";
410         HttpMethod httpMethod = new GetMethod(url);
411         httpMethod.setDoAuthentication(true);
412         httpMethod.getParams().setParameter("Connection", "Keep-Alive");
413         return httpMethod;
414     }

416     private HttpClient getHttpClient() {
417         HttpClient client = new HttpClient();
418         client.getParams().setAuthenticationPreemptive(true);
419         client.getState().setCredentials(
420                 new AuthScope(host, Integer.valueOf(port), "realm"),
421                 new UsernamePasswordCredentials(username, password));
422         return client;
423     }

425     /**
426      * @return total time to run all iterations
427      *         {ingest, addDatastream, modifyDatastreamByReference,
428      *         modifyDatastreamByValue, purgeDatastream, purgeObject}
429      */
430     public long[] runThroughputTests() throws Exception {
431         long ingestTime = 0;
432         long addDsTime = 0;
433         long modifyRefTime = 0;
434         long modifyValTime = 0;
435         long purgeDsTime = 0;
436         long purgeObjectTime = 0;
437         long startTime = 0;
438         long stopTime = 0;
439 
440         // Ingest
441         startTime = System.currentTimeMillis();
442         for (int i = 0; i < iterations; i++) {
443             runIngest(FOXML[i]);
444         }
445         stopTime = System.currentTimeMillis();
446         ingestTime = (stopTime - startTime);
447 
448         // Add Datastream
449         startTime = System.currentTimeMillis();
450         for (int i = 0; i < iterations; i++) {
451             runAddDatastream(PIDS[i], "MDS3");
452         }
453         stopTime = System.currentTimeMillis();
454         addDsTime = (stopTime - startTime);
455 
456         // Modify Datastream By Reference
457         startTime = System.currentTimeMillis();
458         for (int i = 0; i < iterations; i++) {
459             runModifyDatastreamByRef(PIDS[i]);
460         }
461         stopTime = System.currentTimeMillis();
462         modifyRefTime = (stopTime - startTime);
463 
464         // Modify Datastream By Value
465         startTime = System.currentTimeMillis();
466         for (int i = 0; i < iterations; i++) {
467             runModifyDatastreamByValue(PIDS[i]);
468         }
469         stopTime = System.currentTimeMillis();
470         modifyValTime = (stopTime - startTime);
471 
472         // Purge Datastream
473         startTime = System.currentTimeMillis();
474         for (int i = 0; i < iterations; i++) {
475             runPurgeDatastream(PIDS[i], "MDS1");
476         }
477         stopTime = System.currentTimeMillis();
478         purgeDsTime = (stopTime - startTime);
479 
480         // Purge Object
481         startTime = System.currentTimeMillis();
482         for (int i = 0; i < iterations; i++) {
483             runPurgeObject(PIDS[i]);
484         }
485         stopTime = System.currentTimeMillis();
486         purgeObjectTime = (stopTime - startTime);
487 
488         // Get Datastream and Get Datastream REST results
489         // do not change from round-trip tests
490 
491         long[] totals = {ingestTime,
492                          addDsTime,
493                          modifyRefTime,
494                          modifyValTime,
495                          purgeDsTime,
496                          purgeObjectTime};
497         return totals;
498     }

501      * @return total time to run all iterations with threading
502      *         {ingest, addDatastream, modifyDatastreamByReference,
503      *         modifyDatastreamByValue, purgeDatastream, purgeObject,
504      *         getDatastream, getDatastreamREST}
505      */
506     public long[] runThreadedThroughputTests() throws Exception {
507         long ingestTime = 0;
508         long addDsTime = 0;
509         long modifyRefTime = 0;
510         long modifyValTime = 0;
511         long purgeDsTime = 0;
512         long purgeObjectTime = 0;
513         long getDatastreamTime = 0;
514         long getDatastreamRestTime = 0;
515         long startTime = 0;
516         long stopTime = 0;
517 
518         ExecutorService threadPool = Executors.newFixedThreadPool(threads);
519 
520         // Ingest
521         ArrayList<Callable<Boolean>> ingestRunnerList = new ArrayList<Callable<Boolean>>();
522         for (int i = 0; i < iterations; i++) {
523             ingestRunnerList.add(new MethodRunner(MethodType.INGEST, i));
524         }
525         startTime = System.currentTimeMillis();
526         threadPool.invokeAll(ingestRunnerList);
527         stopTime = System.currentTimeMillis();
528         ingestTime = (stopTime - startTime);
529 
530         // Add Datastream
531         ArrayList<Callable<Boolean>> addDatastreamRunnerList = new ArrayList<Callable<Boolean>>();
532         for (int i = 0; i < iterations; i++) {
533             addDatastreamRunnerList.add(new MethodRunner(MethodType.ADD_DATASTREAM, i));
534         }
535         startTime = System.currentTimeMillis();
536         threadPool.invokeAll(addDatastreamRunnerList);
537         stopTime = System.currentTimeMillis();
538         addDsTime = (stopTime - startTime);
539 
540         // Modify Datastream By Reference
541         ArrayList<Callable<Boolean>> modDatastreamRefRunnerList = new ArrayList<Callable<Boolean>>();
542         for (int i = 0; i < iterations; i++) {
543             modDatastreamRefRunnerList.add(new MethodRunner(MethodType.MODIFY_DATASTREAM_REF, i));
544         }
545         startTime = System.currentTimeMillis();
546         threadPool.invokeAll(modDatastreamRefRunnerList);
547         stopTime = System.currentTimeMillis();
548         modifyRefTime = (stopTime - startTime);
549 
550         // Modify Datastream By Value
551         ArrayList<Callable<Boolean>> modDatastreamValRunnerList = new ArrayList<Callable<Boolean>>();
552         for (int i = 0; i < iterations; i++) {
553             modDatastreamValRunnerList.add(new MethodRunner(MethodType.MODIFY_DATASTREAM_VAL, i));
554         }
555         startTime = System.currentTimeMillis();
556         threadPool.invokeAll(modDatastreamValRunnerList);
557         stopTime = System.currentTimeMillis();
558         modifyValTime = (stopTime - startTime);
559 
560         // Purge Datastream
561         ArrayList<Callable<Boolean>> purgeDatastreamRunnerList = new ArrayList<Callable<Boolean>>();
562         for (int i = 0; i < iterations; i++) {
563             purgeDatastreamRunnerList.add(new MethodRunner(MethodType.PURGE_DATASTREAM, i));
564         }
565         startTime = System.currentTimeMillis();
566         threadPool.invokeAll(purgeDatastreamRunnerList);
567         stopTime = System.currentTimeMillis();
568         purgeDsTime = (stopTime - startTime);
569 
570         // Purge Object
571         ArrayList<Callable<Boolean>> purgeObjectRunnerList = new ArrayList<Callable<Boolean>>();
572         for (int i = 0; i < iterations; i++) {
573             purgeObjectRunnerList.add(new MethodRunner(MethodType.PURGE_OBJECT, i));
574         }
575         startTime = System.currentTimeMillis();
576         threadPool.invokeAll(purgeObjectRunnerList);
577         stopTime = System.currentTimeMillis();
578         purgeObjectTime = (stopTime - startTime);
579 
580         // Get Datastream
581         ArrayList<Callable<Boolean>> getDatastreamRunnerList = new ArrayList<Callable<Boolean>>();
582         for (int i = 0; i < iterations; i++) {
583             getDatastreamRunnerList.add(new MethodRunner(MethodType.GET_DATASTREAM, i));
584         }
585         startTime = System.currentTimeMillis();
586         threadPool.invokeAll(getDatastreamRunnerList);
587         stopTime = System.currentTimeMillis();
588         getDatastreamTime = (stopTime - startTime);

591         ArrayList<Callable<Boolean>> getDatastreamRestRunnerList = new ArrayList<Callable<Boolean>>();
592         for (int i = 0; i < iterations; i++) {

594         }
595         startTime = System.currentTimeMillis();
596         threadPool.invokeAll(getDatastreamRestRunnerList);
597         stopTime = System.currentTimeMillis();
598         getDatastreamRestTime = (stopTime - startTime);

600         long[] totals = {ingestTime,

604                          purgeDsTime,
605                          purgeObjectTime,
606                          getDatastreamTime,

623      * Runnable class to use within threads. Executes a
624      * Fedora method based on the MethodType defined.

627     private class MethodRunner implements Callable<Boolean> {
628 
629         MethodType methodType;
630         int index;
631 
632         public MethodRunner(MethodType methodType, int index) {
633             this.methodType = methodType;
634             this.index = index;
635         }
636 
637         @Override
638         public Boolean call() throws Exception {
639             if (methodType.equals(MethodType.INGEST)) {
640                 runIngest(FOXML[index]);
641             } else if (methodType.equals(MethodType.ADD_DATASTREAM)) {
642                 runAddDatastream(PIDS[index], "MDS3");
643             } else if (methodType.equals(MethodType.MODIFY_DATASTREAM_REF)) {
644                 runModifyDatastreamByRef(PIDS[index]);
645             } else if (methodType.equals(MethodType.MODIFY_DATASTREAM_VAL)) {
646                 runModifyDatastreamByValue(PIDS[index]);
647             } else if (methodType.equals(MethodType.PURGE_DATASTREAM)) {
648                 runPurgeDatastream(PIDS[index], "MDS1");
649             } else if (methodType.equals(MethodType.PURGE_OBJECT)) {
650                 runPurgeObject(PIDS[index]);
651             } else if (methodType.equals(MethodType.GET_DATASTREAM)) {
652                 runGetDatastream(PIDS[index]);
653             } else if (methodType.equals(MethodType.GET_DATASTREAM_REST)) {
654                 runGetDatastreamRest(PIDS[index]);
655             }
656             return new Boolean(true);
657         }
658     }

662         BigDecimal bd = new BigDecimal(Double.toString(d));

664         return bd.doubleValue();
