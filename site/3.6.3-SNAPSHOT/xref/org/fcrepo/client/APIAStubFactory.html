39      public static FedoraAPIA getStub(String protocol,
40                                           String host,
41                                           int port,
42                                           String username,
43                                           String password)
44              throws MalformedURLException, ServiceException {
45          if (!"http".equalsIgnoreCase(protocol)
46                  && !"https".equalsIgnoreCase(protocol)) {
47              throw new javax.xml.rpc.ServiceException("The protocol" + " "
48                      + protocol + " is not supported by this service.");
49          }
50  
51          Map<String, Object> props = new HashMap<String, Object>();
52          props.put("mtom-enabled", Boolean.FALSE);
53  
54          org.apache.cxf.jaxws.JaxWsProxyFactoryBean clientFactory =
55                  new org.apache.cxf.jaxws.JaxWsProxyFactoryBean();
56          clientFactory.setAddress(protocol + "://" + host + ":" + port
57                  + "/fedora/services/access");
58          clientFactory.setServiceClass(FedoraAPIA.class);
59          clientFactory.setUsername(username);
60          clientFactory.setPassword(password);
61          clientFactory.setProperties(props);
62  //        LoggingInInterceptor log1 = new LoggingInInterceptor(new PrintWriter(System.out));
63  //        LoggingOutInterceptor log2 = new LoggingOutInterceptor(new PrintWriter(System.out));
64  //        clientFactory.getInInterceptors().add(log1);
65  //        clientFactory.getInInterceptors().add(log2);
66          PrintStream syserr = System.err;
67          System.setErr(System.out);
68          FedoraAPIA service = (FedoraAPIA) clientFactory.create();
69          System.setErr(syserr);
70          syserr = null;
71  
72          if (Administrator.INSTANCE == null) {
73              // if running without Administrator, don't wrap it with the statusbar stuff
74              return service;
75          } else {
76              return new APIAStubWrapper(service);
77          }
78      }
