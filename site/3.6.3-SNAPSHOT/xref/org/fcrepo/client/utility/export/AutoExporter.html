61      public static void export(FedoraAPIAMTOM apia,
62                                FedoraAPIMMTOM apim,
63                                String pid,
64                                String format,
65                                String exportContext,
66                                OutputStream outStream) throws RemoteException,
67              IOException {
68  
69          // Get the repository info from the hash for the repository that APIA is using,
70          // unless it isn't in the hash yet... in which case, ask the server.
71          RepositoryInfo repoinfo = s_repoInfo.get(apia);
72          if (repoinfo == null) {
73              repoinfo = apia.describeRepository();
74              s_repoInfo.put(apia, repoinfo);
75          }
76  
77          byte[] bytes;
78          // For backward compatibility:
79          // Pre-2.0 repositories will only export "metslikefedora1" format.
80          // 2.x repositories will only export "metslikefedora1" and "foxml1.0" formats.
81          // For 3.0+ repositories the format arg is sent to "export" method.
82          StringTokenizer stoken =
83                  new StringTokenizer(repoinfo.getRepositoryVersion(), ".");
84          int majorVersion = new Integer(stoken.nextToken()).intValue();
85          if (majorVersion < 2) {
86              if (format == null || format.equals(METS_EXT1_1.uri)
87                      || format.equals(METS_EXT1_0.uri)
88                      || format.equals("METS_EXT1_0_LEGACY")
89                      || format.equals("default")) {
90                  if (format.equals(METS_EXT1_1.uri)) {
91                      System.out.println("WARNING: Repository does not support METS Fedora " +
92                                         "Extension 1.1; exporting older format (v1.0) instead");
93                  }
94                  bytes = TypeUtility.convertDataHandlerToBytes(apim.export(pid, METS_EXT1_0_LEGACY, exportContext));
95              } else {
96                  throw new IOException("You are connected to a pre-2.0 Fedora repository "
97                          + "which will only export the XML format \"metslikefedora1\".");
98              }
99          } else {
100             if (majorVersion < 3) {
101                 if (format != null) {
102                     if (format.equals(FOXML1_1.uri)) {
103                         System.out.println("WARNING: Repository does not support FOXML 1.1; " +
104                                            "exporting older format (v1.0) instead");
105                         format = FOXML1_0_LEGACY;
106                     } else if (format.equals(FOXML1_0.uri)) {
107                         format = FOXML1_0_LEGACY;
108                     } else if (format.equals(METS_EXT1_1.uri)) {
109                         System.out.println("WARNING: Repository does not support METS Fedora " +
110                                            "Extension 1.1; exporting older format (v1.0) instead");
111                         format = METS_EXT1_0_LEGACY;
112                     } else if (format.equals(METS_EXT1_0.uri)) {
113                         format = METS_EXT1_0_LEGACY;
114                     } else {
115                         throw new IOException("You are connected to a 2.x Fedora repository " +
116                                               "which will only export FOXML and METS XML formats.");
117                     }
118                 }
119             } else { // majorVersion >= 3
120                 if (format != null) {
121                     if (format.equals(FOXML1_0_LEGACY)) {
122                         format = FOXML1_0.uri;
123                     } else if (format.equals(METS_EXT1_0_LEGACY)) {
124                         format = METS_EXT1_0.uri;
125                     }
126                 }
127                 validateFormat(format);
128             }
129             bytes = TypeUtility.convertDataHandlerToBytes(apim.export(pid, format, exportContext));
130         }
131         try {
132             // TODO This export method has assumed the output is always XML,
133             // but with ATOM_ZIP (and in the future, IMS CP/SCORM) this is no
134             // longer the case. Perhaps we move pretty printing into the
135             // serializers themselves.
136             if (format.equals(ATOM_ZIP1_1.uri)) {
137                 FileUtils.copy(new ByteArrayInputStream(bytes), outStream);
138             } else {
139                 // use xerces to pretty print the xml, assuming it's well formed
140                 OutputFormat fmt = new OutputFormat("XML", "UTF-8", true);
141                 fmt.setIndent(2);
142                 fmt.setLineWidth(120);
143                 fmt.setPreserveSpace(false);
144                 XMLSerializer ser = new XMLSerializer(outStream, fmt);
145                 DocumentBuilderFactory factory =
146                         DocumentBuilderFactory.newInstance();
147                 factory.setNamespaceAware(true);
148                 DocumentBuilder builder = factory.newDocumentBuilder();
149                 Document doc = builder.parse(new ByteArrayInputStream(bytes));
150                 ser.serialize(doc);
151             }
152         } catch (Exception e) {
153             System.out.println("ERROR: " + e.getClass().getName() + " : "
154                     + e.getMessage());
155         } finally {
156             outStream.close();
157         }
158     }

165     public static void getObjectXML(FedoraAPIAMTOM apia,
166                                     FedoraAPIMMTOM apim,
167                                     String pid,
168                                     OutputStream outStream)
169             throws RemoteException, IOException {
170 
171         // Get the repository info from the hash for the repository that APIA is using,
172         // unless it isn't in the hash yet... in which case, ask the server.
173         RepositoryInfo repoinfo = s_repoInfo.get(apia);
174         if (repoinfo == null) {
175             repoinfo = apia.describeRepository();
176             s_repoInfo.put(apia, repoinfo);
177         }
178         // get the object XML as it exists in the repository's
179         // persitent storage area.
180         byte[] bytes = TypeUtility.convertDataHandlerToBytes(apim.getObjectXML(pid));
181         try {
182             // use xerces to pretty print the xml, assuming it's well formed
183             OutputFormat fmt = new OutputFormat("XML", "UTF-8", true);
184             fmt.setIndent(2);
185             fmt.setLineWidth(120);
186             fmt.setPreserveSpace(false);
187             XMLSerializer ser = new XMLSerializer(outStream, fmt);
188             DocumentBuilderFactory factory =
189                     DocumentBuilderFactory.newInstance();
190             factory.setNamespaceAware(true);
191             DocumentBuilder builder = factory.newDocumentBuilder();
192             Document doc = builder.parse(new ByteArrayInputStream(bytes));
193             ser.serialize(doc);
194         } catch (Exception e) {
195             System.out.println("ERROR: " + e.getClass().getName() + " : "
196                     + e.getMessage());
197         } finally {
198             outStream.close();
199         }
200     }

202     public static void validateFormat(String format) throws IOException {
203         if (format == null) {
204             return;
205         }
206         if (!format.equals(FOXML1_1.uri) &&
207             !format.equals(FOXML1_0.uri) &&
208             !format.equals(METS_EXT1_1.uri) &&
209             !format.equals(METS_EXT1_0.uri) &&
210             !format.equals(ATOM1_1.uri) &&
211             !format.equals(ATOM_ZIP1_1.uri) &&
212             !format.equals("default")) {
213             throw new IOException("Invalid export format. Valid FORMAT values are: '"
214                     + FOXML1_1.uri
215                     + "' '"
216                     + FOXML1_0.uri
217                     + "' '"
218                     + METS_EXT1_1.uri
219                     + "' '"
220                     + METS_EXT1_0.uri
221                     + "' '"
222                     + ATOM1_1.uri
223                     + "' '"
224                     + ATOM_ZIP1_1.uri
225                     + "' and 'default'");
226         }
