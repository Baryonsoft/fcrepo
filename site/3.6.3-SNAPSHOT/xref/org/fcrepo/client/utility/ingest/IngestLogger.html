15  /**
16   * Methods to create and update a log of ingest activity.
17   * 
18   * @author Chris Wilper
19   */
20  public class IngestLogger {
21  
22      public static File newLogFile(String logRootName) {
23  
24          String fileName =
25                  logRootName + "-" + System.currentTimeMillis() + ".xml";
26          File outFile;
27          String fedoraHome = Constants.FEDORA_HOME;
28          if (fedoraHome == null) {
29              // to current dir
30              outFile = new File(fileName);
31          } else {
32              // to client/log
33              File logDir =
34                      new File(new File(new File(fedoraHome), "client"), "logs");
35              if (!logDir.exists()) {
36                  logDir.mkdir();
37              }
38              outFile = new File(logDir, fileName);
39          }
40          return outFile;
41      }
42  
43      public static void openLog(PrintStream log, String rootName)
44              throws Exception {
45          log.println("<?xml version=\"1.0\" encoding=\"UTF-8\"?>");
46          log.println("<" + rootName + ">");
47      }
48  
49      public static void logFromFile(PrintStream log, File f, String pid)
50              throws Exception {
51          log.println("  <ingested file=\"" + f.getPath() + "\" targetPID=\""
52                  + pid + "\" />");
53      }
54  
55      public static void logFailedFromFile(PrintStream log, File f, Exception e)
56              throws Exception {
57          String message = e.getMessage();
58          if (message == null) {
59              message = e.getClass().getName();
60          }
61          log.println("  <failed file=\"" + f.getPath() + "\">");
62          log.println("    " + StreamUtility.enc(message));
63          log.println("  </failed>");
64      }
65  
66      public static void logFromRepos(PrintStream log,
67                                      String sourcePID,
68                                      String targetPID) {
69          log.println("  <ingested sourcePID=\"" + sourcePID + "\" targetPID=\""
70                  + targetPID + "\"/>");
71      }
72  
73      public static void logFailedFromRepos(PrintStream log,
74                                            String sourcePID,
75                                            Exception e) {
76          String message = e.getMessage();
77          if (message == null) {
78              message = e.getClass().getName();
79          }
80          log.println("  <failed sourcePID=\"" + sourcePID + "\">");
81          log.println("    " + StreamUtility.enc(message));
82          log.println("  </failed>");
83      }
84  
85      public static void closeLog(PrintStream log, String rootName)
86              throws Exception {
87          log.println("</" + rootName + ">");

89      }
