131     /**
132      * Parse the command line arguments and check for validity.
133      */
134     public ValidatorProcessParameters(String[] args) {
135         Map<String, String> parms = parseArgsIntoMap(args);
136 
137         String username = getRequiredParameter(PARAMETER_USERNAME, parms);
138         String password = getRequiredParameter(PARAMETER_PASSWORD, parms);
139         URL serverUrl = getRequiredUrlParameter(PARAMETER_SERVER_URL, parms);
140         serviceInfo = new ServiceInfo(serverUrl, username, password);
141 
142         String query = getOptionalParameter(PARAMETER_QUERY, parms);
143         String terms = getOptionalParameter(PARAMETER_TERMS, parms);
144         String pidfileParm = getOptionalParameter(PARAMETER_PIDFILE, parms);
145 
146         iteratorType = figureOutIteratorType(query, terms, pidfileParm);
147         if (iteratorType == IteratorType.FS_QUERY) {
148             fieldSearchQuery = assembleFieldSearchQuery(query, terms);
149             pidfile = null;
150         } else {
151             fieldSearchQuery = null;
152             pidfile = assemblePidfile(pidfileParm);
153         }
154 
155         if (parms.containsKey(PARAMETER_VERBOSE)) {
156             verbose = true;
157         } else {
158             verbose = false;
159         }
160     }

162     /**
163      * Decide which of the command line arguments are keywords and which are
164      * values, and build a map to hold them.
165      */
166     private Map<String, String> parseArgsIntoMap(String[] args) {
167         Map<String, String> parms = new HashMap<String, String>();
168 
169         for (int i = 0; i < args.length; i++) {
170             String key = args[i];
171             if (!isKeyword(key)) {
172                 throw new ValidatorProcessUsageException("'" + key
173                         + "' is not a keyword.");
174             }
175             if (!ALL_PARAMETERS.contains(key)) {
176                 throw new ValidatorProcessUsageException("'" + key
177                         + "' is not a recognized keyword.");
178             }
179             if (i >= args.length - 1) {
180                 parms.put(key, null);
181             } else {
182                 String value = args[i + 1];
183                 if (isKeyword(value)) {
184                     parms.put(key, null);
185                 } else {
186                     parms.put(key, value);
187                     i++;
188                 }
189             }
190         }
191 
192         return parms;
193     }
194 
195     private boolean isKeyword(String arg) {
196         return arg.startsWith("-");
197     }

199     /**
200      * Get the requested parameter from the map. Complain if it's not found or
201      * has a null value.
202      */
203     private String getRequiredParameter(String keyword,
204                                         Map<String, String> parms) {
205         if (!parms.containsKey(keyword)) {
206             throw new ValidatorProcessUsageException("Parameter '" + keyword
207                     + "' is required.");
208         }
209 
210         String value = parms.get(keyword);
211         if (value == null) {
212             throw new ValidatorProcessUsageException("Parameter '" + keyword
213                     + "' requires a value.");
214         } else {
215             return value;
216         }
217     }
218 
219     /**
220      * Get the requested parameter from the map. Complain if not found, or not a
221      * valid URL.
222      */
223     private URL getRequiredUrlParameter(String keyword,
224                                         Map<String, String> parms) {
225         String urlString = getRequiredParameter(keyword, parms);
226         try {
227             return new URL(urlString);
228         } catch (MalformedURLException e) {
229             throw new ValidatorProcessUsageException("Value '" + urlString
230                     + "' for parameter '" + keyword + "' is not a valid URL: "
231                     + e.getMessage());
232         }
233     }
234 
235     /**
236      * Get the requested parameter from the map. If it's not there, return null,
237      * but if it's there with no value, complain.
238      */
239     private String getOptionalParameter(String keyword,
240                                         Map<String, String> parms) {
241         if (!parms.containsKey(keyword)) {
242             return null;
243         }
244 
245         String value = parms.get(keyword);
246         if (value == null) {
247             throw new ValidatorProcessUsageException("If parameter '" + keyword
248                     + "' is provided, it must have a value.");
249         } else {
250             return value;
251         }
252     }

254     /**
255      * Look at the parameters. Is this a query-based request or a pidfile-based
256      * request? If we put in too many parms, or not enough, that's a problem.
257      */
258     private IteratorType figureOutIteratorType(String query,
259                                                String terms,
260                                                String pidfileParm) {
261         int howMany =
262                 (query == null ? 0 : 1) + (terms == null ? 0 : 1)
263                         + (pidfileParm == null ? 0 : 1);
264         if (howMany == 0) {
265             throw new ValidatorProcessUsageException("You must provide "
266                     + "either '" + PARAMETER_QUERY + "', '" + PARAMETER_TERMS
267                     + "' or '" + PARAMETER_PIDFILE + "'.");
268         }
269         if (howMany > 1) {
270             throw new ValidatorProcessUsageException("You must provide only "
271                     + "one of these parameters: '" + PARAMETER_QUERY + "', '"
272                     + PARAMETER_TERMS + "' or '" + PARAMETER_PIDFILE + "'.");
273         }
274 
275         return pidfileParm == null ? IteratorType.FS_QUERY
276                 : IteratorType.PIDFILE;
277     }

279     /**
280      * A {@link FieldSearchQuery} may be made from a terms string or a query
281      * string, but not both.
282      */
283     private FieldSearchQuery assembleFieldSearchQuery(String query, String terms) {
284         if (terms != null) {
285             return new FieldSearchQuery(terms);
286         } else {
287             try {
288                 return new FieldSearchQuery(Condition.getConditions(query));
289             } catch (QueryParseException e) {
290                 throw new ValidatorProcessUsageException("Value '" + query
291                         + "' of parameter '" + PARAMETER_QUERY
292                         + "' is not a valid query string.");
293             }
294         }
295     }
296 
297     /**
298      * Is there a valid file out there for this parm? We already know that the
299      * parms is not null.
300      */
301     private File assemblePidfile(String pidfileParm) {
302         File pidfile = new File(pidfileParm);
303         if (!pidfile.exists()) {
304             throw new ValidatorProcessUsageException("-pidfile does not exist: '"
305                     + pidfileParm + "'");
306         }
307         if (!pidfile.canRead()) {
308             throw new ValidatorProcessUsageException("-pidfile is not readable: '"
309                     + pidfileParm + "'");
310         }
311         return pidfile;
312     }
