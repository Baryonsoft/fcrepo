28  public class AutoPurger {

30      private final FedoraAPIMMTOM m_apim;
31  
32      public AutoPurger(FedoraAPIMMTOM apim)
33              throws MalformedURLException, ServiceException {
34          m_apim = apim;
35      }
36  
37      public void purge(String pid, String logMessage)
38              throws RemoteException, IOException {
39          purge(m_apim, pid, logMessage);
40      }
41  
42      public void purgeFromList(File pFile, String logMessage)
43              throws RemoteException, IOException {
44          purgeFromList(m_apim, pFile, logMessage);
45      }
46  
47      public static void purgeFromList(FedoraAPIMMTOM skeleton,
48                                 File pFile,
49                                 String logMessage) {
50          PidfileIterator p_iter = new PidfileIterator(pFile);
51          int objs_purged = 0;
52  
53          while ( p_iter.hasNext() ) {
54              String pid = p_iter.next();
55              try {
56                  skeleton.purgeObject(pid, logMessage, false);
57                  System.out.println("'" + pid +"' purged");
58                  objs_purged++;
59              } catch (Exception e) {
60                  System.err.println("ERROR purging '" + pid + "' : " +
61                                     (e.getMessage() == null ? "(no detail provided)" : e.getMessage()));
62              }
63          }
64          System.out.println(objs_purged + " objects successfully purged.");
65      }

67      public static void purge(FedoraAPIMMTOM skeleton,
68                               String pid,
69                               String logMessage) throws RemoteException, IOException {
70          skeleton.purgeObject(pid, logMessage, false);
71      }

73      /**
74       * Print error message and show usage for command-line interface.
75       */
76      public static void showUsage(String msg) {
77          System.err.println("Command: fedora-purge");
78          System.err.println();
79          System.err.println("Summary: Purges an object from the Fedora repository.");
80          System.err.println();
81          System.err.println("Syntax:");
82          System.err.println("  fedora-purge host:port user password pid|fileURI protocol log [context]");
83          System.err.println();
84          System.err.println("Where:");
85          System.err.println("  host          is the target repository hostname.");
86          System.err.println("  port          is the target repository port number.");
87          System.err.println("  user          is the id of the target repository user.");
88          System.err.println("  password      is the password of the target repository user.");
89          System.err.println("  pid|fileURI   is the id of the object to purge from the target repository");
90          System.err.println("                OR the file URL for a file containing pids to purge, one pid per line");
91          System.err.println("  protocol      is the protocol to communicate with repository (http or https)");
92          System.err.println("  log           is a log message.");
93          System.out.println(
94                  "  context  (optional) is a different web application server context of Fedora (default is fedora)");
95  
96          System.err.println();
97          System.err.println("Examples:");
98          System.err.println("fedora-purge myrepo.com:8443 jane janepw demo:5 https \"my message\"");
99          System.err.println();
100         System.err.println("  Purges the object whose pid is demo:5 from the");
101         System.err.println("  target repository at myrepo.com:8443 using the secure https protocol (SSL)");
102         System.err.println();
103         System.err.println("fedora-purge myrepo.com:8443 jane janepw file:///some/dir/purge-pids.txt http \"my message\"");
104         System.err.println();
105         System.err.println("  Purges the objects whose pids are listed in the file /some/dir/purge-pids.txt from the");
106         System.err.println("  target repository at myrepo.com:8443 using the non-secure http protocol");
107         System.err.println();
108         System.err.println("ERROR  : " + msg);
109         System.exit(1);
110     }

112     public static void main(String[] args) {
113         System.setProperty("java.awt.headless", "true");
114         try {
115             if (args.length < 6 || args.length > 7) {
116                 AutoPurger
117                         .showUsage("You must provide six or seven arguments.");
118             } else {
119                 String[] hp = args[0].split(":");
120                 String hostName = hp[0];
121                 int portNum = Integer.parseInt(hp[1]);
122                 String third_param = args[3];
123                 String pid = null;
124                 URL fileURL = null;
125                 if (third_param.startsWith("file:/")) {
126                     fileURL = new URL(third_param);
127                 } else {
128                     pid = third_param;
129                 }
130                 String protocol = args[4];
131                 String logMessage = args[5];
132 
133                 String context = Constants.FEDORA_DEFAULT_APP_CONTEXT;
134 
135                 if (args.length == 7 && !args[6].equals("")) {
136                     context = args[6];
137                 }
138                 // ******************************************
139                 // NEW: use new client utility class
140                 String baseURL =
141                         protocol + "://" + hostName + ":" + portNum + "/"
142                         + context;
143                 FedoraClient fc = new FedoraClient(baseURL, args[1], args[2]);
144                 AutoPurger a = new AutoPurger(fc.getAPIMMTOM());
145                 //*******************************************
146 
147                 /* Single PID:  just purge it */
148                 if (pid != null) {
149                     a.purge(pid, logMessage);
150                 /* read PIDs from a file */
151                 } else {
152                     File pFile = new File(fileURL.toURI()).getCanonicalFile();
153                     a.purgeFromList(pFile, logMessage);
154                 }
155             }
156         } catch (Exception e) {
157             AutoPurger.showUsage(e.getClass().getName()
158                                  + " - "
159                                  + (e.getMessage() == null ? "(no detail provided)" : e
160                     .getMessage()));
161         }
162     }
