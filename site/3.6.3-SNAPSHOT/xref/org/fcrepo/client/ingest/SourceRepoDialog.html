79      public SourceRepoDialog() {
80          super(JOptionPane.getFrameForComponent(Administrator.getDesktop()),

87                          .createCompoundBorder(BorderFactory

95          JLabel protocolLabel = new JLabel("Protocol");
96          JLabel contextLabel = new JLabel("Fedora Server Context");
97          JLabel usernameLabel = new JLabel("Username");
98          JLabel passwordLabel = new JLabel("Password");
99          if (s_lastServer == null) {

104         if (s_lastProtocol == null) {
105             s_lastProtocol = "http";
106         }
107         m_protocolField = new JTextField(s_lastProtocol);
108 
109         if (s_lastContext == null) {
110             s_lastContext = Constants.FEDORA_DEFAULT_APP_CONTEXT;
111         }
112         m_contextField = new JTextField(s_lastContext);
113 
114         if (s_lastUsername == null) {
115             s_lastUsername = "fedoraAdmin";
116         }
117         m_usernameField = new JTextField(s_lastUsername);
118 
119         if (s_lastPassword == null) {
120             s_lastPassword = "";
121         }
122         m_passwordField = new JPasswordField(s_lastPassword);

124         addLabelValueRows(new JLabel[] {serverLabel, protocolLabel,

126                 m_serverField, m_protocolField, m_contextField,
127                 m_usernameField, m_passwordField}, gridBag, inputPane);

129         JButton okButton = new JButton(new AbstractAction() {
130 
131             private static final long serialVersionUID = 1L;
132 
133             @Override
134             public void actionPerformed(ActionEvent evt) {
135                 // construct apia and apim after doing some field validation
136                 if (m_passwordField.getPassword().length == 0
137                         || m_usernameField.getText().equals("")) {
138                     JOptionPane
139                             .showMessageDialog(Administrator.getDesktop(),
140                                                "Username and password must both be non-empty",
141                                                "Error",
142                                                JOptionPane.ERROR_MESSAGE);
143                 } else {
144 
145                     String[] hp = m_serverField.getText().split(":");
146                     if (hp.length == 2) {
147                         try {
148                             m_protocol = m_protocolField.getText();
149                             m_host = hp[0];
150                             m_port = Integer.parseInt(hp[1]);
151 
152                             // ******************************************************
153                             // NEW: use the new client utility class FedoraClient
154                             String baseURL =
155                                     m_protocol + "://" + m_host + ":" + m_port
156                                             + "/" + m_contextField.getText();
157                             FedoraClient fc =
158                                     new FedoraClient(baseURL,
159                                                      m_usernameField.getText(),
160                                                      new String(m_passwordField
161                                                              .getPassword()));
162                             m_apia = fc.getAPIAMTOM();
163                             m_apim = fc.getAPIMMTOM();
164                             //*******************************************************
165 
166                             // Get SOAP stubs for the source repository.
167                             // NOTE! For backward compatibility with Fedora 2.0
168                             // we will immediately try a describe repository
169                             // request on the API-A stub to see if it works.  If it
170                             // fails, we will try obtaining a stub with the OLD
171                             // SOAP URL syntax.  This is because the path in the
172                             // SOAP URLs were changed in Fedora 2.1 to be more standard.
173                             /*
174                              * try { m_apia=APIAStubFactory.getStub(m_protocol,
175                              * m_host, m_port, m_usernameField.getText(), new
176                              * String(m_passwordField.getPassword()));
177                              * m_apim=APIMStubFactory.getStub(m_protocol,
178                              * m_host, m_port, m_usernameField.getText(), new
179                              * String(m_passwordField.getPassword())); // try a
180                              * request to see if things work ok
181                              * m_repositoryInfo=m_apia.describeRepository(); }
182                              * catch (Exception e) { // If request on default
183                              * stub fails, try the old URL path for the service\
184                              * m_apia=APIAStubFactory.getStubAltPath(m_protocol,
185                              * m_host, m_port, "/fedora/access/soap",
186                              * m_usernameField.getText(), new
187                              * String(m_passwordField.getPassword()));
188                              * m_apim=APIMStubFactory.getStubAltPath(m_protocol,
189                              * m_host, m_port, "/fedora/management/soap",
190                              * m_usernameField.getText(), new
191                              * String(m_passwordField.getPassword())); }
192                              */
193                             try {
194                                 m_repositoryInfo = m_apia.describeRepository();
195                                 s_lastServer = m_host + ":" + m_port;
196                                 s_lastProtocol = m_protocol;
197                                 s_lastContext = m_contextField.getText();
198                                 s_lastUsername = m_usernameField.getText();
199                                 s_lastPassword =
200                                         new String(m_passwordField
201                                                 .getPassword());
202                                 dispose();
203                             } catch (Exception e) {
204                                 // earlier repositories won't support these
205                                 // methods... so here we only fail if there's
206                                 // a connection or authentication problem
207                                 boolean retry = false;
208                                 if (e.getMessage().indexOf("Unauthorized") != -1
209                                         || e.getMessage()
210                                                 .indexOf("Unrecognized") != -1) {
211                                     Administrator
212                                             .showErrorDialog(Administrator
213                                                                      .getDesktop(),
214                                                              "Connection Error",
215                                                              "Bad username or password.",
216                                                              e);
217                                     retry = true;
218                                 }
219                                 if (e.getMessage().indexOf("java.net") != -1) {
220                                     Administrator
221                                             .showErrorDialog(Administrator
222                                                                      .getDesktop(),
223                                                              "Connection Error",
224                                                              "Can't connect to "
225                                                                      + m_protocol
226                                                                      + "://"
227                                                                      + m_host
228                                                                      + ":"
229                                                                      + m_port,
230                                                              e);
231                                     retry = true;
232                                 }
233                                 if (!retry) {
234                                     // the exception must have been the result
235                                     // of an unimplemented method on a prior
236                                     // version of fedora, which is ok.
237                                     s_lastServer = m_host + ":" + m_port;
238                                     s_lastProtocol = m_protocol;
239                                     s_lastContext = m_contextField.getText();
240                                     s_lastUsername = m_usernameField.getText();
241                                     s_lastPassword =
242                                             new String(m_passwordField
243                                                     .getPassword());
244                                     dispose();
245                                 } else {
246                                     e.printStackTrace();
247                                 }
248                             }
249                         } catch (NumberFormatException nfe) {
250                             Administrator
251                                     .showErrorDialog(Administrator.getDesktop(),
252                                                      "Error",
253                                                      "Server port must be numeric",
254                                                      nfe);
255                         } catch (Exception e) {
256                             Administrator
257                                     .showErrorDialog(Administrator.getDesktop(),
258                                                      "Error",
259                                                      "Malformed host information",
260                                                      e);
261                         }
262                     } else {
263                         JOptionPane
264                                 .showMessageDialog(Administrator.getDesktop(),
265                                                    "Server should be specified as host:port",
266                                                    "Error",
267                                                    JOptionPane.ERROR_MESSAGE);
268                     }
269                 }
270             }
271         });
272         okButton.setText("OK");
273         JButton cancelButton = new JButton(new AbstractAction() {

321     public void addLabelValueRows(JLabel[] labels,
322                                   JComponent[] values,
323                                   GridBagLayout gridBag,
324                                   Container container) {
325         GridBagConstraints c = new GridBagConstraints();
326         c.insets = new Insets(0, 6, 6, 6);
327         for (int i = 0; i < labels.length; i++) {
328             c.anchor = GridBagConstraints.EAST;
329             c.gridwidth = GridBagConstraints.RELATIVE; //next-to-last
330             c.fill = GridBagConstraints.NONE; //reset to default
331             c.weightx = 0.0; //reset to default
332             gridBag.setConstraints(labels[i], c);
333             container.add(labels[i]);
334 
335             c.gridwidth = GridBagConstraints.REMAINDER; //end row
336             if (!(values[i] instanceof JComboBox)) {
337                 c.fill = GridBagConstraints.HORIZONTAL;
338             } else {
339                 c.anchor = GridBagConstraints.WEST;
340             }
341             c.weightx = 1.0;
342             gridBag.setConstraints(values[i], c);
343             container.add(values[i]);
344         }
345
