283     public class ConditionSelectionListener
284             implements ListSelectionListener {
285 
286         private int m_selectedRow;
287 
288         private final JButton m_modifyButton;
289 
290         private final JButton m_deleteButton;
291 
292         public ConditionSelectionListener(JButton modifyButton,
293                                           JButton deleteButton,
294                                           int selectedRow) {
295             m_selectedRow = selectedRow;
296             m_modifyButton = modifyButton;
297             m_deleteButton = deleteButton;
298             updateButtons();
299         }
300 
301         @Override
302         public void valueChanged(ListSelectionEvent e) {
303             //Ignore extra messages.
304             if (e.getValueIsAdjusting()) {
305                 return;
306             }
307 
308             ListSelectionModel lsm = (ListSelectionModel) e.getSource();
309             if (lsm.isSelectionEmpty()) {
310                 m_selectedRow = -1;
311             } else {
312                 m_selectedRow = lsm.getMinSelectionIndex();
313             }
314             updateButtons();
315         }
316 
317         public int getSelectedRow() {
318             return m_selectedRow;
319         }
320 
321         private void updateButtons() {
322             if (getSelectedRow() == -1) {
323                 m_modifyButton.setEnabled(false);
324                 m_deleteButton.setEnabled(false);
325             } else {
326                 m_modifyButton.setEnabled(true);
327                 m_deleteButton.setEnabled(true);
328             }
329         }
330     }
