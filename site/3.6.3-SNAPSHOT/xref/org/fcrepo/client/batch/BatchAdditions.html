70      private static final String getContents(String filePrefix) {
71          String contents = null;
72          try {
73              File file = new File(filePrefix);
74              FileInputStream fileInputStream = null;
75              if (!file.canRead()) {
76                  throw new Exception("Missing or unreadable file: " + getPath(file));
77              }
78              long lFileLength;
79              try {
80                  lFileLength = file.length();
81              } catch (Exception eCaughtStatFile) { //<== make specific
82                  throw new Exception("file " + getPath(file)
83                          + "couldn't be statted for reading");
84              }
85              if (lFileLength > Integer.MAX_VALUE) {
86                  throw new Exception("file " + getPath(file)
87                          + "too large for reading");
88              }
89              fileInputStream = new FileInputStream(file);
90              byte[] buffer = new byte[BUFFERLENGTH];
91              int bytesRead = 0;
92              StringBuffer sbuffer = new StringBuffer();
93              while ((bytesRead = fileInputStream.read(buffer, 0, BUFFERLENGTH)) != -1) {
94                  String temp = new String(buffer, 0, bytesRead);
95                  sbuffer.append(temp);
96              }
97              contents = new String(sbuffer);
98          } catch (Exception e) {
99              System.err.println("exception in content read " + e.getMessage());
100         }
101         return contents;
102     }

104     private static final void packageMetadata(PrintStream out,
105                                               Properties metadataProperties,
106                                               String context,
107                                               String objectname,
108                                               int indents) {
109         Enumeration elementNames = metadataProperties.propertyNames(); //metadataCategories.keys();
110         String tabs = "STRING NOT ASSIGNED TO";
111         {
112             StringBuffer temp = new StringBuffer();
113             for (int i = 0; i < indents; i++) {
114                 temp.append("\t");
115             }
116             tabs = new String(temp);
117         }
118         if (elementNames.hasMoreElements()) {
119             out.println(tabs + "<metadata>");
120             while (elementNames.hasMoreElements()) {
121                 String elementName = (String) elementNames.nextElement();
122                 String directoryName =
123                         metadataProperties.getProperty(elementName); //metadataCategories.get(elementName);
124                 String parentPath = context + directoryName;
125                 String fileName = getFilename(parentPath, objectname);
126                 if (fileName != null && !fileName.equals("")) {
127                     String metadata = getContents(parentPath + FS + fileName);
128                     if (metadata != null && !metadata.equals("")) {
129                         out.println(tabs + "\t<metadata id=\"" + elementName
130                                 + "\">");
131                         out.println(metadata);
132                         out.println(tabs + "\t</metadata>"); //"+ elementName +">");
133                     }
134                 }
135             }
136             out.println(tabs + "</metadata>");
137         }
138     }

145     static String getFilename(String parentPath, String objectName) {
146         String filename = null;
147         {
148             File temp = new File(parentPath);
149             if (temp.exists()) {
150                 if (!temp.isDirectory()) {
151                     System.err.println("bad directory structure " + parentPath
152                             + " " + objectName);
153                 } else {
154                     String[] filenames = (new File(parentPath)).list();
155                     int nFound = 0;
156                     int iFound = -1;
157                     for (int i = 0; i < filenames.length; i++) {
158                         if (filenames[i].startsWith(objectName)) {
159                             nFound++;
160                             iFound = i;
161                         }
162                     }
163                     if (nFound == 1) {
164                         filename = filenames[iFound];
165                     }
166                 }
167             }
168         }
169         return filename;
170     }

258     final void prep() throws Exception {
259         //get datastream labels from mediaDirectory
260         File mediaDirectory = new File(mediaPath);
261         File[] datastreamDirectories = mediaDirectory.listFiles();
262         datastreams = new String[datastreamDirectories.length];
263         for (int i = 0; i < datastreamDirectories.length; i++) {
264             if (datastreamDirectories[i].isDirectory()) {
265                 String directoryName = datastreamDirectories[i].getName();
266                 if (metadataCategories.get(directoryName) != null) {
267                     throw new Exception();
268                 }
269                 datastreams[i] = datastreamDirectories[i].getName();
270             }
271         }
272 
273         //get objectnames from keyDirectory
274         File[] files = null;
275         {
276             File keyDirectory = new File(keyPath);
277             files = keyDirectory.listFiles();
278         }
279         objectnames = new String[files.length];
280 
281         for (int i = 0; i < files.length; i++) {
282             String objectname = null;
283             {
284                 String filename = files[i].getName();
285                 int j = filename.lastIndexOf('.');
286                 objectname = j >= 0 ? filename.substring(0, j) : filename;
287             }
288             objectnames[i] = objectname;
289         }
290         good2go = true;
291     }

293     final void process() throws Exception {
294         if (good2go) {
295             int object = startObject;
296             for (String objectname : objectnames) {
297                 String objid = objectNameSpace + ":" + object++;
298                 PrintStream out =
299                         new PrintStream(new FileOutputStream(additionsPath + FS
300                                 + objectname));
301                 out.println("<?xml version=\"1.0\" encoding=\"ISO-8859-1\"?>");
302                 out.print("<input OBJID=\"" + objid + "\"");
303                 out.println(" xmlns:METS=\"" + METS.uri + "\" "
304                         + namespaceDeclarations + " >");
305                 packageMetadata(out,
306                                 metadataProperties,
307                                 metadataPath + FS,
308                                 objectname,
309                                 1);
310                 out.println("\t<datastreams>");
311                 Enumeration ddatastreams = dataProperties.propertyNames();
312                 while (ddatastreams.hasMoreElements()) {
313                     String ndatastream = (String) ddatastreams.nextElement();
314                     String datastream = dataProperties.getProperty(ndatastream);
315                     String href =
316                             mediaFileContainsUrl ? getContents(mediaPath + FS
317                                     + datastream + FS + objectname)
318                                     : getHref(stringPrefix, datastream
319                                             + "/"
320                                             + getFilename(mediaPath + "/"
321                                                     + datastream, objectname));
322                     out.println("\t\t<datastream id=\"" + ndatastream /* fixup(datastream) */
323                             + "\" href=\"" + href + "\">");
324                     out.println("\t\t</datastream>");
325                 }
326                 out.println("\t</datastreams>");
327                 out.println("</input>");
328                 out.close();
329             }
330         }
331     }
