134     final void process() throws Exception {
135         //System.err.println("in BatchIngest.process()");
136         pidMaps = new Hashtable<String, String>();
137         keys = new Vector<String>();
138         //AutoIngestor autoIngestor = new AutoIngestor(protocol, host, port, username, password);
139         AutoIngestor autoIngestor = new AutoIngestor(APIA, APIM);
140 
141         //get files from batchDirectory
142         File[] files = null;
143         {
144             File batchDirectory = new File(objectsPath);
145             files = batchDirectory.listFiles();
146         }
147 
148         if (!(pidsFormat.equals("xml") || pidsFormat.equals("text"))) {
149             System.err.println("bad pidsFormat: " + pidsFormat);
150         } else if (!objectFormat.equals(FOXML1_1.uri)
151                 && !objectFormat.equals(METS_EXT1_1.uri)) {
152             System.err.println("bad objectFormat: " + objectFormat);
153         } else {
154             int badFileCount = 0;
155             int succeededIngestCount = 0;
156             int failedIngestCount = 0;
157             String logMessage = "another fedora object";
158             for (int i = 0; i < files.length; i++) {
159                 if (!files[i].isFile()) {
160                     badFileCount++;
161                     System.err
162                             .println("batch directory contains unexpected directory or file: "
163                                     + files[i].getName());
164                 } else {
165                     String pid = null;
166                     try {
167                         pid =
168                                 autoIngestor
169                                         .ingestAndCommit(new FileInputStream(files[i]),
170                                                          objectFormat,
171                                                          logMessage);
172                     } catch (Exception e) {
173                         System.err.println("ingest failed for: "
174                                 + files[i].getName());
175                         System.err.println("\t" + e.getClass().getName());
176                         System.err.println("\t" + e.getMessage());
177                         System.err.println("ingest format specified was: \""
178                                 + objectFormat + "\"");
179                         System.err.println("===BATCH HAS FAILED===");
180                         System.err
181                                 .println("consider manually backing out "
182                                         + "any objects which were already successfully ingested in this batch");
183                         throw e;
184                     }
185                     if (pid == null || pid.equals("")) {
186                         failedIngestCount++;
187                         System.err.println("ingest failed for: "
188                                 + files[i].getName());
189                     } else {
190                         succeededIngestCount++;
191                         System.out.println("ingest succeeded for: "
192                                 + files[i].getName());
193                         keys.add(files[i].getName());
194                         pidMaps.put(files[i].getName(), pid);
195                     }
196                 }
197             }
198             System.err.println("\n" + "Batch Ingest Summary");
199             System.err.println("\n"
200                     + (succeededIngestCount + failedIngestCount + badFileCount)
201                     + " files processed in this batch");
202             System.err.println("\t" + succeededIngestCount
203                     + " objects successfully ingested into Fedora");
204             System.err.println("\t" + failedIngestCount + " objects failed");
205             System.err.println("\t" + badFileCount
206                     + " unexpected files in directory");
207             System.err
208                     .println("\t"
209                             + (files.length - (succeededIngestCount
210                                     + failedIngestCount + badFileCount))
211                             + " files ignored after error");
212         }
213     }
