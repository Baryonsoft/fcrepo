5   package org.fcrepo.client.batch;

7   import java.io.PrintStream;
8   
9   import org.xml.sax.ErrorHandler;
10  import org.xml.sax.SAXException;
11  import org.xml.sax.SAXParseException;
12  
13  /**
14   * @author Ross Wayland
15   */
16  public class BatchModifyValidatorErrorHandler
17          implements ErrorHandler {
18  
19      protected static int errorCount = 0;
20  
21      private static PrintStream out;
22  
23      public BatchModifyValidatorErrorHandler(PrintStream out) {
24          BatchModifyValidatorErrorHandler.out = out;
25          BatchModifyValidatorErrorHandler.errorCount = 0;
26      }
27  
28      public void warning(SAXParseException e) throws SAXException {
29          System.err
30                  .println("BatchModifyValidatorErrorHandler detected SAX WARNING: ");
31          logError(e);
32      }
33  
34      public void error(SAXParseException e) throws SAXException {
35          System.err
36                  .println("BatchModifyValidatorErrorHandler detected SAX ERROR: ");
37          logError(e);
38      }
39  
40      public void fatalError(SAXParseException e) throws SAXException {
41          System.err
42                  .print("BatchModifyValidatorErrorHandler detected SAX FATAL ERROR:");
43          logError(e);
44      }
45  
46      private static void printPubID(SAXParseException e) {
47          if (e.getPublicId() != null) {
48              System.err.print(e.getPublicId() + " ");
49              out.println("    " + e.getPublicId() + " ");
50          }
51          if (e.getLineNumber() != -1) {
52              System.err
53                      .println("[modify-directives-file] Parsing error at line: "
54                              + e.getLineNumber() + " column: "
55                              + e.getColumnNumber() + " ");
56              out.println("    [modify-directives-file] Parsing error at line: "
57                      + e.getLineNumber() + " column: " + e.getColumnNumber()
58                      + " ");
59          }
60      }
61  
62      private static void printMsg(SAXParseException e) {
63          System.err.println(e.getClass().getName()
64                  + " - "
65                  + (e.getMessage() == null ? "(no detail provided)" : e
66                          .getMessage()));
67          out.println("    "
68                  + e.getClass().getName()
69                  + " - "
70                  + (e.getMessage() == null ? "(no detail provided)" : e
71                          .getMessage()));
72      }
73  
74      private static void logError(SAXParseException e) {
75          out.println("  <parser-error>");
76          printPubID(e);
77          printMsg(e);
78          out.println("  </parser-error>");
79          errorCount++;
80      }

82  }
