33  /**
34   * A GUI interface for entering info required to validate a file of modify
35   * directives against the batchModify-1.1.xsd XML Schema. The batch modify
36   * utility has validation turned on so pre-validation is not necessary.
37   * It is provided as a means to pre-validate a modify directives file prior
38   * to running it through the batch modify utility.

53      public BatchModifyValidate() {
54          InputStream in = null;
55          BatchModifyValidator bmv = null;
56          File file = null;
57          long st = System.currentTimeMillis();
58          long et = 0;
59  
60          try {
61              JFileChooser browse = new JFileChooser(Administrator.getLastDir());
62              browse.setFileSelectionMode(JFileChooser.FILES_AND_DIRECTORIES);
63              int returnVal = browse.showOpenDialog(Administrator.getDesktop());
64              if (returnVal == JFileChooser.APPROVE_OPTION) {
65                  file = browse.getSelectedFile();
66                  int n =
67                          JOptionPane
68                                  .showConfirmDialog(Administrator.getDesktop(),
69                                                     "Validate the Modify Directives in file: \n"
70                                                             + file
71                                                                     .getAbsoluteFile()
72                                                             + " ?\n",
73                                                     "Validate Modify Directives File",
74                                                     JOptionPane.YES_NO_OPTION);
75                  if (n == JOptionPane.YES_OPTION) {
76                      Administrator.setLastDir(file);
77                      openLog("validate-modify-directives");
78                      in = new FileInputStream(file);
79                      bmv = new BatchModifyValidator(in, s_log);
80                  }
81              }
82          } catch (Exception e) {
83              Administrator
84                      .showErrorDialog(Administrator.getDesktop(),
85                                       "Error Parsing Directives File.",
86                                       e.getClass().getName()
87                                               + " - "
88                                               + (e.getMessage() == null ? "(no detail provided)"
89                                                       : e.getMessage()),
90                                       e);
91          } finally {
92              try {
93                  if (in != null) {
94                      in.close();
95                  }
96                  if (s_log != null && bmv != null) {
97                      et = System.currentTimeMillis();
98                      if (bmv.isValid()) {
99                          JOptionPane
100                                 .showMessageDialog(Administrator.getDesktop(),
101                                                    "Modify Directives file: \n"
102                                                            + file
103                                                                    .getAbsoluteFile()
104                                                            + "\nis Valid !"
105                                                            + "\nTime elapsed: "
106                                                            + getDuration(et
107                                                                    - st),
108                                                    "Directives File Valid",
109                                                    JOptionPane.INFORMATION_MESSAGE);
110                         closeLog();
111                         return;
112                     }
113                     JOptionPane
114                             .showMessageDialog(Administrator.getDesktop(),
115                                                +bmv.getErrorCount()
116                                                        + " XML validation errors found in Modify Directives file.\n"
117                                                        + "See log file for details.\n"
118                                                        + "Time elapsed: "
119                                                        + getDuration(et - st));
120                     s_log.println("  <summary>");
121                     s_log
122                             .println("    "
123                                     + StreamUtility
124                                             .enc(bmv.getErrorCount()
125                                                     + " XML validation errors found in Modify Directives file.\n"
126                                                     + "    See log file for details.\n"
127                                                     + "    Time elapsed: "
128                                                     + getDuration(et - st)));
129                     s_log.println("  </summary>");
130                     closeLog();
131                     int n =
132                             JOptionPane
133                                     .showConfirmDialog(Administrator
134                                                                .getDesktop(),
135                                                        "A detailed log file was created at\n"
136                                                                + s_logPath
137                                                                + "\n\n"
138                                                                + "View it now?",
139                                                        "View Validation Log?",
140                                                        JOptionPane.YES_NO_OPTION);
141                     if (n == JOptionPane.YES_OPTION) {
142                         JTextComponent textEditor = new JTextArea();
143                         textEditor.setFont(new Font("monospaced",
144                                                     Font.PLAIN,
145                                                     12));
146                         textEditor.setText(fileAsString(s_logPath));
147                         textEditor.setCaretPosition(0);
148                         textEditor.setEditable(false);
149                         JInternalFrame viewFrame =
150                                 new JInternalFrame("Viewing " + s_logPath,
151                                                    true,
152                                                    true,
153                                                    true,
154                                                    true);
155                         ImageIcon editIcon =
156                             new ImageIcon(ClassLoader.
157                                           getSystemResource("images/client/standard/general/Edit16.gif"));
158                         viewFrame.setFrameIcon(editIcon);
159                         viewFrame.getContentPane()
160                                 .add(new JScrollPane(textEditor));
161                         viewFrame.setSize(720, 520);
162                         viewFrame.setVisible(true);
163                         Administrator.getDesktop().add(viewFrame);

165                         try {
166                             viewFrame.setSelected(true);

168                             // ignore
169                         }
170                     }
171                 }
172             } catch (Exception e) {

174                 Administrator

176                                          "Error",
177                                          e.getClass().getName()
178                                                  + " - "

186     /**
187      * Convert the duration time from milliseconds to standard hours, minutes,
188      * and seconds format.
189      *
190      * @param millis
191      *        The time interval to convert in miliseconds.
192      * @return A string with the converted time.
193      */
194     private static String getDuration(long millis) {
195         long tsec = millis / 1000;
196         long h = tsec / 60 / 60;
197         long m = (tsec - h * 60 * 60) / 60;
198         long s = tsec - h * 60 * 60 - m * 60;
199         StringBuffer out = new StringBuffer();
200         if (h > 0) {
201             out.append(h + " hour");
202             if (h > 1) {
203                 out.append('s');
204             }
205         }
206         if (m > 0) {
207             if (h > 0) {
208                 out.append(", ");
209             }
210             out.append(m + " minute");
211             if (m > 1) {
212                 out.append('s');
213             }
214         }
215         if (s > 0 || h == 0 && m == 0) {
216             if (h > 0 || m > 0) {
217                 out.append(", ");
218             }
219             out.append(s + " second");
220             if (s != 1) {
221                 out.append('s');
222             }
223         }
224         return out.toString();
225     }

227     /**
228      * Initializes the log file for writing.
229      *
230      * @param rootName
231      *        The name of the root element for the xml log file.
232      * @throws Exception
233      *         If any type of error occurs in trying to open the log file for
234      *         writing.
235      */
236     private static void openLog(String rootName) throws Exception {
237         s_rootName = rootName;
238         String fileName =
239                 s_rootName + "-" + System.currentTimeMillis() + ".xml";
240         File outFile;
241         String fedoraHome = Constants.FEDORA_HOME;
242         if (fedoraHome == null) {
243             // to current dir
244             outFile = new File(fileName);
245         } else {
246             // to client/log
247             File logDir =
248                     new File(new File(new File(fedoraHome), "client"), "logs");
249             if (!logDir.exists()) {
250                 logDir.mkdir();
251             }
252             outFile = new File(logDir, fileName);
253 
254         }
255         s_logPath = outFile.getPath();
256         s_log = new PrintStream(new FileOutputStream(outFile), true, "UTF-8");
257         s_log.println("<?xml version=\"1.0\" encoding=\"UTF-8\"?>");
258         s_log.println("<" + s_rootName + ">");
259     }

272     /**
273      * Converts file into string.
274      *
275      * @param path
276      *        The absolute file path of the file.
277      * @return The contents of the file as a string.
278      * @throws Exception
279      *         If any type of error occurs during the conversion.
280      */
281     private static String fileAsString(String path) throws Exception {
282         StringBuffer buffer = new StringBuffer();
283         InputStream fis = new FileInputStream(path);
284         InputStreamReader isr = new InputStreamReader(fis, "UTF-8");
285         Reader in = new BufferedReader(isr);
286         int ch;
287         while ((ch = in.read()) > -1) {
288             buffer.append((char) ch);
289         }
290         in.close();
291         return buffer.toString();
292     }
