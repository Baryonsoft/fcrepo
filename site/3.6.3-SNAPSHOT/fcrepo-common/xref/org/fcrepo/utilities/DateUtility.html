50      /**
51       * Converts an instance of java.util.Date into an ISO 8601 String
52       * representation. Uses the date format yyyy-MM-ddTHH:mm:ss.SSSZ or
53       * yyyy-MM-ddTHH:mm:ssZ, depending on whether millisecond precision is
54       * desired.
55       *
56       * @param date
57       *        Instance of java.util.Date.
58       * @param millis
59       *        Whether or not the return value should include milliseconds.
60       * @return ISO 8601 String representation of the Date argument or null if
61       *         the Date argument is null.
62       */
63      public static String convertDateToString(Date date, boolean millis) {
64          if (date == null) {
65              return null;
66          } else {
67              DateFormat df;
68              if (millis) {
69                  df = new SimpleDateFormat("yyyy-MM-dd'T'HH:mm:ss.SSS'Z'");
70              } else {
71                  df = new SimpleDateFormat("yyyy-MM-dd'T'HH:mm:ss'Z'");
72              }
73              df.setTimeZone(TimeZone.getTimeZone("UTC"));
74  
75              if (date.before(ONE_CE)) {
76                  StringBuilder sb = new StringBuilder(df.format(date));
77                  sb.insert(0, "-");
78                  return sb.toString();
79              } else {
80                  return df.format(date);
81              }
82          }
83      }

85      /**
86       * Converts an instance of <code>Date</code> into the canonical lexical
87       * representation of an XSD dateTime with the following exceptions: - Dates
88       * before 1 CE (i.e. 1 AD) are handled according to ISO 8601:2000 Second
89       * Edition: "0000" is the lexical representation of 1 BCE "-0001" is the
90       * lexical representation of 2 BCE
91       *
92       * @param date
93       *        Instance of java.util.Date.
94       * @return the lexical form of the XSD dateTime value, e.g.
95       *         "2006-11-13T09:40:55.001Z".
96       * @see <a
97       *      href="http://www.w3.org/TR/xmlschema-2/#date-canonical-representation">3.2.7.2
98       *      Canonical representation</a>
99       */
100     public static String convertDateToXSDString(Date date) {
101         if (date == null) {
102             return null;
103         }
104         StringBuilder lexicalForm;
105         String dateTime = convertDateToString(date, true);
106         int len = dateTime.length() - 1;
107         if (dateTime.indexOf('.', len - 4) != -1) {
108             while (dateTime.charAt(len - 1) == '0') {
109                 len--; // fractional seconds may not with '0'.
110             }
111             if (dateTime.charAt(len - 1) == '.') {
112                 len--;
113             }
114             lexicalForm = new StringBuilder(dateTime.substring(0, len));
115             lexicalForm.append('Z');
116         } else {
117             lexicalForm = new StringBuilder(dateTime);
118         }
119 
120         if (date.before(ONE_CE)) {
121             DateFormat df = new SimpleDateFormat("yyyy");
122             df.setTimeZone(TimeZone.getTimeZone("UTC"));
123             StringBuilder year =
124                     new StringBuilder(String.valueOf(Integer.parseInt(df
125                             .format(date)) - 1));
126             while (year.length() < 4) {
127                 year.insert(0, '0');
128             }
129             lexicalForm
130                     .replace(0, lexicalForm.indexOf("-", 4), year.toString());
131             if (date.before(ONE_BCE)) {
132                 lexicalForm.insert(0, "-");
133             }
134         }
135         return lexicalForm.toString();
136     }

201     public static Date parseDateOrNull(String dateString) throws IllegalArgumentException {
202         if (dateString == null) {
203             return null;

205             return null;
206         }
207         try {
208             return parseDateStrict(dateString);

214     /**
215      * Attempt to parse the given string of form: yyyy-MM-dd[THH:mm:ss[.SSS][Z]]
216      * as a Date.
217      *
218      * @param dateString the date string to parse
219      * @return a Date representation of the dateString
220      * @throws ParseException if dateString is null, empty or is otherwise
221      * unable to be parsed.
222      */
223     public static Date parseDateStrict(String dateString) throws ParseException {
224         if (dateString == null) {
225             throw new ParseException("Argument cannot be null.", 0);
226         } else if (dateString.isEmpty()) {
227             throw new ParseException("Argument cannot be empty.", 0);
228         } else if (dateString.endsWith(".")) {
229             throw new ParseException("dateString ends with invalid character.", dateString.length() - 1);
230         }
231         SimpleDateFormat formatter = new SimpleDateFormat();
232         formatter.setTimeZone(TimeZone.getTimeZone("UTC"));
233         int length = dateString.length();
234         if (dateString.startsWith("-")) {
235             length--;
236         }
237         if (dateString.endsWith("Z")) {
238             if (length == 11) {
239                 formatter.applyPattern("yyyy-MM-dd'Z'");
240             } else if (length == 20) {
241                 formatter.applyPattern("yyyy-MM-dd'T'HH:mm:ss'Z'");
242             } else if (length > 21 && length < 24) {
243                 // right-pad the milliseconds with 0s up to three places
244                 StringBuilder sb = new StringBuilder(dateString.substring(0, dateString.length() - 1));
245                 int dotIndex = sb.lastIndexOf(".");
246                 int endIndex = sb.length() - 1;
247                 int padding = 3 - (endIndex - dotIndex);
248                 for (int i = 0; i < padding; i++) {
249                     sb.append("0");
250                 }
251                 sb.append("Z");
252                 dateString = sb.toString();
253                 formatter.applyPattern("yyyy-MM-dd'T'HH:mm:ss.SSS'Z'");
254             } else if (length == 24) {
255                 formatter.applyPattern("yyyy-MM-dd'T'HH:mm:ss.SSS'Z'");
256             }
257         } else {
258             if (length == 10) {
259                 formatter.applyPattern("yyyy-MM-dd");
260             } else if (length == 19) {
261                 formatter.applyPattern("yyyy-MM-dd'T'HH:mm:ss");
262             } else if (length > 20 && length < 23) {
263                 // right-pad millis with 0s
264                 StringBuilder sb = new StringBuilder(dateString);
265                 int dotIndex = sb.lastIndexOf(".");
266                 int endIndex = sb.length() - 1;
267                 int padding = 3 - (endIndex - dotIndex);
268                 for (int i = 0; i < padding; i++) {
269                     sb.append("0");
270                 }
271                 dateString = sb.toString();
272                 formatter.applyPattern("yyyy-MM-dd'T'HH:mm:ss.SSS");
273             } else if (length == 23) {
274                 formatter.applyPattern("yyyy-MM-dd'T'HH:mm:ss.SSS");
275             } else if (dateString.endsWith("GMT") || dateString.endsWith("UTC")) {
276                 formatter.applyPattern("EEE, dd MMMM yyyyy HH:mm:ss z");
277             }
278         }
279         return formatter.parse(dateString);
280     }
