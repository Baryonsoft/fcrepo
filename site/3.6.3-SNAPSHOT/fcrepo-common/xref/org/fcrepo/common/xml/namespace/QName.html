9    * A namespace-qualified name in XML.

14  public class QName extends javax.xml.namespace.QName {
15  
16      private static final long serialVersionUID = 6368425528110304020L;
17  
18      /** The namespace to which this name belongs. */
19      public final XMLNamespace namespace;
20  
21      /** The local part of the qualified name. */
22      public final String localName;
23  
24      /**
25       * A string of the form: <code>prefix:localName</code>, acceptable for
26       * use in an instance document. The prefix used will be the preferred prefix
27       * of the namespace.
28       */
29      public final String qName;
30  
31      /**
32       * Constructs an instance.
33       * 
34       * @param namespace
35       *        the namespace to which this name belongs.
36       * @param localName
37       *        the local part of the qualified name.
38       * @throws IllegalArgumentException
39       *         if either parameter is null.
40       */
41      public QName(XMLNamespace namespace, String localName) {
42          super(namespace.uri, localName, namespace.prefix);
43          
44          if (namespace == null) {
45              throw new IllegalArgumentException("namespace cannot be null");
46          }
47          if (localName == null) {
48              throw new IllegalArgumentException("localName cannot be null");
49          }
50          this.namespace = namespace;
51          this.localName = localName;
52          qName = namespace.prefix + ":" + localName;
53      }
