8   /**
9    * An XML namespace.
10   * 
11   * @author Chris Wilper
12   * @version $Id$
13   */
14  public class XMLNamespace {
15      
16      /**
17       * Namespace URI to use to represent that there is no Namespace. 
18       * Defined by the Namespace specification to be "".
19       */
20      public static final String NULL_NS_URI = "";
21      
22      /** The URI of this namespace. */
23      public final String uri;
24  
25      /**
26       * The preferred prefix for this namespace when used in instance documents.
27       */
28      public final String prefix;
29  
30      /**
31       * Constructs an instance.
32       * 
33       * @param uri
34       *        the URI of the namespace.
35       * @param prefix
36       *        the preferred prefix.
37       * @throws IllegalArgumentException
38       *         if either parameter is null.
39       */
40      public XMLNamespace(String uri, String prefix) {
41          if (uri == null) {
42              throw new IllegalArgumentException("uri cannot be null");
43          }
44          if (prefix == null) {
45              throw new IllegalArgumentException("prefix cannot be null");
46          }
47          this.uri = uri;
48          this.prefix = prefix;
49      }
50  
51      //---
52      // Object overrides
53      //---
54  
55      /**
56       * Returns the URI of the namespace. {@inheritDoc}
57       */
58      @Override
59      public String toString() {
60          return uri;
61      }
62  
63      /**
64       * Returns true iff the given object is an instance of this class and has
65       * the same URI. {@inheritDoc}
66       */
67      @Override
68      public boolean equals(Object o) {
69          if (o instanceof XMLNamespace
70                  && o.getClass().getName().equals(this.getClass().getName())) {
71              XMLNamespace n = (XMLNamespace) o;
72              return uri.equals(n.uri);
73          } else {
74              return false;
75          }
76      }
77  
78      /**
79       * {@inheritDoc}
80       */

84      }
