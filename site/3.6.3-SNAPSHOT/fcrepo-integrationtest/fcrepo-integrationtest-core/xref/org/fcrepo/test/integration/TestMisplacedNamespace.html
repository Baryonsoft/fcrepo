22  /**
23   * Tests live behaviour when ingesting malformed foxml.
24   * <p>
25   * This test is to verify repository sanity when ingesting a foxml file with an
26   * inline datastream containing an element such that the namespace of the
27   * element is defined outside the scope of the inline xmlContent. The repository
28   * should reject the file upon ingest without any irreperable harm to the object
29   * registry.
30   * </p>
31   *
32   * @author Aaron Birkland
33   * @version $Id$
34   * @see <a
35   *      href="https://fedora-commons.org/jira/browse/FCREPO-537">FCREPO-537</a>
36   */
37  public class TestMisplacedNamespace
38          extends FedoraTestCase {
39  
40      private FedoraClient m_client;
41  
42      private static final String OFFENDING_FOXML =
43              "test-objects/foxml/TestMisplacedNamespace/DemoFail.xml";
44  
45      private static final String GOOD_FOXML =
46              "test-objects/foxml/TestMisplacedNamespace/DemoSuccess.xml";
47  
48      private static final String PID = "demo:failObject";
49  
50      @Override
51      @Before
52      public void setUp() throws Exception {
53          m_client = getFedoraClient();
54      }
55  
56      @Override
57      @After
58      public void tearDown() throws Exception {
59          m_client.getAPIMMTOM().purgeObject(PID, "Cleanup", false);
60      }
61  
62      @Test
63      public void testIngestAndPurge() throws Exception {
64          try {
65              /* Ingest of the offending foxml should fail */
66              m_client.getAPIMMTOM().ingest(TypeUtility.convertBytesToDataHandler(getFoxml(OFFENDING_FOXML)),
67                                        FOXML1_1.uri,
68                                        "malformed foxml object");
69              fail("Sould have failed initial ingest!");
70          } catch (Exception e) {
71              /* Make sure the RI contains no trace of this object */
72              RISearchUtil.checkSPOCount(m_client, "<info:fedora/" + PID
73                                                   + "> * *", 0);
74          }
75  
76          /* Ingest the good object. Should succeed */
77          m_client.getAPIMMTOM().ingest(TypeUtility.convertBytesToDataHandler(getFoxml(GOOD_FOXML)),
78                                    FOXML1_1.uri,
79                                    "non-malformed foxml object");
80      }
81  
82      private byte[] getFoxml(String file) throws IOException {
83          return StreamUtility.getBytes(getClass().getClassLoader()
84                  .getResourceAsStream(file));
85      }
86  }
