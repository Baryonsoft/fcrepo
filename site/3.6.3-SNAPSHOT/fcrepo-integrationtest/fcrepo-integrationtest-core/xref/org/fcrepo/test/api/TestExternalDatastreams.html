163     @Test
164     public void testAddDatastreamWithChecksum() throws Exception {
165         String pid = "demo:e_ds_test_add";
166         String checksumType = "MD5";
167         apim.ingest(getFoxmlObject(pid, null), FOXML1_1.uri, null);
168         File temp = null;
169 
170         try {
171             temp = getTempFile(true);
172             String dsLocation = temp.getCanonicalFile().toURI().toString();
173             String checksum = computeChecksum(checksumType, new FileInputStream(temp));
174             String dsId = addDatastream(pid, dsLocation, checksumType, checksum);
175             assertEquals("DS", dsId);
176 
177             // Now ensure that bogus checksums do indeed fail
178             apim.purgeDatastream(pid, dsId, null, null, null, false);
179             addDatastream(pid, dsLocation, checksumType, checksum);
180 
181             apim.purgeDatastream(pid, dsId, null, null, null, false);
182             checksum = "bogus";
183             try {
184                 addDatastream(pid, dsLocation, checksumType, checksum);
185                 fail("Adding datastream with bogus checksum should have failed.");
186             } catch (SOAPFaultException e) {
187                 assertTrue(e.getMessage().contains("Checksum Mismatch"));
188             }
189         } finally {
190             apim.purgeObject(pid, "test", false);
191             if (temp != null) temp.delete();
192         }
193 
194         pid = "demo:e_ds_test_add_portable";
195         apim.ingest(getFoxmlObject(pid, null), FOXML1_1.uri, null);
196         try {
197             String dsLocation = "http://local.fedora.server/fedora/objects/fedora-system:ContentModel-3.0/datastreams/DC/content";
198             org.fcrepo.server.types.gen.MIMETypedStream dc_content = apia.getDatastreamDissemination("fedora-system:ContentModel-3.0", "DC", null);
199             String checksum = computeChecksum(checksumType, new ByteArrayInputStream(dc_content.getStream()));
200             String dsId = addDatastream(pid, dsLocation, checksumType, checksum);
201             assertEquals("DS", dsId);
202         } finally {
203             apim.purgeObject(pid, "testAddDatastreamWithChecksumType", false);
204             if (temp != null) temp.delete();
205         }

335     private Foxml11Document createFoxmlObject(String spid, String contentLocation) throws Exception {
336         PID pid = PID.getInstance(spid);
337         Date date = new Date(1);
338 
339         Foxml11Document doc = new Foxml11Document(pid.toString());
340         doc.addObjectProperty(Property.STATE, "A");
341 
342         if (contentLocation != null && contentLocation.length() > 0) {
343             String ds = "DS";
344             String dsv = "DS1.0";
345             doc.addDatastream(ds, State.A, ControlGroup.E, true);
346             doc.addDatastreamVersion(ds, dsv, "text/plain", "label", 1, date);
347             doc.setContentLocation(dsv, contentLocation, org.fcrepo.server.storage.types.Datastream.DS_LOCATION_TYPE_URL);
348         }
349         return doc;
350     }

352     private String computeChecksum(String csType, InputStream is) throws Exception {
353         MessageDigest md = MessageDigest.getInstance(csType);
354         if (is == null) {
355             throw new IllegalArgumentException("InputStream cannot be null");
356         }
357         byte buffer[] = new byte[5000];
358         int numread;
359         while ((numread = is.read(buffer, 0, 5000)) > 0) {
360             md.update(buffer, 0, numread);
361         }
362         return StringUtility.byteArraytoHexString(md.digest());
363     }
