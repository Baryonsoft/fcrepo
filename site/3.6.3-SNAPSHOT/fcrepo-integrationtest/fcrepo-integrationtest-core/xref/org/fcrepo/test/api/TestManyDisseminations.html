127     /**
128      * Starts getting a dissemination, then aborts the connection (forcibly
129      * closes the socket on the client side) 30 times in a row.
130      */
131     private void doDissemTest(String what, boolean redirectOK) throws Exception {
132         final int num = 30;
133         System.out.println("Getting " + what + " " + num + " times...");
134         int i = 0;
135         try {
136             URL url = new URL(BASE_URL + "/get/" + what);
137             for (i = 0; i < num; i++) {
138                 HttpURLConnection conn =
139                         (HttpURLConnection) url.openConnection();
140                 InputStream in = conn.getInputStream();
141                 in.read();
142                 in.close();
143                 conn.disconnect();
144             }
145         } catch (Exception e) {
146             fail("Dissemination of " + what + " failed on iter " + i + ": "
147                  + e.getMessage());
148         }
149     }

265             buf.append("<rdf:RDF xmlns:rdf=\"http://www.w3.org/1999/02/22-"
266                        + "rdf-syntax-ns#\" xmlns:fedora-model=\"info:fedora/"
267                        + "fedora-system:def/model#\">\n  <rdf:Description "
268                        + "rdf:about=\"info:fedora/" + sPID + "\">\n");

561             openFOXML(buf, DATA_OBJECT_PID);
562             final String mimeType = "text/xml";
563             InetAddress addr = InetAddress.getLocalHost();
564             final String url =
565                     "http://" + addr.getHostAddress() + ":"
566                     + FedoraServerTestCase.getPort() + "/"
567                     + FedoraServerTestCase.getFedoraAppServerContext()
568                     + "/get/" + DATA_OBJECT_PID + "/" + X_DS;
569             appendInlineDatastream(buf,
570                                    X_DS,
571                                    "text/xml",
572                                    Constants.OAI_DC2_0.uri,
573                                    getDC());
574             appendInlineDatastream(buf, "XSLT", "text/xml", null, getXSLT());
575             appendRemoteDatastream(buf, E_DS, "E", mimeType, url);
576             appendRemoteDatastream(buf, R_DS, "R", mimeType, url);
577 
578             StringBuilder rdf = new StringBuilder();
579             openRDF(rdf, DATA_OBJECT_PID);
580             appendRel(rdf, MODEL.HAS_MODEL.localName, PID
581                     .getInstance(Models.FEDORA_OBJECT_3_0.uri).toString());
582             appendRel(rdf, MODEL.HAS_MODEL.localName, CMODEL_OBJECT_PID);
583             closeRDF(rdf);
584             appendInlineDatastream(buf,
585                                    "RELS-EXT",
586                                    "application/rdf+xml",
587                                    Constants.RELS_EXT1_0.uri,
588                                    rdf.toString());
589 
590             closeFOXML(buf);
591             return buf.toString().getBytes("UTF-8");
592         }
