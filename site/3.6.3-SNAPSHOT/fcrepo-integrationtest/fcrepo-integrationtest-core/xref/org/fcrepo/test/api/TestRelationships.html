393     private void checkExistsViaGetRelationships(String subject,
394                                                 String predicate,
395                                                 String object) throws Exception {
396         boolean found = false;
397 
398         for (RelationshipTuple tuple : apim
399                 .getRelationships(subject, predicate)) {
400             if (tuple.getSubject().equals(subjectAsURI(subject))
401                     && tuple.getPredicate().equals(predicate)
402                     && tuple.getObject().equals(object)) {
403                 found = true;
404             }
405         }
406         assertTrue("Relationship not found via getRelationships (subject="
407                            + subject + ", predicate=" + predicate + ", object="
408                            + object,
409                    found);
410     }

415     private void addRelationship(String subject,
416                                  String predicate,
417                                  String object,
418                                  boolean isLiteral,
419                                  String datatype) throws Exception {
420         assertTrue(apim.addRelationship(subject,
421                                         predicate,
422                                         object,
423                                         isLiteral,
424                                         datatype));
425         assertFalse("Adding duplicate relationship should return false",
426                     apim.addRelationship(subject,
427                                          predicate,
428                                          object,
429                                          isLiteral,
430                                          datatype));
431 
432         // check resource index
433         String query = "";
434         if (isLiteral) {
435             if (datatype != null) {
436                 query =
437                         String.format("* <%s> '%s'^^%s",
438                                       predicate,
439                                       object,
440                                       datatype);
441             } else {
442                 query = String.format("* <%s> '%s'", predicate, object);
443             }
444         } else {
445             query = String.format("* <%s> <%s>", predicate, object);
446         }
447 
448         TripleIterator triples = queryRI(query);
449         try {
450             assertTrue("Relationship not found in RI (query = " + query + ")",
451                        triples.hasNext());
452             while (triples.hasNext()) {
453                 assertEquals(triples.next().getSubject().stringValue(),
454                              subjectAsURI(subject));
455             }
456         } finally {
457             triples.close();
458         }
459     }

464     private String subjectAsURI(String subj) {
465         // already a uri?
466         if (subj.startsWith(Constants.FEDORA.uri)) {
467             return subj;
468         }
469         // no, convert to uri
470         return PID.toURI(subj);

474     private void getRelationship(String subject,
475                                  String predicate,
476                                  String object,
477                                  boolean isLiteral,
478                                  String datatype) throws Exception {
479         addRelationship(subject, predicate, object, isLiteral, datatype);
480         List<RelationshipTuple> tuples = apim.getRelationships(subject, predicate);
481         assertNotNull(tuples);
482         assertEquals(1, tuples.size());
483         assertEquals(subjectAsURI(subject), tuples.get(0).getSubject());
484         assertEquals(predicate, tuples.get(0).getPredicate());
485         assertEquals(object, tuples.get(0).getObject());
486         assertEquals(isLiteral, tuples.get(0).isIsLiteral());
487         assertEquals(datatype, tuples.get(0).getDatatype());
488     }

503     private TripleIterator queryRI(String query) throws Exception {
504         FedoraClient client = getFedoraClient();
505         InputStream results =
506                 client.get(RISEARCH_QUERY + URLEncoder.encode(query, "UTF-8"),
507                            true);
