115     /**
116      * Validate the provided options and perform any necessary startup tasks.
117      */
118     public void start(Map<String, String> options)
119     throws ResourceIndexException {
120         // validate options
121 
122         // do startup tasks
123 
124         String levelValue;
125         if (m_riConfig == null){ //must have been configured outside fcfg
126             Module riModule = m_context.getBean(moduleName,Module.class);
127             if (riModule != null){
128                 logger.warn("ModuleConfiguration bean unavailable; getting Module bean");
129                 levelValue = riModule.getParameter("level");
130             }
131             else {
132                 logger.error("Cannot load ResourceIndex module definition from Spring config or Fedora config");
133                 throw new ResourceIndexException("Cannot locate ResourceIndex module definition in Spring config or Fedora config");
134             }
135         }
136         else {
137             levelValue = m_riConfig.getParameter("level",Parameter.class).getValue();
138         }
139         int riLevel = Integer.parseInt(levelValue);
140 
141         Map<String, String> aliasMap = new HashMap<String, String>();
142         Iterator<Parameter> it = m_riConfig.getParameters(Parameter.class).iterator();
143         Parameter p;
144         while (it.hasNext()) {
145             p = it.next();
146             String pName = p.getName();
147             String[] parts = pName.split(":");
148             if (parts.length == 2 && parts[0].equals("alias")) {
149                 aliasMap.put(parts[1], p.getValue(p.getIsFilePath()));
150             }
151         }
152 
153 
154         System.out.println("Initializing triplestore interface...");
155         try {
156             if (m_conn instanceof MulgaraConnector){
157                 String path = m_conn.getConfiguration().get("path");
158                 dropIndex(path);
159             }
160 
161             m_ri = new ResourceIndexImpl(m_conn, m_generator, riLevel, false);
162             m_ri.setAliasMap(aliasMap);
163         } catch (Exception e) {
164             logger.error("Failed to initialize new Resource Index",e);
165             e.printStackTrace(System.err);
166             throw new ResourceIndexException("Failed to initialize new Resource Index",
167                                              e);
168         }
169 
170     }

172     private void dropIndex(String tsPath){
173         if (tsPath == null) {
174             BufferedReader reader =
175                 new BufferedReader(new InputStreamReader(System.in));
176             System.out.println();
177             System.out

181             System.out

183             System.out
184             .println("      for this type of triplestore.  Press enter");
185             System.out.println("      when finished.");
186             try {
187                 reader.readLine();
188             } catch (IOException e) {
189             }
190             System.out.println("OK, continuing...");
191         } else {
192             System.out.println("Clearing directory " + tsPath + "...");
193             deleteDirectory(tsPath);

218     private boolean deleteDirectory(String directory) {
219 
220         boolean result = false;
221 
222         if (directory != null) {
223             File file = new File(directory);
224             if (file.exists() && file.isDirectory()) {
225                 // 1. delete content of directory:
226                 File[] files = file.listFiles();
227                 result = true; //init result flag
228                 int count = files.length;
229                 for (int i = 0; i < count; i++) { //for each file:
230                     File f = files[i];
231                     if (f.isFile()) {
232                         result = result && f.delete();
233                     } else if (f.isDirectory()) {
234                         result = result && deleteDirectory(f.getAbsolutePath());
235                     }
236                 }//next file
237 
238                 file.delete(); //finally delete (empty) input directory
239             }//else: input directory does not exist or is not a directory
240         }//else: no input value
241 
242         return result;
243     }//deleteDirectory()
