95      private final String getContextId(EvaluationCtx context) {
96          URI contextIdType = null;
97          URI contextIdId = null;
98          try {
99              contextIdType = new URI(StringAttribute.identifier);
100         } catch (URISyntaxException e) {
101             logger.debug("ContextAttributeFinder:getContextId" + " exit on "
102                     + "couldn't make URI for contextId type");
103         }
104         try {
105             contextIdId = new URI(Constants.ACTION.CONTEXT_ID.uri);
106         } catch (URISyntaxException e) {
107             logger.debug("ContextAttributeFinder:getContextId" + " exit on "
108                     + "couldn't make URI for contextId itself");
109         }
110         logger.debug("ContextAttributeFinder:findAttribute"
111                 + " about to call getAttributeFromEvaluationCtx");
112 
113         EvaluationResult attribute =
114                 context.getActionAttribute(contextIdType, contextIdId, null);
115         Object element = getAttributeFromEvaluationResult(attribute);
116         if (element == null) {
117             logger.debug("ContextAttributeFinder:getContextId" + " exit on "
118                     + "can't get contextId on request callback");
119             return null;
120         }
121 
122         if (!(element instanceof StringAttribute)) {
123             logger.debug("ContextAttributeFinder:getContextId" + " exit on "
124                     + "couldn't get contextId from xacml request "
125                     + "non-string returned");
126             return null;
127         }
128 
129         String contextId = ((StringAttribute) element).getValue();
130 
131         if (contextId == null) {
132             logger.debug("ContextAttributeFinder:getContextId" + " exit on "
133                     + "null contextId");
134             return null;
135         }
136 
137         if (!validContextId(contextId)) {
138             logger.debug("ContextAttributeFinder:getContextId" + " exit on "
139                     + "invalid context-id");
140             return null;
141         }
142 
143         return contextId;
144     }

146     private final boolean validContextId(String contextId) {
147         if (contextId == null) {
148             return false;
149         }
150         if ("".equals(contextId)) {
151             return false;
152         }
153         if (" ".equals(contextId)) {
154             return false;
155         }
156         return true;
157     }

159     @Override
160     protected final Object getAttributeLocally(int designatorType,
161                                                String attributeId,
162                                                URI resourceCategory,
163                                                EvaluationCtx ctx) {
164         logger.debug("getAttributeLocally context");
165         String contextId = getContextId(ctx);
166         logger.debug("contextId=" + contextId + " attributeId=" + attributeId);
167         if (contextId == null || contextId.equals("")) {
168             return null;
169         }
170         Context context = m_contexts.getContext(contextId);
171         logger.debug("got context");
172         Object values = null;
173         logger.debug("designatorType" + designatorType);
174         switch (designatorType) {
175             case AttributeDesignator.SUBJECT_TARGET:
176                 if (0 > context.nSubjectValues(attributeId)) {
177                     values = null;
178                 } else {
179                     logger.debug("getting n values for " + attributeId + "="
180                             + context.nSubjectValues(attributeId));
181                     switch (context.nSubjectValues(attributeId)) {
182                         case 0:
183                             values = null;
184                             /*
185                              * values = new String[1]; ((String[])values)[0] =
186                              * Authorization.UNDEFINED;
187                              */
188                             break;
189                         case 1:
190                             values = new String[1];
191                             ((String[]) values)[0] =
192                                     context.getSubjectValue(attributeId);
193                             break;
194                         default:
195                             values = context.getSubjectValues(attributeId);
196                     }
197                     if (logger.isDebugEnabled()) {
198                         if (values == null) {
199                             logger.debug("RETURNING NO VALUES FOR " + attributeId);
200                         } else {
201                             StringBuffer sb = new StringBuffer();
202                             sb.append("RETURNING " + ((String[]) values).length
203                                     + " VALUES FOR " + attributeId + " ==");

205                                 sb.append(" " + ((String[]) values)[i]);
206                             }

208                         }
209                     }
210                 }
211                 break;
212             case AttributeDesignator.ACTION_TARGET:
213                 if (0 > context.nActionValues(attributeId)) {
214                     values = null;

282         } else if (values instanceof String[]) {

285                 logger.debug("another string value={}", ((String[]) values)[i]);
286             }
287         } else {

289         }
