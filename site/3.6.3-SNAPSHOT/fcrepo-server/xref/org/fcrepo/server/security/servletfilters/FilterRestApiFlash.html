24  /**
25   * Determines if the client is a flash player (based on a query string
26   * value of 'flash' being set to true) and wraps the response to
27   * guarantee only 2xx range response codes.
28   *
29   * Responses which would normally return with a status outside of the 2xx
30   * range are updated such that the response code is set to 200 and the
31   * response body ends with the string "::ERROR(code)" where code
32   * is replaced by the actual response code.
33   *
34   * @author Bill Branan
35   */
36  public class FilterRestApiFlash implements Filter {
37  
38      private static final Logger logger =
39              LoggerFactory.getLogger(FilterRestApiFlash.class);
40  
41      /**
42       * Required Filter method
43       */
44      public void init(FilterConfig arg0) throws ServletException {
45      }
46  
47      /**
48       * Perform flash client response filtering
49       */
50      public void doFilter(ServletRequest request,
51                           ServletResponse response,
52                           FilterChain filterChain)
53      throws IOException, ServletException {
54          if (logger.isDebugEnabled()) {
55              logger.debug("Entering FilterRestApiFlash.doThisSubclass()");
56          }
57  
58          HttpServletRequest httpRequest = (HttpServletRequest)request;
59          HttpServletResponse httpResponse = (HttpServletResponse)response;
60          String queryString = httpRequest.getQueryString();
61          if(queryString != null &&
62             queryString.contains("flash=true")) {
63              StatusHttpServletResponseWrapper sHttpResponse =
64                  new StatusHttpServletResponseWrapper(httpResponse);
65  
66              filterChain.doFilter(request, sHttpResponse);
67  
68              if(sHttpResponse.status != sHttpResponse.realStatus) {
69                  // Append the error indicator with real status
70                  try {
71                      ServletOutputStream out = sHttpResponse.getOutputStream();
72                      out.print("::ERROR("+sHttpResponse.realStatus+")");
73                  } catch(IllegalStateException ise) {
74                      PrintWriter out = sHttpResponse.getWriter();
75                      out.print("::ERROR("+sHttpResponse.realStatus+")");
76                  }
77              }
78          } else {
79              filterChain.doFilter(request, response);
80          }
81      }
82  
83      /**
84       * Required Filter method
85       */
86      public void destroy() {
87      }
88  
89      /**
90       * Ensures that all returned HTTP status codes are in the 2xx range,
91       * this allows Flash-based client applications to have access to
92       * error responses.
93       */
94      private static class StatusHttpServletResponseWrapper
95      extends HttpServletResponseWrapper {
96          private int realStatus;
97          private int status;
98  
99        public StatusHttpServletResponseWrapper(HttpServletResponse response) {
100           super(response);
101       }
102 
103       @Override
104       public void setStatus(final int statusCode) {
105         realStatus = statusCode;
106         if ((statusCode - 200) < 0 ||
107             (statusCode - 200) >= 100) {
108             // Set code to 200 for all responses
109             status = SC_OK;
110             super.setStatus(status);
111         } else {
112             status = statusCode;
113         }
114       }
115     }
