67      private static final HttpMethodBase setup(HttpClient client,
68                                                URL url,
69                                                Map requestParameters,
70                                                Cookie[] requestCookies) {
71          logger.debug("Entered setup()");
72          HttpMethodBase method = null;
73          if (requestParameters == null) {
74              logger.debug("Using GetMethod; requestParameters == null");
75              method = new GetMethod(url.toExternalForm());
76          } else {
77              logger.debug("Using PostMethod; requestParameters specified");
78              method = new PostMethod(url.toExternalForm()); // "http://localhost:8080/"
79  
80              Part[] parts = new Part[requestParameters.size()];
81              Iterator iterator = requestParameters.keySet().iterator();
82              for (int i = 0; iterator.hasNext(); i++) {
83                  String fieldName = (String) iterator.next();
84                  String fieldValue = (String) requestParameters.get(fieldName);
85                  StringPart stringPart = new StringPart(fieldName, fieldValue);
86                  parts[i] = stringPart;
87                  logger.debug("Adding Post parameter {} = {}", fieldName, fieldValue);
88                  ((PostMethod) method).addParameter(fieldName, fieldValue); //old way
89              }
90          }
91          HttpState state = client.getState();
92          for (Cookie cookie : requestCookies) {
93              state.addCookie(cookie);
94          }
95          return method;
96      }

98      public final void connect(String urlString,
99                                Map requestParameters,
100                               Cookie[] requestCookies,
101                               String truststoreLocation,
102                               String truststorePassword) {
103         if (logger.isDebugEnabled()) {
104             logger.debug("Entered .connect() " + " url=="
105                     + urlString + " requestParameters==" + requestParameters
106                     + " requestCookies==" + requestCookies);
107         }
108         responseCookies2 = null;
109         URL url = null;
110         try {
111             url = new URL(urlString);
112         } catch (MalformedURLException mue) {
113             logger.error("Malformed url: " + urlString, mue);
114         }
115 
116         if (urlString.startsWith("https:") && null != truststoreLocation
117                 && !"".equals(truststoreLocation) && null != truststorePassword
118                 && !"".equals(truststorePassword)) {
119             logger.debug("setting " + FilterPubcookie.TRUSTSTORE_LOCATION_KEY
120                     + " to " + truststoreLocation);
121             System.setProperty(FilterPubcookie.TRUSTSTORE_LOCATION_KEY,
122                                truststoreLocation);
123             logger.debug("setting " + FilterPubcookie.TRUSTSTORE_PASSWORD_KEY
124                     + " to " + truststorePassword);
125             System.setProperty(FilterPubcookie.TRUSTSTORE_PASSWORD_KEY,
126                                truststorePassword);
127 
128             logger.debug("setting " + FilterPubcookie.KEYSTORE_LOCATION_KEY
129                     + " to " + truststoreLocation);
130             System.setProperty(FilterPubcookie.KEYSTORE_LOCATION_KEY,
131                                truststoreLocation);
132             logger.debug("setting " + FilterPubcookie.KEYSTORE_PASSWORD_KEY
133                     + " to " + truststorePassword);
134             System.setProperty(FilterPubcookie.KEYSTORE_PASSWORD_KEY,
135                                truststorePassword);
136 
137             System.setProperty("javax.net.debug",
138                                "ssl,handshake,data,trustmanager");
139 
140         } else {
141             logger.debug("DIAGNOSTIC urlString==" + urlString);
142             logger.debug("didn't set " + FilterPubcookie.TRUSTSTORE_LOCATION_KEY
143                     + " to " + truststoreLocation);
144             logger.debug("didn't set " + FilterPubcookie.TRUSTSTORE_PASSWORD_KEY
145                     + " to " + truststorePassword);
146         }
147 
148         HttpClient client = new HttpClient();
149         logger.debug(".connect() requestCookies==" + requestCookies);
150         HttpMethodBase method =
151                 setup(client, url, requestParameters, requestCookies);
152         int statusCode = 0;
153         try {
154             client.executeMethod(method);
155             statusCode = method.getStatusCode();
156         } catch (Exception e) {
157             logger.error("failed original connect, url==" + urlString, e);
158         }
159 
160         logger.debug("status code==" + statusCode);
161 
162         if (302 == statusCode) {
163             Header redirectHeader = method.getResponseHeader("Location");
164             if (redirectHeader != null) {
165                 String redirectString = redirectHeader.getValue();
166                 if (redirectString != null) {
167                     URL redirectURL = null;
168                     try {
169                         redirectURL = new URL(redirectString);
170                         method =
171                                 setup(client,
172                                       redirectURL,
173                                       requestParameters,
174                                       requestCookies);
175                     } catch (MalformedURLException mue) {
176                         logger.error(".connect() malformed redirect url: " + urlString);
177                     }
178                     statusCode = 0;
179                     try {
180                         client.executeMethod(method);
181                         statusCode = method.getStatusCode();
182                         logger.debug(".connect() (on redirect) statusCode==" + statusCode);
183                     } catch (Exception e) {
184                         logger.error(".connect() "
185                                 + "failed redirect connect");
186                     }
187                 }
188             }
189         }
190         if (statusCode == 200) { // this is either the original, non-302, status code or the status code after redirect
191             String content = null;
192             try {
193                 content = method.getResponseBodyAsString();
194             } catch (IOException e) {
195                 logger.error("Error getting content", e);
196                 return;
197             }
198             if (content == null) {
199                 logger.error("Content is null");
200                 return;
201             } else {
202                 Tidy tidy = null;
203                 try {
204                     tidy = new Tidy();
205                 } catch (Throwable t) {
206                     logger.error("Error creating Tidy instance?!", t);
207                 }
208                 byte[] inputBytes = content.getBytes();
209                 ByteArrayInputStream inputStream =
210                         new ByteArrayInputStream(inputBytes);
211                 responseDocument = tidy.parseDOM(inputStream, null); //use returned root node as only output
212             }
213             HttpState state = client.getState();
214             try {
215                 responseCookies2 = method.getRequestHeaders();
216                 if (logger.isDebugEnabled()) {
217                     for (Header element : responseCookies2) {
218                         logger.debug("Header: {}={}", element.getName(), element.getValue());
219                     }
220                 }
221                 responseCookies = state.getCookies();
222                 logger.debug(this.getClass().getName()
223                         + ".connect() responseCookies==" + responseCookies);
224             } catch (Throwable t) {
225                 logger.error(this.getClass().getName() + ".connect() exception=="
226                         + t.getMessage());
227                 if (t.getCause() != null) {
228                     logger.error(this.getClass().getName() + ".connect() cause=="
229                             + t.getCause().getMessage());
230                 }
231             }
232             completedFully = true;
233             logger.debug(this.getClass().getName() + ".connect() completedFully=="
234                     + completedFully);
235         }
236     }
