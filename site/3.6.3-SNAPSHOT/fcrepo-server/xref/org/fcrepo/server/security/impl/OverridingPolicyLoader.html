37      @Override
38      public Map<String, AbstractPolicy> loadPolicies(PolicyParser policyParser, boolean validate, File dir)
39              throws IOException, ValidationException {
40          Map<String,AbstractPolicy> policies = new HashMap<String,AbstractPolicy>();
41          File defaultDir = new File(dir,"default");
42          if (defaultDir.exists()){
43              policies.putAll(loadPolicies(policyParser, validate,defaultDir));
44          }
45          for (File file: dir.listFiles()) {
46              if (!file.getName().equals("default")){
47                  if (file.isDirectory()) {
48                      policies.putAll(loadPolicies(policyParser, validate, file));
49                  } else {
50                      if (file.getName().endsWith(".xml")) {
51                          logger.info("Loading policy: {}", file.getPath());
52                          InputStream policyStream = new FileInputStream(file);
53                          AbstractPolicy policy = policyParser.parse(policyStream, validate);
54                          logger.info("Loaded policy ID: {}", policy.getId());
55                          String key = null;
56                          if (m_override == Strategy.FILENAME) {
57                              key = file.getName();
58                          }
59                          else key = policy.getId().toString();
60                          policies.put(key, policy);
61                      }
62                  }
63              }
64          }
65          return policies;
66      }
