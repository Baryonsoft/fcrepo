99          String m = "L init() ";
100         try {
101             logger.debug(m + ">");
102             super.init(filterConfig);
103             m = FilterSetup.getFilterNameAbbrev(FILTER_NAME) + " init() ";
104             inited = false;
105             if (!initErrors) {
106                 Set temp = new HashSet();
107                 if (ATTRIBUTES2RETURN == null) {
108                     ATTRIBUTES2RETURN = new String[0];
109                 } else {
110                     for (String element : ATTRIBUTES2RETURN) {
111                         temp.add(element);
112                     }
113                 }
114                 if (AUTHENTICATE && PASSWORD != null && !"".equals(PASSWORD)) {
115                     temp.add(PASSWORD);
116                 }
117                 DIRECTORY_ATTRIBUTES_NEEDED =
118                         (String[]) temp.toArray(StringArrayPrototype);
119 
120                 boolean haveBindMethod = false;
121                 if (SECURITY_AUTHENTICATION != null
122                         && !"".equals(SECURITY_AUTHENTICATION)) {
123                     haveBindMethod = true;
124                 }
125 
126                 boolean haveSuperUser = false;
127                 if (SECURITY_PRINCIPAL != null
128                         && !"".equals(SECURITY_PRINCIPAL)) {
129                     haveSuperUser = true;
130                 }
131 
132                 boolean haveSuperUserPassword = false;
133                 if (SECURITY_CREDENTIALS != null
134                         && !"".equals(SECURITY_CREDENTIALS)) {
135                     haveSuperUserPassword = true;
136                 }
137 
138                 boolean haveUserPasswordAttributeName = false;
139                 if (PASSWORD != null && !"".equals(PASSWORD)) {
140                     haveUserPasswordAttributeName = true;
141                 }
142 
143                 boolean commonBindConfigured = false;
144                 if (haveBindMethod && haveSuperUserPassword) {
145                     boolean error = false;
146                     if (!haveSuperUser) {
147                         error = true;
148                     }
149                     if (error) {
150                         initErrors = true;
151                     } else {
152                         commonBindConfigured = true;
153                     }
154                 }
155 
156                 boolean individualBindConfigured = false;
157                 boolean individualBindTestConfigured = false;
158 
159                 if (haveBindMethod && !haveSuperUserPassword) {
160                     if (haveSuperUser) {

165                 }

195         String m =
196                 FilterSetup.getFilterNameAbbrev(FILTER_NAME)
197                         + " initThisSubclass() ";
198         try {
199             logger.debug(m + ">");
200             logger.debug(m + key + "==" + value);

212                 BASE = value;
213                 setLocally = true;
214             } else if (USERID_KEY.equals(key)) {
215                 setLocally = true;
216             } else if (ATTRIBUTES2RETURN_KEY.equals(key)) {
217                 if (value.indexOf(",") < 0) {
218                     if ("".equals(value)) {
219                         ATTRIBUTES2RETURN = null;
220                     } else {
221                         ATTRIBUTES2RETURN = new String[1];
222                         ATTRIBUTES2RETURN[0] = value;
223                     }
224                 } else {
225                     ATTRIBUTES2RETURN = value.split(",");
226                 }
227                 setLocally = true;
228             } else if (GROUPS_NAME_KEY.equals(key)) {
229                 GROUPS_NAME = value;
230                 setLocally = true;
231             } else if (FILTER_KEY.equals(key)) {
232                 FILTER = value;
233                 setLocally = true;
234             } else if (PASSWORD_KEY.equals(key)) {

239                 setLocally = true;

263     private final String applyFilter(String filter, String[] args) {
264         String m =
265                 FilterSetup.getFilterNameAbbrev(FILTER_NAME)
266                         + " applyFilter() ";
267         String result = filter;
268         logger.debug(m + "result==" + result);
269         int i = args.length - 1;
270         for (; i >= 0; i--) {
271             String regex = "\\{" + Integer.toString(i) + "\\}";
272             logger.debug(m + "regex ==" + regex);
273             logger.debug(m + "arg ==" + args[i]);
274             result = result.replaceFirst(regex, args[i]);
275             logger.debug(m + "result==" + result);
276         }
277         return result;
278     }

298     private Hashtable getEnvironment(String userid, String password) {
299         String m =
300                 FilterSetup.getFilterNameAbbrev(FILTER_NAME)
301                         + " getEnvironment() ";
302         Hashtable env = null;
303 
304         try {
305             env = new Hashtable();
306             env.put(Context.INITIAL_CONTEXT_FACTORY,
307                     "com.sun.jndi.ldap.LdapCtxFactory");
308 
309             if (VERSION != null && !"".equals(VERSION)) {
310                 logger.debug(m + "ldap explicit version==" + VERSION);
311                 env.put(CONTEXT_VERSION_KEY, VERSION);
312             }
313             logger.debug(m + "ldap version==" + env.get(CONTEXT_VERSION_KEY));
314 
315             env.put(Context.PROVIDER_URL, URL);
316             logger.debug(m + "ldap url==" + env.get(Context.PROVIDER_URL));
317 
318             if (!bindRequired()) {
319                 logger.debug(m + "\"binding\" anonymously");
320             } else {
321                 env.put(Context.SECURITY_AUTHENTICATION,
322                         SECURITY_AUTHENTICATION);
323 
324                 String userForBind = null;
325                 String passwordForBind = null;
326                 if (!individualUserBind()) {
327                     userForBind = SECURITY_PRINCIPAL;
328                     passwordForBind = SECURITY_CREDENTIALS;
329                     logger.debug(m + "binding to protected directory");
330                 } else {
331                     passwordForBind = password;
332                     if (SECURITY_PRINCIPAL == null
333                             || "".equals(SECURITY_PRINCIPAL)) {
334                         userForBind = userid;
335                         logger.debug(m + "binding for real user");
336                     } else {
337                         //simulate test against user-bind at directory server
338                         userForBind = SECURITY_PRINCIPAL;
339                         logger.debug(m + "binding for --test-- user");
340                     }
341                 }
342                 env.put(Context.SECURITY_CREDENTIALS, passwordForBind);
343                 String[] parms = {userForBind};
344                 String userFormattedForBind = applyFilter(BIND_FILTER, parms);
345                 env.put(Context.SECURITY_PRINCIPAL, userFormattedForBind);
346             }
347             logger.debug(m + "bind w " + env.get(Context.SECURITY_AUTHENTICATION));
348             logger.debug(m + "user== " + env.get(Context.SECURITY_PRINCIPAL));
349             logger.debug(m + "passwd==" + env.get(Context.SECURITY_CREDENTIALS));
350         } catch (Throwable th) {
351             if (LOG_STACK_TRACES) {
352                 logger.error(m + "couldn't set up env for DirContext", th);
353             } else {
354                 logger.error(m + "couldn't set up env for DirContext"
355                         + th.getMessage());
356             }
357         } finally {
358             logger.debug(m + "< " + env);
359         }
360         return env;
361     }

416     private NamingEnumeration getBasicNamingEnumeration(String userid,
417                                                         String password,
418                                                         String filter,
419                                                         SearchControls searchControls,
420                                                         Hashtable env)
421             throws NamingException, Exception {
422         String m =
423                 FilterSetup.getFilterNameAbbrev(FILTER_NAME)
424                         + " getNamingEnumeration() ";
425         logger.debug(m + ">");
426         NamingEnumeration ne = null;
427         try {
428             DirContext ctx;
429             try {
430                 ctx = new InitialDirContext(env);
431             } catch (NamingException th) {
432                 String msg = "exception getting ldap context";
433                 if (LOG_STACK_TRACES) {
434                     logger.error(m + msg, th);
435                 } else {
436                     logger.error(m + msg + " " + th.getMessage());
437                 }
438                 throw th;
439             }
440             if (ctx == null) {
441                 logger.error(m + "unexpected null ldap context");
442                 throw new NamingException("");
443             }
444             try {
445                 ne = ctx.search(BASE, filter, searchControls);
446             } catch (NamingException th) {
447                 String msg = "exception getting ldap enumeration";
448                 if (LOG_STACK_TRACES) {
449                     logger.error(m + msg, th);
450                 } else {
451                     logger.error(m + msg + " " + th.getMessage());
452                 }
453                 throw th;
454             }
455             if (ne == null) {
456                 logger.error(m + "unexpected null ldap enumeration");
457                 throw new NamingException("");
458             }
459         } finally {
460             logger.debug(m + "< " + ne);
461         }
462         return ne;
463     }

465     private NamingEnumeration getNamingEnumeration(String userid,
466                                                    String password,
467                                                    String filter,
468                                                    SearchControls searchControls,
469                                                    Hashtable env)
470             throws NamingException, Exception {
471         String m =
472                 FilterSetup.getFilterNameAbbrev(FILTER_NAME)
473                         + " getNamingEnumeration() ";
474         logger.debug(m + ">");
475         // this condition is to -further- protect against behavior suggested by
476         // log from hull (see below for first-line protection)
477         // the idea here is to steer clear of possible trouble in underlying
478         // code and avoid calling ldap w/o a needed and practical password
479 
480         String msg = "[LDAP: error code 49 - Bind failed: ";
481 
482         if (!individualUserBind()) {
483             logger.info(m + "-not- binding individual user");
484         } else {
485             logger.info(m + "-binding- individual user");
486             if (password == null) {
487                 logger.debug(m + "null password");
488                 if (USE_FILTER.equalsIgnoreCase(PW_NULL)) {
489                     logger.debug(m + "-no- pre null password handling");
490                 } else {
491                     if (AUTHENTICATE) {
492                         logger.info(m + "-doing- pre null password handling");
493                         if (UNAUTHENTICATE_USER_UNCONDITIONALLY
494                                 .equalsIgnoreCase(PW_NULL)) {
495                             logger.info(m
496                                     + "pre unauthenticating for null password");
497                             throw new NamingException(msg + "null password]");
498                         } else if (SKIP_FILTER.equalsIgnoreCase(PW_NULL)) {
499                             logger.info(m + "pre ignoring for null passwd");
500                             throw new Exception(msg + "null password]");
501                         } else {
502                             assert true : "bad value for PW_NULL==" + PW_NULL;
503                         }
504                     }
505                 }
506             } else if ("".equals(password)) {
507                 logger.debug(m + "0-length password");
508                 if (USE_FILTER.equalsIgnoreCase(PW_0)) {
509                     logger.debug(m + "-no- pre 0-length password handling");
510                 } else {
511                     if (AUTHENTICATE) {
512                         logger.info(m + "-doing- pre 0-length password handling");
513                         if (UNAUTHENTICATE_USER_UNCONDITIONALLY
514                                 .equalsIgnoreCase(PW_0)) {
515                             logger.info(m + "pre unauthenticating for 0-length password");
516                             throw new NamingException(msg
517                                     + "0-length password]");
518                         } else if (SKIP_FILTER.equalsIgnoreCase(PW_0)) {
519                             logger.info(m + "pre ignoring for 0-length passwd");
520                             throw new Exception(msg + "0-length password]");
521                         } else {
522                             assert true : "bad value for PW_0==" + PW_0;
523                         }
524                     }
525                 }
526             } else {
527                 assert password.length() > 0;
528             }
529         }
530 
531         NamingEnumeration ne = null;
532         try {
533             ne =
534                     getBasicNamingEnumeration(userid,
535                                               password,
536                                               filter,
537                                               searchControls,
538                                               env);
539             assert ne != null;
540             if (ne.hasMoreElements()) {
541                 logger.debug(m + "enumeration has elements");
542             } else {
543                 logger.debug(m + "enumeration has no elements, yet no exceptions");
544                 if (bindRequired() && !individualUserBind()) {
545                     logger.debug(m + "failed security bind");
546                     throw new NamingException(msg + "failed security bind]");
547                 }
548                 if (!AUTHENTICATE) {
549                     logger.debug(m
550                             + "user authentication -not- done by this filter");
551                 } else {
552                     logger.debug(m + "user authentication -done- by this filter");
553                     if (!bindRequired()) {
554                         logger.debug(m + "but -not- binding");
555                     } else {
556                         logger.debug(m + "-and- binding");
557                         if (SKIP_FILTER.equalsIgnoreCase(EMPTY_RESULTS)) {
558                             logger.debug(m + "passing thru for EMPTY_RESULTS");
559                             throw new Exception(msg + "null password]");
560                         } else if (UNAUTHENTICATE_USER_UNCONDITIONALLY
561                                 .equalsIgnoreCase(EMPTY_RESULTS)) {
562                             logger.debug(m + "failing for EMPTY_RESULTS");
563                             throw new NamingException(msg + "null password]");
564                         } else if (USE_FILTER.equalsIgnoreCase(EMPTY_RESULTS)) {
565                             logger.debug(m + "passing for EMPTY_RESULTS");
566                             //XXXXXXXXXXXXXXXXXXXXXXXXXXXXXXXXX
567                         } else if (UNAUTHENTICATE_USER_CONDITIONALLY
568                                 .equalsIgnoreCase(EMPTY_RESULTS)) {
569                             if (ATTRIBUTES2RETURN == null
570                                     || ATTRIBUTES2RETURN.length < 1) {
571                                 logger.debug(m + "fair enough");
572                             } else {
573                                 throw new NamingException(msg + "expected some");
574                             }
575                         } else {
576                             assert true : "bad value for EMPTY_RESULTS=="
577                                     + EMPTY_RESULTS;
578                         }
579                     }
580                 }
581             }
582         } finally {
583             logger.debug(m + "< " + ne);
584         }
585         return ne;
586     }

588     private static Boolean comparePassword(Attributes attributes,
589                                            String password,
590                                            String passwordAttribute)
591             throws PasswordComparisonException {
592         String m = "- comparePassword() ";
593         logger.debug(m + ">");
594         Boolean rc = null;
595         try {
596             logger.debug(m + "looking for return attribute==" + passwordAttribute);
597             Attribute attribute = attributes.get(passwordAttribute);
598             if (attribute == null) {
599                 logger.error(m + "null object");
600             } else {
601                 int size = attribute.size();
602                 logger.debug(m + "object with n==" + size);
603                 for (int j = 0; j < size; j++) {
604                     Object o = attribute.get(j);
605                     if (password.equals(o.toString())) {
606                         logger.debug(m + "compares true");
607                         if (rc == null) {
608                             logger.debug(m + "1st comp:  authenticate");
609                             rc = Boolean.TRUE;
610                         } else {
611                             logger.error(m + "dup comp:  keep previous rc==" + rc);
612                         }
613                     } else {
614                         logger.debug(m + "compares false, -un-authenticate");
615                         if (rc == null) {
616                             logger.debug(m + "1st comp (fyi)");
617                         } else {
618                             logger.error(m + "dup comp (fyi)");
619                         }
620                         rc = Boolean.FALSE;
621                     }
622                 }
623             }
624         } catch (Throwable th) {
625             logger.error(m + "resetting to null rc==" + rc + th.getMessage());
626             throw new PasswordComparisonException("in ldap servlet filter", th);
627         } finally {
628             logger.debug(m + "< " + rc);
629         }
630         return rc;
631     }

633     private void getAttributes(Attributes attributes, Map map) throws Throwable {
634         String m =
635                 FilterSetup.getFilterNameAbbrev(FILTER_NAME)
636                         + " getAttributes() ";
637         logger.debug(m + ">");
638         try {
639             for (String key : ATTRIBUTES2RETURN) {
640                 logger.debug(m + "looking for return attribute==" + key);
641                 Attribute attribute = attributes.get(key);
642                 if (attribute == null) {
643                     logger.error(m + "null object...continue to next attr sought");
644                     continue;
645                 }
646                 if (GROUPS_NAME != null && !"".equals(GROUPS_NAME)) {
647                     key = GROUPS_NAME;
648                     logger.debug(m
649                             + "values collected and interpreted as groups=="
650                             + key);
651                 }
652                 Set values;
653                 if (map.containsKey(key)) {
654                     logger.debug(m + "already a value-set for attribute==" + key);
655                     values = (Set) map.get(key);
656                 } else {
657                     logger.debug(m + "making+storing a value-set for attribute=="
658                             + key);
659                     values = new HashSet();
660                     map.put(key, values);
661                 }
662                 int size = attribute.size();
663                 logger.debug(m + "object with n==" + size);
664                 for (int j = 0; j < size; j++) {
665                     Object o = attribute.get(j);
666                     values.add(o);
667                     logger.debug(m + "added value==" + o.toString() + ", class=="
668                             + o.getClass().getName());
669                 }
670             }
671         } finally {
672             logger.debug(m + "<");
673         }
674     }

676     private Boolean processNamingEnumeration(NamingEnumeration ne,
677                                              String password,
678                                              Boolean authenticated,
679                                              Map map) {
680         String m =
681                 FilterSetup.getFilterNameAbbrev(FILTER_NAME)
682                         + " processNamingEnumeration() ";
683         logger.debug(m + ">");
684         try {
685             boolean errorOnSomeComparison = false;
686             while (ne.hasMoreElements()) {
687                 logger.debug(m + "another element");
688                 SearchResult s = null;
689                 try {
690                     Object o = ne.nextElement();
691                     logger.debug(m + "got a " + o.getClass().getName());
692                     s = (SearchResult) o;
693                 } catch (Throwable th) {
694                     logger.error(m + "naming enum contains obj not SearchResult");
695                     continue;
696                 }
697                 Attributes attributes = s.getAttributes();
698                 getAttributes(attributes, map);
699                 if (individualUserComparison()) {
700                     Boolean temp = null;
701                     try {
