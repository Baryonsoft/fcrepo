61      /**
62       * Post-Initializes the Module based on configuration parameters. The
63       * implementation of this method is dependent on the schema used to define
64       * the parameter names for the role of
65       * <code>org.fcrepo.server.storage.DefaultBackendSecurity</code>.
66       *
67       * @throws ModuleInitializationException
68       *         If initialization values are invalid or initialization fails for
69       *         some other reason.
70       */
71      @Override
72      public void postInitModule() throws ModuleInitializationException {
73  
74          try {
75              Server s_server = getServer();
76              logger.debug("DefaultBackendSecurity initialized");
77              String fedoraHome = Constants.FEDORA_HOME;
78              if (fedoraHome == null) {
79                  throw new ModuleInitializationException("[DefaultBackendSecurity] Module failed to initialize: "
80                                                                  + "FEDORA_HOME is undefined",
81                                                          getRole());
82              } else {
83                  m_beSecurityPath = fedoraHome + "/server/config/beSecurity.xml";
84              }
85              logger.debug("m_beSecurityPath: " + m_beSecurityPath);
86  
87              String validate = getParameter("beSecurity_validation");
88              if (validate != null) {
89                  if (!validate.equals("true") && !validate.equals("false")) {
90                      logger.warn("Validation setting for backend "
91                                      + "security configuration file must be either \"true\" or \"false\". "
92                                      + "Value specified was: \"" + validate
93                                      + "\". Validation is defaulted to "
94                                      + "\"false\".");
95                  } else {
96                      m_validate = new Boolean(validate).booleanValue();
97                  }
98              } else {
99                  logger.warn("Validation setting for backend "
100                                 + "security configuration file was not specified. Validation is defaulted to "
101                                 + "\"false\".");
102             }
103             logger.debug("beSecurity_validate: " + m_validate);
104 
105             m_encoding = getParameter("beSecurity_char_encoding");
106             if (m_encoding == null) {
107                 m_encoding = "utf-8";
108                 logger.warn("Character encoding for backend "
109                         + "security configuration file was not specified. Encoding defaulted to "
110                         + "\"utf-8\".");
111             }
112             logger.debug("beSecurity_char_encoding: " + m_encoding);
113 
114             // initialize static BackendSecuritySpec instance
115             setBackendSecuritySpec();
116             if (logger.isDebugEnabled()) {
117                 Set roleList = beSS.listRoleKeys();
118                 Iterator iter = roleList.iterator();
119                 while (iter.hasNext()) {
120                     logger.debug("beSecurity ROLE: " + iter.next());
121                 }
122             }
123 
124         } catch (Throwable th) {
125             throw new ModuleInitializationException("[DefaultBackendSecurity] "
126                     + "BackendSecurity "
127                     + "could not be instantiated. The underlying error was a "
128                     + th.getClass().getName() + "The message was \""
129                     + th.getMessage() + "\".", getRole());
130         }
131     }

133     /**
134      * Parses the beSecurity configuration file.
135      *
136      * @throws BackendSecurityParserException
137      *         If an error occurs in attempting to parse the beSecurity
138      *         configuration file.
139      */
140     public BackendSecuritySpec parseBeSecurity()
141             throws BackendSecurityParserException {
142 
143         try {
144             BackendSecurityDeserializer bsd =
145                     new BackendSecurityDeserializer(m_encoding, m_validate);
146             return bsd.deserialize(m_beSecurityPath);
147 
148         } catch (Throwable th) {
149             throw new BackendSecurityParserException("[DefaultBackendSecurity] "
150                     + "An error has occured in parsing the backend security "
151                     + "configuration file located at \""
152                     + m_beSecurityPath
153                     + "\". "
154                     + "The underlying error was a "
155                     + th.getClass().getName()
156                     + "The message was \""
157                     + th.getMessage() + "\".");
158         }
159     }
