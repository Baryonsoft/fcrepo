27       * Prints a "FEDORA STARTUP ERROR" to STDERR along with the stacktrace of
28       * the Throwable (if given) and finally, throws a ServletException.
29       */
30      protected void failStartup(String message, Throwable th)
31              throws ServletException {
32          System.err.println("\n**************************");
33          System.err.println("** FEDORA STARTUP ERROR **");
34          System.err.println("**************************\n");
35          System.err.println(message);
36          if (th == null) {
37              System.err.println();
38              throw new ServletException(message);
39          } else {
40              th.printStackTrace();
41              System.err.println();
42              throw new ServletException(message, th);
43          }
44      }

46      @Override
47      public void init(ServletConfig config) throws ServletException {
48          super.init(config);
49          File fedoraHomeDir = getFedoraHomeDir();
50          // get file for writing startup status
51          try {
52              m_status = new ServerStatusFile(new File(fedoraHomeDir, "server"));
53          } catch (Throwable th) {
54              failStartup("Error initializing server status file", th);
55          }
56  
57          try {
58              m_appContext = WebApplicationContextUtils.getRequiredWebApplicationContext(config.getServletContext());
59              m_server = (Server)m_appContext.getBean("org.fcrepo.server.Server");
60              if (m_server == null) failStartup("Could not retrieve org.fcrepo.server.Server bean",null);
61          } catch (Throwable th) {
62              String msg = "Fedora startup failed";
63              try {
64                  m_status.appendError(ServerState.STARTUP_FAILED, th);
65              } catch (Exception e) {
66              }
67              failStartup(msg, th);
68          }
69      }

77      /**
78       * Validates and returns the value of FEDORA_HOME.
79       *
80       * @return the FEDORA_HOME directory.
81       * @throws ServletException
82       *         if FEDORA_HOME (or fedora.home) was not set, does not denote an
83       *         existing directory, or is not writable by the current user.
84       */
85      private File getFedoraHomeDir() throws ServletException {
86  
87          String fedoraHome = Constants.FEDORA_HOME;
88          if (fedoraHome == null) {
89              failStartup("FEDORA_HOME was not configured properly.  It must be "
90                      + "set via the fedora.home servlet init-param (preferred), "
91                      + "the fedora.home system property, or the FEDORA_HOME "
92                      + "environment variable.", null);
93          }
94          File fedoraHomeDir = new File(fedoraHome);
95          if (!fedoraHomeDir.isDirectory()) {
96              failStartup("The FEDORA_HOME directory, " + fedoraHomeDir.getPath()
97                      + " does not exist", null);
98          }
99          File writeTest = new File(fedoraHomeDir, "writeTest.tmp");
100         String writeErrorMessage =
101                 "The FEDORA_HOME directory, " + fedoraHomeDir.getPath()
102                         + " is not writable by " + "the current user, "
103                         + System.getProperty("user.name");
104         try {
105             writeTest.createNewFile();
106             if (!writeTest.exists()) {
107                 throw new IOException("");
108             }
109             writeTest.delete();
110         } catch (IOException e) {
111             failStartup(writeErrorMessage, null);
112         }
113 
114         return fedoraHomeDir;
115     }
