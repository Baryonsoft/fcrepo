50  /**
51   * This servlet acts as a proxy to resolve the physical location of datastreams.
52   *
53   * <p>It requires a single parameter named <code>id</code> that denotes the
54   * temporary id of the requested datastresm. This id is in the form of a
55   * DateTime stamp. The servlet will perform an in-memory hashtable lookup
56   * using the temporary id to obtain the actual physical location of the
57   * datastream and then return the contents of the datastream as a MIME-typed
58   * stream. This servlet is invoked primarily by external mechanisms needing to
59   * retrieve the contents of a datastream.
60   *
61   * <p>The servlet also requires that an external mechanism request a datastream
62   * within a finite time interval of the tempID's creation. This is to lessen the
63   * risk of unauthorized access. The time interval within which a mechanism must
64   * respond is set by the Fedora configuration parameter named
65   * datastreamMediationLimit and is specified in milliseconds. If this
66   * parameter is not supplied it defaults to 5000 milliseconds.
67   *
68   * @author Ross Wayland
69   */
70  public class DatastreamResolverServlet
71          extends SpringServlet {

106                             .getModule("org.fcrepo.server.storage.DOManager");

108                     m_server.getParameter("datastreamMediationLimit");
109             if (expireLimit == null || expireLimit.equalsIgnoreCase("")) {
110                 logger.info("datastreamMediationLimit unspecified, using default "
111                         + "of 5 seconds");
112                 datastreamMediationLimit = 5000;
113             } else {
114                 datastreamMediationLimit = new Integer(expireLimit).intValue();
115                 logger.info("datastreamMediationLimit: "
116                         + datastreamMediationLimit);
117             }
118         } catch (Throwable th) {
119             logger.error("Error initializing servlet", th);
120         }

123     private static final boolean contains(String[] array, String item) {
124         boolean contains = false;
125         for (String element : array) {
126             if (element.equals(item)) {
127                 contains = true;
128                 break;
129             }
130         }
131         return contains;

207                 logger.debug("**************************** DatastreamResolverServlet dm.dsControlGroupType: {}", dm.dsControlGroupType);

219             // DatastreamResolverServlet maps to two distinct servlet mappings
220             // in fedora web.xml.
221             // getDS - is used when the backend service is incapable of
222             // basicAuth or SSL
223             // getDSAuthenticated - is used when the backend service has
224             // basicAuth and SSL enabled
225             // Since both the getDS and getDSAuthenticated servlet targets map
226             // to the same servlet
227             // code and the Context used to initialize policy enforcement is
228             // based on the incoming
229             // HTTPRequest, the code must provide special handling for requests
230             // using the getDS
231             // target. When the incoming URL to DatastreamResolverServlet
232             // contains the getDS target,
233             // there are several conditions that must be checked to insure that
234             // the correct role is
235             // assigned to the request before policy enforcement occurs.
236             // 1) if the mapped dsPhysicalLocation of the request is actually a
237             // callback to the
238             // Fedora server itself, then assign the role as
239             // BACKEND_SERVICE_CALL_UNSECURE so
240             // the basicAuth and SSL constraints will match those of the getDS
241             // target.
242             // 2) if the mapped dsPhysicalLocation of the request is actually a
243             // Managed Content
244             // or Inline XML Content datastream, then assign the role as
245             // BACKEND_SERVICE_CALL_UNSECURE so
246             // the basicAuth and SSL constraints will match the getDS target.
247             // 3) Otherwise, leave the targetrole unchanged.

249                     && (ServerUtility.isURLFedoraServer(dsPhysicalLocation)

251                             .equals("X"))) {
252                 if (logger.isDebugEnabled()) {
253                     logger.debug("*********************** Changed role from: "
254                             + dm.callbackRole + "  to: "
255                             + BackendPolicies.BACKEND_SERVICE_CALL_UNSECURE);
256                 }
257                 dm.callbackRole = BackendPolicies.BACKEND_SERVICE_CALL_UNSECURE;
258             }

260             // If callback is to fedora server itself and callback is over SSL,
261             // adjust the protocol and port
262             // on the URL to match settings of Fedora server. This is necessary
263             // since the SSL settings for the
264             // backend service may have specified basicAuth=false, but contained
265             // datastreams that are callbacks
266             // to the local Fedora server which requires SSL. The version of
267             // HttpClient currently in use does
268             // not handle autoredirecting from http to https so it is necessary
269             // to set the protocol and port
270             // to the appropriate secure port.
271             if (dm.callbackRole.equals(BackendPolicies.FEDORA_INTERNAL_CALL)) {
272                 if (dm.callbackSSL) {
273                     dsPhysicalLocation =
274                             dsPhysicalLocation.replaceFirst("http:", "https:");
275                     dsPhysicalLocation =
276                             dsPhysicalLocation
277                                     .replaceFirst(fedoraServerPort,
278                                                   fedoraServerRedirectPort);
279                     if (logger.isDebugEnabled()) {
280                         logger
281                                 .debug("*********************** DatastreamResolverServlet -- Was Fedora-to-Fedora call -- modified dsPhysicalLocation: "
282                                         + dsPhysicalLocation);
283                     }
284                 }
285             }
286             keyTimestamp = Timestamp.valueOf(ds.extractTimestamp(id));
287             currentTimestamp = new Timestamp(new Date().getTime());
288             logger.debug("dsPhysicalLocation=" + dsPhysicalLocation
289                     + "dsControlGroupType=" + dsControlGroupType);
290 
291             // Deny mechanism requests that fall outside the specified time
292             // interval.
293             // The expiration limit can be adjusted using the Fedora config
294             // parameter
295             // named "datastreamMediationLimit" which is in milliseconds.
296             long diff = currentTimestamp.getTime() - keyTimestamp.getTime();
297             logger.debug("Timestamp diff for mechanism's reponse: " + diff
298                     + " ms.");
299             if (diff > datastreamMediationLimit) {
300                 out = response.getWriter();
301                 response.setContentType(HTML_CONTENT_TYPE);
302                 out
303                         .println("<br><b>[DatastreamResolverServlet] Error:</b>"
304                                 + "<font color=\"red\"> Deployment has failed to respond "
305                                 + "to the DatastreamResolverServlet within the specified "
306                                 + "time limit of \""
307                                 + datastreamMediationLimit
308                                 + "\""
309                                 + "milliseconds. Datastream access denied.");
310                 logger.error("Deployment failed to respond to "
311                         + "DatastreamResolverServlet within time limit of "
312                         + datastreamMediationLimit);
313                 out.close();
314                 return;
315             }

318                 throw new AuthzOperationalException("no callbackRole for this ticket");
319             }

335                  * request.getUserPrincipal(); if (principal == null) { // no

338                  * if (roles == null) { roles = new String[0]; }
339                  */

342                         + request.getRemoteUser());
343                 /*
344                  * if
345                  * (((ExtendedHttpServletRequest)request).isUserInRole(targetRole)) {
346                  * logger.debug("DatastreamResolverServlet: user has required

348                  * exception in validating user"); throw new
349                  * AuthzDeniedException("wrong user for this ticket"); }
350                  */
351             }

354                 logger.debug("debugging backendService role");
355                 logger.debug("targetRole=" + targetRole);

357                 logger.debug("targetRolesLength=" + targetRolesLength);
358                 if (targetRolesLength > 0) {
359                     logger.debug("targetRoles[0]=" + targetRoles[0]);

363                 if (nSubjectValues > 0) {

478                             "[DatastreamResolverServlet]  The "
479                                     + "internal Fedora datastream id:  \""

492                                             context,

498                 int bytestream = 0;
499                 response.setContentType(d.DSMIME);
500                 outStream = response.getOutputStream();
501                 byte[] buffer = new byte[255];
502                 while ((bytestream = is.read(buffer)) != -1) {
503                     outStream.write(buffer, 0, bytestream);
504                 }
505                 buffer = null;
506                 is.close();
507             } else {
508                 out = response.getWriter();
509                 response.setContentType(HTML_CONTENT_TYPE);

511                         .println("<br>[DatastreamResolverServlet] Unknown "

526                     "[DatastreamResolverServlet] returned an error. The "
527                             + "underlying error was a  \""
528                             + th.getClass().getName() + "  The message was  \""
529                             + th.getMessage() + "\".  ";
530             throw new ServletException(message);
531         } finally {
532             if (out != null) {
533                 out.close();
534             }
535             if (outStream != null) {
536                 outStream.close();
537             }
538             DisseminationService.dsRegistry.remove(id);
