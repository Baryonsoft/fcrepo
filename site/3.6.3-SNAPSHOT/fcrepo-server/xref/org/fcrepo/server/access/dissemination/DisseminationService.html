152 	/**
153 	 * <p>
154 	 * Assembles a dissemination given an instance of <code>
155 	 * DisseminationBindingInfo</code> which has the dissemination-related
156 	 * information from the digital object and its associated Service Deployment
157 	 * object.
158 	 * </p>
159 	 * 
160 	 * @param context
161 	 *            The current context.
162 	 * @param PID
163 	 *            The persistent identifier of the digital object.
164 	 * @param h_userParms
165 	 *            A hashtable of user-supplied method parameters.
166 	 * @param dissBindInfoArray
167 	 *            The associated dissemination binding information.
168 	 * @return A MIME-typed stream containing the result of the dissemination.
169 	 * @throws ServerException
170 	 *             If unable to assemble the dissemination for any reason.
171 	 */
172 	public MIMETypedStream assembleDissemination(Context context, String PID,
173 			Hashtable<String, String> h_userParms,
174 			DisseminationBindingInfo[] dissBindInfoArray, String deploymentPID,
175 			ServiceDeploymentReader bmReader, String methodName)
176 			throws ServerException {
177 
178 		logger.debug("Started assembling dissemination");
179 
180 		String dissURL = null;
181 		String protocolType = null;
182 		DisseminationBindingInfo dissBindInfo = null;
183 		MIMETypedStream dissemination = null;
184 		boolean isRedirect = false;
185 
186 		if (logger.isDebugEnabled()) {
187 			printBindingInfo(dissBindInfoArray);
188 		}
189 
190 		if (dissBindInfoArray != null && dissBindInfoArray.length > 0) {
191 			String replaceString = null;
192 			int numElements = dissBindInfoArray.length;
193 
194 			// Get row(s) of binding info and perform string substitution
195 			// on DSBindingKey and method parameter values in WSDL
196 			// Note: In case where more than one datastream matches the
197 			// DSBindingKey or there are multiple DSBindingKeys for the
198 			// method, multiple rows will be present; otherwise there is only
199 			// a single row.
200 			for (int i = 0; i < dissBindInfoArray.length; i++) {
201 				m_authorization
202 						.enforce_Internal_DSState(context,
203 								dissBindInfoArray[i].dsID,
204 								dissBindInfoArray[i].dsState);
205 				dissBindInfo = dissBindInfoArray[i];
206 
207 				// Before doing anything, check whether we can replace any
208 				// placeholders in the datastream url with parameter values from
209 				// the request. This supports the special case where a
210 				// datastream's URL is dependent on user parameters, such
211 				// as when the datastream is actually a dissemination that
212 				// takes parameters.
213 				if (dissBindInfo.dsLocation != null
214 						&& (dissBindInfo.dsLocation.startsWith("http://") || dissBindInfo.dsLocation
215 								.startsWith("https://"))) {
216 					String[] parts = dissBindInfo.dsLocation.split("=\\("); // regex
217 																			// for
218 																			// =(
219 					if (parts.length > 1) {
220 						StringBuffer replaced = new StringBuffer();
221 						replaced.append(parts[0]);
222 						for (int x = 1; x < parts.length; x++) {
223 							replaced.append('=');
224 							int rightParenPos = parts[x].indexOf(")");
225 							if (rightParenPos != -1 && rightParenPos > 0) {
226 								String key = parts[x].substring(0,
227 										rightParenPos);
228 								String val = h_userParms.get(key);
229 								if (val != null) {
230 									// We have a match... so insert the
231 									// urlencoded value.
232 									try {
233 										replaced.append(URLEncoder.encode(val,
234 												"UTF-8"));
235 									} catch (UnsupportedEncodingException uee) {
236 										// won't happen: java always supports
237 										// UTF-8
238 									}
239 									if (rightParenPos < parts[x].length()) {
240 										replaced.append(parts[x]
241 												.substring(rightParenPos + 1));
242 									}
243 								} else {
244 									replaced.append('(');
245 									replaced.append(parts[x]);
246 								}
247 							} else {
248 								replaced.append('(');
249 								replaced.append(parts[x]);
250 							}
251 						}
252 						dissBindInfo.dsLocation = replaced.toString();
253 					}
254 				}
255 
256 				// Match DSBindingKey pattern in WSDL which is a string of the
257 				// form:
258 				// (DSBindingKey). Rows in DisseminationBindingInfo are sorted
259 				// alphabetically on binding key.
260 				String bindingKeyPattern = "\\(" + dissBindInfo.DSBindKey
261 						+ "\\)";
262 				if (i == 0) {
263 					// If addressLocation has a value of "LOCAL", this indicates
264 					// the associated operationLocation requires no
265 					// addressLocation.
266 					// i.e., the operationLocation contains all information
267 					// necessary
268 					// to perform the dissemination request. This is a special
269 					// case
270 					// used when the web services are generally mechanisms like
271 					// cgi-scripts,
272 					// java servlets, and simple HTTP GETs. Using the value of
273 					// LOCAL
274 					// in the address location also enables one to have
275 					// different methods
276 					// serviced by different hosts. In true web services like
277 					// SOAP, the
278 					// addressLocation specifies the host name of the service
279 					// and all
280 					// methods are served from that single host location.
281 					if (dissBindInfo.AddressLocation
282 							.equalsIgnoreCase(LOCAL_ADDRESS_LOCATION)) {
283 						dissURL = dissBindInfo.OperationLocation;
284 					} else {
285 						dissURL = dissBindInfo.AddressLocation
286 								+ dissBindInfo.OperationLocation;
287 
288 						/*
289 						 * Substitute real app server context if we detect
290 						 * '/fedora'. This is necessary here because
291 						 * DOTranslator does not scrub URLs that result from
292 						 * concatenating fragments from different locations in
293 						 * the file
294 						 */
295 						dissURL = dissURL.replaceAll(m_fedoraServerHost + ":"
296 								+ m_fedoraServerPort + "/fedora/",
297 								m_fedoraServerHost + ":" + m_fedoraServerPort
298 										+ "/" + m_fedoraAppServerContext + "/");
299 					}
300 					protocolType = dissBindInfo.ProtocolType;
301 				}
302 
303 				// Assess beSecurity for backend service and for datastreams
304 				// that may be parameters for the
305 				// backend service.
306 				//
307 				// dsMediatedCallbackHost - when dsMediation is in effect, all
308 				// M, X, and E type datastreams
309 				// are encoded as callbacks to the Fedora server to obtain the
310 				// datastream's contents. dsMediatedCallbackHost contains
311 				// protocol,
312 				// host, and port used for this type of backendservice-to-fedora
313 				// callback.
314 				// The specifics of protocol, host, and port are obtained from
315 				// the
316 				// beSecurity configuration file.
317 				// dsMediatedServletPath - when dsMediation is in effect, all M,
318 				// X, and E type datastreams
319 				// are encoded as callbacks to the Fedora server to obtain the
320 				// datastream's contents. dsMediatedServletPath contains the
321 				// servlet
322 				// path info for this type of backendservice-to-fedora callback.
323 				// The specifics of servlet path are obtained from the
324 				// beSecurity configuration
325 				// file and determines whether the backedservice-to-fedora
326 				// callback
327 				// will use authentication or not.
328 				// callbackRole - contains the role of the backend service (the
329 				// deploymentPID of the service).
330 
331 				String callbackRole = deploymentPID;
332 				Hashtable<String, String> beHash = m_beSS.getSecuritySpec(
333 						callbackRole, methodName);
334 				boolean callbackBasicAuth = new Boolean(
335 						beHash.get("callbackBasicAuth")).booleanValue();
336 				boolean callbackSSL = new Boolean(beHash.get("callbackSSL"))
337 						.booleanValue();
338 				String dsMediatedServletPath = null;
339 				if (callbackBasicAuth) {
340 					dsMediatedServletPath = "/" + m_fedoraAppServerContext
341 							+ "/getDSAuthenticated?id=";
342 				} else {
343 					dsMediatedServletPath = "/" + m_fedoraAppServerContext
344 							+ "/getDS?id=";
345 				}
346 				String dsMediatedCallbackHost = null;
347 				if (callbackSSL) {
348 					dsMediatedCallbackHost = "https://" + m_fedoraServerHost
349 							+ ":" + m_fedoraServerRedirectPort;
350 				} else {
351 					dsMediatedCallbackHost = "http://" + m_fedoraServerHost
352 							+ ":" + m_fedoraServerPort;
353 				}
354 				String datastreamResolverServletURL = dsMediatedCallbackHost
355 						+ dsMediatedServletPath;
356 				if (logger.isDebugEnabled()) {
357 					logger.debug(
358 							"******************Checking backend service dsLocation: {}",
359 							dissBindInfo.dsLocation);
360 					logger.debug(
361 							"******************Checking backend service dsControlGroupType: {}",
362 							dissBindInfo.dsControlGroupType);
363 					logger.debug(
364 							"******************Checking backend service callbackBasicAuth: {}",
365 							callbackBasicAuth);
366 					logger.debug(
367 							"******************Checking backend service callbackSSL: {}",
368 							callbackSSL);
369 					logger.debug(
370 							"******************Checking backend service callbackRole: {}",
371 							callbackRole);
372 					logger.debug(
373 							"******************DatastreamResolverServletURL: {}",
374 							datastreamResolverServletURL);
375 				}
376 
377 				String currentKey = dissBindInfo.DSBindKey;
378 				String nextKey = "";
379 				if (i != numElements - 1) {
380 					// Except for last row, get the value of the next binding
381 					// key
382 					// to compare with the value of the current binding key.
383 					nextKey = dissBindInfoArray[i + 1].DSBindKey;
384 				}
385 				logger.debug("currentKey: '" + currentKey + "', nextKey: '"
386 						+ nextKey + "'");
387 				// In most cases, there is only a single datastream that matches
388 				// a
389 				// given DSBindingKey so the substitution process is to just
390 				// replace
391 				// the occurrence of (BINDING_KEY) with the value of the
392 				// datastream
393 				// location. However, when multiple datastreams match the same
394 				// DSBindingKey, the occurrence of (BINDING_KEY) is replaced
395 				// with the
396 				// value of the datastream location and the value +(BINDING_KEY)
397 				// is
398 				// appended so that subsequent datastreams matching the binding
399 				// key
400 				// will be substituted. The end result is that the binding key
401 				// will
402 				// be replaced by a series of datastream locations separated by
403 				// a
404 				// plus(+) sign. For example, in the case where 3 datastreams
405 				// match
406 				// the binding key for PHOTO:
407 				//
408 				// file=(PHOTO) becomes
409 				// file=dslocation1+dslocation2+dslocation3
410 				//
411 				// It is the responsibility of the Service Deployment to know
412 				// how to
413 				// handle an input parameter with multiple datastream locations.
414 				//
415 				// In the case of a method containing multiple binding keys,
416 				// substitutions are performed on each binding key. For example,
417 				// in
418 				// the case where there are 2 binding keys named PHOTO and
419 				// WATERMARK
420 				// where each matches a single datastream:
421 				//
422 				// image=(PHOTO)&watermark=(WATERMARK) becomes
423 				// image=dslocation1&watermark=dslocation2
424 				//
425 				// In the case with multiple binding keys and multiple
426 				// datastreams,
427 				// the substitution might appear like the following:
428 				//
429 				// image=(PHOTO)&watermark=(WATERMARK) becomes
430 				// image=dslocation1+dslocation2&watermark=dslocation3
431 				if (nextKey.equalsIgnoreCase(currentKey) & i != numElements) {
432 					// Case where binding keys are equal which means that
433 					// multiple
434 					// datastreams matched the same binding key.
435 					if (m_doDatastreamMediation
436 							&& !dissBindInfo.dsControlGroupType
437 									.equalsIgnoreCase("R")) {
438 						// Use Datastream Mediation (except for redirected
439 						// datastreams).
440 
441 						replaceString = datastreamResolverServletURL
442 								+ registerDatastreamLocation(
443 										dissBindInfo.dsLocation,
444 										dissBindInfo.dsControlGroupType,
445 										callbackRole, methodName) + "+("
446 								+ dissBindInfo.DSBindKey + ")";
447 					} else {
448 						// Bypass Datastream Mediation.
449 						if (dissBindInfo.dsControlGroupType
450 								.equalsIgnoreCase("M")
451 								|| dissBindInfo.dsControlGroupType
452 										.equalsIgnoreCase("X")) {
453 							// Use the Default Disseminator syntax to resolve
454 							// the internal
455 							// datastream location for Managed and XML
456 							// datastreams.
457 							replaceString = resolveInternalDSLocation(context,
458 									dissBindInfo.dsLocation,
459 									dissBindInfo.dsCreateDT,
460 									dsMediatedCallbackHost)
461 									+ "+(" + dissBindInfo.DSBindKey + ")";
462 							;
463 						} else {
464 							replaceString = dissBindInfo.dsLocation + "+("
465 									+ dissBindInfo.DSBindKey + ")";
466 						}
467 						if (dissBindInfo.dsControlGroupType
468 								.equalsIgnoreCase("R")
469 								&& dissBindInfo.AddressLocation
470 										.equals(LOCAL_ADDRESS_LOCATION)) {
471 							isRedirect = true;
472 						}
473 					}
474 				} else {
475 					// Case where there are one or more binding keys.
476 					if (m_doDatastreamMediation
477 							&& !dissBindInfo.dsControlGroupType
478 									.equalsIgnoreCase("R")) {
479 						// Use Datastream Mediation (except for Redirected
480 						// datastreams)
481 						replaceString = datastreamResolverServletURL
482 								+ registerDatastreamLocation(
483 										dissBindInfo.dsLocation,
484 										dissBindInfo.dsControlGroupType,
485 										callbackRole, methodName); // this is
486 																	// generic,
487 																	// should be
488 																	// made
489 																	// specific
490 																	// per
491 																	// service
492 					} else {
493 						// Bypass Datastream Mediation.
494 						if (dissBindInfo.dsControlGroupType
495 								.equalsIgnoreCase("M")
496 								|| dissBindInfo.dsControlGroupType
497 										.equalsIgnoreCase("X")) {
498 							// Use the Default Disseminator syntax to resolve
499 							// the internal
500 							// datastream location for Managed and XML
501 							// datastreams.
502 							replaceString = resolveInternalDSLocation(context,
503 									dissBindInfo.dsLocation,
504 									dissBindInfo.dsCreateDT,
505 									dsMediatedCallbackHost);
506 						} else {
507 							replaceString = dissBindInfo.dsLocation;
508 						}
509 						if (dissBindInfo.dsControlGroupType
510 								.equalsIgnoreCase("R")
511 								&& dissBindInfo.AddressLocation
512 										.equals(LOCAL_ADDRESS_LOCATION)) {
513 							isRedirect = true;
514 						}
515 					}
516 				}
517 				try {
518 					
519 					// Here we choose between two different tests for deciding
520 					// whether to URL-encode the datastream URL:
521 					
522 					//Old method:
523 					// If the operationLocation contains datastreamInputParms and also
524 					// contains a "=(" sequence, then
525                     // URLEncode each parameter before substitution. Otherwise, the
526                     // operationLocation has no parameters (i.e., it is a simple URL )
527                     // so bypass URLencoding.
528 					
529 					// New Method:
530 					// If the operationLocation contains datastreamInputParms
531 					// URLEncode each parameter before substitution, except when
532 					// the parameter comprises the first part of the the URL.
533 
534 					boolean useUrlEncoding = m_useNewUrlEncodingTest ? dissURL.indexOf("("
535 							+ bindingKeyPattern + ")") > 0 : dissURL
536 							.indexOf("=(") != -1;
537 
538 					if (useUrlEncoding) {
539 						dissURL = substituteString(dissURL, bindingKeyPattern,
540 								URLEncoder.encode(replaceString, "UTF-8"));
541 					} else {
542 						dissURL = substituteString(dissURL, bindingKeyPattern,
543 								replaceString);
544 					}
545 				} catch (UnsupportedEncodingException uee) {
546 					String message = "[DisseminationService] An error occured. The error "
547 							+ "was \""
548 							+ uee.getClass().getName()
549 							+ "\"  . The Reason was \""
550 							+ uee.getMessage()
551 							+ "\"  . String value: " + replaceString + "  . ";
552 					logger.error(message);
553 					throw new GeneralException(message);
554 				}
555 				logger.debug("Replaced dissURL: " + dissURL.toString()
556 						+ " DissBindingInfo index: " + i);
557 			}
558 
559 			DeploymentDSBindSpec dsBindSpec = bmReader
560 					.getServiceDSInputSpec(null);
561 			DeploymentDSBindRule rules[] = dsBindSpec.dsBindRules;
562 			for (DeploymentDSBindRule element : rules) {
563 				String rulePattern = "(" + element.bindingKeyName + ")";
564 				if (dissURL.indexOf(rulePattern) != -1) {
565 					throw new DisseminationException(null, "Data Object " + PID
566 							+ " missing required datastream: "
567 							+ element.bindingKeyName, null, null, null);
568 				}
569 			}
570 
571 			// Substitute method parameter values in dissemination URL
572 			Enumeration<String> e = h_userParms.keys();
573 			while (e.hasMoreElements()) {
574 				String name = null;
575 				String value = null;
576 				try {
577 					name = URLEncoder.encode(e.nextElement(), "UTF-8");
578 					value = URLEncoder.encode(h_userParms.get(name), "UTF-8");
579 				} catch (UnsupportedEncodingException uee) {
580 					String message = "[DisseminationService] An error occured. The error "
581 							+ "was \""
