58  /**
59   * Implements the three methods GetObjectProfile, GetDissemination, and
60   * GetDatastreamDissemination of the Fedora Access LITE (API-A-LITE) interface
61   * using a java servlet front end. The syntax defined by API-A-LITE defines
62   * three bindings for these methods:
63   * <ol>
64   * <li>GetDissemination URL syntax:
65   * <p>
66   * protocol://hostname:port/fedora/get/PID/sDefPID/methodName[/dateTime][?
67   * parmArray]
68   * </p>
69   * <p>
70   * This syntax requests a dissemination of the specified object using the
71   * specified method of the associated service definition object. The result is
72   * returned as a MIME-typed stream.
73   * </p>
74   * </li>
75   * <ul>
76   * <li>protocol - either http or https.</li>
77   * <li>hostname - required hostname of the Fedora server.</li>
78   * <li>port - required port number on which the Fedora server is running.</li>
79   * <li>fedora - required path name for the Fedora access service.</li>
80   * <li>get - required path name for the Fedora service.</li>
81   * <li>PID - required persistent idenitifer of the digital object.</li>
82   * <li>sDefPID - required persistent identifier of the service definition object
83   * to which the digital object subscribes.</li>
84   * <li>methodName - required name of the method to be executed.</li>
85   * <li>dateTime - optional dateTime value indicating dissemination of a version
86   * of the digital object at the specified point in time.
87   * <li>parmArray - optional array of method parameters consisting of name/value
88   * pairs in the form parm1=value1&parm2=value2...</li>
89   * </ul>
90   * <li>GetObjectProfile URL syntax:
91   * <p>
92   * protocol://hostname:port/fedora/get/PID[/dateTime][?xml=BOOLEAN]
93   * </p>
94   * <p>
95   * This syntax requests an object profile for the specified digital object. The
96   * xml parameter determines the type of output returned. If the parameter is
97   * omitted or has a value of "false", a MIME-typed stream consisting of an html
98   * table is returned providing a browser-savvy means of viewing the object
99   * profile. If the value specified is "true", then a MIME-typed stream
100  * consisting of XML is returned.
101  * </p>
102  * </li>
103  * <ul>
104  * <li>protocol - either http or https</li>
105  * <li>hostname - required hostname of the Fedora server.</li>
106  * <li>port - required port number on which the Fedora server is running.</li>
107  * <li>fedora - required name of the Fedora access service.</li>
108  * <li>get - required verb of the Fedora service.</li>
109  * <li>PID - required persistent identifier of the digital object.</li>
110  * <li>dateTime - optional dateTime value indicating dissemination of a version
111  * of the digital object at the specified point in time.
112  * <li>xml - an optional parameter indicating the requested output format. A
113  * value of "true" indicates a return type of text/xml; the absence of the xml
114  * parameter or a value of "false" indicates format is to be text/html.</li>
115  * </ul>
116  * <li>GetDatastreamDissemination URL syntax:
117  * <p>
118  * protocol://hostname:port/fedora/get/PID/DSID[/dateTime]
119  * </p>
120  * This syntax requests a datastream dissemination for the specified digital
121  * object. It is used to return the contents of a datastream.
122  * <p></li>
123  * <ul>
124  * <li>protocol - either http or https.</li>
125  * <li>hostname - required hostname of the Fedora server.</li>
126  * <li>port - required port number on which the Fedora server is running.</li>
127  * <li>fedora - required name of the Fedora access service.</li>
128  * <li>get - required verb of the Fedora service.</li>
129  * <li>PID - required persistent identifier of the digital object.</li>
130  * <li>DSID - required datastream identifier for the datastream.</li>
131  * <li>dateTime - optional dateTime value indicating dissemination of a version
132  * of the digital object at the specified point in time.
133  * </ul>
134  * </ol>
135  *
136  * @author Ross Wayland
137  */

139         extends SpringAccessServlet

159     /**
160      * <p>
161      * Process Fedora Access Request. Parse and validate the servlet input
162      * parameters and then execute the specified request.
163      * </p>
164      *
165      * @param request
166      *        The servlet request.
167      * @param response
168      *        servlet The servlet response.
169      * @throws ServletException
170      *         If an error occurs that effects the servlet's basic operation.
171      * @throws IOException
172      *         If an error occurrs with an input or output operation.
173      */
174     @Override
175     public void doGet(HttpServletRequest request, HttpServletResponse response)
176             throws ServletException, IOException {
177         String PID = null;
178         String sDefPID = null;
179         String methodName = null;
180         String dsID = null;
181         Date asOfDateTime = null;
182         Date versDateTime = null;
183         Property[] userParms = null;
184         boolean isGetObjectProfileRequest = false;
185         boolean isGetDisseminationRequest = false;
186         boolean isGetDatastreamDisseminationRequest = false;
187         boolean xml = false;
188 
189         requestURI =
190                 request.getRequestURL().toString()
191                         + (request.getQueryString() != null ? "?"
192                                 + request.getQueryString() : "");
193         logger.info("Got request: " + requestURI);
194 
195         // Parse servlet URL.
196         // For the Fedora API-A-LITE "get" syntax, valid entries include:
197         //
198         // For dissemination requests:
199         // http://host:port/fedora/get/pid/sDefPid/methodName
200         // http://host:port/fedora/get/pid/sDefPid/methodName/timestamp
201         // http://host:port/fedora/get/pid/sDefPid/methodName?parm=value[&parm=value]
202         // http://host:port/fedora/get/pid/sDefPid/methodName/timestamp?parm=value[&parm=value]
203         //
204         // For object profile requests:
205         // http://host:port/fedora/get/pid
206         // http://host:port/fedora/get/pid/timestamp
207         //
208         // For datastream dissemination requests:
209         // http://host:port/fedora/get/pid/dsID
210         // http://host:port/fedora/get/pid/dsID/timestamp
211         //
212         String[] URIArray = request.getRequestURL().toString().split("/");
213         if (URIArray.length == 6 || URIArray.length == 7) {
214             // Request is either an ObjectProfile request or a datastream
215             // request
216             if (URIArray.length == 7) {
217                 // They either specified a date/time or a datastream id.
218                 if (URIArray[6].indexOf(":") == -1) {
219                     // If it doesn't contain a colon, they were after a
220                     // datastream,
221                     // so this is a DatastreamDissemination request
222                     dsID = URLDecoder.decode(URIArray[6], "UTF-8");
223                     isGetDatastreamDisseminationRequest = true;
224                 } else {
225                     // If it DOES contain a colon, they were after a
226                     // date/time-stamped object profile
227                     try {
228                         versDateTime = DateUtility.parseDateStrict(URIArray[6]);
229                     } catch (ParseException e) {
230                         String message =
231                                 "ObjectProfile Request Syntax Error: DateTime value "
232                                         + "of \""
233                                         + URIArray[6]
234                                         + "\" is not a valid DateTime format. "
235                                         + " <br></br> The expected format for DateTime is \""
236                                         + "YYYY-MM-DDTHH:MM:SS.SSSZ\".  "
237                                         + " <br></br> The expected syntax for "
238                                         + "ObjectProfile requests is: \""
239                                         + URIArray[0]
240                                         + "//"
241                                         + URIArray[2]
242                                         + "/"
243                                         + URIArray[3]
244                                         + "/"
245                                         + URIArray[4]
246                                         + "/PID[/dateTime] \"  ."
247                                         + " <br></br> Submitted request was: \""
248                                         + requestURI + "\"  .  ";
249                         logger.warn(message);
250                         throw new ServletException("from FedoraAccessServlet"
251                                 + message);
252                         /*
253                          * commented out for exception.jsp test
254                          * response.setStatus
255                          * (HttpServletResponse.SC_INTERNAL_SERVER_ERROR);
256                          * response
257                          * .sendError(HttpServletResponse.SC_INTERNAL_SERVER_ERROR
258                          * , message); return; commented out for exception.jsp
259                          * test
260                          */
261                     }
262                     asOfDateTime = versDateTime;
263                     isGetObjectProfileRequest = true;
264                 }
265             } else {
266                 // URIArray.length==6 so this is a GetObjectProfile request
267                 isGetObjectProfileRequest = true;
268             }
269         } else if (URIArray.length > 7) {
270             // Request is either dissemination request or timestamped get
271             // datastream request
272             methodName = URLDecoder.decode(URIArray[7], "UTF-8");
273             if (URIArray.length == 8) {
274                 if (URIArray[6].indexOf(":") == -1) {
275                     // If it doesn't contain a colon, they were after a
276                     // timestamped
277                     // datastream, so this is a GetDatastreamDissemination
278                     // request.
279                     dsID = URLDecoder.decode(URIArray[6], "UTF-8");
280 
281                     try {
282                         versDateTime = DateUtility.parseDateStrict(URIArray[7]);
283                     } catch (ParseException e) {
284                         String message =
285                                 "GetDatastreamDissemination Request Syntax Error: DateTime value "
286                                         + "of \""
287                                         + URIArray[7]
288                                         + "\" is not a valid DateTime format. "
289                                         + " <br></br> The expected format for DateTime is \""
290                                         + "YYYY-MM-DDTHH:MM:SS.SSSZ\".  "
291                                         + " <br></br> The expected syntax for GetDatastreamDissemination requests is: \""
292                                         + URIArray[0]
293                                         + "//"
294                                         + URIArray[2]
295                                         + "/"
296                                         + URIArray[3]
297                                         + "/"
298                                         + URIArray[4]
299                                         + "/PID/dsID[/dateTime] \"  "
300                                         + " <br></br> Submitted request was: \""
301                                         + requestURI + "\"  .  ";
302                         logger.warn(message);
303                         throw new ServletException("from FedoraAccessServlet"
304                                 + message);
305                     }
306 
307                     asOfDateTime = versDateTime;
308                     isGetDatastreamDisseminationRequest = true;
309                 } else {
310                     isGetDisseminationRequest = true;
311                 }
312             } else if (URIArray.length == 9) {
313                 try {
314                     versDateTime = DateUtility.parseDateStrict(URIArray[8]);
315                 } catch (ParseException e) {
316                     String message =
317                             "Dissemination Request Syntax Error: DateTime value "
318                                     + "of \""
319                                     + URIArray[8]
320                                     + "\" is not a valid DateTime format. "
321                                     + " <br></br> The expected format for DateTime is \""
322                                     + "YYYY-MM-DDTHH:MM:SS.SSS\".  "
323                                     + " <br></br> The expected syntax for Dissemination requests is: \""
324                                     + URIArray[0]
325                                     + "//"
326                                     + URIArray[2]
327                                     + "/"
328                                     + URIArray[3]
329                                     + "/"
330                                     + URIArray[4]
331                                     + "/PID/sDefPID/methodName[/dateTime][?ParmArray] \"  "
332                                     + " <br></br> Submitted request was: \""
333                                     + requestURI + "\"  .  ";
334                     logger.warn(message);
335                     throw new ServletException("from FedoraAccessServlet"
336                             + message);
337                     /*
338                      * commented out for exception.jsp test
339                      * response.setStatus(HttpServletResponse
340                      * .SC_INTERNAL_SERVER_ERROR);
341                      * response.sendError(HttpServletResponse
342                      * .SC_INTERNAL_SERVER_ERROR, message); return; commented
343                      * out for exception.jsp test
344                      */
345                 }
346                 asOfDateTime = versDateTime;
347                 isGetDisseminationRequest = true;
348             }
349             if (URIArray.length > 9) {
350                 String message =
351                         "Dissemination Request Syntax Error: The expected "
352                                 + "syntax for Dissemination requests is: \""
353                                 + URIArray[0]
354                                 + "//"
355                                 + URIArray[2]
356                                 + "/"
357                                 + URIArray[3]
358                                 + "/"
359                                 + URIArray[4]
360                                 + "/PID/sDefPID/methodName[/dateTime][?ParmArray] \"  "
361                                 + " <br></br> Submitted request was: \""
362                                 + requestURI + "\"  .  ";
363                 logger.warn(message);
364                 throw new ServletException("from FedoraAccessServlet" + message);
365                 /*
366                  * commented out for exception.jsp test
367                  * response.setStatus(HttpServletResponse
368                  * .SC_INTERNAL_SERVER_ERROR);
369                  * response.sendError(HttpServletResponse
370                  * .SC_INTERNAL_SERVER_ERROR, message); return; commented out
371                  * for exception.jsp test
372                  */
373             }
374         } else {
375             // Bad syntax; redirect to syntax documentation page.
376             response
377                     .sendRedirect("/userdocs/client/browser/apialite/index.html");
378             return;
379         }
380 
381         // Separate out servlet parameters from method parameters
382         Hashtable<String, String> h_userParms = new Hashtable<String, String>();
383         for (Enumeration<?> e = request.getParameterNames(); e
384                 .hasMoreElements();) {
385             String name = URLDecoder.decode((String) e.nextElement(), "UTF-8");
386             if (isGetObjectProfileRequest && name.equalsIgnoreCase("xml")) {
387                 xml = new Boolean(request.getParameter(name)).booleanValue();
388             } else {
389                 String value =
390                         URLDecoder.decode(request.getParameter(name), "UTF-8");
391                 h_userParms.put(name, value);
392             }
393         }
394 
395         // API-A interface requires user-supplied parameters to be of type
396         // Property[] so create Property[] from hashtable of user parameters.
397         int userParmCounter = 0;
398         userParms = new Property[h_userParms.size()];
399         for (Enumeration<String> e = h_userParms.keys(); e.hasMoreElements();) {
400             Property userParm = new Property();
401             userParm.name = e.nextElement();
402             userParm.value = h_userParms.get(userParm.name);
403             userParms[userParmCounter] = userParm;
404             userParmCounter++;
405         }
406 
407         PID = URIArray[5];
408         String actionLabel = "Access";
409 
410         try {
411             if (isGetObjectProfileRequest) {
412                 logger.debug("Servicing getObjectProfile request " + "(PID="
413                         + PID + ", asOfDate=" + versDateTime + ")");
414 
415                 Context context =
416                         ReadOnlyContext.getContext(HTTP_REQUEST.REST.uri,
417                                                    request);
418                 getObjectProfile(context,
419                                  PID,
420                                  asOfDateTime,
421                                  xml,
422                                  request,
423                                  response);
424 
425                 logger.debug("Finished servicing getObjectProfile request");
426             } else if (isGetDisseminationRequest) {
427                 sDefPID = URIArray[6];
428                 logger.debug("Servicing getDissemination request (PID=" + PID
429                         + ", sDefPID=" + sDefPID + ", methodName=" + methodName
430                         + ", asOfDate=" + versDateTime + ")");
431 
432                 Context context =
433                         ReadOnlyContext.getContext(HTTP_REQUEST.REST.uri,
434                                                    request);
435                 getDissemination(context,
436                                  PID,
437                                  sDefPID,
438                                  methodName,
439                                  userParms,
440                                  asOfDateTime,
441                                  response,
442                                  request);
443 
444                 logger.debug("Finished servicing getDissemination request");
445             } else if (isGetDatastreamDisseminationRequest) {
446                 logger.debug("Servicing getDatastreamDissemination request "
447                         + "(PID=" + PID + ", dsID=" + dsID + ", asOfDate="
448                         + versDateTime + ")");
449 
450                 Context context =
451                         ReadOnlyContext.getContext(HTTP_REQUEST.REST.uri,
452                                                    request);
453                 getDatastreamDissemination(context,
454                                            PID,
455                                            dsID,
456                                            asOfDateTime,
457                                            response,
458                                            request);
459 
460                 logger.debug("Finished servicing getDatastreamDissemination "
461                         + "request");
462             }
463         } catch (MethodNotFoundException e) {
464             logger.error("Method not found for request: " + requestURI
465                     + " (actionLabel=" + actionLabel + ")", e);
466             throw new NotFound404Exception("", e, request, actionLabel, e
467                     .getMessage(), new String[0]);
468         } catch (DatastreamNotFoundException e) {
469             logger.error("Datastream not found for request: " + requestURI
470                     + " (actionLabel=" + actionLabel + ")", e);
471             throw new NotFound404Exception("", e, request, actionLabel, e
472                     .getMessage(), new String[0]);
473         } catch (ObjectNotFoundException e) {
474             logger.error("Object not found for request: " + requestURI
475                     + " (actionLabel=" + actionLabel + ")", e);
476             throw new NotFound404Exception("", e, request, actionLabel, e
477                     .getMessage(), new String[0]);
478         } catch (DisseminationException e) {
479             logger.error("Dissemination failed: " + requestURI
480                     + " (actionLabel=" + actionLabel + ")", e);
481             throw new NotFound404Exception("", e, request, actionLabel, e
482                     .getMessage(), new String[0]);
483         } catch (ObjectNotInLowlevelStorageException e) {
484             logger.error("Object or datastream not found for request: "
485                     + requestURI + " (actionLabel=" + actionLabel + ")", e);
486             throw new NotFound404Exception("", e, request, actionLabel, e
487                     .getMessage(), new String[0]);
488         } catch (AuthzException ae) {
489             logger.error("Authorization failed for request: " + requestURI
490                     + " (actionLabel=" + actionLabel + ")", ae);
491             throw RootException.getServletException(ae,
492                                                     request,
493                                                     actionLabel,
494                                                     new String[0]);
495         } catch (Throwable th) {
496             logger.error("Unexpected error servicing API-A request", th);
497             throw new InternalError500Exception("",
498                                                 th,
499                                                 request,
500                                                 actionLabel,
501                                                 "",
502                                                 new String[0]);
503         }
504     }

506     public void getObjectProfile(Context context,
507                                  String PID,
508                                  Date asOfDateTime,
509                                  boolean xml,
510                                  HttpServletRequest request,
511                                  HttpServletResponse response)
512             throws ServerException {
513 
514         OutputStreamWriter out = null;
515         Date versDateTime = asOfDateTime;
516         ObjectProfile objProfile = null;
517         PipedWriter pw = null;
518         PipedReader pr = null;
519         try {
520             pw = new PipedWriter();
521             pr = new PipedReader(pw);
522             objProfile = m_access.getObjectProfile(context, PID, asOfDateTime);
523             if (objProfile != null) {
524                 // Object Profile found.
525                 // Serialize the ObjectProfile object into XML
526                 new ProfileSerializerThread(context,
527                                             PID,
528                                             objProfile,
529                                             versDateTime,
530                                             pw).start();
531                 if (xml) {
532                     // Return results as raw XML
533                     response.setContentType(CONTENT_TYPE_XML);
534 
535                     // Insures stream read from PipedReader correctly translates
536                     // utf-8
537                     // encoded characters to OutputStreamWriter.
538                     out =
539                             new OutputStreamWriter(response.getOutputStream(),
540                                                    "UTF-8");
541                     char[] buf = new char[BUF];
542                     int len = 0;
543                     while ((len = pr.read(buf, 0, BUF)) != -1) {
544                         out.write(buf, 0, len);
545                     }
546                     out.flush();
547                 } else {
548                     // Transform results into an html table
549                     response.setContentType(CONTENT_TYPE_HTML);
550                     out =
551                             new OutputStreamWriter(response.getOutputStream(),
552                                                    "UTF-8");
553                     File xslFile =
554                             new File(m_server.getHomeDir(),
555                                      "access/viewObjectProfile.xslt");
556                     TransformerFactory factory =
557                             XmlTransformUtility.getTransformerFactory();
558 
559                     Templates template =
560                             factory.newTemplates(new StreamSource(xslFile));
561                     Transformer transformer = template.newTransformer();
562                     transformer.setParameter("fedora", context
563                             .getEnvironmentValue(FEDORA_APP_CONTEXT_NAME));
564                     transformer.transform(new StreamSource(pr),
565                                           new StreamResult(out));
566                 }
567                 out.flush();
568 
569             } else {
570                 throw new GeneralException("No object profile returned");
571             }
572         } catch (ServerException e) {
573             throw e;
574         } catch (Throwable th) {
575             String message = "Error getting object profile";
576             logger.error(message, th);
577             throw new GeneralException(message, th);
578         } finally {
579             try {
580                 if (pr != null) {
581                     pr.close();
582                 }
583                 if (out != null) {
584                     out.close();
585                 }
586             } catch (Throwable th) {
587                 String message = "Error closing output";
588                 logger.error(message, th);
589                 throw new StreamIOException(message);
590             }
591         }
592     }

594     public void getDatastreamDissemination(Context context,
595                                            String PID,
596                                            String dsID,
597                                            Date asOfDateTime,
598                                            HttpServletResponse response,
599                                            HttpServletRequest request)
600             throws IOException, ServerException {
601         ServletOutputStream out = null;
602         MIMETypedStream dissemination = null;
603         dissemination =
604                 m_access.getDatastreamDissemination(context,
605                                                     PID,
606                                                     dsID,
607                                                     asOfDateTime);
608         try {
609             // testing to see what's in request header that might be of interest
610             if (logger.isDebugEnabled()) {
611                 for (Enumeration<?> e = request.getHeaderNames(); e
612                         .hasMoreElements();) {
613                     String name = (String) e.nextElement();
614                     Enumeration<?> headerValues = request.getHeaders(name);
615                     StringBuffer sb = new StringBuffer();
616                     while (headerValues.hasMoreElements()) {
617                         sb.append((String) headerValues.nextElement());
618                     }
619                     String value = sb.toString();
620                     logger.debug("FEDORASERVLET REQUEST HEADER CONTAINED: "
621                             + name + " : " + value);
622                 }
623             }
624 
625             // Dissemination was successful;
626             // Return MIMETypedStream back to browser client
627             if (dissemination.MIMEType
628                     .equalsIgnoreCase("application/fedora-redirect")) {
629                 // A MIME type of application/fedora-redirect signals that
