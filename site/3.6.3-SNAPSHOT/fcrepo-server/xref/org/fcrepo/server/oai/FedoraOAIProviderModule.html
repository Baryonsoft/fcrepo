51      @Override
52      public void postInitModule() throws ModuleInitializationException {
53          String repositoryName = getParameter("repositoryName");
54          if (repositoryName == null) {
55              throw new ModuleInitializationException("repositoryName must be specified.",
56                                                      getRole());
57          }
58          String repositoryDomainName = getParameter("repositoryDomainName");
59          if (repositoryDomainName == null) {
60              throw new ModuleInitializationException("repositoryDomainName must be specified.",
61                                                      getRole());
62          }
63          String host = getServer().getParameter("fedoraServerHost");
64          if (host == null) {
65              throw new ModuleInitializationException("fedoraServerHost must be specified as primary server config element.",
66                                                      getRole());
67          }
68          String context = getServer().getParameter("fedoraAppServerContext");
69          if (context == null) {
70              throw new ModuleInitializationException("fedoraAppServerContext must be specified as primary server config element.",
71                                                      getRole());
72          }
73  
74          String port = getServer().getParameter("fedoraServerPort");
75          if (port == null) {
76              throw new ModuleInitializationException("fedoraServerPort must be specified as primary server config element.",
77                                                      getRole());
78          }
79          Module mgr = getServer().getModule("org.fcrepo.server.storage.DOManager");
80          if (mgr == null) {
81              throw new ModuleInitializationException("DOManager is required (for pidNamespace param), but isn't loaded.",
82                                                      getRole());
83          }
84          String pidNamespace = mgr.getParameter("pidNamespace");
85          if (pidNamespace == null) {
86              throw new ModuleInitializationException("DOManager did not specify a pidNamespace, but this module requires that it does.",
87                                                      getRole());
88          }
89          String aes = getParameter("adminEmails");
90          if (aes == null) {
91              throw new ModuleInitializationException("adminEmails must be specified.",
92                                                      getRole());
93          }
94          HashSet<String> adminEmails = new HashSet<String>();
95          if (aes.indexOf(" ") == -1) {
96              adminEmails.add(aes);
97          } else {
98              String[] emails = aes.split(" ");
99              for (String element : emails) {
100                 adminEmails.add(element);
101             }
102         }
103         HashSet<String> friends = new HashSet<String>();
104         if (getParameter("friends") != null) {
105             String f = getParameter("friends");
106             if (f.indexOf(" ") == -1) {
107                 friends.add(f);
108             } else {
109                 String[] fs = f.split(" ");
110                 for (String element : fs) {
111                     friends.add(element);
112                 }
113             }
114         }
115         FieldSearch fieldSearch =
116                 (FieldSearch) getServer()
117                         .getModule("org.fcrepo.server.search.FieldSearch");
118         if (fieldSearch == null) {
119             throw new ModuleInitializationException("FieldSearch module was not loaded, but is required.",
120                                                     getRole());
121         }
122         Module fsModule =
123                 getServer().getModule("org.fcrepo.server.search.FieldSearch");
124 
125         if (fsModule.getParameter("maxResults") == null) {
126             throw new ModuleInitializationException("maxResults parameter must be specified in FieldSearch module's configuration.",
127                                                     getRole());
128         }
129         int maxResults = 0;
130         try {
131             maxResults = Integer.parseInt(fsModule.getParameter("maxResults"));
132             if (maxResults < 1) {
133                 throw new NumberFormatException("");
134             }
135         } catch (NumberFormatException nfe) {
136             throw new ModuleInitializationException("maxResults specified in FieldSearch module's configuration must be a positive integer.",
137                                                     getRole());
138         }
139 
140         long maxSets = 100; // unused for now, but passed in the constructor anyway
141         long maxRecords = maxResults;
142         long maxHeaders = maxResults;
143         String maxRecordsString = getParameter("maxRecords");
144         if (maxRecordsString != null) {
145             try {
146                 maxRecords = Long.parseLong(maxRecordsString);
147                 if (maxRecords > maxResults) {
148                     logger.warn("maxRecords was over the limit given by the FieldSearch module, using highest possible value: "
149                                     + maxResults);
150                     maxRecords = maxResults;
151                 }
152             } catch (NumberFormatException nfe) {
153                 throw new ModuleInitializationException("maxRecords value is invalid.",
154                                                         getRole());
155             }
156         }
157         String maxHeadersString = getParameter("maxHeaders");
158         if (maxHeadersString != null) {
159             try {
160                 maxHeaders = Long.parseLong(maxHeadersString);
161                 if (maxHeaders > maxResults) {
162                     logger.warn("maxHeaders was over the limit given by the FieldSearch module, using highest possible value: "
163                                     + maxResults);
164                     maxHeaders = maxResults;
165                 }
166             } catch (NumberFormatException nfe) {
167                 throw new ModuleInitializationException("maxHeaders value is invalid.",
168                                                         getRole());
169             }
170         }
171         m_wrappedOAIProvider =
172                 new FedoraOAIProvider(repositoryName,
173                                       repositoryDomainName,
174                                       host,
175                                       "/" + context + "/oai",
176                                       adminEmails,
177                                       friends,
178                                       pidNamespace,
179                                       maxSets,
180                                       maxRecords,
181                                       maxHeaders,
182                                       fieldSearch);
183     }
