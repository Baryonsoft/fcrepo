110         StringBuffer buf = new StringBuffer();

122                 + ":" + namespaceID + ":7654</sampleIdentifier>\n");
123         buf.append("      </oai-identifier>");
124         m_descriptions.add(buf.toString());
125         if (friendBaseURLs != null && friendBaseURLs.size() > 0) {
126             buf = new StringBuffer();
127             buf.append("      <friends xmlns=\"" + OAI_FRIENDS.uri + "\"\n");
128             buf.append("          xmlns:xsi=\"" + XSI.uri + "\"\n");
129             buf.append("          xsi:schemaLocation=\"" + OAI_FRIENDS.uri
130                     + "\n");
131             buf.append("          " + OAI_FRIENDS2_0.xsdLocation + "\">\n");
132             Iterator iter = friendBaseURLs.iterator();
133             while (iter.hasNext()) {
134                 buf.append("        <baseURL>" + (String) iter.next()
135                         + "</baseURL>\n");
136             }
137             buf.append("      </friends>");
138             m_descriptions.add(buf.toString());
139         }
140         m_formats = new HashSet<SimpleMetadataFormat>();
141         m_formats.add(new SimpleMetadataFormat("oai_dc",
142                                                OAI_DC2_0.xsdLocation,
143                                                OAI_DC.uri));
144         m_setInfos = new ArrayList<SimpleSetInfo>();

183     public Record getRecord(String identifier, String metadataPrefix)
184             throws CannotDisseminateFormatException, IDDoesNotExistException,
185             RepositoryException {
186         if (!metadataPrefix.equals("oai_dc")) {
187             throw new CannotDisseminateFormatException("Repository does not provide that format in OAI-PMH responses.");
188         }
189         String pid = getPID(identifier);
190         List l = null;
191         try {
192             //FIXME: use maxResults from... config instead of hardcoding 100?
193             l =
194                     m_fieldSearch
195                             .findObjects(s_headerAndDCFields,
196                                          100,
197                                          new FieldSearchQuery(Condition
198                                                  .getConditions("pid='"
199                                                          + pid
200                                                          + "' dcmDate>'2000-01-01'")))
201                             .objectFieldsList();
202         } catch (ServerException se) {
203             throw new RepositoryException(se.getClass().getName() + ": "
204                     + se.getMessage());
205         }
206         if (l.size() > 0) {
207             ObjectFields f = (ObjectFields) l.get(0);
208             return new SimpleRecord(getHeader(f), getDCXML(f), s_emptySet);
209         } else {
210             // see if it exists
211             try {
212                 l =
213                         m_fieldSearch
214                                 .findObjects(new String[] {"pid"},
215                                              1,
216                                              new FieldSearchQuery(Condition
217                                                      .getConditions("pid='"
218                                                              + pid + "'")))
219                                 .objectFieldsList();
220             } catch (ServerException se) {
221                 throw new RepositoryException(se.getClass().getName() + ": "
222                         + se.getMessage());
223             }
224             if (l.size() == 0) {
225                 throw new IDDoesNotExistException("The provided id does not match any item in the repository.");
226             } else {
227                 throw new CannotDisseminateFormatException("The item doesn't even have dc_oai metadata.");
228             }
229         }
230     }

232     public List getRecords(Date from,
233                            Date until,
234                            String metadataPrefix,
235                            String set) throws CannotDisseminateFormatException,
236             NoRecordsMatchException, NoSetHierarchyException,
237             RepositoryException {
238         if (!metadataPrefix.equals("oai_dc")) {
239             throw new CannotDisseminateFormatException("Repository does not provide that format in OAI-PMH responses.");
240         }
241         List l = null;
242         FieldSearchResult fsr;
243         try {
244             fsr =
245                     m_fieldSearch
246                             .findObjects(s_headerAndDCFields,
247                                          (int) getMaxRecords(),
248                                          new FieldSearchQuery(Condition
249                                                  .getConditions("dcmDate>'2000-01-01'"
250                                                          + getDatePart(from,
251                                                                        until))));
252             l = fsr.objectFieldsList();
253         } catch (ServerException se) {
254             throw new RepositoryException(se.getClass().getName() + ": "
255                     + se.getMessage());
256         }
257         if (l.size() == 0) {
258             throw new NoRecordsMatchException("No records match the given criteria.");
259         }
260         ArrayList<Object> ret = new ArrayList<Object>();
261         for (int i = 0; i < l.size(); i++) {
262             ObjectFields f = (ObjectFields) l.get(i);
263             ret.add(new SimpleRecord(getHeader(f), getDCXML(f), s_emptySet));
264         }
265         if (fsr.getToken() != null) {
266             // add resumptionToken stuff
267             ret.add(new SimpleResumptionToken(fsr.getToken(), fsr
268                     .getExpirationDate(), fsr.getCompleteListSize(), fsr
269                     .getCursor()));
270         }
271         return ret;
272     }

285     public List getRecords(String resumptionToken)
286             throws CannotDisseminateFormatException, NoRecordsMatchException,
287             NoSetHierarchyException, BadResumptionTokenException,
288             RepositoryException {
289         // this is the exact same as the other getRecords, except for the FieldSearch call,
290         // and the fact that we re-throw UnknownSessionTokenException
291         // as a BadResumptionTokenException
292         List l = null;
293         FieldSearchResult fsr;
294         try {
295             fsr = m_fieldSearch.resumeFindObjects(resumptionToken);
296             l = fsr.objectFieldsList();
297         } catch (UnknownSessionTokenException uste) {
298             throw new BadResumptionTokenException("Not a known resumptionToken.");
299         } catch (ServerException se) {
300             throw new RepositoryException(se.getClass().getName() + ": "
301                     + se.getMessage());
302         }
303         if (l.size() == 0) {
304             throw new NoRecordsMatchException("No records match the given criteria.");
305         }
306         ArrayList<Object> ret = new ArrayList<Object>();
307         for (int i = 0; i < l.size(); i++) {
308             ObjectFields f = (ObjectFields) l.get(i);
309             ret.add(new SimpleRecord(getHeader(f), getDCXML(f), s_emptySet));
310         }
311         if (fsr.getToken() != null) {
312             ret.add(new SimpleResumptionToken(fsr.getToken(), fsr
313                     .getExpirationDate(), fsr.getCompleteListSize(), fsr
314                     .getCursor()));
315         }
316         return ret;
317     }

319     private String getDatePart(Date from, Date until) {
320         if (from == null && until == null) {
321             return "";
322         }
323         StringBuffer out = new StringBuffer();
324         // Note OAI only support ISO8601 dates to the seconds
325         // and Fedora stores dates down to the millisecond level.
326         // This should not matter since OAI requests specify
327         // date ranges (from-until), and as long as the requests
328         // are always in the same units, subsequent requests can pick 
329         // up at the last end-point in time (in seconds) without
330         // concern for millisecond granularity.
331         SimpleDateFormat formatter =
332                 new SimpleDateFormat("yyyy-MM-dd'T'HH:mm:ss");
333         if (from != null) {
334             out.append(" dcmDate>='");
335             out.append(formatter.format(from));
336             out.append("'");
337         }
338         if (until != null) {
339             out.append(" dcmDate<='");
340             out.append(formatter.format(until));
341             out.append("'");
342         }
343         return out.toString();
344     }

441     private String getPID(String id) throws IDDoesNotExistException {
442         if (!id.startsWith("oai:" + m_repositoryDomainName + ":")) {
443             throw new IDDoesNotExistException("For this repository, all identifiers in OAI requests should begin with oai:"
444                     + m_repositoryDomainName + ":");
445         }
446         if (id.indexOf("'") != -1) {
447             throw new IDDoesNotExistException("For this repository, no identifiers contain the apostrophe character.");
448         }
449         return id.substring(4 + m_repositoryDomainName.length() + 1);
450     }

452     public Set getMetadataFormats(String id) throws NoMetadataFormatsException,
453             IDDoesNotExistException, RepositoryException {
454         if (id == null) {
455             return m_formats;
456         }
457         String pid = getPID(id);
458         List l = null;
459         try {
460             l =
461                     m_fieldSearch
462                             .findObjects(new String[] {"pid"},
463                                          1,
464                                          new FieldSearchQuery(Condition
465                                                  .getConditions("pid='"
466                                                          + pid
467                                                          + "' dcmDate>'2000-01-01'")))
468                             .objectFieldsList();
469         } catch (ServerException se) {
470             throw new RepositoryException(se.getClass().getName() + ": "
471                     + se.getMessage());
472         }
473         if (l.size() > 0) {
474             return m_formats;
475         }
476         try {
477             l =
478                     m_fieldSearch.findObjects(new String[] {"pid"},
479                                               1,
480                                               new FieldSearchQuery(Condition
481                                                       .getConditions("pid='"
482                                                               + pid + "'")))
483                             .objectFieldsList();
484         } catch (ServerException se) {
485             throw new RepositoryException(se.getClass().getName() + ": "
486                     + se.getMessage());
487         }
488         if (l.size() > 0) {
489             throw new NoMetadataFormatsException("The item doesn't even have dc_oai metadata.");
490         } else {
491             throw new IDDoesNotExistException("The provided id does not match any item in the repository.");
492         }
493     }
