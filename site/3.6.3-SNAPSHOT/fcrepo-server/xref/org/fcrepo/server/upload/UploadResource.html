27  /**
28   * Enables the upload of temp files for the REST API. Takes a POST request with
29   * a message having media type multipart/form-data and consisting of exactly one
30   * entity called "file".
31   *
32   * @version $Id$
33   */
34  @Path("/")
35  @Component
36  public class UploadResource extends BaseRestResource {
37  
38      private static final Logger logger = LoggerFactory
39              .getLogger(UploadServlet.class);
40  
41      public UploadResource(Server server) {
42          super(server);
43      }
44  
45      /**
46       * Uploads a file encoded in a multipart/form message.
47       *
48       * @param multiPart
49       *            the multiPart object containing the file to be uploaded
50       * @return a URI with the (custom) uploaded:// scheme and a unique
51       *         identifier to be used future ingests. Returns 202 if request succeeds
52       */
53      @POST
54      @Consumes(MediaType.MULTIPART_FORM_DATA)
55      public Response upload(MultipartBody multipart){
56          Attachment file = multipart.getAttachment("file");
57          if (file == null) {
58              return Response.status(400).entity("Missing file part").type(MediaType.TEXT_PLAIN).build();
59          }
60          InputStream fileStream = file.getObject(InputStream.class);
61          String uploaded;
62          try {
63              uploaded = m_management.putTempStream(getContext(), fileStream);
64              logger.debug("File uploaded: ", uploaded);
65          } catch (ServerException e) {
66              logger.error(e.toString());
67              return handleException(e, false);
68          }
69          return Response.status(Response.Status.ACCEPTED).entity(uploaded).type(
70                  MediaType.TEXT_PLAIN).build();
71      }
72  }
