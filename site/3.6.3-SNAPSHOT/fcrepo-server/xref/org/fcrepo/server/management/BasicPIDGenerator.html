5   package org.fcrepo.server.management;

7   import java.io.File;
8   import java.io.IOException;
9   
10  import java.util.Map;
11  
12  import org.fcrepo.common.PID;
13  import org.fcrepo.server.Module;
14  import org.fcrepo.server.Server;
15  import org.fcrepo.server.errors.ModuleInitializationException;
16  import org.fcrepo.server.storage.ConnectionPoolManager;
17  import org.slf4j.Logger;
18  import org.slf4j.LoggerFactory;

20  /**
21   * A wrapper around the DBPIDGenerator class that casts it as a Module.
22   *
23   * @author Chris Wilper
24   */
25  public class BasicPIDGenerator
26          extends Module
27          implements PIDGenerator {
28  
29      private static final Logger logger =
30              LoggerFactory.getLogger(BasicPIDGenerator.class);
31  
32      private DBPIDGenerator m_pidGenerator;
33  
34      private File m_oldPidGenDir;
35  
36      /**
37       * Constructs a BasicPIDGenerator.
38       *
39       * @param moduleParameters
40       *        A pre-loaded Map of name-value pairs comprising the intended
41       *        configuration of this Module.
42       * @param server
43       *        The <code>Server</code> instance.
44       * @param role
45       *        The role this module fulfills, a java class name.
46       * @throws ModuleInitializationException
47       *         If initilization values are invalid or initialization fails for
48       *         some other reason.
49       */
50      public BasicPIDGenerator(Map moduleParameters, Server server, String role)
51              throws ModuleInitializationException {
52          super(moduleParameters, server, role);
53      }
54  
55      @Override
56      public void initModule() {
57          // this parameter is no longer required; but if it's specified,
58          // we can automatically upgrade from a pre-1.2 version of Fedora by
59          // making sure the old "last pid generated" value is respected later.
60          String dir = getParameter("pidgen_log_dir");
61          if (dir != null && !dir.equals("")) {
62              if (dir.startsWith("/") || dir.startsWith("\\")
63                      || dir.substring(1).startsWith(":\\")) {
64                  m_oldPidGenDir = new File(dir);
65              } else {
66                  m_oldPidGenDir = new File(getServer().getHomeDir(), dir);
67              }
68          }
69      }
70  
71      /**
72       * Get a reference to the ConnectionPoolManager so we can give the instance
73       * constructor a ConnectionPool later in initializeIfNeeded().
74       */
75      @Override
76      public void postInitModule() throws ModuleInitializationException {
77          ConnectionPoolManager mgr =
78                  (ConnectionPoolManager) getServer()
79                          .getModule("org.fcrepo.server.storage.ConnectionPoolManager");
80          if (mgr == null) {
81              throw new ModuleInitializationException("ConnectionPoolManager module not loaded.",
82                                                      getRole());
83          }
84          try {
85              m_pidGenerator = new DBPIDGenerator(mgr.getPool(), m_oldPidGenDir);
86          } catch (Exception e) {
87              String msg = "Can't get default connection pool";
88              logger.error(msg, e);
89              throw new ModuleInitializationException(msg, getRole());
90          }
91      }
92  
93      public PID generatePID(String namespaceID) throws IOException {
94          return m_pidGenerator.generatePID(namespaceID);
95      }
96  
97      public PID getLastPID() throws IOException {
98          return m_pidGenerator.getLastPID();
99      }
100 
101     public void neverGeneratePID(String pid) throws IOException {
102         m_pidGenerator.neverGeneratePID(pid);
103     }
104 
105 }
