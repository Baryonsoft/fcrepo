82      @Override
83      public void initModule() throws ModuleInitializationException {
84  
85          // how many minutes should we hold on to uploaded files? default=5
86          String min = getParameter("uploadStorageMinutes");
87          if (min == null) {
88              min = "5";
89          }
90          try {
91              m_uploadStorageMinutes = Integer.parseInt(min);
92              if (m_uploadStorageMinutes < 1) {
93                  throw new ModuleInitializationException("uploadStorageMinutes "
94                                                          + "must be 1 or more, if specified.", getRole());
95              }
96          } catch (NumberFormatException nfe) {
97              throw new ModuleInitializationException("uploadStorageMinutes must "
98                                                      + "be an integer, if specified.",
99                                                      getRole());
100         }
101         // initialize storage area by 1) ensuring the directory is there
102         // and 2) reading in the existing files, if any, and setting their
103         // startTime to the current time.
104         try {
105             m_tempDir = getServer().getUploadDir();
106             if (!m_tempDir.isDirectory()) {
107                 m_tempDir.mkdirs();
108                 if (!m_tempDir.isDirectory()) {
109                     throw new ModuleInitializationException(
110                             "Failed to create temp dir at " +
111                                     m_tempDir.toString(), getRole());
112                 }
113             }
114 
115             // put leftovers in hash, while saving highest id as m_lastId
116             m_uploadStartTime = new Hashtable<String, Long>();
117             String[] fNames = m_tempDir.list();
118             Long leftoverStartTime = new Long(System.currentTimeMillis());
119             m_lastId = 0;
120             for (String element : fNames) {
121                 try {
122                     int id = Integer.parseInt(element);
123                     if (id > m_lastId) {
124                         m_lastId = id;
125                     }
126                     m_uploadStartTime.put(element, leftoverStartTime);
127                 } catch (NumberFormatException nfe) {
128                     // skip files that aren't named numerically
129                 }
130             }
131         } catch (Exception e) {
132             throw new ModuleInitializationException("Error while initializing "
133                                                     + "temporary storage area: " + e.getClass().getName()
134                                                     + ": " + e.getMessage(), getRole(), e);
135         }
136 
137         // initialize variables pertaining to checksumming datastreams.
138         String auto = getParameter("autoChecksum");
139         logger.debug("Got Parameter: autoChecksum = " + auto);
140         if (auto != null && auto.equalsIgnoreCase("true")) {
141             Datastream.autoChecksum = true;
142             Datastream.defaultChecksumType = getParameter("checksumAlgorithm");
143         }
144         logger.debug("autoChecksum is " + auto);
145         logger.debug("defaultChecksumType is " + Datastream.defaultChecksumType);
146 
147         // get delay between purge of two uploaded files (default 1 minute)
148         String purgeDelayInMillis = getParameter("purgeDelayInMillis");
149         if (purgeDelayInMillis == null) {
150             purgeDelayInMillis = "60000";
151         }
152         try {
153             this.m_purgeDelayInMillis = Integer.parseInt(purgeDelayInMillis);
154         } catch (NumberFormatException nfe) {
155             throw new ModuleInitializationException(
156                     "purgeDelayInMillis must be an integer, if specified.",
157                     getRole());
158         }
159     }

161     @Override
162     public void postInitModule() throws ModuleInitializationException {
163         // Verify required modules have been loaded
164         m_manager =
165                 getServer()
166                         .getBean("org.fcrepo.server.storage.DOManager", DOManager.class);
167         if (m_manager == null) {
168             throw new ModuleInitializationException("Can't get a DOManager "
169                                                     + "from Server.getModule", getRole());
170         }
171         m_contentManager =
172                 getServer()
173                         .getBean("org.fcrepo.server.storage.ExternalContentManager", ExternalContentManager.class);
174         if (m_contentManager == null) {
175             throw new ModuleInitializationException("Can't get an ExternalContentManager "
176                                                     + "from Server.getModule",
177                                                     getRole());
178         }
179 
180         m_fedoraXACMLModule =
181                 getServer()
182                         .getBean("org.fcrepo.server.security.Authorization", Authorization.class);
183         if (m_fedoraXACMLModule == null) {
184             throw new ModuleInitializationException(
185                     "Can't get Authorization module (in default management) from Server.getModule",
186                     getRole());
187         }
188 
189         Management m =
190                 new DefaultManagement(m_fedoraXACMLModule,
191                                       m_manager,
192                                       m_contentManager,
193                                       m_uploadStorageMinutes,
194                                       m_lastId,
195                                       m_tempDir,
196                                       m_uploadStartTime,
197                                       m_purgeDelayInMillis);
198 
199         mgmt = getProxyChain(m);
200         // FIXME: FCREPO-765
201         // need to keep a reference to DefaultManagement for admin functions
202         // remove this once migrated to an Admin module
203         defaultManagement = (DefaultManagement)m;
204     }

555      * Build a proxy chain as configured by the module parameters.
556      *
557      * @param m The concrete Management implementation to wrap.
558      * @return A proxy chain for Management
559      * @throws ModuleInitializationException
560      */
561     private Management getProxyChain(Management m)
562             throws ModuleInitializationException {
563         try {
564             invocationHandlers = getInvocationHandlers();
565             return (Management) ProxyFactory.getProxy(m, invocationHandlers);
566         } catch (Exception e) {
567             throw new ModuleInitializationException(e.getMessage(), getRole());
568         }

571     /**
572      * Get an <code>Array</code> of <code>AbstractInvocationHandler</code>s. The
573      * ordering is ascending alphabetical, determined by the module parameter
574      * names that begin with the string "decorator", e.g. "decorator1",
575      * "decorator2".
576      *
577      * @return An array InvocationHandlers
578      */
579     private AbstractInvocationHandler[] getInvocationHandlers() throws Exception {
580         List<String> pNames = new ArrayList<String>();
581         Iterator<String> it = parameterNames();
582         String param;
583         while (it.hasNext()) {
584             param = it.next();
585             if (param.startsWith("decorator")) {
586                 pNames.add(param);
587             }
588         }
589         Collections.sort(pNames);
590 
591         AbstractInvocationHandler[] invocationHandlers = new AbstractInvocationHandler[pNames.size()];
592         for (int i = 0; i < pNames.size(); i++) {
593             invocationHandlers[i] =
594                     getInvocationHandler(getParameter(pNames.get(i)));
595             if (invocationHandlers[i] instanceof ModuleConfiguredInvocationHandler) {
596                 ((ModuleConfiguredInvocationHandler)invocationHandlers[i]).init(this.getServer());
597             }
598         }
599 
600         return invocationHandlers;
601     }
602 
603     private static AbstractInvocationHandler getInvocationHandler(String invocationHandler)
604             throws Exception {
605         if (invocationHandler == null) {
606             return null;
607         }
608         Class<?> c = Class.forName(invocationHandler);
609         return (AbstractInvocationHandler) c.newInstance();
610     }
