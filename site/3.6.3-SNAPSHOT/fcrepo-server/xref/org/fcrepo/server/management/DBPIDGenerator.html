30  /**
31   * A PIDGenerator that uses a database to keep track of the highest pid it knows
32   * about for each namespace.
33   *
34   * @author Chris Wilper
35   */
36  public class DBPIDGenerator
37          implements PIDGenerator {
38  
39      private static final Logger logger =
40              LoggerFactory.getLogger(DBPIDGenerator.class);
41  
42      private final HashMap m_highestID;
43  
44      private PID m_lastPID;
45  
46      private final ConnectionPool m_connectionPool;
47  
48      /**
49       * Initialize the DBPIDGenerator. This initializes the memory hash with
50       * values in the database, if any. If oldPidGenDir is not null, the
51       * constructor will then call neverGeneratePID on the most recently
52       * generated PID as reported by the log files in that directory. This is to
53       * support automatic upgrade of this functionality from versions of Fedora
54       * prior to 1.2.
55       */
56      public DBPIDGenerator(ConnectionPool cPool, File oldPidGenDir)
57              throws IOException {
58          m_connectionPool = cPool;
59          m_highestID = new HashMap();
60          // load the values from the database into the m_highestID hash
61          // pidGen:  namespace  highestID
62          Statement s = null;
63          ResultSet results = null;
64          Connection conn = null;
65          try {
66              conn = m_connectionPool.getReadOnlyConnection();
67              String query = "SELECT namespace, highestID FROM pidGen";
68              s = conn.createStatement();
69              results = s.executeQuery(query);
70              while (results.next()) {
71                  m_highestID.put(results.getString("namespace"),
72                                  new Integer(results.getInt("highestID")));
73              }
74          } catch (SQLException sqle) {
75              logger.warn("Unable to read pidGen table; assuming it "
76                      + "will be created shortly");
77          } finally {
78              try {
79                  if (results != null) {
80                      results.close();
81                  }
82                  if (s != null) {
83                      s.close();
84                  }
85                  if (conn != null) {
86                      m_connectionPool.free(conn);
87                  }
88              } catch (SQLException sqle2) {
89                  logger.warn("Error trying to free db "
90                          + "resources in DBPIDGenerator", sqle2);
91              } finally {
92                  results = null;
93                  s = null;
94              }
95          }
96          upgradeIfNeeded(oldPidGenDir);
97      }
98  
99      /**
100      * Read the highest value from the old pidGen directory if it exists, and
101      * ensure it is never used.
102      */
103     private void upgradeIfNeeded(File oldPidGenDir) throws IOException {
104         if (oldPidGenDir != null && oldPidGenDir.isDirectory()) {
105             String[] names = oldPidGenDir.list();
106             Arrays.sort(names);
107             if (names.length > 0) {
108                 BufferedReader in =
109                         new BufferedReader(new InputStreamReader(new FileInputStream(new File(oldPidGenDir,
110                                                                                               names[names.length - 1]))));
111                 String lastLine = null;
112                 String line;
113                 while ((line = in.readLine()) != null) {
114                     lastLine = line;
115                 }
116                 in.close();
117                 if (lastLine != null) {
118                     String[] parts = lastLine.split("|");
119                     if (parts.length == 2) {
120                         neverGeneratePID(parts[0]);
121                     }
122                 }
123             }
124         }
125     }

127     /**
128      * Generate a new pid that is guaranteed to be unique, within the given
129      * namespace.
130      */
131     public synchronized PID generatePID(String namespace) throws IOException {
132         int i = getHighestID(namespace);
133         i++;
134 
135         try {
136             m_lastPID = new PID(namespace + ":" + i);
137         } catch (MalformedPIDException e) {
138             throw new IOException(e.getMessage());
139         }
140 
141         setHighestID(namespace, i);
142 
143         return m_lastPID;
144     }
145 
146     /**
147      * Get the last pid that was generated.
148      */
149     public synchronized PID getLastPID() {
150         return m_lastPID;
151     }
152 
153     /**
154      * Cause the given PID to never be generated by the PID generator.
155      */
156     public synchronized void neverGeneratePID(String pid) throws IOException {
157         logger.debug("Never generating PID: " + pid);
158         try {
159             PID p = new PID(pid);
160             String ns = p.getNamespaceId();
161             int id = Integer.parseInt(p.getObjectId());
162             if (id > getHighestID(ns)) {
163                 setHighestID(ns, id);
164             }
165         } catch (MalformedPIDException mpe) {
166             throw new IOException(mpe.getMessage());
167         } catch (NumberFormatException nfe) {
168             // if the id part is not numeric, we already know we'll
169             // never generate that id because all generated ids are numeric.
170         }
171     }
172 
173     /**
174      * Gets the highest id ever used for the given namespace.
175      */
176     private int getHighestID(String namespace) {
177         Integer i = (Integer) m_highestID.get(namespace);
178         if (i == null) {
179             return 0;
180         }
181         return i.intValue();
182     }
183 
184     /**
185      * Sets the highest id ever used for the given namespace.
186      */
187     private void setHighestID(String namespace, int id) throws IOException {
188         logger.debug("Setting highest ID for " + namespace + " to " + id);
189         m_highestID.put(namespace, new Integer(id));
190         // write the new highest id in the database, too
191         Connection conn = null;
192         try {
193             conn = m_connectionPool.getReadWriteConnection();
194             SQLUtility.replaceInto(conn,
195                                    "pidGen",
196                                    new String[] {"namespace", "highestID"},
197                                    new String[] {namespace, "" + id},
198                                    "namespace",
199                                    new boolean[] {false, true});
200         } catch (SQLException sqle) {
201             throw new IOException("Error setting highest id for "
202                     + "namespace in db: " + sqle.getMessage());
203         } finally {
204             if (conn != null) {
205                 m_connectionPool.free(conn);
206             }
207         }
208     }
