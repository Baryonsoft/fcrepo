50      /**
51       * Get relationships asserted by the object denoted by <i>pid</i>.
52       *
53       * @param pid The pid of the Fedora object, e.g. demo:1.
54       * @param subject The subject uri. If null, defaults to the URI form of pid,
55       *                e.g. info:fedora/demo:1.
56       * @param predicate The predicate uri or null to match any predicate.
57       * @param format one of "rdf/xml", "n-triples", "turtle", or "sparql".
58       *               If null, defaults to rdf/xml.
59       * @return the relationships in the specified format.
60       */
61      @GET
62      @Produces({"application/rdf+xml", "text/plain", "application/x-turtle",
63          "application/sparql-results+xml"})
64      public Response getRelationships(
65              @PathParam(RestParam.PID)
66              String pid,
67              @QueryParam(RestParam.SUBJECT)
68              String subject,
69              @QueryParam(RestParam.PREDICATE)
70              String predicate,
71              @QueryParam(RestParam.FORMAT)
72              @DefaultValue("rdf/xml")
73              String format,
74              @QueryParam(RestParam.FLASH)
75              @DefaultValue("false")
76              boolean flash) {
77          Context context = getContext();
78          if (subject == null) {
79              // assume the subject is the object as denoted by the pid
80              subject = PID.toURI(pid);
81          }
82          try {
83              RelationshipTuple[] tuples = m_management.getRelationships(context, subject, predicate);
84              TripleIterator it = new TupleArrayTripleIterator(new ArrayList<RelationshipTuple>(Arrays.asList(tuples)));
85              ByteArrayOutputStream out = new ByteArrayOutputStream();
86  
87              format = format.toLowerCase();
88              RDFFormat outputFormat;
89              MediaType mediaType;
90              if (format.equalsIgnoreCase("xml") || format.equals("rdf/xml")) {
91                  outputFormat = RDFFormat.RDF_XML;
92                  mediaType = new MediaType("application", "rdf+xml");
93              } else if (format.equals("n-triples") || format.equals("ntriples")) {
94                  outputFormat = RDFFormat.N_TRIPLES;
95                  mediaType = MediaType.TEXT_PLAIN_TYPE;
96              } else if (format.equals("turtle")) {
97                  outputFormat = RDFFormat.TURTLE;
98                  mediaType = new MediaType("application", "x-turtle");
99              } else if (format.equals("sparql")) {
100                 outputFormat = RDFFormat.SPARQL;
101                 mediaType = new MediaType("application", "sparql-results+xml");
102             } else {
103                 throw new IllegalArgumentException("unknown format: " + format);
104             }
105             it.toStream(out, outputFormat);
106             return Response.ok(out.toString("UTF-8"), mediaType).build();
107         } catch (ServerException e) {
108             return handleException(e, flash);
109         } catch (TrippiException e) {
110             return handleException(e, flash);
111         } catch (UnsupportedEncodingException e) {
112             return handleException(e, flash);
113         }
114     }

137             @QueryParam(RestParam.DATATYPE)

139             @QueryParam(RestParam.FLASH)
140             @DefaultValue("false")
141             boolean flash) {
142         Context context = getContext();
143         try {
144             if (subject == null) {
145                 // assume the subject is the object as denoted by the pid
146                 subject = PID.toURI(pid);
147             }
148             boolean result = m_management.addRelationship(context, subject, predicate, object, isLiteral, datatype);
149             return Response.ok(Boolean.toString(result)).build(); // needs an entity to not be overridden with a 204
