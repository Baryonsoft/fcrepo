43      public DefaultSerializer(String fedoraServerHost, Context context) {
44          this.fedoraServerHost = fedoraServerHost;
45          this.fedoraServerPort = context.getEnvironmentValue(Constants.HTTP_REQUEST.SERVER_PORT.uri);
46          this.fedoraAppServerContext = context.getEnvironmentValue(Constants.FEDORA_APP_CONTEXT_NAME);
47  
48          if (Constants.HTTP_REQUEST.SECURE.uri
49                  .equals(context.getEnvironmentValue(Constants.HTTP_REQUEST.SECURITY.uri))) {
50              this.fedoraServerProtocol = "https";
51          } else {
52              this.fedoraServerProtocol = "http";
53          }
54      }

60          xml.append("<pidList "

63                  + " xmlns:xsi=\"http://www.w3.org/2001/XMLSchema-instance\""
64                  + " xsi:schemaLocation=\"" + Constants.PID_LIST1_0.namespace.uri + " "
65                  + Constants.PID_LIST1_0.xsdLocation + "\">");
66  
67          // PID array serialization
68          for (int i = 0; i < pidList.length; i++) {

75      public String objectProfileToXML(
76              ObjectProfile objProfile,
77              Date versDateTime)  {
78          StringBuilder buffer = new StringBuilder();
79  
80          String pid = objProfile.PID;
81          String dateString = "";
82          if (versDateTime != null) {
83              String tmp = DateUtility.convertDateToString(versDateTime);
84              if (tmp != null) {
85                  dateString = String.format("dateTime=\"%s\" ", tmp);
86              }
87          }

89          buffer.append("<?xml version=\"1.0\" encoding=\"UTF-8\"?>");
90          buffer.append("<objectProfile "
91                        + " xmlns=\"" + Constants.OBJ_PROFILE1_0.namespace.uri + "\" "
92                        + " xmlns:xsd=\"http://www.w3.org/2001/XMLSchema\""
93                        + " xmlns:xsi=\"http://www.w3.org/2001/XMLSchema-instance\""
94                        + " xsi:schemaLocation=\"" + Constants.OBJ_PROFILE1_0.namespace.uri + " "
95                        + Constants.OBJ_PROFILE1_0.xsdLocation + "\""
96                        + " pid=\"" + enc(pid) + "\" " + dateString + ">");
97  
98          // PROFILE FIELDS SERIALIZATION
99          buffer.append("<objLabel>" + enc(objProfile.objectLabel)
100                       + "</objLabel>");
101         buffer.append("<objOwnerId>" + enc(objProfile.objectOwnerId)
102                       + "</objOwnerId>");
103         buffer.append("<objModels>");
104         for (String model : objProfile.objectModels) {
105             buffer.append("<model>");
106             buffer.append(enc(model));
107             buffer.append("</model>");
108         }
109         buffer.append("</objModels>");
110         String cDate = DateUtility.convertDateToString(objProfile.objectCreateDate);
111         buffer.append("<objCreateDate>" + cDate + "</objCreateDate>");
112         String mDate = DateUtility.convertDateToString(objProfile.objectLastModDate);
113         buffer.append("<objLastModDate>" + mDate + "</objLastModDate>");
114         buffer.append("<objDissIndexViewURL>"
115                       + enc(objProfile.dissIndexViewURL)
116                       + "</objDissIndexViewURL>");
117         buffer.append("<objItemIndexViewURL>"
118                       + enc(objProfile.itemIndexViewURL)
119                       + "</objItemIndexViewURL>");
120         buffer.append("<objState>"
121                       + enc(objProfile.objectState)
122                       + "</objState>");
123         buffer.append("</objectProfile>");
124 
125         return buffer.toString();
126     }

128     private String datastreamFieldSerialization(Datastream dsProfile, String prefix, boolean validateChecksum) {
129         StringBuffer buffer = new StringBuffer();
130         if (prefix != null) {
131             if (!prefix.equals("")) prefix = prefix + ":";
132         } else prefix = "";
133         buffer.append("<" + prefix + "dsLabel>" + enc(dsProfile.DSLabel) + "</" + prefix + "dsLabel>");
134         buffer.append("<" + prefix + "dsVersionID>" + enc(dsProfile.DSVersionID) + "</" + prefix + "dsVersionID>");
135 
136         String cDate = DateUtility.convertDateToString(dsProfile.DSCreateDT);
137         buffer.append("<" + prefix + "dsCreateDate>" + enc(cDate) + "</" + prefix + "dsCreateDate>");
138 
139         buffer.append("<" + prefix + "dsState>" + enc(dsProfile.DSState) + "</" + prefix + "dsState>");
140         buffer.append("<" + prefix + "dsMIME>" + enc(dsProfile.DSMIME) + "</" + prefix + "dsMIME>");
141         buffer.append("<" + prefix + "dsFormatURI>" + enc(dsProfile.DSFormatURI) + "</" + prefix + "dsFormatURI>");
142         buffer.append("<" + prefix + "dsControlGroup>" + enc(dsProfile.DSControlGrp) + "</" + prefix + "dsControlGroup>");
143         buffer.append("<" + prefix + "dsSize>" + enc(Long.valueOf(dsProfile.DSSize).toString()) + "</" + prefix + "dsSize>");
144         buffer.append(
145                 "<" + prefix + "dsVersionable>"
146                 + enc(Boolean.valueOf(dsProfile.DSVersionable).toString())
147                 + "</" + prefix + "dsVersionable>");
148         buffer.append("<" + prefix + "dsInfoType>" + enc(dsProfile.DSInfoType) + "</" + prefix + "dsInfoType>");
149         buffer.append("<" + prefix + "dsLocation>" + enc(dsProfile.DSLocation) + "</" + prefix + "dsLocation>");
150         buffer.append("<" + prefix + "dsLocationType>" + enc(dsProfile.DSLocationType) + "</" + prefix + "dsLocationType>");
151         buffer.append("<" + prefix + "dsChecksumType>" + enc(dsProfile.DSChecksumType) + "</" + prefix + "dsChecksumType>");
152         buffer.append("<" + prefix + "dsChecksum>" + enc(dsProfile.DSChecksum) + "</" + prefix + "dsChecksum>");
153         if (validateChecksum) {
154             String valid = dsProfile.compareChecksum() ? "true" : "false";
155             buffer.append("<" + prefix + "dsChecksumValid>" + valid + "</" + prefix + "dsChecksumValid>");
156         }
157         String[] dsAltIDs = dsProfile.DatastreamAltIDs;
158         for (int i = 0; i < dsAltIDs.length; i++) {
159             buffer.append("<" + prefix + "dsAltID>" + enc(dsAltIDs[i]) + "</" + prefix + "dsAltID>");
160         }
161         return buffer.toString();
162     }

191     String datastreamProfilesToXML(String pid, Datastream[] dsProfiles, Date versDateTime,
192                                   boolean validateChecksum){
193         StringBuilder builder=new StringBuilder();
194         String dateString = "";
195         if (versDateTime != null) {
196             String tmp = DateUtility.convertDateToString(versDateTime);
197             if (tmp != null) {
198                 dateString = String.format("asOfDateTime=\"%s\" ", tmp);
199             }
200         }
201         String baseUrl = enc(fedoraServerProtocol)
202                 + "://" + enc(fedoraServerHost) + ":"
203                 + enc(fedoraServerPort) + "/" + fedoraAppServerContext;
204         builder.append("<?xml version=\"1.0\" encoding=\"UTF-8\"?>")
205             .append("<objectDatastreams xmlns=\"" + Constants.ACCESS.uri + "\" ")
206             .append("xmlns:apim=\"" + Constants.MANAGEMENT.uri + "\" ")
207             .append("xmlns:xsd=\"http://www.w3.org/2001/XMLSchema\" ")
208             .append("xmlns:xsi=\"http://www.w3.org/2001/XMLSchema-instance\" ")
209             .append("xsi:schemaLocation=\"http://www.fedora.info/definitions/1/0/access/ ")
210             .append(baseUrl).append("/schema/listDatastreams.xsd\"")
211             .append(" pid=\"").append(enc(pid)).append("\" ").append(dateString)
212             .append(" baseURL=\"").append(baseUrl).append("/\" >");
213         for (Datastream ds:dsProfiles){
214             StringBuilder profileBuilder=new StringBuilder();
215             profileBuilder.append("<datastreamProfile pid=\"").append(enc(pid))
216                 .append("\" dsID=\"").append(enc(ds.DatastreamID)).append("\" >");
217             profileBuilder.append(datastreamFieldSerialization(ds, "apim", validateChecksum));
218             profileBuilder.append("</datastreamProfile>");
219             builder.append(profileBuilder.toString());
220         }
221         builder.append("</objectDatastreams>");
222         return builder.toString();
223     }

272     public String objectMethodsToXml(
273             ObjectMethodsDef[] methodDefs,
274             String pid,
275             String sDef,
276             Date versDateTime) {
277         StringBuffer buffer = new StringBuffer();
278 
279         buffer.append("<?xml version=\"1.0\" encoding=\"UTF-8\"?>");
280         String asOfDateTimeElement = "";
281         if (versDateTime != null) {
282             asOfDateTimeElement = "asOfDateTime=\""
283                                   + DateUtility.convertDateToString(versDateTime) + "\" ";
284         }
285         String sDefElement = "";
286         if (sDef != null) {
287             sDefElement = "sDef=\"" + enc(sDef) + "\" ";
288         }
289         buffer.append("<objectMethods "
290                       + "xmlns=\"" + Constants.OBJ_METHODS1_0.namespace.uri + "\" "
291                       + "xmlns:xsd=\"http://www.w3.org/2001/XMLSchema\" "
292                       + "xmlns:xsi=\"http://www.w3.org/2001/XMLSchema-instance\" "
293                       + "xsi:schemaLocation=\"" + Constants.OBJ_METHODS1_0.namespace.uri + " "
294                       + Constants.OBJ_METHODS1_0.xsdLocation + "\" "
295                       + " pid=\"" + enc(pid) + "\" "
296                       + asOfDateTimeElement + sDefElement + "baseURL=\""
297                       + enc(fedoraServerProtocol) + "://"
298                       + enc(fedoraServerHost) + ":"
299                       + enc(fedoraServerPort) + "/" + fedoraAppServerContext + "/\" >");
300 
301         // ObjectMethodsDef SERIALIZATION
302         String nextSdef = "null";
303         String currentSdef = "";
304         for (int i = 0; i < methodDefs.length; i++) {
305             currentSdef = methodDefs[i].sDefPID;
306             if (sDef == null || currentSdef.equals(sDef)) {
307                 if (!currentSdef.equalsIgnoreCase(nextSdef)) {
308                     if (!nextSdef.equals("null")) {
309                         buffer.append("</sDef>");
310                     }
311                     buffer.append("<sDef pid=\"" + enc(methodDefs[i].sDefPID)
312                                   + "\" >");
313                 }
314                 buffer.append("<method name=\"" + enc(methodDefs[i].methodName)
315                               + "\" >");
316                 MethodParmDef[] methodParms = methodDefs[i].methodParmDefs;
317                 for (int j = 0; j < methodParms.length; j++) {
318                     buffer.append("<methodParm parmName=\""
319                                   + enc(methodParms[j].parmName)
320                                   + "\" parmDefaultValue=\""
321                                   + enc(methodParms[j].parmDefaultValue)
322                                   + "\" parmRequired=\"" + methodParms[j].parmRequired
323                                   + "\" parmLabel=\"" + enc(methodParms[j].parmLabel)
324                                   + "\" >");
325                     if (methodParms[j].parmDomainValues.length > 0) {
326                         buffer.append("<methodParmDomain>");
327                         for (int k = 0; k < methodParms[j].parmDomainValues.length; k++) {
328                             buffer.append("<methodParmValue>"
329                                           + enc(methodParms[j].parmDomainValues[k])
330                                           + "</methodParmValue>");
331                         }
332                         buffer.append("</methodParmDomain>");
333                     }
334                     buffer.append("</methodParm>");
335                 }
336 
337                 buffer.append("</method>");
338                 nextSdef = currentSdef;
339             }
340         }
341         if (!nextSdef.equals("null")) {
342             buffer.append("</sDef>");
343         }
344         buffer.append("</objectMethods>");
345 
346         return buffer.toString();
347     }
