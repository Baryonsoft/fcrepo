5   package org.fcrepo.server.utilities;

7   import java.io.File;

9   import javax.servlet.ServletContextEvent;
10  import javax.servlet.ServletContextListener;
11  
12  import org.fcrepo.common.Constants;
13  import org.fcrepo.utilities.LogConfig;
14  import org.slf4j.bridge.SLF4JBridgeHandler;
15  
16  /**
17   * Context listener for logging initialization.
18   * <p>
19   * This ensures that logging is initialized before any filters or servlets are
20   * started.
21   *
22   * @author Chris Wilper
23   */
24  public class LogSetupContextListener implements ServletContextListener {
25  
26      @Override
27      public void contextInitialized(ServletContextEvent event) {
28          // If fedora.home servlet context init param is defined, make sure
29          // it is used for the value of Constants.FEDORA_HOME
30          String contextFH = event.getServletContext().getInitParameter("fedora.home");
31          if (contextFH != null && !contextFH.equals("")) {
32              System.setProperty("servlet.fedora.home", contextFH);
33          }
34  
35          // Configure logging from file
36          System.setProperty("fedora.home", Constants.FEDORA_HOME);
37          LogConfig.initFromFile(new File(new File(Constants.FEDORA_HOME),
38                                          "server/config/logback.xml"));
39  
40          // Replace java.util.logging's default handlers with one that
41          // redirects everything to SLF4J
42          java.util.logging.Logger rootLogger =
43                  java.util.logging.LogManager.getLogManager().getLogger("");
44          java.util.logging.Handler[] handlers = rootLogger.getHandlers();
45          for (int i = 0; i < handlers.length; i++) {
46              rootLogger.removeHandler(handlers[i]);
47          }
48          SLF4JBridgeHandler.install();
49      }
50  
51      @Override
52      public void contextDestroyed(ServletContextEvent event) {
53          // no cleanup needed for this listener
54      }
55  }
