19      /**
20       * Method that attempts to break a string up into lines no longer than the
21       * specified line length.
22       *
23       * <p>The string is assumed to consist of tokens separated by a delimeter.
24       * The default delimiter is a space. If the last token to be added to a
25       * line exceeds the specified line length, it is written on the next line
26       * so actual line length is approximate given the specified line length
27       * and the length of tokens in the string.
28       *
29       * @param in
30       *        The input string to be split into lines.
31       * @param lineLength
32       *        The maximum length of each line.
33       * @param delim
34       *        The character delimiter separating each token in the input string;
35       *        if null, defaults to the space character.
36       * @return A string split into multiple lines whose lenght is less than the
37       *         specified length. Actual length is approximate depending on line
38       *         length, token size, and how many complete tokens will fit into
39       *         the specified line length.
40       */
41      public static String prettyPrint(String in, int lineLength, String delim) {
42          StringBuffer sb = new StringBuffer();
43          if (delim == null) {
44              delim = " ";
45          }
46          StringTokenizer st = new StringTokenizer(in, delim);
47          int charCount = 0;
48          while (st.hasMoreTokens()) {
49              String s = st.nextToken();
50              charCount = charCount + s.length();
51              if (charCount < lineLength) {
52                  sb.append(s + " ");
53                  charCount++;
54              } else {
55                  sb.append("\n");
56                  sb.append(s + " ");
57                  charCount = s.length() + 1;
58              }
59          }
60          return sb.toString();
61      }

63      /**
64       * Method that attempts to break a string up into lines no longer than the
65       * specified line length.
66       *
67       * <p>The string is assumed to a large chunk of undifferentiated text such
68       * as base 64 encoded binary data.
69       *
70       * @param str
71       *        The input string to be split into lines.
72       * @param indent
73       *        The number of spaces to insert at the start of each line.
74       * @param numChars
75       *        The maximum length of each line (not counting the indent spaces).
76       * @return A string split into multiple indented lines whose length is less
77       *         than the specified length + indent amount.
78       */
79      public static String splitAndIndent(String str, int indent, int numChars) {
80          StringBuffer sb = new StringBuffer();
81          StringBuffer ib = new StringBuffer();
82          for (int i = 0; i < indent; i++) {
83              ib.append(" ");
84          }
85          String indentStr = ib.toString();
86          int offset = 0;
87          int totalLength = str.length();
88          while (totalLength > numChars) {
89              String s = str.substring(offset, offset + numChars);
90              sb.append(indentStr);
91              sb.append(s);
92              sb.append("\n");
93              offset += numChars;
94              totalLength -= numChars;
95          }
96          String s = str.substring(offset);
97          sb.append(indentStr);
98          sb.append(s);
99          sb.append("\n");
100 
101         return sb.toString();
102 
103     }

105     public static String byteArraytoHexString(byte[] array) {
106         final String hexByte = "0123456789abcdef";
107         StringBuffer buf = new StringBuffer();
108         for (byte val : array) {
109             int v1 = val >>> 4 & 0x0f;
110             int v2 = val & 0x0f;
111             buf.append(hexByte.substring(v1, v1 + 1)).append(hexByte
112                     .substring(v2, v2 + 1));
113         }
114         return buf.toString();
115     }

117     public static byte[] hexStringtoByteArray(String str) {
118         if ((str.length() & 0x01) != 0) {
119             throw new NumberFormatException();
120         }
121         byte ret[] = new byte[str.length() / 2];
122         for (int i = 0; i < str.length() / 2; i++) {
123             ret[i] =
124                     (byte) Integer
125                             .parseInt(str.substring(i * 2, i * 2 + 2), 16);
126         }
127         return ret;
128     }
