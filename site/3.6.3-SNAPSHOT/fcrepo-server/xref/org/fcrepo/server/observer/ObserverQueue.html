5   package org.fcrepo.server.observer;
6   
7   import java.util.Set;
8   import java.util.concurrent.BlockingQueue;
9   import java.util.concurrent.CopyOnWriteArraySet;
10  import java.util.concurrent.LinkedBlockingQueue;
11  
12  /**
13   * A threaded FIFO queue to improve performance of updates to Observers.
14   * 
15   * @author Edwin Shin
16   * @version $Id$
17   * @see <a
18   *      href="http://www.javaworld.com/javaworld/javatips/jw-javatip29.html">http://www.javaworld.com/javaworld/javatips/jw-javatip29.html</a>
19   */
20  public class ObserverQueue
21          implements Publisher, Subscriber, Runnable {
22  
23      private final Set<Subscriber> subscribers;
24  
25      private BlockingQueue<Object> messages;
26  
27      public ObserverQueue() {
28          subscribers = new CopyOnWriteArraySet<Subscriber>();
29          messages = new LinkedBlockingQueue<Object>();
30      }
31  
32      public void update(Publisher o, Object arg) {
33          messages.add(arg);
34      }
35  
36      /*
37       * (non-Javadoc)
38       * 
39       * @see java.lang.Runnable#run()
40       */
41      public void run() {
42          while (true) {
43              Object obj;
44              try {
45                  obj = messages.take();
46                  notifySubscribers(obj);
47              } catch (InterruptedException e) {
48                  // restore interrupted status
49                  Thread.currentThread().interrupt();
50              }
51          }
52      }
53  
54      public void addSubscriber(Subscriber obs) {
55          subscribers.add(obs);
56      }
57  
58      public void notifySubscribers() {
59          notifySubscribers(null);
60      }
61  
62      /**
63       * Notify subscribers. Notifications are issued using a copy of the
64       * subscriber list. Therefore, Subscribers should not assume that a
65       * notification will not be received even after calling removeSubscriber.
66       * {@inheritDoc}
67       */
68      public void notifySubscribers(Object o) {
69          for (Subscriber subscriber : subscribers) {
70              subscriber.update(this, o);
71          }
72      }
73  
74      public void removeSubscriber(Subscriber obs) {
75          subscribers.remove(obs);
76      }
77  
78  }
