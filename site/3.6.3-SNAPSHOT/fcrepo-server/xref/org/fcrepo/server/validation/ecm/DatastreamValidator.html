36      public DatastreamValidator(RepositoryReader doMgr) {

38          schemaValidator = new SchemaValidator(doMgr);
39          formValidator = new FormValidator();

42      public void validate(Context context, DOReader currentObjectReader, Date asOfDateTime,
43                           Validation validation, ExternalContentManager m_exExternalContentManager) throws ServerException {
44  
45  
46          // /Datastream validation stuff
47  
48          /** Get the objects content models */
49          List<String> contentmodels = currentObjectReader.getContentModels();
50  
51          //For each content model, parse the DS-COMPOSITE-MODEL
52          for (String contentmodel : contentmodels) {
53              contentmodel = contentmodel.substring("info:fedora/".length());
54  
55              DOReader contentmodelReader = doMgr.getReader(false, context, contentmodel);
56              Datastream dscompmodelDS = contentmodelReader.GetDatastream("DS-COMPOSITE-MODEL", asOfDateTime);
57  
58              if (dscompmodelDS == null) {//NO ds composite model, thats okay, continue to next content model
59                  continue;
60              }
61              DsCompositeModel dscompobject = JAXB.unmarshal(dscompmodelDS.getContentStream(), DsCompositeModel.class);
62              for (DsTypeModel typeModel : dscompobject.getDsTypeModel()) {
63                  String DSID = typeModel.getID();
64                  Datastream objectDatastream = currentObjectReader.GetDatastream(DSID, asOfDateTime);
65  
66                  if (objectDatastream == null) {
67                      Boolean optional = typeModel.isOptional();
68                      if (optional != null && optional) {
69                          //optional datastream can be missing, not a problem
70                      } else {
71                          reportMissingDatastreamError(contentmodel, DSID, validation);
72                      }
73                      continue;
74                  }
75  
76  
77                  formValidator.checkFormAndMime(typeModel, objectDatastream, validation, contentmodelReader);
78                  schemaValidator
79                          .validate(context, typeModel, objectDatastream, validation, contentmodelReader, asOfDateTime,m_exExternalContentManager);
80              }
81          }
82  
83  
84      }
85  
86      private void reportMissingDatastreamError(String contentmodel, String dsid, Validation validation) {
87          List<String> problems = validation.getDatastreamProblems().get(dsid);
88          if (problems == null) {
89              problems = new ArrayList<String>();
90              validation.getDatastreamProblems().put(dsid, problems);
91          }
92  
93  
94          problems.add(Errors.missingRequiredDatastream(dsid,contentmodel));
95          validation.setValid(false);
96      }
