30  /**
31   * Validates the RDF/XML content of the RELS-EXT and RELS-INT datastreams.
32   * <p>
33   * The following restrictions are enforced:
34   * <ul>
35   * <li>The RDF must follow a prescribed RDF/XML authoring style where there is
36   * ONE (RELS-EXT) or more (RELS-INT) subjects encoded as RDF &lt;Description&gt;
37   * elements with RDF <code>about</code> attribute containing either a digital
38   * object URI (RELS-EXT) or a datastream URI (RELS-INT). The sub-elements are
39   * the relationship properties of the subject. Each relationship may refer to
40   * any resource (identified by URI) via an RDF 'resource' attribute, or a
41   * literal. Relationship assertions can be from the default Fedora relationship
42   * ontology, or from other namespaces.
43   * <ul>
44   * <li>RELS-EXT example:
45   *
46   * <pre>&lt;rdf:Description about="info:fedora/demo:5"&gt;
47   *   &lt;fedora:isMemberOfCollection resource="info:fedora/demo:100"/&gt;
48   *   &lt;nsdl:isAugmentedBy resource="info:fedora/demo:333"/&gt;
49   *   &lt;example:source resource="http://example.org/bsmith/article1.html"/&gt;
50   *   &lt;example:primaryAuthor&gt;Bob Smith&lt;/example:primaryAuthor&gt;
51   * &lt;/rdf:Description&gt;
52   * </pre>
53   * </li>
54   * <li>RELS-INT example:
55   *
56   * <pre>&lt;rdf:Description about="info:fedora/demo:5/DS1"&gt;
57   *   &lt;nsdl:isAugmentedBy resource="info:fedora/demo:333"/&gt;
58   *   &lt;example:source resource="http://example.org/bsmith/article1.html"/&gt;
59   *   &lt;example:primaryAuthor&gt;Bob Smith&lt;/example:primaryAuthor&gt;
60   * &lt;/rdf:Description&gt;
61   * </pre>
62   * </li>
63   * </ul>
64   * </li>
65   * <li>For RELS-EXT, there must be only ONE &lt;rdf:Description&gt; element.</li>
66   * <li>There must be NO nesting of assertions. In terms of XML depth, the RDF
67   * root element is considered depth of 0. Then, the &lt;rdf:Description&gt;
68   * element must be at depth of 1, and the relationship properties must exist at
69   * depth of 2. That's it.</li>
70   * <li>For RELS-EXT, The RDF <code>about</code> attribute of the RDF
71   * &lt;Description&gt; must be the URI of the digital object in which the
72   * RELS-EXT datastream resides. This means that all relationships are FROM
73   * "this" object to other objects.</li>
74   * <li>For RELS-INT, the RDF <code>about</code> attribute(s) of the RDF
75   * &lt;Description&gt; element(s) must be valid URIs of datastreams for the
76   * digital object in which the RELS-EXT datastream resides. The datastreams do
77   * not actually have to exist, but these URIs must be syntactically valid. This
78   * means that all relationships are FROM datastreams in "this" object to other
79   * objects.</li>
80   * <li>If the target of the statement is a resource (identified by a URI), the
81   * RDF <code>resource</code> attribute must specify a syntactically valid,
82   * absolute URI.</li>
83   * <li>For RELS-EXT, there must NOT be any assertion of properties from the DC
84   * namespace.</li>
85   * <li>There must NOT be any assertions of properties from the Fedora object
86   * properties namespaces (model and view), with the following exceptions for
87   * RELS-EXT only:
88   *
89   * <pre>
90   * fedora-model:hasService
91   * fedora-model:hasModel
92   * fedora-model:isDeploymentOf
93   * fedora-model:isContractorOf
94   *        These assertions are allowed in the RELS-EXT datastream, but all
95   *        others from the <code>fedora-model</code> and <code>fedora-view</code>
96   *        namespaces are inferred from values expressed elsewhere in the
97   *        digital object, and we do not want duplication.
98   * </li>
99   * </ul>
100  *
101  * @author Sandy Payette
102  * @author Eddie Shin
103  * @author Chris Wilper
104  * @author Stephen Bayliss
105  */
106 public class RelsValidator

139     public RelsValidator() {
140         try {
141             SAXParserFactory spf = SAXParserFactory.newInstance();
142             spf.setNamespaceAware(true);
143             m_parser = spf.newSAXParser();
144         } catch (Exception wontHappen) {
145             throw new FaultException(wontHappen);

149     public void validate(PID pid, String dsId, InputStream content)
150             throws ValidationException {
151         try {
152             m_rootRDFFound = false;
153             m_descriptionCount = 0;
154             m_withinDescription = false;
155             m_depth = 0;
156             m_doURI = pid.toURI();
157             if (!dsId.equals(RELS_EXT) && !dsId.equals(RELS_INT)) {
158                 throw new ValidationException("Relationships datastream ID must be RELS-EXT or RELS-INT ("
159                         + dsId + ")");
160             }
161             m_dsId = dsId;
162             m_parser.parse(content, this);
163         } catch (Exception e) {
164             throw new ValidationException(dsId + " validation failed: "
165                     + e.getMessage(), e);
166         }
167     }

169     @Override
170     public void startElement(String nsURI,
171                              String localName,
172                              String qName,
173                              Attributes a) throws SAXException {
174 
175         if (nsURI.equals(RDF.uri) && localName.equalsIgnoreCase("RDF")) {
176             m_rootRDFFound = true;
177         } else if (m_rootRDFFound) {
178             if (nsURI.equals(RDF.uri)
179                     && localName.equalsIgnoreCase("Description")) {
180                 // are we not already within a Description element?
181                 if (!m_withinDescription) {
182                     m_withinDescription = true;
183                     m_descriptionCount++;
184                     m_depth++;
185                     if ((m_descriptionCount > 1) && m_dsId.equals(RELS_EXT)) {
186                         throw new SAXException("RelsExtValidator:"
187                                 + " Only ONE RDF <Description> element is allowed"
188                                 + " in the RELS-EXT datastream.");
189                     }
190                     checkDepth(m_depth, qName);
191                     checkAboutURI(grab(a, RDF.uri, "about"));
192 
193                 } else {
194                     throw new SAXException("RelsExtValidator:"
195                             + " RDF <Description> elements may not be nested"
196                             + " in the " + m_dsId + " datastream.");
197                 }
198             } else if (m_withinDescription) {
199                 m_depth++;
200                 checkDepth(m_depth, qName);
201                 checkBadAssertion(nsURI, localName, qName);
202                 String resourceURI = grab(a, RDF.uri, "resource");
203                 if (resourceURI.length() > 0) {
204                     checkResourceURI(resourceURI, qName);
205                     m_literalType = null;
206                     m_literalValue = null;
207                 } else {
208                     if (nsURI.equals(MODEL.uri)) {
209                         // if it's not a resource, the predicate cannot
210                         // be fedora-model:hasService, hasModel,
211                         // isContractor, or is DeploymentOf
212                         if (localName.equals(MODEL.HAS_SERVICE.localName)
213                                 || localName.equals(MODEL.HAS_MODEL.localName)
214                                 || localName
215                                         .equals(MODEL.IS_CONTRACTOR_OF.localName)
216                                 || localName
217                                         .equals(MODEL.IS_DEPLOYMENT_OF.localName)) {
218                             throw new SAXException("RelsExtValidator: "
219                                     + "Target of " + qName + " statement "
220                                     + "MUST be an rdf:resource");
221                         }
222                     }
223                     String datatypeURI = grab(a, RDF.uri, "datatype");
224                     if (datatypeURI.length() == 0) {
225                         m_literalType = null;
226                     } else {
227                         m_literalType = datatypeURI;
228                     }
229                     m_literalValue = new StringBuffer();
230                 }
231             } else {
232                 throw new SAXException("RelsExtValidator:"
233                         + " Invalid element " + localName
234                         + " found in the RELS-EXT datastream.\n"
235                         + " Relationship assertions must be built"
236                         + " upon an RDF <Description> element.");
237             }
238         } else {
239             throw new SAXException("RelsExtValidator:"
240 
241             + " The 'RDF' root element was not found "
242                     + " in the RELS-EXT datastream.\n"
243                     + " Relationship metadata must be encoded using RDF/XML.");
244         }
245     }

254     @Override
255     public void endElement(String nsURI, String localName, String qName)
256             throws SAXException {
257         if (m_rootRDFFound) {
258             m_depth--;
259         }
260         if (nsURI.equals(RDF.uri) && localName.equalsIgnoreCase("Description")) {
261             m_withinDescription = false;
262         }
263         if (m_literalType != null && m_literalValue != null) {
264             checkTypedValue(m_literalType, m_literalValue.toString(), qName);
265         }
266         m_literalType = null;
267         m_literalValue = null;
268     }

270     private static String grab(Attributes a,
271                                String namespace,
272                                String elementName) {
273         String ret = a.getValue(namespace, elementName);
274         if (ret == null) {
275             ret = a.getValue(elementName);
276         }
277         // set null attribute value to empty string since it's
278         // generally helpful in the code to avoid null pointer exception
279         // when operations are performed on attributes values.
280         if (ret == null) {
281             ret = "";
282         }
283         return ret;
284     }

286     /**
287      * checkDepth: checks that there is NO nesting of relationship assertions.
288      * In terms of XML depth, the RDF root element is considered depth of 0.
289      * Then, the RDF <Description> must be at depth of 1, and the relationship
290      * properties must exist at depth of 2. That's it.
291      *
292      * @param depth
293      *        the depth of the XML element being evaluated
294      * @param qName
295      *        the name of the relationship property being evaluated
296      * @throws SAXException
297      */
298     private void checkDepth(int depth, String qName) throws SAXException {
299 
300         if (depth > 2) {
301             throw new SAXException("RelsExtValidator:"
302                     + " The RELS-EXT datastream has improper"
303                     + " nesting in its relationship assertions.\n"
304                     + " (The XML depth is " + depth
305                     + " which must not exceed a depth of 2.\n"
306                     + " The root <RDF> element should be level 0,"
307                     + " the <Description> element should be level 1,"
308                     + " and relationship elements should be level 2.)");
309         }
310     }

312     /**
313      * checkBadAssertion: checks that the DC and fedora-view namespace are not
314      * being used in RELS-EXT, and that if fedora-model is used, the localName
315      * is hasService, hasModel, isDeploymentOf, or isContractorOf. Also ensures
316      * that fedora-model:hasContentModel is only used once.
317      *
318      * @param nsURI
319      *        the namespace URI of the predicate being evaluated
320      * @param localName
321      *        the local name of the predicate being evaluated
322      * @param qName
323      *        the qualified name of the predicate being evaluated
324      */
325     private void checkBadAssertion(String nsURI, String localName, String qName)
326             throws SAXException {
327 
328         if (m_dsId.equals(RELS_EXT)
329                 && (nsURI.equals(DC.uri) || nsURI.equals(OAI_DC.uri))) {
330             throw new SAXException("RelsExtValidator:"
331                     + " The RELS-EXT datastream has improper"
332                     + " relationship assertion: " + qName + ".\n"
333                     + " No Dublin Core assertions allowed"
334                     + " in Fedora relationship metadata.");
335         } else if (nsURI.equals(MODEL.uri)) {
336             if ((m_dsId.equals(RELS_INT)
337                     && !localName.equals(MODEL.DOWNLOAD_FILENAME.localName)                    )
338                     ||
339                     (m_dsId.equals(RELS_EXT)
340                     && !localName.equals(MODEL.HAS_SERVICE.localName)
341                     && !localName.equals(MODEL.IS_CONTRACTOR_OF.localName)
342                     && !localName.equals(MODEL.HAS_MODEL.localName)
343                     && !localName.equals(MODEL.IS_DEPLOYMENT_OF.localName)
344                     )) {
345                 throw new SAXException("RelsExtValidator:"
346                         + " Disallowed predicate in " + m_dsId + ": "
347                         + qName
348                         + "\n"
349                         + " The only predicates from the fedora-model namespace"
350                         + " allowed in RELS-EXT are "
351                         + MODEL.HAS_SERVICE.localName + ", "
352                         + MODEL.IS_CONTRACTOR_OF.localName + ", "
353                         + MODEL.HAS_MODEL.localName + ", "
354                         + MODEL.IS_DEPLOYMENT_OF.localName + ".  The only predicate allowed "
355                         + "in RELS-INT is "
356                         + MODEL.DOWNLOAD_FILENAME.localName +"."
357                         );
358             }
359         } else if (nsURI.equals(VIEW.uri)) {
360             throw new SAXException("RelsExtValidator:"
361                     + " Disallowed predicate in RELS-EXT: " + qName + "\n"
362                     + " The fedora-view namespace is reserved by Fedora.");
363         }
364     }

366     /**
367      * checkAboutURI: ensure that the RDF <Description> is about the digital
368      * object that contains the RELS-EXT datastream, since the REL-EXT
369      * datastream is only supposed to capture relationships about "this" digital
370      * object.
371      *
372      * @param aboutURI
373      *        the URI value of the RDF 'about' attribute
374      * @throws SAXException
375      */
376     private void checkAboutURI(String aboutURI) throws SAXException {
377         if (m_dsId.equals(RELS_EXT)) {
378             if (!m_doURI.equals(aboutURI)) {
379                 throw new SAXException("RelsExtValidator:"
380                         + " The RELS-EXT datastream refers to"
381                         + " an improper URI in the 'about' attribute of the"
382                         + " RDF <Description> element.\n"
383                         + " The URI must be that of the digital object"
384                         + " in which the RELS-EXT datastream resides" + " ("
385                         + m_doURI + ").");
386             }
387         } else if (m_dsId.equals(RELS_INT)) {
388             if (!aboutURI.startsWith(m_doURI + "/")) {
389                 throw new SAXException("RelsExtValidator:"
390                         + " The RELS-INT datastream refers to"
391                         + " an improper URI in the 'about' attribute of the"
392                         + " RDF <Description> element.\n"
393                         + " The URI must be a datastream in the digital object"
394                         + " in which the RELS-INT datastream resides" + " ("
395                         + m_doURI + ", " + aboutURI + ").");
396 
397             }
398             String dsId = aboutURI.replace(m_doURI + "/", "");
399             // datastream ID must be an XML NCName, implemented using NCName class
400             if (dsId.length() > ValidationConstants.DATASTREAM_ID_MAXLEN
401                     || dsId.length() < 1 || !isValid(dsId)) {
402                 throw new SAXException("RelsExtValidator:"
403                         + " The RELS-INT datastream refers to"
404                         + " an improper URI in the 'about' attribute of the"
405                         + " RDF <Description> element.\n" + " " + dsId
406                         + " is not a valid datastream ID");
407 
408             }
409         }
410     }

412     /**
413      * validate the value against the xsd definition
414      * <p/>
415      * NCName ::=  (Letter | '_') (NCNameChar)* NCNameChar ::=  Letter | Digit | '.' | '-' | '_' |
416      * CombiningChar | Extender
417      */
418     private static boolean isValid(String stValue) {
419         int scan;
420         boolean bValid = true;
421 
422         for (scan = 0; scan < stValue.length(); scan++) {
423             if (scan == 0)
424                 bValid = XMLChar.isNCNameStart(stValue.charAt(scan));
425             else
426                 bValid = XMLChar.isNCName(stValue.charAt(scan));
427             if (!bValid)
428                 break;
429         }
430         return bValid;
431     }

433     /**
434      * checkResourceURI: ensure that the target resource is a proper URI.
435      *
436      * @param resourceURI
437      *        the URI value of the RDF 'resource' attribute
438      * @param relName
439      *        the name of the relationship property being evaluated
440      * @throws SAXException
441      */
442     private void checkResourceURI(String resourceURI, String relName)
443             throws SAXException {
444 
445         URI uri;
446         try {
447             uri = new URI(resourceURI);
448         } catch (Exception e) {
449             throw new SAXException("RelsExtValidator:"
450                     + "Error in relationship '" + relName + "'."
451                     + " The RDF 'resource' is not a valid URI.");
452         }
453 
454         if (!uri.isAbsolute()) {
455             throw new SAXException("RelsValidator:" + "Error in relationship '"
456                     + relName + "'."
457                     + " The specified RDF 'resource' is not an absolute URI.");
458         }
459     }

461     /**
462      * checkTypedValue: ensure that the datatype of a literal is one of the
463      * supported types and that it's a valid value for that type.
464      *
465      * @param datatypeURI
466      *        the URI value of the RDF 'datatype' attribute
467      * @param value
468      *        the value
469      * @param relName
470      *        the name of the property being evaluated
471      * @throws SAXException
472      */
473     private void checkTypedValue(String datatypeURI,
474                                  String value,
475                                  String relName) throws SAXException {
476         if (datatypeURI.equals(RDF_XSD.INT.uri)) {
477             try {
478                 Integer.parseInt(value);
479             } catch (Exception e) {
480                 throw new SAXException("RelsExtValidator:"
481                         + " The value specified for " + relName
482                         + " is not a valid 'int' value");
483             }
484         } else if (datatypeURI.equals(RDF_XSD.LONG.uri)) {
485             try {
486                 Long.parseLong(value);
487             } catch (Exception e) {
488                 throw new SAXException("RelsExtValidator:"
489                         + " The value specified for " + relName
490                         + " is not a valid 'long' value");
491             }
492         } else if (datatypeURI.equals(RDF_XSD.FLOAT.uri)) {
493             try {
494                 Float.parseFloat(value);
495             } catch (Exception e) {
496                 throw new SAXException("RelsExtValidator:"
497                         + " The value specified for " + relName
498                         + " is not a valid 'float' value");
499             }
500         } else if (datatypeURI.equals(RDF_XSD.DOUBLE.uri)) {
501             try {
502                 Double.parseDouble(value);
503             } catch (Exception e) {
504                 throw new SAXException("RelsExtValidator:"
505                         + " The value specified for " + relName
506                         + " is not a valid 'double' value");
507             }
508         } else if (datatypeURI.equals(RDF_XSD.DATE_TIME.uri)) {
509             if (!isValidDateTime(value)) {
510                 throw new SAXException("RelsExtValidator:"
511                         + " The value specified for " + relName
512                         + " is not a valid 'dateTime' value.\n"
513                         + "The following dateTime formats are allowed:\n"
514                         + "  yyyy-MM-ddTHH:mm:ss\n"
515                         + "  yyyy-MM-ddTHH:mm:ss.S\n"
516                         + "  yyyy-MM-ddTHH:mm:ss.SS\n"
517                         + "  yyyy-MM-ddTHH:mm:ss.SSS\n"
518                         + "  yyyy-MM-ddTHH:mm:ss.SSSZ");
519             }
520         } else {
521             throw new SAXException("RelsExtValidator:"
522                     + " Error in relationship '"
523                     + relName
524                     + "'.\n"
525                     + " The RELS-EXT datastream does not support the specified"
526                     + " datatype.\n"
527                     + "If specified, the RDF 'datatype' must be the URI of one of\n"
528                     + "the following W3C XML Schema data types: int, long, float,\n"
529                     + "double, or dateTime");
530         }
531     }

533     /**
534      * Tells whether the given string is a valid lexical representation of a
535      * dateTime value. Passing this test will ensure successful indexing later.
536      */
537     private static boolean isValidDateTime(String lex) {
538         SimpleDateFormat format = new SimpleDateFormat();
539 
540         format.setTimeZone(TimeZone.getTimeZone("UTC"));
541         int length = lex.length();
542         if (lex.startsWith("-")) {
543             length--;
544         }
545         if (lex.endsWith("Z")) {
546             if (length == 20) {
547                 format.applyPattern("yyyy-MM-dd'T'HH:mm:ss'Z'");
548             } else if (length == 22) {
549                 format.applyPattern("yyyy-MM-dd'T'HH:mm:ss.S'Z'");
550             } else if (length == 23) {
551                 format.applyPattern("yyyy-MM-dd'T'HH:mm:ss.SS'Z'");
552             } else if (length == 24) {
553                 format.applyPattern("yyyy-MM-dd'T'HH:mm:ss.SSS'Z'");
554             } else {
555                 logger.warn("Not a valid dateTime: " + lex);
556                 return false;
557             }
558         } else {
559             if (length == 19) {
560                 format.applyPattern("yyyy-MM-dd'T'HH:mm:ss");
561             } else if (length == 21) {
562                 format.applyPattern("yyyy-MM-dd'T'HH:mm:ss.S");
563             } else if (length == 22) {
564                 format.applyPattern("yyyy-MM-dd'T'HH:mm:ss.SS");
565             } else if (length == 23) {
566                 format.applyPattern("yyyy-MM-dd'T'HH:mm:ss.SSS");
567             } else {
568                 logger.warn("Not a valid dateTime: " + lex);
569                 return false;
570             }
571         }
572         try {
573             format.parse(lex);
574             if (logger.isTraceEnabled()) {
575                 logger.trace("Validated dateTime: " + lex);
576             }
577             return true;
578         } catch (ParseException e) {
579             logger.warn("Not a valid dateTime: " + lex);
580             return false;
581         }
582     }
