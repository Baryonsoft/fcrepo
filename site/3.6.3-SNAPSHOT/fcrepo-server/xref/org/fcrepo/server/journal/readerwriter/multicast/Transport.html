8   import java.util.Date;
9   import java.util.Map;
10  
11  import javax.xml.stream.XMLEventWriter;
12  
13  import org.fcrepo.server.journal.JournalException;
14  
15  
16  /**
17   * <p>
18   * Transport.java
19   * </p>
20   * <p>
21   * The abstract superclass of Journal Transport objects. It enforces the
22   * constructors of subclasses, by requiring parent, server, etc.
23   * </p>
24   * <p>
25   * Sub-classes must implement the four life-cycle methods:
26   * {@link #openFile(String, String, Date) openFile},
27   * {@link #getWriter() getWriter}, {@link #closeFile() closeFile}, and
28   * {@link #shutdown() shutdown}. Each of these should call
29   * {@link #testStateChange(State) testStateChange} before operating, in case the
30   * Transport is in an invalid state for that operation. The exception is
31   * {@link #getWriter() getWriter}, which does not change the Transport state,
32   * and should call {@link #testWriterState() testWriterState} instead.
33   * </p>
34   *
35   * @author jblake
36   * @version $Id: Transport.java,v 1.3 2007/06/01 17:21:31 jblake Exp $
37   */
38  public abstract class Transport {
39  
40      /** The states of in the life-cycle of a Transport. */
41      public static enum State {
42          FILE_CLOSED, FILE_OPEN, SHUTDOWN
43      }
44  
45      /** The parameters that were passed to this Transport for its use. */
46      protected final Map<String, String> parameters;
47  
48      /** If this transport throws an Exception, is it a crucial problem? */
49      protected final boolean crucial;
50  
51      /** The parent can format a file header or footer. */
52      protected final TransportParent parent;
53  
54      /** Initial state is not shutdown, but no file is open, either. */
55      private State currentState = State.FILE_CLOSED;
56  
57      public Transport(Map<String, String> parameters,
58                       boolean crucial,
59                       TransportParent parent)
60              throws JournalException {
61          this.parameters = parameters;
62          this.crucial = crucial;
63          this.parent = parent;
64      }
65  
66      Map<String, String> getParameters() {
67          return parameters;
68      }
69  
70      public boolean isCrucial() {
71          return crucial;
72      }
73  
74      /**
75       * <p>
76       * Subclasses should call this before attempting an operation that will
77       * change the current state, so if the change is not permitted, it will not
78       * be performed.
79       * </p>
80       * <p>
81       * Note that a redundant call to Shutdown is not considered an error.
82       * </p>
83       */
84      public void testStateChange(State desiredState) throws JournalException {
85          if (currentState == State.FILE_CLOSED
86                  && desiredState == State.FILE_CLOSED) {
87              throw new JournalException("Request to close Transport when not open.");
88          }
89          if (currentState == State.FILE_OPEN && desiredState == State.FILE_OPEN) {
90              throw new JournalException("Request to open Transport when already open.");
91          }
92          if (currentState == State.FILE_OPEN && desiredState == State.SHUTDOWN) {
93              throw new JournalException("Request to shutdown Transport with file open.");
94          }
95          if (currentState == State.SHUTDOWN && desiredState == State.FILE_OPEN) {
96              throw new JournalException("Request to open Transport after shutdown.");
97          }
98          if (currentState == State.SHUTDOWN && desiredState == State.FILE_CLOSED) {
99              throw new JournalException("Request to close Transport after shutdown.");
100         }
101     }
102 
103     /**
104      * Subclasses should call this before executing a
105      * {@link #getWriter() getWriter} request. This insures that the Transport
106      * is in the proper state.
107      */
108     protected void testWriterState() throws JournalException {
109         if (currentState == State.FILE_CLOSED) {
110             throw new JournalException("Trying to write to a Transport that is not open.");
111         }
112         if (currentState == State.SHUTDOWN) {
113             throw new JournalException("Trying to write to a Transport after shutdown.");
114         }
115     }
116 
117     /**
118      * Subclasses should call this to change the current state, after the
119      * operation has been performed.
120      */
121     public void setState(State newState) throws JournalException {
122         testStateChange(newState);
123         currentState = newState;
124     }
125 
126     public State getState() {
127         return currentState;
128     }
129 
130     /**
131      * Subclasses should implement this to create a new logical journal file.
132      * Check {@link #testStateChange(State) testStateChange} before performing
133      * the operation, and call
134      * {@link TransportParent#writeDocumentHeader(XMLEventWriter, String, Date)}
135      * to do the formatting.
136      */
137     public abstract void openFile(String repositoryHash,
138                                   String filename,
139                                   Date currentDate) throws JournalException;
140 
141     /**
142      * Subclasses should implement this to provite an {@link XMLEventWriter} for
143      * the JournalWriter to write to. Check {@link #testWriterState()} before
144      * performing the operation.
145      */
146     public abstract XMLEventWriter getWriter() throws JournalException;
147 
148     /**
149      * Subclasses should implement this to close a logical journal file. Check
150      * {@link #testStateChange(State) testStateChange} before performing the
151      * operation, and call
152      * {@link TransportParent#writeDocumentTrailer(XMLEventWriter)} to do the
153      * formatting.
154      */
155     public abstract void closeFile() throws JournalException;
156 
157     /**
158      * Subclasses should implement this to close any resources associated with
159      * the Transport (unless it is already shut down). Check
160      * {@link #testStateChange(State) testStateChange} before performing the
161      * operation.
162      */
163     public abstract void shutdown() throws JournalException;
164 
165 }
