9   import org.fcrepo.server.journal.JournalException;
10  import org.fcrepo.server.journal.ServerInterface;
11  import org.fcrepo.server.journal.recoverylog.JournalRecoveryLog;

14  /**
15   * A JournalReader implementation for "following" a leading server, when the
16   * leading server is using a {@link MultiFileJournalWriter}, or the equivalent.
17   * <p>
18   * The recovery is never complete, as the reader continues to poll for
19   * recently-created files, until the server shuts down.
20   * <p>
21   * This class should likely be superceded by
22   * {@link LockingFollowingJournalReader}.
23   * 
24   * @author Jim Blake
25   */
26  public class MultiFileFollowingJournalReader
27          extends MultiFileJournalReader {
28  
29      private final long pollingIntervalMillis;
30  
31      /**
32       * Do the super-class constructor, and then find the polling interval.
33       */
34      public MultiFileFollowingJournalReader(Map<String, String> parameters,
35                                             String role,
36                                             JournalRecoveryLog recoveryLog,
37                                             ServerInterface server)
38              throws JournalException {
39          super(parameters, role, recoveryLog, server);
40          pollingIntervalMillis =
41                  MultiFileJournalHelper
42                          .parseParametersForPollingInterval(parameters);
43      }
44  
45      /**
46       * Ask for a new file, using the superclass method, but if none is found,
47       * wait for a while and ask again. This will continue until we get a server
48       * shutdown signal.
49       */
50      @Override
51      protected synchronized JournalInputFile openNextFile()
52              throws JournalException {
53          while (open) {
54              JournalInputFile nextFile = super.openNextFile();
55              if (nextFile != null) {
56                  return nextFile;
57              }
58              try {
59                  wait(pollingIntervalMillis);
60              } catch (InterruptedException e) {
61                  // no special action on interrupt.
62              }
63          }
64          return null;
65      }
66  
67      /**
68       * If the server requests a shutdown, stop waiting the next file to come in.
69       */
70      @Override
71      public synchronized void shutdown() throws JournalException {
72          super.shutdown();
73          notifyAll();
74      }
75  }
