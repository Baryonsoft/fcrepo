25  /**
26   * A JournalReader implementation for "recovering", when using a
27   * {@link MultiFileJournalWriter}, or the equivalent.
28   * <p>
29   * The recovery is complete when the all of the files in the journal directory
30   * have been processed and moved to the archive directory.
31   * 
32   * @author Jim Blake
33   */
34  public class MultiFileJournalReader
35          extends JournalReader
36          implements MultiFileJournalConstants {
37  
38      // the directory that holds the journal files before they are processed.
39      private final File journalDirectory;
40  
41      // the directory that will hold the journal files after they are processed.
42      private final File archiveDirectory;
43  
44      // journal file names will start with this.
45      private final String filenamePrefix;
46  
47      protected JournalInputFile currentFile;
48  
49      protected boolean open = true;
50  
51      public MultiFileJournalReader(Map<String, String> parameters,
52                                    String role,
53                                    JournalRecoveryLog recoveryLog,
54                                    ServerInterface server)
55              throws JournalException {
56          super(parameters, role, recoveryLog, server);
57          journalDirectory =
58                  ParameterHelper
59                          .parseParametersForWritableDirectory(parameters,
60                                                               PARAMETER_JOURNAL_DIRECTORY);
61          archiveDirectory =
62                  ParameterHelper
63                          .parseParametersForWritableDirectory(parameters,
64                                                               PARAMETER_ARCHIVE_DIRECTORY);
65          filenamePrefix =
66                  ParameterHelper.parseParametersForFilenamePrefix(parameters);
67          checkDirectoriesAreDifferent();
68      }
69  
70      private void checkDirectoriesAreDifferent() throws JournalException {
71          if (archiveDirectory.equals(journalDirectory)) {
72              throw new JournalException("Archive directory and Journal directory are identical: '"
73                      + archiveDirectory.getPath() + "'");
74          }
75      }

77      /*
78       * Close the current file and set the closed flag.
79       */
80      @Override
81      public synchronized void shutdown() throws JournalException {
82          if (open) {
83              recoveryLog.log("Shutdown requested by server.");
84              closeCurrentFile();
85              open = false;
86          }
87      }

89      /*
90       * Advance to the next tag. If its end of file, close and get next file. If
91       * null, return a null entry.
92       */
93      @Override
94      public synchronized ConsumerJournalEntry readJournalEntry()
95              throws JournalException, XMLStreamException {
96          if (!open) {
97              return null;
98          }
99  
100         scanThroughFilesForNextJournalEntry();
101 
102         if (currentFile == null) {
103             return null;
104         } else {
105             String identifier = peekAtJournalEntryIdentifier(currentFile);
106             ConsumerJournalEntry journalEntry =
107                     super.readJournalEntry(currentFile.getReader());
108             journalEntry.setIdentifier(identifier);
109             return journalEntry;
110         }
111     }

113     /**
114      * Create an identifier string for the Journal Entry, so we can easily
115      * connect the entries in the Recovery Logger with those in the Journal. Call
116      * this before calling
117      * {@link JournalReader#readJournalEntry(XMLEventReader)}, because the
118      * reader is positioned at the beginning of the JournalEntry, so a peek()
119      * will give us the start tag, with the info we need.
120      */
121     private String peekAtJournalEntryIdentifier(JournalInputFile file)
122             throws XMLStreamException {
123         String fileName = file.getFilename();
124         XMLEvent event = file.getReader().peek();
125 
126         String timeString = "unknown";
127         if (event.isStartElement()) {
128             StartElement start = event.asStartElement();
129             Attribute timeStamp =
130                     start.getAttributeByName(QNAME_ATTR_TIMESTAMP);
131             if (timeStamp != null) {
132                 timeString = timeStamp.getValue();
133             }
134         }
135 
136         return "file='" + fileName + "', entry='" + timeString + "'";
137     }

139     /**
140      * Advance to the next journal entry if there is one. If we find one, the
141      * current file will be pointing to it. If we don't find one, there will be
142      * no current file.
143      */
144     private void scanThroughFilesForNextJournalEntry() throws JournalException {
145         try {
146             while (true) {
147                 if (currentFile != null) {
148                     // Check to see whether the current file contains any more
149                     // entries. 
150                     advancePastWhitespace(currentFile.getReader());
151 
152                     XMLEvent next = currentFile.getReader().peek();
153                     if (isStartTagEvent(next, QNAME_TAG_JOURNAL_ENTRY)) {
154                         // found it
155                         return;
156                     } else if (isEndTagEvent(next, QNAME_TAG_JOURNAL)) {
157                         // need to get the next file
158                         closeCurrentFile();
159                     } else {
160                         // problems
161                         throw getNotNextMemberOrEndOfGroupException(QNAME_TAG_JOURNAL,
162                                                                     QNAME_TAG_JOURNAL_ENTRY,
163                                                                     next);
164                     }
165                 }
166 
167                 // If we don't have a file open, try to open one.
168                 if (currentFile == null) {
169                     currentFile = openNextFile();
170                 }
171 
172                 // If we still don't have a file open, we're finished.
173                 if (currentFile == null) {
174                     // if there was no next file, we're done.
175                     return;
176                 }
177 
178                 // A new file needs to be advanced before using.
179                 advanceIntoFile(currentFile.getReader());
180             }
181         } catch (XMLStreamException e) {
182             throw new JournalException(e);
183         }
184     }
185 
186     /**
187      * Look in the directory for files that match the prefix. If there are none,
188      * leave with currentFile still null. If we find one, advance into it.
189      */
190     protected JournalInputFile openNextFile() throws JournalException {
191         File[] journalFiles =
192                 MultiFileJournalHelper
193                         .getSortedArrayOfJournalFiles(journalDirectory,
194                                                       filenamePrefix);
195 
196         if (journalFiles.length == 0) {
197             return null;
198         }
199 
200         JournalInputFile nextFile = new JournalInputFile(journalFiles[0]);
201         recoveryLog.log("Opening journal file: '" + nextFile.getFilename()
202                 + "'");
203         return nextFile;
204     }
205 
206     /**
207      * Advance past the document header to the first JournalEntry.
208      */
209     private void advanceIntoFile(XMLEventReader reader)
210             throws XMLStreamException, JournalException {
211         XMLEvent event = reader.nextEvent();
212         if (!event.isStartDocument()) {
213             throw new JournalException("Expecting XML document header, but event was '"
214                     + event + "'");
215         }
216 
217         event = reader.nextTag();
218         if (!isStartTagEvent(event, QNAME_TAG_JOURNAL)) {
219             throw new JournalException("Expecting FedoraJournal start tag, but event was '"
220                     + event + "'");
221         }
222 
223         String hash =
224                 getOptionalAttributeValue(event.asStartElement(),
225                                           QNAME_ATTR_REPOSITORY_HASH);
226         checkRepositoryHash(hash);
227     }
228 
229     private void closeCurrentFile() throws JournalException {
230         if (currentFile != null) {
231             recoveryLog.log("Closing journal file: '"
232                     + currentFile.getFilename() + "'");
233             currentFile.closeAndRename(archiveDirectory);
234             currentFile = null;
235         }
236     }

238     @Override
239     public String toString() {
240         return super.toString() + ", journalDirectory='" + journalDirectory
241                 + "', archiveDirectory='" + archiveDirectory
242                 + "', filenamePrefix='" + filenamePrefix + "'";
243 
244     }
