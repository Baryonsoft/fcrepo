63                                                                  new Object[] {
64                                                                          parameters,
65                                                                          role,

68              logger.info("JournalRecoveryLog is " + recoveryLog.toString());
69              return (JournalRecoveryLog) recoveryLog;

77      /**
78       * Concrete sub-classes must implement this constructor. Checks the server
79       * parameters to find out what Logging Level to use - default is Low.
80       */
81      protected JournalRecoveryLog(Map<String, String> parameters,
82                                   String role,
83                                   ServerInterface server)
84              throws ModuleInitializationException {
85          this.server = server;
86  
87          String level = parameters.get(PARAMETER_RECOVERY_LOG_LEVEL);
88          if (level == null) {
89              logLevel = LEVEL_LOW;
90          } else if (VALUE_RECOVERY_LOG_LEVEL_HIGH.equals(level)) {
91              logLevel = LEVEL_HIGH;
92          } else if (VALUE_RECOVERY_LOG_LEVEL_MEDIUM.equals(level)) {
93              logLevel = LEVEL_MEDIUM;
94          } else if (VALUE_RECOVERY_LOG_LEVEL_LOW.equals(level)) {
95              logLevel = LEVEL_LOW;
96          } else {
97              throw new ModuleInitializationException("'"
98                      + PARAMETER_RECOVERY_LOG_LEVEL + "' parameter must be '"
99                      + VALUE_RECOVERY_LOG_LEVEL_LOW + "'(default), '"
100                     + VALUE_RECOVERY_LOG_LEVEL_MEDIUM + "' or '"
101                     + VALUE_RECOVERY_LOG_LEVEL_HIGH + "'", role);
102         }
103     }

129     public void logHeaderInfo(Map<String, String> parameters) {
130         StringBuffer buffer = new StringBuffer("Recovery parameters:");
131         for (Iterator<String> keys = parameters.keySet().iterator(); keys
132                 .hasNext();) {
133             Object key = keys.next();
134             Object value = parameters.get(key);
135             buffer.append("\n    ").append(key).append("=").append(value);
136         }
137         log(buffer.toString());
138     }

140     /**
141      * Format a journal entry for writing to the logger. Take logging level into
142      * account.
143      */
144     public void log(ConsumerJournalEntry journalEntry) {
145         StringBuffer buffer = new StringBuffer();
146         buffer.append("Event: method='").append(journalEntry.getMethodName())
147                 .append("', ").append(journalEntry.getIdentifier())
148                 .append("\n");
149         if (logLevel == LEVEL_HIGH) {
150             JournalEntryContext context = journalEntry.getContext();
151             buffer.append("    context=").append(context.getClass().getName())
152                     .append("\n");
153             buffer.append(writeMapValues("environmentAttributes", context
154                     .getEnvironmentAttributes()));
155             buffer.append(writeMapValues("subjectAttributes", context
156                     .getSubjectAttributes()));
157             buffer.append(writeMapValues("actionAttributes", context
158                     .getActionAttributes()));
159             buffer.append(writeMapValues("resourceAttributes", context
160                     .getResourceAttributes()));
161             buffer.append(writeMapValues("recoveryAttributes", context
162                     .getRecoveryAttributes()));
163             buffer.append("        password='*********'\n");
164             buffer.append("        noOp=").append(context.getNoOp())
165                     .append("\n");
166         }
167         if (logLevel == LEVEL_HIGH || logLevel == LEVEL_MEDIUM) {
168             buffer.append("        now=" + journalEntry.getContext().getNoOp()
169                     + "\n");
170             buffer.append("    arguments\n");
171             Map argumentsMap = journalEntry.getArgumentsMap();
172             for (Iterator names = argumentsMap.keySet().iterator(); names
173                     .hasNext();) {
174                 String name = (String) names.next();
175                 Object value = argumentsMap.get(name);
176                 if (value instanceof String[]) {
177                     buffer.append(writeStringArray(name, (String[]) value));
178                 } else {
179                     buffer.append("        " + name + "='" + value + "'\n");
180                 }
181             }
182         }
183         log(buffer.toString());
184     }

186     /**
187      * Helper for the {@link #log(ConsumerJournalEntry)} method. Writes the
188      * values of a string array
189      */
190     private String writeStringArray(String name, String[] values) {
191         StringBuffer buffer = new StringBuffer();
192         buffer.append("        ").append(name).append("=[");
193         for (int i = 0; i < values.length; i++) {
194             buffer.append("'").append(values[i]).append("'");
195             if (i < values.length - 1) {
196                 buffer.append(", ");
197             }
198         }
199         buffer.append("]\n");
200         return buffer.toString();
201     }
202 
203     /**
204      * Helper for the {@link #log(ConsumerJournalEntry)} method. Writes the
205      * values of a context multi-map.
206      */
207     private String writeMapValues(String mapName, MultiValueMap map) {
208         StringBuffer buffer = new StringBuffer();
209         buffer.append("        " + mapName + "\n");
210         for (Iterator names = map.names(); names.hasNext();) {
211             String name = (String) names.next();
212             buffer.append("            ").append(name).append("\n");
213             String[] values = map.getStringArray(name);
214             for (String element : values) {
215                 buffer.append("                ").append(element).append("\n");
216             }
217         }
218         return buffer.toString();
219     }

221     /**
222      * Concrete sub-classes call this method to perform the final formatting if
223      * a log entry.
224      */
225     protected void log(String message, Writer writer) {
226         try {
227             writer.write(JournalHelper.formatDate(new Date()) + ": " + message
228                     + "\n");
229         } catch (IOException e) {
230             logger.error("Error writing journal log entry", e);
231         }
232     }
