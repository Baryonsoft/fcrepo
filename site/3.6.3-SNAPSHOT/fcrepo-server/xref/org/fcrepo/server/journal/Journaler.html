54      /**
55       * Augment the parameters with values obtained from System Properties, and
56       * create the proper worker (JournalCreator or JournalConsumer) for the
57       * current mode.
58       */
59      @Override
60      public void initModule() throws ModuleInitializationException {
61          Map<String, String> parameters = getParameters();
62          copyPropertiesOverParameters(parameters);
63          serverInterface = new ServerWrapper(getServer());
64          logger.info("Journaling parameters: " + parameters);
65          parseParameters(parameters);
66  
67          if (inRecoveryMode) {
68              worker =
69                      new JournalConsumer(parameters, getRole(), serverInterface);
70          } else {
71              worker = new JournalCreator(parameters, getRole(), serverInterface);
72          }
73          logger.info("Journal worker module is: " + worker.toString());
74      }
75  
76      /**
77       * Get the ManagementDelegate module and pass it to the worker.
78       */
79      @Override
80      public void postInitModule() throws ModuleInitializationException {
81          ManagementDelegate delegate = serverInterface.getManagementDelegate();
82          if (delegate == null) {
83              throw new ModuleInitializationException("Can't get a ManagementDelegate from Server.getModule()",
84                                                      getRole());
85          }
86          worker.setManagementDelegate(delegate);
87      }
88  
89      /**
90       * Tell the worker to shut down.
91       */
92      @Override
93      public void shutdownModule() throws ModuleShutdownException {
94          worker.shutdown();
95      }

97      /**
98       * Augment, and perhaps override, the server parameters, using any System
99       * Property whose name begins with "fedora.journal.". So, for example, a
100      * System Property of "fedora.journal.mode" will override a server parameter
101      * of "mode".
102      */
103     private void copyPropertiesOverParameters(Map<String, String> parameters) {
104         Properties properties = System.getProperties();
105         for (Object object : properties.keySet()) {
106             String key = (String) object;
107             if (key.startsWith(SYSTEM_PROPERTY_PREFIX)) {
108                 parameters.put(key.substring(SYSTEM_PROPERTY_PREFIX.length()),
109                                properties.getProperty(key));
110             }
111         }
112     }
113 
114     /**
115      * Check the parameters for required values and for acceptable values. At
116      * this point, the only parameter we care about is "mode".
117      */
118     private void parseParameters(Map<String, String> parameters)
119             throws ModuleInitializationException {
120         logger.info("Parameters: " + parameters);
121 
122         String mode = parameters.get(PARAMETER_JOURNAL_MODE);
123         if (mode == null) {
124             inRecoveryMode = false;
125         } else if (mode.equals(VALUE_JOURNAL_MODE_NORMAL)) {
126             inRecoveryMode = false;
127         } else if (mode.equals(VALUE_JOURNAL_MODE_RECOVER)) {
128             inRecoveryMode = true;
129         } else {
130             throw new ModuleInitializationException("'"
131                                                     + PARAMETER_JOURNAL_MODE + "' parameter must be '"
132                                                     + VALUE_JOURNAL_MODE_NORMAL + "'(default) or '"
133                                                     + VALUE_JOURNAL_MODE_RECOVER + "'", getRole());
134         }
135     }
