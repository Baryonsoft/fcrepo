7   import org.fcrepo.server.errors.ServerException;
8   import org.fcrepo.server.journal.JournalConstants;
9   import org.fcrepo.server.journal.JournalException;
10  import org.fcrepo.server.journal.entry.JournalEntry;
11  import org.fcrepo.server.management.ManagementDelegate;

14  /**
15   * <p>
16   * Abstract base class for the classes that act as adapters to the Management
17   * methods.
18   * </p>
19   * <p>
20   * An adapter is needed for each method that modifies the contents of the
21   * repository in any way. If a method is read-only, it will not be written to
22   * the journal, and so doesn't require an adapter class.
23   * </p>
24   * 
25   * @author Jim Blake
26   */
27  public abstract class ManagementMethod
28          implements JournalConstants {
29  
30      /**
31       * Get an instance of the proper class, based on the method name.
32       */
33      public static ManagementMethod getInstance(String methodName,
34                                                 JournalEntry parent) {
35          if (METHOD_INGEST.equals(methodName)) {
36              return new IngestMethod(parent);
37          } else if (METHOD_MODIFY_OBJECT.equals(methodName)) {
38              return new ModifyObjectMethod(parent);
39          } else if (METHOD_PURGE_OBJECT.equals(methodName)) {
40              return new PurgeObjectMethod(parent);
41          } else if (METHOD_ADD_DATASTREAM.equals(methodName)) {
42              return new AddDatastreamMethod(parent);
43          } else if (METHOD_MODIFY_DATASTREAM_BY_REFERENCE.equals(methodName)) {
44              return new ModifyDatastreamByReferenceMethod(parent);
45          } else if (METHOD_MODIFY_DATASTREAM_BY_VALUE.equals(methodName)) {
46              return new ModifyDatastreamByValueMethod(parent);
47          } else if (METHOD_SET_DATASTREAM_STATE.equals(methodName)) {
48              return new SetDatastreamStateMethod(parent);
49          } else if (METHOD_SET_DATASTREAM_VERSIONABLE.equals(methodName)) {
50              return new SetDatastreamVersionableMethod(parent);
51          } else if (METHOD_PURGE_DATASTREAM.equals(methodName)) {
52              return new PurgeDatastreamMethod(parent);
53          } else if (METHOD_PURGE_RELATIONSHIP.equals(methodName)) {
54              return new PurgeRelationshipMethod(parent);
55          } else if (METHOD_PUT_TEMP_STREAM.equals(methodName)) {
56              return new PutTempStreamMethod(parent);
57          } else if (METHOD_GET_NEXT_PID.equals(methodName)) {
58              return new GetNextPidMethod(parent);
59          } else if (METHOD_ADD_RELATIONSHIP.equals(methodName)) {
60              return new AddRelationshipMethod(parent);
61          } else {
62              throw new IllegalArgumentException("Unrecognized method name: '"
63                      + methodName + "'");
64          }
65      }
66  
67      protected final JournalEntry parent;
68  
69      protected ManagementMethod(JournalEntry parent) {
70          this.parent = parent;
71      }
72  
73      /**
74       * Each concrete sub-class should use this method to pull the necessary
75       * arguments from the map of the parent JournalEntry, call the appropriate
76       * method on the ManagementDelegate, and perhaps store the result in the
77       * context of the parent JournalEntry (depends on the sub-class).
78       */
79      public abstract Object invoke(ManagementDelegate delegate)
80              throws ServerException, JournalException;
81  }
