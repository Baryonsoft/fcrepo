5   package org.fcrepo.server.journal.helpers;

7   import org.fcrepo.utilities.Base64;

9   /**
10   * Encipher the password so we aren't writing it to the file in clear.
11   * <p>
12   * The encipher method does not allow options for cipher type. It should always
13   * use the latest and greatest. The decipher method accepts different cipher
14   * types so it can be used to read older Journal Files. Known types include:
15   * <ul>
16   * <li>"" or null - empty cipher, clear text is the same as cipher text</li>
17   * <li>"1" - rotating key XOR cipher</li>
18   * </ul>
19   *
20   * @author Jim Blake
21   */
22  public class PasswordCipher {
23  
24      /**
25       * Two methods: String encode(String key, String clearText); String
26       * decode(String key, String decodedText); encoding: for each character in
27       * the clearText, get the byte from the char, get a byte from the key, XOR,
28       * and write to the byte array. Base64 encode and return. decoding: base64
29       * decode. for each byte in the result, XOR with a byte from the key to
30       * yield a char. combine into a string. return.
31       */
32  
33      /**
34       * Use the key to produce a ciphered String from the clearText.
35       */
36      public static String encipher(String key, String clearText) {
37          if (key == null) {
38              throw new NullPointerException("key may not be null");
39          }
40  
41          if (key == "") {
42              return clearText;
43          }
44  
45          if (clearText == null) {
46              return null;
47          }
48  
49          byte[] keyBytes = convertKeyToByteArray(key);
50          byte[] clearTextBytes = convertClearTextToByteArray(clearText);
51          byte[] cipherBytes = applyCipher(keyBytes, clearTextBytes);
52          return Base64.encodeToString(cipherBytes);
53      }
54  
55      /**
56       * Use the key to produce a clear text String from the cipherText. If no key
57       * is provided, or if no type is specified, just return the text as is.
58       */
59      public static String decipher(String key,
60                                    String cipherText,
61                                    String cipherType) {
62          if (key == null || key == "") {
63              return cipherText;
64          }
65  
66          if (cipherText == null) {
67              return null;
68          }
69  
70          if (cipherType == null || cipherType == "") {
71              return cipherText;
72          } else if ("1".equalsIgnoreCase(cipherType)) {
73              byte[] keyBytes = convertKeyToByteArray(key);
74              byte[] cipherBytes = Base64.decode(cipherText);
75              sanityCheckOnCipherBytes(cipherText, cipherBytes);
76              byte[] clearTextBytes = applyCipher(keyBytes, cipherBytes);
77              return convertByteArrayToClearText(clearTextBytes);
78          } else {
79              throw new IllegalArgumentException("Unrecognized cipher type: '"
80                      + cipherType + "'");
81          }
82      }
83  
84      /**
85       * Convert the key to a byte array by compressing the 16-bit characters to
86       * bytes. This XOR compression insures that the top 8 bits are still
87       * significant, so a key of "\u0020" yields a different cipher than a key of
88       * "\u0120".
89       */
90      private static byte[] convertKeyToByteArray(String key) {
91          byte[] result = new byte[key.length()];
92          for (int i = 0; i < result.length; i++) {
93              char thisChar = key.charAt(i);
94              result[i] = (byte) (thisChar >>> 8 & 0xFF ^ thisChar & 0xFF);
95          }
96          return result;
97      }
98  
99      /**
100      * Convert the clear text to a byte array by splitting each 16-bit character
101      * into 2 bytes. This insures that no data is lost.
102      */
103     private static byte[] convertClearTextToByteArray(String clearText) {
104         byte[] result = new byte[clearText.length() * 2];
105         for (int i = 0; i < clearText.length(); i++) {
106             char thisChar = clearText.charAt(i);
107             int pos = i * 2;
108             result[pos] = (byte) (thisChar >>> 8 & 0xFF);
109             result[pos + 1] = (byte) (thisChar & 0xFF);
110         }
111         return result;
112     }
113 
114     /**
115      * If the cipher text decodes to an odd number of bytes, we can't go on!
116      */
117     private static void sanityCheckOnCipherBytes(String cipherText,
118                                                  byte[] cipherBytes) {
119         if (cipherBytes.length % 2 != 0) {
120             throw new IllegalStateException("Ciphered text decodes to an odd number of bytes! Text='"
121                     + cipherText
122                     + "', decodes to "
123                     + cipherBytes.length
124                     + " bytes.");
125         }
126     }
127 
128     /**
129      * Convert a byte array to text by joining each two bytes into a 16-bit
130      * character.
131      */
132     private static String convertByteArrayToClearText(byte[] bytes) {
133         StringBuffer result = new StringBuffer();
134         for (int i = 0; i < bytes.length; i += 2) {
135             char thisChar = (char) (bytes[i] << 8 | bytes[i + 1]);
136             result.append(thisChar);
137         }
138         return result.toString();
139     }
140 
141     /**
142      * The same algorithm applies for enciphering or deciphering. Go through the
143      * text, XORing with successive bytes of the key. If you apply it twice, you
144      * get the original text back.
145      */
146     private static byte[] applyCipher(byte[] keyBytes, byte[] textBytes) {
147         byte[] result = new byte[textBytes.length];
148         for (int i = 0; i < result.length; i++) {
149             int keyPos = i % keyBytes.length;
150             result[i] = (byte) (textBytes[i] ^ keyBytes[keyPos]);
151         }
152         return result;
153     }
