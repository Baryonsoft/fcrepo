5   package org.fcrepo.server.journal.xmlhelpers;

7   import javax.xml.namespace.QName;
8   import javax.xml.stream.XMLEventReader;
9   import javax.xml.stream.XMLStreamException;
10  import javax.xml.stream.events.Attribute;
11  import javax.xml.stream.events.StartElement;
12  import javax.xml.stream.events.XMLEvent;
13  
14  import org.fcrepo.server.journal.JournalConstants;
15  import org.fcrepo.server.journal.JournalException;
16  
17  
18  /**
19   * An abstract base class that provides some useful methods for the XML reader
20   * classes.
21   * 
22   * @author Jim Blake
23   */
24  public abstract class AbstractXmlReader
25          implements JournalConstants {
26  
27      /**
28       * Advance past any white space. Leave the reader positioned before the next
29       * tag (or non-white-space event).
30       */
31      protected void advancePastWhitespace(XMLEventReader reader)
32              throws XMLStreamException {
33          XMLEvent next = reader.peek();
34          while (next.isCharacters() && next.asCharacters().isWhiteSpace()) {
35              reader.nextEvent();
36              next = reader.peek();
37          }
38      }
39  
40      /**
41       * Read the next event and complain if it is not the Start Tag that we
42       * expected.
43       */
44      protected XMLEvent readStartTag(XMLEventReader reader, QName tagName)
45              throws XMLStreamException, JournalException {
46          XMLEvent event = reader.nextTag();
47          if (!isStartTagEvent(event, tagName)) {
48              throw getNotStartTagException(tagName, event);
49          }
50          return event;
51      }
52  
53      /**
54       * Test an event to see whether it is an start tag with the expected name.
55       */
56      protected boolean isStartTagEvent(XMLEvent event, QName tagName) {
57          return event.isStartElement()
58                  && event.asStartElement().getName().equals(tagName);
59      }
60  
61      /**
62       * Test an event to see whether it is an end tag with the expected name.
63       */
64      protected boolean isEndTagEvent(XMLEvent event, QName tagName) {
65          return event.isEndElement()
66                  && event.asEndElement().getName().equals(tagName);
67      }
68  
69      /**
70       * Get the value of a given attribute from this start tag, or complain if
71       * it's not there.
72       */
73      protected String getRequiredAttributeValue(StartElement start,
74                                                 QName attributeName)
75              throws JournalException {
76          Attribute mapNameAttribute = start.getAttributeByName(attributeName);
77          if (mapNameAttribute == null) {
78              throw new JournalException("Start tag '" + start
79                      + "' must contain a '" + attributeName + "' attribute.");
80          }
81          return mapNameAttribute.getValue();
82      }
83  
84      /**
85       * Get the value of a given attribute from this start tag, or null if the
86       * attribute is not there.
87       */
88      protected String getOptionalAttributeValue(StartElement start,
89                                                 QName attributeName) {
90          Attribute mapNameAttribute = start.getAttributeByName(attributeName);
91          if (mapNameAttribute == null) {
92              return null;
93          } else {
94              return mapNameAttribute.getValue();
95          }
96      }
97  
98      /**
99       * Loop through a series of character events, accumulating the data into a
100      * String. The character events should be terminated by an EndTagEvent with
101      * the expected tag name.
102      */
103     protected String readCharactersUntilEndTag(XMLEventReader reader,
104                                                QName tagName)
105             throws XMLStreamException, JournalException {
106         StringBuffer stringValue = new StringBuffer();
107         while (true) {
108             XMLEvent event = reader.nextEvent();
109             if (event.isCharacters()) {
110                 stringValue.append(event.asCharacters().getData());
111             } else if (isEndTagEvent(event, tagName)) {
112                 break;
113             } else {
114                 throw getNotCharactersException(tagName, event);
115             }
116         }
117         return stringValue.toString();
118     }
119 
120     /**
121      * Complain when we were expecting a start tag, and didn't find it.
122      */
123     protected JournalException getNotStartTagException(QName tagName,
124                                                        XMLEvent event) {
125         return new JournalException("Expecting '" + tagName
126                 + "' start tag, but event was '" + event + "'");
127 
128     }
129 
130     /**
131      * Complain when we were expecting a end tag, and didn't find it.
132      */
133     protected JournalException getNotEndTagException(QName tagName,
134                                                      XMLEvent event) {
135         return new JournalException("Expecting '" + tagName
136                 + "' end tag, but event was '" + event + "'");
137 
138     }
139 
140     /**
141      * If we encounter an unexpected event when reading the journal file, create
142      * an exception with all of the pertinent information.
143      */
144     protected JournalException getNotCharactersException(QName tagName,
145                                                          XMLEvent event) {
146         return new JournalException("Expecting characters or '" + tagName
147                 + "' end tag, but event was '" + event + "'");
148     }
149 
150     /**
151      * While traversing a group of member tags, we expected either the start of
152      * another member tag, or the end of the group.
153      */
154     protected JournalException getNotNextMemberOrEndOfGroupException(QName groupTagName,
155                                                                      QName memberTagName,
156                                                                      XMLEvent event) {
157         return new JournalException("Expecting either '" + memberTagName
158                 + "' start tag, or '" + groupTagName
159                 + "' end tag, but event was '" + event + "'");
160     }
