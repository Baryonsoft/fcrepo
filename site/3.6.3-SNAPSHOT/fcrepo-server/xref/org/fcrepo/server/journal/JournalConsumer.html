24  /**
25   * The JournalWorker class to use in recovery mode or in* following mode.
26   * <p>
27   * Create a <code>JournalConsumerThread</code> to process the journal. If any
28   * calls to Management methods come in from outside, reject them.
29   * </p>
30   *
31   * @author Jim Blake
32   */
33  public class JournalConsumer
34          implements JournalWorker {
35  
36      private final String role;
37  
38      private final JournalConsumerThread consumerThread;
39  
40      private final JournalReader reader;
41  
42      private final JournalRecoveryLog recoveryLog;
43  
44      private ManagementDelegate delegate;
45  
46      /**
47       * Get the appropriate JournalReader and JournalRecoveryLog, based on the
48       * server parameters, and create a JournalConsumerThread that will process
49       * the journal entries, using that reader and that logger.
50       */
51      public JournalConsumer(Map<String, String> parameters,
52                             String role,
53                             ServerInterface server)
54              throws ModuleInitializationException {
55          this.role = role;
56          recoveryLog = JournalRecoveryLog.getInstance(parameters, role, server);
57          reader =
58                  JournalReader
59                          .getInstance(parameters, role, recoveryLog, server);
60          consumerThread =
61                  new JournalConsumerThread(parameters,
62                                            role,
63                                            server,
64                                            reader,
65                                            recoveryLog);
66      }
67  
68      /**
69       * Get the ManagementDelegate module and pass it to the
70       * JournalConsumerThread, so it can start working.
71       */
72      public void setManagementDelegate(ManagementDelegate delegate) {
73          this.delegate = delegate;
74          consumerThread.setManagementDelegate(delegate);
75      }
76  
77      /**
78       * Tell the thread, the reader and the log to shut down.
79       */
80      public void shutdown() throws ModuleShutdownException {
81          try {
82              consumerThread.shutdown();
83              reader.shutdown();
84              recoveryLog.shutdown("Server is shutting down.");
85          } catch (JournalException e) {
86              throw new ModuleShutdownException("Error closing journal reader.",
87                                                role,
88                                                e);
89          }
90      }
