38  /**
39   * SYNCHRONIZATION NOTE: All public methods are synchronized against
40   * {@link JournalWriter.SYNCHRONIZER}, as is the {@link #closeFile() closeFile}
41   * method. This means that an asynchronous call by the timer task will not
42   * interrupt a synchronous operation already in progress, or vice versa.

44   * @author jblake

104     public MulticastJournalWriter(Map<String, String> parameters,
105                                   String role,
106                                   ServerInterface server)
107             throws JournalException {

122     /**
123      * Create a Map of Maps, holding parameters for all of the transports.
124      *
125      * @throws JournalException
126      */
127     private Map<String, Map<String, String>> parseTransportParameters(Map<String, String> parameters)
128             throws JournalException {
129         Map<String, Map<String, String>> allTransports =
130                 new LinkedHashMap<String, Map<String, String>>();
131         for (String key : parameters.keySet()) {
132             if (isTransportParameter(key)) {
133                 Map<String, String> thisTransport =
134                         getThisTransportMap(allTransports,
135                                             getTransportName(key));
136                 thisTransport.put(getTransportParameterName(key), parameters
137                         .get(key));
138             }
139         }
140         return allTransports;
141     }

143     private boolean isTransportParameter(String key) throws JournalException {
144         return key.startsWith(TRANSPORT_PARAMETER_PREFIX);
145     }

147     private int findParameterNameSeparator(String key) throws JournalException {
148         int dotHere = key.indexOf('.', TRANSPORT_PARAMETER_PREFIX.length());
149         if (dotHere < 0) {
150             throw new JournalException("Invalid name for transport parameter '"
151                     + key + "' - requires '.' after transport name.");
152         }
153         return dotHere;
154     }
155 
156     private String getTransportParameterName(String key)
157             throws JournalException {
158         return key.substring(findParameterNameSeparator(key) + 1);
159     }
160 
161     private String getTransportName(String key) throws JournalException {
162         return key.substring(TRANSPORT_PARAMETER_PREFIX.length(),
163                              findParameterNameSeparator(key));
164     }

166     /** If we don't yet have a map for this transport name, create one. */
167     private Map<String, String> getThisTransportMap(Map<String, Map<String, String>> allTransports,
168                                                     String transportName) {
169         if (!allTransports.containsKey(transportName)) {
170             allTransports.put(transportName, new HashMap<String, String>());
171         }
172         return allTransports.get(transportName);
173     }

176     protected void checkTransportParametersForValidity()
177             throws JournalException {
178         checkAtLeastOneTransport();
179         checkAllTransportsHaveClassnames();
180         checkAllTransportsHaveCrucialFlags();
181         checkAtLeastOneCrucialTransport();
182         logger.info("Journal transport parameters validated.");

185     private void checkAtLeastOneTransport() throws JournalException {
186         if (transportParameters.size() == 0) {
187             throw new JournalException("MulticastJournalWriter must have "
188                     + "at least one Transport.");
189         }
190     }

192     private void checkAllTransportsHaveClassnames() throws JournalException {
193         for (String transportName : transportParameters.keySet()) {
194             Map<String, String> thisTransportMap =
195                     transportParameters.get(transportName);
196             if (!thisTransportMap.containsKey(CLASSNAME_PARAMETER_KEY)) {
197                 throw new JournalException("Transport '" + transportName
198                         + "' does not have a '" + CLASSNAME_PARAMETER_KEY
199                         + "' parameter");
200             }
201         }
202     }
203 
204     private void checkAllTransportsHaveCrucialFlags() throws JournalException {
205         for (String transportName : transportParameters.keySet()) {
206             Map<String, String> thisTransportMap =
207                     transportParameters.get(transportName);
208             if (!thisTransportMap.containsKey(CRUCIAL_PARAMETER_KEY)) {
209                 throw new JournalException("Transport '" + transportName
210                         + "' does not have a '" + CRUCIAL_PARAMETER_KEY
211                         + "' parameter");
212             }
213         }
214     }
215 
216     private void checkAtLeastOneCrucialTransport() throws JournalException {
217         for (String transportName : transportParameters.keySet()) {
218             Map<String, String> thisTransportMap =
219                     transportParameters.get(transportName);
220             String crucialString = thisTransportMap.get(CRUCIAL_PARAMETER_KEY);
221             if (Boolean.parseBoolean(crucialString)) {
222                 return;
223             }
224         }
225         throw new JournalException("There must be at least one crucial transport.");
226     }

228     private Map<String, Transport> createTransports() throws JournalException {
229         Map<String, Transport> result = new HashMap<String, Transport>();
230         for (String transportName : transportParameters.keySet()) {
231             Map<String, String> thisTransportMap =
232                     transportParameters.get(transportName);
233             String className = thisTransportMap.get(CLASSNAME_PARAMETER_KEY);
234             boolean crucialFlag =
235                     Boolean.parseBoolean(thisTransportMap
236                             .get(CRUCIAL_PARAMETER_KEY));
237 
238             Object transport =
239                     JournalHelper
240                             .createInstanceFromClassname(className,
241                                                          TRANSPORT_CONSTRUCTOR_ARGUMENT_TYPES,
242                                                          new Object[] {
243                                                                  thisTransportMap,
244                                                                  crucialFlag,
245                                                                  this});
246             logger.info("Transport '" + transportName + "' is " + transport);
247             result.put(transportName, (Transport) transport);
248         }
249         return result;
250     }

256     /**
257      * <p>
258      * Get ready to write a journal entry, insuring that we have an open file.
259      * </p>
260      * <p>
261      * If we are shutdown, ignore this request. Otherwise, check if we need to
262      * shut a file down based on size limit. Then check to see whether we need
263      * to open another file. If so, we'll need a repository hash and a filename.
264      * </p>
265      *
266      * @see org.fcrepo.server.journal.JournalWriter#prepareToWriteJournalEntry()
267      */
268     @Override
269     public void prepareToWriteJournalEntry() throws JournalException {
270         synchronized (JournalWriter.SYNCHRONIZER) {
271             if (state == SHUTDOWN) {
272                 return;
273             }
274 
275             logger.debug("Preparing to write journal entry.");
276 
277             if (state == FILE_OPEN) {
278                 closeFileIfAppropriate();
279             }
280 
281             if (state == FILE_CLOSED) {
282                 openNewFile();
283             }
284         }
285     }

287     /**
288      * <p>
289      * Write a journal entry.
290      * </p>
291      * <p>
292      * If we are shutdown, ignore this request. Otherwise, get an output stream
293      * from each Transport in turn, and write the entry. If this puts the file
294      * size over the limit, close them.
295      * </p>
296      *
297      * @see org.fcrepo.server.journal.JournalWriter#writeJournalEntry(org.fcrepo.server.journal.entry.CreatorJournalEntry)
298      */
299     @Override
300     public void writeJournalEntry(CreatorJournalEntry journalEntry)
301             throws JournalException {
302         synchronized (JournalWriter.SYNCHRONIZER) {
303             if (state == SHUTDOWN) {
304                 return;
305             }
306             logger.debug("Writing journal entry.");
307             sendRequestToAllTransports(new WriteEntryRequest(this, journalEntry));
308             currentSize += sizeEstimator.estimateSize(journalEntry);
309 
310             if (state == FILE_OPEN) {
311                 closeFileIfAppropriate();
312             }
313         }
314     }

316     /**
317      * <p>
318      * Shut it down
319      * </p>
320      * <p>
321      * If the Transports still have files open, close them. Then stop responding
322      * to requests.
323      * </p>
324      *
325      * @see org.fcrepo.server.journal.JournalWriter#shutdown()
326      */
327     @Override
328     public void shutdown() throws JournalException {
329         synchronized (JournalWriter.SYNCHRONIZER) {
330             if (state == SHUTDOWN) {
331                 return;
332             }
333             if (state == FILE_OPEN) {
334                 closeFile();
335             }
336 
337             logger.debug("Shutting down.");
338             sendRequestToAllTransports(new ShutdownRequest());
339             state = SHUTDOWN;
340         }
341     }

343     private void openNewFile() throws JournalException {
344         try {
345             String hash = server.getRepositoryHash();
346             String filename =
347                     JournalHelper.createTimestampedFilename(filenamePrefix,
348                                                             getCurrentDate());
349             timer = createTimer();
350             sendRequestToAllTransports(new OpenFileRequest(hash,
351                                                            filename,
352                                                            getCurrentDate()));
353             currentSize = 0;
354             state = FILE_OPEN;
355         } catch (ServerException e) {
356             throw new JournalException(e);
357         }
358     }

365     /**
366      * Create the timer, and schedule a task that will let us know when the file
367      * is too old to continue. If the age limit is 0 or negative, we treat it as
368      * "no limit".
369      */
370     private Timer createTimer() {
371         Timer fileTimer = new Timer();
372 
373         // if the age limit is 0 or negative, treat it as "no limit".
374         if (ageLimit >= 0) {
375             fileTimer.schedule(new CloseFileTimerTask(), ageLimit);
376         }
377 
378         return fileTimer;
379     }

381     /**
382      * When the timer goes off, close the file.
383      */
384     private final class CloseFileTimerTask
385             extends TimerTask {
386 
387         @Override
388         public void run() {
389             try {
390                 logger.debug("Timer task requests file close.");
391                 closeFile();
392             } catch (JournalException e) {
393                 /*
394                  * What to do with this exception? If we print it, where is the
395                  * console? If we throw it, who will catch it?
396                  */
397                 e.printStackTrace();
398                 throw new IllegalStateException(e);
399             }
400         }
401     }

403     /**
404      * Check to see whether the file size has passed the limit.
405      */
406     private void closeFileIfAppropriate() throws JournalException {
407         if (sizeLimit != 0 && currentSize >= sizeLimit) {
408             closeFile();
409         }
410     }

412     /**
413      * Close the file unconditionally. Called if
414      * <ul>
415      * <li>the file passes the size limit,</li>
416      * <li>the timer expires,</li>
417      * <li>the server commands a shutdown</li>
418      * </ul>
419      * Synchronized so a close request from the timer doesn't conflict with
420      * other processing.
421      */
422     private void closeFile() throws JournalException {
423         synchronized (JournalWriter.SYNCHRONIZER) {
424             // check to be sure that another thread didn't close the file while
425             // we were waiting for the lock.
426             if (state == FILE_OPEN) {
427                 sendRequestToAllTransports(new CloseFileRequest());
428                 currentSize = 0;
429                 state = FILE_CLOSED;
430             }
431 
432             // turn off the timer that is checking the age of this file.
433             if (timer != null) {
434                 timer.cancel();
435             }
436         }
437     }

468     /**
469      * Send a request for some operation to the Transports. Send it to all of
470      * them, even if one or more throws an Exception. Report any exceptions when
471      * all Transports have been attempted.
472      *
473      * @param request
474      *        the request object
475      * @param args
476      *        the arguments to be passed to the request object
477      * @throws JournalException
478      *         if there were any crucial problems.
479      */
480     private void sendRequestToAllTransports(TransportRequest request)
481             throws JournalException {
482         Map<String, JournalException> crucialExceptions =
483                 new LinkedHashMap<String, JournalException>();
484         Map<String, JournalException> nonCrucialExceptions =
485                 new LinkedHashMap<String, JournalException>();
486 
487         /*
488          * Send the request to all transports, accumulating any Exceptions as we
489          * go. That way, we increase the likeihood that at least one Transport
490          * succeeded in the request.
491          */
492         for (String transportName : transports.keySet()) {
493             Transport transport = transports.get(transportName);
494             try {
495                 logger.debug("Sending " + request.getClass().getSimpleName()
496                         + " to transport '" + transportName + "'");
497                 request.performRequest(transport);
498             } catch (JournalException e) {
499                 if (transport.isCrucial()) {
500                     crucialExceptions.put(transportName, e);
501                 } else {
502                     nonCrucialExceptions.put(transportName, e);
503                 }
504             }
505         }
506 
507         /*
508          * Report the Exceptions. Report the non-crucial ones first, in case the
509          * Server decides to take some definitive action on a crucial Exception.
510          */
511         reportNonCrucialExceptions(nonCrucialExceptions);
512         reportCrucialExceptions(crucialExceptions);
513     }
514 
515     private void reportNonCrucialExceptions(Map<String, JournalException> nonCrucialExceptions) {
516         if (nonCrucialExceptions.isEmpty()) {
517             return;
518         }
519         for (String transportName : nonCrucialExceptions.keySet()) {
520             JournalException e = nonCrucialExceptions.get(transportName);
521             logger.error("Exception thrown from non-crucial Journal Transport: '"
522                     + transportName + "'", e);
523         }
524     }
525 
526     private void reportCrucialExceptions(Map<String, JournalException> crucialExceptions)
527             throws JournalException {
528         if (!crucialExceptions.isEmpty()) {
529             JournalOperatingMode.setMode(JournalOperatingMode.READ_ONLY);
530         }
531         for (String transportName : crucialExceptions.keySet()) {
532             JournalException e = crucialExceptions.get(transportName);
533             logger.error("Exception thrown from crucial Journal Transport: '"
534                     + transportName + "'", e);
535         }
536     }
