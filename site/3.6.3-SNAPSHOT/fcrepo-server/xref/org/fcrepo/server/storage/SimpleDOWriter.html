46  /**
47   * A DigitalObject-backed DOWriter.
48   * <p>
49   * This interface supports transaction behavior with the commit(String) and
50   * rollBack() methods. When a DOWriter is instantiated, there is an implicit
51   * transaction. Write methods may be called, but they won't affect the the
52   * underlying data store until commit(String) is invoked. This also has the
53   * effect of creating another implicit transaction. If temporary changes are no
54   * longer wanted, rollBack() may be called to return the object to it's original
55   * form. rollBack() is only valid for the current transaction.
56   * </p>
57   * <p>
58   * The read methods of DOWriter reflect on the composition of the object in the
59   * context of the current transaction.
60   * </p>
61   *
62   * @author Chris Wilper
63   */
64  public class SimpleDOWriter
65          extends SimpleDOReader
66          implements Constants, DOWriter {

175     /**
176      * Removes a datastream from the object.
177      *
178      * @param id
179      *        The id of the datastream.
180      * @param start
181      *        The start date (inclusive) of versions to remove. If
182      *        <code>null</code>, this is taken to be the smallest possible
183      *        value.
184      * @param end
185      *        The end date (inclusive) of versions to remove. If
186      *        <code>null</code>, this is taken to be the greatest possible
187      *        value.
188      * @throws ServerException
189      *         If any type of error occurred fulfilling the request.
190      */
191     public Date[] removeDatastream(String id, Date start, Date end)
192             throws ServerException {
193         assertNotInvalidated();
194         assertNotPendingRemoval();
195         ArrayList<Datastream> removeList = new ArrayList<Datastream>();
196         for (Datastream ds : m_obj.datastreams(id)) {
197             boolean doRemove = false;
198             if (start != null) {
199                 if (end != null) {
200                     if (ds.DSCreateDT.compareTo(start) >= 0
201                             && ds.DSCreateDT.compareTo(end) <= 0) {
202                         doRemove = true;
203                     }
204                 } else {
205                     if (ds.DSCreateDT.compareTo(start) >= 0) {
206                         doRemove = true;
207                     }
208                 }
209             } else {
210                 if (end != null) {
211                     if (ds.DSCreateDT.compareTo(end) <= 0) {
212                         doRemove = true;
213                     }
214                 } else {
215                     doRemove = true;
216                 }
217             }
218             if (doRemove) {
219                 // Note: We don't remove old audit records by design.
220 
221                 // add this datastream to the datastream to-be-removed list.
222                 removeList.add(ds);
223             }
224         }
225 
226         /* Now that we've identified all ds versions to remove, remove 'em */
227         for (Datastream toRemove : removeList) {
228             m_obj.removeDatastreamVersion(toRemove);
229         }
230 
231         // finally, return the dates of each deleted item
232         Date[] deletedDates = new Date[removeList.size()];
233         for (int i = 0; i < removeList.size(); i++) {
234             deletedDates[i] = (removeList.get(i)).DSCreateDT;
235         }
236         return deletedDates;
237     }

239     // from the relationship subject, determine which datastream to modify etc
240     private String resolveSubjectToDatastream(String subject) throws ServerException{
241         String dsId = null;
242         String pidURI = PID.toURI(m_obj.getPid());
243         if (subject.equals(pidURI)) {
244             dsId = "RELS-EXT";
245         } else {
246             if (subject.startsWith(pidURI + "/")) {
247                 dsId = "RELS-INT";
248             } else {
249                 throw new GeneralException("Cannot determine which relationship datastream to update for subject " + subject + ".  Relationship subjects must be the URI of the object or the URI of a datastream within the object.");
250             }
251         }
252         return dsId;
253

270     public boolean addRelationship(String dsId,
271                                    String subject,
272                                    String relationship,
273                                    String object,
274                                    boolean isLiteral,
275                                    String datatype) throws ServerException {
276 
277         Triple toAdd =
278                 createTriple(subject, relationship, object, isLiteral, datatype);
279         Datastream relsDatastream = GetDatastream(dsId, null);
280         XMLDatastreamProcessor dsxml = null;
281         if (relsDatastream == null) {
282             ByteArrayOutputStream out = new ByteArrayOutputStream();
283             Map<String, String> map = new HashMap<String, String>();
284             // namespaces for RELS-EXT
285             if (dsId.equals("RELS-EXT")) {
286                 map.put(RELS_EXT.prefix, RELS_EXT.uri);
287                 map.put(MODEL.prefix, MODEL.uri);
288             }
289             map.put(RDF.prefix, RDF.uri);
290 
291             try {
292                 TripleIterator triples =
293                         new FilteredTripleIterator(map, toAdd, true);
294                 triples.toStream(out, RDFFormat.RDF_XML, false);
295             } catch (TrippiException e) {
296                 throw new GeneralException(e.getMessage(), e);
297             }
298 
299             dsxml = new XMLDatastreamProcessor(dsId);
300             Datastream newds = dsxml.getDatastream();
301             newds.DatastreamAltIDs = new String[0];
302             // formats for internal datastreams
303             if (dsId.equals("RELS-EXT")) {
304                 newds.DSFormatURI = RELS_EXT1_0.uri;
305             } else {
306                 if (dsId.equals("RELS-INT"))
307                     newds.DSFormatURI = RELS_INT1_0.uri;
308             }
309             newds.DSMIME = "application/rdf+xml";
310             //newds.DSControlGrp = "X"; set by XMLDatastreamProcessor instead
311             newds.DSInfoType = null;
312             newds.DSState = "A";
313             newds.DSVersionable = false;
314             newds.DSVersionID = dsId + ".0";
315             newds.DSLabel = "Relationships";
316             newds.DSCreateDT = Server.getCurrentDate(m_context);
317             newds.DSLocation = null;
318             newds.DSLocationType = null;
319             newds.DSChecksumType = Datastream.getDefaultChecksumType();
320             dsxml.setXMLContent(out.toByteArray());
321             newds.DSSize = dsxml.getXMLContent().length;
322 
323             ValidationUtility.validateReservedDatastream(PID.getInstance(m_obj.getPid()),
324                                                          newds.DatastreamID,
325                                                          newds);
326             addDatastream(newds, false);
327         } else { // (relsDatastream != null)
328             dsxml = new XMLDatastreamProcessor(relsDatastream);
329             FilteredTripleIterator newIter = null;
330             try {
331                 ByteArrayOutputStream out = new ByteArrayOutputStream();
332                 TripleIterator iter =
333                         TripleIteratorFactory.defaultInstance().fromStream(relsDatastream.getContentStream(),
334                                                   RDFFormat.RDF_XML);
335                 newIter = new FilteredTripleIterator(iter, toAdd, true);
336                 newIter.toStream(out, RDFFormat.RDF_XML, false);
337 
338                 if (newIter.wasChangeMade()) {
339                     XMLDatastreamProcessor newdsxml = dsxml.newVersion();
340                     Datastream newds = newdsxml.getDatastream();
341 
342                     // TODO: only for XML Metadata datastream
343                     newdsxml.setDSMDClass(dsxml.getDSMDClass());
344 
345                     newds.DatastreamID = relsDatastream.DatastreamID;
346                     newds.DatastreamAltIDs = relsDatastream.DatastreamAltIDs;
347                     newds.DSFormatURI = relsDatastream.DSFormatURI;
348                     newds.DSMIME = relsDatastream.DSMIME;
349                     // newds.DSControlGrp = "X"; set by XMLDatastreamProcessor
350                     newds.DSInfoType = relsDatastream.DSInfoType;
351                     newds.DSState = relsDatastream.DSState;
352                     newds.DSVersionable = relsDatastream.DSVersionable;
353                     newds.DSVersionID = newDatastreamID(dsId);
354                     newds.DSLabel = relsDatastream.DSLabel;
355                     newds.DSCreateDT = Server.getCurrentDate(m_context);
356                     newds.DSLocation = null;
357                     newds.DSLocationType = null;
358                     newds.DSChecksumType = relsDatastream.DSChecksumType;
359                     newdsxml.setXMLContent(out.toByteArray());
360                     newds.DSSize = newdsxml.getXMLContent().length;
361 
362                     ValidationUtility.validateReservedDatastream(PID.getInstance(m_obj.getPid()),
363                                                                  newds.DatastreamID,
364                                                                  newds);
365                     addDatastream(newds, newds.DSVersionable);
366                 } else {
367                     // relationship already exists
368                     return false;
369                 }
370             } catch (TrippiException e) {
371                 throw new GeneralException(e.getMessage(), e);
372             } finally {
373                 try {
374                     if (newIter != null) {
375                         newIter.close();
376                     }
377                 } catch (TrippiException e) {
378                     throw new GeneralException(e.getMessage(), e);
379                 }
380             }
381         }
382         return true;
383     }

399     public boolean purgeRelationship(String dsId,
400                                      String subject,
401                                      String relationship,
402                                      String object,
403                                      boolean isLiteral,
404                                      String datatype) throws ServerException {
405 
406         Triple toPurge =
407                 createTriple(subject, relationship, object, isLiteral, datatype);
408 
409         Datastream relsDatastream = GetDatastream(dsId, null);
410         if (relsDatastream == null) {
411             // relationship does not exist
412             return false;
413         } else { // (relsExt != null)
414             XMLDatastreamProcessor dsxml = new XMLDatastreamProcessor(relsDatastream);
415             InputStream relsDatastreamIS = relsDatastream.getContentStream();
416 
417             TripleIterator iter = null;
418             FilteredTripleIterator newIter = null;
419             try {
420                 iter = TripleIteratorFactory.defaultInstance().fromStream(relsDatastreamIS, RDFFormat.RDF_XML);
421 
422                 newIter = new FilteredTripleIterator(iter, toPurge, false);
423                 ByteArrayOutputStream out = new ByteArrayOutputStream();
424 
425                 newIter.toStream(out, RDFFormat.RDF_XML, false);
426 
427                 if (newIter.wasChangeMade()) {
428                     XMLDatastreamProcessor newdsxml = dsxml.newVersion();
429                     Datastream newds = newdsxml.getDatastream();
430                     // TODO: setting of this on DatastreamAsXML
431                     // TODO: wrap original datastream in handler class
432                     newdsxml.setDSMDClass(dsxml.getDSMDClass());
433                     newds.DatastreamID = dsId;
434                     newds.DatastreamAltIDs = relsDatastream.DatastreamAltIDs;
435                     newds.DSFormatURI = relsDatastream.DSFormatURI;
436                     newds.DSMIME = relsDatastream.DSMIME;
437                     // newds.DSControlGrp = "X"; set by XMLDatastreamProcessor
438                     newds.DSInfoType = relsDatastream.DSInfoType;
439                     newds.DSState = relsDatastream.DSState;
440                     newds.DSVersionable = relsDatastream.DSVersionable;
441                     newds.DSVersionID = newDatastreamID(dsId);
442                     newds.DSLabel = relsDatastream.DSLabel;
443                     newds.DSCreateDT = Server.getCurrentDate(m_context);
444 
445                     newds.DSLocation = null;
446                     newds.DSLocationType = null;
447                     newds.DSChecksumType = relsDatastream.DSChecksumType;
448                     newdsxml.setXMLContent(out.toByteArray());
449                     newds.DSSize = newdsxml.getXMLContent().length;
450 
451                     ValidationUtility.validateReservedDatastream(PID.getInstance(m_obj.getPid()),
452                                                                  newds.DatastreamID,
453                                                                  newds);
454                     addDatastream(newds, newds.DSVersionable);
455                 } else {
456                     // relationship does not exist
457                     return false;
458                 }
459             } catch (TrippiException e) {
460                 throw new GeneralException(e.getMessage(), e);
461             } finally {
462                 try {
463                     if (newIter != null) {
464                         newIter.close(); // also closes the contained iter
465                     }
466                 } catch (TrippiException e) {
467                     throw new GeneralException(e.getMessage(), e);
468                 }
469             }
470         }
471         return true;
472     }

474     private static Triple createTriple(String subject,
475                                        String predicate,
476                                        String object,
477                                        boolean isLiteral,
478                                        String datatype)
479             throws ServerException {
480         ObjectNode o = null;
481         try {
482             if (isLiteral) {
483                 if (datatype == null || datatype.length() == 0) {
484                     o = new SimpleLiteral(object);
485                 } else {
486                     o = new SimpleLiteral(object, new URI(datatype));
487                 }
488             } else {
489                 o = new SimpleURIReference(new URI(object));
490             }
491             return new SimpleTriple(new SimpleURIReference(new URI(subject)),
492                                     new SimpleURIReference(new URI(predicate)),
493                                     o);
494         } catch (URISyntaxException e) {
495             throw new GeneralException(e.getMessage(), e);
496         }
497     }
