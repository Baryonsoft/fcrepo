177     public void addDatastreamVersion(Datastream ds, boolean addNewVersion) {
178         if (!addNewVersion) {
179             Datastream latestCreated = null;
180             long latestCreateTime = -1;
181             for (Datastream d : datastreams(ds.DatastreamID)) {
182                 if (d.DSCreateDT.getTime() > latestCreateTime) {
183                     latestCreateTime = d.DSCreateDT.getTime();
184                     latestCreated = d;
185                 }
186             }
187             remove(latestCreated);
188         }
189         add(ds);
190     }

192     private void add(Datastream d) {
193         String id = d.DatastreamID;
194         if (!m_datastreams.containsKey(id)) {
195             m_datastreams.put(id, new ArrayList<Datastream>());
196         }
197 
198         m_datastreams.get(id).add(d);
199         m_datastreamProcessor.processAdd(d);
200     }
201 
202     private void remove(Datastream d) {
203         if (d == null) return;
204         List<Datastream> datastreams = m_datastreams.get(d.DatastreamID);
205 
206         if (datastreams == null) {
207             return;
208         }
209 
210         int size = datastreams.size();
211         for (int i = 0; i < size; i++) {
212             Datastream v = datastreams.get(i);
213             if (d.DSVersionID.equals(v.DSVersionID)) {
214                 datastreams.remove(i);
215                 m_datastreamProcessor.processRemove(v);
216                 break;
217             }
218         }
219 
220         /* If we've removed the last version, remove the ID from the map */
221         if (datastreams.size() == 0) {
222             m_datastreams.remove(d.DatastreamID);
223         }
224     }

318     public Set<RelationshipTuple> getRelationships(SubjectNode subject,
319                                                    PredicateNode predicate,
320                                                    ObjectNode object) {
321         Set<RelationshipTuple> foundRels = new HashSet<RelationshipTuple>();
322 
323         if (m_rels == null) {
324             readRels();
325         }
326 
327         boolean basicExplicit = false;
328 
329         // Iterate explicit relationships, finding matches and
330         // determining whether the object has an explicit basic cmodel.
331         for (RelationshipTuple t : m_rels) {
332 
333             // Do any hasModel rels point to a basic cmodel?
334             if (Constants.MODEL.HAS_MODEL.uri.equals(t.predicate)
335                     && Models.isBasicModel(t.object)) {
336                 basicExplicit = true;
337             }
338 
339             // Find matching relationships from those that are explicit
340             if (subject != null) {
341                 if (!JRDF.sameSubject(subject, t.subject)) {
342                     continue;
343                 }
344             }
345             if (predicate != null) {
346                 if (!JRDF.samePredicate(predicate, t.predicate)) {
347                     continue;
348                 }
349             }
350             if (object != null) {
351                 if (!JRDF.sameObject(object,
352                                      t.object,
353                                      t.isLiteral,
354                                      t.datatype,
355                                      t.language)) {
356                     continue;
357                 }
358 
359             }
360             foundRels.add(t);
361         }
362 
363         // If necessary, add the current basic cmodel to the set of matches
364         try {
365             if (!basicExplicit
366                     && (subject == null ||
367                             JRDF.sameSubject(subject, new SimpleURIReference(new URI(PID.toURI(m_pid)))))
368                     && (predicate == null ||
369                         JRDF.samePredicate(predicate, Constants.MODEL.HAS_MODEL))
370                     && (object == null ||
371                         JRDF.sameObject(object, Models.FEDORA_OBJECT_CURRENT))) {
372                 foundRels.add(
373                         new RelationshipTuple(Constants.FEDORA.uri + m_pid,
374                                               Constants.MODEL.HAS_MODEL.uri,
375                                               Models.FEDORA_OBJECT_CURRENT.uri,
376                                               false,
377                                               null));
378             }
379         } catch (URISyntaxException e) {
380             // assume that m_pid is a valid pid
381         }
382 
383         return foundRels;
384     }

400     /**
401      * Given an iterator of existing ids, return a new id that starts with
402      * <code>start</code> and is guaranteed to be unique. This algorithm adds
403      * one to the highest existing id that starts with <code>start</code>. If
404      * no such existing id exists, it will return <i>start</i> + "1".
405      */
406     private String newID(Iterator<String> iter, String start) {
407         int highest = 0;
408         while (iter.hasNext()) {
409             String id = iter.next();
410             if (id.startsWith(start) && id.length() > start.length()) {
411                 try {
412                     int num = Integer.parseInt(id.substring(start.length()));
413                     if (num > highest) {
414                         highest = num;
415                     }
416                 } catch (NumberFormatException ignored) {
417                 }
418             }
419         }
420         int newNum = highest + 1;
421         return start + newNum;
422     }

431     /**
432      * Given a relationships datastream name, return the relationships contained
433      * in that datastream
434      */
435     private Set<RelationshipTuple> getRels(String relsDatastreamName) {
436         List<Datastream> relsDatastreamVersions = m_datastreams.get(relsDatastreamName);
437 
438         if (relsDatastreamVersions == null || relsDatastreamVersions.size() == 0) {
439             return new HashSet<RelationshipTuple>();
440         }
441 
442         Datastream latestRels = relsDatastreamVersions.get(0);
443 
444         for (Datastream v : relsDatastreamVersions) {
445            	if (v.DSCreateDT == null){
446            		latestRels = v;
447            		break;
448            	}
449             if (v.DSCreateDT.getTime() > latestRels.DSCreateDT.getTime()) {
450                 latestRels = v;
451             }
452         }
453 
454         try {
455             return RDFRelationshipReader.readRelationships(latestRels);
456         } catch (ServerException e) {
457             throw new RuntimeException("Error reading object relationships in " + relsDatastreamName, e);
458         }
459     }

461     private abstract class DatastreamProcessor {
462 
463         abstract void processAdd(Datastream d);
464 
465         abstract void processRemove(Datastream d);
466 
467         boolean isLatestVersion(Datastream d) {
468 
469             List<Datastream> versions = m_datastreams.get(d.DatastreamID);
470 
471             if (d.DSCreateDT == null || versions == null || versions.size() == 0) return true;
472             
473             long created = d.DSCreateDT.getTime();
474 
475             for (Datastream v : versions) {
476                 if (v.DSCreateDT.getTime() > created) {
477                     return false;
478                 }
479             }
480 
481             return true;
482         }
483     }

485     /**
486      * If the latest RELS-EXT or RELS-INT is added or removed, invalidate the relationship
487      * cache so that it has to be re-read next time it is requested.
488      */
489     private class RelationshipProcessor
490             extends DatastreamProcessor {
491 
492         private static final String RELS_EXT = "RELS-EXT";
493         private static final String RELS_INT = "RELS-INT";
494 
495         @Override
496         void processRemove(Datastream d) {
497             invalidateIfLatestRels(d);
498         }
499 
500         @Override
501         void processAdd(Datastream d) {
502             invalidateIfLatestRels(d);
503         }
504 
505         private void invalidateIfLatestRels(Datastream d) {
506             if ((d.DatastreamID.equals(RELS_EXT) || d.DatastreamID.equals(RELS_INT))&& isLatestVersion(d)) {
507                 m_rels = null;
508             }
509         }
510     }
