19  class TimestampPathAlgorithm
20          extends PathAlgorithm {
21  
22      private final String storeBase;
23  
24      private static final String[] PADDING = {"", "0", "00", "000"};
25  
26      private static final String SEP = File.separator;
27  
28      public TimestampPathAlgorithm(Map<String, ?> configuration) {
29          super(configuration);
30          storeBase = (String) configuration.get("storeBase");
31      }
32  
33      @Override
34      public final String get(String pid) throws LowlevelStorageException {
35          return format(encode(pid));
36      }
37  
38      public String format(String pid) throws LowlevelStorageException {
39          GregorianCalendar calendar = new GregorianCalendar();
40          String year = Integer.toString(calendar.get(Calendar.YEAR));
41          String month = leftPadded(1 + calendar.get(Calendar.MONTH), 2);
42          String dayOfMonth = leftPadded(calendar.get(Calendar.DAY_OF_MONTH), 2);
43          String hourOfDay = leftPadded(calendar.get(Calendar.HOUR_OF_DAY), 2);
44          String minute = leftPadded(calendar.get(Calendar.MINUTE), 2);
45          //String second = leftPadded(calendar.get(Calendar.SECOND),2);
46          return storeBase + SEP + year + SEP + month + dayOfMonth + SEP
47                  + hourOfDay + SEP + minute /* + sep + second */+ SEP + pid;
48      }
49  
50      private final String leftPadded(int i, int n)
51              throws LowlevelStorageException {
52          if (n > 3 || n < 0 || i < 0 || i > 999) {
53              throw new LowlevelStorageException(true, getClass().getName()
54                      + ": faulty date padding");
55          }
56          int m = i > 99 ? 3 : i > 9 ? 2 : 1;
57          int p = n - m;
58          return PADDING[p] + Integer.toString(i);
59      }
60  }
