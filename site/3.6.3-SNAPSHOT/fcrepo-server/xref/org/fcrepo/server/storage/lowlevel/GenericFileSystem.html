40      private final File wrappedNewFile(File file, String suffix)
41              throws LowlevelStorageException {
42          File temp;
43          String path = "";
44          try {
45              path = file.getCanonicalPath() + suffix;
46              temp = new File(path);
47          } catch (Exception e) {
48              throw new LowlevelStorageException(true,
49                                                 "GenericFileSystem.wrappedNewFile(): couldn't create File for ["
50                                                         + path + "]",
51                                                 e);
52          }
53          return temp;
54      }

56      private final String getPath(File file) {
57          String temp;
58          try {
59              temp = file.getCanonicalPath();
60          } catch (Exception eCaughtFiles) {
61              temp = "";
62          }
63          return temp;
64      }

66      @Override
67      public final void write(File file, InputStream content)
68              throws LowlevelStorageException {
69          try {
70              writeIntoExistingDirectory(file, content);
71          } catch (LowlevelStorageException eCaught) {
72              File containingDirectories = null;
73              try {
74                  containingDirectories = file.getParentFile();
75                  containingDirectories.mkdirs();
76              } catch (Exception e) {
77                  throw new LowlevelStorageException(true,
78                                                     "GenericFileSystem.write(): couldn't make directories for ["
79                                                             + getPath(file)
80                                                             + "]",
81                                                     e);
82              }
83              writeIntoExistingDirectory(file, content);
84          }
85      }

87      private final void writeIntoExistingDirectory(File file, InputStream content)
88              throws LowlevelStorageException {
89          FileOutputStream fileOutputStream = null;
90          try {
91              fileOutputStream = new FileOutputStream(file);
92          } catch (Exception eCaughtFileNotCreated) {
93              throw new LowlevelStorageException(true, "couldn't create file "
94                      + getPath(file), eCaughtFileNotCreated);
95          }
96          try {
97              boolean fileCopySuccessful = FileUtils.copy(content, fileOutputStream);
98  
99              if(!fileCopySuccessful) {
100                 throw new LowlevelStorageException(true, "couldn't write new file "
101                         + getPath(file));
102             }
103         } finally {
104             try {
105                 fileOutputStream.close();
106                 content.close();
107             } catch (Exception eCaughtFileNotClosed) {
108                 throw new LowlevelStorageException(true,
109                                                    "couldn't close new file "
110                                                      + getPath(file),
111                                                    eCaughtFileNotClosed);
112             }
113         }
114     }

116     @Override
117     public void rewrite(File file, InputStream content)
118             throws LowlevelStorageException {
119 
120         File backupFile = wrappedNewFile(file, ".bak");
121 
122         if (!file.renameTo(backupFile)) {
123             try {
124                 content.close();
125             } catch (IOException e) {
126             }
127             throw new LowlevelStorageException(true, "failed to rename with "
128                     + ".bak extension " + getPath(file));
129         }
130 
131         boolean needToRevert = false;
132         String err = null;
133         FileOutputStream out = null;
134 
135         try {
136             out = new FileOutputStream(file);
137             boolean fileCopySuccessful = FileUtils.copy(content, out);
138             if(!fileCopySuccessful) {
139                 needToRevert = true;
140                 err = "failed to write content to file " + file.getPath();
141             }
142         } catch (IOException e) {
143             needToRevert = true;
144             err = "failed to write content to file " + file.getPath();
145         } finally {
146             if (out != null) {
147                 try {
148                     out.close();
149                 } catch (IOException e) {
150                     logger.warn("Could not close file for writing "
151                             + file.getPath(), e);
152                 }
153             }
154             try {
155                 content.close();
156             } catch (IOException e) {
157                 logger.warn("Could not close content stream for reading", e);
158             }
159         }
160 
161         if (needToRevert) {
162             if (backupFile.renameTo(file)) {
163                 err += ", so reverted to original";
164             } else {
165                 err += ", AND failed to revert to original from .bak!";
166             }
167             throw new LowlevelStorageException(true, err);
168         } else {
169             if (!backupFile.delete()) {
170                 logger.warn("Could not delete backup file {}",
171                         backupFile.getPath());
172             }
173         }
174 
175     }

177     @Override
178     public final InputStream read(File file) throws LowlevelStorageException {
179         //buffered reader?
180         FileInputStream fileInputStream = null;
181         {
182             if (!file.exists()) {
183                 throw new LowlevelStorageException(true, "file "
184                         + getPath(file) + "doesn't exist for reading");
185             }
186             if (!file.canRead()) {
187                 throw new LowlevelStorageException(true, "file "
188                         + getPath(file) + "not readable");
189             }
190 
191             try {
192                 fileInputStream = new FileInputStream(file);
193             } catch (IOException eCaughtOpenFile) {
194                 throw new LowlevelStorageException(true,
195                                                    "file "
196                                                            + getPath(file)
197                                                            + "couldn't be opened for reading",
198                                                    eCaughtOpenFile);
199             }
200         }
201         return fileInputStream;
202     }
