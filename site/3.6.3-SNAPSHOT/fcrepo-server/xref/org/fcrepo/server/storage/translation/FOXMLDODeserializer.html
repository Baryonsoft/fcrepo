203     /**
204      * Creates a deserializer that reads the given FOXML format.
205      *
206      * @param format
207      *        the version-specific FOXML format.
208      * @throws IllegalArgumentException
209      *         if format is not a known FOXML format.
210      */
211     public FOXMLDODeserializer(XMLFormat format) {
212         if (format.equals(FOXML1_0) || format.equals(FOXML1_1)) {
213             m_format = format;
214         } else {
215             throw new IllegalArgumentException("Not a FOXML format: "
216                     + format.uri);
217         }
218         // initialize sax
219         try {
220             synchronized(spf){
221                 m_parser = spf.newSAXParser();
222             }
223         } catch (Exception e) {
224             throw new RuntimeException("Error initializing SAX parser", e);
225         }
226 
227     }

240     /**
241      * {@inheritDoc}
242      */
243     public synchronized void deserialize(InputStream in,
244                             DigitalObject obj,
245                             String encoding,
246                             int transContext) throws ObjectIntegrityException,
247             StreamIOException, UnsupportedEncodingException {
248         if (logger.isDebugEnabled()) {
249             logger.debug("Deserializing " + m_format.uri + " for transContext: "
250                 + transContext);
251         }
252 
253         m_obj = obj;
254         m_obj.setLabel("");
255         m_obj.setOwnerId("");
256         m_characterEncoding = encoding;
257         m_transContext = transContext;
258         initialize();
259         try {
260             m_parser.parse(in, this);
261         } catch (IOException ioe) {
262             throw new StreamIOException("low-level stream io problem occurred "
263                     + "while sax was parsing this object.");
264         } catch (SAXException se) {
265             throw new ObjectIntegrityException("FOXML IO stream was bad : "
266                     + se.getMessage(), se);
267         }
268         logger.debug("Just finished parse.");
269 
270         if (!m_rootElementFound) {
271             throw new ObjectIntegrityException("FOXMLDODeserializer: Input stream is not valid FOXML."
272                     + " The digitalObject root element was not detected.");
273         }
274 
275         DOTranslationUtility.normalizeDatastreams(m_obj, m_transContext, m_characterEncoding);
276     }

302         }

305     /**
306      * {@inheritDoc}
307      */
308     @Override
309     public void startElement(String uri,
310                              String localName,
311                              String qName,
312                              Attributes a) throws SAXException {
313 
314         // Initialize string buffer to hold content of the new element.
315         // This will start a fresh buffer for every element encountered.
316         m_elementContent = new StringBuffer();
317 
318         if (uri.equals(FOXML.uri) && !m_inXMLMetadata) {
319             // WE ARE NOT INSIDE A BLOCK OF INLINE XML...
320             if (localName.equals("digitalObject")) {
321                 m_rootElementFound = true;
322                 //======================
323                 // OBJECT IDENTIFIERS...
324                 //======================
325                 m_obj.setPid(grab(a, FOXML.uri, "PID"));
326                 //=====================
327                 // OBJECT PROPERTIES...
328                 //=====================
329             } else if (localName.equals("property")
330                     || localName.equals("extproperty")) {
331                 m_objPropertyName = grab(a, FOXML.uri, "NAME");
332                 if (m_objPropertyName.equals(MODEL.STATE.uri)) {
333                     try {
334                         m_obj.setState(DOTranslationUtility
335                                        .readStateAttribute(grab(a, FOXML.uri, "VALUE")));
336                     } catch (ParseException e) {
337                         throw new SAXException("Could not read state", e);
338                     }
339                 } else if (m_objPropertyName.equals(MODEL.LABEL.uri)) {
340                     m_obj.setLabel(grab(a, FOXML.uri, "VALUE"));
341                 } else if (m_objPropertyName.equals(MODEL.OWNER.uri)) {
342                     m_obj.setOwnerId(grab(a, FOXML.uri, "VALUE"));
343                 } else if (m_objPropertyName.equals(MODEL.CREATED_DATE.uri)) {
344                     m_obj.setCreateDate(DateUtility
345                             .convertStringToDate(grab(a, FOXML.uri, "VALUE")));
346                 } else if (m_objPropertyName
347                         .equals(VIEW.LAST_MODIFIED_DATE.uri)) {
348                     m_obj.setLastModDate(DateUtility
349                             .convertStringToDate(grab(a, FOXML.uri, "VALUE")));
350                 } else {
351                     // Legacy object properties from FOXML 1.0, if present,
352                     // will be retained here as external properties in the
353                     // DigitalObject.  This includes fedora-model:contentModel
354                     // and rdf:type.
355                     m_obj.setExtProperty(m_objPropertyName, grab(a,
356                                                                  FOXML.uri,
357                                                                  "VALUE"));
358                 }
359                 //===============
360                 // DATASTREAMS...
361                 //===============
362             } else if (localName.equals("datastream")) {
363                 // get datastream container-level attributes...
364                 // These are common for all versions of the datastream.
365                 m_dsId = grab(a, FOXML.uri, "ID");
366                 m_dsState = grab(a, FOXML.uri, "STATE");
367                 m_dsControlGrp = grab(a, FOXML.uri, "CONTROL_GROUP");
368                 String versionable = grab(a, FOXML.uri, "VERSIONABLE");
369                 // If dsVersionable is null or missing, default to true.
370                 if (versionable == null || versionable.equals("")) {
371                     m_dsVersionable = true;
372                 } else {
373                     m_dsVersionable = new Boolean(versionable).booleanValue();
374                 }
375                 // Never allow the AUDIT datastream to be versioned
376                 // since it naturally represents a system-controlled
377                 // view of changes over time.
378                 if (m_dsId.equals("AUDIT")) {
379                     m_dsVersionable = false;
380                 }
381             } else if (localName.equals("datastreamVersion")) {
382                 // get datastream version-level attributes...
383                 m_dsVersId = grab(a, FOXML.uri, "ID");
384                 m_dsLabel = grab(a, FOXML.uri, "LABEL");
385                 m_dsCreateDate =
386                         DateUtility.convertStringToDate(grab(a,
387                                                              FOXML.uri,
388                                                              "CREATED"));
389                 String altIDsString = grab(a, FOXML.uri, "ALT_IDS");
390                 if (altIDsString.length() == 0) {
391                     m_dsAltIds = new String[0];
392                 } else {
393                     m_dsAltIds = altIDsString.split(" ");
394                 }
395                 m_dsFormatURI = grab(a, FOXML.uri, "FORMAT_URI");
396                 if (m_dsFormatURI.length() == 0) {
397                     m_dsFormatURI = null;
398                 }
399                 checkMETSFormat(m_dsFormatURI);
400                 m_dsMimeType = grab(a, FOXML.uri, "MIMETYPE");
401                 String sizeString = grab(a, FOXML.uri, "SIZE");
402                 if (sizeString != null && !sizeString.equals("")) {
403                     try {
404                         m_dsSize = Long.parseLong(sizeString);
405                     } catch (NumberFormatException nfe) {
406                         throw new SAXException("If specified, a datastream's "
407                                 + "SIZE attribute must be an xsd:long.");
408                     }
409                 } else {
410                     m_dsSize = -1;
411                 }
412                 if (m_dsVersId.equals("AUDIT.0")) {
413                     m_gotAudit = true;
414                 }
415                 m_dsChecksumType = Datastream.CHECKSUMTYPE_DISABLED;
416                 m_dsChecksum = Datastream.CHECKSUM_NONE;
417             } else if (localName.equals("contentDigest")) {
418                 m_dsChecksumType = grab(a, FOXML.uri, "TYPE");
419                 m_dsChecksum = grab(a, FOXML.uri, "DIGEST");
420             }
421             //======================
422             // DATASTREAM CONTENT...
423             //======================
424             // inside a datastreamVersion element, it's either going to be
425             // xmlContent (inline xml), contentLocation (a reference) or binaryContent
426             else if (localName.equals("xmlContent")) {
427                 m_dsXMLBuffer = new StringBuffer();
428                 m_xmlDataLevel = 0;
429                 m_inXMLMetadata = true;
430             } else if (localName.equals("contentLocation")) {
431                 String dsLocation = grab(a, FOXML.uri, "REF");
432                 if (dsLocation == null || dsLocation.equals("")) {
433                     throw new SAXException("REF attribute must be specified in contentLocation element");
434                 }
435                 // check if datastream is ExternalReferenced
436                 if (m_dsControlGrp.equalsIgnoreCase("E")
437                         || m_dsControlGrp.equalsIgnoreCase("R")) {
438 
439                     // URL FORMAT VALIDATION for dsLocation:
440                     // make sure we have a properly formed URL
441                     try {
442                         ValidationUtility.validateURL(dsLocation, m_dsControlGrp);
443                     } catch (ValidationException ve) {
444                         throw new SAXException(ve.getMessage());
445                     }
446                     // system will set dsLocationType for E and R datastreams...
447                     m_dsLocationType = Datastream.DS_LOCATION_TYPE_URL;
448                     m_dsLocation = dsLocation;
449                     instantiateDatastream(new DatastreamReferencedContent());
450                     // check if datastream is ManagedContent
451                 } else if (m_dsControlGrp.equalsIgnoreCase("M")) {
452                     // URL FORMAT VALIDATION for dsLocation:
453                     // For Managed Content the URL is only checked when we are parsing a
454                     // a NEW ingest file because the URL is replaced with an internal identifier
455                     // once the repository has sucked in the content for storage.
456                     if (m_obj.isNew()) {
457                         try {
458                             ValidationUtility.validateURL(dsLocation, m_dsControlGrp);
459                             m_dsLocationType = Datastream.DS_LOCATION_TYPE_URL;
460                         } catch (ValidationException ve) {
461                             throw new SAXException(ve.getMessage());
462                         }
463                     }
464                     else {
465                         m_dsLocationType = Datastream.DS_LOCATION_TYPE_INTERNAL;
466                     }
467                     m_dsLocation = dsLocation;
468                     instantiateDatastream(new DatastreamManagedContent());
469                 }
470             } else if (localName.equals("binaryContent")) {
471                 if (m_dsControlGrp.equalsIgnoreCase("M")) {
472                     m_readingBinaryContent = true;
473                     m_binaryContentTempFile = null;
474                     try {
475                         m_binaryContentTempFile =
476                                 File.createTempFile("binary-datastream", null);
477                     } catch (IOException ioe) {
478                         throw new SAXException(new StreamIOException("Unable to create temporary file for binary content"));
479                     }
480                 }
481             } else if (m_format.equals(FOXML1_0)) {
482                 //==================
483                 // DISSEMINATORS...
484                 //==================
485                 startDisseminators(localName, a);
486             }
487         } else {
488             //===============
489             // INLINE XML...
490             //===============
491             if (m_inXMLMetadata) {
492                 // we are inside an xmlContent element.
493                 // just output it, remembering the number of foxml:xmlContent elements we see,
494                 appendElementStart(uri, localName, qName, a, m_dsXMLBuffer);
495 
496                 // FOXML INSIDE FOXML! we have an inline XML datastream
497                 // that is itself FOXML.  We do not want to parse this!
498                 if (uri.equals(FOXML.uri) && localName.equals("xmlContent")) {
499                     m_xmlDataLevel++;
500                 }
501 
502                 // if AUDIT datastream, initialize new audit record object
503                 if (m_gotAudit) {
504                     if (localName.equals("record")) {
505                         m_auditRec = new AuditRecord();
506                         m_auditRec.id = grab(a, uri, "ID");
507                     } else if (localName.equals("process")) {
508                         m_auditProcessType = grab(a, uri, "type");
509                     }
510                 }
511             } else {
512                 // ignore all else
513             }
514         }
515     }

517     /**
518      * {@inheritDoc}
519      */
520     @Override
521     public void characters(char[] ch, int start, int length) {
522         // read normal element content into a string buffer
523         if (m_elementContent != null) {
524             m_elementContent.append(ch, start, length);
525         }
526         // read entire inline XML metadata chunks into a buffer
527         if (m_inXMLMetadata) {
528             // since this data is encoded straight back to xml,
529             // we need to make sure special characters &, <, >, ", and '
530             // are re-converted to the xml-acceptable equivalents.
531             StreamUtility.enc(ch, start, length, m_dsXMLBuffer);
532         } else if (m_readingBinaryContent) {
533             // append it to something...
534             // FIXME: IMPLEMENT HERE IN POST v2.0
535         }
536     }

538     /**
539      * {@inheritDoc}
540      */
541     @Override
542     public void endElement(String uri, String localName, String qName)
543             throws SAXException {
544         //==================
545         // INLINE XML...
546         //==================
547         if (m_inXMLMetadata) {
548             //=====================
549             // AUDIT DATASTREAM...
550             //=====================
551             if (m_gotAudit) {
552                 // Pick up audit records from the current ds version
553                 // and instantiate audit records array in digital object.
554                 if (localName.equals("action")) {
555                     m_auditAction = m_elementContent.toString();
556                     //} else if (localName.equals("recordID")) {
557                     //    m_auditRecordID=m_elementContent.toString();
558                 } else if (localName.equals("componentID")) {
559                     m_auditComponentID = m_elementContent.toString();
560                 } else if (localName.equals("responsibility")) {
561                     m_auditResponsibility = m_elementContent.toString();
562                 } else if (localName.equals("date")) {
563                     m_auditDate = m_elementContent.toString();
564                 } else if (localName.equals("justification")) {
565                     m_auditJustification = m_elementContent.toString();
566                 } else if (localName.equals("record")) {
567                     //m_auditRec.id=m_auditRecordID;
568                     m_auditRec.processType = m_auditProcessType;
569                     m_auditRec.action = m_auditAction;
570                     m_auditRec.componentID = m_auditComponentID;
571                     m_auditRec.responsibility = m_auditResponsibility;
572                     m_auditRec.date =
573                             DateUtility.convertStringToDate(m_auditDate);
574                     m_auditRec.justification = m_auditJustification;
575                     // add the audit records to the digital object
576                     m_obj.getAuditRecords().add(m_auditRec);
577                     // reinit variables for next audit record
578                     m_auditProcessType = "";
579                     m_auditAction = "";
580                     m_auditComponentID = "";
581                     m_auditResponsibility = "";
582                     m_auditDate = "";
583                     m_auditJustification = "";
584                 } else if (localName.equals("auditTrail")) {
585                     m_gotAudit = false;
586                 }
587                 // process end of xmlContent ONLY if it is NOT embedded within inline XML!
588             } else if (uri.equals(FOXML.uri) && localName.equals("xmlContent")
589                     && m_xmlDataLevel == 0) {
590                 //=====================
591                 // AUDIT DATASTREAM...
592                 //=====================
593                 if (m_dsId.equals("AUDIT")) {
594                     // if we are in the inline XML of the AUDIT datastream just
595                     // end processing and move on.  Audit datastream handled elsewhere.
596                     m_inXMLMetadata = false; // other stuff is re-initted upon
597                     // startElement for next xml metadata
598                     // element
599                     //========================
600                     // ALL OTHER INLINE XML...
601                     //========================
602                 } else {
603                     // Create the right kind of datastream and add to the object
604                     DatastreamXMLMetadata ds = new DatastreamXMLMetadata();
605                     instantiateXMLDatastream(ds);
606                     m_inXMLMetadata = false;
607                     m_localPrefixMap.clear();
608                 }
609             } else {
610                 // finished an element within inline xml metadata
611                 m_dsXMLBuffer.append("</" + qName + ">");
612                 // make sure we know when to pay attention to FOXML again
613                 if (uri.equals(FOXML.uri) && localName.equals("xmlContent")) {
614                     m_xmlDataLevel--;
615                 }
616             }
617             //========================================
618             // ALL OTHER ELEMENTS (NOT INLINE XML)...
619             //========================================
620         } else if (uri.equals(FOXML.uri) && localName.equals("binaryContent")) {
621             if (m_binaryContentTempFile != null) {
622                 try {
623                     FileOutputStream os =
624                             new FileOutputStream(m_binaryContentTempFile);
625                     // remove all spaces and newlines, this might not be necessary.
626                     String elementStr =
627                             m_elementContent.toString().replaceAll("\\s", "");
628                     byte elementBytes[] = Base64.decode(elementStr);
629                     os.write(elementBytes);
630                     os.close();
631                     m_dsLocationType = Datastream.DS_LOCATION_TYPE_INTERNAL;
632                     m_dsLocation =
633                         DatastreamManagedContent.TEMP_SCHEME
634                                     + m_binaryContentTempFile.getAbsolutePath();
635                     instantiateDatastream(new DatastreamManagedContent());
636                 } catch (FileNotFoundException fnfe) {
637                     throw new SAXException(new StreamIOException("Unable to open temporary file created for binary content"));
638                 } catch (IOException fnfe) {
639                     throw new SAXException(new StreamIOException("Error writing to temporary file created for binary content"));
640                 }
641             }
642             m_binaryContentTempFile = null;
643             m_readingBinaryContent = false;
644         } else if (uri.equals(FOXML.uri)
645                 && localName.equals("datastreamVersion")) {
646             // reinitialize datastream version-level attributes...
647             m_dsVersId = "";
648             m_dsLabel = "";
649             m_dsCreateDate = null;
650             m_dsAltIds = new String[0];
651             m_dsFormatURI = "";
652             m_dsMimeType = "";
653             m_dsSize = -1;
654             //m_dsAdmIds=new HashMap();
655             //m_dsDmdIds=null;
656         } else if (uri.equals(FOXML.uri) && localName.equals("datastream")) {
657             // reinitialize datastream attributes ...
658             m_dsId = "";
659             m_dsVersionable = true;
660             m_dsState = "";
661             m_dsInfoType = "";
662             m_dsOtherInfoType = "";
663             m_dsMDClass = 0;
664         } else if (m_format.equals(FOXML1_0)) {
665             endDisseminators(uri, localName);
666         }
667     }

673     private void startDisseminators(String localName, Attributes a) {
674         if (localName.equals("disseminator")) {
675             m_dissID = grab(a, FOXML.uri, "ID");
676             m_sDefID = grab(a, FOXML.uri, "BDEF_CONTRACT_PID");

745     private void appendElementStart(String uri,
746                                     String localName,
747                                     String qName,
748                                     Attributes a,
749                                     StringBuffer out) {
750         out.append("<" + qName);
751         // add the current qName's namespace to m_localPrefixMap
752         // and m_prefixList if it's not already in m_localPrefixMap
753         // This ensures that all namespaces used in inline XML are declared within,
754         // since it's supposed to be a standalone chunk.
755         String[] parts = qName.split(":");
756         if (parts.length == 2) {
757             String nsuri = m_localPrefixMap.get(parts[0]);
758             if (nsuri == null) {
759                 m_localPrefixMap.put(parts[0], parts[1]);
760                 m_prefixList.add(parts[0]);
761             }
762         }
763         // do we have any newly-mapped namespaces?
764         while (m_prefixList.size() > 0) {
765             String prefix = m_prefixList.remove(0);
766             out.append(" xmlns");
767             if (prefix.length() > 0) {
768                 out.append(":");
769             }
770             out.append(prefix + "=\""
771                     + StreamUtility.enc(m_prefixMap.get(prefix))
772                     + "\"");
773         }
774         for (int i = 0; i < a.getLength(); i++) {
775             out.append(" " + a.getQName(i) + "=\""
776                     + StreamUtility.enc(a.getValue(i)) + "\"");
777         }
778         out.append(">");
779     }

926     private void checkMETSFormat(String formatURI) {
927         if (formatURI != null && !formatURI.equals("")) {
928             Matcher m = metsPattern.matcher(formatURI);
929             //Matcher m = metsURI.matcher(formatURI);
930             if (m.lookingAt()) {
931                 int index = m.end();
932                 StringTokenizer st =
933                         new StringTokenizer(formatURI.substring(index), ".");
934                 String mdClass = st.nextToken();
935                 if (st.hasMoreTokens()) {
936                     m_dsInfoType = st.nextToken();
937                 }
938                 if (st.hasMoreTokens()) {
939                     m_dsOtherInfoType = st.nextToken();
940                 }
941                 if (mdClass.equals("techMD")) {
942                     m_dsMDClass = 1;
943                 } else if (mdClass.equals("sourceMD")) {
944                     m_dsMDClass = 2;
945                 } else if (mdClass.equals("rightsMD")) {
946                     m_dsMDClass = 3;
947                 } else if (mdClass.equals("digiprovMD")) {
948                     m_dsMDClass = 4;
949                 } else if (mdClass.equals("descMD")) {
950                     m_dsMDClass = 5;
951                 }
952                 if (m_dsInfoType.equals("OTHER")) {
953                     m_dsInfoType = m_dsOtherInfoType;
954                 }
955             }
956         }
