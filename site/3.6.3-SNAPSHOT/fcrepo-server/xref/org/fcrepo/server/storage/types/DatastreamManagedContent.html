128     // Note: might seem strange to pass through the Context, but DatastreamManagedContent looks after
129     // datastreams before they have been ingested, ie while location is still external (file, URL), so without
130     // Context authz will fail
131     public InputStream getContentStream(Context ctx) throws StreamIOException {
132         try {
133             // For new or modified datastreams, the new bytestream hasn't yet been
134             // committed. However, we need to access it in order to compute
135             // the datastream checksum
136             if (DSLocation.startsWith(UPLOADED_SCHEME)) {
137                 // TODO: refactor to use proper temp file management - FCREPO-718
138                 // for now, just get the file directly (see also DefaultManagement.getTempStream(...))
139                 String internalId = DSLocation.substring(UPLOADED_SCHEME.length());
140                 File uploadedFile = new File(getTempUploadDir(), internalId);
141                 // check it has not been automatically purged (see DefaultManagement.purgeUploadedFiles())
142                 if (uploadedFile.exists()) {
143                     return new FileInputStream(uploadedFile);
144                 } else {
145                     throw new StreamIOException("Uploaded file " + DSLocation + " no longer exists.");
146                 }
147 
148             } else if (DSLocation.startsWith(TEMP_SCHEME)) {
149                 // TODO: refactor to use proper temp file management - FCREPO-718
150                 String fileName = DSLocation.substring(TEMP_SCHEME.length());
151                 File tempFile = new File(fileName);
152                 // check it has not been removed elsewhere (should not happen)
153                 if (tempFile.exists()) {
154                     return new FileInputStream(tempFile);
155                 } else {
156                     throw new StreamIOException("Temp file " + DSLocation + " no longer exists.");
157                 }
158             } else if (Datastream.DS_LOCATION_TYPE_INTERNAL.equals(DSLocationType)) {
159                 return getLLStore().retrieveDatastream(DSLocation);
160             } else if (Datastream.DS_LOCATION_TYPE_URL.equals(DSLocationType)) {
161                 // Managed content can have URL dsLocation on ingest
162                 ValidationUtility.validateURL(DSLocation, this.DSControlGrp);
163                 // If validation has succeeded, assume an external resource.
164                 // Fetch it, store it locally, update DSLocation
165                 if (ctx == null) {
166                     ctx = ReadOnlyContext.getContext(null, null, "", false);
167                     // changed from below, which sets noOp to true, seems to cause an AuthZ permitted exception
168                     // in PolicyEnforcementPoint.enforce(...)
169                     // ctx = ReadOnlyContext.EMPTY;
170                 }
171                ContentManagerParams params = new ContentManagerParams(DSLocation);
172                 params.setContext(ctx);
173                 MIMETypedStream stream = getExternalContentManager()
174                         .getExternalContent(params);
175 
176                 // TODO: refactor temp file management - see FCREPO-718; for now create temp file and write to it
177                 // note - don't use temp upload directory, use (container's) temp dir (upload dir is for uploads)
178                 File tempFile = File.createTempFile("managedcontentupdate", null);
179                 OutputStream os = new FileOutputStream(tempFile);
180                 StreamUtility.pipeStream(stream.getStream(), os, 32768);
181                 DSLocation = TEMP_SCHEME + tempFile.getAbsolutePath();
182                 return new FileInputStream(new File(tempFile.getAbsolutePath()));
183             }
184         } catch (Throwable th) {
185             throw new StreamIOException("[DatastreamManagedContent] returned "
186                     + " the error: \"" + th.getClass().getName()
187                     + "\". Reason: " + th.getMessage(), th);
188         }
189         throw new StreamIOException("[DatastreamManagedContent] could not resolve dsLocation " + DSLocation + " dsLocationType " + DSLocationType);
190     }

192     /**
193      * Set the contents of this managed datastream by storing as a temp file.  If the previous content
194      * was stored in a temp file, clean up this file.
195      * @param stream - the data to store in this datastream
196      * @throws StreamIOException
197      */
198     public void putContentStream(MIMETypedStream stream) throws StreamIOException {
199 
200         // TODO: refactor to use proper temp file management - FCREPO-718
201         // for now, write to new temp file, clean up existing location
202         String oldDSLocation = DSLocation;
203         try {
204             // note: don't use temp upload dir, use (container's) temp dir (upload dir is for uploads)
205             File tempFile = File.createTempFile("managedcontentupdate", null);
206             OutputStream os = new FileOutputStream(tempFile);
207             StreamUtility.pipeStream(stream.getStream(), os, 32768);
208             DSLocation = TEMP_SCHEME + tempFile.getAbsolutePath();
209         } catch (Exception e) {
210             throw new StreamIOException("Error creating new temp file for updated managed content (existing content is:" + oldDSLocation + ")", e);
211         }
212 
213         // if old location was a temp location, clean it up
214         // (if old location was uploaded, DefaultManagement should do this, but refactor up as part of FCREPO-718)
215         if (oldDSLocation != null && oldDSLocation.startsWith(TEMP_SCHEME)) {
216             File oldFile;
217             try {
218                 oldFile = new File(oldDSLocation.substring(TEMP_SCHEME.length()));
219             } catch (Exception e) {
220                 throw new StreamIOException("Error removing old temp file while updating managed content (location: " + oldDSLocation + ")", e);
221             }
222             if (oldFile.exists()) {
223                 if (!oldFile.delete()) {
224                     logger.warn("Failed to delete temp file, marked for deletion when VM closes " + oldFile.getAbsolutePath());
225                     oldFile.deleteOnExit();
226                 }
227             } else
228                 logger.warn("Cannot delete temp file as it no longer exists " + oldFile.getAbsolutePath());
229         }
230     }
