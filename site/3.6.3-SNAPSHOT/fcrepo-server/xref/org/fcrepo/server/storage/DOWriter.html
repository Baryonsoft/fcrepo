5   package org.fcrepo.server.storage;

7   import java.util.Date;

9   import org.fcrepo.server.errors.ServerException;
10  import org.fcrepo.server.storage.types.Datastream;

13  /**
14   * The standard interface for write operations on a digital object.
15   *
16   * <p>A <code>DOWriter</code> instance is a handle on a Fedora digital object,
17   * and is obtained via a <code>getWriter(String)</code> call on a
18   * <code>DOManager</code>.
19   *
20   * <p>Work with a DOWriter ends with either commit() or cancel().
21   *
22   * @author Chris Wilper
23   */
24  public interface DOWriter
25          extends DOReader {
26  
27      /**
28       * Sets the state of the entire digital object.
29       *
30       * @param state
31       *        The state.
32       * @throws ServerException
33       *         If any type of error occurred fulfilling the request.
34       */
35      public void setState(String state) throws ServerException;
36  
37      /**
38       * Sets the ownerId for the digital object.
39       *
40       * @param ownerId
41       *        The ownerId.
42       * @throws ServerException
43       *         If any type of error occurred fulfilling the request.
44       */
45      public void setOwnerId(String ownerId) throws ServerException;
46  
47      /**
48       * Sets the state for all versions of the specified datastream.
49       *
50       * @param id
51       *        The datastream id.
52       * @param state
53       *        The state.
54       * @throws ServerException
55       *         If any type of error occurred fulfilling the request.
56       */
57      public void setDatastreamState(String id, String state)
58              throws ServerException;
59  
60      /**
61       * Sets the indicator as to whether the datastream is subject to versioning.
62       * In Fedora 2.0, the system will not operate on this indicator and all
63       * datastreams will be versioned by default.
64       *
65       * @param id
66       *        The datastream id.
67       * @param versionable
68       *        a boolean indicating if versionable
69       * @throws ServerException
70       *         If any type of error occurred fulfilling the request.
71       */
72      public void setDatastreamVersionable(String id, boolean versionable)
73              throws ServerException;
74  
75      /**
76       * Sets the label of the digital object.
77       *
78       * @param label
79       *        The label.
80       * @throws ServerException
81       *         If any type of error occurred fulfilling the request.
82       */
83      public void setLabel(String label) throws ServerException;
84  
85      /**
86       * Removes the entire digital object.
87       *
88       * @throws ServerException
89       *         If any type of error occurred fulfilling the request.
90       */
91      public void remove() throws ServerException;
92  
93      /**
94       * Adds a datastream to the object.
95       *
96       * @param datastream
97       *        The datastream.
98       * @throws ServerException
99       *         If any type of error occurred fulfilling the request.
100      */
101     public void addDatastream(Datastream datastream, boolean addNewVersion)
102             throws ServerException;
103 
104     /**
105      * Removes a range of datastream versions from an object without leaving
106      * anything behind. If any integrity checks need to be done, they should be
107      * done outside of this code.
108      *
109      * @param id
110      *        The id of the datastream.
111      * @param start
112      *        The start date (inclusive) of versions to remove. If
113      *        <code>null</code>, this is taken to be the smallest possible
114      *        value.
115      * @param end
116      *        The end date (inclusive) of versions to remove. If
117      *        <code>null</code>, this is taken to be the greatest possible
118      *        value.
119      * @throws ServerException
120      *         If any type of error occurred fulfilling the request.
121      */
122     public Date[] removeDatastream(String id, Date start, Date end)
123             throws ServerException;
124 
125     /**
126      * Saves the changes thus far to the permanent copy of the digital object.
127      *
128      * @param logMessage
129      *        An explanation of the change(s).
130      * @throws ServerException
131      *         If any type of error occurred fulfilling the request.
132      */
133     public void commit(String logMessage) throws ServerException;
134 
135     /**
136      * Marks this DOWriter handle invalid (unusable).
137      */
138     public void invalidate();
139 
140     /**
141      * Generate a unique id for a datastream.
142      */
143     public String newDatastreamID();
144 
145     /**
146      * Generate a unique id for a datastream version.
147      */
148     public String newDatastreamID(String dsID);
149 
150     /**
151      * Generate a unique id for an audit record.
152      */
153     public String newAuditRecordID();
154 
155     /**
156      * Marks whether the object has been successfully committed.
157      */
158     public boolean isCommitted();
159 
160     /**
161      * Marks whether the object is new.
162      */
163     public boolean isNew();
164 
165     /**
166      * Adds a RDF triple to a reserved datastream (based on the subject)
167      */
168     public boolean addRelationship(String subject,
169                                    String relationship,
170                                    String object,
171                                    boolean isLiteral,
172                                    String datatype) throws ServerException;
173     /**
174      * Adds a RDF triple to the specified (RDF) datastream
175      */
176     public boolean addRelationship(String dsId,
177                                    String subject,
178                                    String relationship,
179                                    String object,
180                                    boolean isLiteral,
181                                    String datatype) throws ServerException;
182 
183     /**
184      * Purges a RDF triple from a reserved datastream (based on the subject)
185      */
186     public boolean purgeRelationship(String subject,
187                                      String relationship,
188                                      String object,
189                                      boolean isLiteral,
190                                      String datatype) throws ServerException;
191     /**
192      * Purges a RDF triple from the specified (RDF) datastream
193      */
194     public boolean purgeRelationship(String dsId,
195                                      String subject,
196                                      String relationship,
197                                      String object,
198                                      boolean isLiteral,
199                                      String datatype) throws ServerException;
200 
201 }
