99      /**
100      * Initializes the Module based on configuration parameters. The
101      * implementation of this method is dependent on the schema used to define
102      * the parameter names for the role of
103      * <code>org.fcrepo.server.storage.ConnectionPoolManager</code>.
104      *
105      * @throws ModuleInitializationException
106      *         If initialization values are invalid or initialization fails for
107      *         some other reason.
108      */
109     @Override
110     public void initModule() throws ModuleInitializationException {
111         try {
112             Server s_server = getServer();
113             defaultPoolName = this.getParameter("defaultPoolName");
114             if (defaultPoolName == null || defaultPoolName.equalsIgnoreCase("")) {
115                 throw new ModuleInitializationException("Default Connection Pool "
116                                                                 + "Name Not Specified",
117                                                         getRole());
118             }
119             logger.debug("DefaultPoolName: " + defaultPoolName);
120             String poolList = this.getParameter("poolNames");
121 
122             // Pool names should be comma delimited
123             String[] poolNames = poolList.split(",");
124 
125             // Initialize each connection pool
126             for (int i = 0; i < poolNames.length; i++) {
127                 DatastoreConfig config =
128                         s_server.getDatastoreConfig(poolNames[i]);
129                 jdbcDriverClass = config.getParameter("jdbcDriverClass");
130                 dbUsername = config.getParameter("dbUsername");
131                 dbPassword = config.getParameter("dbPassword");
132                 jdbcURL = config.getParameter("jdbcURL");
133                 maxActive =
134                         new Integer(config.getParameter("maxActive"))
135                                 .intValue();
136                 maxIdle =
137                         new Integer(config.getParameter("maxIdle")).intValue();
138                 maxWait =
139                         new Integer(config.getParameter("maxWait")).intValue();
140                 minIdle =
141                         new Integer(config.getParameter("minIdle")).intValue();
142                 numTestsPerEvictionRun =
143                         new Integer(config
144                                 .getParameter("numTestsPerEvictionRun"))
145                                 .intValue();
146                 minEvictableIdleTimeMillis =
147                         new Long(config
148                                 .getParameter("minEvictableIdleTimeMillis"))
149                                 .longValue();
150                 timeBetweenEvictionRunsMillis =
151                         new Long(config
152                                 .getParameter("timeBetweenEvictionRunsMillis"))
153                                 .longValue();
154                 validationQuery = config.getParameter("validationQuery");
155                 testOnBorrow =
156                         new Boolean(config.getParameter("testOnBorrow"))
157                                 .booleanValue();
158                 testOnReturn =
159                         new Boolean(config.getParameter("testOnReturn"))
160                                 .booleanValue();
161                 testWhileIdle =
162                         new Boolean(config.getParameter("testWhileIdle"))
163                                 .booleanValue();
164                 whenExhaustedAction =
165                         new Byte(config.getParameter("whenExhaustedAction"))
166                                 .byteValue();
167                 if (whenExhaustedAction != 0 && whenExhaustedAction != 1
168                         && whenExhaustedAction != 2) {
169                     logger.debug("Valid values for whenExhaustedAction are: 0 - (fail), 1 - (block), or 2 - (grow)");
170                     throw new ModuleInitializationException("A connection pool could "
171                             + "not be instantiated. The underlying error was an "
172                             + "invalid value for the whenExhaustedAction parameter."
173                             + "Valid values are 0 - (fail), 1 - (block), or 2 - (grow). Value specified"
174                             + "was \"" + whenExhaustedAction + "\".", getRole());
175                 }
176 
177                 if (logger.isDebugEnabled()) {
178                     logger.debug("poolName[" + i + "] = " + poolNames[i]);
179                     logger.debug("JDBC driver: " + jdbcDriverClass);
180                     logger.debug("Database username: " + dbUsername);
181                     logger.debug("Database password: " + dbPassword);
182                     logger.debug("JDBC connection URL: " + jdbcURL);
183                     logger.debug("Maximum active connections: " + maxActive);
184                     logger.debug("Maximum idle connections: " + maxIdle);
185                     logger.debug("Maximum wait time: " + maxWait);
186                     logger.debug("Minimum idle time: " + minIdle);
187                     logger.debug("Number of tests per eviction run: "
188                             + numTestsPerEvictionRun);
189                     logger.debug("Minimum Evictable Idle time: "
190                             + minEvictableIdleTimeMillis);
191                     logger.debug("Minimum Evictable Idle time: "
192                             + timeBetweenEvictionRunsMillis);
193                     logger.debug("Validation query: " + validationQuery);
194                     logger.debug("Test on borrow: " + testOnBorrow);
195                     logger.debug("Test on return: " + testOnReturn);
196                     logger.debug("Test while idle: " + testWhileIdle);
197                     logger.debug("whenExhaustedAction: " + whenExhaustedAction);
198                 }
199 
200                 // Treat any parameters whose names start with "connection."
201                 // as connection parameters
202                 Map<String, String> cProps = new HashMap<String, String>();
203                 for (String name : config.getParameters().keySet()) {
204                     if (name.startsWith("connection.")) {
205                         String realName = name.substring(11);
206                         logger.debug("Connection property " + realName + " = "
207                                 + config.getParameter(name));
208                         cProps.put(realName, config.getParameter(name));
209                     }
210                 }
211 
212                 // If a ddlConverter has been specified for the pool,
213                 // try to instantiate it so the ConnectionPool can use
214                 // it when it provides a TableCreatingConnection.
215                 // If a ddlConverter has been specified, it is assumed
216                 // that a failure to initialize (construct) it should
217                 // trigger a ModuleInitializationException (a fatal startup error).
218                 DDLConverter ddlConverter = null;
219                 String ddlConverterClassName =
220                         getServer().getDatastoreConfig(poolNames[i])
221                                 .getParameter("ddlConverter");
222                 if (ddlConverterClassName != null) {
223                     try {
224                         ddlConverter =
225                                 (DDLConverter) Class
226                                         .forName(ddlConverterClassName)
227                                         .newInstance();
228                     } catch (Throwable th) {
229                         throw new ModuleInitializationException("A DDLConverter was "
230                                                                         + "specified for the pool \""
231                                                                         + poolNames[i]
232                                                                         + "\", but it couldn't be instantiated.",
233                                                                 getRole(),
234                                                                 th);
235                     }
236                 }
237 
238                 // Create connection pool
239                 try {
240                     ConnectionPool connectionPool =
241                             new ConnectionPool(jdbcDriverClass,
242                                                jdbcURL,
243                                                dbUsername,
244                                                dbPassword,
245                                                ddlConverter,
246                                                maxActive,
247                                                maxIdle,
248                                                maxWait,
249                                                minIdle,
250                                                minEvictableIdleTimeMillis,
251                                                numTestsPerEvictionRun,
252                                                timeBetweenEvictionRunsMillis,
253                                                validationQuery,
254                                                testOnBorrow,
255                                                testOnReturn,
256                                                testWhileIdle,
257                                                whenExhaustedAction);
258                     connectionPool.setConnectionProperties(cProps);
259                     logger.debug("Initialized Pool: " + connectionPool);
260                     h_ConnectionPools.put(poolNames[i], connectionPool);
261                     logger.debug("putPoolInHash: " + h_ConnectionPools.size());
262                 } catch (SQLException sqle) {
263                     logger.error("Unable to initialize connection pool: "
264                             + poolNames[i] + ": " + sqle.getMessage());
265                 }
266             }
267 
268         } catch (Throwable th) {
269             th.printStackTrace();
270             throw new ModuleInitializationException("A connection pool could "
271                     + "not be instantiated. The underlying error was a "
272                     + th.getClass().getName() + "The message was \""
273                     + th.getMessage() + "\".", getRole());
274         }
275     }

277     /**
278      * <p>
279      * Gets a named connection pool.
280      * </p>
281      *
282      * @param poolName
283      *        The name of the connection pool.
284      * @return The named connection pool.
285      * @throws ConnectionPoolNotFoundException
286      *         If the specified connection pool cannot be found.
287      */
288     public ConnectionPool getPool(String poolName)
289             throws ConnectionPoolNotFoundException {
290         ConnectionPool connectionPool = null;
291 
292         try {
293             if (h_ConnectionPools.containsKey(poolName)) {
294                 connectionPool = h_ConnectionPools.get(poolName);
295             } else {
296                 // Error: pool was never initialized or name could not be found
297                 throw new ConnectionPoolNotFoundException("Connection pool "
298                         + "not found: " + poolName);
299             }
300         } catch (Throwable th) {
301             throw new ConnectionPoolNotFoundException("The specified connection "
302                     + "pool \""
303                     + poolName
304                     + "\" could not be found. The underlying "
305                     + "error was a "
306                     + th.getClass().getName()
307                     + "The message was \""
308                     + th.getMessage() + "\".");
309         }
310 
311         return connectionPool;
312     }

314     /**
315      * <p>
316      * Gets the default Connection Pool. This method overrides <code>
317      * getPool(String poolName)</code>
318      * .
319      * </p>
320      *
321      * @return The default connection pool.
322      * @throws ConnectionPoolNotFoundException
323      *         If the default connection pool cannot be found.
324      */
325     public ConnectionPool getPool() throws ConnectionPoolNotFoundException {
326         ConnectionPool connectionPool = null;
327 
328         try {
329             if (h_ConnectionPools.containsKey(defaultPoolName)) {
330                 connectionPool = h_ConnectionPools.get(defaultPoolName);
331             } else {
332                 // Error: default pool was never initialized or could not be found
333                 throw new ConnectionPoolNotFoundException("Default connection pool "
334                         + "not found: " + defaultPoolName);
335             }
336 
337         } catch (Throwable th) {
338             throw new ConnectionPoolNotFoundException("The default connection "
339                     + "pool \"" + defaultPoolName
340                     + "\" could not be found. The " + "underlying error was a "
341                     + th.getClass().getName() + "The message was \""
342                     + th.getMessage() + "\".");
343         }
344 
345         return connectionPool;
346     }

348     /**
349      * <p>
350      * Closes all connection pools. This method overrides
351      * <code> shutdownModule()</code> .
352      * </p>
353      *
354      * @throws ModuleShutdownException
355      *         If the close operation for the connection pool(s) fails.

357     @Override
358     public void shutdownModule() throws ModuleShutdownException {
359         super.shutdownModule();
360 
361         for (Map.Entry<String, ConnectionPool> e : h_ConnectionPools.entrySet()) {
362             e.getValue().close();
363         }
