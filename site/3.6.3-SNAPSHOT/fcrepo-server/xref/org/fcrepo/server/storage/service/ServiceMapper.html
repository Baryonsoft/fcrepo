30  /**
31   * Controller class for parsing the various kinds of inline metadata datastreams
32   * found in service objects. The intent of this class is to initiate parsing of
33   * these datastreams so that information about a service can be
34   * instantiated in Fedora.
35   * </p>

50      public static void main(String[] args) {
51          if (args.length < 2) {
52              System.err
53                      .println("usage: java ServiceMapper wsdlLocation methodMapLocation "
54                              + "\n"
55                              + "  wsdlLocation: the file path of the wsdl to be parsed"
56                              + "\n"
57                              + "  methodMapLocation: the file path of the method map to be parsed."
58                              + "\n"
59                              + "  dsInputSpecLocation: the file path of the datastream input spec to be parsed."
60                              + "  pid: the PID of the sDef or sDep object for the above files.");
61              System.exit(1);
62          }
63          try {
64              ServiceMapper mapper = new ServiceMapper(args[3]);
65              InputSource wsdl =
66                      new InputSource(new FileInputStream(new File(args[0])));
67              InputSource mmap =
68                      new InputSource(new FileInputStream(new File(args[1])));
69              InputSource dsSpec =
70                      new InputSource(new FileInputStream(new File(args[2])));
71              mapper.getMethodDefs(mmap);
72              mapper.getMethodDefBindings(wsdl, mmap);
73              mapper.getDSInputSpec(dsSpec);
74          } catch (Exception e) {
75              e.printStackTrace();
76          }
77          System.exit(1);
78  
79      }

85      /**
86       * getMethodDefs: creates an array of abstract method definitions in the
87       * form of an array of Fedora MethodDef objects. The creation of a MethodDef
88       * object requires information from a Fedora Method Map.
89       * 
90       * @param methodMapSource :
91       *        Fedora Method Map definition for methods
92       * @return MethodDef[] : an array of abstract method definitions
93       * @throws ObjectIntegrityException
94       * @throws RepositoryConfigurationException
95       * @throws GeneralException
96       */
97      public MethodDef[] getMethodDefs(InputSource methodMapSource)
98              throws ObjectIntegrityException, RepositoryConfigurationException,
99              GeneralException {
100         Mmap methodMap = getMethodMap(methodMapSource);
101         return methodMap.mmapMethods;
102     }

104     /**
105      * getMethodDefBindings: creates an array of operation bindings in the form
106      * of an array of Fedora MethodDefOperationBind objects. The creation of a
107      * MethodDefOperationBind object requires information from a WSDL service
108      * definition and a related Fedora Method Map. The Fedora Method Map is
109      * merged with the WSDL to provide a Fedora-specific view of the WSDL.
110      * 
111      * @param wsdlSource :
112      *        WSDL service definition for methods
113      * @param methodMapSource :
114      *        Fedora Method Map definition for methods
115      * @return MethodDefOperationBind[] : an array of method bindings
116      * @throws ObjectIntegrityException
117      * @throws RepositoryConfigurationException
118      * @throws GeneralException
119      */
120     public MethodDefOperationBind[] getMethodDefBindings(InputSource wsdlSource,
121                                                          InputSource methodMapSource)
122             throws ObjectIntegrityException, RepositoryConfigurationException,
123             GeneralException {
124         return merge(getService(wsdlSource), getMethodMap(methodMapSource));
125     }

149     private Service getService(InputSource wsdlSource)
150             throws ObjectIntegrityException, RepositoryConfigurationException,
151             GeneralException {
152         if (wsdlHandler == null) {
153             wsdlHandler = (WSDLParser) parse(wsdlSource, new WSDLParser());

158     private DefaultHandler parse(InputSource xml, DefaultHandler eventHandler)
159             throws ObjectIntegrityException, RepositoryConfigurationException,
160             GeneralException {
161         try {
162             // XMLSchema validation via SAX parser
163             SAXParserFactory spf = SAXParserFactory.newInstance();
164             spf.setNamespaceAware(true);
165             spf.setValidating(false);
166             SAXParser sp = spf.newSAXParser();
167             DefaultHandler handler = eventHandler;
168             XMLReader xmlreader = sp.getXMLReader();
169             xmlreader.setContentHandler(handler);
170             xmlreader.parse(xml);
171             return handler;
172         } catch (ParserConfigurationException e) {
173             String msg =
174                     "ServiceMapper returned parser error. "
175                             + "The underlying exception was a "
176                             + e.getClass().getName() + ".  "
177                             + "The message was " + "\"" + e.getMessage() + "\"";
178             throw new RepositoryConfigurationException(msg);
179         } catch (SAXException e) {
180             String msg =
181                     "ServiceMapper returned SAXException. "
182                             + "The underlying exception was a "
183                             + e.getClass().getName() + ".  "
184                             + "The message was " + "\"" + e.getMessage() + "\"";
185             throw new ObjectIntegrityException(msg);
186         } catch (Exception e) {
187             String msg =
188                     "ServiceMapper returned error. "
189                             + "The underlying error was a "
190                             + e.getClass().getName() + ".  "
191                             + "The message was " + "\"" + e.getMessage() + "\"";
192             e.printStackTrace();
193             throw new GeneralException(msg);
194         }
195     }

197     private MethodDefOperationBind[] merge(Service service, Mmap methodMap)
198             throws ObjectIntegrityException, GeneralException {
199         Port port = null;
200         Binding binding = null;
201         MethodDefOperationBind[] fedoraMethodDefBindings =
202                 new MethodDefOperationBind[0];
203 
204         // If the WSDL Service defines multiple Ports (with Binding) for the abstract operations
205         // we must CHOOSE ONE binding for Fedora to work with.
206         if (service.ports.length > 1) {
207             port = choosePort(service);
208         } else {
209             port = service.ports[0];
210         }
211 
212         binding = port.binding;
213 
214         // Reflect on the type of binding we are dealing with, then Fedora-ize things.
215         if (binding.getClass().getName()
216                 .equalsIgnoreCase("org.fcrepo.server.storage.service.HTTPBinding")) {
217             // Initialize the array to hold the Fedora-ized operation bindings.
218             fedoraMethodDefBindings =
219                     new MethodDefOperationBind[((HTTPBinding) binding).operations.length];
220 
221             for (int i = 0; i < ((HTTPBinding) binding).operations.length; i++) {
222                 // From methodMap which was previously created by parsing Fedora method map metadata
223                 // which provides a Fedora overlay on the service WSDL.
224                 MmapMethodDef methodDef =
225                         (MmapMethodDef) methodMap.wsdlOperationToMethodDef
226                                 .get(((HTTPBinding) binding).operations[i].operationName);
227                 fedoraMethodDefBindings[i] = new MethodDefOperationBind();
228                 fedoraMethodDefBindings[i].methodName = methodDef.methodName;
229                 fedoraMethodDefBindings[i].methodLabel = methodDef.methodLabel;
230                 fedoraMethodDefBindings[i].methodParms = methodDef.methodParms;
231 
232                 // From WSDL Port found in Service object
233                 fedoraMethodDefBindings[i].serviceBindingAddress =
234                         port.portBaseURL;
235 
236                 // From WSDL Binding found in Service object
237                 fedoraMethodDefBindings[i].protocolType =
238                         MethodDefOperationBind.HTTP_MESSAGE_PROTOCOL;
239                 fedoraMethodDefBindings[i].operationLocation =
240                         ((HTTPBinding) binding).operations[i].operationLocation;
241                 fedoraMethodDefBindings[i].operationURL =
242                         fedoraMethodDefBindings[i].serviceBindingAddress
243                                 .concat(fedoraMethodDefBindings[i].operationLocation);
244 
245                 // Get the list of datastream input binding keys that pertain to the particular
246                 // operation binding.  From the WSDL perspective, the datastream input keys
247                 // are WSDL message parts that, according the the Fedora Method Map, are
248                 // datastream inputs to the operation.
249 
250                 MmapMethodParmDef[] parms = methodDef.wsdlMsgParts;
251                 Vector<String> tmp_dsInputKeys = new Vector<String>();
252                 for (MmapMethodParmDef element : parms) {
253                     if (element.parmType
254                             .equalsIgnoreCase(MethodParmDef.DATASTREAM_INPUT)) {
255                         tmp_dsInputKeys.add(element.parmName);
256                     }
257                 }
258                 fedoraMethodDefBindings[i].dsBindingKeys =
259                         (String[]) tmp_dsInputKeys.toArray(new String[0]);
260 
261                 // Set the outputMIMETypes from the operation's output binding, if any
262                 HTTPOperationInOut oBind =
263                         ((HTTPBinding) binding).operations[i].outputBinding;
264                 if (oBind != null) {
265                     Vector<String> tmp_outputMIMETypes = new Vector<String>();
266                     for (MIMEContent element : oBind.ioMIMEContent) {
267                         tmp_outputMIMETypes.add(element.mimeType);
268                     }
269                     fedoraMethodDefBindings[i].outputMIMETypes =
270                             (String[]) tmp_outputMIMETypes
271                                     .toArray(new String[0]);
272                 }
273 
274             }
275         } else if (binding.getClass().getName()
276                 .equalsIgnoreCase("org.fcrepo.server.storage.service.SOAPBinding")) {
277             // FIXIT!!  Implement this!
278         }
279         return fedoraMethodDefBindings;
280     }

282     private Port choosePort(Service service) {
283         // If there is an HTTP binding, this will be preferred.
284         for (Port element : service.ports) {
285             Binding binding = element.binding;
286             if (binding
287                     .getClass()
288                     .getName()
289                     .equalsIgnoreCase("org.fcrepo.server.storage.service.HTTPBinding")) {
290                 return element;
291             }
292         }
293         // Otherwise, just return the first port for binding
294         return service.ports[0];
295     }
