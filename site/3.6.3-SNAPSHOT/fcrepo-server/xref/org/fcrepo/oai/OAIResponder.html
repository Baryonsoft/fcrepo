52      public void respond(Context context, Map args, OutputStream outStream)
53              throws RepositoryException, AuthzException {
54          if (m_authorization != null) {
55              m_authorization.enforceOAIRespond(context);
56          } else {
57              logger.warn("Serving OAIResponses without Authorization module");
58          }
59          m_granularity = m_provider.getDateGranularitySupport();
60          PrintWriter out = null;
61          try {
62              out =
63                      new PrintWriter(new BufferedWriter(new OutputStreamWriter(outStream,
64                                                                                "UTF-8")));
65          } catch (UnsupportedEncodingException uee) {
66              // not going to happen... all java impls support UTF-8
67          }
68          String verb = (String) args.get("verb");
69  
70          String baseURL =
71                  m_provider
72                          .getBaseURL(context
73                                              .getEnvironmentValue(HTTP_REQUEST.SECURITY.uri)
74                                              .equals(HTTP_REQUEST.SECURE.uri) ? "https"
75                                              : "http",
76                                      context
77                                              .getEnvironmentValue(HTTP_REQUEST.SERVER_PORT.uri));
78          try {
79              if (verb == null) {
80                  throw new BadVerbException("Request did not specify a verb.");
81              }
82              if (verb.equals("GetRecord")) {
83                  String identifier = (String) args.get("identifier");
84                  if (identifier == null) {
85                      throw new BadArgumentException("GetRecord request did not specify 'identifier' argument.");
86                  }
87                  String metadataPrefix = (String) args.get("metadataPrefix");
88                  if (metadataPrefix == null) {
89                      throw new BadArgumentException("GetRecord request did not specify 'metadataPrefix' argument.");
90                  }
91                  if (args.size() > 3) {
92                      throw new BadArgumentException("GetRecord request specified illegal argument(s).");
93                  }
94                  Record record =
95                          m_provider.getRecord(identifier, metadataPrefix);
96                  respondToGetRecord(args, baseURL, record, out);
97              } else if (verb.equals("Identify")) {
98                  if (args.size() > 1) {
99                      throw new BadArgumentException("Identify request specified illegal argument(s).");
100                 }
101                 String repositoryName = m_provider.getRepositoryName();
102                 String protocolVersion = m_provider.getProtocolVersion();
103                 Date earliestDatestamp = m_provider.getEarliestDatestamp();
104                 DeletedRecordSupport deletedRecord =
105                         m_provider.getDeletedRecordSupport();
106                 Set adminEmails = m_provider.getAdminEmails();
107                 Set compressions =
108                         m_provider.getSupportedCompressionEncodings();
109                 Set descriptions = m_provider.getDescriptions();
110                 respondToIdentify(args,
111                                   baseURL,
112                                   repositoryName,
113                                   protocolVersion,
114                                   earliestDatestamp,
115                                   deletedRecord,
116                                   adminEmails,
117                                   compressions,
118                                   descriptions,
119                                   out);
120             } else if (verb.equals("ListIdentifiers")) {
121                 String rToken = (String) args.get("resumptionToken");
122                 List headers;
123                 if (rToken != null) {
124                     if (args.size() > 2) {
125                         throw new BadArgumentException("ListIdentifiers request specified resumptionToken with other arguments.");
126                     }
127                     headers = m_provider.getHeaders(rToken);
128                 } else {
129                     Iterator iter = args.keySet().iterator();
130                     boolean badParam = false;
131                     Date from = null;
132                     Date until = null;
133                     String metadataPrefix = null;
134                     String set = null;
135                     while (iter.hasNext()) {
136                         String name = (String) iter.next();
137                         if (name.equals("metadataPrefix")) {
138                             metadataPrefix = (String) args.get(name);
139                         } else if (name.equals("set")) {
140                             set = (String) args.get(name);
141                         } else if (name.equals("from")) {
142                             from = getUTCDate((String) args.get(name), false);
143                         } else if (name.equals("until")) {
144                             until = getUTCDate((String) args.get(name), true);
145                         } else if (!name.equals("verb")) {
146                             badParam = true;
147                         }
148                     }
149                     if (from != null && until != null) {
150                         assertSameGranularity((String) args.get("from"),
151                                               (String) args.get("until"));
152                     }
153                     if (badParam) {
154                         throw new BadArgumentException("ListIdentifiers request specified illegal argument(s).");
155                     }
156                     if (metadataPrefix == null) {
157                         throw new BadArgumentException("ListIdentifiers request did not specify metadataPrefix argument.");
158                     }
159                     headers =
160                             m_provider.getHeaders(from,
161                                                   until,
162                                                   metadataPrefix,
163                                                   set);
164                 }
165                 if (headers.size() == 0) {
166                     throw new NoRecordsMatchException("No records match the providied criteria.");
167                 }
168                 ResumptionToken resumptionToken = null;
169                 if (m_provider.getMaxHeaders() > 0) {
170                     if (headers.size() > m_provider.getMaxHeaders()) {
171                         resumptionToken =
172                                 (ResumptionToken) headers
173                                         .get(headers.size() - 1);
174                         headers = headers.subList(0, headers.size() - 1);
175                     }
176                 }
177                 respondToListIdentifiers(args,
178                                          baseURL,
179                                          headers,
180                                          resumptionToken,
181                                          out);
182             } else if (verb.equals("ListMetadataFormats")) {
183                 String identifier = (String) args.get("identifier");
184                 if (identifier == null) {
185                     if (args.size() > 1) {
186                         throw new BadArgumentException("ListMetadataFormats request specified illegal argument(s).");
187                     }
188                 } else {
189                     if (args.size() > 2) {
190                         throw new BadArgumentException("ListMetadataFormats request specified illegal argument(s).");
191                     }
192                 }
193                 respondToListMetadataFormats(args, baseURL, m_provider
194                         .getMetadataFormats(identifier), out);
195             } else if (verb.equals("ListRecords")) {
196                 String rToken = (String) args.get("resumptionToken");
197                 List records;
198                 if (rToken != null) {
199                     if (args.size() > 2) {
200                         throw new BadArgumentException("ListRecords request specified resumptionToken with other arguments.");
201                     }
202                     records = m_provider.getRecords(rToken);
203                 } else {
204                     Iterator iter = args.keySet().iterator();
205                     boolean badParam = false;
206                     Date from = null;
207                     Date until = null;
208                     String metadataPrefix = null;
209                     String set = null;
210                     while (iter.hasNext()) {
211                         String name = (String) iter.next();
212                         if (name.equals("metadataPrefix")) {
213                             metadataPrefix = (String) args.get(name);
214                         } else if (name.equals("set")) {
215                             set = (String) args.get(name);
216                         } else if (name.equals("from")) {
217                             from = getUTCDate((String) args.get(name), false);
218                         } else if (name.equals("until")) {
219                             until = getUTCDate((String) args.get(name), true);
220                         } else if (!name.equals("verb")) {
221                             badParam = true;
222                         }
223                     }
224                     if (from != null && until != null) {
225                         assertSameGranularity((String) args.get("from"),
226                                               (String) args.get("until"));
227                     }
228                     if (badParam) {
229                         throw new BadArgumentException("ListRecords request specified illegal argument(s).");
230                     }
231                     if (metadataPrefix == null) {
232                         throw new BadArgumentException("ListRecords request did not specify metadataPrefix argument.");
233                     }
234                     records =
235                             m_provider.getRecords(from,
236                                                   until,
237                                                   metadataPrefix,
238                                                   set);
239                 }
240                 if (records.size() == 0) {
241                     throw new NoRecordsMatchException("No records match the providied criteria.");
242                 }
243                 ResumptionToken resumptionToken = null;
244                 if (m_provider.getMaxRecords() > 0) {
245                     if (records.size() > m_provider.getMaxRecords()) {
246                         resumptionToken =
247                                 (ResumptionToken) records
248                                         .get(records.size() - 1);
249                         records = records.subList(0, records.size() - 1);
250                     }
251                 }
252                 respondToListRecords(args,
253                                      baseURL,
254                                      records,
255                                      resumptionToken,
256                                      out);
257             } else if (verb.equals("ListSets")) {
258                 String rToken = (String) args.get("resumptionToken");
259                 List sets;
260                 if (rToken == null) {
261                     if (args.size() > 1) {
262                         throw new BadArgumentException("ListSets request specified illegal argument(s).");
263                     }
264                     sets = m_provider.getSets();
265                 } else {
266                     if (args.size() > 2) {
267                         throw new BadArgumentException("ListSets request specified illegal argument(s).");
268                     }
269                     sets = m_provider.getSets(rToken);
270                 }
271                 ResumptionToken resumptionToken = null;
272                 if (m_provider.getMaxSets() > 0) {
273                     if (sets.size() > m_provider.getMaxSets()) {
274                         resumptionToken =
275                                 (ResumptionToken) sets.get(sets.size() - 1);
276                         sets = sets.subList(0, sets.size() - 1);
277                     }
278                 }
279                 respondToListSets(args, baseURL, sets, resumptionToken, out);
280             } else {
281                 throw new BadVerbException("Unrecognized verb, '" + verb + "'.");
282             }
283         } catch (OAIException oaie) {
284             respondWithError(oaie, verb, baseURL, out);
285         } finally {
286             out.flush();
287         }
288     }

290     private void assertSameGranularity(String from, String until)
291             throws BadArgumentException {
292         if (from.endsWith("Z") && !until.endsWith("Z") || until.endsWith("Z")
293                 && !from.endsWith("Z")) {
294             throw new BadArgumentException("Date granularities of from and until arguments do not match.");
295         }
296     }

445         Set abouts = record.getAbouts();
446         out.println(indent + "<record>");

454                 out.println(indent + "  <about>");

462     private void appendHeader(String indent, Header header, PrintWriter out) {
463         String identifier = header.getIdentifier();
464         Date datestamp = header.getDatestamp();
465         Set setSpecs = header.getSetSpecs();
466         boolean isAvailable = header.isAvailable();
467         out.print(indent + "<header");

469             out.print(" status=\"deleted\"");

471         out.println(">");
472         out.println(indent + "  <identifier>" + OAIResponder.enc(identifier)
473                 + "</identifier>");
474         out.println(indent
475                 + "  <datestamp>"
476                 + getUTCString(datestamp,
477                                m_granularity == DateGranularitySupport.SECONDS)
478                 + "</datestamp>");
479         Iterator iter = setSpecs.iterator();
480         while (iter.hasNext()) {
481             out.println(indent + "  <setSpec>"
482                     + OAIResponder.enc((String) iter.next()) + "</setSpec>");
483         }
484         out.println(indent + "</header>");
485     }

536                                   String verb,
537                                   String baseURL,
538                                   PrintWriter out) {
539         appendTop(out);

554     private static String getUTCString(Date utcDate, boolean fineGranularity) {
555         SimpleDateFormat formatter;
556         if (fineGranularity) {
557             formatter = new SimpleDateFormat("yyyy-MM-dd'T'HH:mm:ss'Z'");
558         } else {
559             formatter = new SimpleDateFormat("yyyy-MM-dd");
560         }
561         return formatter.format(utcDate);
562     }

564     private Date getUTCDate(String formattedDate, boolean isUntil)
565             throws BadArgumentException {
566         if (formattedDate == null) {
567             return null;
568         }
569         try {
570             if (formattedDate.endsWith("Z")) {
571                 if (m_granularity == DateGranularitySupport.SECONDS) {
572                     SimpleDateFormat formatter =
573                             new SimpleDateFormat("yyyy-MM-dd'T'HH:mm:ss'Z'");
574                     return formatter.parse(formattedDate);
575                 } else {
576                     throw new BadArgumentException("Repository does not support that granularity.");
577                 }
578             } else {
579                 SimpleDateFormat formatter = new SimpleDateFormat("yyyy-MM-dd");
580                 if (isUntil) {
581                     SimpleDateFormat sdf =
582                             new SimpleDateFormat("yyyy-MM-dd'T'HH:mm:ss'Z'");
583                     return sdf.parse(formatter.format(formatter
584                             .parse(formattedDate))
585                             + "T23:59:59Z");
586                 } else {
587                     return formatter.parse(formattedDate);
588                 }
589             }
590         } catch (ParseException pe) {
591             throw new BadArgumentException("Error parsing date.");
592         }
593     }

595     private Date getUTCDate(Date localDate) {
596         Calendar cal = Calendar.getInstance();
597         int tzOffset = cal.get(Calendar.ZONE_OFFSET);
598         TimeZone tz = cal.getTimeZone();
599         if (tz.inDaylightTime(localDate)) {
600             tzOffset += cal.get(Calendar.DST_OFFSET);
601         }
602         Date UTCDate = new Date();
603         UTCDate.setTime(localDate.getTime() + tzOffset);
604         return UTCDate;
605     }

654     /**
655      * Appends an XML-appropriate encoding of the given character to the given
656      * StringBuffer.
657      *
658      * @param in
659      *        The character.
660      * @param out
661      *        The StringBuffer to write to.
662      */
663     private static void enc(char in, StringBuffer out) {
664         if (in == '&') {
665             out.append("&amp;");
666         } else if (in == '<') {
667             out.append("&lt;");
668         } else if (in == '>') {
669             out.append("&gt;");
670         } else if (in == '\"') {
671             out.append("&quot;");
672         } else if (in == '\'') {
673             out.append("&apos;");
674         } else {
675             out.append(in);
676         }
677     }
