73      @Test
74      public void testInlineXMLEncoding() throws Exception {
75          final String TAG = "test";
76          final String OPEN = "<" + TAG + ">";
77          final String CLOSE = "</" + TAG + ">";
78          char[] unicodeContent = new char[1365];
79  
80          StringBuilder payload =
81                  new StringBuilder(unicodeContent.length + OPEN.length()
82                          + CLOSE.length());
83  
84          for (int i = 0; i < unicodeContent.length; i++) {
85              unicodeContent[i] = '\u0e57'; // Thai digit 7
86          }
87  
88          payload.append(OPEN);
89          payload.append(unicodeContent);
90          payload.append(CLOSE);
91  
92          DigitalObject obj = createTestObject(FEDORA_OBJECT_3_0);
93          final String dsID1 = "DS1";
94  
95          /* Populate the object with a test datastream and serialize */
96          DatastreamXMLMetadata ds1 = createXDatastream(dsID1);
97          ds1.xmlContent = payload.toString().getBytes("UTF-8");
98          obj.addDatastreamVersion(ds1, true);
99  
100         String serializedContent =
101                 doSerialize(obj, SERIALIZE_STORAGE_INTERNAL)
102                         .getElementsByTagName(TAG).item(0).getFirstChild()
103                         .getNodeValue();
104 
105         assertTrue("UTF-8 chars are not serialized properly!",
106                    new String(unicodeContent).equals(serializedContent));
107     }

124     /**
125      * Serialize the object, failing the test if an exception is thrown.
126      */
127     protected Document doSerializeOrFail(DigitalObject obj, int transContext) {
128         Document result = null;
129         try {
130             result = doSerialize(obj, transContext);
131         } catch (ObjectIntegrityException e) {
132             e.printStackTrace();
133             fail("Serializer threw ObjectIntegrityException");
134         } catch (SAXException e) {
135             e.printStackTrace();
136             fail("Serialized XML was not well-formed");
137         }
138         return result;
139     }

141     /**
142      * Serialize the object, failing the test only if obviously incorrect
143      * behavior occurs.
144      *
145      * @throws ObjectIntegrityException
146      *         if the serializer fails due to same.
147      * @throws SAXException
148      *         if the result XML is not well-formed.
149      */
150     protected Document doSerialize(DigitalObject obj, int transContext)
151             throws ObjectIntegrityException, SAXException {
152         ByteArrayOutputStream out = new ByteArrayOutputStream();
153         try {
154             m_serializer.serialize(obj, out, "UTF-8", transContext);
155         } catch (StreamIOException e) {
156             fail("Serializer threw StreamIOException");
157         } catch (UnsupportedEncodingException e) {
158             fail("Serializer doesn't support UTF-8!?");
159         }
160 
161         InputStream in = new ByteArrayInputStream(out.toByteArray());
162         try {
163             return XMLUnit.buildControlDocument(new InputSource(in));
164         } catch (SAXException notWellFormed) {
165             throw notWellFormed;
166         } catch (IOException wontHappen) {
167             throw new Error(wontHappen);
168         }
169     }
