34      /** Generic tests should all pass with these patterns. */
35      @Test
36      public void testGoodPatterns() throws Exception {
37          runGenericTests(null);
38          runGenericTests("");
39          runGenericTests("#");
40          runGenericTests("#/#");
41          runGenericTests("##");
42          runGenericTests("##/##");
43          runGenericTests("##/##/##");
44          runGenericTests("################################"); // 32 chars
45      }

47      /** Bad patterns should fail at construction-time. */
48      @Test
49      public void testBadPatterns() throws Exception {
50          assertBadPattern("a");
51          assertBadPattern("a#");
52          assertBadPattern("#a");
53          assertBadPattern("/#");
54          assertBadPattern("#/");
55          assertBadPattern("/#/");
56          assertBadPattern("/#/");
57          assertBadPattern("#################################"); // 33 chars
58      }

60      /** getExternalId should produce the expected URIs. */
61      @Test
62      public void testGetExternalId() {
63          IdMapper mapper = new HashPathIdMapper("##/##/##");
64          assertEquals(URI1, mapper.getExternalId(URI1_ENCODED));
65          assertEquals(URI2, mapper.getExternalId(URI2_ENCODED));
66          assertEquals(URI3, mapper.getExternalId(URI3_ENCODED));
67      }
68  
69      /** getInternalId should produce the expected URIs. */
70      @Test
71      public void testGetInternalId() {
72          IdMapper mapper = new HashPathIdMapper("##/##/##");
73          assertEquals(URI1_ENCODED, mapper.getInternalId(URI1));
74          assertEquals(URI2_ENCODED, mapper.getInternalId(URI2));
75          assertEquals(URI3_ENCODED, mapper.getInternalId(URI3));
76      }

78      /** getInternalPrefix should throw NPE if given null. */

91      /** getInternalPrefix should return null if pattern is nonempty. */
92      @Test
93      public void testGetInternalPrefixWithPattern() {
94          assertNull(new HashPathIdMapper("#").getInternalPrefix("urn:test"));

97      private static void runGenericTests(String pattern)
98              throws Exception {
99          IdMapper mapper = new HashPathIdMapper(pattern);
100         checkRoundtrip(mapper, URI1);
101         checkRoundtrip(mapper, URI2);
102         checkRoundtrip(mapper, URI3);
103     }
104 
105     private static void checkRoundtrip(IdMapper mapper, URI orig) {
106         URI internal = mapper.getInternalId(orig);
107         URI external = mapper.getExternalId(internal);
108         assertEquals(orig, external);
109     }
110 
111     private static void assertBadPattern(String pattern) {
112         try {
113             new HashPathIdMapper(pattern);
114         } catch (Throwable th) {
115             if (!(th instanceof IllegalArgumentException)) {
116                 fail("Bad pattern '" + pattern + "' should have thrown "
117                      + "IllegalArgumentException, but threw "
118                      + th.getClass().getName());
119             }
120         }
121 
122     }
