186     /** Tests the serializers/deserializers when faced with null object property values.
187      * <p>
188      * Currently, this test assures that null iproperty values are handled consistently
189      * among serializers and deserializers.   The expected behaviour is a bit un-intuitive,
190      * but represents the "status quo" that satisfies existing server code:
191      * <dl>
192      * <dt>CreatedDate, LastModifiedDate, External properties</dt>
193      * <dd>Null value should be interpreted as null</dd>
194      * <dt>Label, OwnerId</dt>
195      * <dd>Null value should be interpreted as an empty string ("")</dd>
196      * <dt>State</dt>
197      * <dd>Null value should be interpreted as "Active"</dd>
198      * </dl>
199      * </p>
200      */
201     @Test
202     public void testNullObjectProperties() {
203         final String EXT_PROP = "http://example.org/test";
204         DigitalObject input = createTestObject(FEDORA_OBJECT_3_0);
205         input.setCreateDate(null);
206         input.setLastModDate(null);
207         input.setLabel(null);
208         input.setOwnerId(null);
209         input.setState(null);
210         input.setExtProperty(EXT_PROP, null);
211 
212         DigitalObject obj = doDeserializeOrFail(input);
213 
214         assertNull("Create date should be null", obj.getCreateDate());
215         assertNull("LastMod date should be null", obj.getLastModDate());
216         assertEquals("Null label should be interpreted as empty string", "", obj.getLabel());
217         assertEquals("Null ownerid should be interpreted as empty string", "", obj.getOwnerId());
218         assertEquals("Null state should be interpreted as active", "A", obj.getState());
219         assertNull("Ext property should be null", obj.getExtProperty(EXT_PROP));
220     }

222     /** Tests the serializers/deserializers when faced with empty ("") object property values.
223      * <p>
224      * Currently, this test assures that empty string property values are handled consistently
225      * among serializers and deserializers.   The expected behaviour is as follows:
226      * <dl>
227      * <dt>Label, Ownerid, External properties</dt>
228      * <dd>Empty string value should be interpreted the empty string ("")</dd>
229      * <dt>State</dt>
230      * <dd>Empty string values should be interpreted as "Active"</dd>
231      * </dl>
232      * </p>
233      */
234     @Test
235     public void testEmptyObjectProperties() {
236         final String EXT_PROP_SUPPORTED = "http://example.org/ext-supported";
237         final String EXT_PROP = "http://example.org/test";
238         DigitalObject input = createTestObject(FEDORA_OBJECT_3_0);
239         input.setLabel("");
240         input.setOwnerId("");
241         //input.setState("");
242         input.setExtProperty(EXT_PROP_SUPPORTED, "true");
243         input.setExtProperty(EXT_PROP, "");
244         DigitalObject obj = doDeserializeOrFail(input);
245 
246         assertEquals("Empty label should remain empty", "", obj.getLabel());
247         assertEquals("Empty Ownerid should remain empty", "", obj.getOwnerId());
248         assertEquals("Empty State should be interpreted as active", "A", obj.getState());
249 
250         /* Some formats (METS) don't support ext. properties */
251         if ("true".equals(obj.getExtProperty(EXT_PROP_SUPPORTED))) {
252             assertEquals("Empty Ext property should remain empty", "", obj.getExtProperty(EXT_PROP));
253         }
254     }

278     /**
279      * Copies of an object by deserializing and re-serializing. In theory, there
280      * should be no difference between copy generations..
281      *
282      * @param original
283      *        Object to copy
284      * @return Copy formed by serializing and de-serializing the original.
285      * @throws UnsupportedEncodingException
286      * @throws ObjectIntegrityException
287      * @throws StreamIOException
288      */
289     private DigitalObject translatedCopy(DigitalObject original)
290             throws UnsupportedEncodingException, ObjectIntegrityException,
291             StreamIOException {
292         DigitalObject copy = new BasicDigitalObject();
293         ByteArrayOutputStream out = new ByteArrayOutputStream();
294 
295         m_serializer.serialize(original,
296                                out,
297                                "UTF-8",
298                                SERIALIZE_STORAGE_INTERNAL);
299         ByteArrayInputStream in = new ByteArrayInputStream(out.toByteArray());
300         m_deserializer.deserialize(in,
301                                    copy,
302                                    "UTF-8",
303                                    SERIALIZE_STORAGE_INTERNAL);
304         return copy;
305     }

322     protected DigitalObject doDeserializeOrFail(DigitalObject obj) {
323         DigitalObject result = null;
324         try {
325             result = doDeserialize(obj);
326         } catch (ObjectIntegrityException e) {
327             e.printStackTrace();
328             fail("Deserializer threw ObjectIntegrityException");
329         } catch (StreamIOException e) {

331             fail("Deserializer threw StreamIOException");

336     @SuppressWarnings("unchecked")
337     protected DigitalObject doDeserialize(DigitalObject obj)
338             throws ObjectIntegrityException, StreamIOException {
339         /*
340          * Make RELS-EXT the last datastream, just to make things trickiest for
341          * deserializers (i.e. if serializers need to know relationships before
342          * RELS-EXT has been parsed..)
343          */
344         try {
345 
346             Field dsField =
347                     BasicDigitalObject.class.getDeclaredField("m_datastreams");
348             dsField.setAccessible(true);
349 
350             LinkedHashMap<String, List<Datastream>> nativelyOrdered =
351                     (LinkedHashMap<String, List<Datastream>>) dsField.get(obj);
352 
353             LinkedHashMap<String, List<Datastream>> speciallyOrdered =
354                     new LinkedHashMap<String, List<Datastream>>();
355 
356             Iterator<String> di = obj.datastreamIdIterator();
357 
358             /* Just copy everything EXCEPT rels ext */
359             while (di.hasNext()) {
360                 String id = di.next();
361                 if (!id.equals("RELS-EXT")) {
362                     speciallyOrdered.put(id, nativelyOrdered.get(id));
363                 }
364             }
365 
366             /* Put RELS-EXT last, if defined */
367             List<Datastream> rels = nativelyOrdered.get("RELS-EXT");
368             if (rels != null) {
369                 speciallyOrdered.put("RELS-EXT", rels);
370             }
371 
372             dsField.set(obj, speciallyOrdered);
373         } catch (Exception e) {
374             throw new RuntimeException(e);
375         }
376         return doDeserialize(getStream(obj));
377     }

379     protected DigitalObject doDeserialize(InputStream in)
380             throws ObjectIntegrityException, StreamIOException {
381         BasicDigitalObject obj = new BasicDigitalObject();
382 
383         try {
384             m_deserializer.deserialize(in, obj, "UTF-8", DESERIALIZE_INSTANCE);
385         } catch (UnsupportedEncodingException wontHappen) {
386             fail("Deserializer doesn't support UTF-8?!");
387         }
388         return obj;
389     }
