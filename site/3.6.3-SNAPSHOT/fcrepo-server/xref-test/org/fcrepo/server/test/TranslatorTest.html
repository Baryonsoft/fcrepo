21  /**
22   * Tests the configured DOTranslator instance, deserializing, then 
23   * re-serializing and printing the bytes from the file whose name is passed in.
24   * 
25   * <p>Since DOTranslator is a Module, it's more appropriate to test it by 
26   * starting up the configured server instance.
27   * 
28   * @author Chris Wilper
29   */
30  public class TranslatorTest {
31  
32      public static void main(String args[]) {
33          FileInputStream in = null;
34          try {
35              if (args.length != 3) {
36                  throw new IOException("*Three* parameters needed, filename, format, and encoding.");
37              }
38              if (Constants.FEDORA_HOME == null) {
39                  throw new IOException("FEDORA_HOME is not set. Try using -Dfedora.home=path/to/fedorahome");
40              }
41              in = new FileInputStream(new File(args[0]));
42              Server server;
43              server = Server.getInstance(new File(Constants.FEDORA_HOME));
44              DOTranslator trans =
45                      (DOTranslator) server
46                              .getModule("org.fcrepo.server.storage.translation.DOTranslator");
47              if (trans == null) {
48                  throw new IOException("DOTranslator module not found via getModule");
49              }
50              DigitalObject obj = new BasicDigitalObject();
51              System.out.println("Deserializing...");
52              trans.deserialize(in,
53                                obj,
54                                args[1],
55                                args[2],
56                                DOTranslationUtility.DESERIALIZE_INSTANCE);
57              System.out.println("Done.");
58              ByteArrayOutputStream outStream = new ByteArrayOutputStream();
59              System.out.println("Re-serializing...");
60              trans.serialize(obj,
61                              outStream,
62                              args[1],
63                              args[2],
64                              DOTranslationUtility.SERIALIZE_STORAGE_INTERNAL);
65              System.out.println("Done. Here it is:");
66              System.out.println(outStream.toString(args[2]));
67              server.shutdown(null); //fixup for xacml
68          } catch (Exception e) {
69              System.out.println("Error: " + e.getClass().getName() + " "
70                      + e.getMessage() + "\n");
71              e.printStackTrace();
72          }
73      }
