345     private Date parseDateString(String dateString) {
346         try {
347             return new SimpleDateFormat("yyyyMMdd.HHmmss.SSS")
348                     .parse(dateString);
349         } catch (ParseException e) {
350             // eat the exception!
351             e.printStackTrace();
352             return null;

356     /**
357      * Confirm that both of the transports received the expected numbers of
358      * requests to openFile, getWriter, closeFile and shutdown.
359      */
360     private void assertCorrectNumberOfRequests(int openFileRequests,
361                                                int getWriterRequests,
362                                                int closeFileRequests,
363                                                int shutdownRequests) {
364         assertCorrectNumberOfRequests("transport1",
365                                       transport1,
366                                       openFileRequests,
367                                       getWriterRequests,
368                                       closeFileRequests,
369                                       shutdownRequests);
370         assertCorrectNumberOfRequests("transport2",
371                                       transport2,
372                                       openFileRequests,
373                                       getWriterRequests,
374                                       closeFileRequests,
375                                       shutdownRequests);
376     }

378     private void assertCorrectNumberOfRequests(String name,
379                                                MockTransport transport,
380                                                int openFileRequests,
381                                                int getWriterRequests,
382                                                int closeFileRequests,
383                                                int shutdownRequests) {
384         assertEquals(name + " should get " + openFileRequests
385                 + " openFile() request(s)", openFileRequests, transport1
386                 .getHowManyOpenFileRequests());
387         assertEquals(name + " should get " + getWriterRequests
388                 + " getWriter() request(s)", getWriterRequests, transport1
389                 .getHowManyGetWriterRequests());
390         assertEquals(name + " should get " + closeFileRequests
391                 + " closeFile() request(s)", closeFileRequests, transport1
392                 .getHowManyCloseFileRequests());
393         assertEquals(name + " should get " + shutdownRequests
394                 + " shutdown() request(s)", shutdownRequests, transport1
395                 .getHowManyShutdownRequests());
396     }
