214     private void startMockRmiJournalReceiver(boolean throwException)
215             throws IOException {
216         String exceptionOption =
217                 throwException ? "throwException" : "dontThrow";
218 
219         ProcessBuilder pb =
220                 new ProcessBuilder("java", MockRmiJournalReceiver.class
221                         .getName(), exceptionOption);
222         pb.environment()
223                 .put("CLASSPATH", System.getProperty("java.class.path"));
224         pb.redirectErrorStream(true);
225         rmiRegistryProcess = pb.start();
226 
227         StreamEater outputEater =
228                 new StreamEater(rmiRegistryProcess.getInputStream());
229         rmiRegistryProcessOutput = outputEater.getOutput();
230     }
231 
232     private void assertCorrectNumberOfCalls(int i, int j, int k) {
233         // Give the output a chance to catch up with the registry process.
234         try {
235             Thread.sleep(100);
236         } catch (InterruptedException e) {
237             // ignore an exception
238         }
239         String outputString = rmiRegistryProcessOutput.toString();
240 
241         String[] split1 = outputString.split("openFile\\(");
242         assertEquals("wrong number of openFile() calls", i, split1.length - 1);
243 
244         String[] split2 = outputString.split("writeText\\(");
245         assertEquals("wrong number of writeText() calls", j, split2.length - 1);
246 
247         String[] split3 = outputString.split("closeFile\\(");
248         assertEquals("wrong number of closeFile() calls", k, split3.length - 1);
249     }

251     private static class StreamEater
252             extends Thread {
253 
254         private final InputStream stream;
255 
256         private final StringWriter output = new StringWriter();
257 
258         private final byte[] buffer = new byte[4096];
259 
260         public StreamEater(InputStream stream) {
261             this.stream = stream;
262             start();
263         }
264 
265         @Override
266         public void run() {
267             try {
268                 int howMany = 0;
269                 while (true) {
270                     howMany = stream.read(buffer);
271                     if (howMany > 0) {
272                         output.write(new String(buffer, 0, howMany));
273                     } else if (howMany == 0) {
274                         Thread.yield();
275                     } else {
276                         break;
277                     }
278                 }
279             } catch (IOException e) {
280                 e.printStackTrace();
281             } finally {
282                 try {
283                     stream.close();
284                 } catch (IOException e) {
285                     e.printStackTrace();
286                 }
287             }
288         }
289 
290         public StringWriter getOutput() {
291             return output;
292         }
293     }
