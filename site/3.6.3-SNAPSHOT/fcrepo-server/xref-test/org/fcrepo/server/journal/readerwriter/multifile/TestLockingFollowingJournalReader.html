194             createJournalFileFromString(getSimpleIngestString());
195             createJournalFileFromString(getSimpleIngestString());

203                     new JournalConsumer(parameters, role, server);

241     /**
242      * A lock request created while the system if polling, which should prevent
243      * further processing until it is removed. Create 1 files and watch it
244      * process all of them. Create a lock and wait for the ack. Create a 2nd
245      * file, and it will not be processed. Remove the lock; ack is removed and
246      * last file is processed.
247      */
248     public void disabledtestLockWhilePollingAndResume() {
249         try {
250             // create 1 file, with an ingest
251             createJournalFileFromString(getSimpleIngestString());
252 
253             // create the JournalConsumer and run it.
254             JournalConsumer consumer =
255                     new JournalConsumer(parameters, role, server);
256             startConsumerThread(consumer);
257 
258             // the file should be processed and we being polling.
259             waitWhileThreadRuns(WAIT_INTERVAL);
260             assertEquals("The first file should have been processed.",
261                          1,
262                          delegate.getCallCount());
263             assertEquals("The first file should have been processed.",
264                          0,
265                          howManyFilesInDirectory(journalDirectory));
266             assertEquals("The first file should have been processed.",
267                          1,
268                          howManyFilesInDirectory(archiveDirectory));
269 
270             // create a lock request and wait for the acceptance.
271             createLockRequest();
272             waitForLockAccepted();
273 
274             // create another Journal file, but it won't be processed.
275             createJournalFileFromString(getSimpleIngestString());
276             waitWhileThreadRuns(WAIT_INTERVAL);
277             assertEquals("The second file should not have been processed.",
278                          1,
279                          delegate.getCallCount());
280             assertEquals("The second file should not have been processed.",
281                          1,
282                          howManyFilesInDirectory(journalDirectory));
283             assertEquals("The second file should not have been processed.",
284                          1,
285                          howManyFilesInDirectory(archiveDirectory));
286             int lockMessageIndex = assertLockMessageInLog();
287 
288             // remove the lock and the file is processed.
289             removeLockRequest();
290             waitForLockReleased();
291             waitWhileThreadRuns(WAIT_INTERVAL);
292             consumer.shutdown();
293 
294             assertEquals("Expected to see 2 ingests", 2, delegate
295                     .getCallCount());
296             assertEquals("Journal files not all gone",
297                          0,
298                          howManyFilesInDirectory(journalDirectory));
299             assertEquals("Wrong number of archive files",
300                          2,
301                          howManyFilesInDirectory(archiveDirectory));
302             assertUnlockMessageInLog(lockMessageIndex);
303         } catch (Throwable e) {
304             processException(e);
305         }

322     /**
323      * Wait until the JournalConsumerThread stops, or until the time limit
324      * expires, whichever comes first.
325      */
326     private void waitWhileThreadRuns(int maxSecondsToWait) {
327         for (int i = 0; i < maxSecondsToWait; i++) {
328             if (getNumberOfCurrentThreads() == initialNumberOfThreads) {
329                 return;
330             }
331             try {
332                 Thread.sleep(1000);
333             } catch (InterruptedException e) {
334                 e.printStackTrace();
335             }
336         }
337     }
338 
339     /**
340      * Wait until the lock is accepted, or until the time runs out. If the
341      * latter, complain.
342      */
343     private void waitForLockAccepted() {
344         int maxWait = 3;
345         for (int i = 0; i < maxWait; i++) {
346             if (lockAcceptedFile.exists()) {
347                 return;
348             }
349             try {
350                 Thread.sleep(1000);
351             } catch (InterruptedException e) {
352                 e.printStackTrace();
353             }
354         }
355         fail("Lock was not accepted after " + maxWait + " seconds.");
356     }
357 
358     /**
359      * Wait until the lock is released, or until the time runs out. If the
360      * latter, complain.
361      */
362     private void waitForLockReleased() {
363         int maxWait = 3;
364         for (int i = 0; i < maxWait; i++) {
365             if (!lockAcceptedFile.exists()) {
366                 return;
367             }
368             try {
369                 Thread.sleep(1000);
370             } catch (InterruptedException e) {
371                 e.printStackTrace();
372             }
373         }
374         fail("Lock was not released after " + maxWait + " seconds.");
375     }

417      * unlock message.
418      */
419     private void assertUnlockMessageInLog(int lockMessageIndex) {
420         List<String> messages = MockJournalRecoveryLog.getMessages();
421         int unlockMessageIndex = lockMessageIndex + 1;
422         assertTrue(messages.size() > unlockMessageIndex);
423         String unlockMessage = messages.get(unlockMessageIndex);
424         assertStringStartsWith(unlockMessage, "Lock request removed");

434     private void processException(Throwable e) {
435         if (e instanceof ServerException) {
436             System.err.println("ServerException: code='"
437                     + ((ServerException) e).getCode() + "', class='"
438                     + e.getClass().getName() + "'");
439             StackTraceElement[] traces = e.getStackTrace();
440             for (StackTraceElement element : traces) {
441                 System.err.println(element);
442             }
443             Throwable cause = e.getCause();
444             if (cause != null) {
445                 cause.printStackTrace();
446             }
447             fail("Threw a ServerException");
448         } else {
449             e.printStackTrace();
450             fail("Threw an exception");
451         }
452     }

506     /**
507      * This sub-class of {@link MockManagementDelegate} allows us to insert a
508      * {@link Runnable} that will be executed in the middle of an ingest call.
509      *
510      * @author Firstname Lastname
511      */
512     private static class MyMockManagementDelegate
513             extends MockManagementDelegate {
514 
515         private Runnable ingestOperation;
516 
517         public void setIngestOperation(Runnable ingestOperation) {
518             this.ingestOperation = ingestOperation;
519         }
520 
521         @Override
522         public String ingest(Context context,
523                              InputStream serialization,
524                              String logMessage,
525                              String format,
526                              String encoding,
527                              String pid) throws ServerException {
528             String result =
529                     super.ingest(context,
530                                  serialization,
531                                  logMessage,
532                                  format,
533                                  encoding,
534                                  pid);
535 
536             if (ingestOperation != null) {
537                 ingestOperation.run();
538             }
539 
540             return result;
541         }
