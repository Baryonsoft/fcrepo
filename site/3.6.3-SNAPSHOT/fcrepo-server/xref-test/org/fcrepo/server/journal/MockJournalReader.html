26  /**
27   * Read Journal XML from a String buffer. If the buffer is null or empty, treat
28   * it as an empty Journal.
29   *
30   * @author Jim Blake
31   */
32  public class MockJournalReader
33          extends JournalReader {
34  
35      private static String buffer;
36  
37      public static void setBuffer(String buffer) {
38          MockJournalReader.buffer = buffer;
39      }
40  
41      private StringReader stringReader;
42  
43      private XMLEventReader xmlReader;
44  
45      public MockJournalReader(Map<String, String> parameters,
46                               String role,
47                               JournalRecoveryLog recoveryLog,
48                               ServerInterface server)
49              throws JournalException {
50          super(parameters, role, recoveryLog, server);
51      }
52  
53      @Override
54      public ConsumerJournalEntry readJournalEntry() throws JournalException,
55              XMLStreamException {
56          prepareXmlReader();
57  
58          if (xmlReader == null) {
59              return null;
60          }
61  
62          advancePastWhitespace(xmlReader);
63  
64          XMLEvent next = xmlReader.peek();
65          if (isStartTagEvent(next, QNAME_TAG_JOURNAL_ENTRY)) {
66              ConsumerJournalEntry journalEntry =
67                      super.readJournalEntry(xmlReader);
68              journalEntry.setIdentifier(new Date().toString());
69              return journalEntry;
70          } else if (isEndTagEvent(next, QNAME_TAG_JOURNAL)) {
71              return null;
72          } else {
73              throw getNotNextMemberOrEndOfGroupException(QNAME_TAG_JOURNAL,
74                                                          QNAME_TAG_JOURNAL_ENTRY,
75                                                          next);
76          }
77      }
78  
79      @Override
80      public void shutdown() throws JournalException {
81          try {
82              if (xmlReader != null) {
83                  xmlReader.close();
84              }
85              if (stringReader != null) {
86                  stringReader.close();
87              }
88          } catch (XMLStreamException e) {
89              throw new JournalException(e);
90          }
91      }
92  
93      /**
94       * Advance past the document header to the first JournalEntry.
95       */
96      private void advanceIntoFile() throws XMLStreamException, JournalException {
97          XMLEvent event = xmlReader.nextEvent();
98          if (!event.isStartDocument()) {
99              throw new JournalException("Expecting XML document header, but event was '"
100                     + event + "'");
101         }
102 
103         event = xmlReader.nextTag();
104         if (!isStartTagEvent(event, QNAME_TAG_JOURNAL)) {
105             throw new JournalException("Expecting FedoraJournal start tag, but event was '"
106                     + event + "'");
107         }
108 
109         String hash =
110                 getOptionalAttributeValue(event.asStartElement(),
111                                           QNAME_ATTR_REPOSITORY_HASH);
112         checkRepositoryHash(hash);
113     }
114 
115     private void prepareXmlReader() throws JournalException {
116         if (xmlReader != null) {
117             return;
118         }
119         if (buffer == null || buffer.length() == 0) {
120             return;
121         }
122 
123         stringReader = new StringReader(buffer);
124 
125         try {
126             xmlReader =
127                     XMLInputFactory.newInstance()
128                             .createXMLEventReader(stringReader);
129             advanceIntoFile();
130         } catch (XMLStreamException e) {
131             throw new JournalException(e);
132         } catch (FactoryConfigurationError e) {
133             throw new JournalException(e);
134         }
135     }
136 }
