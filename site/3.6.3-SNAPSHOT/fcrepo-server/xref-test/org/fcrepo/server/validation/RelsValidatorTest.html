441     private void validateAndClear(String dsId) throws Exception {
442         try {
443             TripleIterator iter = new MockTripleIterator(triples);
444             ByteArrayOutputStream out = new ByteArrayOutputStream();
445             iter.toStream(out, RDFFormat.RDF_XML, false);
446             new RelsValidator().validate(pid, dsId, new ByteArrayInputStream(out.toByteArray()));
447         } finally {
448             triples.clear();
449         }

451     private static Triple createTriple(PID pid,
452                                        String predicate,
453                                        String object,
454                                        boolean isLiteral,
455                                        String datatype)
456             throws ServerException {
457         return createTriple(pid.toURI(), predicate, object, isLiteral, datatype);
458     }

460     private static Triple createTriple(String subject,
461                                        String predicate,
462                                        String object,
463                                        boolean isLiteral,
464                                        String datatype)
465             throws ServerException {
466         ObjectNode o = null;
467         try {
468             if (isLiteral) {
469                 if (datatype == null || datatype.length() == 0) {
470                     o = new SimpleLiteral(object);
471                 } else {
472                     o = new SimpleLiteral(object, new URI(datatype));
473                 }
474             } else {
475                 o = new SimpleURIReference(new URI(object));
476             }
477             return new SimpleTriple(new SimpleURIReference(new URI(subject)),
478                                     new SimpleURIReference(new URI(predicate)),
479                                     o);
480         } catch (URISyntaxException e) {
481             throw new GeneralException(e.getMessage(), e);
482         }
483     }
