109     @Test
110     public void testGetDDL() throws InconsistentTableSpecException, IOException {
111         List<TableSpec> tableSpecs =
112                 TableSpec.getTableSpecs(MockTableSpec.getTableSpecStream());
113         verifyInputTableSpec(tableSpecs);
114 
115         for (TableSpec spec : tableSpecs) {
116             List<String> statements = converter.getDDL(spec);
117             for (String stmt : statements) {
118                 verifyAndEliminateStatement(stmt);
119             }
120         }
121 
122         assertTrue("Some statements unverified: " + expectedStmts.size(),
123                    expectedStmts.isEmpty());
124     }
125 
126     private void verifyInputTableSpec(List<TableSpec> tableSpecs)
127             throws InconsistentTableSpecException, IOException {
128 
129         final int NUM_SPECS = 7;
130         int numTableSpecs = tableSpecs.size();
131         assertTrue("There should be " + NUM_SPECS + " tableSpecs: "
132                    + numTableSpecs, numTableSpecs == NUM_SPECS);
133     }
134 
135     private void verifyAndEliminateStatement(String stmt) {
136         assertNotNull(stmt);
137 
138         String foundStmt = null;
139         for (String expected : expectedStmts) {
140             if (stmt.equalsIgnoreCase(expected)) {
141                 foundStmt = expected;
142             }
143         }
144         assertTrue("Creation statement invalid: '" + stmt + "'",
145                    foundStmt != null);
146         expectedStmts.remove(foundStmt);
147     }
