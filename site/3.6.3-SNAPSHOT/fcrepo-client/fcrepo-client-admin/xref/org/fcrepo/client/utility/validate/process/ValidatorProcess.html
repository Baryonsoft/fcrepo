22  /**
23   * A command-line utility that validates objects in a remote repository,
24   * selected by criteria. See the javadoc for {@link ValidatorProcessParameters}
25   * for the usage details.
26   *
27   * @author Jim Blake
28   */
29  public class ValidatorProcess {
30  
31      /**
32       * Open the connection to the Fedora server.
33       */
34      private static RemoteObjectSource openObjectSource(ServiceInfo serviceInfo) {
35          try {
36              return new RemoteObjectSource(serviceInfo);
37          } catch (ServiceException e) {
38              throw new IllegalStateException("Failed to initialize "
39                      + "the ValidatorProcess: ", e);
40          } catch (IOException e) {
41              throw new IllegalStateException("Failed to initialize "
42                      + "the ValidatorProcess: ", e);
43          }
44      }
45  
46      /**
47       * The list of PIDs may come from a file, or from a query against the object
48       * source.
49       */
50      private static Iterator<String> getPidIterator(ValidatorProcessParameters parms,
51                                                     RemoteObjectSource objectSource)
52              throws ObjectSourceException {
53          if (parms.getIteratorType() == IteratorType.FS_QUERY) {
54              return objectSource.findObjectPids(parms.getQuery());
55          } else {
56              return new PidfileIterator(parms.getPidfile());
57          }
58      }
59  
60      public static void main(String[] args) throws ObjectSourceException {
61          System.setProperty("java.awt.headless", "true");
62          try {
63              // Parse the parameters.
64              ValidatorProcessParameters parms =
65                      new ValidatorProcessParameters(args);
66  
67              // Create the tools we will need.
68              RemoteObjectSource objectSource =
69                      openObjectSource(parms.getServiceInfo());
70              ValidationResults results =
71                      new SimpleValidationResults(parms.getVerbose());
72  
73              // Get the list of PIDs.
74              Iterator<String> pids = getPidIterator(parms, objectSource);
75  
76              // Go through the list, validating.
77              ObjectValidator validator = new ObjectValidator(objectSource);
78              while (pids.hasNext()) {
79                  results.record(validator.validate(pids.next()));
80              }
81  
82              // Display the results.
83              results.closeResults();
84          } catch (ValidatorProcessUsageException e) {
85              System.err.println(e.getMessage());
86          }
87      }
