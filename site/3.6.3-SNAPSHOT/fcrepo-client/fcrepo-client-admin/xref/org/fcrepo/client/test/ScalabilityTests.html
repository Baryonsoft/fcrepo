88      public void init(String host, String port, String username, String password, String batch, String batches, String threads, String outputFileLocation, String context) throws Exception {
89  
90          String baseURL =  "http://" + host + ":" + port + "/" + context;
91          FedoraClient fedoraClient = new FedoraClient(baseURL, username, password);
92          apim = fedoraClient.getAPIMMTOM();
93  
94          try {
95              batchSize = Integer.valueOf(batch);
96          } catch (NumberFormatException nfe) {
97              System.err.println("Batch Size value could not be " +
98                                 "converted to an integer, using the default value (" +
99                                 defaultBatchSize + ") instead");
100             batchSize = defaultBatchSize;
101         }
102 
103         try {
104             numBatches = Integer.valueOf(batches);
105         } catch (NumberFormatException nfe) {
106             System.err.println("Number of Batches value could not be " +
107                                "converted to an integer, using the default value (" +
108                                defaultNumBatches + ") instead");
109             numBatches = defaultNumBatches;
110         }
111 
112         try {
113             numThreads = Integer.valueOf(threads);
114         } catch (NumberFormatException nfe) {
115             System.err.println("Number of Threads value could not be " +
116                                "converted to an integer, using the default value (" +
117                                defaultNumThreads + ") instead");
118             numThreads = defaultNumThreads;
119         }
120 
121         File outputFile = new File(outputFileLocation);
122         out = new PrintStream(outputFile);
123         out.println("--- Scalability Test Results ---");
124         out.println("Total Objects Ingested, " +
125                     "Time (ms) To Ingest Batch of " + batchSize + " Objects, " +
126                     "Average Ingest Time (ms) Per Object");
127 
128         threadPool = Executors.newFixedThreadPool(numThreads);
129         ingestRunnerList = new ArrayList<Callable<Boolean>>();
130         for(int i=0; i<batchSize; i++) {
131             ingestRunnerList.add(new IngestRunner());
132         }
133     }

139     /**
140      * Runs the ingest over the set number of batches
141      * or runs continuously if the number of batches
142      * is <= 0.
143      */
144     public void runIngestTest() throws Exception {
145         if(numBatches > 0) {
146             for(int i=0; i<numBatches; i++) {
147                 runIngestBatch();
148             }
149         } else {
150             while(true) {
151                 runIngestBatch();
152             }
153         }
154     }
155 
156     /**
157      * Runs a batch of ingests using a thread pool.
158      * Prints the time required to perform all ingests.
159      */
160     public void runIngestBatch() throws Exception {
161         long startTime = 0;
162         long stopTime = 0;
163 
164         startTime = System.currentTimeMillis();
165         threadPool.invokeAll(ingestRunnerList);
166         stopTime = System.currentTimeMillis();
167 
168         totalIngested += batchSize;
169         long ingestTime = stopTime - startTime;
170 
171         out.println(totalIngested + ", " + ingestTime + ", " + ingestTime/batchSize);
172     }

245            port == null || port.startsWith("$") ||
246            username == null || username.startsWith("$") ||
247            password == null || password.startsWith("$") ||
248            batchSize == null || batchSize.startsWith("$") ||
249            numBatches == null || numBatches.startsWith("$") ||
250            threads == null || threads.startsWith("$") ||
251            output == null || output.startsWith("$")) {
252            usage();
253         }

255         ScalabilityTests tests = new ScalabilityTests();
256         tests.init(host, port, username, password, batchSize, numBatches, threads, output, context);
