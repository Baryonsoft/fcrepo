5   package org.fcrepo.client.batch;

7   import java.io.BufferedReader;
8   import java.io.File;
9   import java.io.FileReader;
10  
11  import java.util.Properties;
12  
13  import org.fcrepo.common.Constants;

15  
16  /**
17   * Auto Batch Build.
18   * 
19   * @author Ross Wayland
20   */
21  public class AutoBatchBuild
22          implements Constants {
23  
24      private final Properties batchProperties = new Properties();
25  
26      public AutoBatchBuild(String objectTemplate,
27                            String objectSpecificDir,
28                            String objectDir,
29                            String logFile,
30                            String logFormat,
31                            String objectFormat)
32              throws Exception {
33  
34          batchProperties.setProperty("template", objectTemplate);
35          batchProperties.setProperty("merge-objects", "yes");
36          batchProperties.setProperty("specifics", objectSpecificDir);
37          batchProperties.setProperty("objects", objectDir);
38          batchProperties.setProperty("pids-format", logFormat);
39          batchProperties.setProperty("ingested-pids", logFile);
40          batchProperties.setProperty("object-format", objectFormat);
41  
42          BatchTool batchTool = new BatchTool(batchProperties, null, null);
43          batchTool.prep();
44          batchTool.process();
45      }
46  
47      public static final void main(String[] args) throws Exception {
48          System.setProperty("java.awt.headless", "true");
49          boolean errors = false;
50          String objectFormat = null;
51          if (args.length == 5) {
52              if (!new File(args[0]).exists() && !new File(args[0]).isFile()) {
53                  System.out.println("Specified object template file path: \""
54                          + args[0] + "\" does not exist.");
55                  errors = true;
56              }
57              if (!new File(args[1]).isDirectory()) {
58                  System.out.println("Specified object specific directory: \""
59                          + args[1] + "\" is not directory.");
60                  errors = true;
61              }
62              if (!new File(args[2]).isDirectory()) {
63                  System.out.println("Specified object directory: \"" + args[2]
64                          + "\" is not a directory.");
65                  errors = true;
66              }
67              if (!args[4].equals("xml") && !args[4].equals("text")) {
68                  System.out
69                          .println("Format for log file must must be either: \""
70                                  + "\"xml\"  or  \"txt\"");
71                  errors = true;
72              }
73              // Verify format of template file to see if it is a METS or FOXML template
74              BufferedReader br = new BufferedReader(new FileReader(args[0]));
75              String line;
76              while ((line = br.readLine()) != null) {
77                  System.out.println(line);
78                  if (line.indexOf("<foxml:") != -1) {
79                      objectFormat = FOXML1_1.uri;
80                      break;
81                  }
82                  if (line.indexOf("<METS:") != -1) {
83                      objectFormat = METS_EXT1_1.uri;
84                      break;
85                  }
86              }
87              br.close();
88              br = null;
89  
90              if (objectFormat == null) {
91                  errors = true;
92              }
93  
94              if (!errors) {
95                  System.out.println("\n*** Format of template files is: "
96                          + objectFormat + " . Generated objects will be in "
97                          + objectFormat + " format.\n");
98                  new AutoBatchBuild(args[0],
99                                     args[1],
100                                    args[2],
101                                    args[3],
102                                    args[4],
103                                    objectFormat);
104             }
105         } else {
106 
107             if (objectFormat == null && args.length == 5) {
108                 System.out.println("\nUnknown format for template file.\n"
109                         + "Template file must either be METS or FOXML.\n");
110             } else {
111                 System.out.println("ERROR: Wrong Number of Arguments, AutoBatchBuild requires 5 arguments.");
112                 System.out.println("");
113                 System.out.println("Command: fedora-batch-build");
114                 System.out.println("");
115                 System.out.println("Syntax:");
116                 System.out.println("  fedora-batch-build [object-template-file] [object-specific-dir] [object-directory] [log-filepath] [log-format]");
117                 System.out.println("");
118                 System.out.println("  Where:");
119                 System.out.println("");
120                 System.out.println("    object-template-file - the full path to the batch template file");
121                 System.out.println("    obj-specific-dir     - the full path to the directory containing the object-specific files");
122                 System.out.println("    object-directory     - the full path to the directory where the generated objects will be built");
123                 System.out.println("    log-filepath         - the full path to the file where logs will be written");
124                 System.out.println("    log-format           - the format of the log file. Valid values are text or xml.");
125             }     
126         }
127     }
