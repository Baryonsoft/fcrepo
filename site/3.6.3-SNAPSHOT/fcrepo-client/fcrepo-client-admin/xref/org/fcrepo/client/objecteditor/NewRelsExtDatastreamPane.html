52      private static Datastream[] MakeBlankRelsExtDatastream() {
53          Datastream[] ds = new Datastream[1];
54          ds[0] = new Datastream();
55          ds[0].setControlGroup(DatastreamControlGroup.fromValue("X"));
56          ds[0].setID("s_dsid");
57          ds[0].setVersionID("RELS-EXT.0");
58          ds[0].setLabel("RDF Statements about this object");
59          ds[0].setVersionable(true);
60          ds[0].setMIMEType("application/rdf+xml");
61          ds[0].setFormatURI(Constants.RELS_EXT1_0.uri);
62          ds[0].setSize(0L);
63          ds[0].setState("A");
64          return ds;
65      }

88      public void saveChanges(String logMessage) throws Exception {
89          if (m_currentVersionPane.isDirty()) {
90              // defer to the currentVersionPane if anything else changed
91              try {
92                  String state =
93                          s_stateComboBoxValues[m_stateComboBox
94                                  .getSelectedIndex()];
95                  String label =
96                          m_currentVersionPane.m_labelTextField.getText().trim();
97                  String mimeType =
98                          m_currentVersionPane.m_MIMETextField.getText().trim();
99                  String formatURI =
100                         m_currentVersionPane.m_formatURITextField.getText()
101                                 .trim();
102                 String[] altIDs =
103                         m_currentVersionPane.m_altIDsTextField.getText().trim()
104                                 .split(" ");
105                 String checksumType =
106                         m_currentVersionPane.m_checksumTypeComboBox
107                                 .getSelectedItem().toString();
108                 if (checksumType.equals("Default")) {
109                     checksumType = null;
110                 }
111                 String location = null;
112                 location =
113                         Administrator.UPLOADER
114                                 .upload(m_currentVersionPane.m_editor
115                                         .getContent());
116                 boolean versionable =
117                         m_versionableComboBox.getSelectedIndex() == NEW_VERSION_ON_UPDATE ? true
118                                 : false;
119                 String newID =
120                         Administrator.APIM.addDatastream(m_pid,
121                                                          s_dsid,
122                                                          TypeUtility.convertStringtoAOS(altIDs),
123                                                          label,
124                                                          versionable, // DEFAULT_VERSIONABLE
125                                                          mimeType,
126                                                          formatURI,
127                                                          location,
128                                                          "X",
129                                                          state,
130                                                          checksumType,
131                                                          null, // checksum type and checksum
132                                                          logMessage); // DEFAULT_LOGMESSAGE
133 
134             } catch (Exception e) {
135                 throw e;
136             }
137             m_owner.addDatastreamTab(s_dsid, false);
138         }
