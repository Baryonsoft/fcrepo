60      /**
61       * Constructor. Queries the server for the object, builds the object and
62       * component tabs, and populates them with the appropriate panels.
63       */
64      public ObjectEditorFrame(String pid, int startTab)
65              throws Exception {
66          super(pid, true, true, true, true);
67          m_pid = pid;
68          // query the server for key object fields
69          ObjectFields o =
70                  Util.getObjectFields(pid, new String[] {"pid", "state",
71                          "label", "cDate", "mDate", "ownerId"});
72          String state = o.getState() != null ? o.getState().getValue() : null;
73          String label = o.getLabel() != null ? o.getLabel().getValue() : null;
74          String cDate = o.getCDate() != null ? o.getCDate().getValue() : null;
75          String mDate = o.getMDate() != null ? o.getMDate().getValue() : null;
76          String ownerId = o.getOwnerId() != null ? o.getOwnerId().getValue() : null;
77  
78          doTitle(false);
79  
80          // set up dirtiness check on close
81          setDefaultCloseOperation(DO_NOTHING_ON_CLOSE);
82          addInternalFrameListener(new ObjectEditorClosingListener(pid));
83  
84          // outerPane(tabbedPane)
85          ImageIcon objIcon =
86              new ImageIcon(this.getClass().getClassLoader().getSystemClassLoader()
87                            .getSystemResource("images/client/standard/general/Information16.gif"));
88  
89          ImageIcon dsIcon =
90              new ImageIcon(this.getClass().getClassLoader().getSystemClassLoader()
91                            .getSystemResource("images/client/standard/general/Copy16.gif"));
92  
93          ImageIcon dissIcon =
94              new ImageIcon(this.getClass().getClassLoader().getSystemClassLoader()
95                            .getSystemResource("images/client/standard/general/Refresh16.gif"));
96  
97          // tabbedPane(ObjectPane, DatastreamsPane, DisseminatorsPane)
98          m_objectPane =
99                  new ObjectPane(this,
100                                pid,
101                                state,
102                                label,
103                                cDate,
104                                mDate,
105                                ownerId);
106         m_datastreamsPane = new DatastreamsPane(this, pid);
107 
108         m_tabbedPane = new JTabbedPane();
109         m_tabbedPane.addTab("Properties", m_objectPane);
110         m_tabbedPane.setBackgroundAt(0, Administrator.DEFAULT_COLOR);
111         m_tabbedPane.setIconAt(0, objIcon);
112         m_tabbedPane.addTab("Datastreams", m_datastreamsPane);
113         m_tabbedPane.setBackgroundAt(1, Administrator.DEFAULT_COLOR);
114         m_tabbedPane.setIconAt(1, dsIcon);
115         m_tabbedPane.setSelectedIndex(startTab);
116 
117         JPanel outerPane = new JPanel();
118         outerPane.setLayout(new BorderLayout());
119         outerPane.setBorder(BorderFactory.createEmptyBorder(4, 4, 4, 4));
120         outerPane.add(m_tabbedPane, BorderLayout.CENTER);
121 
122         getContentPane().setLayout(new BorderLayout());
123         getContentPane().add(outerPane, BorderLayout.CENTER);
124         //ImageIcon openIcon =
125         //    new ImageIcon(ClassLoader.
126         //                  getSystemResource("images/client/standard/general/Open16.gif"));
127         //setFrameIcon(openIcon);
128         pack();
129         Dimension dims = getSize();
130         if (dims.height < 675) {
131             dims.height = 675;
132         } else if (dims.height > 700) {
133             dims.height = 700;
134         }
135         if (dims.width < 925) {
136             dims.width = 925;
137         } else if (dims.width > 950) {
138             dims.width = 950;
139         }
140         setSize(dims);
141         show();
142     }

144     public Datastream[] getCurrentDatastreamVersions() {
145         Collection vColl = m_datastreamsPane.getCurrentVersionMap().values();
146         Datastream[] versions = new Datastream[vColl.size()];
147         Iterator iter = vColl.iterator();
148         int i = 0;
149         while (iter.hasNext()) {
150             versions[i++] = (Datastream) iter.next();
151         }
152         return versions;
153     }

155     private void doTitle(boolean dirty) {
156         String d = "";
157         if (dirty) {
158             d = "*";
159         }
160 
161         setTitle("Object - " + m_pid + d);
162     }

164     @Override

166         return m_objectPane.isDirty() || m_datastreamsPane.isDirty();
167     }

169     public void indicateDirtiness() {
170         int dirtyCount = 0;
171         if (m_objectPane.isDirty()) {
172             dirtyCount++;
173             m_tabbedPane.setTitleAt(0, "Properties*");
174         } else {
175             m_tabbedPane.setTitleAt(0, "Properties");
176         }
177         if (m_datastreamsPane.isDirty()) {
178             dirtyCount++;
179             m_tabbedPane.setTitleAt(1, "Datastreams*");
180         } else {
181             m_tabbedPane.setTitleAt(1, "Datastreams");
182         }
183         //        if (m_disseminatorsPane!=null && m_disseminatorsPane.isDirty())
184         //        {
185         //            dirtyCount++;
186         //            m_tabbedPane.setTitleAt(2, "Disseminators*");
187         //        }
188         //        else
189         //        {
190         //            if (m_disseminatorsPane!=null) m_tabbedPane.setTitleAt(2, "Disseminators");
191         //        }
192         if (dirtyCount > 0) {
193             doTitle(true);
194         } else {
195             doTitle(false);
196         }
197     }

199     /**
200      * Listens for closing events and checks for object and component dirtiness.
201      */
202     protected class ObjectEditorClosingListener
203             extends InternalFrameAdapter {
204 
205         private final String m_pid;
206 
207         public ObjectEditorClosingListener(String pid) {
208             m_pid = pid;
209         }
210 
211         /**
212          * Check if any of the items being edited are dirty. If so, give the
213          * user a chance to keep the editor open so they can save their changes.
214          */
215         @Override
216         public void internalFrameClosing(InternalFrameEvent e) {
217             if (isDirty()) {
218                 Object[] options = {"Yes", "No"};
219                 int selected =
220                         JOptionPane
221                                 .showOptionDialog(null,
222                                                   "Close "
223                                                           + m_pid
224                                                           + " without saving changes?",
225                                                   "Unsaved changes",
226                                                   JOptionPane.DEFAULT_OPTION,
227                                                   JOptionPane.WARNING_MESSAGE,
228                                                   null,
229                                                   options,
230                                                   options[0]);
231                 if (selected == 0) {
232                     e.getInternalFrame().dispose();
233                 }
234             } else {
235                 e.getInternalFrame().dispose();
236             }
237         }
