30  /**
31   * Command-line version of the Batch Modify utility that's available
32   * in the admin GUI client.
33   *
34   * This utility processes an xml input file containing modify directives
35   * enabling mass updating of existing objects. It has six required arguments:
36   * <ol>
37   * <li>host - Name of the Fedora repository server.</li>
38   * <li>port - port number o fthe Fedora server.</li>
39   * <li>user - username of the Fedora server admin user</li>
40   * <li>password - password of the Fedora server admin user</li>
41   * <li>directivesFilePath - absolute file path of the input file containing
42   * modify directives. Note that his file should should validate against the
43   * batchModify schema.</li>
44   * <li>logFilePath - absolute file path of the log file; an xml file providing
45   * a history of the transactions processed.</li>

160     public static void modify(FedoraAPIMMTOM APIM,
161                               Uploader UPLOADER,
162                               FedoraAPIAMTOM APIA,
163                               String directivesFilePath,
164                               String logFilePath,
165                               boolean isValidateOnly) {
166 
167         InputStream in = null;
168         BatchModifyParser bmp = null;
169         BatchModifyValidator bmv = null;
170         long st = System.currentTimeMillis();
171         long et = 0;
172         try {
173             in = new FileInputStream(directivesFilePath);
174             if (isValidateOnly) {
175                 openLog(logFilePath, "validate-modify-directives");
176                 bmv = new BatchModifyValidator(in, s_log);
177             } else {
178                 openLog(logFilePath, "modify-batch");
179                 bmp = new BatchModifyParser(UPLOADER, APIM, APIA, in, s_log);
180             }
181 
182         } catch (Exception e) {
183             System.out.println(e.getClass().getName()
184                     + " - "
185                     + (e.getMessage() == null ? "(no detail provided)" : e
186                             .getMessage()));
187 
188         } finally {
189             try {
190                 if (in != null) {
191                     in.close();
192                 }
193                 if (s_log != null) {
194                     et = System.currentTimeMillis();
195                     if (bmp != null) {
196                         if (bmp.getFailedCount() == -1) {
197                             System.out
198                                     .println("\n\n"
199                                             + bmp.getSucceededCount()
200                                             + " modify directives successfully processed.\n"
201                                             + "Parser error encountered.\n"
202                                             + "An unknown number of modify directives were not processed.\n"
203                                             + "See log file for details of those directives processed before the error.\n"
204                                             + "Time elapsed: "
205                                             + getDuration(et - st));
206                             s_log.println("  <summary>");
207                             s_log
208                                     .println("    "
209                                             + StreamUtility
210                                                     .enc(bmp
211                                                             .getSucceededCount()
212                                                             + " modify directives successfully processed.\n"
213                                                             + "    Parser error encountered.\n"
214                                                             + "    An unknown number of modify directives were not processed.\n"
215                                                             + "    Time elapsed: "
216                                                             + getDuration(et
217                                                                     - st)));
218                             s_log.println("  </summary>");
219                         } else {
220                             System.out
221                                     .println("\n\n"
222                                             + bmp.getSucceededCount()
223                                             + " modify directives successfully processed.\n"
224                                             + bmp.getFailedCount()
225                                             + " modify directives failed.\n"
226                                             + "See log file for details.\n"
227                                             + "Time elapsed: "
228                                             + getDuration(et - st));
229                             s_log.println("  <summary>");
230                             s_log
231                                     .println("    "
232                                             + StreamUtility
233                                                     .enc(bmp
234                                                             .getSucceededCount()
235                                                             + " modify directives successfully processed.\n    "
236                                                             + bmp
237                                                                     .getFailedCount()
238                                                             + " modify directives failed.\n"
239                                                             + "    Time elapsed: "
240                                                             + getDuration(et
241                                                                     - st)));
242                             s_log.println("  </summary>");
243                         }
244                     } else if (bmv != null) {
245                         et = System.currentTimeMillis();
246                         if (bmv.isValid()) {
247                             System.out
248                                     .println("Modify Directives File in \n"
249                                             + directivesFilePath
250                                             + "\n is Valid !"
251                                             + "\nTime elapsed: "
252                                             + getDuration(et - st));
253                             s_log.println("  <summary>");
254                             s_log.println("    Modify Directives File: \n    "

256                                     + "\n    Time elapsed: "
257                                     + getDuration(et - st));
258                             s_log.println("  </summary>");
259                         } else {
260                             System.out
261                                     .println(bmv.getErrorCount()
262                                             + " XML validation Errors found in Modify Directives file.\n"
263                                             + "See log file for details.\n"
264                                             + "Time elapsed: "

266                             s_log.println("  <summary>");

271                                                             + " XML validation Errors found in Modify Directives file.\n"
272                                                             + "    See log file for details.\n"
273                                                             + "    Time elapsed: "
274                                                             + getDuration(et
275                                                                     - st)));
276                             s_log.println("  </summary>");

279                     closeLog();

292     /**
293      * <p>
294      * Convert the duration time from milliseconds to standard hours, minutes,
295      * and seconds format.
296      * </p>
297      *
298      * @param millis -
299      *        The time interval to convert in miliseconds.
300      * @return A string with the converted time.
301      */
302     private static String getDuration(long millis) {
303         long tsec = millis / 1000;
304         long h = tsec / 60 / 60;
305         long m = (tsec - h * 60 * 60) / 60;
306         long s = tsec - h * 60 * 60 - m * 60;
307         StringBuffer out = new StringBuffer();
308         if (h > 0) {
309             out.append(h + " hour");
310             if (h > 1) {
311                 out.append('s');
312             }
313         }
314         if (m > 0) {
315             if (h > 0) {
316                 out.append(", ");
317             }
318             out.append(m + " minute");
319             if (m > 1) {
320                 out.append('s');
321             }
322         }
323         if (s > 0 || h == 0 && m == 0) {
324             if (h > 0 || m > 0) {
325                 out.append(", ");
326             }
327             out.append(s + " second");
328             if (s != 1) {
329                 out.append('s');
330             }
331         }
332         return out.toString();
333     }

387     public static void main(String[] args) {
388         System.setProperty("java.awt.headless", "true");
389 
390         String logFilePath = null;
391         String directivesFilePath = null;
392         String protocol = null;
393         String hostName = null;
394         String username = null;
395         String password = null;
396         int portNum = 0;
397         boolean isValidateOnly = true;
398 
399         try {
400             if (args.length < 6 || args.length > 8) {
401                 AutoModify
402                         .showUsage("You must provide either 6, 7 or 8 arguments.");
403             } else {
404 
405                 String[] hostPort = args[0].split(":");
406                 if (hostPort.length != 2) {
407                     AutoModify
408                             .showUsage("First argument must contain target"
409                                     + " Fedora server hostname and port using the syntax"
410                                     + " \"hostname:port\"");
411                 }
412                 hostName = hostPort[0];
413                 portNum = Integer.parseInt(hostPort[1]);
414                 username = args[1];
415                 password = args[2];
416                 directivesFilePath = args[3];
417                 logFilePath = args[4];
418                 protocol = args[5];
419 
420                 if (args.length >= 7) {
421                     isValidateOnly = false;
422                 } else {
423                     isValidateOnly = true;
424                 }
425 
426                 String context = Constants.FEDORA_DEFAULT_APP_CONTEXT;
427                 if (args.length == 8 && !args[7].equals("")){
428                     context = args[7];
429                 }
430 
431                 if (new File(directivesFilePath).exists()) {
432                     System.out.println("\nCONNECTING to Fedora server....");
433 
434                     // ******************************************
435                     // NEW: use new client utility class
436                     String baseURL =
437                             protocol + "://" + hostName + ":" + portNum
438                                     + "/" + context;
439                     FedoraClient fc =
440                             new FedoraClient(baseURL, username, password);
441                     APIA = fc.getAPIAMTOM();
442                     APIM = fc.getAPIMMTOM();
443                     //*******************************************
444 
445                     AutoModify am =
446                             new AutoModify(APIA,
447                                            APIM,
448                                            protocol,
449                                            hostName,
450                                            portNum,
451                                            context,
452                                            username,
453                                            password);
454 
455                     if (isValidateOnly) {
456                         System.out
457                                 .println("\n----- VALIDATING DIRECTIVES FILE ONLY -----\n");
458                     } else {
459                         System.out
460                                 .println("\n----- PROCESSING DIRECTIVES FILE -----\n");
461                     }
462                     am.modify(directivesFilePath, logFilePath, isValidateOnly);
463                 } else {
464                     AutoModify
465                             .showUsage("Directives input file does not exist: "
466                                     + directivesFilePath + " .");
467                 }
468             }
469         } catch (Exception e) {
470             AutoModify.showUsage(e.getClass().getName()
471                     + " - "
472                     + (e.getMessage() == null ? "(no detail provided)" : e
473                             .getMessage()));
474         }
475     }
