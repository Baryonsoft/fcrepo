52          good2go = true;

55      public final void process() throws Exception {
56          if (good2go) {
57              BatchAdditions batchAdditions = null;
58              BatchXforms batchXforms = null;
59              BatchIngest batchIngest = null;
60  
61              //make each phase
62  
63              if (miscProperties.getProperty(AGGREGATE) != null
64                      && miscProperties.getProperty(AGGREGATE).equals("yes")) {
65                  batchAdditions =
66                          new BatchAdditions(batchAdditionsValues,
67                                             datastreamProperties,
68                                             metadataProperties);
69              }
70              if (miscProperties.getProperty(DISCRETE) != null
71                      && miscProperties.getProperty(DISCRETE).equals("yes")) {
72                  batchXforms = new BatchXforms(batchXformsValues);
73              }
74              if (miscProperties.getProperty(EAT) != null
75                      && miscProperties.getProperty(EAT).equals("yes")) {
76                  batchIngest = new BatchIngest(batchIngestValues);
77              }
78  
79              //check in with each phase
80              if (miscProperties.getProperty(AGGREGATE) != null
81                      && miscProperties.getProperty(AGGREGATE).equals("yes")) {
82                  batchAdditions.prep();
83              }
84              if (miscProperties.getProperty(DISCRETE) != null
85                      && miscProperties.getProperty(DISCRETE).equals("yes")) {
86                  batchXforms.prep();
87              }
88              if (miscProperties.getProperty(EAT) != null
89                      && miscProperties.getProperty(EAT).equals("yes")) {
90                  batchIngest.prep();
91              }
92  
93              //perform each phase
94              if (miscProperties.getProperty(AGGREGATE) != null
95                      && miscProperties.getProperty(AGGREGATE).equals("yes")) {
96                  batchAdditions.process();
97              }
98  
99              Vector buildKeys = null;
100             if (miscProperties.getProperty(DISCRETE) == null
101                     || !miscProperties.getProperty(DISCRETE).equals("yes")) {
102                 buildKeys = new Vector();
103             } else {
104                 batchXforms.process();
105                 buildKeys = batchXforms.getKeys();
106             }
107 
108             Hashtable ingestMaps = null;
109             Vector ingestKeys = null;
110             if (miscProperties.getProperty(EAT) == null
111                     || !miscProperties.getProperty(EAT).equals("yes")) {
112                 ingestMaps = new Hashtable();
113                 ingestKeys = new Vector();
114             } else {
115                 batchIngest.process();
116                 ingestMaps = batchIngest.getPidMaps();
117                 ingestKeys = batchIngest.getKeys();
118             }
119 
120             String buildPath2directory =
121                     batchXformsValues.getProperty(BatchTool.ADDITIONSPATH);
122             String ingestPath2directory =
123                     batchIngestValues.getProperty(BatchTool.OBJECTSPATH);
124             String pidsPath = batchIngestValues.getProperty(BatchTool.PIDSPATH);
125 
126             String pidsFormat =
127                     miscProperties.getProperty(BatchTool.PIDSFORMAT);
128             String objectFormat =
129                     miscProperties.getProperty(BatchTool.OBJECTFORMAT);
130             PrintStream out = new PrintStream(new FileOutputStream(pidsPath)); //= System.err;
131 
132             //System.out.println("pidsFormat = [" + pidsFormat + "]");
133             if (pidsFormat.equals("xml")) {
134                 out.println("<" + XMLREPORTROOT + ">");
135             }
136             for (int i = 0, j = 0; i < buildKeys.size()
137                     || j < ingestKeys.size();) {
138                 //System.out.println("it = [" + i + "] [" + j + "]");
139                 String buildPath2file = "";
140                 String ingestPath2file = "";
141                 String pid = "";
142                 try {
143                     String buildFilename = "";
144                     String ingestFilename = "";
145                     int compared = 0;
146                     if (i < buildKeys.size()) {
147                         buildFilename = (String) buildKeys.get(i);
148                         compared = -1;
149                     }
150                     if (j < ingestKeys.size()) {
151                         ingestFilename = (String) ingestKeys.get(j);
152                         compared = 1;
153                     }
154                     if (i < buildKeys.size() & j < ingestKeys.size()) {
155                         compared = buildFilename.compareTo(ingestFilename);
156                     }
157                     if (compared <= 0) {
158                         buildPath2file =
159                                 buildPath2directory + File.separator
160                                         + buildFilename;
161                         ingestPath2file =
162                                 ingestPath2directory + File.separator
163                                         + buildFilename;
164                         i++;
165                     }
166                     if (compared >= 0) {
167                         ingestPath2file =
168                                 ingestPath2directory + File.separator
169                                         + ingestFilename;
170                         pid = (String) ingestMaps.get(ingestFilename);
171                         j++;
172                     }
173                     //System.out.println("no exceptions");
174                 } catch (Exception e) {
175                     System.out.println(e.getMessage());
176                     //System.out.println("exception = [" + e.getMessage() + "]");
177                     Thread.dumpStack();
178                     throw e;
179                 }
180 
181                 if (pidsFormat.equals("xml")) {
182                     //System.out.println("in loop, think it's xml i'm after]");
183                     out.print("\t<map ");
184                     if (buildPath2file != null && !buildPath2file.equals("")) {
185                         out.print("path2spec=\"" + buildPath2file + "\" ");
186                     }
187                     if (ingestPath2file != null && !ingestPath2file.equals("")) {
188                         out.print("path2object=\"" + ingestPath2file + "\" ");
189                     }
190                     if (pid != null && !pid.equals("")) {
191                         out.print("pid=\"" + pid + "\" ");
192                     }
193                     //out.print("\t<map path2spec=\"" + buildPath2file + "\" path2object=\"" + ingestPath2file + "\" pid=\"" + pid + "\" />");
194                     out.println("/>");
195                 } else if (pidsFormat.equals("text")) {
196                     out.println(buildPath2file + "\t" + ingestPath2file + "\t"
197                             + pid);
198                 }
199             }
200             if (pidsFormat.equals("xml")) {
201                 out.println("</" + XMLREPORTROOT + ">");
202             }
203             out.close();
204         }
205     }
