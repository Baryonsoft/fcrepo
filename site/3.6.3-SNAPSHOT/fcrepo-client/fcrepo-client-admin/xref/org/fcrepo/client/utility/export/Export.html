38      public static String getDuration(long millis) {
39          long tsec = millis / 1000;
40          long h = tsec / 60 / 60;
41          long m = (tsec - h * 60 * 60) / 60;
42          long s = tsec - h * 60 * 60 - m * 60;
43          StringBuffer out = new StringBuffer();
44          if (h > 0) {
45              out.append(h + " hour");
46              if (h > 1) {
47                  out.append('s');
48              }
49          }
50          if (m > 0) {
51              if (h > 0) {
52                  out.append(", ");
53              }
54              out.append(m + " minute");
55              if (m > 1) {
56                  out.append('s');
57              }
58          }
59          if (s > 0 || h == 0 && m == 0) {
60              if (h > 0 || m > 0) {
61                  out.append(", ");
62              }
63              out.append(s + " second");
64              if (s != 1) {
65                  out.append('s');
66              }
67          }
68          return out.toString();
69      }

71      public static void one(FedoraAPIAMTOM apia,
72                             FedoraAPIMMTOM apim,
73                             String pid,
74                             String format,
75                             String exportContext,
76                             File dir) throws Exception {
77          String suffix;
78          if (format.equals(ATOM_ZIP1_1.uri)) {
79              suffix = ".zip";
80          } else {
81              suffix = ".xml";
82          }
83          String fName = pid.replaceAll(":", "_") + suffix;
84          File file = new File(dir, fName);
85          System.out.println("Exporting " + pid + " to " + file.getPath());
86          AutoExporter.export(apia,
87                              apim,
88                              pid,
89                              format,
90                              exportContext,
91                              new FileOutputStream(file));
92      }

94      public static int multi(FedoraAPIAMTOM apia,
95                              FedoraAPIMMTOM apim,
96                              String format,
97                              String exportContext,
98                              File dir) throws Exception {
99          int count = 0;
100 
101         // prepare the FieldSearch query
102         FieldSearchQuery query = new FieldSearchQuery();
103         query.setTerms(null);
104 
105         ArrayOfString resultFields = new ArrayOfString();
106         resultFields.getItem().add("pid");
107 
108         // get the first chunk of search results
109 
110         FieldSearchResult result =
111                 AutoFinder.findObjects(apia, resultFields, 100, query);
112 
113         while (result != null && result.getResultList() != null) {
114 
115             List<ObjectFields> ofs = result.getResultList().getObjectFields();
116 
117             // export all objects from this chunk of search results
118             for (ObjectFields element : ofs) {
119                 String pid = element.getPid().getValue();
120                 one(apia, apim, pid, format, exportContext, dir);
121                 count++;
122             }
123 
124             // get the next chunk of search results, if any
125             String token = null;
126             try {
127                 token = result.getListSession().getValue().getToken();
128             } catch (Throwable th) {
129             }
130 
131             if (token != null) {
132                 result = AutoFinder.resumeFindObjects(apia, token);
133             } else {
134                 result = null;
135             }
136         }
137 
138         return count;
139     }

198      * Command-line interface for doing exports.

200     public static void main(String[] args) {
201         System.setProperty("java.awt.headless", "true");
202         try {
203             // USAGE: fedora-export host:port user password pid|ftyps format econtext path protocol [context]
204             if (args.length < 8 || args.length > 9) {
205                 Export.badArgs("Wrong number of arguments.");
206             }
207             String[] hp = args[0].split(":");
208             if (hp.length != 2) {
209                 Export.badArgs("First arg must be of the form 'host:port'");
210             }
211 
212             //SDP - HTTPS
213             String protocol = args[7];
214             if (!protocol.equals("http") && !protocol.equals("https")) {
215                 Export.badArgs("protocol arg must be 'http' or 'https'");
216             }
217 
218             String context = Constants.FEDORA_DEFAULT_APP_CONTEXT;
219             if (args.length == 9 && !args[8].equals("")) {
220                 context = args[8];
221             }
222 
223             // ******************************************
224             // NEW: use new client utility class
225             String baseURL =
226                     protocol + "://" + hp[0] + ":" + Integer.parseInt(hp[1])
227                             + "/" + context;
228             FedoraClient fc = new FedoraClient(baseURL, args[1], args[2]);
229             FedoraAPIAMTOM sourceRepoAPIA = fc.getAPIAMTOM();
230             FedoraAPIMMTOM sourceRepoAPIM = fc.getAPIMMTOM();
231             //*******************************************
232 
233             String exportFormat = args[4];
234             String exportContext = args[5];
235             if (!exportFormat.equals(FOXML1_1.uri)
236                     && !exportFormat.equals(FOXML1_0.uri)
237                     && !exportFormat.equals(METS_EXT1_1.uri)
238                     && !exportFormat.equals(METS_EXT1_0.uri)
239                     && !exportFormat.equals(ATOM1_1.uri)
240                     && !exportFormat.equals(ATOM_ZIP1_1.uri)
241                     && !exportFormat.equals("default")) {
242                 Export.badArgs(exportFormat + " is not a valid export format.");
243             }
244             if (!exportContext.equals("public")
245                     && !exportContext.equals("migrate")
246                     && !exportContext.equals("archive")
247                     && !exportContext.equals("default")) {
248                 Export
249                         .badArgs("econtext arg must be 'public', 'migrate', 'archive', or 'default'");
250             }
251 
252             RepositoryInfo repoinfo = sourceRepoAPIA.describeRepository();
253             StringTokenizer stoken =
254                     new StringTokenizer(repoinfo.getRepositoryVersion(), ".");
255             int majorVersion = new Integer(stoken.nextToken()).intValue();
256             if (majorVersion < 2 // pre-2.0 repo
257                     && !exportFormat.equals(METS_EXT1_0.uri)
258                     && !exportFormat.equals("default")) {
259                 Export.badArgs("format arg must be '" + METS_EXT1_0.uri
260                         + "' or 'default' for pre-2.0 repository.");
261             }
262 
263             if (exportFormat.equals("default")) {
264                 exportFormat = null;
265             }
266             if (exportContext.equals("default")) {
267                 exportContext = null;
268             }
269             if (args[3].indexOf(":") == -1) {
270                 // assume args[3] is FTYPS... so multi-export
271                 int count =
272                         Export.multi(sourceRepoAPIA,
273                                      sourceRepoAPIM,
274                                      exportFormat,
275                                      exportContext,
276                                      //args[4], // format
277                                      //args[5], // export context
278                                      new File(args[6])); // path
279                 System.out.print("Exported " + count + " objects.");
280             } else {
281                 // assume args[3] is a PID...they only want to export one object
282 
283                 Export.one(sourceRepoAPIA, sourceRepoAPIM, args[3], // PID
284                            exportFormat,
285                            exportContext,
286                            //args[4], // format
287                            //args[5], // export context
288                            new File(args[6])); // path
289                 System.out.println("Exported " + args[3]);
290             }
291         } catch (Exception e) {
292             System.err.print("Error  : " + e);
293             if (e.getMessage() == null) {
294                 e.printStackTrace();
295             } else {
296                 System.err.print(e.getMessage());
297             }
298         }
299     }
