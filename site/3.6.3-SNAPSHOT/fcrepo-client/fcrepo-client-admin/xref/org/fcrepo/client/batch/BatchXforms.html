48      BatchXforms(Properties optValues)
49              throws Exception {
50  
51          // The template format controls which xsl stylesheet is to be applied
52          // Valid values are METS_EXT1_1.uri and FOXML1_1.uri
53          if (optValues.getProperty(BatchTool.OBJECTFORMAT)
54                  .equalsIgnoreCase(FOXML1_1.uri)) {
55              xformPath = Constants.FEDORA_HOME + "/client/lib/foxml-merge.xsl";
56              formatMap.put(BatchTool.OBJECTFORMAT, "FOXML");
57          } else if (optValues.getProperty(BatchTool.OBJECTFORMAT)
58                  .equalsIgnoreCase(METS_EXT1_1.uri)) {
59              xformPath = Constants.FEDORA_HOME + "/client/lib/mets-merge.xsl";
60              formatMap.put(BatchTool.OBJECTFORMAT, "METS");
61          } else {
62              System.err.println("Unknown objectTemplate format: "
63                      + optValues.getProperty(BatchTool.OBJECTFORMAT));
64              throw new Exception();
65          }
66          additionsPath = optValues.getProperty(BatchTool.ADDITIONSPATH);
67          objectsPath = optValues.getProperty(BatchTool.OBJECTSPATH);
68          modelPath = optValues.getProperty(BatchTool.CMODEL);
69          if (!BatchTool.argOK(additionsPath)) {
70              System.err.println("additionsPath required");
71              throw new Exception();
72          }
73          if (!BatchTool.argOK(objectsPath)) {
74              System.err.println("objectsPath required");
75              throw new Exception();
76          }
77          if (!BatchTool.argOK(xformPath)) {
78              System.err.println("xformPath required");
79              throw new Exception();
80          }
81  
82      }
