7   import java.io.IOException;
8   import java.io.InputStream;

10  import javax.media.jai.JAI;

12  import javax.swing.ImageIcon;
13  import javax.swing.JComponent;
14  import javax.swing.JLabel;
15  import javax.swing.JScrollPane;

17  import com.sun.media.jai.codec.MemoryCacheSeekableStream;
18  
19  /**
20   * Views content of common image formats in a JComponent.
21   */
22  public class ImageContentViewer
23          extends ContentViewer {
24  
25      private JLabel m_label;
26  
27      private JScrollPane m_pane;
28  
29      private boolean s_registered;
30  
31      public ImageContentViewer() {
32          if (!s_registered) {
33              ContentHandlerFactory.register(this);
34              s_registered = true;
35          }
36      }
37  
38      /**
39       * Get the JComponent.
40       */
41      @Override
42      public JComponent getComponent() {
43          return m_pane;
44      }
45  
46      /**
47       * Returns a list of content types that this component can handle. This will
48       * usually be a list of MIME Types, but may also include other notions of
49       * type known to be understood by the users of ContentHandlerFactory.
50       */
51      @Override
52      public String[] getTypes() {
53          return new String[] {"image/gif", "image/jpeg", "image/tiff",
54                  "image/bmp", "image/x-ms-bmp", "image/x-bitmap", "image/png"};
55      }
56  
57      /**
58       * Initializes the handler. This should only be called once per instance,
59       * and is guaranteed to have been called when this component is provided by
60       * the ContentHandlerFactory. The viewOnly parameter signals to
61       * ContentEditor implementations that editing capabilities are not desired
62       * by the caller.
63       */
64      @Override
65      public void init(String type, InputStream data, boolean viewOnly)
66              throws IOException {
67          setContent(data);
68      }
69  
70      /**
71       * Re-initializes the handler given new input data. The old data can be
72       * discarded.
73       */
74      @Override
75      public void setContent(InputStream data) throws IOException {
76          try {
77              ImageIcon image =
78                      new ImageIcon(JAI
79                              .create("stream",
80                                      new MemoryCacheSeekableStream(data))
81                              .getAsBufferedImage());
82              if (m_pane == null) {
83                  m_label = new JLabel(image);
84                  m_pane = new JScrollPane(m_label);
85              } else {
86                  m_label.setIcon(image);
87              }
88          } catch (Exception e) {
89              throw new IOException(e.getMessage());
90          }
91      }
92  
93  }
