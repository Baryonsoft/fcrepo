279     public void buildBatch() {
280         try {
281             if (!m_templateField.getText().equals("")
282                     && !m_specsField.getText().equals("")
283                     && !m_objectsField.getText().equals("")
284                     && !m_pidsField.getText().equals("")) {
285                 Properties properties = new Properties();
286                 properties.setProperty("merge-objects", "yes");
287                 properties.setProperty("ingest", "yes");
288                 properties.setProperty("template", m_templateField.getText());
289                 properties.setProperty("specifics", m_specsField.getText());
290                 properties.setProperty("objects", m_objectsField.getText());
291                 properties.setProperty("ingested-pids", m_pidsField.getText());
292                 properties.setProperty("pids-format",
293                                        m_xmlMap.isSelected() ? "xml" : "text");
294                 properties.setProperty("server-fqdn", host);
295                 properties.setProperty("server-port", port);
296                 properties.setProperty("username", user);
297                 properties.setProperty("password", pass);
298                 properties.setProperty("server-protocol", Administrator
299                         .getProtocol());
300                 properties.setProperty("context", context);
301 
302                 // Verify format of template file to see if it is a METS or
303                 // FOXML template
304                 BufferedReader br =
305                         new BufferedReader(new FileReader(m_templateField
306                                 .getText()));
307                 String line;
308                 String objectFormat = null;
309                 while ((line = br.readLine()) != null) {
310                     if (line.indexOf("<foxml:") != -1) {
311                         objectFormat = FOXML1_1.uri;
312                         break;
313                     }
314                     if (line.indexOf("<METS:") != -1) {
315                         objectFormat = METS_EXT1_1.uri;
316                         break;
317                     }
318                 }
319                 br.close();
320                 br = null;
321 
322                 // Verify that object format of template file is either foxml or
323                 // mets.
324                 // If objectFormat is null, format could not be determined from
325                 // the template file.
326                 if (objectFormat == null) {
327                     JOptionPane
328                             .showMessageDialog(Administrator.getDesktop(),
329                                                "Template "
330                                                        + "object file format not recognized as either \""
331                                                        + FOXML1_1.uri
332                                                        + "\" or \""
333                                                        + METS_EXT1_1.uri
334                                                        + "\".\n"
335                                                        + "Please verify contents of template file.",
336                                                "Unknown Object Format",
337                                                JOptionPane.OK_OPTION);
338                     throw new Exception("Unknown object format in template file.");
339                 }
340                 // Set object format property based on template file format.
341                 // Query user to be sure they have selected the correct template
342                 // file.
343                 properties.setProperty("object-format", objectFormat);
344                 int n =
345                         JOptionPane
346                                 .showConfirmDialog(Administrator.getDesktop(),
347                                                    "Based on template, "
348                                                            + "generated objects will be in "
349                                                            + formatMap
350                                                                    .get(objectFormat)
351                                                            + " format. Is this correct?",
352                                                    "Generated Object Format",
353                                                    JOptionPane.YES_NO_CANCEL_OPTION);
354                 if (n == JOptionPane.YES_OPTION) {
355 
356                     batchOutput.setDirectoryPath(properties
357                             .getProperty("ingested-pids")); // 2003.12.03 niebel
358                     // -- duplicate
359                     // output to file
360 
361                     try {
362                         mdiDesktopPane.add(batchOutput);
363                     } catch (Exception eee) { // illegal component position
364                         // occurs ~ every other time ?!?
365                         mdiDesktopPane.add(batchOutput);
366                     }
367                     try {
368                         batchOutput.setSelected(true);
369                     } catch (java.beans.PropertyVetoException e) {
370                         System.err.println("BatchBuildIngestGUI"
371                                 + " frame select vetoed " + e.getMessage());
372                     }
373                     BatchThread batchThread = null;
374                     try {
375                         batchThread =
376                                 new BatchThread(batchOutput,
377                                                 batchOutput.getJTextArea(),
378                                                 "Building and Ingesting Batch . . .");
379                     } catch (Exception e) {
380                         System.err.println("BatchBuildIngestGUI"
381                                 + " couldn't instantiate BatchThread "
382                                 + e.getMessage());
383                     }
384                     batchThread.setProperties(properties);
385                     batchThread.start();
386                 }
387             }
388         } catch (Exception e) {
389             System.err.println("BatchBuildIngestGUI" + " general error "
390                     + e.getMessage());
391         }
392     }

394     protected File selectFile(File lastDir, boolean directoriesOnly)
395             throws Exception {
396         File selection = null;
397         JFileChooser browse;
398         if (Administrator.batchtoolLastDir == null) {
399             browse = new JFileChooser();
400         } else {
401             browse = new JFileChooser(Administrator.batchtoolLastDir);
402         }
403         if (directoriesOnly) {
404             browse.setFileSelectionMode(JFileChooser.DIRECTORIES_ONLY);
405         }
406         int returnVal = browse.showOpenDialog(this);
407         if (returnVal == JFileChooser.APPROVE_OPTION) {
408             selection = browse.getSelectedFile();
409             Administrator.batchtoolLastDir = selection.getParentFile(); // remember
410             // the
411             // dir
412             // for
413             // next
414             // time
415         }
416         return selection;
417     }
