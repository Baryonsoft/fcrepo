285     @Override
286     public boolean isDirty() {
287         if (m_done) {
288             return false;
289         }
290         int stateIndex = 0;
291         if (m_mostRecent.getState().equals("I")) {
292             stateIndex = 1;
293         }
294         if (m_mostRecent.getState().equals("D")) {
295             stateIndex = 2;
296         }
297         if (stateIndex != m_stateComboBox.getSelectedIndex()) {
298             return true;
299         }
300         int versionableIndex =
301                 m_mostRecent.isVersionable() ? NEW_VERSION_ON_UPDATE
302                         : REPLACE_ON_UPDATE;
303         if (versionableIndex != m_versionableComboBox.getSelectedIndex()) {
304             return true;
305         }
306         if (m_currentVersionPane.isDirty()) {
307             return true;
308         }
309         return false;
310     }

335     @Override
336     public void saveChanges(String logMessage) throws Exception {
337         String state = null;
338         int i = m_stateComboBox.getSelectedIndex();
339         state = s_stateComboBoxValues[i]; // "A" "I"  or "D"
340         if (!state.equals(m_mostRecent.getState())) {
341             Administrator.APIM.setDatastreamState(m_pid,
342                                                   m_mostRecent.getID(),
343                                                   state,
344                                                   logMessage);
345         }
346         if ((m_mostRecent.isVersionable() ? NEW_VERSION_ON_UPDATE
347                 : REPLACE_ON_UPDATE) != m_versionableComboBox
348                 .getSelectedIndex()) {
349             boolean newVersionableSetting =
350                     m_versionableComboBox.getSelectedIndex() == NEW_VERSION_ON_UPDATE ? true
351                             : false;
352             Administrator.APIM.setDatastreamVersionable(m_pid,
353                                                         m_mostRecent.getID(),
354                                                         newVersionableSetting,
355                                                         logMessage);
356         }
357         if (m_currentVersionPane.isDirty()) {
358             // defer to the currentVersionPane if anything else changed
359             try {
360                 m_currentVersionPane.saveChanges(logMessage, false);
361             } catch (Exception e) {
362                 if (e.getMessage() == null
363                         || e.getMessage().indexOf(" would invalidate ") == -1) {
364                     throw e;
365                 }
366                 // ask if they want to force it.
367                 Object[] options = {"Yes", "No"};
368                 int selected =
369                         JOptionPane
370                                 .showOptionDialog(null,
371                                                   e.getMessage()
372                                                           + "\n\nForce it?",
373                                                   "Warning",
374                                                   JOptionPane.DEFAULT_OPTION,
375                                                   JOptionPane.WARNING_MESSAGE,
376                                                   null,
377                                                   options,
378                                                   options[1]);
379                 if (selected == 0) {
380                     m_currentVersionPane.saveChanges(logMessage, true);
381                 }
382             }
383         }
384     }

400         } else if (m_mostRecent.getState().equals("D")) {

402             m_stateComboBox.setBackground(Administrator.DELETED_COLOR);
403         }

405                 : REPLACE_ON_UPDATE) != m_versionableComboBox
406                 .getSelectedIndex()) {
407             m_versionableComboBox
408                     .setSelectedIndex(m_mostRecent.isVersionable() ? NEW_VERSION_ON_UPDATE
409                             : REPLACE_ON_UPDATE);
410         }
411         m_owner.colorTabForState(m_mostRecent.getID(), m_mostRecent.getState());
412         m_currentVersionPane.undoChanges();
413     }

415     protected String getFedoraURL(Datastream ds, boolean withDate) {
416         StringBuffer buf = new StringBuffer();
417         buf.append(Administrator.getProtocol() + "://");
418         buf.append(Administrator.getHost());
419         if (Administrator.getPort() != 80) {
420             buf.append(':');
421             buf.append(Administrator.getPort());
422         }
423         buf.append("/" + Administrator.getAppServContext() + "/get/");
424         buf.append(m_pid);
425         buf.append('/');
426         buf.append(ds.getID());
427         if (withDate) {
428             buf.append('/');
429             buf.append(ds.getCreateDate());
430         }
431         return buf.toString();
432     }
