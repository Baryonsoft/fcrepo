45      // if logMessage is null, will use original path in logMessage
46      public static String oneFromFile(File file,
47                                       String ingestFormat,
48                                       FedoraAPIAMTOM targetRepoAPIA,
49                                       FedoraAPIMMTOM targetRepoAPIM,
50                                       String logMessage) throws Exception {
51          LAST_PATH = file.getPath();
52          String pid =
53                  AutoIngestor.ingestAndCommit(targetRepoAPIA,
54                                               targetRepoAPIM,
55                                               new FileInputStream(file),
56                                               ingestFormat,
57                                               getMessage(logMessage, file));
58          return pid;
59      }

65      public static void multiFromDirectory(File dir,
66                                            String ingestFormat,
67                                            FedoraAPIAMTOM targetRepoAPIA,
68                                            FedoraAPIMMTOM targetRepoAPIM,
69                                            String logMessage,
70                                            PrintStream log,
71                                            IngestCounter c) throws Exception {
72          File[] files = dir.listFiles();
73          if (files == null) {
74              throw new RuntimeException("Could not read files from directory "
75                      + dir.getPath());
76          }
77  
78          Arrays.sort(files, _FILE_COMPARATOR);
79          for (File element : files) {
80              if (!element.isHidden() && !element.getName().startsWith(".")) {
81                  if (element.isDirectory()) {
82                      multiFromDirectory(element,
83                                         ingestFormat,
84                                         targetRepoAPIA,
85                                         targetRepoAPIM,
86                                         logMessage,
87                                         log,
88                                         c);
89                  } else {
90                      try {
91                          String pid =
92                                  oneFromFile(element,
93                                              ingestFormat,
94                                              targetRepoAPIA,
95                                              targetRepoAPIM,
96                                              logMessage);
97                          c.successes++;
98                          IngestLogger.logFromFile(log, element, pid);
99                      } catch (Exception e) {
100                         // failed... just log it and continue
101                         c.failures++;
102                         IngestLogger.logFailedFromFile(log, element, e);
103                     }
104                 }
105             }
106         }
107     }

113     // if logMessage is null, will make informative one up
114     public static String oneFromRepository(FedoraAPIAMTOM sourceRepoAPIA,
115                                            FedoraAPIMMTOM sourceRepoAPIM,
116                                            String sourceExportFormat,
117                                            String pid,
118                                            FedoraAPIAMTOM targetRepoAPIA,
119                                            FedoraAPIMMTOM targetRepoAPIM,
120                                            String logMessage) throws Exception {
121 
122         // EXPORT from source repository
123         // The export context is set to "migrate" since the intent
124         // of ingest from repository is to migrate an object from
125         // one repository to another.  The "migrate" option will
126         // ensure that URLs that were relative to the "exporting"
127         // repository are made relative to the "importing" repository.
128         ByteArrayOutputStream out = new ByteArrayOutputStream();
129         AutoExporter.export(sourceRepoAPIA,
130                             sourceRepoAPIM,
131                             pid,
132                             sourceExportFormat,
133                             "migrate",
134                             out);
135 
136         // Convert old format values to URIs for ingest
137         String ingestFormat = sourceExportFormat;
138         if (sourceExportFormat.equals(METS_EXT1_0_LEGACY)) {
139             ingestFormat = METS_EXT1_0.uri;
140         } else if (sourceExportFormat.equals(FOXML1_0_LEGACY)) {
141             ingestFormat = FOXML1_0.uri;
142         }
143 
144         // INGEST into target repository
145         String realLogMessage = logMessage;
146         if (realLogMessage == null) {
147             realLogMessage = "Ingested from source repository with pid " + pid;
148         }
149         return AutoIngestor.ingestAndCommit(targetRepoAPIA,
150                                             targetRepoAPIM,
151                                             new ByteArrayInputStream(out
152                                                     .toByteArray()),
153                                             ingestFormat,
154                                             realLogMessage);
155     }

157     public static void multiFromRepository(String sourceProtocol,
158                                            String sourceHost,
159                                            int sourcePort,
160                                            FedoraAPIAMTOM sourceRepoAPIA,
161                                            FedoraAPIMMTOM sourceRepoAPIM,
162                                            String sourceExportFormat,
163                                            FedoraAPIAMTOM targetRepoAPIA,
164                                            FedoraAPIMMTOM targetRepoAPIM,
165                                            String logMessage,
166                                            PrintStream log,
167                                            IngestCounter c) throws Exception {
168         // prepare the FieldSearch query
169         FieldSearchQuery query = new FieldSearchQuery();
170         FieldSearchQuery.Conditions condi = new FieldSearchQuery.Conditions();
171         org.fcrepo.server.types.gen.Condition cond =
172                 new org.fcrepo.server.types.gen.Condition();
173         cond.setProperty("pid");
174         cond.setOperator(ComparisonOperator.fromValue("has"));
175         condi.getCondition().add(cond);
176         org.fcrepo.server.types.gen.ObjectFactory factory =
177                 new org.fcrepo.server.types.gen.ObjectFactory();
178         query.setConditions(factory.createFieldSearchQueryConditions(condi));
179         query.setTerms(null);
180         ArrayOfString resultFields = new ArrayOfString();
181         resultFields.getItem().add("pid");
182 
183         // get the first chunk of search results
184         FieldSearchResult result =
185                 AutoFinder
186                         .findObjects(sourceRepoAPIA, resultFields, 100, query);
187 
188         while (result != null && result.getResultList() != null) {
189 
190             List<ObjectFields> ofs = result.getResultList().getObjectFields();
191 
192             // ingest all objects from this chunk of search results
193             for (ObjectFields element : ofs) {
194                 String pid = element.getPid().getValue();
195                 try {
196                     String newPID =
197                             oneFromRepository(sourceRepoAPIA,
198                                               sourceRepoAPIM,
199                                               sourceExportFormat,
200                                               pid,
201                                               targetRepoAPIA,
202                                               targetRepoAPIM,
203                                               logMessage);
204                     c.successes++;
205                     IngestLogger.logFromRepos(log, pid, newPID);
206                 } catch (Exception e) {
207                     // failed... just log it and continue
208                     c.failures++;
209                     IngestLogger.logFailedFromRepos(log, pid, e);
210                 }
211             }
212 
213             // get the next chunk of search results, if any
214             String token = null;
215             try {
216                 token = result.getListSession().getValue().getToken();
217             } catch (Throwable th) {
218             }
219 
220             if (token != null) {
221                 result = AutoFinder.resumeFindObjects(sourceRepoAPIA, token);
222             } else {
223                 result = null;
224             }
225         }
226 
227     }

229     /**
230      * Determine the default export format of the source repository. For
231      * backward compatibility: with pre-2.0 repositories assume the
232      * "metslikefedora1" format
233      */
234     public static String getExportFormat(RepositoryInfo repoinfo)
235             throws Exception {
236         String sourceExportFormat = null;
237         StringTokenizer stoken =
238                 new StringTokenizer(repoinfo.getRepositoryVersion(), ".");
239         int majorVersion = new Integer(stoken.nextToken()).intValue();
240         if (majorVersion < 2) {
241             sourceExportFormat = METS_EXT1_0_LEGACY;
242         } else {
243             sourceExportFormat = repoinfo.getDefaultExportFormat();
244         }
245 
246         return sourceExportFormat;
247     }

256     // FIXME: this isn't ingest-specific... it doesn't belong here
257     public static String getDuration(long millis) {
258         long tsec = millis / 1000;
259         long h = tsec / 60 / 60;
260         long m = (tsec - h * 60 * 60) / 60;
261         long s = tsec - h * 60 * 60 - m * 60;
262         StringBuffer out = new StringBuffer();
263         if (h > 0) {
264             out.append(h + " hour");
265             if (h > 1) {
266                 out.append('s');
267             }
268         }
269         if (m > 0) {
270             if (h > 0) {
271                 out.append(", ");
272             }
273             out.append(m + " minute");
274             if (m > 1) {
275                 out.append('s');
276             }
277         }
278         if (s > 0 || h == 0 && m == 0) {
279             if (h > 0 || m > 0) {
280                 out.append(", ");
281             }
282             out.append(s + " second");
283             if (s != 1) {
284                 out.append('s');
285             }
286         }
287         return out.toString();
288     }

410     private static void summarize(IngestCounter counter, File logFile) {
411         System.out.println();
412         if (counter.failures > 0) {
413             System.out.println("WARNING: " + counter.failures + " of "
414                     + counter.getTotal() + " objects failed.  Check log.");
415         } else {
416             System.out.println("SUCCESS: All " + counter.getTotal()
417                     + " objects were ingested.");
418         }
419         System.out.println();
420         System.out.println("A detailed log is at " + logFile.getPath());
421     }

423     /**
424      * Command-line interface for doing ingests.
425      */
426     public static void main(String[] args) {
427         System.setProperty("java.awt.headless", "true");
428         try {
429             if (args.length < 1) {
430                 Ingest.badArgs("No arguments entered!");
431             }
432             String context = Constants.FEDORA_DEFAULT_APP_CONTEXT;
433             PrintStream log = null;
434             File logFile = null;
435             String logRootName = null;
436             IngestCounter counter = new IngestCounter();
437             char kind = args[0].toLowerCase().charAt(0);
438             if (kind == 'f') {
439                 // USAGE: fedora-ingest f[ile] path format targetHost:targetProtocol targetUser targetPassword targetProtocol [log] [context]
440                 if (args.length < 7 || args.length > 9) {
441                     Ingest.badArgs("Wrong number of arguments for file ingest.");
442                     System.out
443                             .println("USAGE: fedora-ingest f[ile] path format targetHost:targetProtocol targetUser targetPassword targetProtocol [log] [context]");
444                 }
445                 File f = new File(args[1]);
446                 String ingestFormat = args[2];
447                 String logMessage = null;
448 
449                 if (args.length == 8) {
450                     logMessage = args[7];
451                 }
452 
453                 if (args.length == 9 && !args[8].equals("")) {
454                     context = args[8];
455                 }
456 
457                 String protocol = args[6];
458                 String[] hp = args[3].split(":");
459 
460                 // ******************************************
461                 // NEW: use new client utility class
462                 String baseURL =
463                         protocol + "://" + hp[0] + ":"
464                                 + Integer.parseInt(hp[1]) + "/" + context;
465                 FedoraClient fc = new FedoraClient(baseURL, args[4], args[5]);
466                 FedoraAPIAMTOM targetRepoAPIA = fc.getAPIAMTOM();
467                 FedoraAPIMMTOM targetRepoAPIM = fc.getAPIMMTOM();
468                 //*******************************************
469 
470                 String pid =
471                         Ingest.oneFromFile(f,
472                                            ingestFormat,
473                                            targetRepoAPIA,
474                                            targetRepoAPIM,
475                                            logMessage);
476                 if (pid == null) {
477                     System.out.print("ERROR: ingest failed for file: "
478                             + args[1]);
479                 } else {
480                     System.out.println("Ingested pid: " + pid);
481                 }
482             } else if (kind == 'd') {
483                 // USAGE: fedora-ingest d[ir] path format targetHost:targetPort targetUser targetPassword targetProtocol [log] [context]
484                 if (args.length < 7 || args.length > 9) {
485                     Ingest.badArgs("Wrong number of arguments (" + args.length
486                             + ") for directory ingest.");
487                     System.out
488                             .println("USAGE: fedora-ingest d[ir] path format targetHost:targetPort targetUser targetPassword targetProtocol [log] [context]");
489                 }
490                 File d = new File(args[1]);
491                 String ingestFormat = args[2];
492                 String logMessage = null;
493 
494                 if (args.length == 8) {
495                     logMessage = args[7];
496                 }
497 
498                 if (args.length == 9 && !args[8].equals("")) {
499                     context = args[8];
500                 }
501 
502                 String protocol = args[6];
503                 String[] hp = args[3].split(":");
504 
505                 // ******************************************
506                 // NEW: use new client utility class
507                 String baseURL =
508                         protocol + "://" + hp[0] + ":"
509                                 + Integer.parseInt(hp[1]) + "/" + context;
510                 FedoraClient fc = new FedoraClient(baseURL, args[4], args[5]);
511                 FedoraAPIAMTOM targetRepoAPIA = fc.getAPIAMTOM();
512                 FedoraAPIMMTOM targetRepoAPIM = fc.getAPIMMTOM();
513                 //*******************************************
514 
515                 logRootName = "ingest-from-dir";
516                 logFile = IngestLogger.newLogFile(logRootName);
517                 log =
518                         new PrintStream(new FileOutputStream(logFile),
519                                         true,
520                                         "UTF-8");
521                 IngestLogger.openLog(log, logRootName);
522                 Ingest.multiFromDirectory(d,
523                                           ingestFormat,
524                                           targetRepoAPIA,
525                                           targetRepoAPIM,
526                                           logMessage,
527                                           log,
528                                           counter);
529                 IngestLogger.closeLog(log, logRootName);
530                 summarize(counter, logFile);
531             } else if (kind == 'r') {
532                 // USAGE: fedora-ingest r[epos] sourceHost:sourcePort sourceUser sourcePassword pid|* targetHost:targetPort targetUser targetPassword sourceProtocol targetProtocol [log] [context]
533                 if (args.length < 10 || args.length > 12) {
534                     Ingest.badArgs("Wrong number of arguments for repository ingest.");
535                 }
536                 String logMessage = null;
537                 if (args.length == 11) {
538                     logMessage = args[10];
539                 }
540 
541                 if (args.length == 12 && !args[11].equals("")) {
542                     context = args[11];
543                 }
544                 //Source repository
545                 String[] shp = args[1].split(":");
546                 String source_host = shp[0];
547                 String source_port = shp[1];
548                 String source_user = args[2];
549                 String source_password = args[3];
550                 String source_protocol = args[8];
551 
552                 // ******************************************
553                 // NEW: use new client utility class
554                 String sourceBaseURL =
555                         source_protocol + "://" + source_host + ":"
556                                 + Integer.parseInt(source_port) + "/" + context;
557                 FedoraClient sfc =
558                         new FedoraClient(sourceBaseURL,
559                                          source_user,
560                                          source_password);
561                 FedoraAPIAMTOM sourceRepoAPIA = sfc.getAPIAMTOM();
562                 FedoraAPIMMTOM sourceRepoAPIM = sfc.getAPIMMTOM();
563                 //*******************************************
564 
565                 //Target repository
566                 String[] thp = args[5].split(":");
567                 String target_host = thp[0];
568                 String target_port = thp[1];
569                 String target_user = args[6];
570                 String target_password = args[7];
571                 String target_protocol = args[9];
572 
573                 // ******************************************
574                 // NEW: use new client utility class
575                 String targetBaseURL =
576                         target_protocol + "://" + target_host + ":"
577                                 + Integer.parseInt(target_port) + "/" + context;
578                 FedoraClient tfc =
579                         new FedoraClient(targetBaseURL,
580                                          target_user,
581                                          target_password);
582                 FedoraAPIAMTOM targetRepoAPIA = tfc.getAPIAMTOM();
583                 FedoraAPIMMTOM targetRepoAPIM = tfc.getAPIMMTOM();
584                 //*******************************************
585 
586                 // Determine export format
587                 RepositoryInfo repoinfo = sourceRepoAPIA.describeRepository();
588                 System.out
589                         .println("Ingest: exporting from a source repo version "
590                                 + repoinfo.getRepositoryVersion());
591                 String sourceExportFormat = getExportFormat(repoinfo);
592                 System.out.println("Ingest: source repo is using "
593                         + sourceExportFormat + " export format.");
594 
595                 if (args[4].indexOf(":") != -1) {
596                     // single object
597                     String successfulPID =
598                             Ingest.oneFromRepository(sourceRepoAPIA,
599                                                      sourceRepoAPIM,
600                                                      sourceExportFormat,
601                                                      args[4],
602                                                      targetRepoAPIA,
603                                                      targetRepoAPIM,
604                                                      logMessage);
605                     if (successfulPID == null) {
606                         System.out
607                                 .print("ERROR: ingest from repo failed for pid="
608                                         + args[4]);
609                     } else {
610                         System.out.println("Ingested pid: " + successfulPID);
611                     }
612                 } else {
613                     // multi-object
614                     //hp=args[1].split(":");
615                     logRootName = "ingest-from-repository";
616                     logFile = IngestLogger.newLogFile(logRootName);
617                     log =
618                             new PrintStream(new FileOutputStream(logFile),
619                                             true,
620                                             "UTF-8");
621                     IngestLogger.openLog(log, logRootName);
622                     Ingest.multiFromRepository(source_protocol,
623                                                source_host,
624                                                Integer.parseInt(source_port),
625                                                sourceRepoAPIA,
626                                                sourceRepoAPIM,
627                                                sourceExportFormat,
628                                                targetRepoAPIA,
629                                                targetRepoAPIM,
630                                                logMessage,
631                                                log,
632                                                counter);
633                     IngestLogger.closeLog(log, logRootName);
634                     summarize(counter, logFile);
635                 }
636 
637             } else {
638                 Ingest.badArgs("First argument must start with f, d, or r.");
639             }
