174      * (non-Javadoc)
175      * @see
176      * org.fcrepo.server.security.xacml.pep.rest.filters.RESTFilter#handleResponse(javax.servlet
177      * .http.HttpServletRequest, javax.servlet.http.HttpServletResponse)
178      */
179     @Override
180     public RequestCtx handleResponse(HttpServletRequest request,
181                                      HttpServletResponse response)
182             throws IOException, ServletException {
183         if (request.getParameter("terms") == null
184                 && request.getParameter("query") == null
185                 && request.getParameter("sessionToken") == null) {
186             return null;
187         }
188 
189         DataResponseWrapper res = (DataResponseWrapper) response;
190         byte[] data = res.getData();
191 
192         String result = null;
193         String body = new String(data);
194 
195         if (body.startsWith("<html>")) {
196             logger.debug("filtering html");
197             result = filterHTML(request, res);
198         } else if (body.startsWith("<?xml")) {
199             logger.debug("filtering xml");
200             result = filterXML(request, res);
201         } else {
202             logger.debug("not filtering due to unexpected output: {}", body);
203             result = body;
204         }
205 
206         res.setData(result.getBytes());
207 
208         return null;
209     }

211     /**
212      * Parses an XML based response and removes the items that are not
213      * permitted.
214      *
215      * @param request
216      *        the http servlet request
217      * @param response
218      *        the http servlet response
219      * @return the new response body without non-permissable objects.
220      * @throws ServletException
221      */
222     private String filterXML(HttpServletRequest request,
223                              DataResponseWrapper response)
224             throws ServletException {
225         String body = new String(response.getData());
226         DocumentBuilder docBuilder = null;
227         Document doc = null;
228 
229         try {
230             synchronized(BUILDER_FACTORY){
231                 docBuilder =
232                         BUILDER_FACTORY.newDocumentBuilder();
233             }
234             doc = docBuilder.parse(new ByteArrayInputStream(response.getData()));
235         } catch (Exception e) {
236             throw new ServletException(e);
237         }
238 
239         XPath xpath;
240         synchronized(XPATH_FACTORY){
241             xpath = XPATH_FACTORY.newXPath();
242         }
243         xpath.setNamespaceContext(TYPES_NAMESPACE);
244         NodeList rows = null;
245         try {
246             rows =
247                     (NodeList) xpath
248                             .evaluate("/:result/:resultList/:objectFields/:pid",
249                                       doc,
250                                       XPathConstants.NODESET);
251         } catch (XPathExpressionException xpe) {
252             throw new ServletException("Error parsing XML for search results: ",
253                                        xpe);
254         }
255 
256         if (rows.getLength() == 0) {
257             logger.debug("No results to filter.");
258 
259             return body;
260         }
261 
262         Map<String, Node> pids = new HashMap<String, Node>();
263         for (int x = 0; x < rows.getLength(); x++) {
264             Node pid = rows.item(x);
265             pids.put(pid.getFirstChild().getNodeValue(), pid.getParentNode());
266         }
267 
268         Set<Result> results = evaluatePids(pids.keySet(), request, response);
269 
270         for (Result r : results) {
271             if (r.getResource() == null || "".equals(r.getResource())) {
272                 logger.warn("This resource has no resource identifier in the xacml response results!");
273             } else if (logger.isDebugEnabled()) {
274                 logger.debug("Checking: {}", r.getResource());
275             }
276 
277             String[] ridComponents = r.getResource().split("\\/");
278             String rid = ridComponents[ridComponents.length - 1];
279 
280             if (r.getStatus().getCode().contains(Status.STATUS_OK)
281                     && r.getDecision() != Result.DECISION_PERMIT) {
282                 Node node = pids.get(rid);
283                 node.getParentNode().removeChild(node);
284                 logger.debug("Removing: {} [{}]", r.getResource(), rid);
285             }
286         }
287         // since namespaces are disabled, set the attribute explicitly
288         doc.getDocumentElement().setAttribute("xmlns", "http://www.fedora.info/definitions/1/0/types/");
289         Source src = new DOMSource(doc);
290         ByteArrayOutputStream os = new ByteArrayOutputStream();
291         javax.xml.transform.Result dst = new StreamResult(os);
292         try {
293             xFormer.transform(src, dst);
294         } catch (TransformerException te) {
295             throw new ServletException("error generating output", te);
296         }
297 
298         return new String(os.toByteArray());
299     }

301     /**
302      * Parses an HTML based response and removes the items that are not
303      * permitted.
304      *
305      * @param request
306      *        the http servlet request
307      * @param response
308      *        the http servlet response
309      * @return the new response body without non-permissable objects.
310      * @throws ServletException
311      */
312     private String filterHTML(HttpServletRequest request,
313                               DataResponseWrapper response)
314             throws ServletException {
315         String body = new String(response.getData());
316 
317         InputStream is = new ByteArrayInputStream(body.getBytes());
318         Document doc = tidy.parseDOM(is, null);
319 
320         XPath xpath;
321         synchronized(XPATH_FACTORY) {
322             xpath = XPATH_FACTORY.newXPath();
323         }
324         NodeList rows = null;
325         try {
326             rows =
327                     (NodeList) xpath
328                             .evaluate("/html/body/center/center/table/tr",
329                                       doc,
330                                       XPathConstants.NODESET);
331         } catch (XPathExpressionException xpe) {
332             throw new ServletException("Error parsing HTML for search results: ",
333                                        xpe);
334         }
335 
336         // only the header row, no results.
337         if (rows.getLength() == 1) {
338             logger.debug("No results to filter.");
339             return body;
340         }
341 
342         NodeList headers = rows.item(0).getChildNodes();
343         int numHeaders = headers.getLength();
344         int pidHeader = -1;
345 
346         // ensure we have 'pid' in the list and also that it exists
347         for (int x = 0; x < numHeaders; x++) {
348             String header =
349                     headers.item(x).getFirstChild().getFirstChild()
350                             .getNodeValue();
351             if ("pid".equals(header)) {
352                 pidHeader = x;
353             }
354         }
355 
356         if (pidHeader == -1) {
357             throw new ServletException("pid field not in result list!");
358         }
359 
360         Map<String, Node> pids = new HashMap<String, Node>();
361 
362         // start from 1 to skip the header row.
363         for (int x = 1; x < rows.getLength(); x++) {
364             NodeList elements = rows.item(x).getChildNodes();
365             Node pidParentA = elements.item(pidHeader).getFirstChild();
366             if (pidParentA != null && pidParentA.getNodeName().equals("a")) {
367                 String pid = pidParentA.getFirstChild().getNodeValue();
368                 pids.put(pid, rows.item(x));
369             }
370         }
371 
372         Set<Result> results = evaluatePids(pids.keySet(), request, response);
373 
374         for (Result r : results) {
375             if (r.getResource() == null || "".equals(r.getResource())) {
376                 logger.warn("This resource has no resource identifier in the xacml response results!");
377             } else if (logger.isDebugEnabled()) {
378                 logger.debug("Checking: {}", r.getResource());
379             }
380 
381             String[] ridComponents = r.getResource().split("\\/");
382             String rid = ridComponents[ridComponents.length - 1];
383 
384             if (r.getStatus().getCode().contains(Status.STATUS_OK)
385                     && r.getDecision() != Result.DECISION_PERMIT) {
386                 Node node = pids.get(rid);
387                 node.getParentNode().removeChild(node.getNextSibling());
388                 node.getParentNode().removeChild(node);
389                 logger.debug("Removing: {} [{}]", r.getResource(), rid);
390             }
391         }
392 
393         Source src = new DOMSource(doc);
394         ByteArrayOutputStream os = new ByteArrayOutputStream();
395         javax.xml.transform.Result dst = new StreamResult(os);
396         try {
397             xFormer.transform(src, dst);
398         } catch (TransformerException te) {
399             throw new ServletException("error generating output", te);
400         }
401 
402         return new String(os.toByteArray());
403     }

405     /**
406      * Takes a given list of PID's and evaluates them.
407      *
408      * @param pids
409      *        the list of pids to check
410      * @param request
411      *        the http servlet request
412      * @param response
413      *        the http servlet resposne
414      * @return a set of XACML results.
415      * @throws ServletException
416      */
417     private Set<Result> evaluatePids(Set<String> pids,
418                                      HttpServletRequest request,
419                                      DataResponseWrapper response)
420             throws ServletException {
421         Set<String> requests = new HashSet<String>();
422         for (String pid : pids) {
423             logger.debug("Checking: {}", pid);
424 
425             Map<URI, AttributeValue> actions =
426                     new HashMap<URI, AttributeValue>();
427             Map<URI, AttributeValue> resAttr;
428 
429             try {
430                 actions
431                         .put(Constants.ACTION.ID.getURI(),
432                              Constants.ACTION.LIST_OBJECT_IN_FIELD_SEARCH_RESULTS
433                                      .getStringAttribute());
434                 actions.put(Constants.ACTION.API.getURI(),
435                             Constants.ACTION.APIA.getStringAttribute());
436                 resAttr = ResourceAttributes.getResources(pid);
437 
438                 RequestCtx req =
439                         getContextHandler()
440                                 .buildRequest(getSubjects(request),
441                                               actions,
442                                               resAttr,
443                                               getEnvironment(request));
444 
445                 String r = m_contextUtil.makeRequestCtx(req);
446                 logger.debug(r);
447 
448                 requests.add(r);
449             } catch (Exception e) {
450                 logger.error(e.getMessage(), e);
451                 throw new ServletException(e.getMessage(), e);
452             }
453         }
454 
455         String res = null;
456         ResponseCtx resCtx = null;
457         try {
458             logger.debug("Number of requests: {}", requests.size());
459 
460             res =
461                     getContextHandler().evaluateBatch(requests
462                             .toArray(new String[requests.size()]));
463 
464             logger.debug("Response: {}", res);
465 
466             resCtx = m_contextUtil.makeResponseCtx(res);
467         } catch (MelcoeXacmlException pe) {
468             throw new ServletException("Error evaluating pids: "
469                     + pe.getMessage(), pe);
470         }
471 
472         @SuppressWarnings("unchecked")
473         Set<Result> results = resCtx.getResults();
474 
475         return results;
476     }

478     static class TypesNamespaceContext implements NamespaceContext {
479         static List<String> PREFIXES = Arrays.asList(new String[]{"types",""});
480         static List<String> XSI_PREFIXES = Arrays.asList(new String[]{"xsi"});
481         static ArrayList<String> EMPTY = new ArrayList<String>(0);
482         @Override
483         public String getNamespaceURI(String prefix) {
484             if ("types".equals(prefix) || "".equals(prefix)) {
485                return "http://www.fedora.info/definitions/1/0/types/";
486             }
487             if ("xsi".equals(prefix)){
488                 return "http://www.w3.org/2001/XMLSchema-instance";
489             }
490             return "";
491         }
492 
493         @Override
494         public String getPrefix(String uri) {
495             if ("http://www.fedora.info/definitions/1/0/types/".equals(uri)) {
496                 return "types";
497             }
498             if ("http://www.w3.org/2001/XMLSchema-instance".equals(uri)){
499                 return "xsi";
500             }
501             return null;
502         }
503 
504         @Override
505         public Iterator getPrefixes(String uri) {
506             if("http://www.fedora.info/definitions/1/0/types/".equals(uri)) {
507                 return PREFIXES.iterator();
508             }
509             if ("http://www.w3.org/2001/XMLSchema-instance".equals(uri)){
510                 return XSI_PREFIXES.iterator();
511             }
512             return EMPTY.iterator();
513         }
