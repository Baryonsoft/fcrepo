51  /**
52   * This class handles the filtering of search results for both the findObjects
53   * and resumeFindObjects operations and is called by each of them.
54   *
55   * @author nishen@melcoe.mq.edu.au
56   */
57  public class FieldSearchResultHandler
58          extends AbstractOperationHandler {
59  
60      private static final Logger logger =
61              LoggerFactory.getLogger(FieldSearchResultHandler.class);
62  
63      private ContextUtil m_contextUtil = null;
64  
65      /**
66       * Default constructor.
67       *
68       * @throws PEPException
69       */
70      public FieldSearchResultHandler(ContextHandler contextHandler)
71              throws PEPException {
72          super(contextHandler);
73      }
74  
75      public void setContextUtil(ContextUtil contextUtil) {
76          m_contextUtil = contextUtil;
77      }
78  
79      /**
80       * Removes non-permissable objects from the result list.
81       *
82       * @param context
83       *        the message context
84       * @param result
85       *        the search result object
86       * @return the new search result object without non-permissable items
87       * @throws PEPException
88       */
89      public FieldSearchResult filter(SOAPMessageContext context,
90                                      FieldSearchResult result)
91              throws PEPException {
92          if (result == null || result.getResultList() == null || result.getResultList().getObjectFields() == null || result.getResultList().getObjectFields().isEmpty()) {
93              return result;
94          }
95          List<ObjectFields> objs = result.getResultList().getObjectFields();
96          List<String> requests = new ArrayList<String>();
97          Map<String, ObjectFields> objects = new HashMap<String, ObjectFields>();
98  
99  
100 
101         for (ObjectFields o : objs) {
102             if (logger.isDebugEnabled()) {
103                 logger.debug("Checking: " + o.getPid());
104             }
105 
106             Map<URI, AttributeValue> actions =
107                     new HashMap<URI, AttributeValue>();
108             Map<URI, AttributeValue> resAttr;
109 
110             String pid = o.getPid() != null ? o.getPid().getValue() : null;
111             if (pid != null && !"".equals(pid)) {
112                 objects.put(pid, o);
113 
114                 try {
115                     actions
116                             .put(Constants.ACTION.ID.getURI(),
117                                  new StringAttribute(Constants.ACTION.LIST_OBJECT_IN_FIELD_SEARCH_RESULTS
118                                          .getURI().toASCIIString()));
119                     actions.put(Constants.ACTION.API.getURI(),
120                                 new StringAttribute(Constants.ACTION.APIA.getURI()
121                                         .toASCIIString()));
122 
123                     resAttr = ResourceAttributes.getResources(pid);
124 
125                     RequestCtx req =
126                             getContextHandler()
127                                     .buildRequest(getSubjects(context),
128                                                   actions,
129                                                   resAttr,
130                                                   getEnvironment(context));
131 
132                     requests.add(m_contextUtil.makeRequestCtx(req));
133                 } catch (Exception e) {
134                     logger.error(e.getMessage(), e);
135                     throw new OperationHandlerException(e.getMessage(), e);
136                 }
137             }
138         }
139 
140         String response =
141                 getContextHandler().evaluateBatch(requests
142                         .toArray(new String[requests.size()]));
143         ResponseCtx resCtx;
144         try {
145             resCtx = m_contextUtil.makeResponseCtx(response);
146         } catch (MelcoeXacmlException e) {
147             throw new PEPException(e);
148         }
149 
150         @SuppressWarnings("unchecked")
151         Set<Result> results = resCtx.getResults();
152 
153         List<ObjectFields> resultObjects = new ArrayList<ObjectFields>();
154         for (Result r : results) {
155             if (r.getResource() == null || "".equals(r.getResource())) {
156                 logger.warn("This resource has no resource identifier in the xacml response results!");
157             } else {
158                 logger.debug("Checking: {}", r.getResource());
159             }
160 
161             String[] ridComponents = r.getResource().split("\\/");
162             String rid = ridComponents[ridComponents.length - 1];
163 
164             if (r.getStatus().getCode().contains(Status.STATUS_OK)
165                     && r.getDecision() == Result.DECISION_PERMIT) {
166                 ObjectFields tmp = objects.get(rid);
167                 if (tmp != null) {
168                     resultObjects.add(tmp);
169                     if (logger.isDebugEnabled()) {
170                         logger.debug("Adding: " + r.getResource() + "[" + rid
171                                 + "]");
172                     }
173                 } else {
174                     logger.warn("Not adding this object as no object found for this key: "
175                                     + r.getResource() + "[" + rid + "]");
176                 }
177             }
178         }
179         FieldSearchResult.ResultList rl = new FieldSearchResult.ResultList();
180         rl.getObjectFields().addAll(resultObjects);
181         result.setResultList(rl);
182         return result;
183     }

187      * @see
188      * org.fcrepo.server.security.xacml.pep.ws.operations.OperationHandler#handleRequest(SOAPMessageContext)
189      */
190     @Override
191     public RequestCtx handleRequest(SOAPMessageContext context)
192             throws OperationHandlerException {
193         RequestCtx req = null;
194 
195         Map<URI, AttributeValue> resAttr;
196         Map<URI, AttributeValue> actions = new HashMap<URI, AttributeValue>();
197 
198         try {
199             resAttr = ResourceAttributes.getRepositoryResources();
200 
201             actions.put(Constants.ACTION.ID.getURI(),
202                         new StringAttribute(Constants.ACTION.FIND_OBJECTS
203                                 .getURI().toASCIIString()));
204             actions.put(Constants.ACTION.API.getURI(),
205                         new StringAttribute(Constants.ACTION.APIA.getURI()
206                                 .toASCIIString()));
207 
208             req =
209                     getContextHandler().buildRequest(getSubjects(context),
210                                                      actions,
211                                                      resAttr,
212                                                      getEnvironment(context));
213         } catch (Exception e) {
214             logger.error(e.getMessage(), e);
215             throw new OperationHandlerException(e.getMessage(), e);
216         }
217 
218         return req;
219     }

221     /*
222      * (non-Javadoc)
223      * @see
224      * org.fcrepo.server.security.xacml.pep.ws.operations.OperationHandler#handleResponse(SOAPMessageContext)
225      */
226     @Override
227     public RequestCtx handleResponse(SOAPMessageContext context)
228             throws OperationHandlerException {
229         try {
230             Object response = getSOAPResponseObject(context);
231             org.fcrepo.server.types.gen.FieldSearchResult result =
232                 (org.fcrepo.server.types.gen.FieldSearchResult) callGetter("getResult",response);
233             result =
234                 filter(context, result);
235             Method setter = response.getClass().getDeclaredMethod("setResult",
236                     org.fcrepo.server.types.gen.FieldSearchResult.class);
237             setter.invoke(response, result);
238             setSOAPResponseObject(context, response);
239         } catch (Exception e) {
240             logger.error("Error filtering Objects", e);
241             throw new OperationHandlerException("Error filtering Objects", e);
242         }
243 
244         return null;
245     }
