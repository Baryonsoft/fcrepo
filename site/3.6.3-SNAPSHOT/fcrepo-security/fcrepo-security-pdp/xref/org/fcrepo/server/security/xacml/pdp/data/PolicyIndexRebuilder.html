27          Iterable<Datastream> policyDatastreams = object.datastreams(FedoraPolicyStore.FESL_POLICY_DATASTREAM);
28  
29          // try to get the latest policy datastream version
30          Date latest = null;
31          Datastream policyDatastream = null;
32          for (Datastream ds : policyDatastreams) {
33              if (latest == null || ds.DSCreateDT.after(latest)) {
34                  latest = ds.DSCreateDT;
35                  policyDatastream = ds;
36              }
37          }
38          // null means no policy datastream found
39          if (policyDatastream != null ) {
40              // add to cache
41              System.out.println("   Adding " + object.getPid() + " to index.");
42              String policy = new String(IOUtils.toByteArray(policyDatastream.getContentStream()), "UTF-8");
43              // TODO: PolicyIndex would benefit from methods that can accept streams
44              m_policyIndex.addPolicy(object.getPid(), policy);
45          }

49      }
