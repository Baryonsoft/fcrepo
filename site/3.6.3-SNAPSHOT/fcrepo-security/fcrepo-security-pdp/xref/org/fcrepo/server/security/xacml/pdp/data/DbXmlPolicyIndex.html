101      * (non-Javadoc)
102      * @seemelcoe.xacml.pdp.data.PolicyDataManager#getPolicies(com.sun.xacml.
103      * EvaluationCtx)
104      */
105     @Override
106     public Map<String, AbstractPolicy> getPolicies(EvaluationCtx eval, PolicyFinder policyFinder)
107             throws PolicyIndexException {
108         long a = 0;
109         long b = 0;
110         long total = 0;
111 
112         Map<String, AbstractPolicy> documents = new HashMap<String, AbstractPolicy>();
113 
114         XmlQueryExpression qe = null;
115         XmlQueryContext context = null;
116 
117         try {
118             // Get the query (query gets prepared if necesary)
119             a = System.nanoTime();
120 
121             Map<String, Set<AttributeBean>> attributeMap =
122                 getAttributeMap(eval);
123 
124             context = m_dbXmlManager.manager.createQueryContext();
125             context.setDefaultCollection(m_dbXmlManager.CONTAINER);
126 
127             for (String prefix : namespaces.keySet()) {
128                 context.setNamespace(prefix, namespaces.get(prefix));
129             }
130 
131 
132             // not clear why this is needed.... but it is used in hashing the queries
133             int resourceComponentCount = 0;
134             Map<String, String> variables = getXpathVariables(attributeMap);
135             for (String variable : variables.keySet()) {
136                 context.setVariableValue(variable, new XmlValue(variables.get(variable)));
137                 if (variable.equals(XACML_RESOURCE_ID)) {
138                     resourceComponentCount++;
139                 }
140             }
141 
142 
143             qe =
144                 getQuery(attributeMap, context, resourceComponentCount);
145 
146 
147         } catch (XmlException xe) {
148             throw new PolicyIndexException("Error while constructing query", xe);
149         } catch (URISyntaxException e) {
150             throw new PolicyIndexException("Error while constructing query", e);
151         }
152 
153         DbXmlManager.readLock.lock();
154         try {
155             b = System.nanoTime();
156             total += b - a;
157             if (log.isDebugEnabled()) {
158                 log.debug("Query prep. time: " + (b - a) + "ns");
159             }
160 
161             // execute the query
162             a = System.nanoTime();
163 
164             XmlResults results = qe.execute(context);
165 
166             b = System.nanoTime();
167             total += b - a;
168             if (log.isDebugEnabled()) {
169                 log.debug("Query exec. time: " + (b - a) + "ns");
170             }
171 
172             // process results
173             while (results.hasNext()) {
174 
175                 XmlValue value = results.next();
176                 byte[] content = value.asDocument().getContent();
177                 if (content.length > 0) {
178                     documents.put(value.asDocument().getName(),
179                                   handleDocument(m_policyReader.readPolicy(content),policyFinder));
180                 } else {
181                     throw new PolicyIndexException("Zero-length result found");
182                 }
183             }
184             results.delete();
185         } catch (XmlException xe) {
186             log.error("Error getting query results." + xe.getMessage());
187             throw new PolicyIndexException("Error getting query results." + xe.getMessage(), xe);
188         } catch (ParsingException pe) {
189             log.error("Error getting query results." + pe.getMessage());
190             throw new PolicyIndexException("Error getting query results." + pe.getMessage(), pe);
191         } finally {
192             DbXmlManager.readLock.unlock();
193         }
194 
195         if (log.isDebugEnabled()) {
196             log.debug("Total exec. time: " + total + "ns");
197         }
198 
199         return documents;
200     }

202     /**
203      * Either returns a query that has previously been generated, or generates a
204      * new one if it has not.
205      *
206      * @param attributeMap
207      *        the Map of attributes, type and values upon which this query is
208      *        based
209      * @param context
210      *        the context for the query
211      * @return an XmlQueryExpression that can be executed
212      * @throws XmlException
213      * @throws XmlException
214      * @throws PolicyIndexException
215      */
216     private XmlQueryExpression getQuery(Map<String, Set<AttributeBean>> attributeMap,
217                                         XmlQueryContext context,
218                                         int r) throws XmlException  {
219         // The dimensions for this query.
220         StringBuilder sb = new StringBuilder();
221         for (Set<AttributeBean> attributeBeans : attributeMap.values()) {
222             sb.append(attributeBeans.size() + ":");
223             for (AttributeBean bean : attributeBeans) {
224                 sb.append(bean.getValues().size() + "-");
225             }
226         }
227 
228         // If a query of these dimensions already exists, then just return it.
229         String hash = sb.toString() + r;
230         XmlQueryExpression result = m_queries.get(hash);
231         if (result != null) {
232             return result;
233         }
234 
235         // We do not have a query of those dimensions. We must make one.
236         String query = createQuery(attributeMap);
237 
238         if (log.isDebugEnabled()) {
239             log.debug("Query [" + hash + "]:\n" + query);
240         }
241 
242         // Once we have created a query, we can parse it and store the
243         // execution plan. This is an expensive operation that we do
244         // not want to have to do more than once for each dimension
245         result = m_dbXmlManager.manager.prepare(query, context);
246 
247         m_queries.put(hash, result);
248 
249         return result;
250     }

273     @Override
274     public String addPolicy(String name, String document)
275             throws PolicyIndexException {
276 
277         String docName = null;
278         DbXmlManager.writeLock.lock();
279         try {
280 
281             XmlDocument doc = makeDocument(name, document);
282             docName = doc.getName();
283             log.debug("Adding document: " + docName);
284             m_dbXmlManager.container.putDocument(doc,
285                                                m_dbXmlManager.updateContext);
286             setLastUpdate(System.currentTimeMillis());
287         } catch (XmlException xe) {
288             if (xe.getErrorCode() == XmlException.UNIQUE_ERROR) {
289                 throw new PolicyIndexException("Document already exists: "
290                                                + docName);
291             } else {
292                 throw new PolicyIndexException("Error adding policy: "
293                                                + xe.getMessage(), xe);
294             }
295         } finally {
296             DbXmlManager.writeLock.unlock();
297         }
298 
299         return docName;
300     }

310         log.debug("Deleting document: " + name);
311 
312         DbXmlManager.writeLock.lock();
313         try {
314             m_dbXmlManager.container.deleteDocument(name, m_dbXmlManager.updateContext);
315             setLastUpdate(System.currentTimeMillis());
316         } catch (XmlException xe) {
317             // safe delete - only warn if not found
318             if (xe.getDbError() == XmlException.DOCUMENT_NOT_FOUND){
319                 log.warn("Error deleting document: " + name + " - document does not exist");
320             } else {
321                 throw new PolicyIndexException("Error deleting document: " + name + xe.getMessage(), xe);
322             }
323         } finally {
324             DbXmlManager.writeLock.unlock();
325         }
326         return true;
327     }

414     @Override
415     public boolean contains(String policyName)
416             throws PolicyIndexException {
417         log.debug("Determining if document exists: " + policyName);
418         DbXmlManager.readLock.lock();
419         try {
420             m_dbXmlManager.container.getDocument(policyName,
421                                                new XmlDocumentConfig().setLazyDocs(true));
422         } catch (XmlException e) {
423             if (e.getErrorCode() == XmlException.DOCUMENT_NOT_FOUND) {
424                 return false;
425             } else {
426                 throw new PolicyIndexException("Error executing contains. " + e.getMessage(), e);
427             }
428         } finally {
429             DbXmlManager.readLock.unlock();
430         }
431 
432         return true;
433     }

435     /**
436      * Creates an instance of an XmlDocument for storage in the database.
437      *
438      * @param name
439      *        the name of the document (policy)
440      * @param document
441      *        the document data as a String
442      * @return the XmlDocument instance
443      * @throws XmlException
444      * @throws PolicyStoreException
445      */
446     private XmlDocument makeDocument(String name, String document)
447             throws XmlException, PolicyIndexException {
448         Map<String, String> metadata =
449                 m_utils.getDocumentMetadata(document
450                         .getBytes());
451         XmlDocument doc = m_dbXmlManager.manager.createDocument();
452         String docName = name;
453 
454         if (docName == null || "".equals(docName)) {
455             docName = metadata.get("PolicyId");
456         }
457 
458         if (docName == null || "".equals(docName)) {
459             throw new PolicyIndexException("Could not extract PolicyID from document.");
460         }
461 
462         doc.setMetaData("metadata", "PolicyId", new XmlValue(XmlValue.STRING,
463                                                              docName));
464         doc.setContent(document);
465         doc.setName(docName);
466 
467         // FIXME:
468         // this is probably redundant as the xpath queries now directly query the policy
469         // for the "any" scenarios
470         String item = null;
471         item = metadata.get("anySubject");
472         if (item != null) {
473             doc.setMetaData("metadata",
474                             "anySubject",
475                             new XmlValue(XmlValue.STRING, item));
476         }
477 
478         item = metadata.get("anyResource");
479         if (item != null) {
480             doc.setMetaData("metadata",
481                             "anyResource",
482                             new XmlValue(XmlValue.STRING, item));
483         }
484 
485         item = metadata.get("anyAction");
486         if (item != null) {
487             doc.setMetaData("metadata",
488                             "anyAction",
489                             new XmlValue(XmlValue.STRING, item));
490         }
491 
492         item = metadata.get("anyEnvironment");
493         if (item != null) {
494             doc.setMetaData("metadata",
495                             "anyEnvironment",
496                             new XmlValue(XmlValue.STRING, item));
497         }
498 
499         return doc;
500     }

533     private boolean deleteDirectory(String directory) {
534 
535         boolean result = false;
536 
537         if (directory != null) {
538             File file = new File(directory);
539             if (file.exists() && file.isDirectory()) {
540                 // 1. delete content of directory:
541                 File[] files = file.listFiles();
542                 result = true; //init result flag
543                 int count = files.length;
544                 for (int i = 0; i < count; i++) { //for each file:
545                     File f = files[i];
546                     if (f.isFile()) {
547                         result = result && f.delete();
548                     } else if (f.isDirectory()) {
549                         result = result && deleteDirectory(f.getAbsolutePath());
550                     }
551                 }//next file
552 
553                 file.delete(); //finally delete (empty) input directory
554             }//else: input directory does not exist or is not a directory
555         }//else: no input value
556 
557         return result;
558     }//deleteDirectory()
