53  /**
54   * A Servlet Filter for protecting resources. This filter uses JAAS for
55   * performing user authentication. Once a user is authenticated, a user
56   * principal object that is returned from the JAAS login module is created and
57   * added to the servlet request. The parameters of this filter are as follows:
58   * <ul>
59   * <li>
60   * <p>
61   * <strong>jaas.config.location</strong>
62   * </p>
63   * <p>
64   * This specifies the location of the jaas configuration file. The default is
65   * $FEDORA_HOME/server/config/jaas.conf
66   * </p>
67   * </li>
68   * <li>
69   * <p>
70   * <strong>jaas.config.name</strong>
71   * </p>
72   * <p>
73   * The name of the jaas configuration to use. The default is fedora-auth
74   * </p>
75   * </li>
76   * </ul>
77   *
78   * @author nish.naidoo@gmail.com
79   */

149     public void init() throws ServletException {
150         // get FEDORA_HOME. This being set is mandatory.
151         String fedoraHome = Constants.FEDORA_HOME;
152         if (fedoraHome == null || "".equals(fedoraHome)) {
153             String msg = "FEDORA_HOME environment variable not set";
154             throw new ServletException(msg);
155         }
156 
157         logger.info("using FEDORA_HOME: " + fedoraHome);
158 
159         // Get the jaas.conf file to use and the config to use from the
160         // jaas.conf file. This defaults to $FEDORA_HOME/server/config/jaas.conf
161         // and 'fedora-auth' for the configuration.
162         String jaasConfigLocation = fedoraHome + "/server/config/jaas.conf";
163         jaasConfigName = JAAS_CONFIG_DEFAULT;
164 
165         String tmp = null;
166 
167         tmp = filterConfig.getInitParameter("jaas.config.location");
168         if (tmp != null && !"".equals(tmp)) {
169             jaasConfigLocation = tmp;
170             if (logger.isDebugEnabled()) {
171                 logger.debug("using location from init file: "
172                         + jaasConfigLocation);
173             }
174         }
175 
176         tmp = filterConfig.getInitParameter("jaas.config.name");
177         if (tmp != null && !"".equals(tmp)) {
178             jaasConfigName = tmp;
179             if (logger.isDebugEnabled()) {
180                 logger.debug("using name from init file: " + jaasConfigName);
181             }
182         }
183 
184         tmp = filterConfig.getInitParameter("authnAPIA");
185         authnAPIA = Boolean.parseBoolean(tmp);
186 
187         tmp = filterConfig.getInitParameter("userClassNames");
188         userClassNames = new HashSet<String>();
189         if (tmp != null) {
190             String[] names = tmp.split(" *, *");
191             if (names != null && names.length > 0) {
192                 for (String n : names) {
193                     userClassNames.add(n);
194                 }
195             }
196         }
197 
198         tmp = filterConfig.getInitParameter("roleClassNames");
199         roleClassNames = new HashSet<String>();
200         if (tmp != null) {
201             String[] names = tmp.split(" *, *");
202             if (names != null && names.length > 0) {
203                 for (String n : names) {
204                     roleClassNames.add(n);
205                 }
206             }
207         }
208 
209         tmp = filterConfig.getInitParameter("roleAttributeNames");
210         roleAttributeNames = new HashSet<String>();
211         roleAttributeNames.add(ROLE_KEY);
212         roleAttributeNames.add(FEDORA_ROLE_KEY);
213         if (tmp != null) {
214             String[] names = tmp.split(" *, *");
215             if (names != null && names.length > 0) {
216                 for (String n : names) {
217                     roleAttributeNames.add(n);
218                 }
219             }
220         }
221 
222         File jaasConfig = new File(jaasConfigLocation);
223         if (!jaasConfig.exists()) {
224             String msg =
225                     "JAAS config file not at: " + jaasConfig.getAbsolutePath();
226             logger.error(msg);
227             throw new ServletException(msg);
228         }
229 
230         System.setProperty(JAAS_CONFIG_KEY, jaasConfig.getAbsolutePath());
231 
232         logger.info("initialised servlet filter: " + this.getClass().getName());
233     }

238      * javax.servlet.ServletResponse, javax.servlet.FilterChain)

240     @Override
241     public void doFilter(ServletRequest request,
242                          ServletResponse response,
243                          FilterChain chain) throws IOException,
244             ServletException {
245         HttpServletRequest req = (HttpServletRequest) request;
246         HttpServletResponse res = (HttpServletResponse) response;
247 
248         // This is a hack to skip challenge authentication of API-A methods via the
249         // REST API if API-A AuthN is off (as indicated by the "authnAPIA"
250         // filter init-param).
251         // If API-A AuthN is off, for all GET requests via the REST API,
252         // except those which are known to be part of API-M, don't challenge for authentication
253         // (but do pick up any preemptive credentials that are supplied)
254         // FIXME: As other servlets that require authn also go through this filter,
255         // there's probably a neater way to ensure we are only catching API-A methods
256         // currently we are explicitly testing for other management URLs/paths
257         // (probably should test fullPath for {appcontext}/objects (REST API) and then
258         // test path for API-A methods, maybe regex to make it explicit)
259         boolean doChallenge = true;
260         if (!authnAPIA) {
261             if (req.getMethod().equals("GET")) {
262                 String requestPath = req.getPathInfo();
263                 if (requestPath == null) requestPath = ""; // null is returned eg for /fedora/objects? - API-A, so we still want to do the below...
264                 String fullPath = req.getRequestURI();
265                 // API-M methods
266                 // potentially extra String evals, but aiming for clarity
267                 boolean isExport = requestPath.endsWith("/export");
268                 boolean isObjectXML = requestPath.endsWith("/objectXML");
269                 boolean isGetDatastream =
270                         requestPath.contains("/datastreams/")
271                                 && !requestPath.endsWith("/content");
272                 isGetDatastream = isGetDatastream || (requestPath.endsWith("/datastreams")
273                                 && Boolean.valueOf(request.getParameter("profiles")));
274                 boolean isGetRelationships =
275                         requestPath.endsWith("/relationships");
276                 boolean isValidate = requestPath.endsWith("/validate");
277                 // management get methods (LITE API, control)
278                 boolean isManagement =
279                         fullPath.endsWith("/management/control")
280                                 || fullPath.endsWith("/management/getNextPID");
281                 // user servlet
282                 boolean isUserServlet = fullPath.endsWith("/user");
283                 // challenge if API-M or one of the above other services (otherwise we assume it is API-A)
284                 doChallenge =
285                         isExport || isObjectXML || isGetDatastream
286                                 || isGetRelationships || isValidate
287                                 || isManagement || isUserServlet;
288             }
289         }
290 
291         if (logger.isDebugEnabled()) {
292             logger.debug("incoming filter: " + this.getClass().getName());
293             logger.debug("session-id: " + req.getSession().getId());
294         }
295 
296         Subject subject = authenticate(req);
297 
298         if (subject == null) {
299             if (doChallenge) {
300                 loginForm(res);
301                 return;
302             } else {
303                 // no auth required, and none supplied, do rest of chain
304                 chain.doFilter(request, response);
305                 return;
306             }
307         }
308 
309         // obtain the user principal from the subject and add to servlet.
310         Principal userPrincipal = getUserPrincipal(subject);
311 
312         // obtain the user roles from the subject and add to servlet.
313         Set<String> userRoles = getUserRoles(subject);
314 
315         // wrap the request in one that has the ability to store role
316         // and principal information and store this information.
317         AuthHttpServletRequestWrapper authRequest =
318                 new AuthHttpServletRequestWrapper(req);
319         authRequest.setUserPrincipal(userPrincipal);
320         authRequest.setUserRoles(userRoles);
321 
322         // add the roles that were obtained to the Subject.
323         addRolesToSubject(subject, userRoles);
324 
325         // populate FEDORA_AUX_SUBJECT_ATTRIBUTES with fedoraRole
326         // and any additional Subject attributes
327         populateFedoraAttributes(subject, userRoles, authRequest);
328 
329         chain.doFilter(authRequest, response);
330 
331         if (logger.isDebugEnabled()) {
332             logger.debug("outgoing filter: " + this.getClass().getName());
333         }
334     }

342     /**
343      * Sends a 401 error to the browser. This forces a login box to be displayed
344      * allowing the user to login.
345      *
346      * @param response
347      *        the response to set the headers and status
348      */
349     private void loginForm(HttpServletResponse response) throws IOException {
350         response.reset();
351         response.addHeader("WWW-Authenticate",
352                            "Basic realm=\"!!Fedora Repository Server\"");
353         response.setStatus(HttpServletResponse.SC_UNAUTHORIZED);
354         OutputStream out = response.getOutputStream();
355         out.write("Fedora: 401 ".getBytes());
356         out.flush();
357         out.close();
358     }

360     /**
361      * Performs the authentication. Once a Subject is obtained, it is stored in
362      * the users session. Subsequent requests check for the existence of this
363      * object before performing the authentication again.
364      *
365      * @param req
366      *        the servlet request.
367      * @return a user principal that was extracted from the login context.
368      */
369     private Subject authenticate(HttpServletRequest req) {
370         String authorization = req.getHeader("authorization");
371         if (authorization == null || "".equals(authorization.trim())) {
372             return null;
373         }
374 
375         // subject from session instead of re-authenticating
376         // can't change username/password for this session.
377         Subject subject =
378                 (Subject) req.getSession().getAttribute(authorization);
379         if (subject != null) {
380             return subject;
381         }
382 
383         String auth = null;
384         try {
385             byte[] data = Base64.decode(authorization.substring(6));
386             auth = new String(data);
387         } catch (IOException e) {
388             logger.error(e.toString());
389             return null;
390         }
391 
392         String username = auth.substring(0, auth.indexOf(':'));
393         String password = auth.substring(auth.indexOf(':') + 1);
394 
395         if (logger.isDebugEnabled()) {
396             logger.debug("auth username: " + username);
397         }
398 
399         LoginContext loginContext = null;
400         try {
401             CallbackHandler handler =
402                     new UsernamePasswordCallbackHandler(username, password);
403             loginContext = new LoginContext(jaasConfigName, handler);
404             loginContext.login();
405         } catch (LoginException le) {
406             logger.error(le.toString());
407             return null;
408         }
409 
410         // successfully logged in
411         subject = loginContext.getSubject();
412 
413         // object accessable by a fixed key for usage
414         req.getSession().setAttribute(SESSION_SUBJECT_KEY, subject);
415 
416         // object accessable only by base64 encoded username:password that was
417         // initially used - prevents some dodgy stuff
418         req.getSession().setAttribute(authorization, subject);
419 
420         return subject;
421     }

423     /**
424      * Given a subject, obtain the userPrincipal from it. The user principal is
425      * defined by a Principal class that can be defined in the web.xml file. If
426      * this is undefined, the first principal found is assumed to be the
427      * userPrincipal.
428      *
429      * @param subject
430      *        the subject returned from authentication.
431      * @return the userPrincipal associated with the given subject.
432      */
433     private Principal getUserPrincipal(Subject subject) {
434         Principal userPrincipal = null;
435 
436         Set<Principal> principals = subject.getPrincipals();
437 
438         // try and get userPrincipal based on userClassNames
439         if (userClassNames != null && userClassNames.size() > 0) {
440             for (Principal p : principals) {
441                 if (userPrincipal == null
442                         && userClassNames.contains(p.getClass().getName())) {
443                     userPrincipal = p;
444                 }
445             }
446         }
447 
448         // no userPrincipal found using userClassNames, just grab first principal
449         if (userPrincipal == null) {
450             Iterator<Principal> i = principals.iterator();
451             // should always have 1 at least and 1st should be user principal
452             if (i.hasNext()) {
453                 userPrincipal = i.next();
454             }
455         }
456 
457         if (logger.isDebugEnabled()) {
458             logger.debug("found userPrincipal ["
459                     + userPrincipal.getClass().getName() + "]: "
460                     + userPrincipal.getName());
461         }
462 
463         return userPrincipal;
464     }

466     /**
467      * Obtains the roles for the user based on the class names and attribute
468      * names provided in the web.xml file.
469      *
470      * @param subject
471      *        the subject returned from authentication.
472      * @return a set of strings that represent the users roles.
473      */
474     private Set<String> getUserRoles(Subject subject) {
475         Set<String> userRoles = new HashSet<String>();
476 
477         // get roles from specified classes
478         Set<Principal> principals = subject.getPrincipals();
479         if (roleClassNames != null && roleClassNames.size() > 0) {
480             for (Principal p : principals) {
481                 if (roleClassNames.contains(p.getClass().getName())) {
482                     userRoles.add(p.getName());
483                 }
484             }
485         }
486 
487         // get roles from specified attributes
488         Map<String, Set<String>> attributes =
489                 SubjectUtils.getAttributes(subject);
490         if (attributes != null) {
491             for (String key : attributes.keySet()) {
492                 if (roleAttributeNames.contains(key)) {
493                     userRoles.addAll(attributes.get(key));
494                 }
495             }
496         }
497 
498         if (logger.isDebugEnabled()) {
499             for (String r : userRoles) {
500                 logger.debug("found role: " + r);
501             }
502         }
503 
504         return userRoles;
505     }

507     /**
508      * Adds roles to the Subject object.
509      *
510      * @param subject
511      *        the subject that was returned from authentication.
512      * @param userRoles
513      *        the set of user roles that were found.
514      */
515     private void addRolesToSubject(Subject subject, Set<String> userRoles) {
516         if (userRoles == null) {
517             userRoles = new HashSet<String>();
518         }
519 
520         Map<String, Set<String>> attributes =
521                 SubjectUtils.getAttributes(subject);
522 
523         Set<String> roles = attributes.get(ROLE_KEY);
524         if (roles == null) {
525             roles = new HashSet<String>();
526             attributes.put(ROLE_KEY, roles);
527         }
528 
529         for (String role : userRoles) {
530             roles.add(role);
531             if (logger.isDebugEnabled()) {
532                 logger.debug("added role: " + role);
533             }
534         }
535     }

537     /**
538      * Add roles and other subject attributes where Fedora expects them - a Map
539      * called FEDORA_AUX_SUBJECT_ATTRIBUTES. Roles will be put in "fedoraRole"
540      * and others will be named as-is.
541      *
542      * @param subject
543      *        the subject from authentication.
544      * @param userRoles
545      *        the set of user roles.
546      * @param request
547      *        the request in which to place the attributes.
548      */
549     private void populateFedoraAttributes(Subject subject,
550                                           Set<String> userRoles,
551                                           HttpServletRequest request) {
552         Map<String, Set<String>> attributes =
553                 SubjectUtils.getAttributes(subject);
554         if (attributes == null) {
555             attributes = new HashMap<String, Set<String>>();
556         }
557 
558         // get the fedoraRole attribute or create it.
559         Set<String> roles = attributes.get(FEDORA_ROLE_KEY);
560         if (roles == null) {
561             roles = new HashSet<String>();
562             attributes.put(FEDORA_ROLE_KEY, roles);
563         }
564 
565         roles.addAll(userRoles);
566 
567         request.setAttribute(FEDORA_ATTRIBUTES_KEY, attributes);
568     }
