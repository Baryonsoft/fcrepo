112     /**
113      * Converts a String parameter to a JAXP Source object.
114      * @param param a String parameter
115      * @return Source the generated Source object
116      */
117     protected Source convertString2Source(String param) {
118         Source src;
119         try {
120             src = uriResolver.resolve(param, null);
121         } catch (TransformerException e) {
122             src = null;
123         }
124         if (src == null) {
125             src = new StreamSource(new File(param));
126         }
127         return src;
128     }

130     private void sendPDF(byte[] content, HttpServletResponse response) throws IOException {
131         //Send the result back to the client
132         response.setContentType("application/pdf");
133         response.setContentLength(content.length);
134         response.getOutputStream().write(content);
135         response.getOutputStream().flush();
136     }

138     /**
139      * Renders an XSL-FO file into a PDF file. The PDF is written to a byte
140      * array that is returned as the method's result.
141      * @param fo the XSL-FO file
142      * @param response HTTP response object
143      * @throws FOPException If an error occurs during the rendering of the
144      * XSL-FO
145      * @throws TransformerException If an error occurs while parsing the input
146      * file
147      * @throws IOException In case of an I/O problem
148      */
149     protected void renderFO(String fo, HttpServletResponse response)
150                 throws FOPException, TransformerException, IOException {
151 
152         //Setup source
153         Source foSrc = convertString2Source(fo);
154 
155         //Setup the identity transformation
156         Transformer transformer = this.transFactory.newTransformer();
157         transformer.setURIResolver(this.uriResolver);
158 
159         //Start transformation and rendering process
160         render(foSrc, transformer, response);
161     }

163     /**
164      * Renders an XML file into a PDF file by applying a stylesheet
165      * that converts the XML to XSL-FO. The PDF is written to a byte array
166      * that is returned as the method's result.
167      * @param xml the XML file
168      * @param xslt the XSLT file
169      * @param response HTTP response object
170      * @throws FOPException If an error occurs during the rendering of the
171      * XSL-FO
172      * @throws TransformerException If an error occurs during XSL
173      * transformation
174      * @throws IOException In case of an I/O problem
175      */
176     protected void renderXML(String xml, String xslt, HttpServletResponse response)
177                 throws FOPException, TransformerException, IOException {
178 
179         //Setup sources
180         Source xmlSrc = convertString2Source(xml);
181         Source xsltSrc = convertString2Source(xslt);
182 
183         //Setup the XSL transformation
184         Transformer transformer = this.transFactory.newTransformer(xsltSrc);
185         transformer.setURIResolver(this.uriResolver);
186 
187         //Start transformation and rendering process
188         render(xmlSrc, transformer, response);
189     }

191     /**
192      * Renders an input file (XML or XSL-FO) into a PDF file. It uses the JAXP
193      * transformer given to optionally transform the input document to XSL-FO.
194      * The transformer may be an identity transformer in which case the input
195      * must already be XSL-FO. The PDF is written to a byte array that is
196      * returned as the method's result.
197      * @param src Input XML or XSL-FO
198      * @param transformer Transformer to use for optional transformation
199      * @param response HTTP response object
200      * @throws FOPException If an error occurs during the rendering of the
201      * XSL-FO
202      * @throws TransformerException If an error occurs during XSL
203      * transformation
204      * @throws IOException In case of an I/O problem
205      */
206     protected void render(Source src, Transformer transformer, HttpServletResponse response)
207                 throws FOPException, TransformerException, IOException {
208 
209         FOUserAgent foUserAgent = getFOUserAgent();
210 
211         //Setup output
212         ByteArrayOutputStream out = new ByteArrayOutputStream();
213 
214         //Setup FOP
215         Fop fop = fopFactory.newFop(MimeConstants.MIME_PDF, foUserAgent, out);
216 
217         //Make sure the XSL transformation's result is piped through to FOP
218         Result res = new SAXResult(fop.getDefaultHandler());
219 
220         //Start the transformation and rendering process
221         transformer.transform(src, res);
222 
223         //Return the result
224         sendPDF(out.toByteArray(), response);
225     }
226     
227     /** @return a new FOUserAgent for FOP */
228     protected FOUserAgent getFOUserAgent() {
229         FOUserAgent userAgent = fopFactory.newFOUserAgent();
230         //Configure foUserAgent as desired
231         return userAgent;
232     }
