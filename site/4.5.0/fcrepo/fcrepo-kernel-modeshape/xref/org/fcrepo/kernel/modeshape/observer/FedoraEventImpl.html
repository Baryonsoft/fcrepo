51  /**
52   * A very simple abstraction to prevent event-driven machinery downstream from the repository from relying directly
53   * on a JCR interface {@link Event}. Can represent either a single JCR event or several.
54   *
55   * @author ajs6f
56   * @since Feb 19, 2013
57   */
58  public class FedoraEventImpl implements FedoraEvent {
59  
60      private final Event e;
61      private final String eventID;
62  
63      private final Set<EventType> eventTypes = new HashSet<>();
64      private final Set<String> eventProperties = new HashSet<>();
65  
66      private static final List<Integer> PROPERTY_TYPES = asList(javax.jcr.observation.Event.PROPERTY_ADDED,
67              javax.jcr.observation.Event.PROPERTY_CHANGED,
68              javax.jcr.observation.Event.PROPERTY_REMOVED);
69  
70      private static final UniqueValueSupplier pidMinter = new DefaultPathMinter();
71  
72      /**
73       * Wrap a JCR Event with our FedoraEvent decorators
74       *
75       * @param e the JCR event
76       */
77      public FedoraEventImpl(final Event e) {
78          checkArgument(e != null, "null cannot support a FedoraEvent!");
79          eventID = pidMinter.get();
80          this.e = e;
81      }
82  
83      /**
84       * Create a FedoraEvent from an existing FedoraEvent object
85       * Note: Only the wrapped JCR event is passed on to the new object.
86       *
87       * @param e the given fedora event
88       */
89      public FedoraEventImpl(final FedoraEvent e) {
90          checkArgument(e != null, "null cannot support a FedoraEvent!");
91          eventID = e.getEventID();
92          this.e = ((FedoraEventImpl)e).e;
93      }
94  
95      /**
96       * @return the event types of the underlying JCR {@link Event}s
97       */
98      @Override
99      public Set<EventType> getTypes() {
100         final EventType type = valueOf(e.getType());
101         return eventTypes != null ? union(singleton(type), eventTypes) : singleton(type);
102     }
103 
104     /**
105      * @param type the type
106      * @return this object for continued use
107      */
108     @Override
109     public FedoraEvent addType(final EventType type) {
110         eventTypes.add(type);
111         return this;
112     }
113 
114     /**
115      * @return the property names of the underlying JCR property {@link Event}s
116     **/
117     @Override
118     public Set<String> getProperties() {
119         return eventProperties;
120     }
121 
122     /**
123      * Add a property name to this event
124      * @param property property name
125      * @return this object for continued use
126     **/
127     @Override
128     public FedoraEvent addProperty( final String property ) {
129         eventProperties.add(property);
130         return this;
131     }
132 
133     /**
134      * @return the path of the underlying JCR {@link Event}s
135      */
136     @Override
137     public String getPath() {
138         return getPath(e);
139     }
140 
141     /**
142      * Get the path of the node related to this event (removing property names
143      * from the end of property nodes).
144      * @param e JCR Event
145      * @return the node path for this event
146     **/
147     public static String getPath(final Event e) {
148         try {
149             final String path;
150             if (PROPERTY_TYPES.contains(e.getType())) {
151                 path = e.getPath().substring(0, e.getPath().lastIndexOf("/"));
152             } else {
153                 path = e.getPath();
154             }
155             return path.replaceAll("/" + JCR_CONTENT, "");
156         } catch (final RepositoryException e1) {
157             throw new RepositoryRuntimeException("Error getting event path!", e1);
158         }
159     }
160 
161     /**
162      * @return the user ID of the underlying JCR {@link Event}s
163      */
164     @Override
165     public String getUserID() {
166         return e.getUserID();
167     }
168 
169     /**
170      * @return the user data of the underlying JCR {@link Event}s
171      */
172     @Override
173     public String getUserData() {
174         try {
175             return e.getUserData();
176         } catch (final RepositoryException e1) {
177             throw new RepositoryRuntimeException("Error getting event userData!", e1);
178         }
179     }
180 
181     /**
182      * @return the date of the underlying JCR {@link Event}s
183      */
184     @Override
185     public long getDate() {
186         try {
187             return e.getDate();
188         } catch (final RepositoryException e1) {
189             throw new RepositoryRuntimeException("Error getting event date!", e1);
190         }
191     }
192 
193     /**
194      * Get the event ID.
195      * @return Event identifier to use for building event URIs (e.g., in an external triplestore).
196     **/
197     @Override
198     public String getEventID() {
199         return eventID;
200     }
201 
202     @Override
203     public String toString() {
204 
205         return toStringHelper(this)
206             .add("Event types:", String.join(",", getTypes().stream()
207                             .map(EventType::getName)
208                             .collect(toList())))
209             .add("Event properties:", String.join(",", eventProperties))
210             .add("Path:", getPath())
211             .add("Date: ", getDate()).toString();
212     }
213 
214     private static final Map<Integer, EventType> translation = ImmutableMap.<Integer, EventType>builder()
215             .put(javax.jcr.observation.Event.NODE_ADDED, NODE_ADDED)
216             .put(javax.jcr.observation.Event.NODE_REMOVED, NODE_REMOVED)
217             .put(javax.jcr.observation.Event.PROPERTY_ADDED, PROPERTY_ADDED)
218             .put(javax.jcr.observation.Event.PROPERTY_REMOVED, PROPERTY_REMOVED)
219             .put(javax.jcr.observation.Event.PROPERTY_CHANGED, PROPERTY_CHANGED)
220             .put(javax.jcr.observation.Event.NODE_MOVED, NODE_MOVED)
221             .put(javax.jcr.observation.Event.PERSIST, PERSIST).build();
222 
223     /**
224      * Get the Fedora event type for a JCR type
225      *
226      * @param i the integer value of a JCR type
227      * @return EventType
228      */
229     public static EventType valueOf(final Integer i) {
230         return forMap(translation).apply(i);
231     }
232 
233     private static class DefaultPathMinter implements HierarchicalIdentifierSupplier { }
234 }
