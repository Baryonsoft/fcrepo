37  /**
38   * Predicate to match nodes with any of the given mixin types
39   * @author armintor@gmail.com
40   * @author ajs6f
41   *
42   */
43  public class AnyTypesPredicate implements UncheckedPredicate<Node> {
44      protected final Collection<String> nodeTypes;
45  
46      /**
47       * True if any of the types specified match.
48       * @param types the types
49       */
50      public AnyTypesPredicate(final String...types) {
51          nodeTypes = asList(types);
52      }
53  
54      @Override
55      public boolean testThrows(final Node input) throws RepositoryException {
56          requireNonNull(input, "null node has no types!");
57          if (isFrozen.test(input) && input.hasProperty(FROZEN_MIXIN_TYPES)) {
58              final Property frozenTypesProperty = input.getProperty(FROZEN_MIXIN_TYPES);
59              final List<Value> values = copyOf(property2values.apply(frozenTypesProperty));
60              if (values.stream().map(uncheck(Value::getString)).anyMatch(nodeTypes::contains)) {
61                  return true;
62              }
63          }
64          return nodeTypes.stream().anyMatch(uncheck(input::isNodeType));
65      }
66  }
