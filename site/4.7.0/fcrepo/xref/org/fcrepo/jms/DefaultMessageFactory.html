18  package org.fcrepo.jms;
19  
20  import static java.lang.String.join;
21  import static java.util.stream.Collectors.joining;
22  import static org.fcrepo.kernel.api.observer.OptionalValues.BASE_URL;
23  import static org.fcrepo.kernel.api.observer.OptionalValues.USER_AGENT;
24  import static org.slf4j.LoggerFactory.getLogger;
25  
26  import java.util.Set;
27  import javax.jms.JMSException;
28  import javax.jms.Message;
29  import javax.jms.Session;
30  
31  import org.fcrepo.kernel.api.observer.EventType;
32  import org.fcrepo.kernel.api.observer.FedoraEvent;
33  import org.fcrepo.event.serialization.EventSerializer;
34  import org.fcrepo.event.serialization.JsonLDSerializer;
35  
36  import org.slf4j.Logger;
37  
38  /**
39   * Generates JMS {@link Message}s composed entirely of headers, based entirely
40   * on information found in the {@link FedoraEvent} that triggers publication.
41   *
42   * @author ajs6f
43   * @author escowles
44   * @since Dec 2, 2013
45   */
46  public class DefaultMessageFactory implements JMSEventMessageFactory {
47  
48      public static final String JMS_NAMESPACE = "org.fcrepo.jms.";
49  
50      public static final String TIMESTAMP_HEADER_NAME = JMS_NAMESPACE
51              + "timestamp";
52  
53      public static final String IDENTIFIER_HEADER_NAME = JMS_NAMESPACE
54              + "identifier";
55  
56      public static final String EVENT_TYPE_HEADER_NAME = JMS_NAMESPACE
57              + "eventType";
58  
59      public static final String BASE_URL_HEADER_NAME = JMS_NAMESPACE
60              + "baseURL";
61  
62      public static final String RESOURCE_TYPE_HEADER_NAME = JMS_NAMESPACE + "resourceType";
63  
64      public static final String USER_HEADER_NAME = JMS_NAMESPACE + "user";
65      public static final String USER_AGENT_HEADER_NAME = JMS_NAMESPACE + "userAgent";
66      public static final String EVENT_ID_HEADER_NAME = JMS_NAMESPACE + "eventID";
67  
68      @Override
69      public Message getMessage(final FedoraEvent event, final Session jmsSession)
70              throws JMSException {
71  
72          final EventSerializer serializer = new JsonLDSerializer();
73          final String body = serializer.serialize(event);
74          final Message message = jmsSession.createTextMessage(body);
75  
76          message.setLongProperty(TIMESTAMP_HEADER_NAME, event.getDate().toEpochMilli());
77  
78          if (event.getInfo().containsKey(BASE_URL)) {
79              message.setStringProperty(BASE_URL_HEADER_NAME, event.getInfo().get(BASE_URL));
80          } else {
81              LOGGER.warn("Could not extract baseUrl from FedoraEvent!");
82          }
83          if (event.getInfo().containsKey(USER_AGENT)) {
84              message.setStringProperty(USER_AGENT_HEADER_NAME, event.getInfo().get(USER_AGENT));
85          }
86  
87          message.setStringProperty(IDENTIFIER_HEADER_NAME, event.getPath());
88          message.setStringProperty(EVENT_TYPE_HEADER_NAME, getEventURIs(event.getTypes()));
89          message.setStringProperty(USER_HEADER_NAME, event.getUserID());
90          message.setStringProperty(RESOURCE_TYPE_HEADER_NAME, join(",", event.getResourceTypes()));
91          message.setStringProperty(EVENT_ID_HEADER_NAME, event.getEventID());
92  
93          LOGGER.trace("getMessage() returning: {}", message);
94          return message;
95      }
96  
97      private static String getEventURIs(final Set<EventType> types) {
98          final String uris = types.stream()
99                                   .map(EventType::getType)
100                                  .collect(joining(","));
101 
102         LOGGER.debug("Constructed event type URIs: {}", uris);
103         return uris;
104     }
105 
106     private static final Logger LOGGER = getLogger(DefaultMessageFactory.class);
107 
108 }
