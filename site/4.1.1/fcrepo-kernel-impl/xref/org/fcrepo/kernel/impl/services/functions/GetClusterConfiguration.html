18  import static com.google.common.base.Preconditions.checkNotNull;
19  import static org.slf4j.LoggerFactory.getLogger;
20  
21  import java.util.LinkedHashMap;
22  import java.util.List;
23  import java.util.Map;
24  
25  import javax.jcr.Repository;

27  import org.infinispan.Cache;
28  import org.infinispan.manager.DefaultCacheManager;
29  import org.modeshape.jcr.GetBinaryStore;
30  import org.modeshape.jcr.value.binary.BinaryStore;
31  import org.modeshape.jcr.value.binary.infinispan.InfinispanBinaryStore;
32  import org.slf4j.Logger;
33  
34  import com.google.common.base.Function;

36  /**
37   * Extract the Infinispan cluster configuration and state
38   * from a running Modeshape repository
39   * @author Gregory Jansen
40   * @since Apr 26, 2013
41   */
42  public class GetClusterConfiguration implements
43          Function<Repository, Map<String, String>> {
44  
45      private static final Logger LOGGER =
46              getLogger(GetClusterConfiguration.class);
47  
48      public static final String CLUSTER_NAME = "clusterName";
49      public static final String CACHE_MODE = "clusterCacheMode";
50      public static final String NODE_ADDRESS = "clusterNodeAddress";
51      public static final String PHYSICAL_ADDRESS = "clusterPhysicalAddress";
52      public static final String NODE_VIEW = "clusterNodeView";
53      public static final String CLUSTER_SIZE = "clusterSize";
54      public static final String CLUSTER_MEMBERS = "clusterMembers";
55      public static final int UNKNOWN_NODE_VIEW = -1;
56  
57      private GetBinaryStore getBinaryStore = new GetBinaryStore();
58  
59      /**
60       * Extract the BinaryStore out of Modeshape
61       * (infinspan, jdbc, file, transient, etc)
62       * @return the binary store configuration as a map
63       */
64      @Override
65      public Map<String, String> apply(final Repository input) {
66          checkNotNull(input, "null cannot have a BinaryStore!");
67  
68          final Map<String, String> result =
69              new LinkedHashMap<>();
70          final BinaryStore store = getBinaryStore.apply(input);
71  
72          if (!(store instanceof InfinispanBinaryStore)) {
73              return result;
74          }
75  
76          final InfinispanBinaryStore ispnStore = (InfinispanBinaryStore) store;
77  
78          final List<Cache<?, ?>> caches = ispnStore.getCaches();
79          final DefaultCacheManager cm =
80              (DefaultCacheManager) caches.get(0).getCacheManager();
81  
82          if (cm == null) {
83              LOGGER.debug("Could not get cluster configuration information");
84              return result;
85          }
86  
87          final int nodeView;
88  
89          if (cm.getTransport() != null) {
90              nodeView = cm.getTransport().getViewId() + 1;
91          } else {
92              nodeView = UNKNOWN_NODE_VIEW;
93          }
94  
95          result.put(CLUSTER_NAME, cm.getClusterName());
96          result.put(CACHE_MODE, cm.getCache().getCacheConfiguration()
97                                     .clustering().cacheMode().toString());
98          result.put(NODE_ADDRESS, cm.getNodeAddress());
99          result.put(PHYSICAL_ADDRESS, cm.getPhysicalAddresses());
100         result.put(NODE_VIEW, nodeView == UNKNOWN_NODE_VIEW ?
101                                   "Unknown" :
102                                   String.valueOf(nodeView));
103         result.put(CLUSTER_SIZE, String.valueOf(cm.getClusterSize()));
104         result.put(CLUSTER_MEMBERS, cm.getClusterMembers());
105         return result;
106     }
107 
108 }
