18  package org.fcrepo.kernel.modeshape.utils;
19  
20  import java.util.function.Predicate;
21  
22  import javax.jcr.RepositoryException;
23  
24  import org.fcrepo.kernel.api.exception.RepositoryRuntimeException;
25  
26  /**
27   * Operations that throw {@link RepositoryException} cannot be used as lambdas without "unchecking" those exceptions.
28   *
29   * @author ajs6f
30   * @param <T> the type of the input to the predicate
31   */
32  @FunctionalInterface
33  public interface UncheckedPredicate<T> extends Predicate<T> {
34  
35      @Override
36      default boolean test(final T elem) {
37          try {
38              return testThrows(elem);
39          } catch (final RepositoryException e) {
40              throw new RepositoryRuntimeException(e);
41          }
42      }
43  
44      /**
45       * The same semantic as {@link #test(Object)}, but allowed to throw a {@link RepositoryException}
46       *
47       * @param elem the input argument
48       * @return true if the input matches the predicate, otherwise false
49       * @throws RepositoryException a repository-related exception
50       */
51      boolean testThrows(T elem) throws RepositoryException;
52  
53      /**
54       * @param <T> the type of the input to the predicate
55       * @param p a lambda expression
56       * @return an unchecked version of that lambda
57       */
58      static <T> UncheckedPredicate<T> uncheck(final UncheckedPredicate<T> p) {
59          return p;
60      }
61  }
