18  package org.fcrepo.kernel.modeshape.utils;
19  
20  import org.apache.jena.graph.Node;
21  import org.apache.jena.rdf.model.Resource;
22  import org.apache.jena.rdf.model.Statement;
23  import org.fcrepo.kernel.api.identifiers.IdentifierConverter;
24  import org.fcrepo.kernel.api.models.FedoraResource;
25  import org.fcrepo.kernel.api.utils.RelaxedPropertiesHelper;
26  import org.slf4j.Logger;
27  
28  import javax.jcr.Session;
29  
30  import java.util.ArrayList;
31  import java.util.Calendar;
32  import java.util.List;
33  
34  import static org.fcrepo.kernel.api.RdfLexicon.isRelaxed;
35  import static org.slf4j.LoggerFactory.getLogger;
36  
37  /**
38   * A subclass of JcrPropertyStatementListener that intercepts all requests to modify
39   * "relaxed" server-managed triples and provides methods to get those intercepted values.
40   *
41   * A "relaxed" server-managed triple is one that is typically server-managed but for which
42   * the current configuration allows them to be set explicitly under certain circumstances.
43   *
44   * This class behaves exactly like FilteringJcrPropertyStatementListener in the event that
45   * there are no "relaxed" server managed triples.
46   *
47   * @author Mike Durbin
48   */
49  public class FilteringJcrPropertyStatementListener extends JcrPropertyStatementListener {
50  
51      private static final Logger LOGGER = getLogger(FilteringJcrPropertyStatementListener.class);
52  
53      private List<Statement> filteredAddStatements;
54  
55      /**
56       * Construct a statement listener within the given session that filters out changes to
57       * any relaxed server managed triples.
58       *
59       * @param idTranslator the id translator
60       * @param session the session
61       * @param topic the topic of the RDF statement
62       */
63      public FilteringJcrPropertyStatementListener(final IdentifierConverter<Resource, FedoraResource> idTranslator,
64                                                   final Session session, final Node topic) {
65          super(idTranslator, session, topic);
66          filteredAddStatements = new ArrayList<>();
67      }
68  
69      @Override
70      public void addedStatement(final Statement input) {
71          if (isRelaxed.test(input.getPredicate())) {
72              filteredAddStatements.add(input);
73              LOGGER.trace("The added statement {} was intercepted from the update request.", input);
74          } else {
75              super.addedStatement(input);
76          }
77      }
78  
79      @Override
80      public void removedStatement(final Statement input) {
81          if (!isRelaxed.test(input.getPredicate())) {
82              super.removedStatement(input);
83          } else {
84              LOGGER.trace("The removed statement {} was intercepted from the update request.", input);
85          }
86      }
87  
88      /**
89       * Gets the created date (if any) that was specified to be applied as part of the statements
90       * made to the model to which this StatementListener is listening.
91       * @return the date that should be set for the CREATED_DATE or null if it should be
92       *         untouched
93       */
94      public Calendar getAddedCreatedDate() {
95          return RelaxedPropertiesHelper.getCreatedDate(filteredAddStatements);
96      }
97  
98      /**
99       * Gets the created by user (if any) that was specified to be applied as part of the statements
100      * made to the model to which this StatementListener is listening.
101      * @return the user that should be set for the CREATED_BY or null if it should be
102      *         untouched
103      */
104     public String getAddedCreatedBy() {
105         return RelaxedPropertiesHelper.getCreatedBy(filteredAddStatements);
106     }
107 
108     /**
109      * Gets the modified date (if any) that was specified to be applied as part of the statements
110      * made to the model to which this StatementListener is listening.
111      * @return the date that should be set for the LAST_MODIFIED_DATE or null if it should be
112      *         untouched
113      */
114     public Calendar getAddedModifiedDate() {
115         return RelaxedPropertiesHelper.getModifiedDate(filteredAddStatements);
116     }
117 
118     /**
119      * Gets the modified by user (if any) that was specified to be applied as part of the statements
120      * made to the model to which this StatementListener is listening.
121      * @return the user that should be set for the MODIFIED_BY or null if it should be
122      *         untouched
123      */
124     public String getAddedModifiedBy() {
125         return RelaxedPropertiesHelper.getModifiedBy(filteredAddStatements);
126     }
127 }
