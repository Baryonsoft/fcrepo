36  import com.google.common.base.Converter;
37  import org.apache.jena.graph.impl.LiteralLabel;
38  import org.apache.jena.rdf.model.Resource;
39  
40  import org.fcrepo.kernel.api.exception.RepositoryRuntimeException;
41  import org.fcrepo.kernel.api.models.FedoraResource;
42  import org.fcrepo.kernel.modeshape.rdf.converters.PropertyConverter;
43  import org.fcrepo.kernel.modeshape.rdf.converters.ValueConverter;
44  import org.apache.jena.datatypes.RDFDatatype;
45  import org.apache.jena.graph.Triple;
46  import org.slf4j.Logger;

48  /**
49   * Utility for moving from JCR properties to RDF triples.
50   *
51   * @author ajs6f
52   * @since Oct 10, 2013
53   */
54  public class PropertyToTriple implements Function<Property, Stream<Triple>> {
55  
56      private static final Logger LOGGER = getLogger(PropertyToTriple.class);
57  
58      private static final PropertyConverter propertyConverter = new PropertyConverter();
59      private final ValueConverter valueConverter;
60      private final Converter<Node, Resource> translator;
61  
62      /**
63       * Default constructor. We require a {@link Converter} in order to construct the RDF subjects of our triples.
64       *
65       * @param converter a converter between RDF and the Fedora model
66       * @param session the JCR session
67       */
68      public PropertyToTriple(final Session session, final Converter<Resource, FedoraResource> converter) {
69          this.valueConverter = new ValueConverter(session, converter);
70          this.translator = nodeToResource(converter);
71      }
72  
73      @Override
74      public Stream<Triple> apply(final Property p) {
75          try {
76              final org.apache.jena.graph.Node subject = translator.convert(p.getParent()).asNode();
77              final org.apache.jena.graph.Node propPredicate = propertyConverter.convert(p).asNode();
78              final String propertyName = p.getName();
79  
80              return iteratorToStream(new PropertyValueIterator(p)).filter(this::valueCanBeConverted).map(v -> {
81                  final org.apache.jena.graph.Node object = valueConverter.convert(v).asNode();
82                  if (object.isLiteral()) {
83                      // unpack the name of the property for information about what kind of literal
84                      final int i = propertyName.indexOf('@');
85                      if (i > 0) {
86                          final LiteralLabel literal = object.getLiteral();
87                          final RDFDatatype datatype = literal.getDatatype();
88                          final String datatypeURI = datatype.getURI();
89                          if (datatypeURI.isEmpty() || datatype.equals(XSDstring)) {
90                              // this is an RDF string literal and could involve an RDF lang tag
91                              final String lang = propertyName.substring(i + 1);
92                              final String lex = literal.getLexicalForm();
93                              return create(subject, propPredicate, createLiteral(lex, lang, datatype));
94                          }
95                      }
96                  }
97                  return create(subject, propPredicate, object);
98              });
99          } catch (final RepositoryException e) {
100             throw new RepositoryRuntimeException(e);
101         }
102     }
103 
104     /**
105      * This method tests if a given value can be converted.
106      * The scenario when this may not be true is for (weak)reference properties that target an non-existent resource.
107      * This scenario generally should not be possible, but the following bug introduced the possibility:
108      *   https://jira.duraspace.org/browse/FCREPO-2323
109      *
110      * @param value to be tested for whether it can be converted to an RDFNode or not
111      * @return true if value can be converted
112      */
113     private boolean valueCanBeConverted(final Value value) {
114         try {
115             valueConverter.convert(value);
116             return true;
117         } catch (final RepositoryRuntimeException e) {
118             LOGGER.warn("Reference to non-existent resource encounterd: {}", value);
119             return false;
120         }
121     };
122 }
