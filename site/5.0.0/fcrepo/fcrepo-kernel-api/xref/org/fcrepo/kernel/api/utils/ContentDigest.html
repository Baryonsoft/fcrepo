18  package org.fcrepo.kernel.api.utils;
19  
20  import static org.apache.commons.codec.binary.Hex.encodeHexString;
21  import static org.fcrepo.kernel.api.utils.ContentDigest.DIGEST_ALGORITHM.SHA1;
22  import static org.slf4j.LoggerFactory.getLogger;
23  
24  import java.net.URI;
25  import java.net.URISyntaxException;
26  import java.util.Arrays;
27  
28  import org.fcrepo.kernel.api.exception.RepositoryRuntimeException;
29  import org.slf4j.Logger;
30  
31  /**
32   * Digest helpers to convert digests (checksums) into URI strings
33   * (based loosely on Magnet URIs)
34   * @author Chris Beer
35   * @since Mar 6, 2013
36   */
37  public final class ContentDigest {
38  
39      private static final Logger LOGGER = getLogger(ContentDigest.class);
40  
41      public enum DIGEST_ALGORITHM {
42          SHA1("SHA", "urn:sha1"), SHA256("SHA-256", "urn:sha-256"), MD5("MD5", "urn:md5"), MISSING("NONE", "missing");
43  
44          final public String algorithm;
45          final private String scheme;
46  
47          DIGEST_ALGORITHM(final String alg, final String scheme) {
48              this.algorithm = alg;
49              this.scheme = scheme;
50          }
51  
52          /**
53           * Return the scheme associated with the provided algorithm (e.g. SHA-1 returns urn:sha1)
54           *
55           * @param alg for which scheme is requested
56           * @return scheme
57           */
58          public static String getScheme(final String alg) {
59              return Arrays.stream(values()).filter(value ->
60                      value.algorithm.equalsIgnoreCase(alg) || value.algorithm.replace("-", "").equalsIgnoreCase(alg)
61              ).findFirst().orElse(MISSING).scheme;
62          }
63  
64          /**
65           * Return enum value for the provided scheme (e.g. urn:sha1 returns SHA-1)
66           *
67           * @param argScheme for which enum is requested
68           * @return enum value associated with the arg scheme
69           */
70          public static DIGEST_ALGORITHM fromScheme(final String argScheme) {
71              return Arrays.stream(values()).filter(value -> value.scheme.equalsIgnoreCase(argScheme)
72              ).findFirst().orElse(MISSING);
73          }
74  
75          /**
76           * Return true if the provided algorithm is included in this enum
77           *
78           * @param alg to test
79           * @return true if arg algorithm is supported
80           */
81          public static boolean isSupportedAlgorithm(final String alg) {
82              return !getScheme(alg).equals(MISSING.scheme);
83          }
84      }
85  
86      public static final String DEFAULT_ALGORITHM = DIGEST_ALGORITHM.SHA1.algorithm;
87  
88      private ContentDigest() {
89      }
90  
91      /**
92       * Convert a MessageDigest algorithm and checksum value to a URN
93       * @param algorithm the message digest algorithm
94       * @param value the checksum value
95       * @return URI
96       */
97      public static URI asURI(final String algorithm, final String value) {
98          try {
99              final String scheme = DIGEST_ALGORITHM.getScheme(algorithm);
100 
101             return new URI(scheme, value, null);
102         } catch (final URISyntaxException unlikelyException) {
103             LOGGER.warn("Exception creating checksum URI: {}",
104                                unlikelyException);
105             throw new RepositoryRuntimeException(unlikelyException);
106         }
107     }
108 
109     /**
110      * Convert a MessageDigest algorithm and checksum byte-array data to a URN
111      * @param algorithm the message digest algorithm
112      * @param data the checksum byte-array data
113      * @return URI
114      */
115     public static URI asURI(final String algorithm, final byte[] data) {
116         return asURI(algorithm, asString(data));
117     }
118 
119     /**
120      * Given a digest URI, get the corresponding MessageDigest algorithm
121      * @param digestUri the digest uri
122      * @return MessageDigest algorithm
123      */
124     public static String getAlgorithm(final URI digestUri) {
125         if (digestUri == null) {
126             return DEFAULT_ALGORITHM;
127         }
128         return DIGEST_ALGORITHM.fromScheme(digestUri.getScheme() + ":" +
129              digestUri.getSchemeSpecificPart().split(":", 2)[0]).algorithm;
130     }
131 
132     private static String asString(final byte[] data) {
133         return encodeHexString(data);
134     }
135 
136     /**
137      * Placeholder checksum value.
138      * @return URI
139      */
140     public static URI missingChecksum() {
141         return asURI(SHA1.algorithm, SHA1.scheme);
142     }
143 
144 }
