34  /**
35   * @author cabeer
36   * @since 10/15/14
37   */
38  public class NodeResourceConverter extends Converter<Node, FedoraResource> {
39      public static final NodeResourceConverterourceConverter.html#NodeResourceConverter">NodeResourceConverter nodeConverter = new NodeResourceConverter();
40  
41      /**
42       * Get a converter that can transform a Node to a Resource
43       * @param c the given node
44       * @return the converter that can transform a node to resource
45       */
46      public static Converter<Node, Resource> nodeToResource(final Converter<Resource, FedoraResource> c) {
47          return nodeConverter.andThen(c.reverse());
48      }
49  
50      @Override
51      protected FedoraResource doForward(final Node node) {
52  
53          final FedoraResource fedoraResource;
54  
55          if (NonRdfSourceDescriptionImpl.hasMixin(node)) {
56              fedoraResource = new NonRdfSourceDescriptionImpl(node);
57          } else if (FedoraBinaryImpl.hasMixin(node)) {
58              fedoraResource = new FedoraBinaryImpl(node);
59          } else if (TombstoneImpl.hasMixin(node)) {
60              fedoraResource = new TombstoneImpl(node);
61          } else if (FedoraTimeMapImpl.hasMixin(node)) {
62              fedoraResource = new FedoraTimeMapImpl(node);
63          } else if (FedoraWebacAclImpl.hasMixin(node)) {
64              fedoraResource = new FedoraWebacAclImpl(node);
65          } else {
66              fedoraResource = new ContainerImpl(node);
67          }
68  
69          return fedoraResource;
70      }
71  
72      @Override
73      protected Node doBackward(final FedoraResource resource) {
74          return getJcrNode(resource);
75      }
76  }
