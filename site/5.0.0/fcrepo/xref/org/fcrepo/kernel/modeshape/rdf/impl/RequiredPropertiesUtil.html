18  package org.fcrepo.kernel.modeshape.rdf.impl;

20  import static java.util.Arrays.asList;
21  import static org.fcrepo.kernel.api.RdfLexicon.CONTAINER;
22  import static org.fcrepo.kernel.api.RdfLexicon.CREATED_BY;
23  import static org.fcrepo.kernel.api.RdfLexicon.CREATED_DATE;
24  import static org.fcrepo.kernel.api.RdfLexicon.FEDORA_BINARY;
25  import static org.fcrepo.kernel.api.RdfLexicon.FEDORA_CONTAINER;
26  import static org.fcrepo.kernel.api.RdfLexicon.FEDORA_RESOURCE;
27  import static org.fcrepo.kernel.api.RdfLexicon.LAST_MODIFIED_BY;
28  import static org.fcrepo.kernel.api.RdfLexicon.LAST_MODIFIED_DATE;
29  import static org.fcrepo.kernel.api.RdfLexicon.NON_RDF_SOURCE;
30  import static org.fcrepo.kernel.api.RdfLexicon.RDF_SOURCE;
31  
32  import java.util.Collection;
33  import org.apache.jena.rdf.model.Model;
34  import org.apache.jena.rdf.model.Property;
35  import org.apache.jena.rdf.model.RDFNode;
36  import org.apache.jena.rdf.model.Resource;
37  import org.apache.jena.vocabulary.RDF;
38  import org.fcrepo.kernel.api.exception.ConstraintViolationException;
39  
40  /**
41   * Helper methods for determining if resources contain all required server managed triples.
42   *
43   * @author bbpennel
44   */
45  public class RequiredPropertiesUtil {
46  
47      private RequiredPropertiesUtil() {
48      }
49  
50      private final static Collection<Property> REQUIRED_PROPERTIES = asList(
51              CREATED_BY, CREATED_DATE, LAST_MODIFIED_DATE, LAST_MODIFIED_BY);
52  
53      private final static Collection<Resource> CONTAINER_TYPES = asList(
54              RDF_SOURCE, CONTAINER, FEDORA_CONTAINER, FEDORA_RESOURCE);
55  
56      private final static Collection<Resource> BINARY_TYPES = asList(
57              NON_RDF_SOURCE, FEDORA_BINARY, FEDORA_RESOURCE);
58  
59      /**
60       * Throws a ConstraintViolationException if the model does not contain all required server-managed triples for a
61       * container.
62       *
63       * @param model rdf to validate
64       * @throws ConstraintViolationException if model does not contain all required server-managed triples for container
65       */
66      public static void assertRequiredContainerTriples(final Model model)
67              throws ConstraintViolationException {
68          assertContainsRequiredProperties(model, REQUIRED_PROPERTIES);
69          assertContainsRequiredTypes(model, CONTAINER_TYPES);
70      }
71  
72      /**
73       * Throws a ConstraintViolationException if the model does not contain all required server-managed triples for a
74       * binary description.
75       *
76       * @param model rdf to validate
77       * @throws ConstraintViolationException if model does not contain all required server-managed triples for binary
78       * description
79       */
80      public static void assertRequiredDescriptionTriples(final Model model)
81              throws ConstraintViolationException {
82          assertContainsRequiredProperties(model, REQUIRED_PROPERTIES);
83          assertContainsRequiredTypes(model, BINARY_TYPES);
84      }
85  
86      private static void assertContainsRequiredProperties(final Model model, final Collection<Property> properties) {
87          for (final Property property : properties) {
88              if (!model.contains(null, property, (RDFNode) null)) {
89                  throw new ConstraintViolationException("Missing required property " +
90                          property.getURI());
91              }
92          }
93      }
94  
95      private static void assertContainsRequiredTypes(final Model model, final Collection<Resource> types) {
96          for (final Resource type : types) {
97              if (!model.contains(null, RDF.type, type)) {
98                  throw new ConstraintViolationException("Missing required type " +
99                          type.getURI());
100             }
101         }
102     }
103 }
