18  package org.fcrepo.auth.common;
19  
20  import static org.slf4j.LoggerFactory.getLogger;
21  
22  import java.util.Set;
23  
24  import org.apache.shiro.authc.AuthenticationException;
25  import org.apache.shiro.authc.AuthenticationInfo;
26  import org.apache.shiro.authc.AuthenticationToken;
27  import org.apache.shiro.authc.SimpleAuthenticationInfo;
28  import org.apache.shiro.realm.AuthenticatingRealm;
29  import org.apache.shiro.subject.SimplePrincipalCollection;
30  import org.fcrepo.auth.common.ContainerRolesPrincipalProvider.ContainerRolesPrincipal;
31  import org.slf4j.Logger;
32  
33  /**
34   * @author peichman
35   */
36  public class ServletContainerAuthenticatingRealm extends AuthenticatingRealm {
37  
38      private static final Logger log = getLogger(ServletContainerAuthenticatingRealm.class);
39  
40      @Override
41      public String getName() {
42          return "servlet container authentication";
43      }
44  
45      @Override
46      protected AuthenticationInfo doGetAuthenticationInfo(final AuthenticationToken token)
47              throws AuthenticationException {
48          final ContainerAuthToken/fcrepo/auth/common/ContainerAuthToken.html#ContainerAuthToken">ContainerAuthToken authToken = (ContainerAuthToken) token;
49          final SimplePrincipalCollection principals = new SimplePrincipalCollection();
50          log.debug("Creating principals from servlet container principal and roles");
51          // container-managed auth username
52          principals.add(authToken.getPrincipal(), getName());
53          // container-managed auth roles
54          final Set<ContainerRolesPrincipal> roles = authToken.getRoles();
55          if (!roles.isEmpty()) {
56              principals.addAll(roles, getName());
57          }
58          return new SimpleAuthenticationInfo(principals, ContainerAuthToken.AUTHORIZED);
59      }
60  
61      @Override
62      public boolean supports(final AuthenticationToken token) {
63          return token instanceof ContainerAuthToken;
64      }
65  
66  }
