018package org.fcrepo.persistence.ocfl.impl;
019
020import static org.fcrepo.persistence.ocfl.impl.OcflPersistentStorageUtils.createFilesystemRepository;
021import static org.fcrepo.persistence.ocfl.impl.OcflPersistentStorageUtils.createS3Repository;
022
023import java.io.IOException;
024import java.util.concurrent.TimeUnit;
025
026import javax.inject.Inject;
027import javax.sql.DataSource;
028
029import org.fcrepo.config.MetricsConfig;
030import org.fcrepo.config.OcflPropsConfig;
031import org.fcrepo.config.Storage;
032import org.fcrepo.storage.ocfl.CommitType;
033import org.fcrepo.storage.ocfl.DefaultOcflObjectSessionFactory;
034import org.fcrepo.storage.ocfl.validation.ObjectValidator;
035import org.fcrepo.storage.ocfl.OcflObjectSessionFactory;
036import org.fcrepo.storage.ocfl.ResourceHeaders;

038import org.fcrepo.storage.ocfl.cache.CaffeineCache;
039import org.fcrepo.storage.ocfl.cache.NoOpCache;
040
041import org.apache.commons.lang3.StringUtils;

043import org.springframework.context.annotation.Configuration;
044
045import com.github.benmanes.caffeine.cache.Caffeine;
046
047import edu.wisc.library.ocfl.api.MutableOcflRepository;
048import io.micrometer.core.instrument.MeterRegistry;
049import io.micrometer.core.instrument.binder.cache.CaffeineCacheMetrics;
050import software.amazon.awssdk.auth.credentials.AwsBasicCredentials;
051import software.amazon.awssdk.auth.credentials.StaticCredentialsProvider;
052import software.amazon.awssdk.regions.Region;
053import software.amazon.awssdk.services.s3.S3Client;
054
055/**
056 * A Configuration for OCFL dependencies
057 *
058 * @author dbernstein
059 * @since 6.0.0
060 */
061
062@Configuration
063public class OcflPersistenceConfig {
064
065    @Inject
066    private OcflPropsConfig ocflPropsConfig;
067
068    @Inject
069    private MetricsConfig metricsConfig;
070
071    @Inject
072    private MeterRegistry meterRegistry;
073
074    @Inject
075    private DataSource dataSource;
076
077    /**
078     * Create an OCFL Repository
079     * @return the repository
080     */
081    @Bean
082    public MutableOcflRepository repository() throws IOException {
083        if (ocflPropsConfig.getStorage() == Storage.OCFL_S3) {
084            return createS3Repository(
085                    dataSource,
086                    s3Client(),
087                    ocflPropsConfig.getOcflS3Bucket(),
088                    ocflPropsConfig.getOcflS3Prefix(),
089                    ocflPropsConfig.getOcflTemp(),
090                    ocflPropsConfig.getDefaultDigestAlgorithm());
091        } else {
092            return createFilesystemRepository(ocflPropsConfig.getOcflRepoRoot(), ocflPropsConfig.getOcflTemp(),
093                    ocflPropsConfig.getDefaultDigestAlgorithm());
094        }
095    }
096
097    @Bean
098    public OcflObjectSessionFactory ocflObjectSessionFactory() throws IOException {
099        final var objectMapper = OcflPersistentStorageUtils.objectMapper();
100
101        return new DefaultOcflObjectSessionFactory(repository(),
102                ocflPropsConfig.getFedoraOcflStaging(),
103                objectMapper,
104                resourceHeadersCache(),
105                commitType(),
106                "Authored by Fedora 6",
107                "fedoraAdmin",
108                "info:fedora/fedoraAdmin");
109    }
110
111    @Bean
112    public ObjectValidator objectValidator() throws IOException {
113        final var objectMapper = OcflPersistentStorageUtils.objectMapper();
114        return new ObjectValidator(repository(), objectMapper.readerFor(ResourceHeaders.class));
115    }
116
117    private CommitType commitType() {
118        if (ocflPropsConfig.isAutoVersioningEnabled()) {
119            return CommitType.NEW_VERSION;
120        }
121        return CommitType.UNVERSIONED;
122    }
123
124    private S3Client s3Client() {
125        final var builder = S3Client.builder();
126
127        if (StringUtils.isNotBlank(ocflPropsConfig.getAwsRegion())) {
128            builder.region(Region.of(ocflPropsConfig.getAwsRegion()));
129        }
130
131        if (StringUtils.isNoneBlank(ocflPropsConfig.getAwsAccessKey(), ocflPropsConfig.getAwsSecretKey())) {
132            builder.credentialsProvider(StaticCredentialsProvider.create(
133                    AwsBasicCredentials.create(ocflPropsConfig.getAwsAccessKey(), ocflPropsConfig.getAwsSecretKey())));
134        }
135
136        // May want to do additional HTTP client configuration, connection pool, etc
137
138        return builder.build();
139    }
140
141    private Cache<String, ResourceHeaders> resourceHeadersCache() {
142        if (ocflPropsConfig.isResourceHeadersCacheEnabled()) {
143            final var builder = Caffeine.newBuilder();
144
145            if (metricsConfig.isMetricsEnabled()) {
146                builder.recordStats();
147            }
148
149            final var cache = builder
150                    .maximumSize(ocflPropsConfig.getResourceHeadersCacheMaxSize())
151                    .expireAfterAccess(ocflPropsConfig.getResourceHeadersCacheExpireAfterSeconds(), TimeUnit.SECONDS)
152                    .build();
153
154            if (metricsConfig.isMetricsEnabled()) {
155                CaffeineCacheMetrics.monitor(meterRegistry, cache, "resourceHeadersCache");
156            }
157
158            return new CaffeineCache<>(cache);
159        }
160
161        return new NoOpCache<>();
162    }
163
164}
