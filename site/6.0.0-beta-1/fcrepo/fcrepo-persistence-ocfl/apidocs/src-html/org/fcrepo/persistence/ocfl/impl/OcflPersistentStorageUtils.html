053/**
054 * A set of utility functions for supporting OCFL persistence activities.
055 *
056 * @author dbernstein
057 * @since 6.0.0
058 */
059public class OcflPersistentStorageUtils {
060
061    private static final Logger log = LoggerFactory.getLogger(OcflPersistentStorageUtils.class);
062
063    private OcflPersistentStorageUtils() {
064    }
065
066    /**
067     * The default RDF on disk format
068     * TODO Make this value configurable
069     */
070
071    private static RDFFormat DEFAULT_RDF_FORMAT = NTRIPLES;
072
073    /**
074     * @return the RDF Format. By default NTRIPLES are returned.
075     */
076    public static RDFFormat getRdfFormat() {
077        return DEFAULT_RDF_FORMAT;
078    }
079
080    /**
081     * @return the RDF file extension.
082     */
083    public static String getRDFFileExtension() {
084        return "." + DEFAULT_RDF_FORMAT.getLang().getFileExtensions().get(0);
085    }
086
087    /**
088     * Create a new ocfl repository backed by the filesystem
089     * @param ocflStorageRootDir The ocfl storage root directory
090     * @param ocflWorkDir The ocfl work directory
091     * @param algorithm the algorithm for the OCFL repository
092     * @return the repository
093     */
094    public static MutableOcflRepository createFilesystemRepository(final Path ocflStorageRootDir,
095                                                                   final Path ocflWorkDir,
096                                                                   final org.fcrepo.config.DigestAlgorithm algorithm)
097            throws IOException {
098        Files.createDirectories(ocflStorageRootDir);
099
100        final var storage = FileSystemOcflStorage.builder().repositoryRoot(ocflStorageRootDir).build();
101
102        return createRepository(ocflWorkDir, builder -> {
103            builder.storage(storage);
104        }, algorithm);
105    }
106
107    /**
108     * Create a new ocfl repository backed by s3
109     *
110     * @param dataSource the datasource to keep inventories in and use as a lock
111     * @param s3Client aws s3 client
112     * @param bucket the bucket to store objects in
113     * @param prefix the prefix within the bucket to store objects under
114     * @param ocflWorkDir the local directory to stage objects in
115     * @param algorithm the algorithm for the OCFL repository
116     * @return the repository
117     */
118    public static MutableOcflRepository createS3Repository(final DataSource dataSource,
119                                                           final S3Client s3Client,
120                                                           final String bucket,
121                                                           final String prefix,
122                                                           final Path ocflWorkDir,
123                                                           final org.fcrepo.config.DigestAlgorithm algorithm)
124            throws IOException {
125        Files.createDirectories(ocflWorkDir);
126
127        final var storage = CloudOcflStorage.builder()
128                .cloudClient(OcflS3Client.builder()
129                        .s3Client(s3Client)
130                        .bucket(bucket)
131                        .repoPrefix(prefix)
132                        .build())
133                .build();
134
135        return createRepository(ocflWorkDir, builder -> {
136            builder.contentPathConstraints(ContentPathConstraints.cloud())
137                    .objectLock(lock -> lock.dataSource(dataSource))
138                    .objectDetailsDb(db -> db.dataSource(dataSource))
139                    .storage(storage);
140        }, algorithm);
141    }
142
143    private static MutableOcflRepository createRepository(final Path ocflWorkDir,
144                                                          final Consumer<OcflRepositoryBuilder> configurer,
145                                                          final org.fcrepo.config.DigestAlgorithm algorithm)
146            throws IOException {
147        Files.createDirectories(ocflWorkDir);
148
149        final DigestAlgorithm ocflDigestAlg = translateFedoraDigestToOcfl(algorithm);
150        if (ocflDigestAlg == null) {
151            throw new UnsupportedDigestAlgorithmException(
152                    "Unable to map Fedora default digest algorithm " + algorithm + " into OCFL");
153        }
154
155        final var logicalPathMapper = SystemUtils.IS_OS_WINDOWS ?
156                LogicalPathMappers.percentEncodingWindowsMapper() : LogicalPathMappers.percentEncodingLinuxMapper();
157
158        final var builder = new OcflRepositoryBuilder()
159                .defaultLayoutConfig(new HashedNTupleLayoutConfig())
160                .ocflConfig(new OcflConfig().setDefaultDigestAlgorithm(ocflDigestAlg))
161                .logicalPathMapper(logicalPathMapper)
162                .workDir(ocflWorkDir);
163
164        configurer.accept(builder);
165
166        return builder.buildMutable();
167    }
168
169    /**
170     * @return new object mapper with default config
171     */
172    public static ObjectMapper objectMapper() {
173        return new ObjectMapper()
174                .configure(WRITE_DATES_AS_TIMESTAMPS, false)
175                .registerModule(new JavaTimeModule())
176                .setSerializationInclusion(JsonInclude.Include.NON_NULL);
177    }
178
179    /**
180     * Translates the provided fedora digest algorithm enum into a OCFL client digest algorithm
181     *
182     * @param fcrepoAlg fedora digest algorithm
183     * @return OCFL client DigestAlgorithm, or null if no match could be made
184     */
185    public static DigestAlgorithm translateFedoraDigestToOcfl(final org.fcrepo.config.DigestAlgorithm fcrepoAlg) {
186        return fcrepoAlg.getAliases().stream()
187                .map(alias -> DigestAlgorithmRegistry.getAlgorithm(alias))
188                .filter(alg -> alg != null)
189                .findFirst()
190                .orElse(null);
191    }
