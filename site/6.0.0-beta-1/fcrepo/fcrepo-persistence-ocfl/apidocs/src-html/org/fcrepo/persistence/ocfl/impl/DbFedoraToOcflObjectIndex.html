047/**
048 * Maps Fedora IDs to the OCFL IDs of the OCFL objects the Fedora resource is stored in. This implementation is backed
049 * by a relational database.
050 *
051 * @author pwinckles
052 */
053@Component("ocflIndexImpl")
054public class DbFedoraToOcflObjectIndex implements FedoraToOcflObjectIndex {
055
056    private static final Logger LOGGER = LoggerFactory.getLogger(DbFedoraToOcflObjectIndex.class);
057
058    private static final String MAPPING_TABLE = "ocfl_id_map";
059
060    private static final String FEDORA_ID_COLUMN = "fedora_id";
061
062    private static final String FEDORA_ROOT_ID_COLUMN = "fedora_root_id";
063
064    private static final String OCFL_ID_COLUMN = "ocfl_id";
065
066    private static final String TRANSACTION_OPERATIONS_TABLE = "ocfl_id_map_session_operations";
067
068    private static final String TRANSACTION_ID_COLUMN = "session_id";
069
070    private static final String OPERATION_COLUMN = "operation";
071
072    /*
073     * Lookup all mappings for the resource id.
074     */
075    private static final String LOOKUP_MAPPING = "SELECT " + FEDORA_ROOT_ID_COLUMN + ", " + OCFL_ID_COLUMN + " FROM " +
076            MAPPING_TABLE + " WHERE " + FEDORA_ID_COLUMN + " = :fedoraId";
077
078    /*
079     * Lookup all mappings from the mapping table as well as any new 'add's and excluding any 'delete's in this
080     * transaction.
081     */
082    private static final String LOOKUP_MAPPING_IN_TRANSACTION = "SELECT x." + FEDORA_ROOT_ID_COLUMN + "," +
083            " x." + OCFL_ID_COLUMN + " FROM" +
084            " (SELECT " + FEDORA_ROOT_ID_COLUMN + ", " + OCFL_ID_COLUMN + " FROM " + MAPPING_TABLE + " WHERE " +
085            FEDORA_ID_COLUMN + " = :fedoraId" +
086            " UNION SELECT " + FEDORA_ROOT_ID_COLUMN + ", " + OCFL_ID_COLUMN + " FROM " + TRANSACTION_OPERATIONS_TABLE +
087            " WHERE " + FEDORA_ID_COLUMN + " = :fedoraId AND " + TRANSACTION_ID_COLUMN + " = :transactionId" +
088            " AND " + OPERATION_COLUMN + " = 'add') x";
089
090    /*
091     * Add an 'add' operation to the transaction table.
092     */
093    private static final String UPSERT_MAPPING_TX_POSTGRESQL = "INSERT INTO " + TRANSACTION_OPERATIONS_TABLE +
094            " ( " + FEDORA_ID_COLUMN + ", " + FEDORA_ROOT_ID_COLUMN + ", " + OCFL_ID_COLUMN + ", " +
095            TRANSACTION_ID_COLUMN + ", " + OPERATION_COLUMN + ") VALUES (:fedoraId, :fedoraRootId, :ocflId," +
096            " :transactionId, :operation) ON CONFLICT (" + FEDORA_ID_COLUMN + ", " + TRANSACTION_ID_COLUMN + ")" +
097            " DO UPDATE SET " + FEDORA_ROOT_ID_COLUMN + " = EXCLUDED." + FEDORA_ROOT_ID_COLUMN + ", " +
098            OCFL_ID_COLUMN + " = EXCLUDED." + OCFL_ID_COLUMN + ", " + OPERATION_COLUMN + " = EXCLUDED." +
099            OPERATION_COLUMN;
100
101    private static final String UPSERT_MAPPING_TX_MYSQL_MARIA = "INSERT INTO " + TRANSACTION_OPERATIONS_TABLE +
102            " (" + FEDORA_ID_COLUMN + ", " + FEDORA_ROOT_ID_COLUMN + ", " + OCFL_ID_COLUMN + ", " +
103            TRANSACTION_ID_COLUMN + ", " + OPERATION_COLUMN + ")" +
104            " VALUES (:fedoraId, :fedoraRootId, :ocflId, :transactionId, :operation) ON DUPLICATE KEY UPDATE " +
105            FEDORA_ROOT_ID_COLUMN + " = VALUES(" + FEDORA_ROOT_ID_COLUMN + "), " + OCFL_ID_COLUMN + " = VALUES(" +
106            OCFL_ID_COLUMN + "), " + OPERATION_COLUMN + " = VALUES(" + OPERATION_COLUMN + ")";
107
108    private static final String UPSERT_MAPPING_TX_H2 = "MERGE INTO " + TRANSACTION_OPERATIONS_TABLE +
109            " (" + FEDORA_ID_COLUMN + ", " + FEDORA_ROOT_ID_COLUMN + ", " + OCFL_ID_COLUMN + ", " +
110            TRANSACTION_ID_COLUMN + ", " + OPERATION_COLUMN + ")" +
111            " KEY (" + FEDORA_ID_COLUMN + ", " + TRANSACTION_ID_COLUMN + ")" +
112            " VALUES (:fedoraId, :fedoraRootId, :ocflId, :transactionId, :operation)";
113
114    /**
115     * Map of database product to UPSERT into operations table SQL.
116     */
117    private static final Map<DbPlatform, String> UPSERT_MAPPING_TX_MAP = Map.of(
118            DbPlatform.MYSQL, UPSERT_MAPPING_TX_MYSQL_MARIA,
119            DbPlatform.H2, UPSERT_MAPPING_TX_H2,
120            DbPlatform.POSTGRESQL, UPSERT_MAPPING_TX_POSTGRESQL,
121            DbPlatform.MARIADB, UPSERT_MAPPING_TX_MYSQL_MARIA
122    );
123
124    private static final String COMMIT_ADD_MAPPING_POSTGRESQL = "INSERT INTO " + MAPPING_TABLE +
125            " ( " + FEDORA_ID_COLUMN + ", " + FEDORA_ROOT_ID_COLUMN + ", " + OCFL_ID_COLUMN + ") SELECT " +
126            FEDORA_ID_COLUMN + ", " + FEDORA_ROOT_ID_COLUMN + ", " + OCFL_ID_COLUMN + " FROM " +
127            TRANSACTION_OPERATIONS_TABLE + " WHERE " + OPERATION_COLUMN + " = 'add' AND " + TRANSACTION_ID_COLUMN +
128            " = :transactionId ON CONFLICT ( " +  FEDORA_ID_COLUMN + " )" +
129            " DO UPDATE SET " + FEDORA_ROOT_ID_COLUMN + " = EXCLUDED." + FEDORA_ROOT_ID_COLUMN + ", " +
130            OCFL_ID_COLUMN + " = EXCLUDED." + OCFL_ID_COLUMN;
131
132    private static final String COMMIT_ADD_MAPPING_MYSQL_MARIA = "INSERT INTO " + MAPPING_TABLE +
133            " (" + FEDORA_ID_COLUMN + ", " + FEDORA_ROOT_ID_COLUMN + ", " + OCFL_ID_COLUMN + ") SELECT " +
134            FEDORA_ID_COLUMN + ", " + FEDORA_ROOT_ID_COLUMN + ", " + OCFL_ID_COLUMN + " FROM " +
135            TRANSACTION_OPERATIONS_TABLE + " WHERE " + OPERATION_COLUMN + " = 'add' AND " + TRANSACTION_ID_COLUMN +
136            " = :transactionId ON DUPLICATE KEY UPDATE " +
137            FEDORA_ROOT_ID_COLUMN + " = VALUES(" + FEDORA_ROOT_ID_COLUMN + "), " + OCFL_ID_COLUMN + " = VALUES(" +
138            OCFL_ID_COLUMN + ")";
139
140    private static final String COMMIT_ADD_MAPPING_H2 = "MERGE INTO " + MAPPING_TABLE +
141            " (" + FEDORA_ID_COLUMN + ", " + FEDORA_ROOT_ID_COLUMN + ", " + OCFL_ID_COLUMN + ")" +
142            " SELECT " + FEDORA_ID_COLUMN + ", " + FEDORA_ROOT_ID_COLUMN + ", " + OCFL_ID_COLUMN + " FROM " +
143            TRANSACTION_OPERATIONS_TABLE + " WHERE " + OPERATION_COLUMN + " = 'add'";
144
145    /**
146     * Map of database product name to COMMIT to mapping table from operations table
147     */
148    private static final Map<DbPlatform, String> COMMIT_ADD_MAPPING_MAP = Map.of(
149            DbPlatform.MYSQL, COMMIT_ADD_MAPPING_MYSQL_MARIA,
150            DbPlatform.H2, COMMIT_ADD_MAPPING_H2,
151            DbPlatform.POSTGRESQL, COMMIT_ADD_MAPPING_POSTGRESQL,
152            DbPlatform.MARIADB, COMMIT_ADD_MAPPING_MYSQL_MARIA
153    );
154
155    /*
156     * Delete records from the mapping table that are to be deleted in this transaction.
157     */
158    private static final String COMMIT_DELETE_RECORDS = "DELETE FROM " + MAPPING_TABLE + " WHERE " +
159            "EXISTS (SELECT * FROM " + TRANSACTION_OPERATIONS_TABLE + " WHERE " +
160            TRANSACTION_ID_COLUMN + " = :transactionId AND " +  OPERATION_COLUMN + " = 'delete' AND " +
161            MAPPING_TABLE + "." + FEDORA_ID_COLUMN + " = " + TRANSACTION_OPERATIONS_TABLE + "." + FEDORA_ID_COLUMN +
162            ")";
163
164    private static final String TRUNCATE_MAPPINGS = "TRUNCATE TABLE " + MAPPING_TABLE;
165
166    private static final String TRUNCATE_TRANSACTIONS = "TRUNCATE TABLE " + TRANSACTION_OPERATIONS_TABLE;
167
168    /*
169     * Delete all records from the transaction table for the specified transaction.
170     */
171    private static final String DELETE_ENTIRE_TRANSACTION = "DELETE FROM " + TRANSACTION_OPERATIONS_TABLE + " WHERE " +
172            TRANSACTION_ID_COLUMN + " = :transactionId";
173
174    /*
175     * Row mapper for the Lookup queries.
176     */
177    private static final RowMapper<FedoraOcflMapping> GET_MAPPING_ROW_MAPPER = (resultSet, i) -> new FedoraOcflMapping(
178            FedoraId.create(resultSet.getString(1)),
179            resultSet.getString(2)
180    );
181
182    private final DataSource dataSource;
183
184    private final NamedParameterJdbcTemplate jdbcTemplate;
185
186    private DbPlatform dbPlatform;
187
188    public DbFedoraToOcflObjectIndex(@Autowired final DataSource dataSource) {
189        this.dataSource = dataSource;
190        this.jdbcTemplate = new NamedParameterJdbcTemplate(dataSource);
191    }
192
193    @PostConstruct
194    public void setup() {
195        dbPlatform = DbPlatform.fromDataSource(dataSource);
196    }
197
198    @Override
199    public FedoraOcflMapping getMapping(final String transactionId, final FedoraId fedoraId)
200            throws FedoraOcflMappingNotFoundException {
201        try {
202            final MapSqlParameterSource parameterSource = new MapSqlParameterSource();
203            parameterSource.addValue("fedoraId", fedoraId.getResourceId());
204            if (transactionId != null) {
205                parameterSource.addValue("transactionId", transactionId);
206                return jdbcTemplate.queryForObject(LOOKUP_MAPPING_IN_TRANSACTION, parameterSource,
207                        GET_MAPPING_ROW_MAPPER);
208            } else {
209                return jdbcTemplate.queryForObject(LOOKUP_MAPPING, parameterSource, GET_MAPPING_ROW_MAPPER);
210            }
211        } catch (final EmptyResultDataAccessException e) {
212            throw new FedoraOcflMappingNotFoundException("No OCFL mapping found for " + fedoraId);
213        }
214    }
215
216    @Override
217    public FedoraOcflMapping addMapping(@Nonnull final String transactionId, final FedoraId fedoraId,
218                                        final FedoraId fedoraRootId, final String ocflId) {
219        upsert(transactionId, fedoraId, "add", fedoraRootId, ocflId);
220        return new FedoraOcflMapping(fedoraRootId, ocflId);
221    }
222
223    @Override
224    public void removeMapping(@Nonnull final String transactionId, final FedoraId fedoraId) {
225        upsert(transactionId, fedoraId, "delete");
226    }
227
228    private void upsert(final String transactionId, final FedoraId fedoraId, final String operation) {
229        upsert(transactionId, fedoraId, operation, null, null);
230    }
231
232    /**
233     * Perform the upsert to the operations table.
234     *
235     * @param transactionId the transaction/session id.
236     * @param fedoraId the resource id.
237     * @param operation the operation we are performing (add or delete)
238     * @param fedoraRootId the fedora root id (for add only)
239     * @param ocflId the ocfl id (for add only).
240     */
241    private void upsert(final String transactionId, final FedoraId fedoraId, final String operation,
242                        final FedoraId fedoraRootId, final String ocflId) {
243        final MapSqlParameterSource parameterSource = new MapSqlParameterSource();
244        parameterSource.addValue("fedoraId", fedoraId.getResourceId());
245        parameterSource.addValue("fedoraRootId", fedoraRootId == null ? null : fedoraRootId.getResourceId());
246        parameterSource.addValue("ocflId", ocflId);
247        parameterSource.addValue("transactionId", transactionId);
248        parameterSource.addValue("operation", operation);
249        try {
250            jdbcTemplate.update(UPSERT_MAPPING_TX_MAP.get(dbPlatform), parameterSource);
251        } catch (final DataIntegrityViolationException | BadSqlGrammarException e) {
252            if (e.getMessage().contains("too long for")) {
253                throw new InvalidResourceIdentifierException("Database error - Fedora ID path too long",e);
254            } else {
255                throw new RepositoryRuntimeException("Database error - error during upsert",e);
256            }
257        }
258    }
259
260    @Transactional
261    @Override
262    public void reset() {
263        try {
264            jdbcTemplate.update(TRUNCATE_MAPPINGS, Collections.emptyMap());
265            jdbcTemplate.update(TRUNCATE_TRANSACTIONS, Collections.emptyMap());
266        } catch (final Exception e) {
267            throw new RepositoryRuntimeException("Failed to truncate FedoraToOcfl index tables", e);
268        }
269    }
270
271    @Transactional
272    @Override
273    public void commit(@Nonnull final String sessionId) {
274        LOGGER.debug("Committing FedoraToOcfl index changes from transaction {}", sessionId);
275        final Map<String, String> map = Map.of("transactionId", sessionId);
276        try {
277            jdbcTemplate.update(COMMIT_DELETE_RECORDS, map);
278            jdbcTemplate.update(COMMIT_ADD_MAPPING_MAP.get(dbPlatform), map);
279            jdbcTemplate.update(DELETE_ENTIRE_TRANSACTION, map);
280        } catch (final Exception e) {
281            LOGGER.warn("Unable to commit FedoraToOcfl index transaction {}: {}", sessionId, e.getMessage());
282            throw new RepositoryRuntimeException("Unable to commit FedoraToOcfl index transaction", e);
283        }
284    }
285
286    @Override
287    public void rollback(@Nonnull final String sessionId) {
288        jdbcTemplate.update(DELETE_ENTIRE_TRANSACTION, Map.of("transactionId", sessionId));
289    }
