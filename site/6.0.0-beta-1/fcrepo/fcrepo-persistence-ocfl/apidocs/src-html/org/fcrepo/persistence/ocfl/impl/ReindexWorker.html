018package org.fcrepo.persistence.ocfl.impl;
019
020import static org.slf4j.LoggerFactory.getLogger;
021
022import java.util.List;
023import java.util.concurrent.TimeUnit;
024
025import org.slf4j.Logger;
026
027import com.google.common.base.Stopwatch;
028
029/**
030 * A reindexing worker thread.
031 * @author whikloj
032 */
033public class ReindexWorker implements Runnable {
034
035    private static final Logger LOGGER = getLogger(ReindexWorker.class);
036
037    private static final long REPORTING_INTERVAL_SECS = 30;
038
039    private Thread t;
040    private ReindexManager manager;
041    private ReindexService service;
042    private String transactionId;
043    private boolean running = true;
044    private boolean failOnError;
045
046    /**
047     * Basic Constructor
048     * @param reindexManager the manager service.
049     * @param reindexService the reindexing service.
050     * @param txId the transaction id.
051     * @param failOnError whether the thread should fail on an error or log and continue.
052     */
053    public ReindexWorker(final ReindexManager reindexManager, final ReindexService reindexService,
054                         final String txId, final boolean failOnError) {
055        manager = reindexManager;
056        service = reindexService;
057        transactionId = txId;
058        this.failOnError = failOnError;
059        t = new Thread(this, "ReindexWorker");
060    }
061
062    /**
063     * Join the thread.
064     * @throws InterruptedException if the current thread is interrupted.
065     */
066    public void join() throws InterruptedException {
067        t.join();
068    }
069
070    /**
071     * Start the thread with this Runnable
072     */
073    public void start() {
074        t.start();
075    }
076
077    @Override
078    public void run() {
079        final var stopwatch = Stopwatch.createStarted();
080        while (running) {
081            final List<String> ids = manager.getIds();
082            if (ids.isEmpty()) {
083                stopThread();
084                break;
085            }
086
087            int completed = 0;
088            int errors = 0;
089
090            for (final var id : ids) {
091                if (!running) {
092                    break;
093                }
094                if (stopwatch.elapsed(TimeUnit.SECONDS) > REPORTING_INTERVAL_SECS) {
095                    manager.updateComplete(completed, errors);
096                    completed = 0;
097                    errors = 0;
098                    stopwatch.reset().start();
099                }
100                try {
101                    service.indexOcflObject(transactionId, id);
102                    completed += 1;
103                } catch (final Exception e) {
104                    errors += 1;
105                    if (failOnError) {
106                        stopThread();
107                        manager.updateComplete(completed, errors);
108                        manager.stop();
109                        service.cleanupSession(transactionId);
110                        throw e;
111                    }
112                    LOGGER.error("Reindexing of OCFL id {} failed", id, e);
113                }
114            }
115            manager.updateComplete(completed, errors);
116            service.cleanupSession(transactionId);
117        }
118    }
119
120    /**
121     * Stop this thread from running once it has completed its current batch.
122     */
123    public void stopThread() {
124        this.running = false;
125    }
126
127}
