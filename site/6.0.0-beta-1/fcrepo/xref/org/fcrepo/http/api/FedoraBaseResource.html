18  package org.fcrepo.http.api;
19  
20  import org.fcrepo.http.commons.AbstractResource;
21  import org.fcrepo.http.commons.api.rdf.HttpIdentifierConverter;
22  import org.fcrepo.http.commons.session.TransactionProvider;
23  import org.fcrepo.kernel.api.Transaction;
24  import org.fcrepo.kernel.api.TransactionManager;
25  import org.fcrepo.kernel.api.exception.PathNotFoundException;
26  import org.fcrepo.kernel.api.identifiers.FedoraId;
27  import org.fcrepo.kernel.api.models.FedoraResource;
28  import org.fcrepo.kernel.api.models.ResourceFactory;
29  import org.fcrepo.kernel.api.models.ResourceHelper;
30  import org.slf4j.Logger;
31  
32  import javax.inject.Inject;
33  import javax.servlet.http.HttpServletRequest;
34  import javax.ws.rs.core.Context;
35  import javax.ws.rs.core.SecurityContext;
36  import java.security.Principal;
37  
38  import static org.slf4j.LoggerFactory.getLogger;
39  
40  /**
41   * @author cabeer
42   * @since 10/5/14
43   */
44  abstract public class FedoraBaseResource extends AbstractResource {
45  
46      private static final Logger LOGGER = getLogger(FedoraBaseResource.class);
47  
48      @Context
49      protected SecurityContext securityContext;
50  
51      @Inject
52      protected ResourceFactory resourceFactory;
53  
54      @Inject
55      protected ResourceHelper resourceHelper;
56  
57      @Context protected HttpServletRequest servletRequest;
58  
59      @Inject
60      protected TransactionManager txManager;
61  
62      private TransactionProvider txProvider;
63  
64      private Transaction transaction;
65  
66      protected HttpIdentifierConverter identifierConverter;
67  
68      protected HttpIdentifierConverter identifierConverter() {
69          if (identifierConverter == null) {
70              identifierConverter = new HttpIdentifierConverter(
71                      uriInfo.getBaseUriBuilder().clone().path(FedoraLdp.class));
72          }
73  
74          return identifierConverter;
75      }
76  
77      /**
78       * Gets a fedora resource by id. Uses the provided transaction if it is uncommitted,
79       * or uses a new transaction.
80       *
81       * @param transaction the fedora transaction
82       * @param fedoraId    identifier of the resource
83       * @return the requested FedoraResource
84       */
85      protected FedoraResource getFedoraResource(final Transaction transaction, final FedoraId fedoraId)
86              throws PathNotFoundException {
87          return resourceFactory.getResource(transaction, fedoraId);
88      }
89  
90      /**
91       * @param transaction the transaction in which to check
92       * @param fedoraId identifier of the object to check
93       * @param includeDeleted Whether to check for deleted resources too.
94       * @return Returns true if an object with the provided id exists
95       */
96      protected boolean doesResourceExist(final Transaction transaction, final FedoraId fedoraId,
97                                          final boolean includeDeleted) {
98          return resourceHelper.doesResourceExist(transaction, fedoraId, includeDeleted);
99      }
100 
101     /**
102      *
103      * @param transaction the transaction in which to check
104      * @param fedoraId identifier of the object to check
105      * @return Returns true if object does not exist but whose ID starts other resources that do exist.
106      */
107     protected boolean isGhostNode(final Transaction transaction, final FedoraId fedoraId) {
108         return resourceHelper.isGhostNode(transaction, fedoraId);
109     }
110 
111     protected String getUserPrincipal() {
112         final Principal p = securityContext.getUserPrincipal();
113         return p == null ? null : p.getName();
114     }
115 
116     protected Transaction transaction() {
117         if (transaction == null) {
118             txProvider = new TransactionProvider(txManager, servletRequest,
119                     uriInfo.getBaseUri(), fedoraPropsConfig.getJmsBaseUrl());
120             transaction = txProvider.provide();
121         }
122         return transaction;
123     }
124 
125 }
