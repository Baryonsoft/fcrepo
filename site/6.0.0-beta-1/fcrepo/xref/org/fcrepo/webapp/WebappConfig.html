19  package org.fcrepo.webapp;
20  
21  import org.fcrepo.config.FedoraPropsConfig;
22  import org.fcrepo.http.api.ExternalContentHandlerFactory;
23  import org.fcrepo.http.api.ExternalContentPathValidator;
24  import org.fcrepo.kernel.api.rdf.RdfNamespaceRegistry;
25  
26  import org.apache.http.conn.HttpClientConnectionManager;
27  import org.apache.http.impl.conn.PoolingHttpClientConnectionManager;
28  import org.springframework.context.annotation.Bean;
29  import org.springframework.context.annotation.Configuration;
30  import org.springframework.scheduling.concurrent.ThreadPoolTaskScheduler;
31  
32  import com.google.common.eventbus.EventBus;
33  
34  /**
35   * Spring config for the webapp
36   *
37   * @author pwinckles
38   */
39  @Configuration
40  public class WebappConfig {
41  
42      /**
43       * Task scheduler used for cleaning up transactions
44       *
45       * @return scheduler
46       */
47      @Bean
48      public ThreadPoolTaskScheduler taskScheduler() {
49          final var scheduler = new ThreadPoolTaskScheduler();
50          scheduler.setPoolSize(1);
51          scheduler.setThreadNamePrefix("ScheduledTask");
52          return scheduler;
53      }
54  
55      /**
56       * HTTP connection manager
57       *
58       * @return connection manager
59       */
60      @Bean
61      public HttpClientConnectionManager connectionManager() {
62          return new PoolingHttpClientConnectionManager();
63      }
64  
65      /**
66       * Fedora's lightweight internal event bus. Currently memory-resident.
67       *
68       * @return event bus
69       */
70      @Bean
71      public EventBus eventBus() {
72          // TODO this should be an async event bus: https://jira.lyrasis.org/browse/FCREPO-3587
73          return new EventBus();
74      }
75  
76      /**
77       * Configuration of namespace prefixes
78       *
79       * @param propsConfig config properties
80       * @return rdf namespace registry
81       */
82      @Bean(initMethod = "init", destroyMethod = "shutdown")
83      public RdfNamespaceRegistry rdfNamespaceRegistry(final FedoraPropsConfig propsConfig) {
84          final var registry = new RdfNamespaceRegistry();
85          registry.setConfigPath(propsConfig.getNamespaceRegistry());
86          registry.setMonitorForChanges(true);
87          return registry;
88      }
89  
90      /**
91       * External content configuration
92       *
93       * @param propsConfig config properties
94       * @return external content path validator
95       */
96      @Bean(initMethod = "init", destroyMethod = "shutdown")
97      public ExternalContentPathValidator externalContentPathValidator(final FedoraPropsConfig propsConfig) {
98          final var validator = new ExternalContentPathValidator();
99          validator.setConfigPath(propsConfig.getExternalContentAllowed());
100         validator.setMonitorForChanges(true);
101         return validator;
102     }
103 
104     @Bean
105     public ExternalContentHandlerFactory externalContentHandlerFactory(final ExternalContentPathValidator validator) {
106         final var factory = new ExternalContentHandlerFactory();
107         factory.setValidator(validator);
108         return factory;
109     }
110 
111 }
