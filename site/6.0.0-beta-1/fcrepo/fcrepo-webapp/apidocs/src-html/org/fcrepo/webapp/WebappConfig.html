019package org.fcrepo.webapp;
020
021import org.fcrepo.config.FedoraPropsConfig;
022import org.fcrepo.http.api.ExternalContentHandlerFactory;
023import org.fcrepo.http.api.ExternalContentPathValidator;
024import org.fcrepo.kernel.api.rdf.RdfNamespaceRegistry;
025
026import org.apache.http.conn.HttpClientConnectionManager;
027import org.apache.http.impl.conn.PoolingHttpClientConnectionManager;
028import org.springframework.context.annotation.Bean;
029import org.springframework.context.annotation.Configuration;
030import org.springframework.scheduling.concurrent.ThreadPoolTaskScheduler;
031
032import com.google.common.eventbus.EventBus;
033
034/**
035 * Spring config for the webapp
036 *
037 * @author pwinckles
038 */
039@Configuration
040public class WebappConfig {
041
042    /**
043     * Task scheduler used for cleaning up transactions
044     *
045     * @return scheduler
046     */
047    @Bean
048    public ThreadPoolTaskScheduler taskScheduler() {
049        final var scheduler = new ThreadPoolTaskScheduler();
050        scheduler.setPoolSize(1);
051        scheduler.setThreadNamePrefix("ScheduledTask");
052        return scheduler;
053    }
054
055    /**
056     * HTTP connection manager
057     *
058     * @return connection manager
059     */
060    @Bean
061    public HttpClientConnectionManager connectionManager() {
062        return new PoolingHttpClientConnectionManager();
063    }
064
065    /**
066     * Fedora's lightweight internal event bus. Currently memory-resident.
067     *
068     * @return event bus
069     */
070    @Bean
071    public EventBus eventBus() {
072        // TODO this should be an async event bus: https://jira.lyrasis.org/browse/FCREPO-3587
073        return new EventBus();
074    }
075
076    /**
077     * Configuration of namespace prefixes
078     *
079     * @param propsConfig config properties
080     * @return rdf namespace registry
081     */
082    @Bean(initMethod = "init", destroyMethod = "shutdown")
083    public RdfNamespaceRegistry rdfNamespaceRegistry(final FedoraPropsConfig propsConfig) {
084        final var registry = new RdfNamespaceRegistry();
085        registry.setConfigPath(propsConfig.getNamespaceRegistry());
086        registry.setMonitorForChanges(true);
087        return registry;
088    }
089
090    /**
091     * External content configuration
092     *
093     * @param propsConfig config properties
094     * @return external content path validator
095     */
096    @Bean(initMethod = "init", destroyMethod = "shutdown")
097    public ExternalContentPathValidator externalContentPathValidator(final FedoraPropsConfig propsConfig) {
098        final var validator = new ExternalContentPathValidator();
099        validator.setConfigPath(propsConfig.getExternalContentAllowed());
100        validator.setMonitorForChanges(true);
101        return validator;
102    }
103
104    @Bean
105    public ExternalContentHandlerFactory externalContentHandlerFactory(final ExternalContentPathValidator validator) {
106        final var factory = new ExternalContentHandlerFactory();
107        factory.setValidator(validator);
108        return factory;
109    }
110
111}
