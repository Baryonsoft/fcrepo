018package org.fcrepo.http.commons.exceptionhandlers;
019
020import static javax.ws.rs.core.Response.status;
021import static javax.ws.rs.core.Response.Status.BAD_REQUEST;
022import static org.fcrepo.http.commons.domain.RDFMediaType.TEXT_PLAIN_WITH_CHARSET;
023import static org.slf4j.LoggerFactory.getLogger;
024
025import java.util.regex.Matcher;
026import java.util.regex.Pattern;
027
028import javax.ws.rs.core.Response;
029import javax.ws.rs.ext.ExceptionMapper;
030import javax.ws.rs.ext.Provider;
031
032import org.slf4j.Logger;
033
034import org.apache.jena.query.QueryParseException;
035
036
037/**
038 * Handles Sparql query parsing exceptions thrown when querying or updating.
039 *
040 * @author whikloj
041 * @since September 9, 2014
042 */
043@Provider
044public class QueryParseExceptionMapper implements
045        ExceptionMapper<QueryParseException>, ExceptionDebugLogging {
046
047    private static final Logger LOGGER = getLogger(QueryParseExceptionMapper.class);
048
049    @Override
050    public Response toResponse(final QueryParseException e) {
051
052        LOGGER.error("Captured a query parse exception {}", e.getMessage());
053        debugException(this, e, LOGGER);
054        if (e.getMessage().matches(".* Unresolved prefixed name: .*")) {
055            final Pattern namespacePattern =
056                Pattern.compile("Unresolved prefixed name: (\\w+:\\w+)");
057            final Matcher namespaceMatch =
058                namespacePattern.matcher(e.getMessage());
059            if (namespaceMatch.find()) {
060                final String msg =
061                    String.format(
062                        "There are one or more undefined namespace(s) in your request [ %s ], " +
063                        "please define them before retrying",
064                        namespaceMatch.group(1));
065                return status(BAD_REQUEST).entity(msg).type(TEXT_PLAIN_WITH_CHARSET).build();
066            }
067        }
068
069        return status(BAD_REQUEST).entity(e.getMessage()).type(TEXT_PLAIN_WITH_CHARSET).build();
070    }
071
072}
