176    @Override
177    public SearchResult doSearch(final SearchParameters parameters) throws InvalidQueryException {
178        //translate parameters into a SQL query
179        final MapSqlParameterSource parameterSource = new MapSqlParameterSource();
180        final var whereClauses = new ArrayList<String>();
181        final var conditions = parameters.getConditions();
182        for (int i = 0; i < conditions.size(); i++) {
183            addWhereClause(i, parameterSource, whereClauses, conditions.get(i));
184        }
185
186        final var fields = parameters.getFields().stream().map(Condition.Field::toString).collect(Collectors.toList());
187        final boolean containsRDFTypeField = fields.contains(RDF_TYPE.toString());
188        if (containsRDFTypeField) {
189            whereClauses.add("s.id = r.resource_id");
190            whereClauses.add("r.resource_id = r_filter.resource_id");
191        }
192
193        final var sql =
194                new StringBuilder("SELECT %PLACEHOLDER% FROM " + SIMPLE_SEARCH_TABLE + " s");
195
196        if (containsRDFTypeField) {
197            sql.append(rdfTables);
198            var rdfTypeUriParamValue = "*";
199            for (final Condition condition: conditions) {
200                if (condition.getField().equals(RDF_TYPE)) {
201                    rdfTypeUriParamValue = condition.getObject();
202                    break;
203                }
204            }
205            parameterSource.addValue(RDF_TYPE_URI_PARAM, convertToSqlLikeWildcard(rdfTypeUriParamValue));
206        }
207
208        if (!whereClauses.isEmpty()) {
209            sql.append(" WHERE ");
210            for (final var it = whereClauses.iterator(); it.hasNext(); ) {
211                sql.append(it.next());
212                if (it.hasNext()) {
213                    sql.append(" AND ");
214                }
215            }
216        }
217
218        final String countQuery = sql.toString().replace("%PLACEHOLDER%", "count(*) as count");
219
220        sql.append(" ORDER BY " + parameters.getOrderBy() + " " + parameters.getOrder());
221        sql.append(" LIMIT :limit OFFSET :offset");
222
223        parameterSource.addValue("limit", parameters.getMaxResults());
224        parameterSource.addValue("offset", parameters.getOffset());
225
226        final var rowMapper = new RowMapper<Map<String, Object>>() {
227            @Override
228            public Map<String, Object> mapRow(final ResultSet rs, final int rowNum) throws SQLException {
229                final Map<String, Object> map = new HashMap<>();
230                for (final String fieldStr : fields) {
231                    var value = rs.getObject(fieldStr);
232                    if (value instanceof Timestamp) {
233                        //format as iso instant if timestamp
234                        value = ISO_INSTANT.format(Instant.ofEpochMilli(((Timestamp) value).getTime()));
235                    } else if (fieldStr.equals(RDF_TYPE.toString())) {
236                        //convert the comma-separate string to an array for rdf_type
237                        value = value.toString().split(",");
238                    }
239                    map.put(fieldStr, value);
240                }
241                return map;
242            }
243        };
244
245        final String selectQuery = sql.toString().replace("%PLACEHOLDER%", String.join(",", fields));
246        final Integer totalResults = jdbcTemplate.queryForObject(countQuery, parameterSource, Integer.class);
247        final List<Map<String, Object>> items = jdbcTemplate.query(selectQuery, parameterSource, rowMapper);
248        final var pagination = new PaginationInfo(parameters.getMaxResults(), parameters.getOffset(),
249                (totalResults != null ? totalResults : 0));
250        LOGGER.debug("Search query with parameters: {} - {}", sql, parameters);
251        return new SearchResult(items, pagination);
252    }

254    private void addWhereClause(final int paramCount, final MapSqlParameterSource parameterSource,
255                                final List<String> whereClauses,
256                                final Condition condition) throws InvalidQueryException {
257        final var field = condition.getField();
258        final var operation = condition.getOperator();
259        var object = condition.getObject();
260        final var paramName = "param" + paramCount;
261        if ((field.equals(FEDORA_ID) || field.equals(MIME_TYPE)) &&
262                condition.getOperator().equals(Condition.Operator.EQ)) {
263            if (!object.equals("*")) {
264                final String whereClause;
265                if (object.contains("*")) {
266                    object = convertToSqlLikeWildcard(object);
267                    if (object.contains("_")) {
268                        object = object.replaceAll("_", "\\\\_");
269                    }
270                    whereClause = field + " like :" + paramName;
271                } else {
272                    whereClause = field + " = :" + paramName;
273                }
274
275                whereClauses.add("s." +  whereClause);
276                parameterSource.addValue(paramName, object);
277            }
278        } else if (field.equals(Condition.Field.CREATED) || field.equals(Condition.Field.MODIFIED)) {
279            //parse date
280            try {
281                final var instant = InstantParser.parse(object);
282                whereClauses.add("s." + field + " " + operation.getStringValue() + " :" + paramName);
283                parameterSource.addValue(paramName, new Timestamp(instant.toEpochMilli()), Types.TIMESTAMP);
284            } catch (final Exception ex) {
285                throw new InvalidQueryException(ex.getMessage());
286            }
287        } else if (field.equals(CONTENT_SIZE)) {
288            try {
289                whereClauses.add(field + " " + operation.getStringValue() +
290                        " :" + paramName);
291                parameterSource.addValue(paramName, Long.parseLong(object), Types.INTEGER);
292            } catch (final Exception ex) {
293                throw new InvalidQueryException(ex.getMessage());
294            }
295        } else if (field.equals(RDF_TYPE) && condition.getOperator().equals(Condition.Operator.EQ) ) {
296           //allowed but no where clause added here.
297        } else {
298            throw new InvalidQueryException("Condition not supported: \"" + condition + "\"");
299        }
300    }
301
302    private String convertToSqlLikeWildcard(final String value) {
303        return value.replace("*", "%");
304    }

311    @Transactional
312    @Override
313    public void addUpdateIndex(final String txId, final ResourceHeaders resourceHeaders) {
314        final var fedoraId = resourceHeaders.getId();
315        final var fullId = fedoraId.getFullId();
316
317        if (fedoraId.isAcl() || fedoraId.isMemento()) {
318            LOGGER.debug("The search index does not include acls or mementos. Ignoring resource {}", fullId);
319            return;
320        }
321
322        final var selectParams = new MapSqlParameterSource();
323        selectParams.addValue(FEDORA_ID_PARAM, fullId);
324        final var result =
325                jdbcTemplate.queryForList(SELECT_BY_FEDORA_ID,
326                        selectParams);
327        try {
328            final var fedoraResource = resourceFactory.getResource(txId, fedoraId);
329            final var rdfTypes = fedoraResource.getTypes();
330            final var rdfTypeIds = findOrCreateRdfTypesInDb(rdfTypes);
331            final var params = new MapSqlParameterSource();
332            params.addValue(FEDORA_ID_PARAM, fullId);
333            params.addValue(MODIFIED_PARAM, new Timestamp(resourceHeaders.getLastModifiedDate().toEpochMilli()));
334            params.addValue(MIME_TYPE_PARAM, resourceHeaders.getMimeType());
335            params.addValue(CONTENT_SIZE_PARAM, resourceHeaders.getContentSize());
336            final var exists = result.size() > 0;
337            final Long resourcePrimaryKey;
338            if (exists) {
339                resourcePrimaryKey = (Long) result.get(0).get(ID_COLUMN);
340                jdbcTemplate.update(UPDATE_INDEX_SQL, params);
341                //delete rdf_type associations
342                deleteRdfTypeAssociations(resourcePrimaryKey);
343            } else {
344                params.addValue(CREATED_PARAM, new Timestamp(resourceHeaders.getCreatedDate().toEpochMilli()));
345                resourcePrimaryKey = jdbcInsertResource.executeAndReturnKey(params).longValue();
346            }
347            insertRdfTypeAssociations(rdfTypeIds, resourcePrimaryKey);
348        } catch (final Exception e) {
349            throw new RepositoryRuntimeException("Failed add/updated the search index for : " + fullId, e);
350        }
351    }

353    private void insertRdfTypeAssociations(final List<Long> rdfTypeIds, final Long resourceId) {
354        //add rdf type associations
355        final List<MapSqlParameterSource> parameterSourcesList = new ArrayList<>();
356        for (final var rdfTypeId : rdfTypeIds) {
357            final var assocParams = new MapSqlParameterSource();
358            assocParams.addValue(RESOURCE_ID_PARAM, resourceId);
359            assocParams.addValue(RDF_TYPE_ID_PARAM, rdfTypeId);
360            parameterSourcesList.add(assocParams);
361        }
362        final MapSqlParameterSource[] psArray = parameterSourcesList.toArray(new MapSqlParameterSource[0]);
363        jdbcTemplate.batchUpdate(INSERT_RDF_TYPE_ASSOC, psArray);
364    }
365
366    private void deleteRdfTypeAssociations(final Long resourceId) {
367        final var deleteParams = new MapSqlParameterSource();
368        deleteParams.addValue(RESOURCE_ID_PARAM, resourceId);
369        jdbcTemplate.update(DELETE_RDF_TYPE_ASSOCIATIONS,
370                deleteParams);
371    }

373    private List<Long> findOrCreateRdfTypesInDb(final List<URI> rdfTypes) {
374        final List<String> rdfTypes_str = rdfTypes.stream().map(URI::toString).collect(Collectors.toList());
375
376        final List<RdfType> results = jdbcTemplate.query(SELECT_RDF_TYPE_ID,
377                Map.of(RDF_TYPE_URI_PARAM, rdfTypes_str), RDF_TYPE_ROW_MAPPER);
378        // List of existing type ids.
379        final List<Long> rdfTypeIds = new ArrayList<>();
380        // List of existing type uris.
381        final Set<String> rdfTypeUris = new HashSet<>();
382        for (final RdfType type : results) {
383            rdfTypeIds.add(type.getTypeId());
384            rdfTypeUris.add(type.getTypeUri());
385        }
386        // Type uris that don't already have a record. Needs to be a set to avoid inserting the same URI and
387        final var missingUris = rdfTypes_str.stream().filter(t -> !rdfTypeUris.contains(t))
388                .collect(Collectors.toSet());
389
390        if (!missingUris.isEmpty()) {
391            final List<MapSqlParameterSource> parameterSourcesList = new ArrayList<>();
392            for (final var uri : missingUris) {
393                LOGGER.debug("Adding rdf type uri: " + uri);
394                final var ps = new MapSqlParameterSource();
395                ps.addValue(RDF_TYPE_URI_PARAM, uri);
396                parameterSourcesList.add(ps);
397            }
398            // Batch insert all the records.
399            final MapSqlParameterSource[] psArray = parameterSourcesList.toArray(new MapSqlParameterSource[0]);
400            jdbcTemplate.batchUpdate(INSERT_RDF_TYPE.get(this.dbPlatForm), psArray);
401            // Do a single query for the ID to all the URIs we just inserted.
402            final List<RdfType> createdIds = jdbcTemplate.query(SELECT_RDF_TYPE_ID,
403                    Map.of(RDF_TYPE_URI_PARAM, missingUris), RDF_TYPE_ROW_MAPPER);
404            if (createdIds.size() != missingUris.size()) {
405                throw new RepositoryRuntimeException("Did not select all the items we inserted into the table");
406            }
407            rdfTypeIds.addAll(createdIds.stream().map(RdfType::getTypeId).collect(Collectors.toList()));
408        }
409        return rdfTypeIds;
410    }

433    @Override
434    public void removeFromIndex(final FedoraId fedoraId) {
435        try {
436            final var params = new MapSqlParameterSource();
437            params.addValue(FEDORA_ID_PARAM, fedoraId.getFullId());
438            jdbcTemplate.update(DELETE_FROM_INDEX_SQL, params);
439        } catch (final DataAccessException ex) {
440            throw new RepositoryRuntimeException("Failed to delete search index entry for " + fedoraId.getFullId());
441        }

444    @Override
445    public void reset() {
446        try (final var conn = this.dataSource.getConnection();
447             final var statement = conn.createStatement()) {
448            for (final var sql : toggleForeignKeyChecks(false)) {
449                statement.addBatch(sql);
450            }
451            statement.addBatch(truncateTable(SEARCH_RDF_TYPE_TABLE));
452            statement.addBatch(truncateTable(SEARCH_RESOURCE_RDF_TYPE_TABLE));
453            statement.addBatch(truncateTable(SIMPLE_SEARCH_TABLE));
454            for (final var sql : toggleForeignKeyChecks(true)) {
455                statement.addBatch(sql);
456            }
457            statement.executeBatch();
458        } catch (final SQLException e) {
459            throw new RepositoryRuntimeException("Failed to truncate search index tables", e);
460        }
461    }
462
463    private List<String> toggleForeignKeyChecks(final boolean enable) {
464
465        if (isPostgres()) {
466            return EMPTY_LIST;
467        } else {
468            return List.of("SET FOREIGN_KEY_CHECKS = " + (enable ? 1 : 0) + ";");
469        }
470    }
471
472    private boolean isPostgres() {
473        return dbPlatForm.equals(POSTGRESQL);
474    }
475
476    private String truncateTable(final String tableName) {
477        final var addCascade = isPostgres();
478        return "TRUNCATE TABLE " + tableName + (addCascade ? " CASCADE" : "") + ";";
479    }
