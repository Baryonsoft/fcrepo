018package org.fcrepo.kernel.api.exception;
019
020/**
021 * Thrown in circumstances where a client has used an unknown or unsupported hash algorithm
022 * in a request, e.g. with `Digest` or `Want-Digest`.
023 *
024 * @author harring
025 * @since 2017-09-12
026 */
027public class UnsupportedAlgorithmException extends RepositoryRuntimeException {
028
029    private static final long serialVersionUID = 1L;
030
031    /**
032     * Exception with message
033     * @param message the message
034     */
035    public UnsupportedAlgorithmException(final String message) {
036        super(message);
037    }
038
039    /**
040     * Ordinary constructor.
041     *
042     * @param message the message
043     * @param rootCause the root cause
044     */
045    public UnsupportedAlgorithmException(final String message, final Throwable rootCause) {
046        super(message, rootCause);
047    }
048}
