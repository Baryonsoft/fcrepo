095@Produces({TEXT_HTML_WITH_CHARSET})
096public class StreamingBaseHtmlProvider implements MessageBodyWriter<RdfNamespacedStream> {

121    private Transaction transaction() {
122        if (request.getHeader(ATOMIC_ID_HEADER) != null) {
123            return transactionManager.get(request.getHeader(ATOMIC_ID_HEADER));
124        }
125        return null;
126    }

128    private HttpIdentifierConverter identifierConverter() {
129        if (identifierConverter == null) {
130            final UriBuilder uriBuilder =
131                    uriInfo.getBaseUriBuilder().clone().path(FedoraLdp.class);
132            identifierConverter = new HttpIdentifierConverter(uriBuilder);
133        }
134        return identifierConverter;
135    }

150    private Map<String, Template> templatesMap;

155            "/velocity.properties";

159    private static final Logger LOGGER =
160        getLogger(StreamingBaseHtmlProvider.class);

162    @PostConstruct
163    void init() throws IOException {
164        LOGGER.trace("Velocity engine initializing...");
165        final Properties properties = new Properties();
166        final var velocityLog = fedoraPropsConfig.getVelocityLog().toString();
167        autoVersioningEnabled = ocflPropsConfig.isAutoVersioningEnabled();
168        LOGGER.debug("Setting Velocity runtime log: {}", velocityLog);
169        properties.setProperty("runtime.log", velocityLog);
170
171        final URL propertiesUrl =
172                getClass().getResource(velocityPropertiesLocation);
173        LOGGER.debug("Using Velocity configuration from {}", propertiesUrl);
174        try (final InputStream propertiesStream = propertiesUrl.openStream()) {
175            properties.load(propertiesStream);
176        }
177        velocity.init(properties);
178        LOGGER.trace("Velocity engine initialized.");
179
180        LOGGER.trace("Assembling a map of node primary types -> templates...");
181        final ImmutableMap.Builder<String, Template> templatesMapBuilder = builder();
182
183        of("fcr:versions", "fcr:fixity", "default")
184            .forEach(key -> templatesMapBuilder.put(key, velocity.getTemplate(getTemplateLocation(key))));
185
186        templatesMap = templatesMapBuilder
187            .put(REPOSITORY_ROOT.toString(), velocity.getTemplate(getTemplateLocation("root")))
188            .put(NON_RDF_SOURCE.toString(), velocity.getTemplate(getTemplateLocation("binary")))
189            .put(RDF_SOURCE.toString(), velocity.getTemplate(getTemplateLocation("resource"))).build();
190
191        LOGGER.trace("Assembled template map.");
192        LOGGER.trace("HtmlProvider initialization complete.");
193    }

195    @Override
196    public void writeTo(final RdfNamespacedStream nsStream, final Class<?> type,
197                        final Type genericType, final Annotation[] annotations,
198                        final MediaType mediaType,
199                        final MultivaluedMap<String, Object> httpHeaders,
200                        final OutputStream entityStream) throws IOException {
201
202        final Node subject = ViewHelpers.getContentNode(nsStream.stream.topic());
203
204        final Model model = nsStream.stream.collect(toModel());
205        model.setNsPrefixes(nsStream.namespaces);
206
207        final Template nodeTypeTemplate = getTemplate(model, subject, Arrays.asList(annotations));
208
209        final Context context = getContext(model, subject);
210
211        final FedoraId fedoraID = FedoraId.create(identifierConverter().toInternalId(subject.toString()));
212        try {
213            final FedoraResource resource = getResourceFromSubject(fedoraID);
214            context.put("isOriginalResource", (resource != null && resource.isOriginalResource()));
215            context.put("isArchivalGroup", (resource != null && resource.getSystemTypes(false)
216                    .contains(URI.create(ARCHIVAL_GROUP.getURI()))));
217            context.put("isVersion", (resource != null && resource.isMemento()));
218            context.put("isLDPNR", (resource != null && (resource instanceof Binary || !resource
219                    .getDescribedResource().equals(resource))));
220        } catch (final PathNotFoundException e) {
221            final var baseId = FedoraId.create(fedoraID.getBaseId());
222            if (fedoraID.isRepositoryRoot() || baseId.isRepositoryRoot()) {
223                // We have requested the root resource or default ACL.
224                context.put("isOriginalResource", true);
225                context.put("isArchivalGroup", false);
226                context.put("isVersion", false);
227                context.put("isLDPNR", false);
228            } else {
229                throw new RepositoryRuntimeException(e.getMessage(), e);
230            }
231        }
232        context.put("autoVersioningEnabled", autoVersioningEnabled);
233
234        // the contract of MessageBodyWriter<T> is _not_ to close the stream
235        // after writing to it
236        final Writer outWriter = new OutputStreamWriter(entityStream, StandardCharsets.UTF_8);
237        nodeTypeTemplate.merge(context, outWriter);
238        outWriter.flush();
239    }

241    /**
242     * Get a FedoraResource for the subject of the graph, if it exists.
243     *
244     * @param resourceId the FedoraId of the subject
245     * @return FedoraResource if exists or null
246     * @throws PathNotFoundException if the OCFL mapping is not found.
247     */
248    private FedoraResource getResourceFromSubject(final FedoraId resourceId) throws PathNotFoundException {
249
250        final var tx = transaction();
251        if (tx == null || tx.isCommitted()) {
252            return resourceFactory.getResource(resourceId);
253        } else {
254            return resourceFactory.getResource(tx.getId(), resourceId);
255        }
256    }
257
258    private Context getContext(final Model model, final Node subject) {
259        final FieldTool fieldTool = new FieldTool();
260
261        final Context context = new VelocityContext();
262        final String[] baseUrl = uriInfo.getBaseUri().getPath().split("/");
263        if (baseUrl.length > 0) {
264            final String staticBaseUrl = String.join("/", Arrays.copyOf(baseUrl, baseUrl.length - 1));
265            context.put("staticBaseUrl", staticBaseUrl);
266        } else {
267            context.put("staticBaseUrl", "/");
268        }
269        context.put("rdfLexicon", fieldTool.in(RdfLexicon.class));
270        context.put("helpers", VIEW_HELPERS);
271        context.put("esc", escapeTool);
272        context.put("rdf", model.getGraph());
273
274        context.put("model", model);
275        context.put("subjects", model.listSubjects());
276        context.put("nodeany", ANY);
277        context.put("topic", subject);
278        context.put("originalResource", VIEW_HELPERS.getOriginalResource(subject));
279        context.put("uriInfo", uriInfo);
280        return context;
281    }

283    private Template getTemplate(final Model rdf, final Node subject,
284                                 final List<Annotation> annotations) {
285
286        final String tplName = annotations.stream().filter(x -> x instanceof HtmlTemplate)
287            .map(x -> ((HtmlTemplate) x).value()).filter(templatesMap::containsKey).findFirst()
288            .orElseGet(() -> {
289                final List<String> types = multiValueURI(rdf.getResource(subject.getURI()), type);
290                if (types.contains(REPOSITORY_ROOT.toString())) {
291                    return REPOSITORY_ROOT.toString();
292                }
293                return types.stream().filter(templatesMap::containsKey).findFirst().orElse("default");
294            });
295        LOGGER.debug("Using template: {}", tplName);
296        return templatesMap.get(tplName);
297    }

299    @Override
300    public boolean isWriteable(final Class<?> type, final Type genericType,
301                               final Annotation[] annotations, final MediaType mediaType) {
302        LOGGER.debug(
303                "Checking to see if type: {} is serializable to mimeType: {}",
304                type.getName(), mediaType);
305        return isTextHtml(mediaType) && RdfNamespacedStream.class.isAssignableFrom(type);
306    }
307
308    @Override
309    public long getSize(final RdfNamespacedStream t, final Class<?> type,
310                        final Type genericType, final Annotation[] annotations,
311                        final MediaType mediaType) {
312        // we don't know in advance how large the result might be
313        return -1;
314    }
315
316    private static String getTemplateLocation(final String nodeTypeName) {
317        return templatesLocation + "/" +
318            nodeTypeName.replace(':', '-') + templateFilenameExtension;
319    }
320
321    private static boolean isTextHtml(final MediaType mediaType) {
322        return mediaType.getType().equals(TEXT_HTML_TYPE.getType()) &&
323            mediaType.getSubtype().equals(TEXT_HTML_TYPE.getSubtype());
324    }
