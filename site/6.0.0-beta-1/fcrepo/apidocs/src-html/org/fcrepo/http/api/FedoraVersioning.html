113    /**
114     * Create a new version of a resource. If a memento-datetime header is provided, then the new version will be
115     * based off the provided body using that datetime. If one was not provided, then a version is created based off
116     * the current version of the resource.
117     *
118     * @return response
119     * @throws InvalidChecksumException thrown if one of the provided digests does not match the content
120     * @throws MementoDatetimeFormatException if the header value of memento-datetime is not RFC-1123 format
121     */
122    @POST
123    public Response addVersion() {
124
125        if (headers.getHeaderString("Slug") != null) {
126            throw new BadRequestException("Slug header is no longer supported for versioning label.");
127        }
128
129        if (headers.getHeaderString(MEMENTO_DATETIME_HEADER) != null) {
130            throw new CannotCreateMementoException(MEMENTO_DATETIME_HEADER +
131                    " header is no longer supported on versioning.");
132        }
133
134        final var transaction = transaction();
135
136        if (!transaction.isShortLived()) {
137            throw new BadRequestException("Version creation is not allowed within transactions.");
138        }
139
140        final var resource = resource();
141
142        try {
143            LOGGER.debug("Request to create version for <{}>", externalPath);
144
145            versionService.createVersion(transaction, resource.getFedoraId(), getUserPrincipal());
146
147            // need to commit the transaction before loading the memento otherwise it won't exist
148            transaction.commitIfShortLived();
149
150            final var versions = reloadResource().getTimeMap().getChildren().collect(Collectors.toList());
151
152            if (versions.isEmpty()) {
153                throw new RepositoryRuntimeException(String.format("Failed to create a version for %s", externalPath));
154            }
155
156            final var memento = versions.get(versions.size() - 1);
157
158            return createUpdateResponse(memento, true);
159        } catch (final Exception e) {
160            checkForInsufficientStorageException(e, e);
161            return null; // not reachable
162        } finally {
163            transaction.releaseResourceLocksIfShortLived();
164        }
165    }

167    /**
168     * Get the list of versions for the object
169     *
170     * @param acceptValue the rdf media-type
171     * @return List of versions for the object as RDF
172     * @throws IOException in case of error extracting content
173     */
174    @GET
175    @HtmlTemplate(value = "fcr:versions")
176    @Produces({ TURTLE_WITH_CHARSET + ";qs=1.0", JSON_LD + ";qs=0.8",
177        N3_WITH_CHARSET, N3_ALT2_WITH_CHARSET, RDF_XML, NTRIPLES, TEXT_PLAIN_WITH_CHARSET,
178        TURTLE_X, TEXT_HTML_WITH_CHARSET, APPLICATION_LINK_FORMAT })
179    public Response getVersionList(@HeaderParam("Accept") final String acceptValue) throws IOException {
180
181        final FedoraResource theTimeMap = resource().getTimeMap();
182        checkCacheControlHeaders(request, servletResponse, theTimeMap, transaction());
183
184        LOGGER.debug("GET resource '{}'", externalPath());
185
186        addResourceHttpHeaders(theTimeMap);
187
188        if (acceptValue != null && acceptValue.equalsIgnoreCase(APPLICATION_LINK_FORMAT)) {
189            final String extUrl = identifierConverter().toDomain(externalPath());
190
191            final URI parentUri = URI.create(extUrl);
192            final List<Link> versionLinks = new ArrayList<>();
193            versionLinks.add(Link.fromUri(parentUri).rel("original").build());
194            versionLinks.add(Link.fromUri(parentUri).rel("timegate").build());
195            // So we don't collect the children twice, store them in an array.
196            final FedoraResource[] children = theTimeMap.getChildren().toArray(FedoraResource[]::new);
197
198            Arrays.stream(children).forEach(t -> {
199                final URI childUri = getUri(t);
200                versionLinks.add(Link.fromUri(childUri).rel("memento")
201                                     .param("datetime",
202                        MEMENTO_RFC_1123_FORMATTER.format(t.getMementoDatetime()))
203                                     .build());
204            });
205            // Based on the dates of the above mementos, add the range to the below link.
206            final Instant[] mementos = Arrays.stream(children).map(FedoraResource::getMementoDatetime)
207                .sorted(Comparator.naturalOrder())
208                .toArray(Instant[]::new);
209            final Builder linkBuilder =
210                Link.fromUri(parentUri + "/" + FCR_VERSIONS).rel("self").type(APPLICATION_LINK_FORMAT);
211            if (mementos.length >= 2) {
212                // There are 2 or more Mementos so make a range.
213                linkBuilder.param("from", MEMENTO_RFC_1123_FORMATTER.format(mementos[0].atZone(ZoneId.of("UTC"))));
214                linkBuilder.param("until",
215                    MEMENTO_RFC_1123_FORMATTER.format(mementos[mementos.length - 1].atZone(ZoneId.of("UTC"))));
216            }
217            versionLinks.add(linkBuilder.build());
218            return ok(new LinkFormatStream(versionLinks.stream())).build();
219        } else {
220            return getContent(getChildrenLimit(), theTimeMap);
221        }
222    }

237    /**
238     * Can't delete TimeMaps
239     *
240     * @return the response to a delete request.
241     */
242    @DELETE
243    @Produces({TEXT_PLAIN_WITH_CHARSET})
244    public Response delete() {
245        final FedoraResource theTimeMap = resource().getTimeMap();
246        addResourceHttpHeaders(theTimeMap);
247        final String message = "Timemaps are deleted with their associated resource.";
248        return status(METHOD_NOT_ALLOWED).entity(message).type(TEXT_PLAIN_WITH_CHARSET).build();
249    }
