018package org.fcrepo.http.api.services;
019
020import static org.slf4j.LoggerFactory.getLogger;
021
022import java.util.Collection;
023import java.util.stream.Collectors;
024
025import javax.inject.Inject;
026import javax.ws.rs.core.MediaType;
027
028import org.fcrepo.kernel.api.ContainmentIndex;
029import org.fcrepo.kernel.api.models.FedoraResource;
030import org.fcrepo.kernel.api.rdf.LdpTriplePreferences;
031import org.fcrepo.kernel.api.services.MembershipService;
032
033import org.apache.commons.codec.digest.DigestUtils;
034import org.slf4j.Logger;
035import org.springframework.stereotype.Component;
036
037/**
038 * Service for computing etags for request responses
039 *
040 * @author bbpennel
041 */
042@Component
043public class EtagService {
044    private static final Logger LOGGER = getLogger(EtagService.class);
045
046    @Inject
047    private ContainmentIndex containmentIndex;
048
049    @Inject
050    private MembershipService membershipService;
051
052    /**
053     * Produces etag for a request for an RDF resource. It is based on factors related to the
054     * current state of the resource, as well as request options which change the
055     * representation of the resource.
056     *
057     * @param txId transaction id
058     * @param resource resource
059     * @param prefers LDP preference headers for the request
060     * @param acceptableMediaTypes collection of acceptable media types for the response
061     * @return etag for the request
062     */
063    public String getRdfResourceEtag(final String txId, final FedoraResource resource,
064            final LdpTriplePreferences prefers, final Collection<MediaType> acceptableMediaTypes) {
065        // Start etag based on the current state of the fedora resource, using the state token
066        final StringBuilder etag = new StringBuilder(resource.getStateToken());
067
068        // Factor in the requested mimetype(s)
069        final String mimetype = acceptableMediaTypes.stream()
070                .map(MediaType::toString)
071                .sorted()
072                .collect(Collectors.joining(";"));
073        addComponent(etag, mimetype);
074
075        // Factor in preferences which change which triples are included in the response
076        etag.append('|');
077        if (prefers.displayContainment()) {
078            final var lastUpdated = containmentIndex.containmentLastUpdated(txId, resource.getFedoraId());
079            if (lastUpdated != null) {
080                etag.append(lastUpdated);
081            }
082        }
083        etag.append('|');
084        if (prefers.displayMembership()) {
085            final var lastUpdated = membershipService.getLastUpdatedTimestamp(txId, resource.getFedoraId());
086            if (lastUpdated != null) {
087                etag.append(lastUpdated);
088            }
089        }
090        addComponent(etag, prefers.displayEmbed());
091        addComponent(etag, prefers.displayUserRdf());
092        addComponent(etag, prefers.displayReferences());
093        addComponent(etag, prefers.displayServerManaged());
094
095        // Compute a digest of all these components to use as the etag
096        final String etagMd5 = DigestUtils.md5Hex(etag.toString()).toUpperCase();
097        LOGGER.debug("Produced etag {} for {} from {}", etagMd5, resource.getId(), etag);
098
099        return etagMd5;
100    }
101
102    private void addComponent(final StringBuilder etag, final Object component) {
103        etag.append('|').append(component);
104    }
105}
