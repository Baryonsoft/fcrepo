018package org.fcrepo.common.metrics;
019
020import io.micrometer.core.instrument.Timer;
021import org.fcrepo.common.lang.UncheckedCallable;
022
023/**
024 * Helper for recording metrics
025 *
026 * @author pwinckles
027 */
028public final class MetricsHelper {
029
030    private MetricsHelper() {
031        // static class
032    }
033
034    /**
035     * Records a timing metric around the code in the closure.
036     *
037     * @param timer the timer to record to
038     * @param callable the closure to time
039     * @param <T> the return type
040     * @return the result of the closure
041     */
042    public static <T> T time(final Timer timer, final UncheckedCallable<T> callable) {
043        final var stopwatch = Timer.start();
044        try {
045            return callable.call();
046        } finally {
047            stopwatch.stop(timer);
048        }
049    }
050
051}
