018package org.fcrepo.http.api;
019
020import org.fcrepo.http.commons.AbstractResource;
021import org.fcrepo.http.commons.api.rdf.HttpIdentifierConverter;
022import org.fcrepo.http.commons.session.TransactionProvider;
023import org.fcrepo.kernel.api.Transaction;
024import org.fcrepo.kernel.api.TransactionManager;
025import org.fcrepo.kernel.api.exception.PathNotFoundException;
026import org.fcrepo.kernel.api.identifiers.FedoraId;
027import org.fcrepo.kernel.api.models.FedoraResource;
028import org.fcrepo.kernel.api.models.ResourceFactory;
029import org.fcrepo.kernel.api.models.ResourceHelper;
030import org.slf4j.Logger;
031
032import javax.inject.Inject;
033import javax.servlet.http.HttpServletRequest;
034import javax.ws.rs.core.Context;
035import javax.ws.rs.core.SecurityContext;
036import java.security.Principal;
037
038import static org.slf4j.LoggerFactory.getLogger;
039
040/**
041 * @author cabeer
042 * @since 10/5/14
043 */
044abstract public class FedoraBaseResource extends AbstractResource {
045
046    private static final Logger LOGGER = getLogger(FedoraBaseResource.class);
047
048    @Context
049    protected SecurityContext securityContext;
050
051    @Inject
052    protected ResourceFactory resourceFactory;
053
054    @Inject
055    protected ResourceHelper resourceHelper;
056
057    @Context protected HttpServletRequest servletRequest;
058
059    @Inject
060    protected TransactionManager txManager;
061
062    private TransactionProvider txProvider;
063
064    private Transaction transaction;
065
066    protected HttpIdentifierConverter identifierConverter;
067
068    protected HttpIdentifierConverter identifierConverter() {
069        if (identifierConverter == null) {
070            identifierConverter = new HttpIdentifierConverter(
071                    uriInfo.getBaseUriBuilder().clone().path(FedoraLdp.class));
072        }
073
074        return identifierConverter;
075    }
076
077    /**
078     * Gets a fedora resource by id. Uses the provided transaction if it is uncommitted,
079     * or uses a new transaction.
080     *
081     * @param transaction the fedora transaction
082     * @param fedoraId    identifier of the resource
083     * @return the requested FedoraResource
084     */
085    protected FedoraResource getFedoraResource(final Transaction transaction, final FedoraId fedoraId)
086            throws PathNotFoundException {
087        return resourceFactory.getResource(transaction, fedoraId);
088    }
089
090    /**
091     * @param transaction the transaction in which to check
092     * @param fedoraId identifier of the object to check
093     * @param includeDeleted Whether to check for deleted resources too.
094     * @return Returns true if an object with the provided id exists
095     */
096    protected boolean doesResourceExist(final Transaction transaction, final FedoraId fedoraId,
097                                        final boolean includeDeleted) {
098        return resourceHelper.doesResourceExist(transaction, fedoraId, includeDeleted);
099    }
100
101    /**
102     *
103     * @param transaction the transaction in which to check
104     * @param fedoraId identifier of the object to check
105     * @return Returns true if object does not exist but whose ID starts other resources that do exist.
106     */
107    protected boolean isGhostNode(final Transaction transaction, final FedoraId fedoraId) {
108        return resourceHelper.isGhostNode(transaction, fedoraId);
109    }
110
111    protected String getUserPrincipal() {
112        final Principal p = securityContext.getUserPrincipal();
113        return p == null ? null : p.getName();
114    }
115
116    protected Transaction transaction() {
117        if (transaction == null) {
118            txProvider = new TransactionProvider(txManager, servletRequest,
119                    uriInfo.getBaseUri(), fedoraPropsConfig.getJmsBaseUrl());
120            transaction = txProvider.provide();
121        }
122        return transaction;
123    }
124
125}
