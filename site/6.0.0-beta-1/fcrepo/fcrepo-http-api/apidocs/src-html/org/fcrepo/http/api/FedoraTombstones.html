018package org.fcrepo.http.api;
019
020import com.google.common.annotations.VisibleForTesting;
021import io.micrometer.core.annotation.Timed;
022import org.fcrepo.kernel.api.exception.PathNotFoundException;
023import org.fcrepo.kernel.api.exception.PathNotFoundRuntimeException;
024import org.fcrepo.kernel.api.identifiers.FedoraId;
025import org.fcrepo.kernel.api.models.FedoraResource;
026import org.fcrepo.kernel.api.models.Tombstone;
027import org.fcrepo.kernel.api.services.PurgeResourceService;
028import org.slf4j.Logger;
029import org.springframework.context.annotation.Scope;
030
031import javax.inject.Inject;
032import javax.ws.rs.DELETE;
033import javax.ws.rs.GET;
034import javax.ws.rs.OPTIONS;
035import javax.ws.rs.POST;
036import javax.ws.rs.PUT;
037import javax.ws.rs.Path;
038import javax.ws.rs.PathParam;
039import javax.ws.rs.core.Response;
040
041import static javax.ws.rs.core.HttpHeaders.ALLOW;
042import static javax.ws.rs.core.Response.noContent;
043import static org.slf4j.LoggerFactory.getLogger;
044
045/**
046 * CRUD operations on Fedora tombstones
047 *
048 * @author cbeer
049 */
050@Timed
051@Scope("request")
052@Path("/{path: .*}/fcr:tombstone")
053public class FedoraTombstones extends ContentExposingResource {
054
055    private static final Logger LOGGER = getLogger(FedoraTombstones.class);
056
057    @PathParam("path") protected String externalPath;
058
059    @Inject
060    private PurgeResourceService purgeResourceService;
061
062    /**
063     * Default JAX-RS entry point
064     */
065    public FedoraTombstones() {
066        super();
067    }
068
069    /**
070     * Create a new FedoraTombstones instance for a given path
071     * @param externalPath the external path
072     */
073    @VisibleForTesting
074    public FedoraTombstones(final String externalPath) {
075        this.externalPath = externalPath;
076    }
077
078    /**
079     * Delete a tombstone resource (freeing the original resource to be reused)
080     * @return the free resource
081     */
082    @DELETE
083    public Response delete() {
084        final FedoraResource resource = resource();
085        if (!(resource instanceof Tombstone)) {
086            // If the resource is not deleted there is no tombstone.
087            return Response.status(Response.Status.NOT_FOUND).build();
088        }
089        try {
090            final Tombstone tombstone = (Tombstone) resource;
091            LOGGER.info("Delete tombstone: {}", resource.getFedoraId());
092            purgeResourceService.perform(transaction(), tombstone.getDeletedObject(), getUserPrincipal());
093            transaction().commitIfShortLived();
094            return noContent().build();
095        } finally {
096            transaction().releaseResourceLocksIfShortLived();
097        }
098    }
099
100    /*
101     * These methods are disallowed, but need to exist here or the path gets caught by the FedoraLdp path matcher.
102     */
103    @GET
104    public Response get() {
105        return methodNotAllowed();
106    }
107
108    @POST
109    public Response post() {
110        return methodNotAllowed();
111    }
112    @PUT
113    public Response put() {
114        return methodNotAllowed();
115    }
116
117    @OPTIONS
118    public Response options() {
119        return Response.ok().header(ALLOW, "DELETE").build();
120    }
121
122    @Override
123    protected FedoraResource resource() {
124        final FedoraId resourceId = identifierConverter().pathToInternalId(externalPath);
125        try {
126            return getFedoraResource(transaction(), resourceId);
127        } catch (final PathNotFoundException e) {
128            throw new PathNotFoundRuntimeException(e.getMessage(), e);
129        }
130    }
131
132    @Override
133    protected String externalPath() {
134        return null;
135    }
136
137    private Response methodNotAllowed() {
138        return Response.status(Response.Status.METHOD_NOT_ALLOWED).header(ALLOW, "DELETE").build();
139    }
140}
