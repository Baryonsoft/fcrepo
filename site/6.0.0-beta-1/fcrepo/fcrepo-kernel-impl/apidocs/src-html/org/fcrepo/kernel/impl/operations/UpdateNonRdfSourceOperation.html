018package org.fcrepo.kernel.impl.operations;
019
020import static org.fcrepo.kernel.api.operations.ResourceOperationType.UPDATE;
021
022import java.io.InputStream;
023import java.net.URI;
024
025import org.fcrepo.kernel.api.identifiers.FedoraId;
026import org.fcrepo.kernel.api.operations.ResourceOperationType;
027
028/**
029 * Operation for updating a non-rdf source
030 *
031 * @author bbpennel
032 */
033public class UpdateNonRdfSourceOperation extends AbstractNonRdfSourceOperation {
034
035    /**
036     * Constructor for internal binaries.
037     *
038     * @param rescId the internal identifier.
039     * @param content the stream of the content.
040     */
041    protected UpdateNonRdfSourceOperation(final FedoraId rescId, final InputStream content) {
042        super(rescId, content);
043    }
044
045    /**
046     * Constructor for external content.
047     *
048     * @param rescId the internal identifier.
049     * @param externalContentURI the URI of the external content.
050     * @param externalHandling the type of external content handling (REDIRECT, PROXY)
051     */
052    protected UpdateNonRdfSourceOperation(final FedoraId rescId, final URI externalContentURI,
053            final String externalHandling) {
054        super(rescId, externalContentURI, externalHandling);
055    }
056
057    @Override
058    public ResourceOperationType getType() {
059        return UPDATE;
060    }
061}
