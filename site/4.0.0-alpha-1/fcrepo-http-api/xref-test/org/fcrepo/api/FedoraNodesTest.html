154     @Test
155     public void testCreateDatastream() throws RepositoryException, IOException,
156             InvalidChecksumException, URISyntaxException {
157         final String pid = "FedoraDatastreamsTest1";
158         final String dsId = "testDS";
159         final String dsContent = "asdf";
160         final String dsPath = "/" + pid + "/" + dsId;
161         final InputStream dsContentStream = IOUtils.toInputStream(dsContent);
162         final Node mockNode = mock(Node.class);
163         when(mockNode.getSession()).thenReturn(mockSession);
164         when(
165                 mockDatastreams.createDatastreamNode(any(Session.class),
166                         eq(dsPath), anyString(), eq(dsContentStream),
167                         any(URI.class))).thenReturn(mockNode);
168         final Response actual =
169                 testObj.createObject(createPathList(pid, dsId),
170                         FEDORA_DATASTREAM, null, null, getUriInfoImpl(),
171                         dsContentStream);
172         assertEquals(CREATED.getStatusCode(), actual.getStatus());
173         verify(mockDatastreams)
174                 .createDatastreamNode(any(Session.class), eq(dsPath),
175                         anyString(), any(InputStream.class), any(URI.class));
176         verify(mockSession).save();
177     }
