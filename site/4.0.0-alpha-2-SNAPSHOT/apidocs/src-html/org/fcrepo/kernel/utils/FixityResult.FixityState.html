016package org.fcrepo.kernel.utils;
017
018import static java.util.Objects.hash;
019
020import java.net.URI;
021import java.util.EnumSet;
022
023/**
024 * Structure for presenting the results of a fixity check
025 * (and any repair operations that may have occurred)
026 */
027public class FixityResult {
028
029    /**
030     * The possible fixity states (which may be ORed together later)
031     */
032    public static enum FixityState {
033        SUCCESS, REPAIRED, BAD_CHECKSUM, BAD_SIZE
034    }
035
036    /**
037     * This is a little weird here, and is vestigal from when
038     * this was a JAX-B model as well.
039     *
040     * The "state" of the fixity object is one of:
041     * - SUCCESS: the fixity check was declared successful
042     * - BAD_CHECKSUM and/or BAD_SIZE: either the checksum or the size of the
043     *       bitstream didn't match the stored size
044     * - REPAIRED and BAD_*: the checksum or size failed to match, but it
045     *       was automatically recovered from a different copy
046     */
047    public EnumSet<FixityState> status = EnumSet.noneOf(FixityState.class);
048
049
050    /**
051     * the size computed by the fixity check
052     * @todo make this private
053     */
054    public long computedSize;
055
056    /**
057     * the checksum computed by the fixity check
058     * @todo make this private
059     */
060    public URI computedChecksum;
061
062    private final LowLevelCacheEntry entry;
063
064    /**
065     * Initialize an empty fixity result
066     */
067    public FixityResult() {
068        this(null);
069    }
070
071    /**
072     * Prepare a fixity result for a Low-Level cache entry
073     * @param entry
074     */
075    public FixityResult(final LowLevelCacheEntry entry) {
076        this.entry = entry;
077    }
078
079    /**
080     * Prepare a fixity result given the computed checksum and size
081     * @param size
082     * @param checksum
083     */
084    public FixityResult(final long size, final URI checksum) {
085        this(null, size, checksum);
086    }
087
088    /**
089     * Prepare a fixity result with the expected size and checksum
090     * @param entry
091     * @param size
092     * @param checksum
093     */
094    public FixityResult(final LowLevelCacheEntry entry, final long size,
095                        final URI checksum) {
096        this.entry = entry;
097        computedSize = size;
098        computedChecksum = checksum;
099    }
100
101    /**
102     * Get the identifier for the entry's store
103     * @return
104     */
105    public String getStoreIdentifier() {
106        return entry.getExternalIdentifier();
107    }
108
109    @Override
110    public boolean equals(final Object obj) {
111
112        boolean result = false;
113        if (obj instanceof FixityResult) {
114            final FixityResult that = (FixityResult) obj;
115            result =
116                computedSize == that.computedSize &&
117                computedChecksum.equals(that.computedChecksum);
118        }
119
120        return result;
121    }
122
123    @Override
124    public int hashCode() {
125        return hash(computedSize, computedChecksum);
126    }
127
128    @Override
129    public String toString() {
130        return "Fixity: checksum: " + computedChecksum.toString() + " / " +
131            Long.toString(computedSize);
132    }
133
134    /**
135     * Get the underlying Low-Level cache entry
136     * @return
137     */
138    public LowLevelCacheEntry getEntry() {
139        return entry;
140    }
141
142    /**
143     * Check if the fixity result matches the given checksum URI
144     * @param checksum
145     * @return
146     */
147    public boolean matches(final URI checksum) {
148        return computedChecksum.equals(checksum);
149    }
150
151    /**
152     * Check if the fixity result matches the given size
153     * @param size
154     * @return
155     */
156    public boolean matches(final long size) {
157        return computedSize == size;
158    }
159
160    /**
161     * Does the fixity entry match the given size and checksum?
162     * @param size bitstream size in bytes
163     * @param checksum checksum URI in the form urn:DIGEST:RESULT
164     * @return true if both conditions matched
165     */
166    public boolean matches(final long size, final URI checksum) {
167        return matches(size) && matches(checksum);
168    }
169
170    /**
171     * Was the fixity declared a success
172     * @return
173     */
174    public boolean isSuccess() {
175        return status.contains(FixityState.SUCCESS);
176    }
177
178    /**
179     * Mark the fixity result as been automatically repaired
180     */
181    public void setRepaired() {
182        status.add(FixityState.REPAIRED);
183    }
184}
