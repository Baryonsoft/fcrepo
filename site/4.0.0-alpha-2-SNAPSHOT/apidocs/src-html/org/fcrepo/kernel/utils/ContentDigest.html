016package org.fcrepo.kernel.utils;
017
018import static com.google.common.base.Throwables.propagate;
019import static org.apache.commons.codec.binary.Hex.encodeHexString;
020import static org.slf4j.LoggerFactory.getLogger;
021
022import java.net.URI;
023import java.net.URISyntaxException;
024import java.util.Map;
025
026import org.slf4j.Logger;
027
028import com.google.common.collect.ImmutableMap;
029
030/**
031 * Digest helpers to convert digests (checksums) into URI strings
032 * (based loosely on Magnet URIs)
033 * @author Chris Beer
034 * @date Mar 6, 2013
035 */
036public abstract class ContentDigest {
037
038    private static final Logger LOGGER = getLogger(ContentDigest.class);
039
040    public static final Map<String, String> algorithmToScheme = ImmutableMap
041            .of("SHA-1", "urn:sha1", "SHA1", "urn:sha1");
042
043    public static final Map<String, String> schemeToAlgorithm =
044        ImmutableMap.of("urn:sha1", "SHA-1");
045
046    /**
047     * Convert a MessageDigest algorithm and checksum value to a URN
048     * @param algorithm
049     * @param value
050     * @return
051     */
052    public static URI asURI(final String algorithm, final String value) {
053        try {
054            final String scheme = algorithmToScheme.get(algorithm);
055
056            return new URI(scheme, value, null);
057        } catch (final URISyntaxException unlikelyException) {
058            LOGGER.warn("Exception creating checksum URI: {}",
059                               unlikelyException);
060            throw propagate(unlikelyException);
061        }
062    }
063
064    /**
065     * Convert a MessageDigest algorithm and checksum byte-array data to a URN
066     * @param algorithm
067     * @param data
068     * @return
069     */
070    public static URI asURI(final String algorithm, final byte[] data) {
071        return asURI(algorithm, asString(data));
072    }
073
074    /**
075     * Given a digest URI, get the corresponding MessageDigest algorithm
076     * @param digestUri
077     * @return
078     */
079    public static String getAlgorithm(URI digestUri) {
080        return schemeToAlgorithm
081            .get(digestUri.getScheme() + ":" +
082                 digestUri.getSchemeSpecificPart().split(":", 2)[0]);
083    }
084
085    private static String asString(final byte[] data) {
086        return encodeHexString(data);
087    }
088
089}
