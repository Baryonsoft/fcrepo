070/**
071 * Controller for manipulating binary streams in larger batches
072 * by using multipart requests and responses
073 */
074@Component
075@Scope("prototype")
076@Path("/{path: .*}/fcr:datastreams")
077public class FedoraDatastreams extends AbstractResource {
078
079    @InjectedSession
080    protected Session session;
081
082    private final Logger logger = getLogger(FedoraDatastreams.class);
083
084    /**
085     * Update the content of multiple datastreams from a multipart POST. The
086     * datastream to update is given by the name of the content disposition.
087     * 
088     * @param pathList
089     * @param dsidList
090     * @param multipart
091     * @return
092     * @throws RepositoryException
093     * @throws IOException
094     * @throws InvalidChecksumException
095     */
096    @POST
097    @Timed
098    public Response modifyDatastreams(@PathParam("path")
099            final List<PathSegment> pathList,
100            @QueryParam("delete")
101            final List<String> dsidList,
102            final MultiPart multipart)
103        throws RepositoryException, IOException, InvalidChecksumException,
104        URISyntaxException {
105
106        final String path = toPath(pathList);
107        try {
108            for (final String dsid : dsidList) {
109                logger.debug("Purging datastream: " + dsid);
110                nodeService.deleteObject(session, path + "/" + dsid);
111            }
112
113            for (final BodyPart part : multipart.getBodyParts()) {
114                final String dsid =
115                        part.getContentDisposition().getParameters()
116                                .get("name");
117                logger.debug("Adding datastream: " + dsid);
118                final String dsPath = path + "/" + dsid;
119                final Object obj = part.getEntity();
120                InputStream src = null;
121                if (obj instanceof BodyPartEntity) {
122                    final BodyPartEntity entity =
123                            (BodyPartEntity) part.getEntity();
124                    src = entity.getInputStream();
125                } else if (obj instanceof InputStream) {
126                    src = (InputStream) obj;
127                }
128                datastreamService.createDatastreamNode(session, dsPath, part
129                        .getMediaType().toString(), src);
130            }
131
132            session.save();
133
134            final HttpGraphSubjects subjects =
135                    new HttpGraphSubjects(session, FedoraNodes.class, uriInfo);
136
137            return created(
138                    new URI(subjects.getGraphSubject(session.getNode(path))
139                            .getURI())).build();
140
141        } finally {
142            session.logout();
143        }
144    }
145
146    /**
147     * Delete multiple datastreams given by the dsid query parameter
148     * 
149     * @param pathList
150     * @param dsidList
151     * @return
152     * @throws RepositoryException
153     */
154    @DELETE
155    @Timed
156    public Response deleteDatastreams(@PathParam("path")
157            final List<PathSegment> pathList,
158            @QueryParam("dsid")
159            final List<String> dsidList) throws RepositoryException {
160        try {
161            final String path = toPath(pathList);
162            for (final String dsid : dsidList) {
163                logger.debug("purging datastream {}", path + "/" + dsid);
164                nodeService.deleteObject(session, path + "/" + dsid);
165            }
166            session.save();
167            return noContent().build();
168        } finally {
169            session.logout();
170        }
171    }
172
173    /**
174     * Retrieve multiple datastream bitstreams in a single request as a
175     * multipart/mixed response.
176     * 
177     * @param pathList
178     * @param requestedDsids
179     * @param request
180     * @return
181     * @throws RepositoryException
182     * @throws IOException
183     * @throws NoSuchAlgorithmException
184     */
185    @GET
186    @Produces("multipart/mixed")
187    @Timed
188    public Response getDatastreamsContents(@PathParam("path")
189            final List<PathSegment> pathList,
190            @QueryParam("dsid")
191            final List<String> requestedDsids,
192            @Context
193            final Request request) throws RepositoryException, IOException,
194        NoSuchAlgorithmException {
195
196        final ArrayList<Datastream> datastreams = new ArrayList<Datastream>();
197
198        try {
199            final String path = toPath(pathList);
200            // TODO: wrap some of this JCR logic in an fcrepo abstraction;
201
202            final Node node = nodeService.getObject(session, path).getNode();
203
204            Date date = new Date();
205
206            final MessageDigest digest = MessageDigest.getInstance("SHA-1");
207
208            final NodeIterator ni;
209
210            if (requestedDsids.isEmpty()) {
211                ni = node.getNodes();
212            } else {
213                ni =
214                        node.getNodes(requestedDsids
215                                .toArray(new String[requestedDsids.size()]));
216            }
217
218            // transform the nodes into datastreams, and calculate cache header
219            // data
220            while (ni.hasNext()) {
221
222                final Node dsNode = ni.nextNode();
223                final Datastream ds = datastreamService.asDatastream(dsNode);
224
225                if (!ds.hasContent()) {
226                    continue;
227                }
228
229                digest.update(ds.getContentDigest().toString().getBytes(
230                        StandardCharsets.UTF_8));
231
232                if (ds.getLastModifiedDate().after(date)) {
233                    date = ds.getLastModifiedDate();
234                }
235
236                datastreams.add(ds);
237            }
238
239            final URI digestURI =
240                    ContentDigest.asURI(digest.getAlgorithm(), digest.digest());
241            final EntityTag etag = new EntityTag(digestURI.toString());
242
243            final Date roundedDate = new Date();
244            roundedDate.setTime(date.getTime() - date.getTime() % 1000);
245
246            Response.ResponseBuilder builder =
247                    request.evaluatePreconditions(roundedDate, etag);
248
249            final CacheControl cc = new CacheControl();
250            cc.setMaxAge(0);
251            cc.setMustRevalidate(true);
252
253            if (builder == null) {
254                final MultiPart multipart = new MultiPart();
255
256                for (final Datastream ds : datastreams) {
257                    final BodyPart bodyPart =
258                            new BodyPart(ds.getContent(), MediaType.valueOf(ds
259                                    .getMimeType()));
260                    bodyPart.setContentDisposition(ContentDisposition.type(
261                            "attachment").fileName(ds.getPath()).creationDate(
262                            ds.getCreatedDate()).modificationDate(
263                            ds.getLastModifiedDate()).size(ds.getContentSize())
264                            .build());
265                    multipart.bodyPart(bodyPart);
266                }
267
268                builder = Response.ok(multipart, MULTIPART_FORM_DATA);
269            }
270
271            return builder.cacheControl(cc).lastModified(date).tag(etag)
272                    .build();
273
274        } finally {
275            session.logout();
276        }
277    }
