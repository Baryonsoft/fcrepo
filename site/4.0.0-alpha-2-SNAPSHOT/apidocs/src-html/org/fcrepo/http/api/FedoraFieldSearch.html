070/**
071 * Basic administrative search across the repository
072 *
073 * @author Frank Asseg
074 * @author ajs6f
075 */
076
077@Component
078@Scope("prototype")
079@Path("/fcr:search")
080public class FedoraFieldSearch extends AbstractResource implements
081        FedoraJcrTypes {
082
083    @InjectedSession
084    protected Session session;
085
086    private static final Logger LOGGER = getLogger(FedoraFieldSearch.class);
087
088    /**
089     * A stub method so we can return a text/html representation using
090     * the right template.
091     *
092     * {@link #searchSubmitHtml(String, long, int, javax.ws.rs.core.Request,
093     * javax.ws.rs.core.UriInfo)}
094     *
095     * @param terms
096     * @param offset
097     * @param limit
098     * @param request
099     * @param uriInfo
100     * @return
101     * @throws RepositoryException
102     */
103    @GET
104    @Timed
105    @HtmlTemplate("search:results")
106    @Produces({TEXT_HTML})
107    public Dataset searchSubmitHtml(@QueryParam("q")
108            final String terms,
109            @QueryParam("offset") @DefaultValue("0")
110            final long offset,
111            @QueryParam("limit")
112            @DefaultValue("25")
113            final int limit,
114            @Context
115            final Request request,
116            @Context HttpServletResponse servletResponse,
117            @Context
118            final UriInfo uriInfo) throws RepositoryException {
119        return getSearchDataset(terms, offset, limit, servletResponse, uriInfo);
120    }
121
122    /**
123     * Execute a basic full-text search across the repository
124     *
125     * GET /fcr:search?q=term
126     *
127     * @param terms
128     * @param offset
129     * @param limit
130     * @param request
131     * @param uriInfo
132     * @return
133     * @throws RepositoryException
134     */
135    @GET
136    @Timed
137    @Produces({TURTLE, N3, N3_ALT1, N3_ALT2, RDF_XML, RDF_JSON, NTRIPLES})
138    public Dataset searchSubmitRdf(@QueryParam("q") final String terms,
139            @QueryParam("offset") @DefaultValue("0") final long offset,
140            @QueryParam("limit") @DefaultValue("25") final int limit,
141            @Context final Request request,
142            @Context HttpServletResponse servletResponse,
143            @Context final UriInfo uriInfo) throws RepositoryException {
144
145        if (terms == null) {
146            LOGGER.trace("Received search request, but terms was empty. Aborting.");
147            throw new WebApplicationException(status(BAD_REQUEST).entity(
148                    "q parameter is mandatory").build());
149        }
150
151        return getSearchDataset(terms, offset, limit, servletResponse, uriInfo);
152    }
153
154    private Dataset getSearchDataset(final String terms,
155                                     final long offset,
156                                     final int limit,
157                                     final HttpServletResponse servletResponse,
158                                     final UriInfo uriInfo)
159        throws RepositoryException {
160
161        try {
162            LOGGER.debug(
163                    "Received search request with search terms {}, offset {}, and limit {}",
164                    terms, offset, limit);
165
166            final Resource searchResult;
167
168            if (terms == null) {
169                searchResult = createResource(uriInfo.getBaseUriBuilder()
170                                                  .path(FedoraFieldSearch.class)
171                                                  .build().toString());
172            } else {
173                searchResult = createResource(uriInfo.getBaseUriBuilder()
174                                                  .path(FedoraFieldSearch.class)
175                                                  .queryParam("q", terms)
176                                                  .build().toString());
177            }
178
179            final HttpGraphSubjects subjects = new HttpGraphSubjects(session, FedoraNodes.class, uriInfo);
180
181            final Dataset dataset =
182                    nodeService.searchRepository(subjects, searchResult,
183                            session, terms, limit, offset);
184
185            final Model searchModel = createDefaultModel();
186            if (terms != null) {
187                final Resource pageResource = createResource(uriInfo.getRequestUri().toASCIIString());
188                searchModel.add(pageResource, RDF.type, SEARCH_PAGE);
189                searchModel.add(pageResource, RDF.type, PAGE);
190                searchModel.add(pageResource, PAGE_OF, searchResult);
191
192
193                searchModel.add(pageResource,
194                                   SEARCH_ITEMS_PER_PAGE,
195                                   searchModel.createTypedLiteral(limit));
196                searchModel.add(pageResource,
197                                   SEARCH_OFFSET,
198                                   searchModel.createTypedLiteral(offset));
199                searchModel.add(pageResource, SEARCH_TERMS, terms);
200
201                if (dataset.getDefaultModel()
202                        .contains(searchResult,
203                                     SEARCH_HAS_MORE,
204                                     searchModel.createTypedLiteral(true))) {
205
206                    final Resource nextPageResource =
207                        searchModel.createResource(uriInfo
208                                                       .getBaseUriBuilder()
209                                                       .path(FedoraFieldSearch.class)
210                                                       .queryParam("q", terms)
211                                                       .queryParam("offset", offset + limit)
212                                                       .queryParam("limit", limit)
213                                                       .build()
214                                                       .toString());
215                    searchModel.add(pageResource, NEXT_PAGE, nextPageResource);
216                } else {
217                    searchModel.add(pageResource, NEXT_PAGE, RDF.nil);
218                }
219
220                final String firstPage = uriInfo
221                                       .getBaseUriBuilder()
222                                       .path(FedoraFieldSearch.class)
223                                       .queryParam("q", terms)
224                                       .queryParam("offset", 0)
225                                       .queryParam("limit", limit)
226                                       .build()
227                                       .toString();
228                final Resource firstPageResource =
229                    searchModel.createResource(firstPage);
230                searchModel.add(subjects.getContext(), FIRST_PAGE, firstPageResource);
231
232                servletResponse.addHeader("Link", firstPage + ";rel=\"first\"");
233
234                dataset.addNamedModel("search-pagination", searchModel);
235            }
236
237            return dataset;
238
239        } finally {
240            session.logout();
241        }
242    }
