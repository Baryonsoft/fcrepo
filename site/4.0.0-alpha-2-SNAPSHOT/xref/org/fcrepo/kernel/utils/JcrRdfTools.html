209     /**
210      * Create a default Jena Model populated with the registered JCR namespaces
211      *
212      * @return
213      * @throws RepositoryException
214      */
215     public Model getJcrPropertiesModel()
216         throws RepositoryException {
217         final Model model = ModelFactory.createDefaultModel();
218 
219         final javax.jcr.NamespaceRegistry namespaceRegistry =
220                 NamespaceTools.getNamespaceRegistry(session);
221         assert (namespaceRegistry != null);
222 
223         for (final String prefix : namespaceRegistry.getPrefixes()) {
224             final String nsURI = namespaceRegistry.getURI(prefix);
225             if (nsURI != null && !nsURI.equals("") && !prefix.equals("xmlns")) {
226 
227                 if (prefix.equals("jcr")) {
228                     model.setNsPrefix("fcrepo",
229                             getRDFNamespaceForJcrNamespace(nsURI));
230                 } else {
231                     model.setNsPrefix(prefix,
232                             getRDFNamespaceForJcrNamespace(nsURI));
233                 }
234             }
235         }
236 
237         return model;
238     }

240     /**
241      * Get an RDF model for the given JCR NodeIterator
242      *
243      * @param nodeIterator
244      * @param iteratorSubject
245      * @return
246      * @throws RepositoryException
247      */
248     public Model getJcrPropertiesModel(final Iterator<Node> nodeIterator,
249                                        final Resource iteratorSubject)
250         throws RepositoryException {
251 
252         if (!nodeIterator.hasNext()) {
253             return ModelFactory.createDefaultModel();
254         }
255 
256         final PeekingIterator<Node> iterator =
257                 Iterators.peekingIterator(nodeIterator);
258         final Model model =
259                 getJcrPropertiesModel();
260 
261         while (iterator.hasNext()) {
262             final Node node = iterator.next();
263             addJcrPropertiesToModel(node, model);
264             if (iteratorSubject != null) {
265                 model.add(iteratorSubject, RdfLexicon.HAS_MEMBER_OF_RESULT,
266                              graphSubjects.getGraphSubject(node));
267             }
268         }
269 
270         return model;
271     }

298     /**
299      * Get a Jena RDF model for the JCR version history information for a node
300      *
301      *
302      * @param versionHistory
303      * @return
304      * @throws RepositoryException
305      */
306     public Model getJcrPropertiesModel(final VersionHistory versionHistory, final Resource subject) throws RepositoryException {
307 
308         final Model model = getJcrPropertiesModel();
309 
310         final VersionIterator versionIterator = versionHistory.getAllVersions();
311 
312         while (versionIterator.hasNext()) {
313             final Version version = versionIterator.nextVersion();
314             final Node frozenNode = version.getFrozenNode();
315             final Resource versionSubject =
316                 graphSubjects.getGraphSubject(frozenNode);
317 
318             model.add(subject, HAS_VERSION, versionSubject);
319 
320             final String[] versionLabels =
321                 versionHistory.getVersionLabels(version);
322             for (final String label : versionLabels) {
323                 model.add(versionSubject, HAS_VERSION_LABEL, label);
324             }
325             final javax.jcr.PropertyIterator properties =
326                 frozenNode.getProperties();
327 
328             while (properties.hasNext()) {
329                 final Property property = properties.nextProperty();
330 
331                 addPropertyToModel(versionSubject, model, property);
332             }
333 
334         }
335 
336         return model;
337     }

339     /**
340      * Serialize the JCR fixity information in a Jena Model
341      *
342      * @param node
343      * @param blobs
344      * @return
345      * @throws RepositoryException
346      */
347     public Model getJcrPropertiesModel(final Node node,
348                                        final Iterable<FixityResult> blobs)
349         throws RepositoryException {
350 
351         final Model model = getJcrPropertiesModel();
352 
353         addJcrPropertiesToModel(node, model);
354 
355         for (final FixityResult result : blobs) {
356             // fixity results are just blank nodes
357             final Resource resultSubject = createResource();
358 
359             model.add(resultSubject, IS_FIXITY_RESULT_OF, graphSubjects
360                                                               .getGraphSubject(node));
361             model.add(graphSubjects.getGraphSubject(node), HAS_FIXITY_RESULT,
362                          resultSubject);
363 
364             model.add(resultSubject, HAS_LOCATION, createResource(result
365                                                                       .getStoreIdentifier()));
366 
367             for (final FixityResult.FixityState state : result.status) {
368                 model.add(resultSubject, HAS_FIXITY_STATE,
369                              createTypedLiteral(state.toString()));
370             }
371 
372             final String checksum = result.computedChecksum.toString();
373             model.add(resultSubject, HAS_COMPUTED_CHECKSUM,
374                          createResource(checksum));
375             model.add(resultSubject, HAS_COMPUTED_SIZE,
376                          createTypedLiteral(result.computedSize));
377         }
378         return model;
379     }

381     /**
382      * Get an RDF model of the registered JCR namespaces
383      *
384      * @return
385      * @throws RepositoryException
386      */
387     public Model getJcrNamespaceModel()
388         throws RepositoryException {
389         final Model model = getJcrPropertiesModel();
390 
391         final Map<String, String> prefixMap = model.getNsPrefixMap();
392 
393         for (final Map.Entry<String, String> entry : prefixMap.entrySet()) {
394             if (entry.getKey().isEmpty()) {
395                 continue;
396             }
397             final Resource nsSubject =
398                 ResourceFactory.createResource(entry.getValue());
399 
400             model.add(nsSubject, RDF.type, RdfLexicon.VOAF_VOCABULARY);
401             model.add(nsSubject,
402                          RdfLexicon.HAS_NAMESPACE_PREFIX, ResourceFactory
403                                                               .createPlainLiteral(entry.getKey()));
404 
405             model.add(nsSubject,
406                          RdfLexicon.HAS_NAMESPACE_URI, ResourceFactory
407                                                            .createPlainLiteral(entry.getValue()));
408         }
409 
410         return model;
411     }

413     /**
414      * Add the properties of a Node's parent and immediate children (as well as
415      * the jcr:content of children) to the given RDF model
416      *
417      * @param node
418      * @param offset
419      * @param limit @throws RepositoryException
420      */
421     public Model getJcrTreeModel(final Node node, final long offset, final int limit)
422         throws RepositoryException {
423 
424         final Model model = getJcrPropertiesModel();
425 
426         if (limit < -1) {
427             return model;
428         }
429 
430         final Resource subject = graphSubjects.getGraphSubject(node);
431         final Resource pageContext = graphSubjects.getContext();
432 
433         model.add(pageContext, RDF.type, model.createResource("http://www.w3.org/ns/ldp#Page"));
434         model.add(pageContext, PAGE_OF, subject);
435 
436         if (isContainer(node)) {
437             model.add(pageContext, model.createProperty("http://www.w3.org/ns/ldp#membersInlined"), model.createTypedLiteral(true));
438 
439             model.add(subject, RDF.type, model.createResource("http://www.w3.org/ns/ldp#Container"));
440             model.add(subject, model.createProperty("http://www.w3.org/ns/ldp#membershipSubject"), subject);
441             model.add(subject, model.createProperty("http://www.w3.org/ns/ldp#membershipPredicate"), RdfLexicon.HAS_CHILD);
442             model.add(subject, model.createProperty("http://www.w3.org/ns/ldp#membershipObject"), model.createResource("http://www.w3.org/ns/ldp#MemberSubject"));
443         }
444 
445         // don't do this if the node is the root node.
446         if (node.getDepth() > 0) {
447 
448             final Node parentNode = node.getParent();
449             final Resource parentNodeSubject =
450                 graphSubjects.getGraphSubject(parentNode);
451             model.add(subject, RdfLexicon.HAS_PARENT, parentNodeSubject);
452             model.add(parentNodeSubject, RdfLexicon.HAS_CHILD, subject);
453             addJcrPropertiesToModel(parentNode, model);
454             model.add(pageContext,
455                          model.createProperty("http://www.w3.org/ns/ldp#inlinedResource"),
456                          parentNodeSubject);
457         }
458 
459         if (node.hasNodes()) {
460 
461             if (limit == -1) {
462                 model.add(pageContext, PAGE, RDF.nil);
463             }
464 
465             final javax.jcr.NodeIterator nodeIterator = node.getNodes();
466 
467             int i = 0;
468             long excludedNodeCount = 0;
469 
470             while (nodeIterator.hasNext()) {
471                 final Node childNode = nodeIterator.nextNode();
472 
473                 // exclude jcr system nodes or jcr:content nodes
474                 if (FedoraTypesUtils.isInternalNode.apply(childNode) ||
475                         childNode.getName().equals(JcrConstants.JCR_CONTENT)) {
476                     excludedNodeCount++;
477                 } else {
478                     final Resource childNodeSubject =
479                             graphSubjects.getGraphSubject(childNode);
480 
481                     if (i >= offset && (limit == -1 || i < (offset + limit))) {
482                         addJcrPropertiesToModel(childNode, model);
483                         model.add(pageContext, model.createProperty("http://www.w3.org/ns/ldp#inlinedResource"), childNodeSubject);
484                         model.add(childNodeSubject, RdfLexicon.HAS_PARENT, subject);
485                     }
486 
487                     i++;
488 
489                     model.add(subject, RdfLexicon.HAS_CHILD, childNodeSubject);
490                 }
491 
492             }
493 
494             model.add(subject, RdfLexicon.HAS_CHILD_COUNT, ResourceFactory
495                     .createTypedLiteral(nodeIterator.getSize() - excludedNodeCount));
496         }
497 
498         return model;
499     }

501     private boolean isContainer(final Node node) throws RepositoryException {
502         return HAS_CHILD_NODE_DEFINITIONS.apply(node.getPrimaryNodeType())
503                    || Iterables.any(ImmutableList.copyOf(node.getMixinNodeTypes()),
504                                        HAS_CHILD_NODE_DEFINITIONS);
505     }
506 
507     Predicate<NodeType> HAS_CHILD_NODE_DEFINITIONS = new Predicate<NodeType>() {
508         @Override
509         public boolean apply(final NodeType input) {
510             return input.getChildNodeDefinitions().length > 0;
511         }
512     };

514     /**
515      * Determine if a predicate is an internal property of a node (and
516      * should not be modified from external sources)
517      * @param subjectNode
518      * @param predicate
519      * @return
520      */
521     public boolean isInternalProperty(final Node subjectNode, final Resource predicate) {
522         switch (predicate.getNameSpace()) {
523             case RdfLexicon.REPOSITORY_NAMESPACE:
524             case "http://www.jcp.org/jcr/1.0":
525             case "http://www.w3.org/ns/ldp#":
526                 return true;
527             default:
528                 return false;
529         }
530     }

532     /**
533      * Add all of a node's properties to the given model
534      *
535      * @param node
536      * @param model
537      * @throws RepositoryException
538      */
539     private void addJcrPropertiesToModel(final Node node, final Model model)
540         throws RepositoryException {
541 
542         final Resource subject = graphSubjects.getGraphSubject(node);
543         final javax.jcr.PropertyIterator properties = node.getProperties();
544 
545         while (properties.hasNext()) {
546             final Property property = properties.nextProperty();
547 
548             addPropertyToModel(subject, model, property);
549         }
550 
551         // always include the jcr:content node information
552         if (node.hasNode(JcrConstants.JCR_CONTENT)) {
553             final Node contentNode = node.getNode(JcrConstants.JCR_CONTENT);
554             final Resource contentSubject =
555                     graphSubjects.getGraphSubject(contentNode);
556 
557             model.add(subject, RdfLexicon.HAS_CONTENT, contentSubject);
558             model.add(contentSubject, RdfLexicon.IS_CONTENT_OF, subject);
559 
560             addJcrPropertiesToModel(contentNode, model);
561             addJcrContentLocationInformationToModel(node, model);
562         }
563     }

565     /**
566      * Add repository metrics data to the given JCR model
567      *
568      * @param node
569      * @param model
570      * @throws RepositoryException
571      */
572     private void addRepositoryMetricsToModel(
573                                                 final Node node, final Model model)
574         throws RepositoryException {
575 
576         final Repository repository = node.getSession().getRepository();
577         /* retreive the metrics from the service */
578         final SortedMap<String, Counter> counters = getMetrics().getCounters();
579 
580         final Resource subject = graphSubjects.getGraphSubject(node);
581         for (final String key : repository.getDescriptorKeys()) {
582             final String descriptor = repository.getDescriptor(key);
583             if (descriptor != null) {
584                 final String uri =
585                     RdfLexicon.REPOSITORY_NAMESPACE + "repository/" + key;
586                 model.add(subject, model.createProperty(uri), descriptor);
587             }
588         }
589 
590         final NodeTypeManager nodeTypeManager = FedoraTypesUtils.getNodeTypeManager(
591                 node);
592 
593         final NodeTypeIterator nodeTypes = nodeTypeManager.getAllNodeTypes();
594 
595         while (nodeTypes.hasNext()) {
596             final NodeType nodeType = nodeTypes.nextNodeType();
597             model.add(subject, RdfLexicon.HAS_NODE_TYPE, nodeType.getName());
598         }
599         model.add(subject, RdfLexicon.HAS_OBJECT_COUNT, ResourceFactory
600                  .createTypedLiteral(
601                          FedoraTypesUtils
602                                  .getRepositoryCount(
603                                          repository)));
604         model.add(subject, RdfLexicon.HAS_OBJECT_SIZE, ResourceFactory
605                 .createTypedLiteral(
606                         FedoraTypesUtils
607                                 .getRepositorySize(
608                                         repository)));
609 
610         /* TODO Get and add the Storage policy to the RDF response */
611 
612         /* add the configuration information */
613 
614         /* Get the cluster configuration for the RDF response */
615         final Map<String, String> config =
616             getClusterConfiguration.apply(repository);
617 
618         assert (config != null);
619 
620         for (final Map.Entry<String, String> entry : config.entrySet()) {
621             model.add(subject, model
622                     .createProperty(RdfLexicon.REPOSITORY_NAMESPACE +
623                             entry.getKey()), entry.getValue());
624         }
625 
626         /* and add the repository metrics to the RDF model */
627         if (counters
628                 .containsKey("LowLevelStorageService.fixity-check-counter")) {
629             model.add(subject, RdfLexicon.HAS_FIXITY_CHECK_COUNT,
630                          ResourceFactory.createTypedLiteral(counters.get(
631                                                                             "org.fcrepo.services." + "LowLevelStorageService."
632                                                                                 + "fixity-check-counter").getCount()));
633         }
634 
635         if (counters
636                 .containsKey("LowLevelStorageService.fixity-error-counter")) {
637             model.add(subject, RdfLexicon.HAS_FIXITY_ERROR_COUNT,
638                          ResourceFactory.createTypedLiteral(counters.get(
639                                                                             "org.fcrepo.services." + "LowLevelStorageService."
640                                                                                 + "fixity-error-counter").getCount()));
641         }
642 
643         if (counters
644                 .containsKey("LowLevelStorageService.fixity-repaired-counter")) {
645 
646             model.add(subject, RdfLexicon.HAS_FIXITY_REPAIRED_COUNT,
647                          ResourceFactory.createTypedLiteral(counters.get(
648                                                                             "org.fcrepo.services." + "LowLevelStorageService."
649                                                                                 + "fixity-repaired-counter").getCount()));
650         }
651 
652     }

656      *
657      * @param node
658      * @param model
659      * @throws RepositoryException
660      */
661     private void addJcrContentLocationInformationToModel(final Node node,
662         final Model model)
663         throws RepositoryException {
664         final Node contentNode = node.getNode(JcrConstants.JCR_CONTENT);
665         final Resource contentNodeSubject =
666             graphSubjects.getGraphSubject(contentNode);
667 
668         // TODO get this from somewhere else.
669 
670         if (llstore == null) {
671             llstore = new LowLevelStorageService();
672             llstore.setRepository(node.getSession().getRepository());
673         }
674 
675         final Set<LowLevelCacheEntry> cacheEntries =
676             llstore.getLowLevelCacheEntries(contentNode);
677 
678         for (final LowLevelCacheEntry e : cacheEntries) {
679             model.add(contentNodeSubject, RdfLexicon.HAS_LOCATION, e
680                                                                        .getExternalIdentifier());
681         }
682 
683     }

689      *
690      * @param data an RDF Node (possibly with a DataType)
691      * @param type a JCR PropertyType value
692      * @return a JCR Value
693      * @throws javax.jcr.RepositoryException
694      */
695     Value createValue(final Node node, final RDFNode data,
696                       final int type) throws RepositoryException {
697         final ValueFactory valueFactory = FedoraTypesUtils.getValueFactory.apply(node);
698         assert (valueFactory != null);
699 
700         if (data.isURIResource() &&
701                 (type == REFERENCE || type == WEAKREFERENCE)) {
702             // reference to another node (by path)
703             final Node nodeFromGraphSubject = graphSubjects.getNodeFromGraphSubject(data.asResource());
704             return valueFactory.createValue(nodeFromGraphSubject, type == WEAKREFERENCE);
705         } else if (data.isURIResource() || type == URI) {
706             // some random opaque URI
707             return valueFactory.createValue(data.toString(), PropertyType.URI);
708         } else if (data.isResource()) {
709             // a non-URI resource (e.g. a blank node)
710             return valueFactory.createValue(data.toString(), UNDEFINED);
711         } else if (data.isLiteral() && type == UNDEFINED) {
712             // the JCR schema doesn't know what this should be; so introspect
713             // the RDF and try to figure it out
714             final Literal literal = data.asLiteral();
715             final RDFDatatype dataType = literal.getDatatype();
716             final Object rdfValue = literal.getValue();
717 
718             if (rdfValue instanceof Boolean) {
719                 return valueFactory.createValue((Boolean) rdfValue);
720             } else if (rdfValue instanceof Byte ||
721                            (dataType != null && dataType.getJavaClass() == Byte.class)) {
722                 return valueFactory.createValue(literal.getByte());
723             } else if (rdfValue instanceof Double) {
724                 return valueFactory.createValue((Double) rdfValue);
725             } else if (rdfValue instanceof Float) {
726                 return valueFactory.createValue((Float) rdfValue);
727             } else if (rdfValue instanceof Long ||
728                            (dataType != null && dataType.getJavaClass() == Long.class)) {
729                 return valueFactory.createValue(literal.getLong());
730             } else if (rdfValue instanceof Short ||
731                            (dataType != null && dataType.getJavaClass() == Short.class)) {
732                 return valueFactory.createValue(literal.getShort());
733             } else if (rdfValue instanceof Integer) {
734                 return valueFactory.createValue((Integer) rdfValue);
735             } else if (rdfValue instanceof XSDDateTime) {
736                 return valueFactory.createValue(((XSDDateTime) rdfValue)
737                                                     .asCalendar());
738             } else {
739                 return valueFactory.createValue(literal.getString(), STRING);
740             }
741 
742         } else {
743             return valueFactory.createValue(data.asLiteral().getString(), type);
744         }
745     }

779      * @throws RepositoryException
780      */
781     void addPropertyToModel(final Resource subject,
782                             final Model model, final Property property, final Value v)
783         throws RepositoryException {
784 
785         if (v.getType() == BINARY) {
786             // exclude binary types from property serialization
787             return;
788         }
789 
790         final com.hp.hpl.jena.rdf.model.Property predicate =
791             FedoraTypesUtils.getPredicateForProperty.apply(property);
792 
793         switch (v.getType()) {
794             case BOOLEAN:
795                 model.addLiteral(subject, predicate, v.getBoolean());
796                 break;
797             case DATE:
798                 model.add(subject, predicate, ResourceFactory
799                                                   .createTypedLiteral(v.getDate()));
800                 break;
801             case DECIMAL:
802                 model.add(subject, predicate, ResourceFactory
803                                                   .createTypedLiteral(v.getDecimal()));
804                 break;
805             case DOUBLE:
806                 model.addLiteral(subject, predicate, v.getDouble());
807                 break;
808             case LONG:
809                 model.addLiteral(subject, predicate, v.getLong());
810                 break;
811             case URI:
812                 model.add(subject, predicate, model.createResource(v
813                                                                        .getString()));
814                 return;
815             case REFERENCE:
816             case WEAKREFERENCE:
817                 final Node refNode = session.getNodeByIdentifier(v.getString());
818                 model.add(subject,
819                              predicate,
820                              graphSubjects.getGraphSubject(refNode));
821                 break;
822             case PATH:
823                 model.add(subject,
824                              predicate,
825                              graphSubjects.getGraphSubject(v.getString()));
826                 break;
827 
828             default:
829                 model.add(subject, predicate, v.getString());
830 
831         }

852     /**
853      * Given an RDF predicate value (namespace URI + local name), figure out
854      * what JCR property to use
855      *
856      * @param node the JCR node we want a property for
857      * @param predicate the predicate to map to a property name
858      * @param namespaceMapping prefix => uri namespace mapping
859      * @return the JCR property name
860      * @throws RepositoryException
861      */
862     String getPropertyNameFromPredicate(final Node node,
863                                         final com.hp.hpl.jena.rdf.model.Property predicate,
864                                         final Map<String, String> namespaceMapping)
865         throws RepositoryException {
866 
867         final String prefix;
868 
869         final String namespace =
870             getJcrNamespaceForRDFNamespace(predicate.getNameSpace());
871 
872         final NamespaceRegistry namespaceRegistry =
873             NamespaceTools.getNamespaceRegistry(node);
874 
875         assert (namespaceRegistry != null);
876 
877         if (namespaceRegistry.isRegisteredUri(namespace)) {
878             prefix = namespaceRegistry.getPrefix(namespace);
879         } else {
880             final ImmutableBiMap<String, String> nsMap = ImmutableBiMap.copyOf(namespaceMapping);
881             if (nsMap.containsValue(namespace)
882                     && !namespaceRegistry.isRegisteredPrefix(nsMap.inverse().get(namespace))) {
883                 prefix = nsMap.inverse().get(namespace);
884                 namespaceRegistry.registerNamespace(prefix, namespace);
885             } else {
886                 prefix = namespaceRegistry.registerNamespace(namespace);
887             }
888         }
889 
890         final String localName = predicate.getLocalName();
891 
892         final String propertyName = prefix + ":" + localName;
893 
894         LOGGER.trace("Took RDF predicate {} and translated it to "
895                          + "JCR property {}", predicate, propertyName);
896 
897         return propertyName;
898
