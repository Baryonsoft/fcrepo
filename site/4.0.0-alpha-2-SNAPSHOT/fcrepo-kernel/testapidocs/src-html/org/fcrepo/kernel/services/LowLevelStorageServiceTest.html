176    @SuppressWarnings("unchecked")
177    @Test
178    public void shouldRetrieveLowLevelCacheStoresForCompositeStore()
179            throws Exception {
180
181        mockStatic(ServiceHelpers.class);
182
183        final Cache<?, ?> ispnCache1 = mock(Cache.class);
184        final Cache<?, ?> ispnCache2 = mock(Cache.class);
185        final CacheStore ispnCacheStore1 = mock(CacheStore.class);
186        final CacheStore ispnCacheStore2 = mock(CacheStore.class);
187        final BinaryStore plainBinaryStore = mock(BinaryStore.class);
188        final BinaryStore plainBinaryStore2 = mock(BinaryStore.class);
189
190        final GetCacheStore mockCacheStoreFunc = mock(GetCacheStore.class);
191        when(mockCacheStoreFunc.apply(ispnCache1)).thenReturn(ispnCacheStore1);
192        when(mockCacheStoreFunc.apply(ispnCache2)).thenReturn(ispnCacheStore2);
193
194        final CompositeBinaryStore mockStore = mock(CompositeBinaryStore.class);
195
196        final HashMap<String, BinaryStore> map =
197                new HashMap<String, BinaryStore>();
198        final List<Cache<?, ?>> caches = new ArrayList<Cache<?, ?>>();
199        caches.add(ispnCache1);
200        caches.add(ispnCache2);
201
202        map.put("default", plainBinaryStore);
203        map.put("a", plainBinaryStore2);
204        final InfinispanBinaryStore infinispanBinaryStore =
205                mock(InfinispanBinaryStore.class);
206        when(infinispanBinaryStore.getCaches()).thenReturn(caches);
207        map.put("b", infinispanBinaryStore);
208        when(mockStore.getNamedStoreIterator()).thenReturn(
209                map.entrySet().iterator());
210
211        final DistributedExecutorService mockCluster =
212                mock(DistributedExecutorService.class);
213        when(getClusterExecutor(infinispanBinaryStore)).thenReturn(mockCluster);
214
215        final BinaryKey key = new BinaryKey("key-123");
216
217        final LowLevelCacheEntry cacheEntry1 =
218                new CacheStoreEntry(ispnCacheStore1, "cache1", key);
219        final Set<LowLevelCacheEntry> cacheResponse1 = of(cacheEntry1);
220        final LowLevelCacheEntry cacheEntry2 =
221                new CacheStoreEntry(ispnCacheStore2, "cache2", key);
222        final Set<LowLevelCacheEntry> cacheResponse2 = of(cacheEntry2);
223        final Future<Collection<LowLevelCacheEntry>> future1 =
224                mock(Future.class);
225        final Future<Collection<LowLevelCacheEntry>> future2 =
226                mock(Future.class);
227        when(future1.get(any(Long.class), eq(MILLISECONDS))).thenReturn(
228                cacheResponse1);
229        when(future2.get(any(Long.class), eq(MILLISECONDS))).thenReturn(
230                cacheResponse2);
231
232        final List<Future<?>> mockClusterResults = new ArrayList<Future<?>>(2);
233        mockClusterResults.add(future1);
234        mockClusterResults.add(future2);
235
236        when(
237                mockCluster
238                        .submitEverywhere(any(CacheLocalTransform.class)))
239                .thenReturn(mockClusterResults);
240
241        final LowLevelStorageService testObj = new LowLevelStorageService();
242
243        when(plainBinaryStore.hasBinary(key)).thenReturn(true);
244        when(plainBinaryStore2.hasBinary(key)).thenReturn(false);
245        when(infinispanBinaryStore.hasBinary(key)).thenReturn(true);
246        final Set<LowLevelCacheEntry> entries =
247                testObj.getLowLevelCacheEntriesFromStore(mockStore, key);
248
249        assertEquals(3, entries.size());
250
251        assertTrue(entries.contains(new LocalBinaryStoreEntry(plainBinaryStore,
252                key)));
253        assertFalse(entries.contains(new LocalBinaryStoreEntry(
254                plainBinaryStore2, key)));
255        assertTrue(entries.contains(new CacheStoreEntry(ispnCacheStore1,
256                "cache1", key)));
257        assertTrue(entries.contains(new CacheStoreEntry(ispnCacheStore2,
258                "cache2", key)));
259
260    }
