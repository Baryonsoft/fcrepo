155    @Before
156    public void setUp() {
157        initMocks(this);
158        final String relPath = "/" + TESTPID + "/" + TESTDSID;
159        final String[] mockPrefixes = {MOCKPREFIX};
160        expectedNS = new HashMap<String, String>();
161        try {
162            when(mockSession.getRootNode()).thenReturn(mockRootNode);
163            when(mockRootNode.getNode(relPath)).thenReturn(mockDsNode);
164            when(mockRootNode.getProperty("fedora:size")).thenReturn(mockProp);
165            when(mockProp.getLong()).thenReturn(EXPECTED_SIZE);
166            when(mockRepo.login()).thenReturn(mockSession);
167
168            testObj = new RepositoryService();
169            testObj.setRepository(mockRepo);
170
171            when(mockSession.getNode("/objects")).thenReturn(mockRootNode);
172            when(mockRootNode.getNodes()).thenReturn(mockNI);
173            when(mockNI.getSize()).thenReturn(EXPECTED_SIZE);
174            when(testObj.findOrCreateNode(mockSession, "/objects")).thenReturn(
175                    mockRootNode);
176            when(mockSession.getWorkspace()).thenReturn(mockWorkspace);
177            when(mockWorkspace.getQueryManager()).thenReturn(mockQueryManager);
178            when(mockWorkspace.getNodeTypeManager()).thenReturn(
179                    mockNodeTypeManager);
180            when(mockWorkspace.getNamespaceRegistry()).thenReturn(
181                    mockNamespaceRegistry);
182            when(mockNamespaceRegistry.getPrefixes()).thenReturn(mockPrefixes);
183            when(mockNodeTypeManager.getAllNodeTypes()).thenReturn(mockNTI);
184            when(mockQueryManager.createQuery(anyString(), eq(Query.JCR_SQL2)))
185                    .thenReturn(mockQuery);
186            when(mockQuery.execute()).thenReturn(mockQueryResult);
187            when(mockQueryResult.getRows()).thenReturn(mockRI);
188            when(mockRI.hasNext()).thenReturn(true, false);
189            when(mockRI.nextRow()).thenReturn(mockRow);
190            when(mockRow.getValue(CONTENT_SIZE)).thenReturn(mockValue);
191            when(mockValue.getLong()).thenReturn(EXPECTED_SIZE);
192
193            expectedNS
194                    .put(MOCKPREFIX, mockNamespaceRegistry.getURI(MOCKPREFIX));
195
196        } catch (final RepositoryException e) {
197            e.printStackTrace();
198            fail(e.getMessage());
199        }

234    @Test
235    public void testSearchRepository() throws Exception {
236
237        mockStatic(JcrRdfTools.class);
238        JcrRdfTools mockJcrRdfTools = mock(JcrRdfTools.class);
239        when(JcrRdfTools.withContext(mockSubjectFactory, mockSession)).thenReturn(mockJcrRdfTools);
240
241        final Resource subject = createResource(
242                RdfLexicon.RESTAPI_NAMESPACE + "search/request");
243
244        when(mockSession.getValueFactory()).thenReturn(mockFactory);
245        when(mockSession.getWorkspace()).thenReturn(mockWorkspace);
246        when(mockWorkspace.getQueryManager()).thenReturn(mockQueryManager);
247        when(mockQueryManager.getQOMFactory()).thenReturn(mockQOMFactory);
248        when(mockQOMFactory.createQuery(null, null, null, null)).thenReturn(
249                mockQueryOM);
250        when(mockQueryOM.execute()).thenReturn(mockQueryResult);
251        when(mockQueryResult.getNodes()).thenReturn(mockNI);
252        when(mockNI.getSize()).thenReturn(500L);
253        when(mockNI.next()).thenReturn("");
254        when(
255                mockJcrRdfTools.getJcrPropertiesModel(any(org.fcrepo.kernel.utils.NodeIterator.class), eq(subject)))
256                .thenReturn(createDefaultModel());
257
258        testObj.searchRepository(mockSubjectFactory, subject, mockSession,
259                "search terms", 10, 0L);
260
261        // n+1
262        verify(mockQueryOM).setLimit(11);
263        verify(mockQueryOM).setOffset(0);
264        verify(mockQueryOM).execute();
265
266    }
