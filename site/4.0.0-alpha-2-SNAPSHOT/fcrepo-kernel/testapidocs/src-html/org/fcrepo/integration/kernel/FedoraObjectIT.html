018import static java.util.regex.Pattern.compile;
019import static org.junit.Assert.assertEquals;
020import static org.junit.Assert.assertFalse;
021import static org.junit.Assert.assertNotNull;
022import static org.junit.Assert.assertTrue;

024import java.io.IOException;

026import javax.inject.Inject;
027import javax.jcr.Repository;
028import javax.jcr.RepositoryException;
029import javax.jcr.Session;
030import javax.jcr.Value;
031
032import org.fcrepo.kernel.FedoraObject;
033import org.fcrepo.kernel.RdfLexicon;
034import org.fcrepo.kernel.rdf.impl.DefaultGraphSubjects;
035import org.fcrepo.kernel.services.ObjectService;
036import org.junit.Test;
037import org.springframework.test.context.ContextConfiguration;
038
039import com.hp.hpl.jena.query.Dataset;
040import com.hp.hpl.jena.update.UpdateAction;
041
042@ContextConfiguration({"/spring-test/repo.xml"})
043public class FedoraObjectIT extends AbstractIT {
044
045    @Inject
046    Repository repo;
047
048    @Inject
049    ObjectService objectService;
050
051    @Test
052    public void testCreatedObject() throws RepositoryException, IOException {
053        Session session = repo.login();
054        objectService.createObject(session, "/testObject");
055        session.save();
056        session.logout();
057        session = repo.login();
058        final FedoraObject obj =
059            objectService.getObject(session, "/testObject");
060        assertNotNull("Couldn't find object!", obj);
061    }
062
063    @Test
064    public void testGetSizeWhenInATree() throws Exception {
065
066        final Session session = repo.login();
067        final FedoraObject object =
068            objectService.createObject(session, "/parentObject");
069        final long originalSize = object.getSize();
070        objectService.createObject(session, "/parentObject/testChildObject");
071
072        session.save();
073
074        assertTrue(objectService
075                   .getObject(session, "/parentObject")
076                   .getSize() > originalSize);
077
078    }
079
080    @Test
081    public void testObjectGraph() throws Exception {
082        final Session session = repo.login();
083        final FedoraObject object =
084            objectService.createObject(session, "/graphObject");
085        final Dataset graphStore = object.getPropertiesDataset(new DefaultGraphSubjects(session));
086
087        final String graphSubject =
088                RdfLexicon.RESTAPI_NAMESPACE + "/graphObject";
089
090        assertFalse("Graph store should not contain JCR prefixes",
091                    compile("jcr").matcher(graphStore.toString()).find());
092        assertFalse("Graph store should contain our fcrepo prefix",
093                    compile("fcrepo")
094                    .matcher(graphStore.toString()).find());
095
096        UpdateAction
097            .parseExecute("PREFIX dc: <http://purl.org/dc/terms/>\n" +
098                          "INSERT { <http://example/egbook> dc:title " +
099                          "\"This is an example of an update that will be " +
100                          "ignored\" } WHERE {}", graphStore);
101
102        UpdateAction
103            .parseExecute("PREFIX dc: <http://purl.org/dc/terms/>\n" +
104                          "INSERT { <" + graphSubject + "> dc:title " +
105                          "\"This is an example title\" } WHERE {}",
106                          graphStore);
107
108        assertTrue(object.getNode().getProperty("dc:title").getValues()[0]
109                   .getString(),
110                   object.getNode().getProperty("dc:title").getValues()[0]
111                   .getString().equals("This is an example title"));
112
113
114        UpdateAction
115            .parseExecute("PREFIX myurn: <info:myurn/>\n" +
116                          "INSERT { <" + graphSubject + "> myurn:info " +
117                          "\"This is some example data\";" +
118                          "myurn:info  \"And so it this\"     } WHERE {}",
119                          graphStore);
120
121        final Value[] values =
122            object.getNode().getProperty(object.getNode().getSession()
123                                         .getNamespacePrefix("info:myurn/") +
124                                         ":info").getValues();
125
126        assertEquals("This is some example data", values[0].getString());
127        assertEquals("And so it this", values[1].getString());
128
129
130        UpdateAction
131            .parseExecute("PREFIX fedora-rels-ext: <"
132                    + RdfLexicon.RELATIONS_NAMESPACE + ">\n" +
133                    "INSERT { <" + graphSubject + "> fedora-rels-ext:" +
134                    "isPartOf <" + graphSubject + "> } WHERE {}", graphStore);
135        assertTrue(object.getNode().getProperty("fedorarelsext:isPartOf")
136                   .getValues()[0].getString(),
137                   object.getNode().getProperty("fedorarelsext:isPartOf")
138                   .getValues()[0].getString()
139                   .equals(object.getNode().getIdentifier()));
140
141
142        UpdateAction
143            .parseExecute("PREFIX dc: <http://purl.org/dc/terms/>\n" +
144                          "DELETE { <" + graphSubject + "> dc:title " +
145                          "\"This is an example title\" } WHERE {}",
146                          graphStore);
147
148        assertFalse("Found unexpected dc:title",
149                    object.getNode().hasProperty("dc:title"));
150
151        UpdateAction
152            .parseExecute("PREFIX fedora-rels-ext: <" +
153                    RdfLexicon.RELATIONS_NAMESPACE + ">\n" +
154                    "DELETE { <" + graphSubject + "> " +
155                    "fedora-rels-ext:isPartOf <" + graphSubject + "> " +
156                    "} WHERE {}", graphStore);
157        assertFalse("found unexpected reference",
158                    object.getNode().hasProperty("fedorarelsext:isPartOf"));
159
160        session.save();
161
162    }
