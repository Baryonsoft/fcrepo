095    @Test
096    public void testPutContent() throws RepositoryException, IOException,
097            InvalidChecksumException, URISyntaxException {
098        final String pid = "FedoraDatastreamsTest1";
099        final String dsId = "testDS";
100        final String dsContent = "asdf";
101        final String dsPath = "/" + pid + "/" + dsId;
102        final InputStream dsContentStream = IOUtils.toInputStream(dsContent);
103        when(mockNode.isNew()).thenReturn(true);
104        when(mockNode.getNode(JCR_CONTENT)).thenReturn(mockContentNode);
105        when(mockContentNode.getPath()).thenReturn(dsPath + "/jcr:content");
106        when(mockNodeService.exists(mockSession, dsPath)).thenReturn(false);
107        when(
108                mockDatastreams.createDatastreamNode(any(Session.class),
109                        eq(dsPath), anyString(), any(InputStream.class)))
110                .thenReturn(mockNode);
111        when(mockDatastreams.exists(mockSession, dsPath)).thenReturn(true);
112        final Response actual =
113                testObj.modifyContent(createPathList(pid, dsId), null,
114                        dsContentStream, null);
115        assertEquals(CREATED.getStatusCode(), actual.getStatus());
116        verify(mockDatastreams).createDatastreamNode(any(Session.class),
117                eq(dsPath), anyString(), any(InputStream.class));
118        verify(mockSession).save();
119    }

177    @Test
178    public void testModifyContent() throws RepositoryException, IOException,
179            InvalidChecksumException, URISyntaxException {
180        final String pid = "FedoraDatastreamsTest1";
181        final String dsId = "testDS";
182        final String dsContent = "asdf";
183        final String dsPath = "/" + pid + "/" + dsId;
184        final InputStream dsContentStream = IOUtils.toInputStream(dsContent);
185        when(mockNode.isNew()).thenReturn(false);
186        final Datastream mockDs = mockDatastream(pid, dsId, dsContent);
187        when(mockDatastreams.getDatastream(mockSession, dsPath)).thenReturn(
188                mockDs);
189        final Request mockRequest = mock(Request.class);
190        when(
191                mockRequest.evaluatePreconditions(any(Date.class),
192                        any(EntityTag.class))).thenReturn(null);
193        when(
194                mockDatastreams.createDatastreamNode(any(Session.class),
195                        eq(dsPath), anyString(), any(InputStream.class)))
196                .thenReturn(mockNode);
197        when(mockDatastreams.exists(mockSession, dsPath)).thenReturn(true);
198        final Response actual =
199                testObj.modifyContent(createPathList(pid, dsId), null,
200                        dsContentStream, mockRequest);
201        assertEquals(NO_CONTENT.getStatusCode(), actual.getStatus());
202        verify(mockDatastreams).createDatastreamNode(any(Session.class),
203                eq(dsPath), anyString(), any(InputStream.class));
204        verify(mockSession).save();
205    }

207    @Test
208    public void testGetContent() throws RepositoryException, IOException {
209        final String pid = "FedoraDatastreamsTest1";
210        final String dsId = "testDS";
211        final String path = "/" + pid + "/" + dsId;
212        final String dsContent = "asdf";
213        final Datastream mockDs = mockDatastream(pid, dsId, dsContent);
214        when(mockDatastreams.getDatastream(mockSession, path)).thenReturn(
215                mockDs);
216        when(mockDs.getNode()).thenReturn(mockNode);
217        when(mockNode.getPath()).thenReturn(path);
218        final Request mockRequest = mock(Request.class);
219        final Response actual =
220                testObj.getContent(createPathList(pid, dsId), null, mockRequest);
221        verify(mockDs).getContent();
222        verify(mockSession, never()).save();
223        final String actualContent =
224                IOUtils.toString((InputStream) actual.getEntity());
225        assertEquals("http://localhost/fcrepo" + path + ";rel=\"meta\"", actual.getMetadata().getFirst("Link"));
226        assertEquals("asdf", actualContent);
227    }
