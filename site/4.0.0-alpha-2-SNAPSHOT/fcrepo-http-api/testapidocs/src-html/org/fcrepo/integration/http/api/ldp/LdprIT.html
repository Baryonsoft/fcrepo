081    // 4.2.6
082    @Test
083    public void testMaySupportStandardRepresentations() {
084        final ImmutableList<String> formats = ImmutableList.of(WebContent.contentTypeTurtle,
085                                                                  WebContent.contentTypeNTriples,
086                                                                  WebContent.contentTypeN3,
087                                                                  WebContent.contentTypeRDFJSON,
088                                                                  WebContent.contentTypeRDFXML,
089                                                                  TEXT_HTML);
090
091        assertTrue(Iterables.all(formats, new Predicate<String>() {
092            @Override
093            public boolean apply(final String input) {
094                final HttpGet testMethod = new HttpGet(serverAddress + "");
095                testMethod.setHeader("Accept", input);
096                try {
097                    final HttpResponse response = client.execute(testMethod);
098                } catch (IOException e) {
099                    throw new RuntimeException("Failed on format " + input);
100                } finally {
101                    testMethod.releaseConnection();
102                }
103                return true;
104            }
105        }));
106
107    }
