114    /**
115     * Retrieve the node profile
116     * 
117     * @param pathList
118     * @param offset with limit, control the pagination window of details for
119     *        child nodes
120     * @param limit with offset, control the pagination window of details for
121     *        child nodes
122     * @param request
123     * @param uriInfo
124     * @return
125     * @throws RepositoryException
126     * @throws IOException
127     */
128    @GET
129    @Produces({TURTLE, N3, N3_ALT1, N3_ALT2, RDF_XML, RDF_JSON, NTRIPLES,
130            TEXT_HTML})
131    public Dataset describe(@PathParam("path") final List<PathSegment> pathList,
132            @QueryParam("offset") @DefaultValue("0") final long offset,
133            @QueryParam("limit") @DefaultValue("-1") final int limit,
134            @QueryParam("non-member-properties") final String nonMemberProperties,
135            @Context final Request request,
136            @Context HttpServletResponse servletResponse,
137            @Context final UriInfo uriInfo) throws RepositoryException, IOException {
138        final String path = toPath(pathList);
139        logger.trace("Getting profile for {}", path);
140
141        try {
142            final FedoraResource resource =
143                    nodeService.getObject(session, path);
144
145
146            final EntityTag etag = new EntityTag(resource.getEtagValue());
147            final Date date = resource.getLastModifiedDate();
148            final Date roundedDate = new Date();
149            if (date != null) {
150                roundedDate.setTime(date.getTime() - date.getTime() % 1000);
151            }
152            final ResponseBuilder builder =
153                    request.evaluatePreconditions(roundedDate, etag);
154            if (builder != null) {
155                final CacheControl cc = new CacheControl();
156                cc.setMaxAge(0);
157                cc.setMustRevalidate(true);
158                // here we are implicitly emitting a 304
159                // the exception is not an error, it's genuinely
160                // an exceptional condition
161                throw new WebApplicationException(builder.cacheControl(cc)
162                        .lastModified(date).tag(etag).build());
163            }
164            final HttpGraphSubjects subjects =
165                    new HttpGraphSubjects(session, FedoraNodes.class, uriInfo);
166
167            final int realLimit;
168            if (nonMemberProperties != null && limit == -1) {
169                realLimit = -2;
170            } else {
171                realLimit = limit;
172            }
173
174            final Dataset propertiesDataset =
175                    resource.getPropertiesDataset(subjects, offset, realLimit);
176
177            final Model treeModel = propertiesDataset.getNamedModel(propertiesDataset.getContext().getAsString(INLINED_RESOURCES_MODEL, "NO SUCH MODEL"));
178            if (limit > 0 && treeModel != null && treeModel
179                    .contains(subjects.getGraphSubject(resource.getNode()),
180                                 HAS_CHILD_COUNT)) {
181
182                Model requestModel = ModelFactory.createDefaultModel();
183
184                final long childCount = treeModel
185                                            .listObjectsOfProperty(subjects.getGraphSubject(resource.getNode()), HAS_CHILD_COUNT)
186                                            .nextNode().asLiteral().getLong();
187
188                if (childCount > (offset + limit)) {
189
190                    final Resource nextPageResource =
191                        requestModel.createResource(uriInfo
192                                                       .getRequestUriBuilder()
193                                                       .replaceQueryParam("offset", offset + limit)
194                                                       .replaceQueryParam("limit", limit)
195                                                       .build()
196                                                       .toString());
197                    requestModel.add(subjects.getContext(), NEXT_PAGE, nextPageResource);
198                }
199
200                final String firstPage = uriInfo
201                                       .getRequestUriBuilder()
202                                       .replaceQueryParam("offset", 0)
203                                       .replaceQueryParam("limit", limit)
204                                       .build().toString();
205                final Resource firstPageResource =
206                    requestModel.createResource(firstPage);
207                servletResponse.addHeader("Link", firstPage + ";rel=\"first\"");
208                requestModel.add(subjects.getContext(), FIRST_PAGE, firstPageResource);
209
210                propertiesDataset.addNamedModel("requestModel", requestModel);
211
212            }
213
214            if (!etag.getValue().isEmpty()) {
215                servletResponse.addHeader("ETag", etag.toString());
216            }
217
218            servletResponse.addHeader("Accept-Patch", WebContent.contentTypeSPARQLUpdate);
219            servletResponse.addHeader("Link", "http://www.w3.org/ns/ldp/Resource;rel=\"type\"");
220
221            addResponseInformationToDataset(resource, propertiesDataset,
222                                               uriInfo, subjects);
223
224            return propertiesDataset;
225
226        } finally {
227            session.logout();
228        }
229
230    }

232    /**
233     * Update an object using SPARQL-UPDATE
234     * 
235     * @param pathList
236     * @return 201
237     * @throws RepositoryException
238     * @throws org.fcrepo.kernel.exception.InvalidChecksumException
239     * @throws IOException
240     */
241    @PATCH
242    @Consumes({contentTypeSPARQLUpdate})
243    @Timed
244    public Response updateSparql(@PathParam("path")
245            final List<PathSegment> pathList,
246            @Context
247            final UriInfo uriInfo,
248            final InputStream requestBodyStream,
249            @Context final Request request)
250        throws RepositoryException, IOException {
251
252        final String path = toPath(pathList);
253        logger.debug("Attempting to update path: {}", path);
254
255        try {
256
257            if (requestBodyStream != null) {
258
259                final FedoraResource resource =
260                        nodeService.getObject(session, path);
261
262
263                final EntityTag etag = new EntityTag(resource.getEtagValue());
264                final Date date = resource.getLastModifiedDate();
265                final Date roundedDate = new Date();
266
267                if (date != null) {
268                    roundedDate.setTime(date.getTime() - date.getTime() % 1000);
269                }
270
271                final ResponseBuilder builder =
272                    request.evaluatePreconditions(roundedDate, etag);
273
274                if (builder != null) {
275                    throw new WebApplicationException(builder.build());
276                }
277
278                Dataset properties = resource.updatePropertiesDataset(new HttpGraphSubjects(
279                        session, FedoraNodes.class, uriInfo), IOUtils
280                        .toString(requestBodyStream));
281                Model problems = properties.getNamedModel(PROBLEMS_MODEL_NAME);
282                if (problems.size() > 0) {
283                    logger.info(
284                            "Found these problems updating the properties for {}: {}",
285                            path, problems);
286                    return status(FORBIDDEN).entity(problems.toString())
287                            .build();
288
289                }
290
291                session.save();
292
293                return status(SC_NO_CONTENT).build();
294            } else {
295                return status(SC_BAD_REQUEST).entity(
296                        "SPARQL-UPDATE requests must have content!").build();
297            }
298
299        } finally {
300            session.logout();
301        }
302    }

304    /**
305     * Replace triples with triples from a new model
306     * @param pathList
307     * @param uriInfo
308     * @param requestContentType
309     * @param requestBodyStream
310     * @return
311     * @throws RepositoryException
312     */
313    @PUT
314    @Consumes({TURTLE, N3, N3_ALT1, N3_ALT2, RDF_XML, RDF_JSON, NTRIPLES})
315    @Timed
316    public Response createOrReplaceObjectRdf(
317            @PathParam("path") final List<PathSegment> pathList,
318            @Context final UriInfo uriInfo,
319            @HeaderParam("Content-Type")
320            final MediaType requestContentType,
321            final InputStream requestBodyStream,
322            @Context final Request request) throws RepositoryException, URISyntaxException {
323        final String path = toPath(pathList);
324        logger.debug("Attempting to replace path: {}", path);
325        try {
326            final FedoraResource resource =
327                nodeService.getObject(session, path);
328
329            final Date date = resource.getLastModifiedDate();
330            final Date roundedDate = new Date();
331
332
333            final EntityTag etag = new EntityTag(resource.getEtagValue());
334
335            if (date != null) {
336                roundedDate.setTime(date.getTime() - date.getTime() % 1000);
337            }
338
339            final ResponseBuilder builder =
340                request.evaluatePreconditions(roundedDate, etag);
341
342            if (builder != null) {
343                throw new WebApplicationException(builder.build());
344            }
345
346            final HttpGraphSubjects subjects = new HttpGraphSubjects(session, FedoraNodes.class, uriInfo);
347
348            if (requestContentType != null && requestBodyStream != null)  {
349                final String contentType = requestContentType.toString();
350
351                final String format = contentTypeToLang(contentType).getName()
352                                          .toUpperCase();
353
354                final Model inputModel = ModelFactory.createDefaultModel()
355                                             .read(requestBodyStream,
356                                                      subjects.getGraphSubject(resource.getNode()).toString(),
357                                                      format);
358
359                resource.replacePropertiesDataset(subjects, inputModel);
360            }
361
362            session.save();
363
364            return status(SC_NO_CONTENT).build();
365        } finally {
366            session.logout();
367        }
368    }

370    /**
371     * Creates a new object.
372     * 
373     * @param pathList
374     * @return 201
375     * @throws RepositoryException
376     * @throws InvalidChecksumException
377     * @throws IOException
378     */
379    @POST
380    @Timed
381    public Response createObject(@PathParam("path")
382            final List<PathSegment> pathList,
383            @QueryParam("mixin")
384            final String mixin,
385            @QueryParam("checksum")
386            final String checksum,
387            @HeaderParam("Content-Type")
388            final MediaType requestContentType,
389            @HeaderParam("Slug")
390            final String slug,
391            @Context
392            final UriInfo uriInfo, final InputStream requestBodyStream)
393        throws RepositoryException, IOException, InvalidChecksumException, URISyntaxException {
394
395        final String newObjectPath;
396        final String path = toPath(pathList);
397
398
399        if (nodeService.exists(session, path)) {
400            final String pid;
401
402            if (slug != null) {
403                pid = slug;
404            }  else {
405                pid = pidMinter.mintPid();
406            }
407
408            newObjectPath = path + "/" + pid;
409        } else {
410            newObjectPath = path;
411        }
412
413        logger.debug("Attempting to ingest with path: {}", newObjectPath);
414
415        try {
416            if (nodeService.exists(session, newObjectPath)) {
417                return status(SC_CONFLICT).entity(
418                        path + " is an existing resource!").build();
419            }
420
421            final URI checksumURI;
422
423            if (checksum != null && !checksum.equals("")) {
424                checksumURI = new URI(checksum);
425            } else {
426                checksumURI = null;
427            }
428
429            final HttpGraphSubjects subjects = new HttpGraphSubjects(session, FedoraNodes.class, uriInfo);
430
431            final String objectType;
432
433            if (mixin != null) {
434                objectType = mixin;
435            } else {
436                if (requestContentType != null) {
437                    final String s = requestContentType.toString();
438                    if (s.equals(WebContent.contentTypeSPARQLUpdate) || contentTypeToLang(s) != null) {
439                        objectType = FedoraJcrTypes.FEDORA_OBJECT;
440                    } else {
441                        objectType = FedoraJcrTypes.FEDORA_DATASTREAM;
442                    }
443                } else {
444                    objectType = FedoraJcrTypes.FEDORA_OBJECT;
445                }
446            }
447
448            final FedoraResource result;
449
450            switch (objectType) {
451                case FedoraJcrTypes.FEDORA_OBJECT:
452                    result = objectService.createObject(session, newObjectPath);
453                    break;
454                case FedoraJcrTypes.FEDORA_DATASTREAM:
455                    result = datastreamService.createDatastream(session, newObjectPath);
456                    break;
457                default:
458                    throw new WebApplicationException(clientError().entity("Unknown object type " + objectType).build());
459            }
460
461            if (requestBodyStream != null) {
462
463                final MediaType contentType =
464                    requestContentType != null ? requestContentType
465                        : APPLICATION_OCTET_STREAM_TYPE;
466
467                final String contentTypeString = contentType.toString();
468
469                if (contentTypeString.equals(WebContent.contentTypeSPARQLUpdate)) {
470                    result.updatePropertiesDataset(subjects, IOUtils.toString(requestBodyStream));
471                } else if (contentTypeToLang(contentTypeString) != null) {
472
473                    final Lang lang = contentTypeToLang(contentTypeString);
474
475                    if (lang == null) {
476                        throw new WebApplicationException(notAcceptable().entity("Invalid Content type " + contentType).build());
477                    }
478
479                    final String format = lang.getName()
480                                              .toUpperCase();
481
482                    final Model inputModel = ModelFactory.createDefaultModel()
483                                                 .read(requestBodyStream,
484                                                          subjects.getGraphSubject(result.getNode()).toString(),
485                                                          format);
486
487                    result.replacePropertiesDataset(subjects, inputModel);
488                } else if (result instanceof Datastream) {
489
490                    final Node node =
491                        datastreamService.createDatastreamNode(session, newObjectPath,
492                                                                  contentTypeString, requestBodyStream,
493                                                                  checksumURI);
494
495
496                }
497            }
498
499            session.save();
500
501            logger.debug("Finished creating {} with path: {}", mixin, newObjectPath);
502
503            final URI location;
504            if (result.hasContent()) {
505                location = new URI(subjects.getGraphSubject(result.getNode().getNode(JcrConstants.JCR_CONTENT))
506                                                 .getURI());
507            } else {
508                location = new URI(subjects.getGraphSubject(result.getNode())
509                                                 .getURI());
510            }
511
512            return created(location).entity(newObjectPath).build();
513
514        } finally {
515            session.logout();
516        }
517    }
