061/**
062 * This class exposes the JCR workspace functionality. It may be
063 * too JCR-y in the long run, but this lets us exercise the functionality.
064 */
065@Component
066@Scope("prototype")
067@Path("/fcr:workspaces")
068public class FedoraRepositoryWorkspaces extends AbstractResource {
069
070    private static final Logger logger = getLogger(FedoraRepositoryWorkspaces.class);
071
072    @InjectedSession
073    protected Session session;
074
075    /**
076     * Get the list of accessible workspaces in this repository.
077     *
078     * @return
079     * @throws RepositoryException
080     */
081    @GET
082    @Produces({TURTLE, N3, N3_ALT1, N3_ALT2, RDF_XML, RDF_JSON, NTRIPLES,
083               TEXT_HTML})
084    @HtmlTemplate("jcr:workspaces")
085    public Dataset getWorkspaces() throws RepositoryException {
086
087        Model workspaceModel =
088                JcrRdfTools.withContext(null, session).getJcrPropertiesModel();
089
090        String[] workspaces =
091                session.getWorkspace().getAccessibleWorkspaceNames();
092
093        for (String workspace : workspaces) {
094            final Resource resource =
095                    createResource(uriInfo.getBaseUriBuilder()
096                                           .path("/workspace:" + workspace)
097                                           .build()
098                                           .toString());
099
100            workspaceModel.add(resource, RDF.type, NOT_IMPLEMENTED);
101        }
102
103        try {
104            return DatasetFactory.create(workspaceModel);
105        } finally {
106            session.logout();
107        }
108    }
109
110    /**
111     * Create a new workspace in the repository
112     *
113     * @param path
114     * @param uriInfo
115     * @return
116     * @throws RepositoryException
117     * @throws MalformedURLException
118     */
119    @POST
120    @Path("{path}")
121    public Response createWorkspace(@PathParam("path")
122            final String path,
123            @Context
124            final UriInfo uriInfo) throws RepositoryException,
125        MalformedURLException {
126        final Workspace workspace = session.getWorkspace();
127        workspace.createWorkspace(path);
128
129        return Response.created(
130                uriInfo.getAbsolutePathBuilder().path(FedoraNodes.class)
131                        .buildFromMap(ImmutableMap.of("path", path))).build();
132
133    }
134}
