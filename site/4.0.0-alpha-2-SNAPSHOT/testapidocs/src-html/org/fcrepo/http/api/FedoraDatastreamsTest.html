110    @Test
111    public void testModifyDatastreams() throws RepositoryException,
112            IOException, InvalidChecksumException, URISyntaxException {
113        final String pid = "FedoraDatastreamsTest1";
114        final String dsId1 = "testDs1";
115        final String dsId2 = "testDs2";
116        final HashMap<String, String> atts = new HashMap<String, String>(2);
117        atts.put(dsId1, "asdf");
118        atts.put(dsId2, "sdfg");
119        final MultiPart multipart = getStringsAsMultipart(atts);
120        when(mockNode.getPath()).thenReturn("/FedoraDatastreamsTest1");
121        when(mockSession.getNode("/FedoraDatastreamsTest1")).thenReturn(
122                mockNode);
123        final Response actual =
124                testObj.modifyDatastreams(createPathList(pid), Arrays.asList(
125                        dsId1, dsId2), multipart);
126        assertEquals(CREATED.getStatusCode(), actual.getStatus());
127        verify(mockDatastreams).createDatastreamNode(any(Session.class),
128                eq("/" + pid + "/" + dsId1), anyString(),
129                any(InputStream.class));
130        verify(mockDatastreams).createDatastreamNode(any(Session.class),
131                eq("/" + pid + "/" + dsId2), anyString(),
132                any(InputStream.class));
133        verify(mockSession).save();
134    }

136    @Test

138        final String pid = "FedoraDatastreamsTest1";

145        verify(mockNodes).deleteObject(mockSession, path + "/" + "ds2");
146        verify(mockSession).save();

149    @Test
150    public void testGetDatastreamsContents() throws RepositoryException,
151            IOException, NoSuchAlgorithmException {
152        final String pid = "FedoraDatastreamsTest1";
153        final String dsId = "testDS";
154        final String dsContent = "asdf";
155        final Datastream mockDs = mockDatastream(pid, dsId, dsContent);
156        when(mockDs.hasContent()).thenReturn(true);
157        when(mockIterator.hasNext()).thenReturn(true, false);
158        when(mockIterator.nextNode()).thenReturn(mockDsNode);
159        when(mockObject.getNode()).thenReturn(mockNode);
160        when(mockNode.getNodes(new String[] {dsId})).thenReturn(mockIterator);
161        when(mockNodes.getObject(mockSession, "/FedoraDatastreamsTest1"))
162                .thenReturn(mockObject);
163        when(mockDatastreams.asDatastream(mockDsNode)).thenReturn(mockDs);
164
165        final Response resp =
166                testObj.getDatastreamsContents(createPathList(pid),
167                        asList(dsId), mockRequest);
168        final MultiPart multipart = (MultiPart) resp.getEntity();
169
170        verify(mockDs).getContent();
171        verify(mockSession, never()).save();
172        assertEquals(1, multipart.getBodyParts().size());
173        final InputStream actualContent =
174                (InputStream) multipart.getBodyParts().get(0).getEntity();
175        assertEquals("/FedoraDatastreamsTest1/testDS", multipart.getBodyParts()
176                .get(0).getContentDisposition().getFileName());
177        assertEquals("asdf", IOUtils.toString(actualContent, "UTF-8"));
178    }

181    public void testGetDatastreamsContentsCached() throws RepositoryException,
182            IOException, NoSuchAlgorithmException {
183        final String pid = "FedoraDatastreamsTest1";
184        final String dsId = "testDS";
185        final String dsContent = "asdf";
186        final Datastream mockDs = mockDatastream(pid, dsId, dsContent);
187        when(mockIterator.hasNext()).thenReturn(true, false);
188        when(mockIterator.nextNode()).thenReturn(mockDsNode);
189        when(mockObject.getNode()).thenReturn(mockNode);
190        when(mockNode.getNodes(new String[] {dsId})).thenReturn(mockIterator);
191        when(mockNodes.getObject(mockSession, "/FedoraDatastreamsTest1"))
192                .thenReturn(mockObject);
193        when(mockDatastreams.asDatastream(mockDsNode)).thenReturn(mockDs);
194        when(
195                mockRequest.evaluatePreconditions(any(Date.class),
196                        any(EntityTag.class))).thenReturn(notModified());
197
198        final Response resp =
199                testObj.getDatastreamsContents(createPathList(pid),
200                        asList(dsId), mockRequest);
201        verify(mockDs, never()).getContent();
202        verify(mockSession, never()).save();
203        assertEquals(NOT_MODIFIED.getStatusCode(), resp.getStatus());
204    }

206    public static MultiPart getStringsAsMultipart(
207            final Map<String, String> contents) {
208        final MultiPart multipart = new MultiPart();
209        for (final Map.Entry<String, String> e : contents.entrySet()) {
210            final String id = e.getKey();
211            final String content = e.getValue();
212            final InputStream src = IOUtils.toInputStream(content);
213            final StreamDataBodyPart part = new StreamDataBodyPart(id, src);
214            try {
215                final FormDataContentDisposition cd =
216                        new FormDataContentDisposition("form-data;name=" + id +
217                                ";filename=" + id + ".txt");
218                part.contentDisposition(cd);
219            } catch (final ParseException ex) {
220                ex.printStackTrace();
221            }
222            multipart.bodyPart(part);
223        }
224        return multipart;
225    }
