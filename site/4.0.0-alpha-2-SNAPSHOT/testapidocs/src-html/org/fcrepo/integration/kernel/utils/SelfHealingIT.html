104    private void tamperWithNode(final Node node) throws Exception {
105
106        logger.info("Tampering with node " + node.toString());
107        final Set<LowLevelCacheEntry> binaryBlobs =
108            lowLevelService
109            .getLowLevelCacheEntries(node.getNode(JcrConstants.JCR_CONTENT));
110
111        final Iterator<LowLevelCacheEntry> it = binaryBlobs.iterator();
112
113        final LowLevelCacheEntry entryToTamper = it.next();
114        CacheStore store = ((CacheStoreEntry)entryToTamper).getLowLevelStore();
115        if (store instanceof ChainingCacheStore) {
116            store = ((ChainingCacheStore)store).getStores().keySet().iterator().next();
117            final OutputStream outputStream =
118                    new StoreChunkOutputStream(store, entryToTamper.getKey().toString() +
119                            "-data");
120            IOUtils.copy(new ByteArrayInputStream("qwerty".getBytes()), outputStream);
121            outputStream.close();
122        } else {
123            entryToTamper.storeValue(new ByteArrayInputStream("qwerty".getBytes()));
124        }
125        Thread.sleep(1000);
126
127    }

129    private Collection<FixityResult> getNodeFixity(final Datastream ds)
130        throws NoSuchAlgorithmException, RepositoryException {
131
132        return datastreamService

134                       ds.getContentDigest(),
135                       ds.getContentSize());

139    @Test
140    public void testEddiesMagicSelfHealingRepository() throws Exception {
141        final Session session = repo.login();
142
143        final String contentA =
144            "qn8y34jweuytgopfv3oevo29r7ajrp6r7q21jrxkkciggheh7rqqjbolsq09";
145        final String contentB =
146            "2e6sxpys67dslongzydxosx6ndze5vbgb6fnj1rr53buk405i1380a868xsb";
147
148        final URI shaA = ContentDigest.asURI("SHA-1",
149            Hex.encodeHexString(MessageDigest.getInstance("SHA-1")
150                                .digest(contentA.getBytes())));
151        final URI shaB = ContentDigest.asURI("SHA-1",
152            Hex.encodeHexString(MessageDigest.getInstance("SHA-1")
153                                .digest(contentB.getBytes())));
154        objectService.createObject(session, "/testSelfHealingObject");
155
156        datastreamService
157            .createDatastreamNode(session,
158                                  "/testSelfHealingObject/testDatastreamNode4",
159                                  "application/octet-stream",
160                                  new ByteArrayInputStream(contentA.getBytes()),
161                                  shaA);
162        datastreamService
163            .createDatastreamNode(session,
164                                  "/testSelfHealingObject/testDatastreamNode5",
165                                  "application/octet-stream",
166                                  new ByteArrayInputStream(contentB.getBytes()),
167                                  shaB);
168
169        session.save();
170
171        Collection<FixityResult> nodeFixity;
172        Collection<FixityResult> nodeFixity2;
173
174        Thread.sleep(1000);
175
176        final Datastream ds =
177            datastreamService
178            .getDatastream(session,
179                           "/testSelfHealingObject/testDatastreamNode4");
180
181        final Datastream ds2 =
182            datastreamService
183            .getDatastream(session,
184                           "/testSelfHealingObject/testDatastreamNode5");
185
186        logger.info("checking that our setup succeeded");
187        nodeFixity = getNodeFixity(ds);
188
189        nodeFixity2 = getNodeFixity(ds2);
190
191        assertNotEquals(0, nodeFixity.size());
192
193        logger.info("ds1");
194        boolean fixityOk = true;
195
196        for (final FixityResult fixityResult : nodeFixity) {
197            fixityOk &=
198                fixityResult.computedChecksum.equals(shaA);
199        }
200
201        assertTrue("Expected the fixity check to pass.", fixityOk);
202
203        logger.info("ds2");
204
205        fixityOk = true;
206
207        for (final FixityResult fixityResult : nodeFixity2) {
208            fixityOk &=
209                fixityResult.computedChecksum.equals(shaB);
210        }
211
212        assertTrue("Expected the fixity check to pass.", fixityOk);
213
214        tamperWithNode(ds.getNode());
215
216        nodeFixity = getNodeFixity(ds);
217
218        fixityOk = true;
219        for (final FixityResult fixityResult : nodeFixity) {
220            fixityOk &=
221                fixityResult.computedChecksum.equals(shaA);
222        }
223
224        assertFalse("Expected the fixity check to fail.", fixityOk);
225
226        datastreamService.runFixityAndFixProblems(ds);
227
228        nodeFixity = getNodeFixity(ds);
229
230        fixityOk = true;
231        for (final FixityResult fixityResult : nodeFixity) {
232            fixityOk &=
233                fixityResult.computedChecksum.equals(shaA);
234        }
235
236        assertTrue("Expected the fixity check to pass.", fixityOk);
237
238    }
