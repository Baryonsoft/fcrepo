053/**
054 * Listen to the event bus and record the most recent 10
055 * events, and provide them as an RSS feed.
056 */
057@Component
058@Path("/fcr:rss")
059public class RSSPublisher extends AbstractResource {
060
061    private static final Integer FEED_LENGTH = 10;
062
063    private static final String FEED_TYPE = "rss_2.0";
064
065    private static final String FEED_TITLE = "What's happening in Fedora";
066
067    private static final String FEED_DESCRIPTION = FEED_TITLE;
068
069    private final BlockingQueue<Event> feedQueue =
070            new ArrayBlockingQueue<Event>(FEED_LENGTH);
071
072    private final SyndFeed feed = new SyndFeedImpl();
073
074    /**
075     * Get the RSS feed
076     * 
077     * @return
078     * @throws FeedException
079     */
080    @GET
081    @Produces("application/rss+xml")
082    public StreamSource getFeed() throws FeedException {
083        feed.setLink(uriInfo.getBaseUri().toString());
084        feed.setEntries(transform(copyOf(feedQueue).reverse(), event2entry));
085        // TODO ought to make this stream, not go through a string
086        return new StreamSource(new ByteArrayInputStream(new SyndFeedOutput()
087                .outputString(feed).getBytes(StandardCharsets.UTF_8)));
088    }
089
090    private final Function<Event, SyndEntry> event2entry =
091            new Function<Event, SyndEntry>() {
092
093                @Override
094                public SyndEntry apply(final Event event) {
095                    final SyndEntry entry = new SyndEntryImpl();
096                    try {
097                        entry.setTitle(event.getIdentifier());
098                        entry.setLink(event.getPath());
099                        entry.setPublishedDate(new DateTime(event.getDate())
100                                .toDate());
101                        final SyndContent description = new SyndContentImpl();
102                        description.setType("text/plain");
103                        description.setValue(getEventType(event.getType())
104                                .toString());
105                        entry.setDescription(description);
106                    } catch (final RepositoryException e) {
107                        throw propagate(e);
108                    }
109                    return entry;
110                }
111
112            };
113
114    /**
115     * Engage the eventbus listener and set basic feed properties
116     */
117    @Override
118    @PostConstruct
119    public void initialize() {
120        eventBus.register(this);
121        feed.setFeedType(FEED_TYPE);
122        feed.setTitle(FEED_TITLE);
123        feed.setDescription(FEED_DESCRIPTION);
124    }
125
126    /**
127     * Remove our EventBus listener
128     */
129    @PreDestroy
130    public void shutDown() {
131        eventBus.unregister(this);
132    }
133
134    /**
135     * When a new event is received, add it to the buffer.
136     * 
137     * @param event
138     */
139    @Subscribe
140    public void newEvent(final Event event) {
141        if (feedQueue.remainingCapacity() > 0) {
142            feedQueue.offer(event);
143        } else {
144            feedQueue.poll();
145            feedQueue.offer(event);
146        }
147    }
148
149}
