016package org.fcrepo.auth.oauth.impl;
017
018import static com.google.common.collect.Sets.intersection;
019import static com.google.common.collect.Sets.newHashSet;
020import static org.fcrepo.auth.oauth.Constants.CLIENT_PROPERTY;
021import static org.fcrepo.auth.oauth.Constants.OAUTH_WORKSPACE;
022import static org.fcrepo.auth.oauth.Constants.SCOPES_PROPERTY;
023import static org.fcrepo.kernel.utils.FedoraTypesUtils.map;
024import static org.fcrepo.kernel.utils.FedoraTypesUtils.value2string;
025import static org.slf4j.LoggerFactory.getLogger;
026
027import java.security.AccessControlException;
028import java.util.Set;
029
030import javax.jcr.Node;
031import javax.jcr.PathNotFoundException;
032import javax.jcr.RepositoryException;
033import javax.jcr.Session;
034
035import org.apache.oltu.oauth2.as.request.OAuthTokenRequest;
036import org.fcrepo.auth.oauth.TokenRequestValidations;
037import org.fcrepo.http.commons.session.SessionFactory;
038import org.slf4j.Logger;
039
040
041public class DefaultTokenRequestValidations implements TokenRequestValidations {
042
043    private static final Logger LOGGER =
044        getLogger(DefaultTokenRequestValidations.class);
045
046    private SessionFactory sessions;
047
048    /**
049     * Instantiate the default audits and validations for an OAuth
050     * token request, using information stored in a jcr workspace
051     * accessible via the given session factory.
052     * @param sessions
053     */
054    public DefaultTokenRequestValidations(SessionFactory sessions) {
055        this.sessions = sessions;
056    }
057
058    /**
059     * @param oauthRequest
060     * @return An answer to whether or not this request matches up with an
061     *         authorization code issued at the {@link AuthzEndpoint}
062     * @throws RepositoryException
063     */
064    @Override
065    public boolean isValidAuthCode(OAuthTokenRequest oauthRequest)
066        throws RepositoryException {
067        final String client = oauthRequest.getClientId();
068        LOGGER.debug("Request has authorization client: {}", client);
069        final String code = oauthRequest.getCode();
070        if (code == null) {
071            return false;
072        }
073        final Set<String> scopes = oauthRequest.getScopes();
074        final Session session = sessions.getInternalSession(OAUTH_WORKSPACE);
075        try {
076            final Node authCodeNode =
077                    session.getNode("/authorization-codes/" + code);
078            LOGGER.debug("Found authorization code node stored: {}",
079                    authCodeNode.getPath());
080            // if the client is right
081            if (authCodeNode.getProperty(CLIENT_PROPERTY).getString().equals(
082                    client)) {
083                if (authCodeNode.getProperty(SCOPES_PROPERTY) != null) {
084
085                    final Set<String> storedScopes =
086                        newHashSet(
087                            map(authCodeNode.getProperty(SCOPES_PROPERTY)
088                                .getValues(),
089                                value2string));
090                    // and if there is at least one scope in common
091                    return (storedScopes.size() == 0 || intersection(storedScopes, scopes).size() > 0);
092                } else {
093                    return true;
094                }
095            }
096        } catch (final PathNotFoundException e) {
097            // this wasn't a code we stored
098            return false;
099        } finally {
100            session.logout();
101        }
102        throw new AccessControlException(
103                "Could not establish validity or invalidity of" +
104                 " authorization code! Code:" + code);
105    }
106
107    @Override
108    public boolean isValidClient(final OAuthTokenRequest oauthRequest) {
109        // TODO actually do some checking of client ID and secret and so forth
110        return oauthRequest != null;
111    }
112
113    @Override
114    public boolean isValidSecret(final OAuthTokenRequest oauthRequest) {
115        // TODO actually do some checking of client ID and secret and so forth
116        return oauthRequest != null;
117    }
118
119    @Override
120    public boolean isValidCredentials(final OAuthTokenRequest oauthRequest) {
121        // TODO actually do some credentials?
122        return oauthRequest != null;
123    }
124
125}
