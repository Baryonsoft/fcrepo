78      @Test
79      public void testGetNamespaceRegistryGraph() throws Exception {
80          Session session = repository.login();
81  
82          final Dataset registryGraph = objectService.getNamespaceRegistryGraph(session);
83  
84          final NamespaceRegistry namespaceRegistry = session.getWorkspace().getNamespaceRegistry();
85  
86          logger.info(namespaceRegistry.toString());
87          logger.info(registryGraph.toString());
88          for (String s : namespaceRegistry.getPrefixes()) {
89              if (s.isEmpty() || s.equals("xmlns") || s.equals("jcr")) {
90                  continue;
91              }
92              final String uri = namespaceRegistry.getURI(s);
93              assertTrue("expected to find JCR namespaces " + s + " in graph", registryGraph.asDatasetGraph().contains(Node.ANY, ResourceFactory.createResource(uri).asNode(), RdfLexicon.HAS_NAMESPACE_PREFIX.asNode(), ResourceFactory.createPlainLiteral(s).asNode()));
94          }
95          session.logout();
96      }

107         assertEquals("abc", namespaceRegistry.getPrefix("info:abc"));
108         session.logout();
