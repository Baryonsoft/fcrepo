80      /**
81       * Update the content of multiple datastreams from a multipart POST. The
82       * datastream to update is given by the name of the content disposition.
83       * 
84       * @param pathList
85       * @param dsidList
86       * @param multipart
87       * @return
88       * @throws RepositoryException
89       * @throws IOException
90       * @throws InvalidChecksumException
91       */
92      @POST
93      @Timed
94      public Response modifyDatastreams(@PathParam("path")
95      final List<PathSegment> pathList, @QueryParam("delete")
96      final List<String> dsidList, final MultiPart multipart)
97          throws RepositoryException, IOException, InvalidChecksumException,
98          URISyntaxException {
99  
100         final String path = toPath(pathList);
101         try {
102             for (final String dsid : dsidList) {
103                 logger.debug("Purging datastream: " + dsid);
104                 nodeService.deleteObject(session, path + "/" + dsid);
105             }
106 
107             for (final BodyPart part : multipart.getBodyParts()) {
108                 final String dsid =
109                         part.getContentDisposition().getParameters()
110                                 .get("name");
111                 logger.debug("Adding datastream: " + dsid);
112                 final String dsPath = path + "/" + dsid;
113                 final Object obj = part.getEntity();
114                 InputStream src = null;
115                 if (obj instanceof BodyPartEntity) {
116                     final BodyPartEntity entity =
117                             (BodyPartEntity) part.getEntity();
118                     src = entity.getInputStream();
119                 } else if (obj instanceof InputStream) {
120                     src = (InputStream) obj;
121                 }
122                 datastreamService.createDatastreamNode(session, dsPath, part
123                         .getMediaType().toString(), src);
124             }
125 
126             session.save();
127 
128             final HttpGraphSubjects subjects =
129                     new HttpGraphSubjects(FedoraNodes.class, uriInfo, session);
130 
131             return created(
132                     new URI(subjects.getGraphSubject(session.getNode(path))
133                             .getURI())).build();
134 
135         } finally {
136             session.logout();
137         }
138     }

143      * @param pathList

148     @DELETE

150     public Response deleteDatastreams(@PathParam("path")
151     final List<PathSegment> pathList, @QueryParam("dsid")
152     final List<String> dsidList) throws RepositoryException {
153         try {
154             final String path = toPath(pathList);
155             for (final String dsid : dsidList) {
156                 logger.debug("purging datastream {}", path + "/" + dsid);
157                 nodeService.deleteObject(session, path + "/" + dsid);
158             }
159             session.save();
160             return noContent().build();
161         } finally {
162             session.logout();
163         }
164     }

166     /**
167      * Retrieve multiple datastream bitstreams in a single request as a
168      * multipart/mixed response.
169      * 
170      * @param pathList
171      * @param requestedDsids
172      * @param request
173      * @return
174      * @throws RepositoryException
175      * @throws IOException
176      * @throws NoSuchAlgorithmException
177      */
178     @GET
179     @Produces("multipart/mixed")
180     @Timed
181     public Response getDatastreamsContents(@PathParam("path")
182     final List<PathSegment> pathList, @QueryParam("dsid")
183     final List<String> requestedDsids, @Context
184     final Request request) throws RepositoryException, IOException,
185         NoSuchAlgorithmException {
186 
187         final ArrayList<Datastream> datastreams = new ArrayList<Datastream>();
188 
189         try {
190             final String path = toPath(pathList);
191             // TODO: wrap some of this JCR logic in an fcrepo abstraction;
192 
193             final Node node = nodeService.getObject(session, path).getNode();
194 
195             Date date = new Date();
196 
197             final MessageDigest digest = MessageDigest.getInstance("SHA-1");
198 
199             final NodeIterator ni;
200 
201             if (requestedDsids.isEmpty()) {
202                 ni = node.getNodes();
203             } else {
204                 ni =
205                         node.getNodes(requestedDsids
206                                 .toArray(new String[requestedDsids.size()]));
207             }
208 
209             // transform the nodes into datastreams, and calculate cache header
210             // data
211             while (ni.hasNext()) {
212 
213                 final Node dsNode = ni.nextNode();
214                 final Datastream ds = datastreamService.asDatastream(dsNode);
215 
216                 if (!ds.hasContent()) {
217                     continue;
218                 }
219 
220                 digest.update(ds.getContentDigest().toString().getBytes(
221                         StandardCharsets.UTF_8));
222 
223                 if (ds.getLastModifiedDate().after(date)) {
224                     date = ds.getLastModifiedDate();
225                 }
226 
227                 datastreams.add(ds);
228             }
229 
230             final URI digestURI =
231                     ContentDigest.asURI(digest.getAlgorithm(), digest.digest());
232             final EntityTag etag = new EntityTag(digestURI.toString());
233 
234             final Date roundedDate = new Date();
235             roundedDate.setTime(date.getTime() - date.getTime() % 1000);
236 
237             Response.ResponseBuilder builder =
238                     request.evaluatePreconditions(roundedDate, etag);
239 
240             final CacheControl cc = new CacheControl();
241             cc.setMaxAge(0);
242             cc.setMustRevalidate(true);
243 
244             if (builder == null) {
245                 final MultiPart multipart = new MultiPart();
246 
247                 for (final Datastream ds : datastreams) {
248                     final BodyPart bodyPart =
249                             new BodyPart(ds.getContent(), MediaType.valueOf(ds
250                                     .getMimeType()));
251                     bodyPart.setContentDisposition(ContentDisposition.type(
252                             "attachment").fileName(ds.getPath()).creationDate(
253                             ds.getCreatedDate()).modificationDate(
254                             ds.getLastModifiedDate()).size(ds.getContentSize())
255                             .build());
256                     multipart.bodyPart(bodyPart);
257                 }
258 
259                 builder = Response.ok(multipart, MULTIPART_FORM_DATA);
260             }
261 
262             return builder.cacheControl(cc).lastModified(date).tag(etag)
263                     .build();
264 
265         } finally {
266             session.logout();
267         }
268     }
