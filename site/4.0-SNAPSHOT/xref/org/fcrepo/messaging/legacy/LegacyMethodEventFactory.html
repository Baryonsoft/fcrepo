18  public class LegacyMethodEventFactory implements JMSEventMessageFactory {

20      final private Logger logger = LoggerFactory
21              .getLogger(LegacyMethodEventFactory.class);
22  
23      public LegacyMethodEventFactory() {
24      }

26      @Override
27      public Message getMessage(Event jcrEvent, javax.jcr.Session jcrSession,
28              javax.jms.Session jmsSession) throws RepositoryException,
29              IOException, JMSException {
30          logger.trace("Received an event to transform.");
31          String path = jcrEvent.getPath();
32          logger.trace("Retrieved path from event.");
33          Node resource = jcrSession.getNode(path);
34          logger.trace("Retrieved node from event.");
35          LegacyMethod legacy = new LegacyMethod(jcrEvent, resource);
36          StringWriter writer = new StringWriter();
37          legacy.writeTo(writer);
38          String atomMessage = writer.toString();
39          logger.debug("Constructed serialized Atom message from event.");
40          TextMessage tm = jmsSession.createTextMessage(atomMessage);
41          String pid = legacy.getPid();
42          if (pid != null) tm.setStringProperty("pid", pid);
43          tm.setStringProperty("methodName", legacy.getMethodName());
44          tm.setJMSType(LegacyMethod.FORMAT);
45          tm.setStringProperty("fcrepo.server.version",
46                  LegacyMethod.SERVER_VERSION);
47          logger.trace("Successfully created JMS message from event.");
48          return tm;
49      }
