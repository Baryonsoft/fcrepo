73      @GET
74      @Path("/")
75      @Produces({TEXT_XML, APPLICATION_JSON})
76      public ObjectDatastreams getDatastreams(@PathParam("pid")
77      final String pid) throws RepositoryException, IOException {
78  
79          final ObjectDatastreams objectDatastreams = new ObjectDatastreams();
80          final Builder<DatastreamElement> datastreams = builder();
81  
82          NodeIterator i = getObjectNode(pid).getNodes();
83          while (i.hasNext()) {
84              final Node ds = i.nextNode();
85              datastreams.add(new DatastreamElement(ds.getName(), ds.getName(),
86                      getDSMimeType(ds)));
87          }
88          objectDatastreams.datastreams = datastreams.build();
89          return objectDatastreams;
90

93      @POST
94      @Path("/")
95      public Response addDatastreams(@PathParam("pid")
96      final String pid, final List<Attachment> attachmentList)
97              throws RepositoryException, IOException {
98  
99          final Session session = repo.login();
100         try {
101             Long oldObjectSize =
102                     getObjectSize(session.getNode(getObjectJcrNodePath(pid)));
103 
104             for (final Attachment a : attachmentList) {
105                 final String dsid =
106                         a.getContentDisposition().getParameter("name");
107                 final String dsPath = getDatastreamJcrNodePath(pid, dsid);
108                 createDatastreamNode(session, dsPath, a.getDataHandler()
109                         .getContentType(), a.getDataHandler().getInputStream());
110 
111             }
112             session.save();
113 
114             /*
115              * we save before updating the repo size because the act of
116              * persisting session state creates new system-curated nodes and
117              * properties which contribute to the footprint of this resource
118              */
119             updateRepositorySize(getObjectSize(session
120                     .getNode(getObjectJcrNodePath(pid))) -
121                     oldObjectSize, session);
122             // now we save again to persist the repo size
123             session.save();
124         } finally {
125             session.logout();
126         }
127 
128         return created(uriInfo.getAbsolutePath()).build();
129     }

146     @POST

148     public Response addDatastream(@PathParam("pid")
149     final String pid, @PathParam("dsid")
150     final String dsid, @HeaderParam("Content-Type")
151     MediaType contentType, InputStream requestBodyStream)
152             throws RepositoryException, IOException {
153         final Session session = repo.login();
154 
155         contentType =
156                 contentType != null ? contentType
157                         : APPLICATION_OCTET_STREAM_TYPE;
158         String dspath = getDatastreamJcrNodePath(pid, dsid);
159 
160         if (!session.nodeExists(dspath)) {
161             return created(
162                     addDatastreamNode(pid, dspath, contentType,
163                             requestBodyStream, session)).build();
164         } else {
165             session.getNode(dspath).remove();
166             session.save();
167             return created(
168                     addDatastreamNode(pid, dspath, contentType,
169                             requestBodyStream, session)).build();
170         }
171 
172     }

208     private URI addDatastreamNode(final String pid, final String dsPath,
209             final MediaType contentType, final InputStream requestBodyStream,
210             final Session session) throws RepositoryException, IOException {
211 
212         Long oldObjectSize =
213                 getObjectSize(session.getNode(getObjectJcrNodePath(pid)));
214         logger.debug("Attempting to add datastream node at path: " + dsPath);
215         try {
216             boolean created = session.nodeExists(dsPath);
217             createDatastreamNode(session, dsPath, contentType.toString(),
218                     requestBodyStream);
219             session.save();
220             if (created) {
221                 /*
222                  * we save before updating the repo size because the act of
223                  * persisting session state creates new system-curated nodes and
224                  * properties which contribute to the footprint of this resource
225                  */
226                 updateRepositorySize(getObjectSize(session
227                         .getNode(getObjectJcrNodePath(pid))) -
228                         oldObjectSize, session);
229                 // now we save again to persist the repo size
230                 session.save();
231             }
232         } finally {
233             session.logout();
234         }
235         logger.debug("Finished adding datastream node at path: " + dsPath);
236         return uriInfo.getAbsolutePath();
237     }

360     private DatastreamProfile getDSProfile(Datastream ds)
361             throws RepositoryException, IOException {
362         logger.trace("Executing getDSProfile() with node: " + ds.getDsId());
363         final DatastreamProfile dsProfile = new DatastreamProfile();
364         dsProfile.dsID = ds.getDsId();
365         dsProfile.pid = ds.getObject().getName();
366         logger.trace("Retrieved datastream " + ds.getDsId() + "'s parent: " +
367                 dsProfile.pid);
368         dsProfile.dsLabel = ds.getLabel();
369         logger.trace("Retrieved datastream " + ds.getDsId() + "'s label: " +
370                 ds.getLabel());
371         dsProfile.dsState = A;
372         dsProfile.dsMIME = ds.getMimeType();
373         dsProfile.dsSize =
374                 getNodePropertySize(ds.getNode()) + ds.getContentSize();
375         dsProfile.dsCreateDate =
376                 ds.getNode().getProperty("jcr:created").getString();
377         return dsProfile;
378     }
