32  /**
33   * @todo Add Documentation.
34   * @author barmintor
35   * @date May 15, 2013
36   */
37  public class DefaultGraphSubjects implements GraphSubjects {
38  
39  
40      @Override
41      public Resource getGraphSubject(String absPath) throws RepositoryException {
42          if (absPath.endsWith(JcrConstants.JCR_CONTENT)) {
43              return ResourceFactory
44                             .createResource("info:fedora" +
45                                                     absPath.replace(JcrConstants.JCR_CONTENT,
46                                                                            FedoraJcrTypes.FCR_CONTENT));
47          } else {
48              return ResourceFactory.createResource("info:fedora" + absPath);
49          }
50      }
51  
52      /**
53       * @todo Add Documentation.
54       */
55      @Override
56      public Resource getGraphSubject(Node node) throws RepositoryException {
57          return getGraphSubject(node.getPath());
58      }
59  
60      /**
61       * @todo Add Documentation.
62       */
63      @Override
64      public Node getNodeFromGraphSubject(Session session, Resource subject)
65          throws RepositoryException {
66          if (!isFedoraGraphSubject(subject)) {
67              return null;
68          }
69  
70          final String absPath =
71              subject.getURI().substring("info:fedora".length());
72  
73          if (absPath.endsWith(FCR_CONTENT)) {
74              return session.getNode(absPath.replace(FedoraJcrTypes.FCR_CONTENT,
75                                                     JcrConstants.JCR_CONTENT));
76          } else if (session.nodeExists(absPath)) {
77              return session.getNode(absPath);
78          } else {
79              return null;
80          }
81      }
82  
83      /**
84       * @todo Add Documentation.
85       */
86      @Override
87      public boolean isFedoraGraphSubject(Resource subject) {
88          checkArgument(subject != null, "null cannot be a Fedora object!");
89          assert(subject != null);
90  
91          return subject.isURIResource() &&
92              subject.getURI().startsWith("info:fedora/");
93      }
94  
95  }
