718    private void validateHTML(final String path) throws Exception {
719        final HttpGet getMethod = new HttpGet(serverAddress + path);
720        getMethod.addHeader("Accept", "text/html");
721        final HttpResponse response = client.execute(getMethod);
722        assertEquals(OK.getStatusCode(), response.getStatusLine()
723                .getStatusCode());
724        final String content = EntityUtils.toString(response.getEntity());
725        logger.debug("Retrieved HTML view:\n" + content);
726
727        final HtmlParser htmlParser = new HtmlParser(ALLOW);
728        htmlParser.setDoctypeExpectation(NO_DOCTYPE_ERRORS);
729        htmlParser.setErrorHandler(new HTMLErrorHandler());
730        htmlParser.setContentHandler(new TreeBuilder());
731        try (final InputStream htmlStream =
732                new ByteArrayInputStream(content.getBytes())) {
733            htmlParser.parse(new InputSource(htmlStream));
734        }
735        logger.info("HTML found to be valid.");
736    }
