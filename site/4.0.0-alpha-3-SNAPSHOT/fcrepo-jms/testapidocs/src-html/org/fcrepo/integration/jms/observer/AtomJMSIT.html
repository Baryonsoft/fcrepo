087    @Before
088    public void acquireConnection() throws JMSException {
089        logger.debug(this.getClass().getName() + " acquiring JMS connection.");
090        connection = connectionFactory.createConnection();
091        connection.start();
092        session = connection.createSession(false, AUTO_ACKNOWLEDGE);
093        consumer = session.createConsumer(session.createTopic("fedora"));
094        consumer.setMessageListener(this);
095    }
096
097    @After
098    public void releaseConnection() throws JMSException {
099        logger.debug(this.getClass().getName() + " releasing JMS connection.");
100        consumer.close();
101        session.close();
102        connection.close();
103    }

105    @Test
106    public void testAtomStream() throws LoginException, RepositoryException,
107        InterruptedException {
108        Session session = repository.login();
109        session.getRootNode().addNode("test1").addMixin(FEDORA_OBJECT);
110        session.save();
111
112        waitForEntry();
113
114        session.logout();
115
116        if (entry == null) fail("Waited a second, got no messages");
117        List<Category> categories = copyOf(entry.getCategories("xsd:string"));
118        final String title = entry.getTitle();
119        entry = null;
120        String path = null;
121        for (Category cat : categories) {
122            if (cat.getLabel().equals("fedora-types:pid")) {
123                logger.debug("Found Category with term: " + cat.getTerm());
124                path = cat.getTerm();
125            }
126        }
127        assertEquals("Got wrong pid!", "test1", path);
128        assertEquals("Got wrong method!", "ingest", title);
129    }

156    @Test
157    public void testDatastreamTerm() throws NoSuchNodeTypeException,
158        VersionException, ConstraintViolationException, LockException,
159        ItemExistsException, PathNotFoundException, RepositoryException,
160        InterruptedException {
161        logger.trace("BEGIN: testDatastreamTerm()");
162        Session session = repository.login();
163        final Node object = session.getRootNode().addNode("testDatastreamTerm");
164        object.addMixin(FEDORA_OBJECT);
165        session.save();
166        logger.trace("testDatastreamTerm called session.save()");
167
168        waitForEntry();
169        if (entry == null) fail("Waited a second, got no messages");
170        List<Category> categories = copyOf(entry.getCategories("xsd:string"));
171        entry = null;
172
173        String path = null;
174
175        logger.trace("Matched {} categories with scheme xsd:string", categories
176                .size());
177        for (Category cat : categories) {
178            if (cat.getLabel().equals("fedora-types:pid")) {
179                logger.debug("Found Category with term: " + cat.getTerm());
180                path = cat.getTerm();
181            }
182        }
183        assertEquals("Got wrong object PID!", "testDatastreamTerm", path);
184
185        final Node ds = object.addNode("DATASTREAM");
186        ds.addMixin(FEDORA_DATASTREAM);
187        ds.addNode(JCR_CONTENT).setProperty(JCR_DATA, "fake data");
188        session.save();
189        logger.trace("testDatastreamTerm called session.save()");
190        session.logout();
191        logger.trace("testDatastreamTerm called session.logout()");
192
193        waitForEntry();
194        if (entry == null) fail("Waited a second, got no messages");
195        categories = copyOf(entry.getCategories("xsd:string"));
196        entry = null;
197
198        logger.trace("Matched {} categories with scheme xsd:string", categories
199                .size());
200        for (Category cat : categories) {
201            if (cat.getLabel().equals("fedora-types:dsID")) {
202                logger.debug("Found Category with term: " + cat.getTerm());
203                path = cat.getTerm();
204            }
205        }
206
207        assertEquals("Got wrong datastream ID!", "DATASTREAM", path);
208        logger.trace("END: testDatastreamTerm()");
209    }

211    @Override
212    public void onMessage(Message message) {
213        logger.debug("Received JMS message: " + message.toString());
214
215        TextMessage tMessage = (TextMessage) message;
216        try {
217            if (LegacyMethod.canParse(message)) {
218                LegacyMethod legacy = new LegacyMethod(tMessage.getText());
219                entry = legacy.getEntry();
220                logger.debug("Parsed Entry: {}", entry.toString());
221            } else {
222                logger.warn("Could not parse message: {}", message);
223            }
224        } catch (Exception e) {
225            logger.error("Exception receiving message: {}", e);
226            fail(e.getMessage());
227        }
228        synchronized (this) {
229            this.notify();
230        }
231    }
232
233    private void waitForEntry() throws InterruptedException {
234        for (int i = 0; i < 5; i++) {
235            if (entry == null) { // must not have rec'vd event yet
236                synchronized (this) {
237                    this.wait(1000);
238                }
239            } else {
240                break;
241            }
242        }
243    }
