017package org.fcrepo.jms.legacy;
018
019import java.io.Reader;
020import java.util.UUID;
021
022import org.apache.abdera.Abdera;
023import org.apache.abdera.model.Entry;
024import org.apache.abdera.parser.Parser;
025
026/**
027 * Create and parse ATOM events
028 */
029public abstract class EntryFactory {
030
031    private static final Abdera ABDERA = new Abdera();
032
033    private static final Parser ABDERA_PARSER = ABDERA.getParser();
034
035    public static final String FORMAT =
036            "info:fedora/fedora-system:ATOM-APIM-1.0";
037
038    // TODO get this out of the build properties
039    public static final String SERVER_VERSION = "4.0.0-SNAPSHOT";
040
041    private static final String TYPES_NS =
042            "http://www.fedora.info/definitions/1/0/types/";
043
044    public static final String VERSION_PREDICATE =
045            "info:fedora/fedora-system:def/view#version";
046
047    private static final String XSD_NS = "http://www.w3.org/2001/XMLSchema";
048
049    public static final String FORMAT_PREDICATE =
050            "http://www.fedora.info/definitions/1/0/types/formatURI";
051
052    /**
053     * Create a new base Abdera document for our ATOM entry
054     * @return
055     */
056    static Entry newEntry() {
057        final Entry entry = ABDERA.newEntry();
058        entry.declareNS(XSD_NS, "xsd");
059        entry.declareNS(TYPES_NS, "fedora-types");
060        entry.setId("urn:uuid:" + UUID.randomUUID().toString());
061        entry.addCategory(FORMAT_PREDICATE, FORMAT, "format");
062        entry.addCategory(VERSION_PREDICATE, SERVER_VERSION, "version");
063        return entry;
064    }
065
066    /**
067     * Parse an ATOM entry document into Abdera
068     * @param input
069     * @return
070     */
071    static Entry parse(Reader input) {
072        return (Entry) ABDERA_PARSER.parse(input).getRoot();
073    }
074
075}
