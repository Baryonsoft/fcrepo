017package org.fcrepo.http.commons.responses;
018
019import static com.hp.hpl.jena.graph.NodeFactory.createLiteral;
020import static com.hp.hpl.jena.graph.NodeFactory.createURI;
021import static com.hp.hpl.jena.rdf.model.ModelFactory.createDefaultModel;
022import static javax.ws.rs.core.MediaType.TEXT_HTML_TYPE;
023import static javax.ws.rs.core.MediaType.valueOf;
024import static org.fcrepo.http.commons.responses.RdfSerializationUtils.primaryTypePredicate;
025import static org.junit.Assert.assertEquals;
026import static org.junit.Assert.assertFalse;
027import static org.junit.Assert.assertTrue;
028import static org.mockito.Mockito.mock;
029
030import java.io.ByteArrayOutputStream;
031import java.io.IOException;
032import java.lang.reflect.Type;
033
034import javax.ws.rs.WebApplicationException;
035import javax.ws.rs.core.MultivaluedMap;
036
037import org.fcrepo.http.commons.responses.RdfProvider;
038import org.junit.Test;
039
040import com.hp.hpl.jena.graph.Triple;
041import com.hp.hpl.jena.query.Dataset;
042import com.hp.hpl.jena.sparql.core.DatasetImpl;
043
044public class RdfProviderTest {
045
046    final RdfProvider rdfProvider = new RdfProvider();
047
048    Dataset testData = new DatasetImpl(createDefaultModel());
049
050    {
051        testData.asDatasetGraph().getDefaultGraph().add(
052                new Triple(createURI("test:subject"),
053                        createURI("test:predicate"),
054                        createLiteral("test:object")));
055        testData.asDatasetGraph().getDefaultGraph().add(
056                new Triple(createURI("test:subject"), primaryTypePredicate,
057                        createLiteral("nt:file")));
058
059    }
060
061    @Test
062    public void testIsWriteable() {
063        assertTrue(
064                "Gave false response to RdfProvider.isWriteable() that contained a legitimate combination of parameters!",
065                rdfProvider.isWriteable(Dataset.class, Dataset.class, null,
066                        valueOf("application/rdf+xml")));
067        assertFalse(
068                "RdfProvider.isWriteable() should return false if asked to serialize anything other than Dataset!",
069                rdfProvider.isWriteable(RdfProvider.class, RdfProvider.class,
070                        null, valueOf("application/rdf+xml")));
071        assertFalse(
072                "RdfProvider.isWriteable() should return false to text/html!",
073                rdfProvider.isWriteable(Dataset.class, Dataset.class, null,
074                        TEXT_HTML_TYPE));
075    }
076
077    @Test
078    public void testGetSize() {
079        assertEquals("Returned wrong size from RdfProvider!", rdfProvider
080                .getSize(null, null, null, null, null), -1);
081
082    }
083
084    @SuppressWarnings("unchecked")
085    @Test
086    public void testWriteTo() throws WebApplicationException,
087            IllegalArgumentException, IOException {
088        final ByteArrayOutputStream outStream = new ByteArrayOutputStream();
089
090        rdfProvider.writeTo(testData, Dataset.class, mock(Type.class), null,
091                valueOf("application/rdf+xml"), mock(MultivaluedMap.class),
092                outStream);
093        final byte[] results = outStream.toByteArray();
094        assertTrue("Got no output from serialization!", results.length > 0);
095        assertTrue("Couldn't find test RDF-object mentioned!", new String(
096                results).contains("test:object"));
097    }
098}
