017package org.fcrepo.http.commons.test.util;
018
019import static java.lang.reflect.Proxy.newProxyInstance;
020
021import java.io.IOException;
022import java.lang.reflect.InvocationHandler;
023import java.lang.reflect.Method;
024import java.security.Principal;
025import java.util.Collections;
026import java.util.Set;
027
028import javax.servlet.Filter;
029import javax.servlet.FilterChain;
030import javax.servlet.FilterConfig;
031import javax.servlet.ServletException;
032import javax.servlet.ServletRequest;
033import javax.servlet.ServletResponse;
034import javax.servlet.http.HttpServletRequest;
035import javax.ws.rs.WebApplicationException;
036
037import org.glassfish.grizzly.http.server.GrizzlyPrincipal;
038import org.slf4j.Logger;
039import org.slf4j.LoggerFactory;
040
041import com.sun.jersey.core.util.Base64;
042import com.sun.jersey.spi.container.ContainerRequest;
043
044/**
045 * @author Gregory Jansen
046 */
047public class TestAuthenticationRequestFilter implements Filter {
048
049    private static final Logger log = LoggerFactory
050            .getLogger(TestAuthenticationRequestFilter.class);
051
052    // @Context
053    // UriInfo uriInfo;
054
055    private static final String FEDORA_ADMIN_USER = "fedoraAdmin";
056
057    /*
058     * (non-Javadoc)
059     * @see
060     * com.sun.jersey.spi.container.ContainerRequestFilter#filter(com.sun.jersey
061     * .spi.container.ContainerRequest)
062     */
063    @Override
064    public void doFilter(final ServletRequest request,
065            final ServletResponse response, final FilterChain chain)
066            throws IOException, ServletException {
067        final HttpServletRequest req = (HttpServletRequest) request;
068        final String username = getUsername(req);
069        // Validate the extracted credentials
070        Set<String> containerRoles = Collections.emptySet();
071        if (FEDORA_ADMIN_USER.equals(username)) {
072            containerRoles = Collections.singleton("fedoraAdmin");
073            log.info("ADMIN AUTHENTICATED");
074        } else {
075            containerRoles = Collections.singleton("fedoraUser");
076            log.info("USER AUTHENTICATED");
077        }
078        final ServletRequest proxy = proxy(req, username, containerRoles);
079        chain.doFilter(proxy, response);
080    }
081
082    /**
083     * @param request
084     * @param username
085     * @param containerRoles
086     * @return
087     */
088    private ServletRequest proxy(final HttpServletRequest request,
089            final String username, final Set<String> containerRoles) {
090        final Principal user = new GrizzlyPrincipal(username);
091        final HttpServletRequest result =
092                (HttpServletRequest) newProxyInstance(request.getClass()
093                        .getClassLoader(),
094                        new Class[] {HttpServletRequest.class},
095                        new InvocationHandler() {
096
097                            @Override
098                            public Object invoke(final Object proxy,
099                                    final Method method, final Object[] args)
100                                    throws Throwable {
101                                if (method.getName().equals("isUserInRole")) {
102                                    final String role = (String) args[0];
103                                    return containerRoles.contains(role);
104                                } else if (method.getName().equals(
105                                        "getUserPrincipal")) {
106                                    return user;
107                                } else if (method.getName().equals(
108                                        "getRemoteUser")) {
109                                    return username;
110                                }
111                                return method.invoke(request, args);
112                            }
113                        });
114        return result;
115    }
116
117    private String getUsername(final HttpServletRequest request) {
118        // Extract authentication credentials
119        String authentication =
120                request.getHeader(ContainerRequest.AUTHORIZATION);
121        if (authentication == null) {
122            return null;
123        }
124        if (!authentication.startsWith("Basic ")) {
125            return null;
126        }
127        authentication = authentication.substring("Basic ".length());
128        final String[] values =
129                new String(Base64.base64Decode(authentication)).split(":");
130        if (values.length < 2) {
131            throw new WebApplicationException(400);
132            // "Invalid syntax for username and password"
133        }
134        final String username = values[0];
135        final String password = values[1];
136        if ((username == null) || (password == null)) {
137            return null;
138        }
139        return username;
140    }
141
142    /*
143     * (non-Javadoc)
144     * @see javax.servlet.Filter#init(javax.servlet.FilterConfig)
145     */
146    @Override
147    public void init(final FilterConfig filterConfig) throws ServletException {
148    }
149
150    /*
151     * (non-Javadoc)
152     * @see javax.servlet.Filter#destroy()
153     */
154    @Override
155    public void destroy() {
156    }
157}
