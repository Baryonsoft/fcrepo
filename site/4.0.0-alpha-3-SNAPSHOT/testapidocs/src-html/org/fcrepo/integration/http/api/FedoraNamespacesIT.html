039    @Test
040    public void testGet() throws Exception {
041
042        HttpGet get = new HttpGet(serverAddress + "fcr:namespaces");
043        get.addHeader("Accept", "application/n-triples");
044        HttpResponse response = execute(get);
045        int status = response.getStatusLine().getStatusCode();
046        assertEquals(200, status);
047
048        final GraphStore graphStore =
049                TestHelpers.parseTriples(response.getEntity().getContent());
050
051        logger.debug("Got store {}", graphStore);
052        assertTrue("expected to find nt property in response", graphStore
053                .contains(Node.ANY, ResourceFactory.createResource(
054                        "http://www.jcp.org/jcr/nt/1.0").asNode(),
055                        RdfLexicon.HAS_NAMESPACE_PREFIX.asNode(),
056                        ResourceFactory.createPlainLiteral("nt").asNode()));
057    }

167    @Test
168    public void testDeleteNamespace() throws Exception {
169        HttpPost post = new HttpPost(serverAddress + "fcr:namespaces");
170        BasicHttpEntity entity = new BasicHttpEntity();
171        entity.setContent(new ByteArrayInputStream(
172                ("INSERT { <http://example.com/namespace/abc> <" +
173                        RdfLexicon.HAS_NAMESPACE_PREFIX.toString() + "> \"abc\"} WHERE { }")
174                        .getBytes()));
175
176        post.setEntity(entity);
177        execute(post);
178
179        post = new HttpPost(serverAddress + "fcr:namespaces");
180        entity = new BasicHttpEntity();
181        entity.setContent(new ByteArrayInputStream(
182                ("DELETE { <http://example.com/namespace/abc> <" +
183                        RdfLexicon.HAS_NAMESPACE_PREFIX.toString() + "> \"abc\"} WHERE { }")
184                        .getBytes()));
185
186        post.setEntity(entity);
187        execute(post);
188
189        HttpGet get = new HttpGet(serverAddress + "fcr:namespaces");
190        get.addHeader("Accept", "application/n-triples");
191        HttpResponse response = execute(get);
192        int status = response.getStatusLine().getStatusCode();
193        assertEquals(200, status);
194
195        final GraphStore graphStore =
196                TestHelpers.parseTriples(response.getEntity().getContent());
197
198        logger.debug("Got store {}", graphStore);
199        assertFalse("should not find deleted property in response", graphStore
200                .contains(Node.ANY, ResourceFactory.createResource(
201                        "http://example.com/namespace/abc").asNode(),
202                        RdfLexicon.HAS_NAMESPACE_PREFIX.asNode(),
203                        ResourceFactory.createPlainLiteral("abc").asNode()));
204    }
