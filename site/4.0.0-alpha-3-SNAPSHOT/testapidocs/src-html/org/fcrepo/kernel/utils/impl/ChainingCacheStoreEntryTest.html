017
018import static org.junit.Assert.assertEquals;
019import static org.junit.Assert.assertNull;
020import static org.junit.Assert.fail;
021import static org.mockito.Mockito.mock;
022import static org.mockito.Mockito.times;
023import static org.mockito.Mockito.verify;
024import static org.mockito.Mockito.when;
025import static org.mockito.MockitoAnnotations.initMocks;
026
027import java.io.IOException;
028import java.io.InputStream;
029import java.util.LinkedHashMap;
030import java.util.Set;
031
032import org.fcrepo.kernel.utils.LowLevelCacheEntry;
033import org.infinispan.configuration.cache.AbstractStoreConfiguration;
034import org.infinispan.configuration.cache.CacheStoreConfiguration;
035import org.infinispan.configuration.cache.FileCacheStoreConfiguration;
036import org.infinispan.loaders.CacheStore;
037import org.infinispan.loaders.decorators.ChainingCacheStore;
038import org.infinispan.util.TypedProperties;
039import org.junit.Before;
040import org.junit.Test;
041import org.mockito.Mock;
042import org.modeshape.jcr.value.BinaryKey;
043import org.modeshape.jcr.value.binary.BinaryStoreException;
044
045public class ChainingCacheStoreEntryTest {
046
047    private static final String DUMMY_CACHE_NAME = "dummy-cache-name";
048
049    @Mock
050    private ChainingCacheStore mockStore;
051
052    @Mock
053    private BinaryKey mockKey;
054    
055    private ChainingCacheStoreEntry testObj;
056
057    @Before
058    public void setUp(){
059        initMocks(this);
060        testObj = new ChainingCacheStoreEntry(mockStore, DUMMY_CACHE_NAME, mockKey);
061    }
062
063    @Test
064    public void testNullAndUnimplementedMethodImpls()
065        throws BinaryStoreException, IOException{
066
067        try {
068            testObj.getInputStream();
069            fail("Unexpected completion of " +
070                 "ChainingCacheStoreEntry#getInputStream");
071        } catch (UnsupportedOperationException e) {
072            // expected
073        }
074        try {
075            InputStream mockStream = mock(InputStream.class);
076            testObj.storeValue(mockStream);
077            fail("Unexpected completion of " +
078                 "ChainingCacheStoreEntry#storeValue");
079        } catch (UnsupportedOperationException e) {
080            // expected
081        }
082        assertNull("Unexpected non-null return from ChainingCacheStoreEntry#" +
083                   "getExternalIdentifier",
084                testObj.getExternalIdentifier());
085    }
086
087    @Test
088    public void testChainedEntries() {
089        CacheStore mockChainedFileStore = mock(CacheStore.class);
090        CacheStore mockChainedAbstractStore = mock(CacheStore.class);
091        CacheStore mockUnknownStore = mock(CacheStore.class);
092        FileCacheStoreConfiguration mockFileConfig =
093            mock(FileCacheStoreConfiguration.class);
094        when(mockFileConfig.location()).thenReturn("/foo/bar");
095        AbstractStoreConfiguration mockAbstractConfig =
096                mock(AbstractStoreConfiguration.class);
097        CacheStoreConfiguration mockUnknownConfig =
098                mock(CacheStoreConfiguration.class);
099        TypedProperties mockProps = mock(TypedProperties.class);
100        TypedProperties mockNoProps = mock(TypedProperties.class);
101        when(mockProps.get("id")).thenReturn("dummy-abstract");
102        when(mockAbstractConfig.properties()).thenReturn(mockProps);
103        when(mockFileConfig.properties()).thenReturn(mockNoProps);
104        when(mockUnknownConfig.properties()).thenReturn(mockNoProps);
105        LinkedHashMap<CacheStore, CacheStoreConfiguration> mockStores =
106                new LinkedHashMap<CacheStore, CacheStoreConfiguration>(3);
107        mockStores.put(mockChainedFileStore, mockFileConfig);
108        mockStores.put(mockChainedAbstractStore, mockAbstractConfig);
109        mockStores.put(mockUnknownStore, mockUnknownConfig);
110        when(mockStore.getStores()).thenReturn(mockStores);
111        Set<LowLevelCacheEntry> actual = testObj.chainedEntries();
112        assertEquals(3, actual.size());
113        verify(mockProps).get("id");
114        verify(mockFileConfig).location();
115        verify(mockNoProps, times(0)).get("id");
116    }
117}
