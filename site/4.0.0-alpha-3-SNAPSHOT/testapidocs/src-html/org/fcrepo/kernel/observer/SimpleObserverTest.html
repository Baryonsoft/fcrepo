017package org.fcrepo.kernel.observer;
018
019import static com.google.common.collect.Iterables.filter;
020import static java.util.Arrays.asList;
021import static org.fcrepo.kernel.observer.SimpleObserver.EVENT_TYPES;
022import static org.mockito.Matchers.any;
023import static org.mockito.Matchers.eq;
024import static org.mockito.Mockito.mock;
025import static org.mockito.Mockito.never;
026import static org.mockito.Mockito.verify;
027import static org.mockito.Mockito.when;
028import static org.mockito.MockitoAnnotations.initMocks;
029import static org.powermock.api.mockito.PowerMockito.mockStatic;
030
031import java.lang.reflect.Field;
032import java.util.List;
033
034import javax.jcr.Node;
035import javax.jcr.Session;
036import javax.jcr.Workspace;
037import javax.jcr.observation.Event;
038import javax.jcr.observation.EventIterator;
039import javax.jcr.observation.ObservationManager;
040
041import org.junit.Before;
042import org.junit.Test;
043import org.junit.runner.RunWith;
044import org.mockito.Mock;
045import org.modeshape.jcr.api.Repository;
046import org.powermock.core.classloader.annotations.PowerMockIgnore;
047import org.powermock.core.classloader.annotations.PrepareForTest;
048import org.powermock.modules.junit4.PowerMockRunner;
049
050import com.google.common.collect.Iterables;
051import com.google.common.eventbus.EventBus;
052
053@RunWith(PowerMockRunner.class)
054@PowerMockIgnore({"org.slf4j.*", "javax.xml.parsers.*", "org.apache.xerces.*"})
055@PrepareForTest({com.google.common.collect.Iterables.class})
056public class SimpleObserverTest {
057
058    private SimpleObserver testObj;
059
060    ObservationManager mockOM;
061
062    @Mock
063    Repository mockRepository;
064
065    @Mock
066    Session mockSession;
067
068    @Mock
069    Workspace mockWS;
070
071    @Mock
072    EventBus mockBus;
073
074    @Mock
075    EventFilter mockFilter;
076
077    @Before
078    public void setUp() throws Exception {
079        initMocks(this);
080        testObj = new SimpleObserver();
081    }
082
083    @Test
084    public void testBuildListener() throws Exception {
085        mockOM = mock(ObservationManager.class);
086        when(mockWS.getObservationManager()).thenReturn(mockOM);
087        when(mockSession.getWorkspace()).thenReturn(mockWS);
088        when(mockRepository.login()).thenReturn(mockSession);
089        setField("repository", testObj, mockRepository);
090        testObj.buildListener();
091        verify(mockOM).addEventListener(testObj, EVENT_TYPES, "/", true, null,
092                null, false);
093    }
094
095    @SuppressWarnings("unchecked")
096    @Test
097    public void testOnEvent() throws Exception {
098        setField("eventBus", testObj, mockBus);
099        setField("eventFilter", testObj, mockFilter);
100        setField("session", testObj, mockSession);
101        final Event mockEvent = mock(Event.class);
102        when(mockEvent.getPath()).thenReturn("/foo/bar");
103        final Node mockNode = mock(Node.class);
104        when(mockNode.isNode()).thenReturn(true);
105        when(mockSession.getItem(any(String.class))).thenReturn(mockNode);
106        final EventIterator mockEvents = mock(EventIterator.class);
107        final List<Event> iterable = asList(new Event[] {mockEvent});
108        mockStatic(Iterables.class);
109        when(filter(any(Iterable.class), eq(mockFilter))).thenReturn(iterable);
110        testObj.onEvent(mockEvents);
111        verify(mockBus).post(any(Event.class));
112    }
113
114    @SuppressWarnings("unchecked")
115    @Test
116    public void testOnEventAllFiltered() throws Exception {
117        setField("eventBus", testObj, mockBus);
118        setField("eventFilter", testObj, mockFilter);
119        final EventIterator mockEvents = mock(EventIterator.class);
120        final List<Event> iterable = asList(new Event[0]);
121        mockStatic(Iterables.class);
122        when(filter(any(Iterable.class), eq(mockFilter))).thenReturn(iterable);
123        testObj.onEvent(mockEvents);
124        verify(mockBus, never()).post(any(Event.class));
125    }
126
127    private static void setField(final String name, final SimpleObserver obj,
128            final Object val) throws Exception {
129        final Field field = SimpleObserver.class.getDeclaredField(name);
130        field.setAccessible(true);
131        field.set(obj, val);
132    }
133
134}
