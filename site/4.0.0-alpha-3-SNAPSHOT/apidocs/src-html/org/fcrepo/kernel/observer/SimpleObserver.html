016package org.fcrepo.kernel.observer;
017
018import static com.codahale.metrics.MetricRegistry.name;
019import static com.google.common.base.Throwables.propagate;
020import static com.google.common.collect.Iterables.filter;
021import static javax.jcr.observation.Event.NODE_ADDED;
022import static javax.jcr.observation.Event.NODE_MOVED;
023import static javax.jcr.observation.Event.NODE_REMOVED;
024import static javax.jcr.observation.Event.PROPERTY_ADDED;
025import static javax.jcr.observation.Event.PROPERTY_CHANGED;
026import static javax.jcr.observation.Event.PROPERTY_REMOVED;
027import static org.fcrepo.metrics.RegistryService.getMetrics;
028import static org.slf4j.LoggerFactory.getLogger;
029
030import java.util.HashSet;
031import java.util.Set;
032
033import javax.annotation.PostConstruct;
034import javax.inject.Inject;
035import javax.jcr.Item;
036import javax.jcr.Node;
037import javax.jcr.PathNotFoundException;
038import javax.jcr.RepositoryException;
039import javax.jcr.Session;
040import javax.jcr.observation.Event;
041import javax.jcr.observation.EventListener;
042
043import org.fcrepo.kernel.utils.EventIterator;
044import org.modeshape.jcr.api.Repository;
045import org.slf4j.Logger;
046
047import com.codahale.metrics.Counter;
048import com.google.common.eventbus.EventBus;
049
050
051/**
052 * Simple JCR EventListener that filters JCR Events through a Fedora
053 * EventFilter and puts the resulting stream onto the internal
054 * Fedora EventBus as a stream of FedoraEvents.
055 *
056 * @author eddies
057 * @date Feb 7, 2013
058 */
059public class SimpleObserver implements EventListener {
060
061    private static final Logger LOGGER = getLogger(SimpleObserver.class);
062
063    /**
064     * A simple counter of events that pass through this observer
065     */
066    static final Counter EVENT_COUNTER = getMetrics().counter(
067            name(
068            SimpleObserver.class, "onEvent"));
069
070    static final Integer EVENT_TYPES = NODE_ADDED + NODE_REMOVED + NODE_MOVED +
071            PROPERTY_ADDED + PROPERTY_CHANGED + PROPERTY_REMOVED;
072
073    @Inject
074    private Repository repository;
075
076    @Inject
077    private EventBus eventBus;
078
079    @Inject
080    private EventFilter eventFilter;
081
082    private Session session;
083
084    /**
085     * Register this observer with the JCR event listeners
086     * @throws RepositoryException
087     */
088    @PostConstruct
089    public void buildListener() throws RepositoryException {
090        session = repository.login();
091        session.getWorkspace().getObservationManager().addEventListener(this,
092                EVENT_TYPES, "/", true, null, null, false);
093        session.save();
094    }
095
096    /**
097     * Filter JCR events and transform them into our own FedoraEvents.
098     *
099     * @param events
100     */
101    @Override
102    public void onEvent(final javax.jcr.observation.EventIterator events) {
103        // keep track of nodes that trigger events to prevent duplicates
104        final Set<Node> posted = new HashSet<Node>();
105
106        // post non-duplicate events approved by the filter
107        for (final Event e : filter(new EventIterator(events), eventFilter)) {
108            try {
109                final Item item = session.getItem(e.getPath());
110                Node n = null;
111                if ( item.isNode() ) {
112                    n = (Node)item;
113                } else {
114                    n = item.getParent();
115                }
116                if ( n != null && !posted.contains(n) ) {
117                    EVENT_COUNTER.inc();
118                    LOGGER.debug("Putting event: " + e.toString()
119                        + " on the bus.");
120                    eventBus.post(new FedoraEvent(e));
121                    posted.add(n);
122                } else {
123                    LOGGER.debug("Skipping: " + e);
124                }
125            } catch (final PathNotFoundException ex) {
126                // we can ignore these
127                LOGGER.trace("Not a node in the Fedora workspace: " + e);
128            } catch ( RepositoryException ex ) {
129                throw propagate(ex);
130            }
131        }
132    }
133
134}
