017package org.fcrepo.kernel.services;
018
019import static org.modeshape.jcr.api.JcrConstants.JCR_CONTENT;
020import static org.modeshape.jcr.api.JcrConstants.JCR_DATA;
021import static org.modeshape.jcr.api.JcrConstants.NT_FILE;
022
023import java.net.URI;
024
025import javax.jcr.Node;
026import javax.jcr.NodeIterator;
027import javax.jcr.Property;
028import javax.jcr.PropertyIterator;
029import javax.jcr.RepositoryException;
030import javax.jcr.Value;
031
032import org.fcrepo.kernel.services.functions.CheckCacheEntryFixity;
033import org.fcrepo.kernel.utils.FixityResult;
034import org.fcrepo.kernel.utils.LowLevelCacheEntry;
035import org.infinispan.distexec.DefaultExecutorService;
036import org.infinispan.distexec.DistributedExecutorService;
037import org.modeshape.jcr.value.binary.infinispan.InfinispanBinaryStore;
038import org.springframework.stereotype.Component;
039
040import com.google.common.base.Function;
041
042/**
043 * Uncategorized helper methods
044 */
045@Component
046public abstract class ServiceHelpers {
047
048    /**
049     * Get the total size of a Node's properties
050     * 
051     * @param node
052     * @return size in bytes
053     * @throws RepositoryException
054     */
055    public static Long getNodePropertySize(final Node node)
056        throws RepositoryException {
057        Long size = 0L;
058        for (final PropertyIterator i = node.getProperties(); i.hasNext();) {
059            final Property p = i.nextProperty();
060            if (p.isMultiple()) {
061                for (final Value v : p.getValues()) {
062                    size += v.getBinary().getSize();
063                }
064            } else {
065                size += p.getBinary().getSize();
066            }
067        }
068        return size;
069    }
070
071    /**
072     * @param obj
073     * @return object size in bytes
074     * @throws RepositoryException
075     */
076    public static Long getObjectSize(final Node obj) throws RepositoryException {
077        return getNodePropertySize(obj) + getObjectDSSize(obj);
078    }
079
080    /**
081     * @param obj
082     * @return object's datastreams' total size in bytes
083     * @throws RepositoryException
084     */
085    private static Long getObjectDSSize(final Node obj)
086        throws RepositoryException {
087        Long size = 0L;
088        for (final NodeIterator i = obj.getNodes(); i.hasNext();) {
089            final Node node = i.nextNode();
090            if (node.isNodeType(NT_FILE)) {
091                size += getDatastreamSize(node);
092            }
093        }
094        return size;
095    }
096
097    /**
098     * Get the size of a datastream by calculating the size of the properties
099     * and the binary properties
100     * 
101     * @param ds
102     * @return
103     * @throws RepositoryException
104     */
105    public static Long getDatastreamSize(final Node ds)
106        throws RepositoryException {
107        return getNodePropertySize(ds) + getContentSize(ds);
108    }
109
110    /**
111     * Get the size of the JCR content binary property
112     * 
113     * @param ds
114     * @return
115     * @throws RepositoryException
116     */
117    public static Long getContentSize(final Node ds) throws RepositoryException {
118        long size = 0L;
119        if (ds.hasNode(JCR_CONTENT)) {
120            final Node contentNode = ds.getNode(JCR_CONTENT);
121
122            if (contentNode.hasProperty(JCR_DATA)) {
123                size =
124                        ds.getNode(JCR_CONTENT).getProperty(JCR_DATA)
125                                .getBinary().getSize();
126            }
127        }
128
129        return size;
130    }
131
132    /**
133     * A static factory function to insulate services from the details of
134     * building a DistributedExecutorService
135     * 
136     * @param cache
137     * @return
138     */
139    public static DistributedExecutorService getClusterExecutor(
140            final InfinispanBinaryStore cacheStore) {
141        // Watch out! This is trying to pluck out the blob cache store. This
142        // works as long as
143        // modeshape continues to be ordered..
144        return new DefaultExecutorService(cacheStore.getCaches().get(1));
145    }
146
147    /**
148     * Get the fixity function to map a low-level cache entry to its fixity
149     * result
150     * 
151     * @param dsChecksum
152     * @param dsSize
153     * @return
154     */
155    public static
156    Function<LowLevelCacheEntry, FixityResult>
157    getCheckCacheFixityFunction(final URI dsChecksum, final long dsSize) {
158        return new CheckCacheEntryFixity(dsChecksum, dsSize);
159    }
160
161}
