233    @Test
234    public void testAddRdfType() throws RepositoryException {
235
236        final Resource resource = ResourceFactory.createResource();
237        when(mockSubjects.isFedoraGraphSubject(resource)).thenReturn(true);
238        when(mockSubjects.getNodeFromGraphSubject(resource))
239            .thenReturn(mockSubjectNode);
240
241        when(mockSubjectNode.getSession()).thenReturn(mockSession);
242        when(mockSession.getWorkspace()).thenReturn(mockWorkspace);
243        when(mockWorkspace.getNodeTypeManager()).thenReturn(mockNodeTypeManager);
244        when(mockNodeTypeManager.hasNodeType("fedora:object")).thenReturn(true);
245
246        when(mockSession.getNamespacePrefix(RdfLexicon.RESTAPI_NAMESPACE))
247                .thenReturn("fedora");
248        final Model model = ModelFactory.createDefaultModel();
249        model.add(resource, RDF.type,
250                model.createResource(RdfLexicon.RESTAPI_NAMESPACE+"object"));
251        when(mockSubjectNode.canAddMixin("fedora:object")).thenReturn(true);
252        testObj.addedStatements(model);
253        verify(mockSubjectNode).addMixin("fedora:object");
254    }
