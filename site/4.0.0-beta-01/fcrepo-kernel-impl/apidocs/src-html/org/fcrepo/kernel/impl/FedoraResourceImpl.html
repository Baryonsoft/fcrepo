099    /**
100     * Create or find a FedoraObject at the given path
101     * @param session the JCR session to use to retrieve the object
102     * @param path the absolute path to the object
103     * @throws RepositoryException
104     */
105    public FedoraResourceImpl(final Session session, final String path,
106        final String nodeType) throws RepositoryException {
107        this();
108        initializeNewResourceProperties(session, path, nodeType);
109    }
110
111    private void initializeNewResourceProperties(final Session session,
112                                                 final String path,
113                                                 final String nodeType) throws RepositoryException {
114        this.node = findOrCreateNode(
115                session, path, NT_FOLDER, nodeType);
116
117        if (node.isNew()) {
118
119            if (!isFedoraResource.apply(node) && !isFrozen.apply(node)) {
120                node.addMixin(FEDORA_RESOURCE);
121            }
122
123            node.setProperty(JCR_LASTMODIFIED, Calendar.getInstance());
124        }
125    }

151    /* (non-Javadoc)
152     * @see org.fcrepo.kernel.FedoraResource#getCreatedDate()
153     */
154    @Override
155    public Date getCreatedDate() throws RepositoryException {
156        if (node.hasProperty(JCR_CREATED)) {
157            return new Date(node.getProperty(JCR_CREATED).getDate()
158                            .getTimeInMillis());
159        }
160        LOGGER.debug("Node {} does not have a createdDate", node);
161        return null;
162    }
163
164    /* (non-Javadoc)
165     * @see org.fcrepo.kernel.FedoraResource#getLastModifiedDate()
166     */
167    @Override
168    public Date getLastModifiedDate() throws RepositoryException {
169        if (node.hasProperty(JCR_LASTMODIFIED)) {
170            return new Date(node.getProperty(JCR_LASTMODIFIED).getDate()
171                            .getTimeInMillis());
172        }
173        LOGGER.debug(
174                    "Could not get last modified date property for node {}",
175                    node);
176
177        final Date createdDate = getCreatedDate();
178
179        if (createdDate != null) {
180            LOGGER.trace(
181                        "Using created date for last modified date for node {}",
182                        node);
183            return createdDate;
184        }
185
186        return null;
187    }

200    @Override
201    public Collection<String> getModels() throws RepositoryException {
202        if (isFrozen.apply(node)) {
203            return newArrayList(
204                Iterators.transform(
205                    property2values.apply(node.getProperty(FROZEN_MIXIN_TYPES)), value2string));
206        }

210    /* (non-Javadoc)
211     * @see org.fcrepo.kernel.FedoraResource#updatePropertiesDataset
212     *     (org.fcrepo.kernel.rdf.IdentifierTranslator, java.lang.String)
213     */
214    @Override
215    public Dataset updatePropertiesDataset(final IdentifierTranslator subjects,
216            final String sparqlUpdateStatement) throws RepositoryException {
217        final Dataset dataset = getPropertiesDataset(subjects);
218        final UpdateRequest request =
219            create(sparqlUpdateStatement, dataset.getContext().getAsString(
220                    URI_SYMBOL));
221        dataset.getDefaultModel().setNsPrefixes(request.getPrefixMapping());
222        execute(request, dataset);
223        return dataset;
224    }
225
226    /* (non-Javadoc)
227     * @see org.fcrepo.kernel.FedoraResource#getPropertiesDataset(org.fcrepo.kernel.rdf.IdentifierTranslator, int, int)
228     */
229    @Override
230    public Dataset getPropertiesDataset(final IdentifierTranslator graphSubjects,
231        final int offset, final int limit)
232        throws RepositoryException {
233
234        final JcrRdfTools jcrRdfTools =
235            JcrRdfTools.withContext(graphSubjects, getNode().getSession());
236
237        final RdfStream propertiesStream =
238            jcrRdfTools.getJcrTriples(getNode());
239
240        final HierarchyRdfContextOptions serializationOptions = new HierarchyRdfContextOptions(limit, offset);
241
242        propertiesStream.concat(jcrRdfTools.getTreeTriples(getNode(), serializationOptions));
243
244        final Dataset dataset = DatasetFactory.create(propertiesStream.asModel());
245
246        final Model problemsModel = createDefaultModel();
247
248        final JcrPropertyStatementListener listener =
249            JcrPropertyStatementListener.getListener(graphSubjects, node
250                    .getSession(), problemsModel);
251
252        dataset.getDefaultModel().register(listener);
253
254        dataset.addNamedModel(PROBLEMS_MODEL_NAME, problemsModel);
255
256        dataset.getContext().set(URI_SYMBOL, graphSubjects.getSubject(getNode().getPath()));
257
258
259        return dataset;
260    }

351    /* (non-Javadoc)
352     * @see org.fcrepo.kernel.FedoraResource#replaceProperties
353     *     (org.fcrepo.kernel.rdf.IdentifierTranslator, com.hp.hpl.jena.rdf.model.Model)
354     */
355    @Override
356    public RdfStream replaceProperties(final IdentifierTranslator graphSubjects,
357        final Model inputModel) throws RepositoryException {
358        final RdfStream originalTriples = getTriples(graphSubjects);
359
360        final RdfStream replacementStream = RdfStream.fromModel(inputModel);
361
362        final Set<Triple> replacementTriples =
363            copyOf(replacementStream.iterator());
364
365        final DifferencingIterator<Triple> differencer =
366            new DifferencingIterator<>(replacementTriples, originalTriples);
367
368        new RdfRemover(graphSubjects, getNode().getSession(), replacementStream
369                .withThisContext(differencer)).consume();
370
371        new RdfAdder(graphSubjects, getNode().getSession(), replacementStream
372                .withThisContext(differencer.notCommon())).consume();
373
374        return replacementStream.withThisContext(Iterables.concat(differencer
375                .common(), differencer.notCommon()));
376    }

378    /* (non-Javadoc)
379     * @see org.fcrepo.kernel.FedoraResource#getEtagValue()
380     */
381    @Override
382    public String getEtagValue() throws RepositoryException {
383        final Date lastModifiedDate = getLastModifiedDate();
384
385        if (lastModifiedDate != null) {
386            return shaHex(node.getPath() + lastModifiedDate);
387        }
388        return "";
389    }
