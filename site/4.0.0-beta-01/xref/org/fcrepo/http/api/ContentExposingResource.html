41  /**
42   * An abstract class that sits between AbstractResource and any resource that
43   * wishes to share the routines for building responses containing binary
44   * content.
45   *
46   * @author Mike Durbin
47   */
48  public abstract class ContentExposingResource extends AbstractResource {
49  
50      protected static final int REQUESTED_RANGE_NOT_SATISFIABLE = 416;
51  
52      protected static final int PARTIAL_CONTENT = 206;
53  
54      private static long MAX_BUFFER_SIZE = 10240000;
55  
56      /**
57       * A helper method that does most of the work associated with processing a request
58       * for content (or a range of the content) into a Response
59       */
60      protected Response getDatastreamContentResponse(final Datastream ds, final String rangeValue, final Request request,
61                                                      final HttpServletResponse servletResponse,
62                                                      final HttpIdentifierTranslator subjects) throws
63              RepositoryException, IOException {
64  
65          // we include an explicit etag, because the default behavior is to use the JCR node's etag, not
66          // the jcr:content node digest.
67          checkCacheControlHeaders(request, servletResponse, ds);
68  
69          final CacheControl cc = new CacheControl();
70          cc.setMaxAge(0);
71          cc.setMustRevalidate(true);
72          Response.ResponseBuilder builder;
73  
74          if (rangeValue != null && rangeValue.startsWith("bytes")) {
75  
76              final Range range = Range.convert(rangeValue);
77  
78              final long contentSize = ds.getContentSize();
79  
80              final String endAsString;
81  
82              if (range.end() == -1) {
83                  endAsString = Long.toString(contentSize - 1);
84              } else {
85                  endAsString = Long.toString(range.end());
86              }
87  
88              final String contentRangeValue =
89                  String.format("bytes %s-%s/%s", range.start(),
90                                   endAsString, contentSize);
91  
92              if (range.end() > contentSize ||
93                      (range.end() == -1 && range.start() > contentSize)) {
94                  builder = status(REQUESTED_RANGE_NOT_SATISFIABLE)
95                          .header("Content-Range", contentRangeValue);
96              } else {
97                  final long maxBufferSize = MAX_BUFFER_SIZE; // 10MB max buffer size?
98                  final long rangeSize = range.size();
99                  final long rangeStart = range.start();
100                 final long remainingBytes = contentSize - rangeStart;
101                 final long bufSize = rangeSize < remainingBytes ? rangeSize : remainingBytes;
102 
103                 if (bufSize < maxBufferSize) {
104                     // Small size range content retrieval use javax.jcr.Binary to improve performance
105                     final byte[] buf = new byte[(int) bufSize];
106 
107                     final Binary binaryContent = ds.getBinaryContent();
108                     binaryContent.read(buf, rangeStart);
109                     binaryContent.dispose();
110 
111                     builder = status(PARTIAL_CONTENT).entity(buf)
112                             .header("Content-Range", contentRangeValue);
113                 } else {
114                     // For large range content retrieval, go with the InputStream class to balance
115                     // the memory usage, though this is a rare case in range content retrieval.
116                     final InputStream content = ds.getContent();
117                     final RangeRequestInputStream rangeInputStream =
118                             new RangeRequestInputStream(content, range.start(), range.size());
119 
120                         builder = status(PARTIAL_CONTENT).entity(rangeInputStream)
121                                 .header("Content-Range", contentRangeValue);
122                 }
123             }
124 
125         } else {
126             final InputStream content = ds.getContent();
127             builder = ok(content);
128         }
129 
130         final ContentDisposition contentDisposition = ContentDisposition.type("attachment")
131                 .fileName(ds.getFilename())
132                 .creationDate(ds.getCreatedDate())
133                 .modificationDate(ds.getLastModifiedDate())
134                 .size(ds.getContentSize())
135                 .build();
136 
137         return builder.type(ds.getMimeType()).header(
138                 "Link",
139                 "<" + subjects.getSubject(ds.getNode().getPath()) +
140                         ">;rel=\"describedby\"").header("Accept-Ranges",
141                 "bytes").cacheControl(cc)
142                 .header("Content-Disposition", contentDisposition)
143                 .build();
144     }

146     /**
147      * Evaluate the cache control headers for the request to see if it can be served from
148      * the cache.
149      *
150      * @param request
151      * @param servletResponse
152      * @param resource
153      * @throws javax.jcr.RepositoryException
154      */
155     protected static void checkCacheControlHeaders(final Request request,
156                                                    final HttpServletResponse servletResponse,
157                                                    final Datastream resource) throws RepositoryException {
158 
159         final EntityTag etag = new EntityTag(resource.getContentDigest().toString());
160         final Date date = resource.getLastModifiedDate();
161 
162         final Date roundedDate = new Date();
163         if (date != null) {
164             roundedDate.setTime(date.getTime() - date.getTime() % 1000);
165         }
166         final Response.ResponseBuilder builder =
167             request.evaluatePreconditions(roundedDate, etag);
168 
169         if (builder != null) {
170             final CacheControl cc = new CacheControl();
171             cc.setMaxAge(0);
172             cc.setMustRevalidate(true);
173             // here we are implicitly emitting a 304
174             // the exception is not an error, it's genuinely
175             // an exceptional condition
176             throw new WebApplicationException(builder.cacheControl(cc)
177                                                   .lastModified(date).tag(etag).build());
178         }
179 
180         addCacheControlHeaders(servletResponse, resource);
181     }
182 
183     /**
184      * Add ETag and Last-Modified cache control headers to the response
185      * @param servletResponse
186      * @param resource
187      * @throws RepositoryException
188      */
189     protected static void addCacheControlHeaders(final HttpServletResponse servletResponse,
190                                                  final Datastream resource) throws RepositoryException {
191 
192         final EntityTag etag = new EntityTag(resource.getContentDigest().toString());
193         final Date date = resource.getLastModifiedDate();
194 
195         if (!etag.getValue().isEmpty()) {
196             servletResponse.addHeader("ETag", etag.toString());
197         }
198 
199         if (date != null) {
200             servletResponse.addDateHeader("Last-Modified", date.getTime());
201         }
202     }

204     /**
205      * Setter that set the max buffer size for range content retrieval,
206      * which could help for unit testing.
207      * @param maxBufferSize
208      */
209     public static void setMaxBufferSize(final long maxBufferSize) {
210         MAX_BUFFER_SIZE = maxBufferSize;
211     }
