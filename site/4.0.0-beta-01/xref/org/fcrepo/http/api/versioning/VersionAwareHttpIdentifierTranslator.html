40  /**
41   * An extension of HttpIdentifierTranslator that includes mapping from JCR frozen nodes
42   * that represent versions of versionable nodes to a URI pattern that presents
43   * them as children of the original node.  The URL patterns are as follows:
44   *
45   * If node "node" is versionable, and has the unversionable subgraph
46   * "path/to/versioned/part" and a version snapshot exists with the label
47   * "label", the path to access the frozen nodes
48   * is:
49   * " ../node/fcr:versions/label/path/to/versioned/part".
50   *
51   * The "label" referred to here is either the cannonical identifier assigned
52   * to the version or the user label optionally assinged at version creation
53   * time.  When this class translates JCR nodes to subjects, it will always
54   * return that cannonical identifier, but when mapping from graph subjects to
55   * JCR nodes, either the user label or cannonical ID are supported.
56   *
57   * @author Mike Durbin
58   * @author ajs6f
59   */
60  public class VersionAwareHttpIdentifierTranslator extends HttpIdentifierTranslator {
61  
62      private static final Logger LOGGER = getLogger(VersionAwareHttpIdentifierTranslator.class);
63  
64      private Session internalSession;
65  
66      /**
67       * Creates a GraphSubject implementation that has all of the functionality
68       * of HttpIdentifierTranslator with the additional mapping versioned frozen nodes
69       * to URIs that appear in the hierarchy of the original node.
70       *
71       * This mapping can be heavyweight in that it has to do version lookups and
72       * other expensive operations for nodes that appear to be historic versions.
73       *
74       * @param session the current session (may contain transaction information)
75       * @param internalSession a session with read access to all content
76       * @param relativeTo a class whose path annotation will be used as the
77       *                   base for all relative paths
78       * @param uris a UriInfo implementation with information about the request
79       *             URI.
80       */
81      public VersionAwareHttpIdentifierTranslator(final Session session,
82          final Session internalSession, final Class<?> relativeTo, final UriInfo uris) {
83          super(session, relativeTo, uris);
84          this.internalSession = internalSession;
85      }
86  
87      @Override
88      public String getPathFromGraphSubject(@NotNull final String subjectUri) throws RepositoryException {
89          if (subjectUri.matches("^.*\\Q" + FCR_VERSIONS + "/\\E.+$")) {
90              final Node node = getNodeFromGraphSubjectForVersionNode(subjectUri);
91              if (node == null) {
92                  return null;
93              }
94              return node.getPath();
95          }
96          return super.getPathFromGraphSubject(subjectUri);
97      }
98  
99      @Override
100     public Resource getSubject(final String absPath) throws RepositoryException {
101         if (absPath.contains("jcr:versionStorage")) {
102             final Node probableFrozenNode = internalSession.getNode(absPath);
103             if (probableFrozenNode.getPrimaryNodeType().getName().equals("nt:frozenNode")) {
104                 final URI result = uriBuilder.buildFromMap(getPathMapForVersionNode(probableFrozenNode));
105                 LOGGER.debug("Translated path {} into RDF subject {}", absPath, result);
106                 return createResource(result.toString());
107             }
108             LOGGER.debug("{} was not a frozen node... no version-specific translation required", absPath);
109         }
110         return super.getSubject(absPath);
111     }

113     /**
114      * For frozen nodes (which represent version snapshots) we translate the
115      * paths from their frozen storage URI to
116      * "../node/fcr:versions/label/path/to/versioned/part".
117      * @see #getNodeFromGraphSubjectForVersionNode
118      * @param frozenNode an nt:frozenNode whose fedora-specific URI path is to be
119      *             retrieved
120      * @return a map with a single entry with the key "path" and the value
121      *         equal to the fedora path for the version node.
122      * @throws NullPointerException if session is null
123      */
124     private Map<String,String> getPathMapForVersionNode(final Node frozenNode)
125         throws RepositoryException {
126         Node versionableFrozenNode = frozenNode;
127         Node versionableNode = internalSession.getNodeByIdentifier(
128                 versionableFrozenNode.getProperty("jcr:frozenUuid").getString());
129         String pathWithinVersionable = "";
130         while (!versionableNode.isNodeType("mix:versionable")) {
131             LOGGER.debug("node {} is not versionable, checking parent...", versionableNode.getIdentifier());
132             pathWithinVersionable = "/" + getRelativePath(versionableNode,
133                     versionableNode.getParent()) + pathWithinVersionable;
134             versionableFrozenNode = versionableFrozenNode.getParent();
135             versionableNode = internalSession.getNodeByIdentifier(
136                     versionableFrozenNode.getProperty("jcr:frozenUuid").getString());
137         }
138 
139         pathWithinVersionable = versionableFrozenNode.getIdentifier()
140                 + (pathWithinVersionable.length() > 0 ? pathWithinVersionable : "");
141         final String pathToVersionable = versionableNode.getPath();
142         LOGGER.debug("frozen node found with id {}.", versionableFrozenNode.getIdentifier());
143         String path =  pathToVersionable + "/" + FCR_VERSIONS + "/" + pathWithinVersionable;
144         if (path.endsWith(JCR_CONTENT)) {
145             path = path.replace(JCR_CONTENT, FCR_CONTENT);
146         }
147         return singletonMap("path", path.startsWith("/") ? path.substring(1) : path);
148     }
149 
150     private static String getRelativePath(final Node node, final Node ancestor) throws RepositoryException {
151         return node.getPath().substring(ancestor.getPath().length() + 1);
152     }
153 
154     /**
155      * Gets the frozen node in the system for the version described as the
156      * passed subject.
157      * @see #getPathMapForVersionNode
158      * @param subjectUri a subject containing the fedora identifier as a URI
159      * @return the node that represents the version snapshot of the subject.
160      * @throws RepositoryException
161      * @throws NullPointerException if session is null
162      */
163     public Node getNodeFromGraphSubjectForVersionNode(final String subjectUri)
164         throws RepositoryException {
165 
166         /*
167          * Get the part of the URI after fcr:versions/ that represents the label
168          * optionally followed by a path into the versioned subgraph).
169          */
170         final String labelAndPath
171             = subjectUri.substring(subjectUri.indexOf(FCR_VERSIONS + "/") + FCR_VERSIONS.length() + 1);
172 
173         if (labelAndPath.contains("/")) {
174             /*
175              * The subjectUri references a node within a version snapshot
176              * (identified by label) of a subgraph.
177              */
178             final int firstSlash = labelAndPath.indexOf('/');
179             final String label = labelAndPath.substring(0, firstSlash);
180             final String pathIntoVersionedGraph = labelAndPath.substring(firstSlash + 1);
181             LOGGER.trace("subjectUri={}, label={}, pathIntoVersionedGraph={}", subjectUri, label,
182                     pathIntoVersionedGraph);
183             final Node versionedFrozenNode = getFrozenNodeByLabel(subjectUri, label);
184             if (versionedFrozenNode == null) {
185                 // there is no version with the given label!
186                 return null;
187             }
188             return internalSession.getNode(versionedFrozenNode.getPath() + "/"
189                     + pathIntoVersionedGraph.replace(FCR_CONTENT, JCR_CONTENT));
190         }
191         /**
192          * The subjectUri references a version of a verisonable node identified
193          * by label (the root of the versioned subgraph).
194          */
195         final String label = labelAndPath;
196         LOGGER.trace("subjectUri={}, label={}", subjectUri, label);
197         return getFrozenNodeByLabel(subjectUri, label);
198     }

200     /**
201      * A private helper method that tries to look up frozen node for the given subject
202      * by a label.  That label may either be one that was assigned at creation time
203      * (and is a version label in the JCR sense) or a system assigned identifier that
204      * was used for versions created without a label.  The current implementation
205      * uses the JCR UUID for the frozen node as the system-assigned label.
206      */
207     private Node getFrozenNodeByLabel(final String subjectUri, final String label) throws RepositoryException {
208         final String baseResourcePath
209             = getPathFromGraphSubject(subjectUri.substring(0, subjectUri.indexOf("/" + FCR_VERSIONS)));
210         try {
211             final Node frozenNode = internalSession.getNodeByIdentifier(label);
212 
213             /*
214              * We found a node whose identifier is the "label" for the version.  Now
215              * we must do due dilligence to make sure it's a frozen node representing
216              * a version of the subject node.
217              */
218             final Property p = frozenNode.getProperty("jcr:frozenUuid");
219             if (p != null) {
220                 final Node subjectNode = internalSession.getNode(baseResourcePath);
221                 if (p.getString().equals(subjectNode.getIdentifier())) {
222                     return frozenNode;
223                 }
224             }
225             /*
226              * Though a node with an id of the label was found, it wasn't the
227              * node we were looking for, so fall through and look for a labeled
228              * node.
229              */
230         } catch (final ItemNotFoundException ex) {
231             /*
232              * the label wasn't a uuid of a frozen node but
233              * instead possibly a version label.
234              */
235         }
236 
237         final VersionHistory hist =
238             internalSession.getWorkspace().getVersionManager().getVersionHistory(baseResourcePath);
239         if (hist.hasVersionLabel(label)) {
240             LOGGER.debug("Found version for {} by label {}.",  subjectUri, label);
241             return hist.getVersionByLabel(label).getFrozenNode();
242         }
243         LOGGER.warn("Unknown version {} with label or uuid {}!", subjectUri, label);
244         return null;
245     }
