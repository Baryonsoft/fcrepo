064public class FedoraFileSystemConnectorTest {
065
066    private FedoraFileSystemConnector connector;
067
068    private static Path directoryPath;
069
070    private static File tmpFile;
071
072    @Mock
073    private NamespaceRegistry mockRegistry;
074
075    @Mock
076    private NodeTypeManager mockNodeTypeManager;
077
078    @Mock
079    private DocumentTranslator mockTranslator;
080
081    @Mock
082    private NameFactory mockNameFactory;
083
084    @Mock
085    private ValueFactories mockValueFactories;
086
087    @Mock
088    private ExtraPropertiesStore mockExtraPropertiesStore;
089
090    @Mock
091    private Property binaryProperty;
092
093    @Mock
094    private BinaryValue binaryValue;
095
096    private ExecutionContext mockContext = new ExecutionContext();
097
098    private static final Logger logger =
099        getLogger(FedoraFileSystemConnectorTest.class);
100
101    @BeforeClass
102    public static void beforeClass() throws IOException {
103        directoryPath = createTempDirectory("fedora-filesystemtest");
104        tmpFile =
105            createTempFile(directoryPath, "fedora-filesystemtestfile",
106                    "txt").toFile();
107        try (FileOutputStream outputStream = new FileOutputStream(tmpFile)) {
108            outputStream.write("hello".getBytes());
109        } catch (final IOException e) {
110            logger.error("Error creating: {} - {}", tmpFile.getAbsolutePath(),
111                    e.getMessage());
112        }
113    }
114
115    @AfterClass
116    public static void afterClass() {
117        try {
118            tmpFile.delete();
119        } catch (final Exception e) {
120            logger.error("Error deleting: " + tmpFile.getAbsolutePath()
121                    + " - " + e.getMessage());
122        }
123    }
124
125    @Before
126    public void setUp() throws Exception {
127        initMocks(this);
128
129        connector = new FedoraFileSystemConnector();
130        setField(connector, "directoryPath", directoryPath.toString());
131        setField(connector, "translator", mockTranslator);
132        setField(connector, "context", mockContext);
133        setField(connector, "extraPropertiesStore", mockExtraPropertiesStore);
134        setField(mockTranslator, "names", mockNameFactory);
135        setField(connector, "readonly", true);
136        connector.initialize(mockRegistry, mockNodeTypeManager);
137        mockContext.getNamespaceRegistry().register("fedora", "http://fedora.info/definitions/v4/repository#");
138        mockContext.getNamespaceRegistry().register("premis", "http://www.loc.gov/premis/rdf/v1#");
139    }
140
141    @Test(expected = RepositoryException.class)
142    public void testThatReadOnlyIsRequired() throws IOException, RepositoryException {
143        final FedoraFileSystemConnector c = new FedoraFileSystemConnector();
144        setField(c, "directoryPath", directoryPath.toString());
145        setField(c, "translator", mockTranslator);
146        setField(c, "context", mockContext);
147        setField(c, "extraPropertiesStore", mockExtraPropertiesStore);
148        setField(mockTranslator, "names", mockNameFactory);
149        c.initialize(mockRegistry, mockNodeTypeManager);
150    }
151
152    @Test
153    public void testGetDocumentByIdNull() throws Exception {
154        final Document doc = connector.getDocumentById(null);
155        assertNull(doc);
156    }
157
158    @Test
159    public void testGetDocumentByIdDatastream() throws Exception {
160        when(mockTranslator.getPrimaryTypeName(any(Document.class)))
161                .thenReturn(NT_FILE);
162        when(mockNameFactory.create(anyString())).thenReturn(
163                new BasicName("", tmpFile.getName()));
164
165        final Document doc = connector.getDocumentById("/" + tmpFile.getName());
166        assertNotNull(doc);
167    }
168
169    @Test
170    public void testGetDocumentByIdContent() throws Exception {
171        when(mockTranslator.getPrimaryTypeName(any(Document.class)))
172                .thenReturn(NT_RESOURCE);
173        when(mockNameFactory.create(anyString())).thenReturn(
174                new BasicName("", tmpFile.getName()));
175
176        when(binaryProperty.getFirstValue()).thenReturn(binaryValue);
177        when(mockTranslator.getProperty(any(Document.class), eq(JCR_DATA)))
178                .thenReturn(binaryProperty);
179
180        final Document doc = connector.getDocumentById("/" + tmpFile.getName());
181        assertNotNull(doc);
182    }
183
184    @Test
185    public void testSha1WhenContentDigestIsCached() {
186        when(mockTranslator.getPrimaryTypeName(any(Document.class)))
187                .thenReturn(NT_RESOURCE);
188        when(mockNameFactory.create(anyString())).thenReturn(new BasicName("", tmpFile.getName()));
189        when(binaryProperty.getFirstValue()).thenReturn(binaryValue);
190        when(mockTranslator.getProperty(any(Document.class), eq(JCR_DATA)))
191                .thenReturn(binaryProperty);
192
193        final String chksum = "aaf4c61ddcc5e8a2dabede0f3b482cd9aea9434d";
194
195        when(mockTranslator.getProperty(any(Document.class), eq(CONTENT_DIGEST)))
196                .thenReturn(new BasicSingleValueProperty(new BasicName("", CONTENT_DIGEST.toString()),
197                                     chksum));
198
199        final String sha1 = connector.sha1(tmpFile);
200
201        assertNotNull(sha1);
202        assert(sha1.contains(chksum));
203    }
204
205    @Test
206    public void testSha1ContentDigestIsNotCached() {
207        when(mockTranslator.getPrimaryTypeName(any(Document.class)))
208                .thenReturn(NT_RESOURCE);
209        when(mockNameFactory.create(anyString())).thenReturn(new BasicName("", tmpFile.getName()));
210        when(binaryProperty.getFirstValue()).thenReturn(binaryValue);
211        when(mockTranslator.getProperty(any(Document.class), eq(JCR_DATA)))
212                .thenReturn(binaryProperty);
213
214        final String chksum = "aaf4c61ddcc5e8a2dabede0f3b482cd9aea9434d";
215
216        when(mockTranslator.getProperty(any(Document.class), eq(CONTENT_DIGEST)))
217                .thenReturn(null);
218
219        final String sha1 = connector.sha1(tmpFile);
220
221        assertNotNull(sha1);
222        assert(sha1.contains(chksum));
223    }
