16  package org.fcrepo.integration.connector.file;

18  import org.fcrepo.kernel.FedoraResource;
19  import org.junit.Test;
20  
21  import javax.jcr.Node;
22  import javax.jcr.NodeIterator;
23  import javax.jcr.RepositoryException;
24  import javax.jcr.Session;
25  import java.io.File;
26  
27  import static org.junit.Assert.assertEquals;
28  import static org.junit.Assert.assertTrue;
29  
30  /**
31   * @author Mike Durbin
32   */
33  
34  public class ReadOnlyExternalPropertiesFedoraFileSystemConnectorIT extends AbstractFedoraFileSystemConnectorIT {
35  
36      @Override
37      protected String federationName() {
38          return "readonly-federated";
39      }
40  
41      @Override
42      protected String getFederationRoot() {
43          return getReadOnlyFederationRoot();
44      }
45  
46      protected String testFilePath() {
47          return "/" + federationName() + "/repo.xml";
48      }
49  
50      @Test
51      public void verifyThatPropertiesAreExternal() throws RepositoryException {
52          final Session session = repo.login();
53          final FedoraResource object = nodeService.getObject(session, testFilePath());
54          assertEquals("There should be exactly as many visible nodes as actual files (ie, no hidden sidecar files).",
55                  fileForNode(object.getNode()).getParentFile().list().length,
56                  getChildCount(object.getNode().getParent()));
57          final File propFile = propertyFileForNode(object.getNode());
58          assertTrue("There should be properties file (" + propFile + ") in the external directory.",
59                  propFile.exists());
60      }
61  
62      protected int getChildCount(final Node node) throws RepositoryException {
63          int childCount = 0;
64          final NodeIterator childrenIt = node.getNodes();
65          while (childrenIt.hasNext()) {
66              childCount ++;
67              childrenIt.next();
68          }
69          return childCount;
70      }
