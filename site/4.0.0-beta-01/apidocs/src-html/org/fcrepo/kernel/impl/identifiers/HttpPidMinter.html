016package org.fcrepo.kernel.impl.identifiers;
017
018import static org.slf4j.LoggerFactory.getLogger;
019import static org.apache.commons.lang.StringUtils.isBlank;
020import static com.google.common.base.Preconditions.checkArgument;
021
022import org.slf4j.Logger;
023import com.codahale.metrics.annotation.Timed;
024
025import java.io.ByteArrayInputStream;
026import java.io.IOException;
027import java.net.URI;
028
029import org.w3c.dom.Document;
030import javax.xml.parsers.DocumentBuilder;
031import javax.xml.parsers.DocumentBuilderFactory;
032import javax.xml.xpath.XPathException;
033import javax.xml.xpath.XPathExpression;
034import javax.xml.xpath.XPathFactory;
035
036import org.apache.http.HttpResponse;
037import org.apache.http.client.HttpClient;
038import org.apache.http.client.methods.HttpGet;
039import org.apache.http.client.methods.HttpPost;
040import org.apache.http.client.methods.HttpPut;
041import org.apache.http.client.methods.HttpUriRequest;
042import org.apache.http.impl.client.HttpClientBuilder;
043import org.apache.http.impl.conn.PoolingHttpClientConnectionManager;
044import org.apache.http.util.EntityUtils;
045
046import org.apache.http.auth.AuthScope;
047import org.apache.http.auth.UsernamePasswordCredentials;
048import org.apache.http.client.CredentialsProvider;
049import org.apache.http.impl.client.BasicCredentialsProvider;
050
051
052/**
053 * PID minter that uses an external REST service to mint PIDs.
054 *
055 * @author escowles
056 * @since 04/28/2014
057 */
058public class HttpPidMinter extends BasePidMinter {
059
060    private static final Logger log = getLogger(HttpPidMinter.class);
061    protected final String url;
062    protected final String method;
063    protected final String username;
064    protected final String password;
065    private final String regex;
066    private XPathExpression xpath;
067
068    protected HttpClient client;
069
070    /**
071     * Create a new HttpPidMinter.
072     * @param url The URL for the minter service.  This is the only required argument -- all
073     *    other parameters can be blank.
074     * @param method The HTTP method (POST, PUT or GET) used to generate a new PID (POST will
075     *    be used if the method is blank.
076     * @param username If not blank, use this username to connect to the minter service.
077     * @param password If not blank, use this password used to connect to the minter service.
078     * @param regex If not blank, use this regular expression used to remove unwanted text from the
079     *    minter service response.  For example, if the response text is "/foo/bar:baz" and the
080     *    desired identifier is "baz", then the regex would be ".*:".
081     * @param xpath If not blank, use this XPath expression used to extract the desired identifier
082     *    from an XML minter response.
083    **/
084    public HttpPidMinter( final String url, final String method, final String username,
085        final String password, final String regex, final String xpath ) {
086
087        checkArgument( !isBlank(url), "Minter URL must be specified!" );
088
089        this.url = url;
090        this.method = method;
091        this.username = username;
092        this.password = password;
093        this.regex = regex;
094        if ( xpath != null ) {
095            try {
096                this.xpath = XPathFactory.newInstance().newXPath().compile(xpath);
097            } catch ( XPathException ex ) {
098                log.warn("Error parsing xpath ({}): {}", xpath, ex );
099            }
100        }
101        this.client = buildClient();
102    }
103
104    /**
105     * Setup authentication in httpclient.
106    **/
107    protected HttpClient buildClient() {
108        HttpClientBuilder builder = HttpClientBuilder.create().useSystemProperties().setConnectionManager(
109            new PoolingHttpClientConnectionManager());
110        if (!isBlank(username) && !isBlank(password)) {
111            final URI uri = URI.create(url);
112            final CredentialsProvider credsProvider = new BasicCredentialsProvider();
113            credsProvider.setCredentials(new AuthScope(uri.getHost(), uri.getPort()),
114                new UsernamePasswordCredentials(username, password));
115            builder = builder.setDefaultCredentialsProvider(credsProvider);
116        }
117        return builder.build();
118    }
119
120    /**
121     * Instantiate a request object based on the method variable.
122    **/
123    private HttpUriRequest minterRequest() {
124        if ( method != null && method.equalsIgnoreCase("GET") ) {
125            return new HttpGet(url);
126        } else if ( method != null && method.equalsIgnoreCase("PUT") ) {
127            return new HttpPut(url);
128        } else {
129            return new HttpPost(url);
130        }
131    }
132
133    /**
134     * Remove unwanted text from the minter service response to produce the desired identifer.
135     * Override this method for processing more complex than a simple regex replacement.
136    **/
137    protected String responseToPid( final String responseText ) throws Exception {
138        log.debug("responseToPid({})", responseText);
139        if ( !isBlank(regex) ) {
140            return responseText.replaceFirst(regex,"");
141        } else if ( xpath != null ) {
142            return xpath( responseText, xpath );
143        } else {
144            return responseText;
145        }
146    }
147
148    /**
149     * Extract the desired identifier value from an XML response using XPath
150    **/
151    private static String xpath( final String xml, final XPathExpression xpath ) throws Exception {
152        final DocumentBuilder builder = DocumentBuilderFactory.newInstance().newDocumentBuilder();
153        final Document doc = builder.parse(new ByteArrayInputStream(xml.getBytes()));
154        return xpath.evaluate(doc);
155    }
156
157    /**
158     * Mint a unique identifier using an external HTTP API.
159     * @return The generated identifier.
160     */
161    @Timed
162    @Override
163    public String mintPid() {
164        try {
165            log.debug("mintPid()");
166            final HttpResponse resp = client.execute( minterRequest() );
167            return responseToPid( EntityUtils.toString(resp.getEntity()) );
168        } catch ( IOException ex ) {
169            log.warn("Error minting pid from {}: {}", url, ex);
170            throw new RuntimeException("Error minting pid", ex);
171        } catch ( Exception ex ) {
172            log.warn("Error processing minter response", ex);
173            throw new RuntimeException("Error processing minter response", ex);
174        }
175    }
176}
