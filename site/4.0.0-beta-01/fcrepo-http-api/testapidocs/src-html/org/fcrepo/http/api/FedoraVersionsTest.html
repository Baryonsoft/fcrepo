064public class FedoraVersionsTest {
065
066    private FedoraVersions testObj;
067
068    @Mock
069    private NodeService mockNodes;
070
071    @Mock
072    VersionService mockVersions;
073
074    @Mock
075    SessionFactory mockSessionFactory;
076
077    @Mock
078    private Node mockNode;
079
080    @Mock
081    private NodeType mockNodeType;
082
083    private Session mockSession;
084
085    @Mock
086    private FedoraResourceImpl mockResource;
087
088    private RdfStream mockRdfStream = new RdfStream();
089
090    @Mock
091    private Request mockRequest;
092
093    @Mock
094    private Variant mockVariant;
095
096    @Mock
097    private Dataset mockDataset;
098
099    @Before
100    public void setUp() throws Exception {
101        initMocks(this);
102        testObj = new FedoraVersions();
103        mockSession = mockSession(testObj);
104        setField(testObj, "nodeService", mockNodes);
105        setField(testObj, "uriInfo", getUriInfoImpl());
106        setField(testObj, "session", mockSession);
107        setField(testObj, "versionService", mockVersions);
108        setField(testObj, "sessionFactory", mockSessionFactory);
109        when(mockSessionFactory.getInternalSession()).thenReturn(mockSession);
110        when(mockNode.getPath()).thenReturn("/test/path");
111        when(mockResource.getNode()).thenReturn(mockNode);
112        when(mockNodeType.getName()).thenReturn("nt:folder");
113        when(mockNode.getPrimaryNodeType()).thenReturn(mockNodeType);
114    }
115
116    @Test
117    public void testGetVersionList() throws RepositoryException {
118        final String pid = "FedoraVersioningTest";
119        when(mockRequest.selectVariant(POSSIBLE_RDF_VARIANTS)).thenReturn(
120                mockVariant);
121        when(mockNodes.getObject(any(Session.class), anyString())).thenReturn(
122                mockResource);
123        when(mockResource.getVersionTriples(any(HttpIdentifierTranslator.class)))
124                .thenReturn(mockRdfStream);
125        when(mockVariant.getMediaType()).thenReturn(
126                new MediaType("text", "turtle"));
127        final RdfStream response =
128            testObj.getVersionList(createPathList(pid), mockRequest,
129                    getUriInfoImpl());
130        assertEquals("Got wrong RdfStream!", mockRdfStream, response);
131    }
132
133    @Test
134    public void testAddVersionLabel() throws RepositoryException {
135        final String pid = "FedoraVersioningTest";
136        final String versionLabel = "FedoraVersioningTest1/fcr:versions/v0.0.1";
137        when(mockNodes.getObject(any(Session.class), anyString())).thenReturn(
138                mockResource);
139
140        final Response response =
141            testObj.addVersion(createPathList(pid), versionLabel);
142        verify(mockResource).addVersionLabel(anyString());
143        verify(mockVersions).createVersion(any(Workspace.class),
144                Matchers.<Collection<String>>any());
145        assertNotNull(response);
146    }
147
148    @Test
149    public void testRevertToVersion() throws RepositoryException {
150        final String pid = UUID.randomUUID().toString();
151        final String versionLabel = UUID.randomUUID().toString();
152        when(mockNodes.getObject(any(Session.class), anyString())).thenReturn(
153                mockResource);
154        final Response response = testObj.revertToVersion(createPathList(pid), versionLabel);
155        verify(mockVersions).revertToVersion(testObj.session.getWorkspace(), "/" + pid, versionLabel);
156        assertNotNull(response);
157    }
158
159    @Test (expected = PathNotFoundException.class)
160    public void testRevertToVersionFailure() throws RepositoryException {
161        final String pid = UUID.randomUUID().toString();
162        final String versionLabel = UUID.randomUUID().toString();
163        when(mockNodes.getObject(any(Session.class), anyString())).thenReturn(
164                mockResource);
165        doThrow(PathNotFoundException.class)
166                .when(mockVersions).revertToVersion(any(Workspace.class), anyString(), anyString());
167        testObj.revertToVersion(createPathList(pid), versionLabel);
168    }
169
170    @Test
171    public void testRemoveVersion() throws RepositoryException {
172        final String pid = UUID.randomUUID().toString();
173        final String versionLabel = UUID.randomUUID().toString();
174        when(mockNodes.getObject(any(Session.class), anyString())).thenReturn(
175                mockResource);
176        final Response response = testObj.removeVersion(createPathList(pid), versionLabel);
177        verify(mockVersions).removeVersion(testObj.session.getWorkspace(), "/" + pid, versionLabel);
178        assertNotNull(response);
179    }
180
181    @Test (expected = PathNotFoundException.class)
182    public void testRemoveVersionFailure() throws RepositoryException {
183        final String pid = UUID.randomUUID().toString();
184        final String versionLabel = UUID.randomUUID().toString();
185        when(mockNodes.getObject(any(Session.class), anyString())).thenReturn(
186                mockResource);
187        doThrow(PathNotFoundException.class)
188                .when(mockVersions).removeVersion(any(Workspace.class), anyString(), anyString());
189        testObj.removeVersion(createPathList(pid), versionLabel);
190    }
