081    @Test
082    public void testGetGraphSubjectChildNode() throws RepositoryException {
083        mockVersion(UUID.randomUUID().toString());
084
085        final String uri = "http://localhost:8080/fcrepo/rest" + mockVersionableNode.getPath()
086                + "/fcr:versions/" + mockFrozenNode.getIdentifier() + "/fcr:content";
087
088        final String contentNodeId = UUID.randomUUID().toString();
089        final Node mockVersionChildNode = mock(Node.class);
090        when(mockSession.getNodeByIdentifier(contentNodeId)).thenReturn(mockVersionChildNode);
091        when(mockVersionChildNode.isNodeType("mix:versionable")).thenReturn(false);
092        final String mockVersionedChildNodePath = mockVersionableNode.getPath() + "/" + FCR_CONTENT;
093        when(mockVersionChildNode.getPath()).thenReturn(mockVersionedChildNodePath);
094        when(mockVersionChildNode.getParent()).thenReturn(mockVersionableNode);
095
096        final Node mockFrozenChildNode = mock(Node.class);
097        final String mockFrozenChildNodePath = mockFrozenNode.getPath() + "/" + JCR_CONTENT;
098        when(mockFrozenChildNode.getPath()).thenReturn(mockFrozenChildNodePath);
099        final NodeType mockFrozenNodeType = mock(NodeType.class);
100        when(mockFrozenNodeType.getName()).thenReturn("nt:frozenNode");
101        when(mockFrozenChildNode.getPrimaryNodeType()).thenReturn(mockFrozenNodeType);
102        when(mockSession.nodeExists(mockFrozenChildNodePath)).thenReturn(true);
103        when(mockSession.getNode(mockFrozenChildNodePath)).thenReturn(mockFrozenChildNode);
104        final Property mockProperty = mock(Property.class);
105        when(mockProperty.getString()).thenReturn(contentNodeId);
106        when(mockFrozenChildNode.getProperty("jcr:frozenUuid")).thenReturn(mockProperty);
107        when(mockFrozenChildNode.getPath()).thenReturn(mockFrozenChildNodePath);
108        when(mockWorkspace.getName()).thenReturn("default");
109        when(mockFrozenChildNode.getParent()).thenReturn(mockFrozenNode);
110
111        final Resource actual = testObj.getSubject(mockFrozenChildNodePath);
112        assertEquals(uri, actual.getURI());
113    }

141    public void testGetNodeFromGraphSubjectForVersionByLabel() throws PathNotFoundException,
142            RepositoryException {
143        mockVersion(UUID.randomUUID().toString());
144
145        final String label = UUID.randomUUID().toString();
146
147        when(mockSession.getNodeByIdentifier(label)).thenThrow(new ItemNotFoundException());
148        final Version mockVersion = mock(Version.class);

152        when(mockVersionHistory.getVersionByLabel(label)).thenReturn(mockVersion);
153        when(mockVersionManager.getVersionHistory(mockVersionableNode.getPath())).thenReturn(mockVersionHistory);
154
155        final String uri = "http://localhost:8080/fcrepo/rest" + mockVersionableNode.getPath()
156                + "/fcr:versions/" + label;
157        mockSubject(uri);
158
159        final String actual = testObj.getPathFromSubject(mockSubject);
160        assertEquals(mockFrozenNode.getPath(), actual);

184    private void mockSubject(final String uri) {
185        when(mockSubject.getURI()).thenReturn(uri);
186        when(mockSubject.isURIResource()).thenReturn(true);
187    }

189    /**
190     * Sets up the mocks so that mockVersionableNode represents a new
191     * versionable node and mockFrozenNode represents a historical version
192     * of that node.
193     */
194    private void mockVersion(final String nodePath) throws RepositoryException {
195        final String frozenNodeUUID = UUID.randomUUID().toString();
196        final String versionedNodeUUID = UUID.randomUUID().toString();
197
198        final String frozenPath = "/jcr:versionStorage/" + frozenNodeUUID;
199        final String versionedPath = "/" + nodePath;
200
201        when(mockSession.nodeExists(frozenPath)).thenReturn(true);
202        when(mockSession.nodeExists(versionedPath)).thenReturn(true);
203        when(mockSession.getNode(frozenPath)).thenReturn(mockFrozenNode);
204        when(mockSession.getNodeByIdentifier(frozenNodeUUID)).thenReturn(mockFrozenNode);
205        when(mockSession.getNode(versionedPath)).thenReturn(mockVersionableNode);
206        when(mockSession.getNodeByIdentifier(versionedNodeUUID)).thenReturn(mockVersionableNode);
207        when(mockVersionableNode.getIdentifier()).thenReturn(versionedNodeUUID);
208        when(mockFrozenNode.getIdentifier()).thenReturn(frozenNodeUUID);
209        when(mockVersionableNode.getPath()).thenReturn(versionedPath);
210
211        final NodeType mockFrozenNodeType = mock(NodeType.class);
212        when(mockFrozenNodeType.getName()).thenReturn("nt:frozenNode");
213        when(mockFrozenNode.getPrimaryNodeType()).thenReturn(mockFrozenNodeType);
214        when(mockFrozenNode.getPath()).thenReturn(frozenPath);
215        final Property mockProperty = mock(Property.class);
216        when(mockProperty.getString()).thenReturn(versionedNodeUUID);
217        when(mockFrozenNode.getProperty("jcr:frozenUuid")).thenReturn(mockProperty);
218
219        when(mockVersionableNode.isNodeType("mix:versionable")).thenReturn(true);
220
221    }
