091    @Test
092    public void testSearchRdf() throws Exception {
093        /* first post two objects which can be used for the search */
094        createObject("testobj");
095        createObject("testobj2");
096
097        /* and add a dc title to the object so the query returns a result */
098        setProperty("testobj", "http://purl.org/dc/elements/1.1/title",
099                "testobj");
100
101        final HttpGet method = new HttpGet(serverAddress + "fcr:search");
102        final URI uri =
103            new URIBuilder(method.getURI()).addParameter("q", "testobj")
104                    .addParameter("offset", "0").addParameter("limit", "1")
105                    .build();
106
107        method.setURI(uri);
108
109        final GraphStore graphStore =  getGraphStore(method);
110
111        logger.debug("Got search results graph: {}", graphStore);
112        assertTrue(graphStore.contains(ANY, createResource(
113                serverAddress + "fcr:search?q=testobj").asNode(),
114                SEARCH_HAS_TOTAL_RESULTS.asNode(), createTypedLiteral(1)
115                        .asNode()));
116
117        assertTrue(graphStore.contains(ANY, createResource(
118                serverAddress + "fcr:search?q=testobj&offset=0&limit=1")
119                .asNode(), PAGE_OF.asNode(), createResource(
120                serverAddress + "fcr:search?q=testobj").asNode()));
121
122        assertTrue(graphStore.contains(ANY, createResource(
123                serverAddress + "fcr:search?q=testobj&offset=0&limit=1")
124                .asNode(), SEARCH_OFFSET.asNode(), createTypedLiteral(0)
125                .asNode()));
126
127        assertTrue(graphStore.contains(ANY, createResource(
128                serverAddress + "fcr:search?q=testobj&offset=0&limit=1")
129                .asNode(), SEARCH_ITEMS_PER_PAGE.asNode(),
130                createTypedLiteral(1).asNode()));
131
132        assertTrue(graphStore.contains(ANY, createResource(
133                serverAddress + "fcr:search?q=testobj&offset=0&limit=1")
134                .asNode(), NEXT_PAGE.asNode(), nil.asNode()));
135    }

144    @Test
145    public void testResponseContentTypes() throws Exception {
146        final String responseTypes[] = {TURTLE, N3, N3_ALT2, RDF_XML, NTRIPLES, TEXT_PLAIN, TURTLE_X, JSON_LD};
147        for (final String type : responseTypes) {
148            final HttpGet method =
149                    new HttpGet(serverAddress + "fcr:search?q=testtype");
150
151            method.addHeader("Accept", type);
152            assertEquals(type, getContentType(method));
153        }
154    }
