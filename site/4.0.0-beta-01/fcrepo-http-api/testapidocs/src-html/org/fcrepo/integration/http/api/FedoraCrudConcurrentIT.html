034/**
035 * This "test" is a utility for collecting the timing of concurrent operations operations.
036 * It takes roughly 2 minutes to complete and should only be run if the timing metrics are wanted.
037 * In order to activate this utility, the following System Property must be set:
038 * <p/>
039 * mvn -DRUN_TEST_CRUD_CONCURRENT install
040 *
041 * @author lsitu
042 */
043public class FedoraCrudConcurrentIT extends AbstractResourceIT {
044
045    private static final String TEST_ACTIVATION_PROPERTY = "RUN_TEST_CRUD_CONCURRENT";
046
047    @Test
048    public void testConcurrentIngest() throws Exception {
049        setLogger();
050
051        if (System.getProperty(TEST_ACTIVATION_PROPERTY) == null) {
052            logger.info("Not running tests because system property not set: {}", TEST_ACTIVATION_PROPERTY);
053            return;
054        }
055
056        final int[] numThreadsToTest = {2, 4, 8, 16, 32};
057        logger.info("# Start CRUD concurrent performance testing...");
058        for (int i = 0; i < numThreadsToTest.length; i++) {
059            startCrudConcurrentPerformanceTest(numThreadsToTest[i]);
060        }
061    }
062
063    /**
064     * Test CRUD concurrent access performance:
065     * create/update/delete object, create/update/delete content file
066     *
067     * @param numThreads
068     * @throws Exception
069     */
070    private void startCrudConcurrentPerformanceTest(final int numThreads) throws Exception {
071        String pid = null;
072
073        // Tasks to run
074        final List<HttpRunner> tasks = new ArrayList<>();
075        final List<String> pids = new ArrayList<>();
076
077        // Create object
078        logger.info("# Starting " + numThreads + " concurrent threads to create object...");
079        for (int i = 0; i < numThreads; i++) {
080            pid = getRandomUniquePid();
081            pids.add(pid);
082            final String taskName = "Thread " + (i + 1) + " to create object " + pid;
083            final HttpRequestBase request = postObjMethod("/");
084            request.addHeader("Slug", pid);
085            final HttpRunner task = new HttpRunner(request, taskName);
086            task.setExpectedStatusCode(CREATED.getStatusCode());
087            tasks.add(task);
088        }
089        startThreads(tasks);
090        long totalResponseTime = getTotalResponseTime(numThreads, tasks);
091        logger.info("** Average response time for {} concurrent threads to CREATE object: {} ms",
092                    numThreads,
093                    totalResponseTime / numThreads);
094
095        tasks.clear();
096        // Update objects
097        logger.info("# Starting " + numThreads + " concurrent threads to update object...");
098        for (int i = 0; i < numThreads; i++) {
099            pid = pids.get(i);
100            final String taskName = "Thread " + (i + 1) + " to update object";
101            final HttpPatch request = patchObjMethod(pid);
102            request.addHeader("Content-Type", "application/sparql-update");
103            final String subjectUri = request.getURI().toString();
104            final BasicHttpEntity e = new BasicHttpEntity();
105            e.setContent(new ByteArrayInputStream(
106                    ("INSERT { <" + subjectUri + "> <http://purl.org/dc/elements/1.1/title> "
107                            + "\"Title: " + taskName + pid + "\" } WHERE {}"
108                    ).getBytes()));
109            request.setEntity(e);
110            final HttpRunner task = new HttpRunner(request, taskName);
111            task.setExpectedStatusCode(204);
112            tasks.add(task);
113        }
114        startThreads(tasks);
115        totalResponseTime = getTotalResponseTime(numThreads, tasks);
116        logger.info("** Average response time for {} concurrent threads to UPDATE object: {} ms",
117                    numThreads,
118                    totalResponseTime / numThreads);
119
120
121        tasks.clear();
122        // Ingest new content
123        logger.info("# Starting " + numThreads + " concurrent threads to inget content...");
124        for (int i = 0; i < numThreads; i++) {
125            pid = pids.get(i);
126            final String taskName = "Thread " + (i + 1) + " to ingest content file to object";
127            final HttpRequestBase request = postDSMethod(pid, "ds", "This is a content file: " + taskName + pid);
128            final HttpRunner task = new HttpRunner(request, taskName);
129            task.setExpectedStatusCode(CREATED.getStatusCode());
130            tasks.add(task);
131        }
132        startThreads(tasks);
133        totalResponseTime = getTotalResponseTime(numThreads, tasks);
134        logger.info("** Average response time for {} concurrent threads to INGEST content file: {} ms",
135                    numThreads,
136                    totalResponseTime / numThreads);
137
138
139        tasks.clear();
140        // Update content
141        logger.info("# Starting " + numThreads + " concurrent threads to update content...");
142        for (int i = 0; i < numThreads; i++) {
143            pid = pids.get(i);
144            final String taskName = "Thread " + (i + 1) + " to update content file in object";
145            final HttpRequestBase request = putDSMethod(pid,
146                                                        "ds",
147                                                        "This is an updated content file: " + taskName + pid);
148            final HttpRunner task = new HttpRunner(request, taskName);
149            task.setExpectedStatusCode(204);
150            tasks.add(task);
151        }
152        startThreads(tasks);
153        totalResponseTime = getTotalResponseTime(numThreads, tasks);
154        logger.info("** Average response time for {} concurrent threads to UPDATE content file: {} ms",
155                    numThreads,
156                    totalResponseTime / numThreads);
157
158
159        tasks.clear();
160        // Retrieve content
161        logger.info("# Starting " + numThreads + " concurrent threads to retrieve content...");
162        for (int i = 0; i < numThreads; i++) {
163            pid = pids.get(i);
164            final String taskName = "Thread " + (i + 1) + " to retrieve content file in object";
165            final HttpRequestBase request = getDSMethod(pid, "ds");
166            final HttpRunner task = new HttpRunner(request, taskName);
167            task.setExpectedStatusCode(200);
168            tasks.add(task);
169        }
170        startThreads(tasks);
171        totalResponseTime = getTotalResponseTime(numThreads, tasks);
172        logger.info("** Average response time for {} concurrent threads to RETRIEVE content file: {} ms",
173                    numThreads,
174                    totalResponseTime / numThreads);
175
176
177        tasks.clear();
178        // Delete content file
179        logger.info("# Starting " + numThreads + " concurrent threads to delete content file...");
180        for (int i = 0; i < numThreads; i++) {
181            pid = pids.get(i);
182            final String taskName = "Thread " + (i + 1) + " to delete content file in object";
183            final HttpRequestBase request = deleteObjMethod(pid + "/ds");
184            final HttpRunner task = new HttpRunner(request, taskName);
185            task.setExpectedStatusCode(204);
186            tasks.add(task);
187        }
188        startThreads(tasks);
189        totalResponseTime = getTotalResponseTime(numThreads, tasks);
190        logger.info("** Average response time for {} concurrent threads to DELETE content file: {} ms",
191                    numThreads,
192                    totalResponseTime / numThreads);
193
194
195        tasks.clear();
196        // Retrieve objects
197        logger.info("# Starting " + numThreads + " concurrent threads to retrieve object...");
198        for (int i = 0; i < numThreads; i++) {
199            pid = pids.get(i);
200            final String taskName = "Thread " + (i + 1) + " to retrieve object";
201            final HttpGet request = getObjMethod(pid);
202            final HttpRunner task = new HttpRunner(request, taskName);
203            task.setExpectedStatusCode(200);
204            tasks.add(task);
205        }
206        startThreads(tasks);
207        totalResponseTime = getTotalResponseTime(numThreads, tasks);
208        logger.info("** Average response time for {} concurrent threads to RETRIEVE object: {} ms",
209                    numThreads,
210                    totalResponseTime / numThreads);
211
212
213        tasks.clear();
214        // Delete objects
215        logger.info("# Starting " + numThreads + " concurrent threads to delete object...");
216        for (int i = 0; i < numThreads; i++) {
217            pid = pids.get(i);
218            final String taskName = "Thread " + (i + 1) + " to delete object";
219            final HttpRequestBase request = deleteObjMethod(pid);
220            final HttpRunner task = new HttpRunner(request, taskName);
221            task.setExpectedStatusCode(204);
222            tasks.add(task);
223        }
224        startThreads(tasks);
225        totalResponseTime = getTotalResponseTime(numThreads, tasks);
226        logger.info("** Average response time for {} concurrent threads to DELETE object: {} ms",
227                    numThreads,
228                    totalResponseTime / numThreads);
229
230    }
231
232    private long getTotalResponseTime(final int numThreads,
233                                      final List<HttpRunner> tasks) throws InterruptedException {
234        Thread.sleep(1000);
235        long totalResponseTime = 0;
236        for (int i = 0; i < numThreads; i++) {
237            totalResponseTime += tasks.get(i).responseTime;
238        }
239        return totalResponseTime;
240    }
241
242    private void startThreads(final List<HttpRunner> tasks) throws InterruptedException {
243        final int taskSize = tasks.size();
244        for (int i = 0; i < taskSize; i++) {
245
246            final Thread thread = new Thread(tasks.get(i));
247            thread.run();
248            thread.join();
249        }
250    }
251
252    /**
253     * Task to run http request for CRUD concurrent performance test.
254     *
255     * @author lsitu
256     */
257    class HttpRunner implements Runnable {
258
259        private HttpClient httpClient = null;
260
261        private HttpResponse response = null;
262
263        private HttpRequestBase request = null;
264
265        private String taskName = null;
266
267        private long responseTime = 0;
268
269        private int statusCode = 0;
270
271        private int expectedStatusCode = 0;
272
273        public HttpRunner(final HttpRequestBase request, final String taskName) {
274            this.taskName = taskName;
275            this.request = request;
276            // Use its own HttpClient instance to make sure each performance test
277            // won't affected by a single HttpClient instance with multiple connections.
278            httpClient = createClient();
279        }
280
281        @Override
282        public void run() {
283            try {
284                final long startTime = System.currentTimeMillis();
285                response = httpClient.execute(request);
286                final long endTime = System.currentTimeMillis();
287                responseTime = endTime - startTime;
288                statusCode = response.getStatusLine().getStatusCode();
289                logger.info("{} {} with status {} in {} ms.",
290                            taskName, request.getURI().toString(),
291                            statusCode, String.valueOf(responseTime));
292                assertEquals(taskName + " exited abnormally.", expectedStatusCode, statusCode);
293            } catch (IOException e) {
294                logger.error("Error {} {} got IOException: {}", taskName, request.getURI().toString(), e.getMessage());
295            } finally {
296                request.releaseConnection();
297            }
298        }
299
300        public HttpResponse getResponse() {
301            return response;
302        }
303
304
305        public HttpRequestBase getRequest() {
306            return request;
307        }
308
309
310        public int getStatusCode() {
311            return statusCode;
312        }
313
314
315        public void setExpectedStatusCode(final int expectedStatusCode) {
316            this.expectedStatusCode = expectedStatusCode;
317        }
318
319    }
