084    @Test
085    public void testCreateAndTimeoutTransaction() throws Exception {
086
087        /* create a short-lived tx */
088        final long testTimeout = min(500, REAP_INTERVAL / 2);
089        System.setProperty(TIMEOUT_SYSTEM_PROPERTY, Long.toString(testTimeout));
090
091        /* create a tx */
092        final String location = createTransaction();
093
094        final HttpGet getWithinTx = new HttpGet(location);
095        HttpResponse resp = execute(getWithinTx);
096        IOUtils.toString(resp.getEntity().getContent());
097        assertEquals(200, resp.getStatusLine().getStatusCode());
098        assertTrue(Iterators.any(Iterators.forArray(resp.getHeaders("Link")), new Predicate<Header>() {
099            @Override
100            public boolean apply(final Header input) {
101                return input.getValue().contains("<" + serverAddress + ">;rel=\"canonical\"");
102            }
103        }));
104
105        int statusCode = 0;
106
107        sleep(REAP_INTERVAL * 2);
108        final HttpGet getWithExpiredTx = new HttpGet(location);
109        resp = execute(getWithExpiredTx);
110        IOUtils.toString(resp.getEntity().getContent());
111        statusCode = resp.getStatusLine().getStatusCode();
112
113        try {
114            assertEquals("Transaction did not expire", 410, statusCode);
115        } finally {
116            System.setProperty(TIMEOUT_SYSTEM_PROPERTY, Long
117                    .toString(DEFAULT_TIMEOUT));
118            System.clearProperty("fcrepo4.tx.timeout");
119        }
120    }

282    /**
283     * Tests whether a Sparql update is visible within a transaction
284     * and if the update is made persistent along with the commit.
285     * @throws Exception
286     */
287    @Test
288    public void testIngestNewWithSparqlPatchWithinTransaction() throws Exception {
289        final String objectInTxCommit = randomUUID().toString();
290
291        /* create new tx */
292        final String txLocation = createTransaction();
293
294        client = createClient();
295        final HttpPost postNew =
296                new HttpPost(txLocation);
297        postNew.addHeader("Slug", objectInTxCommit);
298        HttpResponse resp = execute(postNew);
299        assertEquals(201, resp.getStatusLine().getStatusCode());
300        final String newObjectLocation = resp.getFirstHeader("Location").getValue();
301
302        /* update sparql */
303        final HttpPatch method = new HttpPatch(newObjectLocation);
304        method.addHeader("Content-Type", "application/sparql-update");
305        final BasicHttpEntity entity = new BasicHttpEntity();
306        final String title = "this is a new title";
307        entity.setContent(new ByteArrayInputStream(
308                ("INSERT { <> <http://purl.org/dc/elements/1.1/title> \"" + title + "\" } WHERE {}")
309                        .getBytes()));
310        method.setEntity(entity);
311        final HttpResponse responseFromPatch = client.execute(method);
312        final int status = responseFromPatch.getStatusLine().getStatusCode();
313        assertEquals("Didn't get a 204 status! Got status:\n" + status,
314                NO_CONTENT.getStatusCode(), status);
315
316        /* make sure the change was made within the tx */
317        final HttpGet httpGet = new HttpGet(newObjectLocation);
318        final GraphStore graphStore = getGraphStore(httpGet);
319        assertTrue("The sparql update did not succeed within a transaction",
320                graphStore.contains(ANY, createResource(newObjectLocation).asNode(),
321                DC_TITLE.asNode(), createPlainLiteral(title)
322                .asNode()));
323
324        /* commit */
325        client = createClient();
326        final HttpPost commitTx =
327                new HttpPost(txLocation + "/fcr:tx/fcr:commit");
328        resp = execute(commitTx);
329
330        assertEquals(204, resp.getStatusLine().getStatusCode());
331
332        /* it must exist after commit */
333        client = createClient();
334        final HttpGet getObjCommitted =
335                new HttpGet(serverAddress + objectInTxCommit);
336        final GraphStore graphStoreAfterCommit = getGraphStore(getObjCommitted);
337        assertTrue("The inserted triple does not exist after the transaction has committed",
338                graphStoreAfterCommit.contains(ANY, ANY,
339                DC_TITLE.asNode(), createPlainLiteral(title)
340                .asNode()));
341
342

358     * Tests that transactions cannot be hijacked
359     */
360    @Test
361    public void testTransactionHijackingNotPossible() throws Exception {
362
363        /* "fedoraAdmin" creates a transaction */
364        final HttpPost createTx = new HttpPost(serverAddress + "fcr:tx");
365        final HttpResponse response = executeWithBasicAuth(createTx, "fedoraAdmin", "fedoraAdmin");
366        assertEquals("Status should be 201 after creating a transaction with user fedoraAdmin",
367                201, response.getStatusLine().getStatusCode());
368        final String txLocation = response.getFirstHeader("Location").getValue();
369
370        /* "fedoraUser" puts to "fedoraAdmin"'s transaction and fails */
371        final HttpPut putFedoraUser = new HttpPut(txLocation);
372        final HttpResponse responseFedoraUser = executeWithBasicAuth(putFedoraUser, "fedoraUser", "fedoraUser");
373        assertEquals("Status should be 410 because putting on a transaction of a different user is not allowed",
374                410, responseFedoraUser.getStatusLine().getStatusCode());
375
376        /* anonymous user puts to "fedoraAdmin"'s transaction and fails */
377        final HttpPut putTxAnon = new HttpPut(txLocation);
378        final HttpResponse responseTxAnon = execute(putTxAnon);
379        assertEquals("Status should be 410 because putting on a transaction of a different user is not allowed",
380                410, responseTxAnon.getStatusLine().getStatusCode());
381
382        /* transaction is still intact and "fedoraAdmin" - the owner - can successfully put to it */
383        final String objectInTxCommit = randomUUID().toString();
384        final HttpPut putToExistingTx = new HttpPut(txLocation + "/" + objectInTxCommit);
385        final HttpResponse responseFromPutToTx = executeWithBasicAuth(putToExistingTx, "fedoraAdmin", "fedoraAdmin");
386        assertEquals("Status should be 201 after putting", 201, responseFromPutToTx.getStatusLine().getStatusCode());
387
388    }

422    /**
423     * Tests that transactions are treated as atomic with regards to nodes.
424     *
425     * A common use case for applications written against fedora is that an
426     * operation checks some property of a fedora object and acts on it
427     * accordingly.  In order for this to work in a multi-client or
428     * multi-threaded environment that comparison+action combination needs to
429     * be atomic.
430     *
431     * Imagine a scenario where we have one process that deletes all objects
432     * in the repository that don't have a "preserve" property set to the
433     * literal "true", and we have any number of other clients that add such
434     * a property.
435     *
436     * We want to ensure that there is no way for a client to successfully
437     * add this property between when the "deleter" process has determined that
438     * no such property exists and when it deletes the object.
439     *
440     * In other words, if there are only clients adding properties and the
441     * "deleter" deleting objects it should not be possible for an object
442     * to be deleted if a client has added a title and received a successful
443     * http response code.
444     */
445    @Test
446    @Ignore("Until we implement some kind of record level locking.")
447    public void testTransactionAndConcurrentConflictingUpdate() throws Exception {
448        final String preserveProperty = "preserve";
449        final String preserveValue = "true";
450
451        /* create the object in question */
452        final String objPid = randomUUID().toString();
453        createObject(objPid);
454
455         /* create the deleter transaction */
456        final String deleterTxLocation = createTransaction();
457        final String deleterTxId = deleterTxLocation.substring(serverAddress.length());
458
459        /* assert that the object is eligible for delete in the transaction */
460        verifyProperty("No preserve property should be set!", objPid, deleterTxId,
461                preserveProperty, preserveValue, false);
462
463        /* delete that object in the transaction */
464        final HttpDelete delete = new HttpDelete(deleterTxLocation + "/" + objPid);
465        assertEquals(204, execute(delete).getStatusLine().getStatusCode());
466
467        /* fetch the object-deleted-in-tx outside of the tx */
468        final HttpGet getObj =
469                new HttpGet(serverAddress + objPid);
470        HttpResponse resp = execute(getObj);
471        assertEquals("Expected to find our object outside the scope of the tx,"
472                + " despite it being deleted in the uncommitted transaction.",
473                200, resp.getStatusLine().getStatusCode());
474
475        /* mark the object as not deletable outside the context of the transaction */
476        setProperty(objPid, preserveProperty, preserveValue);
477
478        /* commit that transaction */
479        final HttpPost commitDeleteTx =
480                new HttpPost(deleterTxLocation + "/fcr:tx/fcr:commit");
481        resp = execute(commitDeleteTx);
482        assertNotEquals("Transaction is not atomic with regards to the object!",
483                204, resp.getStatusLine().getStatusCode());
484    }

493    private void verifyProperty(final String assertionMessage, final String pid, final String txId,
494                                final String propertyUri, final String propertyValue,
495                                final boolean shouldExist) throws IOException {
496        client = createClient();
497        final HttpGet getObjCommitted =
498                new HttpGet(serverAddress + (txId != null ? txId + "/" : "") + pid);
499        final GraphStore graphStore = getGraphStore(getObjCommitted);
500        final boolean exists = graphStore.contains(ANY, createResource(serverAddress + pid).asNode(),
501                ResourceFactory.createProperty(propertyUri).asNode(), createPlainLiteral(propertyValue)
502                .asNode());
503        if (shouldExist) {
504            assertTrue(assertionMessage, exists);
505        } else {
506            assertFalse(assertionMessage, exists);
507        }
508    }
