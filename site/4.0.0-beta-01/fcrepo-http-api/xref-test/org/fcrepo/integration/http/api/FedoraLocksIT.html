96       * Test whether a created lock can be viewed by both
97       * the creator (supplying the token) and by a request
98       * without the token and that the token is only visible
99       * when it was supplied to the session.
100      */
101     @Test
102     public void testLockMetadataAvailability() throws IOException {
103         final String pid = getRandomUniquePid();
104         createObject(pid);
105 
106         final String lockToken = getLockToken(lockObject(pid));
107 
108         final Node lockURI = createURI(serverAddress + pid + "/" + FCR_LOCK);
109         final Node nodeURI = createURI(serverAddress + pid);
110 
111         GraphStore lockTriples = getLockProperties(pid, lockToken);
112 
113         Assert.assertTrue("Lock relationship must be present!",
114                 lockTriples.contains(ANY, lockURI, createURI(LOCKS.getURI()), nodeURI));
115 
116         Assert.assertTrue("Lock token must be present!",
117                 lockTriples.contains(ANY, lockURI, createURI(HAS_LOCK_TOKEN.getURI()), createLiteral(lockToken)));
118 
119         lockTriples = getLockProperties(pid, null);
120 
121         Assert.assertTrue("Lock relationship must be present!",
122                 lockTriples.contains(ANY, lockURI, createURI(LOCKS.getURI()), nodeURI));
123 
124         Assert.assertFalse("Lock token must be hidden!",
125                 lockTriples.contains(ANY, lockURI, createURI(HAS_LOCK_TOKEN.getURI()), createLiteral(lockToken)));
126 
127         assertUnlockWithToken(pid, lockToken);
128     }

158     /**
159      * A basic test to ensure that deep locking prevents updates to child nodes
160      * while a shallow lock does not.
161      */
162     @Test
163     public void testDeepVsShallowLocks() throws IOException {
164         final String pid = getRandomUniquePid();
165         createObject(pid);
166         final String childPid = pid + "/" + getRandomUniquePid();
167         createObject(childPid);
168 
169         // Test shallow lock
170         final String shallowLockToken = getLockToken(lockObject(pid));
171 
172         assertCannotSetPropertyWithoutLockToken(pid);
173         assertCanSetProperty(pid, shallowLockToken);
174         assertCanSetProperty("Properties of a child of a shallow locked node"
175                 + " should be updatable without the lock token!", childPid, null);
176         assertUnlockWithToken(pid, shallowLockToken);
177 
178         // Test deep lock
179         final String deepLockToken = getLockToken(lockObject(pid, true));
180 
181         assertCannotSetPropertyWithoutLockToken(pid);
182         assertCannotSetPropertyWithoutLockToken("Deep lock must prevent property updates on child nodes!", childPid);
183         assertCanSetProperty(pid, deepLockToken);
184         assertCanSetProperty("Properties of a child of a shallow locked node"
185                 + " should be updatable without the lock token!", childPid, deepLockToken);
186         assertUnlockWithToken(pid, deepLockToken);
187         assertCanSetProperty("Child of unlocked object must be able to be updated now!", childPid, null);
188     }

309     @Test
310     public void testResponseContentTypes() throws Exception {
311         final String pid = getRandomUniquePid();
312         createObject(pid);
313         final String lockToken = getLockToken(lockObject(pid));
314 
315         for (final String type : RDFMediaType.POSSIBLE_RDF_RESPONSE_VARIANTS_STRING) {
316             final HttpGet method =
317                     new HttpGet(serverAddress + pid + "/" + FCR_LOCK);
318 
319             method.addHeader("Accept", type);
320             addLockToken(method, lockToken);
321             final HttpResponse response = execute(method);
322             Assert.assertEquals(200, response.getStatusLine().getStatusCode());
323 
324             // If you don't read the whole response, the session
325             // may not be closed (if it's mid-stream) and your
326             // next request will fail because the token is still
327             // attached to that session and cannot be attached
328             // again.
329             EntityUtils.consume(response.getEntity());
330 
331             assertContentType(response, type);
332         }
333     }

432      * Assumes (and asserts) that a request to lock a resource was
433      * successful and returns the lock token from the response.
434      */
435     private String getLockToken(final HttpResponse response) {
436         final StatusLine status = response.getStatusLine();
437         Assert.assertEquals(CREATED.getStatusCode(), status.getStatusCode());
438         final Header lockToken = response.getFirstHeader("Lock-Token");
439         Assert.assertNotNull("Lock-Token was not provided in response!", lockToken);

500     /**
501      * Attempts to set a property on an object.
502      * @param pid the pid
503      * @param txId a transaction ID (may be null)
504      * @param lockToken a lock token (may be null)
505      * @param propertyUri the property to set
506      * @param value the value to set
507      * @return the HttpResponse
508      */
509     private HttpResponse setProperty(final String pid,
510                                      final String txId,
511                                      final String lockToken,
512                                      final String propertyUri,
513                                      final String value) throws IOException {
514         final HttpPatch postProp = new HttpPatch(serverAddress
515                 + (txId != null ? txId + "/" : "") + pid);
516         addLockToken(postProp, lockToken);
517         postProp.setHeader("Content-Type", "application/sparql-update");
518         final String updateString =
519                 "INSERT { <"
520                         + serverAddress + pid
521                         + "> <" + propertyUri + "> \"" + value + "\" } WHERE { }";
522         postProp.setEntity(new StringEntity(updateString));
523         return execute(postProp);
524     }
