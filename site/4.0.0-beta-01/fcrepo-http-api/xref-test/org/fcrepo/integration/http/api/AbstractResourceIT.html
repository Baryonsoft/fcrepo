159     // Executes requests with preemptive basic authentication
160     protected HttpResponse executeWithBasicAuth(final HttpUriRequest request,
161                                                 final String username,
162                                                 final String password)
163         throws IOException {
164         final HttpHost target = new HttpHost(HOSTNAME, SERVER_PORT, PROTOCOL);
165         final CredentialsProvider credsProvider = new BasicCredentialsProvider();
166         credsProvider.setCredentials(
167                 new AuthScope(target.getHostName(), target.getPort()),
168                 new UsernamePasswordCredentials(username, password));
169         final CloseableHttpClient httpclient = HttpClients.custom()
170                 .setDefaultCredentialsProvider(credsProvider).build();
171 
172         final AuthCache authCache = new BasicAuthCache();
173         final BasicScheme basicAuth = new BasicScheme();
174         authCache.put(target, basicAuth);
175 
176         final HttpClientContext localContext = HttpClientContext.create();
177         localContext.setAuthCache(authCache);
178 
179         final CloseableHttpResponse response = httpclient.execute(request, localContext);
180         return response;
181     }

184     protected int getStatus(final HttpUriRequest method)
185         throws ClientProtocolException, IOException {
186         final HttpResponse response = execute(method);
187         final int result = response.getStatusLine().getStatusCode();
188         if (!(result > 199) || !(result < 400)) {
189             logger.warn(EntityUtils.toString(response.getEntity()));
190         }
191         return result;

202     protected GraphStore getGraphStore(final HttpClient client, final HttpUriRequest method) throws IOException {
203 
204         if (method.getFirstHeader("Accept") == null) {
205             method.addHeader("Accept", "application/n-triples");
206         } else {
207             logger.debug("Retrieving RDF in mimeType: {}", method
208                     .getFirstHeader("Accept"));
209         }
210 
211         final HttpResponse response = client.execute(method);
212         assertEquals(OK.getStatusCode(), response.getStatusLine()
213                                              .getStatusCode());
214         final GraphStore result = parseTriples(response.getEntity());
215         logger.trace("Retrieved RDF: {}", result);
216         return result;
217 
218     }

257     protected HttpResponse setProperty(final String pid, final String txId,
258                                        final String propertyUri,
259                                        final String value) throws IOException {
260         final HttpPatch postProp = new HttpPatch(serverAddress
261                 + (txId != null ? txId + "/" : "") + pid);
262         postProp.setHeader("Content-Type", "application/sparql-update");
263         final String updateString =
264                 "INSERT { <"
265                         + serverAddress + pid
266                         + "> <" + propertyUri + "> \"" + value + "\" } WHERE { }";
267         postProp.setEntity(new StringEntity(updateString));
268         final HttpResponse dcResp = execute(postProp);
269         assertEquals(dcResp.getStatusLine().toString(),
270                 204, dcResp.getStatusLine().getStatusCode());
271         postProp.releaseConnection();
272         return dcResp;
273     }

275     protected static void addMixin(final String pid, final String mixinUrl) throws IOException {
276         final HttpPatch updateObjectGraphMethod =
277                 new HttpPatch(serverAddress + pid);
278         updateObjectGraphMethod.addHeader("Content-Type",
279                 "application/sparql-update");
280         final BasicHttpEntity e = new BasicHttpEntity();
281 
282         e.setContent(new ByteArrayInputStream(
283                 ("INSERT DATA { <> <http://www.w3.org/1999/02/22-rdf-syntax-ns#type> <" + mixinUrl + "> . } ")
284                         .getBytes()));
285         updateObjectGraphMethod.setEntity(e);
286         final HttpResponse response = client.execute(updateObjectGraphMethod);
287         assertEquals(NO_CONTENT.getStatusCode(), response.getStatusLine()
288                 .getStatusCode());
289     }
