565     /**
566      * Verifies that one version exists with each supplied value.  This method
567      * makes assertions that each of the provided values is the content of a
568      * version node and nothing else.  Order isn't important, and no assumption
569      * is made about whether extra versions exist.
570      */
571     private void verifyVersions(final GraphStore graph, final Node subject, final String ... values)
572             throws IOException {
573         final ArrayList<String> remainingValues = newArrayList(values);
574         final Iterator<Quad> versionIt = graph.find(ANY, subject, HAS_VERSION.asNode(), ANY);
575 
576         while (versionIt.hasNext() && !remainingValues.isEmpty()) {
577             final String value =
578                 EntityUtils.toString(client
579                         .execute(new HttpGet(versionIt.next().getObject().getURI() + "/fcr:content")).getEntity());
580             remainingValues.remove(value);
581         }
582 
583         if (!remainingValues.isEmpty()) {
584             fail(remainingValues.get(0) + " was not preserved in the version history!");
585         }
586     }
