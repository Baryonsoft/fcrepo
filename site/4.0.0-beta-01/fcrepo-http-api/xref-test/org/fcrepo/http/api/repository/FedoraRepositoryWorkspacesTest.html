76      public void setUp() throws Exception {
77          initMocks(this);
78          mockUriInfo = getUriInfoImpl();
79          workspaces = new FedoraRepositoryWorkspaces();
80          setField(workspaces, "session", mockSession);

90                  new String[] {"xxx"});

95

98                                        createResource("http://localhost/fcrepo/workspace:xxx" )));

102     @Test
103     public void testCreateWorkspace() throws Exception {
104         final Repository mockRepository = mockRepository();
105         when(mockSession.getRepository()).thenReturn(mockRepository);
106         when(mockWorkspaceSession.getRepository()).thenReturn(mockRepository);
107         when(mockRepository.login("xxx")).thenReturn(mockWorkspaceSession);
108         when(mockWorkspaceSession.getWorkspace()).thenReturn(mockOtherWorkspace);
109         when(mockOtherWorkspace.getName()).thenReturn("xxx");
110         final Response response = workspaces.createWorkspace("xxx", mockUriInfo);
111         verify(mockWorkspace).createWorkspace("xxx");
112         assertEquals(201, response.getStatus());
113         final String location = response.getMetadata().getFirst("Location").toString();
114         assertEquals("http://localhost/fcrepo/workspace:xxx/", location);
115     }
