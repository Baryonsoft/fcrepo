059/**
060 * Inject our HTTP API methods into the object graphs
061 *
062 * @author awoods
063 */
064@Component
065public class HttpApiResources implements UriAwareResourceModelFactory {
066
067    @Autowired
068    protected SerializerUtil serializers;
069
070    @Override
071    public Model createModelForResource(final FedoraResource resource,
072        final UriInfo uriInfo, final IdentifierTranslator graphSubjects)
073        throws RepositoryException {
074
075        final Model model = createDefaultModel();
076
077        final Resource s = graphSubjects.getSubject(resource.getNode().getPath());
078
079        if (resource.getNode().getPrimaryNodeType().isNodeType(ROOT)) {
080            addRepositoryStatements(uriInfo, model, s);
081        } else {
082            addNodeStatements(resource, uriInfo, model, s);
083        }
084
085        if (resource.hasContent()) {
086            addContentStatements(resource, uriInfo, model, s);
087        }
088
089        return model;
090    }
091
092    private static void addContentStatements(final FedoraResource resource, final UriInfo uriInfo,
093        final Model model, final Resource s) throws RepositoryException {
094        // fcr:fixity
095        final Map<String, String> pathMap =
096            singletonMap("path", resource.getPath().substring(1));
097        model.add(s, HAS_FIXITY_SERVICE, createResource(uriInfo
098                .getBaseUriBuilder().path(FedoraFixity.class).buildFromMap(
099                        pathMap).toASCIIString()));
100    }
101
102    private void addNodeStatements(final FedoraResource resource, final UriInfo uriInfo,
103        final Model model, final Resource s) throws RepositoryException {
104
105        final Map<String, String> pathMap =
106                singletonMap("path", resource.getPath().substring(1));
107
108        // hasLock
109        if (resource.getNode().isLocked()) {
110            final String path = resource.getNode().getPath();
111            final Node lockHoldingNode
112                = resource.getNode().getSession().getWorkspace()
113                    .getLockManager().getLock(path).getNode();
114            final Map<String, String> lockedNodePathMap =
115                    singletonMap("path", lockHoldingNode.getPath().substring(1));
116            model.add(s, HAS_LOCK, createResource(uriInfo
117                .getBaseUriBuilder().path(FedoraLocks.class).buildFromMap(
118                        lockedNodePathMap).toASCIIString()));
119        }
120
121        // fcr:versions
122        if (resource.getNode().isNodeType(NodeType.MIX_VERSIONABLE)) {
123            model.add(s, HAS_VERSION_HISTORY, createResource(uriInfo
124                    .getBaseUriBuilder().path(FedoraVersions.class).buildFromMap(
125                            pathMap).toASCIIString()));
126        }
127
128        // fcr:exports?format=xyz
129        for (final String key : serializers.keySet()) {
130            final Resource format =
131                    createResource(uriInfo.getBaseUriBuilder().path(
132                            FedoraExport.class).queryParam("format", key)
133                            .buildFromMap(pathMap).toASCIIString());
134            model.add(s, HAS_SERIALIZATION, format);
135        }
136    }
137
138    private void addRepositoryStatements(final UriInfo uriInfo, final Model model,
139        final Resource s) {
140        // fcr:search
141        model.add(s, HAS_SEARCH_SERVICE, createResource(uriInfo
142                .getBaseUriBuilder().path(FedoraFieldSearch.class).build()
143                .toASCIIString()));
144
145        // sitemap
146        model.add(s, HAS_SITEMAP, createResource(uriInfo.getBaseUriBuilder()
147                .path(FedoraRepositorySitemap.class).build().toASCIIString()));
148
149        // fcr:tx
150        model.add(s, HAS_TRANSACTION_SERVICE, createResource(uriInfo
151                .getBaseUriBuilder().path(FedoraRepositoryTransactions.class)
152                .build().toASCIIString()));
153
154        // fcr:namespaces
155        model.add(s, HAS_NAMESPACE_SERVICE, createResource(uriInfo
156                .getBaseUriBuilder().path(FedoraRepositoryNamespaces.class)
157                .build().toASCIIString()));
158
159        // fcr:workspaces
160        model.add(s, HAS_WORKSPACE_SERVICE, createResource(uriInfo
161                .getBaseUriBuilder().path(FedoraRepositoryWorkspaces.class)
162                .build().toASCIIString()));
163
164        // fcr:export?format=xyz
165        for (final String key : serializers.keySet()) {
166            model.add(s, HAS_SERIALIZATION, createResource(uriInfo
167                    .getBaseUriBuilder().path(FedoraRepositoryExport.class)
168                    .queryParam("format", key).build().toASCIIString()));
169        }
170    }
