139    /**
140     * Convert a JAX-RS list of PathSegments to a JCR path
141     *
142     * @param paths
143     * @return String jcr path
144     */
145    public static final String toPath(final List<PathSegment> paths) {
146        final StringBuilder result = new StringBuilder();
147        LOGGER.trace("converting URI path to JCR path: {}", paths);
148
149        int i = 0;
150
151        for (final PathSegment path : paths) {
152            final String p = path.getPath();
153
154            if (p.equals("")) {
155                LOGGER.trace("Ignoring empty segment {}", p);
156            } else if (i == 0 &&
157                    (p.startsWith("tx:") || p.startsWith("workspace:"))) {
158                LOGGER.trace("Ignoring internal segment {}", p);
159                i++;
160            } else {
161
162                LOGGER.trace("Adding segment {}", p);
163
164                if (!p.startsWith("[")) {
165                    result.append('/');
166                }
167                result.append(p);
168                i++;
169            }
170        }
171
172        final String path = result.toString();
173
174        if (path.isEmpty()) {
175            return "/";
176        }
177        return path;
178    }

180    /**
181     * Evaluate the cache control headers for the request to see if it can be served from
182     * the cache.
183     *
184     * @param request
185     * @param servletResponse
186     * @param resource
187     * @throws javax.jcr.RepositoryException
188     */
189    protected static void checkCacheControlHeaders(final Request request,
190                                                   final HttpServletResponse servletResponse,
191                                                   final FedoraResource resource) throws RepositoryException {
192        evaluateRequestPreconditions(request, resource, true);
193        addCacheControlHeaders(servletResponse, resource);
194    }
195
196    /**
197     * Add ETag and Last-Modified cache control headers to the response
198     * @param servletResponse
199     * @param resource
200     * @throws RepositoryException
201     */
202    protected static void addCacheControlHeaders(final HttpServletResponse servletResponse,
203                                                 final FedoraResource resource) throws RepositoryException {
204
205        final EntityTag etag = new EntityTag(resource.getEtagValue());
206        final Date date = resource.getLastModifiedDate();
207
208        if (!etag.getValue().isEmpty()) {
209            servletResponse.addHeader("ETag", etag.toString());
210        }
211
212        if (date != null) {
213            servletResponse.addDateHeader("Last-Modified", date.getTime());
214        }
215    }

223                                                       final FedoraResource resource) throws RepositoryException {
224        evaluateRequestPreconditions(request, resource, false);
225    }

242                    uriInfo, subjects);

246    private static void evaluateRequestPreconditions( final Request request, final FedoraResource resource,
247       final boolean cacheControl ) throws RepositoryException {
248
249        final EntityTag etag = new EntityTag(resource.getEtagValue());
250        final Date date = resource.getLastModifiedDate();
251        final Date roundedDate = new Date();
252
253        if (date != null) {
254            roundedDate.setTime(date.getTime() - date.getTime() % 1000);
255        }
256
257        Response.ResponseBuilder builder = request.evaluatePreconditions(etag);
258        if ( builder != null ) {
259            builder = builder.entity("ETag mismatch");
260        } else {
261            builder = request.evaluatePreconditions(roundedDate);
262            if ( builder != null ) {
263                builder = builder.entity("Date mismatch");
264            }
265        }
266
267        if (builder != null && cacheControl ) {
268            final CacheControl cc = new CacheControl();
269            cc.setMaxAge(0);
270            cc.setMustRevalidate(true);
271            // here we are implicitly emitting a 304
272            // the exception is not an error, it's genuinely
273            // an exceptional condition
274            builder = builder.cacheControl(cc).lastModified(date).tag(etag);
275        }
276        if (builder != null) {
277            throw new WebApplicationException(builder.build());
278        }
279    }

281    /**
282     * Parse Content-Disposition header and extract original filename.
283    **/
284    protected static String originalFileName( final String contentDisposition ) throws ParseException {
285        if (!isBlank(contentDisposition)) {
286            return new ContentDisposition(contentDisposition).getFileName();
287        }
288        return null;
289    }

291    /**
292     * Create a checksum URI object.
293    **/
294    protected static URI checksumURI( final String checksum ) throws URISyntaxException {
295        if (!isBlank(checksum)) {
296            return new URI(checksum);
297        }
298        return null;
299    }
