46  /**
47   * An {@link org.fcrepo.kernel.api.RdfStream} containing information about the fixity of a
48   * {@link org.fcrepo.kernel.api.models.FedoraBinary}.
49   *
50   * @author ajs6f
51   * @since Oct 15, 2013
52   */
53  public class FixityRdfContext extends NodeRdfContext {
54  
55      /**
56       * Ordinary constructor.
57       *
58       * @param resource the resource
59       * @param idTranslator the id translator
60       * @param blobs the blobs
61       * @param digest the digest uri
62       * @param size the size
63       */
64      public FixityRdfContext(final FedoraResource resource,
65                              final IdentifierConverter<Resource, FedoraResource> idTranslator,
66                              final Iterable<FixityResult> blobs,
67                              final URI digest,
68                              final long size) {
69          super(resource, idTranslator);
70  
71          concat(StreamSupport.stream(blobs.spliterator(), false).flatMap(uncheck(blob -> {
72              final org.apache.jena.graph.Node resultSubject =
73                      createURI(subject().getURI() + "#fixity/" + now().toEpochMilli());
74              final List<Triple> b = new ArrayList<>();
75  
76              b.add(create(subject(), HAS_FIXITY_RESULT.asNode(), resultSubject));
77              b.add(create(resultSubject, type.asNode(), FIXITY_TYPE.asNode()));
78              b.add(create(resultSubject, type.asNode(), EVENT_OUTCOME_INFORMATION.asNode()));
79  
80              blob.getStatus(size, digest).stream().map(state -> createLiteral(state.toString()))
81                      .map(state -> create(resultSubject, HAS_FIXITY_STATE.asNode(), state)).forEach(b::add);
82  
83              b.add(create(resultSubject, HAS_MESSAGE_DIGEST_ALGORITHM.asNode(),
84                      createTypedLiteral(blob.getUsedAlgorithm()).asNode()));
85              b.add(create(resultSubject, HAS_MESSAGE_DIGEST.asNode(), createURI(blob.getComputedChecksum().toString())));
86              b.add(create(resultSubject, HAS_SIZE.asNode(), createTypedLiteral(blob.getComputedSize()).asNode()));
87  
88              return b.stream();
89          })));
90      }
91  
92  }
