40  /**
41   * Helper for writing Datasets out in a variety of serialization formats
42   * with cache headers.
43   */
44  @Provider
45  @Component
46  public class RdfProvider implements MessageBodyWriter<Dataset> {
47  
48      private static final Logger logger = getLogger(RdfProvider.class);
49  
50      @Override
51      public void writeTo(final Dataset rdf, final Class<?> type,
52              final Type genericType, final Annotation[] annotations,
53              final MediaType mediaType,
54              final MultivaluedMap<String, Object> httpHeaders,
55              final OutputStream entityStream) throws IOException,
56          WebApplicationException {
57  
58          logger.debug("Writing a response for: {} with MIMEtype: {}", rdf,
59                  mediaType);
60  
61          // add standard headers
62          httpHeaders.put("Content-type", of((Object) mediaType.toString()));
63  
64          setCachingHeaders(httpHeaders, rdf);
65  
66          new GraphStoreStreamingOutput(rdf, mediaType).write(entityStream);
67      }
68  
69      @Override
70      public boolean isWriteable(final Class<?> type, final Type genericType,
71              final Annotation[] annotations, final MediaType mediaType) {
72  
73          // we can return a result for any MIME type that Jena can serialize
74          final Boolean appropriateMimeType =
75                  contentTypeToLang(mediaType.toString()) != null;
76          return appropriateMimeType &&
77                  (Dataset.class.isAssignableFrom(type) || Dataset.class
78                          .isAssignableFrom(genericType.getClass()));
79      }
80  
81      @Override
82      public long getSize(final Dataset rdf, final Class<?> type,
83              final Type genericType, final Annotation[] annotations,
84              final MediaType mediaType) {
85          // we don't know in advance how large the result might be
86          return -1;
87      }

89  }
