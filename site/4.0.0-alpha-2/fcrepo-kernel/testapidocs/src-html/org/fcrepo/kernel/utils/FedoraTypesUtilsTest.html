148    @Test
149    public void testIsMultipleValuedProperty() throws RepositoryException {
150        final Property mockYes = mock(Property.class);
151        when(mockYes.isMultiple()).thenReturn(true);
152        final Property mockNo = mock(Property.class);
153        final Predicate<Property> test = isMultipleValuedProperty;
154        try {
155            test.apply(null);
156            fail("Null values should throw an IllegalArgumentException");
157        } catch (final IllegalArgumentException e) {
158        }
159        boolean actual = test.apply(mockYes);
160        assertEquals(true, actual);
161        actual = test.apply(mockNo);
162        assertEquals(false, actual);
163        when(mockYes.isMultiple()).thenThrow(new RepositoryException());
164        try {
165            test.apply(mockYes);
166            fail("Unexpected completion after RepositoryException!");
167        } catch (final RuntimeException e) {} // expected
168    }

306    @Test
307    public void testGetObjectSize() throws RepositoryException {
308
309        when(mockRepository.login()).thenReturn(mockSession);
310        when(mockSession.getWorkspace()).thenReturn(mockWS);
311        when(mockWS.getQueryManager()).thenReturn(mockQueryManager);
312        when(
313                mockQueryManager.createQuery("SELECT [" + CONTENT_SIZE +
314                        "] FROM [" + FEDORA_BINARY + "]", JCR_SQL2))
315                .thenReturn(mockQuery);
316        when(mockQuery.execute()).thenReturn(mockResults);
317        when(mockResults.getRows()).thenReturn(mockIterator);
318
319        when(mockIterator.hasNext()).thenReturn(true, true, true, false);
320        when(mockIterator.nextRow()).thenReturn(mockRow, mockRow, mockRow);
321
322        when(mockRow.getValue(CONTENT_SIZE)).thenReturn(mockValue);
323        when(mockValue.getLong()).thenReturn(5L, 10L, 1L);
324
325        final long count = getRepositorySize(mockRepository);
326        assertEquals("Got wrong count!", 16L, count);
327        verify(mockSession).logout();
328        verify(mockSession, never()).save();
329    }
