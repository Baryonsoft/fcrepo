016package org.fcrepo.kernel;
017
018import static org.fcrepo.kernel.utils.FedoraTypesUtils.isFedoraObject;
019import static org.slf4j.LoggerFactory.getLogger;
020
021import javax.jcr.Node;
022import javax.jcr.RepositoryException;
023import javax.jcr.Session;
024
025import org.modeshape.jcr.api.JcrConstants;
026import org.slf4j.Logger;
027
028/**
029 * An abstraction that represents a Fedora Object backed by
030 * a JCR node.
031 *
032 * @author ajs6f
033 * @date Feb 21, 2013
034 */
035public class FedoraObject extends FedoraResource {
036
037    static final Logger logger = getLogger(FedoraObject.class);
038
039    /**
040     * Construct a FedoraObject from an existing JCR Node
041     * @param node an existing JCR node to treat as an fcrepo object
042     */
043    public FedoraObject(final Node node) {
044        super(node);
045        mixinTypeSpecificCrap();
046    }
047
048    /**
049     * Create or find a FedoraObject at the given path
050     * @param session the JCR session to use to retrieve the object
051     * @param path the absolute path to the object
052     * @throws RepositoryException
053     */
054    public FedoraObject(final Session session, final String path,
055                        final String nodeType) throws RepositoryException {
056        super(session, path, nodeType);
057        mixinTypeSpecificCrap();
058    }
059    /**
060     * Create or find a FedoraDatastream at the given path
061     * @param session the JCR session to use to retrieve the object
062     * @param path the absolute path to the object
063     * @throws RepositoryException
064     */
065    public FedoraObject(final Session session, final String path)
066        throws RepositoryException {
067        this(session, path, JcrConstants.NT_FOLDER);
068    }
069
070
071    private void mixinTypeSpecificCrap() {
072        try {
073            if (node.isNew() || !hasMixin(node)) {
074                logger.debug("Setting {} properties on a {} node {}...",
075                             FEDORA_OBJECT,
076                             JcrConstants.NT_FOLDER,
077                             node.getPath());
078                node.addMixin(FEDORA_OBJECT);
079            }
080        } catch (RepositoryException e) {
081            logger.warn("Could not decorate {} with {} properties: {} ",
082                        JcrConstants.JCR_CONTENT, FEDORA_OBJECT, e);
083        }
084    }
085
086    /**
087     * @return The JCR name of the node that backs this object.
088     * @throws RepositoryException
089     */
090    public String getName() throws RepositoryException {
091        return node.getName();
092    }
093
094    /**
095     * Check if the node has a fedora:object mixin
096     * @param node
097     * @return
098     * @throws RepositoryException
099     */
100    public static boolean hasMixin(final Node node) throws RepositoryException {
101        return isFedoraObject.apply(node);
102    }
103
104}
