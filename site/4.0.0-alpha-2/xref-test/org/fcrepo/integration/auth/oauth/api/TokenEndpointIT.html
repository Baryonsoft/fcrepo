18  import static java.util.regex.Pattern.compile;

20  import static javax.ws.rs.core.MediaType.APPLICATION_FORM_URLENCODED;
21  import static javax.ws.rs.core.MediaType.APPLICATION_JSON;

23  import java.io.IOException;

25  import java.util.regex.Pattern;

27  import org.apache.http.HttpResponse;
28  import org.apache.http.client.ClientProtocolException;
29  import org.apache.http.client.methods.HttpPost;
30  import org.apache.http.util.EntityUtils;
31  import org.junit.Assert;
32  import org.junit.Test;
33  
34  public class TokenEndpointIT extends AbstractOAuthResourceIT {
35  
36      private final String tokenRegexp = "access_token\":\"(.+?)\"";
37  
38      private final Pattern tokenPattern = compile(tokenRegexp);
39  
40      @Test
41      public void testGetToken() throws Exception {
42          logger.trace("Entering testGetToken()...");
43          final HttpPost post =
44                  new HttpPost(
45                          tokenEndpoint +
46                                  "?grant_type=password&username=foo&password=bar&client_secret=foo&client_id=bar");
47          post.addHeader("Accept", APPLICATION_JSON);
48          post.addHeader("Content-type", APPLICATION_FORM_URLENCODED);
49          final HttpResponse tokenResponse = client.execute(post);
50          logger.debug("Got a token response: \n{}", EntityUtils
51                  .toString(tokenResponse.getEntity()));
52          Assert.assertEquals("Couldn't retrieve a token from token endpoint!",
53                              200,
54                              tokenResponse.getStatusLine().getStatusCode());
55  
56      }
57  
58      @Test
59      public void testUseToken() throws ClientProtocolException, IOException {
60          logger.trace("Entering testUseToken()...");
61          logger.debug("Trying to write an object to authenticated area without authentication via token...");
62          final HttpResponse failure =
63                  client.execute(postObjMethod("authenticated/testUseToken"));
64          Assert.assertEquals(
65                  "Was able to write to an authenticated area when I shouldn't be able to",
66                  SC_UNAUTHORIZED,
67                  failure.getStatusLine().getStatusCode());
68          logger.debug("Failed as expected.");
69          logger.debug("Now trying with authentication via token...");
70          final HttpPost post =
71                  new HttpPost(
72                          tokenEndpoint +
73                                  "?grant_type=password&username=foo&password=bar&client_secret=foo&client_id=bar");
74          post.addHeader("Accept", APPLICATION_JSON);
75          post.addHeader("Content-type", APPLICATION_FORM_URLENCODED);
76          final HttpResponse tokenResponse = client.execute(post);
77          final String tokenResponseString =
78                  EntityUtils.toString(tokenResponse.getEntity());
79          logger.debug("Got a token response: \n{}", tokenResponseString);
80          Assert.assertEquals("Couldn't retrieve a token from token endpoint!",
81                              200,
82                              tokenResponse.getStatusLine().getStatusCode());
83  
84          final Matcher tokenMatcher = tokenPattern.matcher(tokenResponseString);
85          Assert.assertTrue("Couldn't find token in token response!", tokenMatcher
86                  .find());
87          final String token = tokenMatcher.group(1);
88          logger.debug("Found token: {}", token);
89          final HttpPost successMethod =
90                  postObjMethod("authenticated/testUseToken?access_token=" +
91                          token);
92          final HttpResponse success = client.execute(successMethod);
93          Assert.assertEquals("Failed to create object even with authentication!",
94                              201,
95                              success.getStatusLine().getStatusCode());
96      }
