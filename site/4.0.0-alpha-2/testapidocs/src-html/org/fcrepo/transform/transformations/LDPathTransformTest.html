069
070        final NodeType mockNodeType = mock(NodeType.class);
071        final NodeType mockNtBase = mock(NodeType.class);
072        when(mockNodeType.getSupertypes()).thenReturn(new NodeType[] { mockNtBase });
073        when(mockNode.getPrimaryNodeType()).thenReturn(mockNodeType);
074        final LDPathTransform nodeTypeSpecificLdpathProgramStream = LDPathTransform.getNodeTypeTransform(mockNode, "some-program");

076        assertNull(nodeTypeSpecificLdpathProgramStream);

079    @Test
080    public void testGetNodeTypeSpecificLdpathProgramForNodeTypeProgram() throws RepositoryException {
081        final Node mockConfigNode = mock(Node.class);
082        final Node mockTypeConfigNode = mock(Node.class, Mockito.RETURNS_DEEP_STUBS);
083        when(mockSession.getNode(LDPathTransform.CONFIGURATION_FOLDER + "some-program")).thenReturn(mockConfigNode);
084
085        final NodeType mockNodeType = mock(NodeType.class);
086        final NodeType mockNtBase = mock(NodeType.class);
087        when(mockNodeType.getSupertypes()).thenReturn(new NodeType[] { mockNtBase });
088        when(mockNode.getPrimaryNodeType()).thenReturn(mockNodeType);
089        when(mockNodeType.toString()).thenReturn("custom:type");
090        when(mockConfigNode.hasNode("custom:type")).thenReturn(true);
091        when(mockConfigNode.getNode("custom:type")).thenReturn(mockTypeConfigNode);
092        final InputStream mockInputStream = mock(InputStream.class);
093        when(mockTypeConfigNode.getNode("jcr:content").getProperty("jcr:data").getBinary().getStream()).thenReturn(mockInputStream);
094        final LDPathTransform nodeTypeSpecificLdpathProgramStream = LDPathTransform.getNodeTypeTransform(mockNode, "some-program");
095
096        assertEquals(new LDPathTransform(mockInputStream), nodeTypeSpecificLdpathProgramStream);
097    }

125    @Test
126    public void testProgramQuery() throws LDPathParseException, RepositoryException {
127
128        final Model model = ModelFactory.createDefaultModel();
129        model.add(model.createResource("abc"), model.createProperty("http://purl.org/dc/elements/1.1/title"), model.createLiteral("some-title"));
130        final Dataset testDataset = DatasetFactory.create(model);
131        testDataset.getContext().set(Symbol.create("uri"), "abc");
132        final InputStream testReader = new ByteArrayInputStream("title = dc:title :: xsd:string ;".getBytes());
133
134        testObj = new LDPathTransform(testReader);
135        final List<Map<String,Collection<Object>>> stringCollectionMap = testObj.apply(testDataset);
136
137        assert(stringCollectionMap != null);
138        assertEquals(1, stringCollectionMap.size());
139        assertEquals(1, stringCollectionMap.get(0).get("title").size());
140        assertTrue(stringCollectionMap.get(0).get("title").contains("some-title"));
141    }
