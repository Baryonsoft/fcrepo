286    @Test
287    public void testHasMixin() throws RepositoryException {
288        final NodeType mockYes = mock(NodeType.class);
289        when(mockYes.getName()).thenReturn(FEDORA_DATASTREAM);
290        final NodeType mockNo = mock(NodeType.class);
291        when(mockNo.getName()).thenReturn("not" + FEDORA_DATASTREAM);
292        final NodeType[] types = new NodeType[] {mockYes};
293        final Node test = mock(Node.class);
294        when(test.getMixinNodeTypes()).thenReturn(types);
295        assertEquals(true, hasMixin(test));
296        types[0] = mockNo;
297        assertEquals(false, hasMixin(test));
