120    @Test
121    public void shouldConvertRdfObjectsToStrings() {
122
123        final DatasetGraph mem = createMem();
124        mem.add(createAnon(), createURI("subject"), createURI("a/b/c"),
125                NodeFactory.createLiteral("abc"));
126        mem.add(createAnon(), createURI("subject"),
127                createURI("a-numeric-type"), createTypedLiteral(0).asNode());
128        mem.add(createAnon(), createURI("subject"),
129                createURI("an-empty-string"), createLiteral(""));
130        mem.add(createAnon(), createURI("subject"), createURI("a-uri"),
131                createURI("some-uri"));
132
133        assertEquals("abc", testObj.getObjectsAsString(mem,
134                createURI("subject"), createResource("a/b/c")));
135        assertEquals("0", testObj.getObjectsAsString(mem, createURI("subject"),
136                createResource("a-numeric-type")));
137        assertEquals("<empty>", testObj.getObjectsAsString(mem,
138                createURI("subject"), createResource("an-empty-string")));
139        assertEquals("&lt;<a href=\"some-uri\">some-uri</a>&gt;", testObj
140                .getObjectsAsString(mem, createURI("subject"),
141                        createResource("a-uri")));
142        assertEquals("", testObj.getObjectsAsString(mem, createURI("subject"),
143                createResource("a-nonexistent-uri")));
144
145    }

157    @Test
158    public void shouldSortTriplesForDisplay() {
159        final Model model = createDefaultModel();
160
161        model.setNsPrefix("prefix", "namespace");
162        final Property propertyA = model.createProperty("namespace", "a");
163        final Property propertyB = model.createProperty("namespace", "b");
164        final Property propertyC = model.createProperty("c");
165        final Literal literalA = model.createLiteral("a");
166        final Literal literalB = model.createLiteral("b");
167        final Resource resourceB = model.createResource("b");
168        model.add(resourceB, propertyA, literalA);
169
170        final Resource a = model.createResource("a");
171        model.add(a, propertyC, literalA);
172
173        model.add(a, propertyB, literalA);
174
175        model.add(a, propertyA, literalA);
176        model.add(a, propertyA, literalB);
177
178        final Iterator<Quad> iterator =
179                DatasetFactory.create(model).asDatasetGraph().find();
180
181        final List<Quad> sortedTriples =
182                testObj.getSortedTriples(model, iterator);
183
184        sortedTriples.get(0).matches(ANY, a.asNode(), propertyA.asNode(),
185                literalA.asNode());
186        sortedTriples.get(1).matches(ANY, a.asNode(), propertyA.asNode(),
187                literalB.asNode());
188        sortedTriples.get(2).matches(ANY, a.asNode(), propertyB.asNode(),
189                literalA.asNode());
190        sortedTriples.get(3).matches(ANY, a.asNode(), propertyC.asNode(),
191                literalA.asNode());
192        sortedTriples.get(4).matches(ANY, resourceB.asNode(),
193                propertyC.asNode(), literalA.asNode());
194
195    }
