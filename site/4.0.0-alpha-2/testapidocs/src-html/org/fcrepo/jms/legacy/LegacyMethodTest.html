023import static org.mockito.Matchers.eq;
024import static org.mockito.Mockito.mock;
025import static org.mockito.Mockito.times;
026import static org.mockito.Mockito.verify;
027import static org.mockito.Mockito.when;
028
029import java.io.IOException;
030import java.io.Writer;
031import java.util.Arrays;
032import java.util.Date;
033import java.util.List;
034
035import javax.jcr.Node;
036import javax.jcr.RepositoryException;
037import javax.jcr.nodetype.NodeType;
038import javax.jcr.observation.Event;
039
040import org.apache.abdera.model.Category;
041import org.apache.abdera.model.Entry;
042import org.apache.abdera.model.Person;
043import org.apache.abdera.model.Text;
044import org.fcrepo.jcr.FedoraJcrTypes;
045import org.junit.Before;
046import org.junit.Test;
047import org.junit.runner.RunWith;
048import org.powermock.api.mockito.PowerMockito;
049import org.powermock.core.classloader.annotations.PowerMockIgnore;
050import org.powermock.core.classloader.annotations.PrepareForTest;
051import org.powermock.modules.junit4.PowerMockRunner;
052
053@RunWith(PowerMockRunner.class)
054@PowerMockIgnore({"org.slf4j.*", "javax.xml.parsers.*", "org.apache.xerces.*"})
055@PrepareForTest({EntryFactory.class})
056public class LegacyMethodTest {
057
058    private LegacyMethod testObj;
059
060    private Entry mockDelegate;
061
062    private Event mockEvent;
063
064    private Node mockSource;
065
066    private Category mockPidCategory;
067
068    private Category mockDsidCategory;
069
070    private static final String SOURCE_DSID = "sourceDsid";
071
072    private static final String SOURCE_PID = "sourcePid";
073
074    @Before
075    public void setUp() throws RepositoryException {
076        // set up the supporting mocks for the constructor
077        mockEvent = mock(Event.class);
078        Node mockParent = mock(Node.class);
079        when(mockParent.getName()).thenReturn(SOURCE_PID);
080        mockSource = mock(Node.class);
081        when(mockSource.getName()).thenReturn(SOURCE_DSID);
082        when(mockSource.getParent()).thenReturn(mockParent);
083        NodeType mockDSType = mock(NodeType.class);
084        when(mockDSType.getName()).thenReturn(FedoraJcrTypes.FEDORA_DATASTREAM);
085        NodeType[] mockTypes = new NodeType[] {mockDSType};
086        when(mockSource.getMixinNodeTypes()).thenReturn(mockTypes);
087        mockDelegate = mock(Entry.class);
088        Text mockText = mock(Text.class);
089        when(mockDelegate.setTitle(anyString())).thenReturn(mockText);
090        // make sure the delegate Entry can be instrumented for tests
091        PowerMockito.mockStatic(EntryFactory.class);
092        when(EntryFactory.newEntry()).thenReturn(mockDelegate);
093        mockPidCategory = mock(Category.class);
094        when(mockPidCategory.getLabel()).thenReturn(
095                LegacyMethod.PID_CATEGORY_LABEL);
096        when(mockPidCategory.getTerm()).thenReturn(SOURCE_PID);
097        mockDsidCategory = mock(Category.class);
098        when(mockDsidCategory.getLabel()).thenReturn(
099                LegacyMethod.DSID_CATEGORY_LABEL);
100        when(mockDsidCategory.getTerm()).thenReturn(SOURCE_DSID);
101        List<Category> categories =
102                Arrays.asList(new Category[] {mockPidCategory, mockDsidCategory});
103        when(mockDelegate.getCategories(LegacyMethod.FEDORA_ID_SCHEME))
104                .thenReturn(categories);
105        // construct the test object
106        testObj = new LegacyMethod(mockEvent, mockSource);
107    }
108
109    @Test
110    public void testPidAccessors() {
111        String newPid = "newPid";
112        assertEquals(SOURCE_PID, testObj.getPid());
113        testObj.setPid(newPid);
114        verify(mockPidCategory).setTerm(SOURCE_PID);
115        verify(mockPidCategory).setTerm(newPid);
116    }
117
118    @Test
119    public void testDsidAccessors() {
120        String newDsid = "newDsid";
121        assertEquals(SOURCE_DSID, testObj.getDsId());
122        testObj.setDsId(newDsid);
123        verify(mockDsidCategory).setTerm(SOURCE_DSID);
124        verify(mockDsidCategory).setTerm(newDsid);
125    }
126
127    @Test
128    public void textGetEntry() {
129        Entry mockEntry = mock(Entry.class);
130        LegacyMethod to = new LegacyMethod(mockEntry);
131        assertEquals(mockEntry, to.getEntry());
132    }
133
134    @Test
135    public void testMethodNameAccessors() {
136        when(mockDelegate.getTitle()).thenReturn("foo");
137        testObj.getMethodName();
138        // called once in the constructor, once in the accessor
139        verify(mockDelegate, times(2)).getTitle();
140        testObj.setMethodName("foo");
141        Text mockText = mock(Text.class);
142        when(mockDelegate.setTitle(anyString())).thenReturn(mockText);
143        String newTitle = "bar";
144        testObj.setMethodName(newTitle);
145        verify(mockDelegate.setTitle(newTitle));
146    }
147
148    @Test
149    public void testModifiedAccesors() {
150        testObj.getModified();
151        verify(mockDelegate).getUpdated();
152        testObj.setModified(new Date());
153        // called once in the constructor, once in the accessor
154        verify(mockDelegate, times(2)).setUpdated(any(Date.class));
155    }
156
157    @Test
158    public void testUserIdAccessors() {
159        Person mockPerson = mock(Person.class);
160        when(mockDelegate.getAuthor()).thenReturn(mockPerson);
161        testObj.getUserID();
162        testObj.setUserId("foo");
163        verify(mockDelegate).addAuthor(eq("foo"), anyString(), anyString());
164        testObj.setUserId(null);
165        // called once in the constructor, once in the accessor
166        verify(mockDelegate, times(2)).addAuthor(eq("unknown"), anyString(),
167                anyString());
168    }
169
170    @Test
171    public void testSetContent() {
172        testObj.setContent("foo");
173        verify(mockDelegate).setContent("foo");
174    }
175
176    @Test
177    public void testWriteTo() throws IOException {
178        Writer mockWriter = mock(Writer.class);
179        testObj.writeTo(mockWriter);
180        verify(mockDelegate).writeTo(mockWriter);
181    }
182
183    @Test
184    public void testGetBaseURL() {
185        String url = testObj.getBaseURL();
186        assertNotNull(url);
187        assertEquals("http://localhost:8080/rest", url);
188
189        final String host = "my-host";
190        System.setProperty("fcrepo.host", host);
191
192        url = testObj.getBaseURL();
193        assertNotNull(url);
194        assertEquals("http://" + host + ":8080/rest", url);
195
196        final String port = "443";
197        System.setProperty("fcrepo.port", port);
198
199        url = testObj.getBaseURL();
200        assertNotNull(url);
201        assertEquals("https://" + host + ":" + port + "/rest", url);
202
203        final String ctxt = "sleep";
204        System.setProperty("fcrepo.ctxt", ctxt);
205
206        url = testObj.getBaseURL();
207        assertNotNull(url);
208        assertEquals("https://" + host + ":" + port + "/" + ctxt, url);
209
210        System.clearProperty("fcrepo.host");
211        System.clearProperty("fcrepo.port");
212        System.clearProperty("fcrepo.ctxt");
213    }
