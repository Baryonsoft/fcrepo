017package org.fcrepo.integration.generator;
018
019import static java.lang.System.err;
020import static java.util.UUID.randomUUID;
021import static java.util.regex.Pattern.DOTALL;
022import static java.util.regex.Pattern.compile;
023import static javax.ws.rs.core.MediaType.TEXT_XML;
024import static org.junit.Assert.assertEquals;
025import static org.junit.Assert.assertTrue;
026
027import org.apache.http.HttpResponse;
028import org.apache.http.client.methods.HttpGet;
029import org.apache.http.client.methods.HttpPatch;
030import org.apache.http.entity.BasicHttpEntity;
031import org.apache.http.util.EntityUtils;
032import org.junit.Test;
033import org.springframework.test.context.ContextConfiguration;
034
035import java.io.ByteArrayInputStream;
036
037@ContextConfiguration({"/spring-test/test-container.xml"})
038public class DublinCoreGeneratorIT extends AbstractResourceIT {
039
040    @Test
041    public void testJcrPropertiesBasedOaiDc() throws Exception {
042        final int status = getStatus(postObjMethod("DublinCoreTest1"));
043        assertEquals(201, status);
044        final HttpPatch post = new HttpPatch(serverAddress + "DublinCoreTest1");
045        post.setHeader("Content-Type", "application/sparql-update");
046        final BasicHttpEntity entity = new BasicHttpEntity();
047        final String subjectURI = serverAddress + "DublinCoreTest1";
048        entity.setContent(new ByteArrayInputStream(
049                ("INSERT { <" + subjectURI + "> <http://purl.org/dc/terms/identifier> \"this is an identifier\" } WHERE {}")
050                        .getBytes()));
051        post.setEntity(entity);
052        assertEquals(204, getStatus(post));
053        final HttpGet getWorstCaseOaiMethod =
054                new HttpGet(serverOAIAddress + "DublinCoreTest1/oai:dc");
055        getWorstCaseOaiMethod.setHeader("Accept", TEXT_XML);
056        final HttpResponse response = client.execute(getWorstCaseOaiMethod);
057
058        assertEquals(200, response.getStatusLine().getStatusCode());
059
060        final String content = EntityUtils.toString(response.getEntity());
061        logger.debug("Got content: {}", content);
062        assertTrue("Didn't find oai_dc!", compile("oai_dc", DOTALL).matcher(
063                content).find());
064
065        assertTrue("Didn't find dc:identifier!", compile("dc:identifier",
066                DOTALL).matcher(content).find());
067    }
068
069    @Test
070    public void testWellKnownPathOaiDc() throws Exception {
071
072        final String pid = randomUUID().toString();
073
074        HttpResponse response = client.execute(postObjMethod(pid));
075        assertEquals(201, response.getStatusLine().getStatusCode());
076        response =
077            client.execute(postDSMethod(pid, "DC", "marbles for everyone"));
078        final int status = response.getStatusLine().getStatusCode();
079        if (status != 201) {
080            err.println(EntityUtils.toString(response.getEntity()));
081        }
082        assertEquals(201, status);
083
084        final HttpGet getWorstCaseOaiMethod =
085            new HttpGet(serverOAIAddress + pid + "/oai:dc");
086        getWorstCaseOaiMethod.setHeader("Accept", TEXT_XML);
087        response = client.execute(getWorstCaseOaiMethod);
088        assertEquals(200, response.getStatusLine().getStatusCode());
089        final String content = EntityUtils.toString(response.getEntity());
090        assertTrue("Didn't find our datastream!", compile(
091                "marbles for everyone", DOTALL).matcher(content).find());
092    }
093}
