35      @Test
36      public void testGetAuthzCode() throws ClientProtocolException, IOException {
37          // we want to grab the auth code, not get redirected
38          client.getParams().setBooleanParameter(HANDLE_REDIRECTS, false);
39          logger.debug("Entering testGetAuthzCode()...");
40          final HttpResponse response =
41                  client.execute(new HttpGet(authzEndpoint +
42                          "?client_id=CLIENT-ID&redirect_uri=http://example.com&response_type=code"));
43          logger.debug("Retrieved authorization endpoint response.");
44          final String redirectHeader =
45                  response.getFirstHeader("Location").getValue();
46          final String authCode =
47                  URI.create(redirectHeader).getQuery().split("&")[0].split("=")[1];
48          Assert.assertNotNull("Didn't find authorization code!", authCode);
49          logger.debug("with authorization code: {}", authCode);
50  
51      }

53      @Test
54      public void testUseAuthCode() throws ClientProtocolException, IOException {
55          // we want to grab the auth code, not get redirected
56          client.getParams().setBooleanParameter(HANDLE_REDIRECTS, false);
57          logger.debug("Entering testUseAuthCode()...");
58          final HttpResponse response =
59                  client.execute(new HttpGet(authzEndpoint +
60                          "?client_id=CLIENT-ID&redirect_uri=http://example.com&response_type=code"));
61          logger.debug("Retrieved authorization endpoint response.");
62          final String redirectHeader =
63                  response.getFirstHeader("Location").getValue();
64          logger.debug("Redirect header '{}'", redirectHeader);
65          final String authCode =
66                  URI.create(redirectHeader).getQuery().split("&")[4].split("=")[1];
67          Assert.assertNotNull("Didn't find authorization code!", authCode);
68          logger.debug("with authorization code: {}", authCode);
69          final HttpPost post =
70                  new HttpPost(tokenEndpoint +
71                          "?grant_type=authorization_code&code=" + authCode +
72                          "&client_secret=foo&client_id=CLIENT-ID&redirect_uri=http://example.com");
73          post.addHeader("Accept", APPLICATION_JSON);
74          post.addHeader("Content-type", APPLICATION_FORM_URLENCODED);
75          final HttpResponse tokenResponse = client.execute(post);
76          logger.debug("Got a token response: \n{}", EntityUtils
77                  .toString(tokenResponse.getEntity()));
78          Assert.assertEquals("Couldn't retrieve a token from token endpoint!",
79                              200,
80                              tokenResponse.getStatusLine().getStatusCode());
81      }
