016package org.fcrepo.integration.auth.oauth.api;
017
018import static javax.ws.rs.core.MediaType.APPLICATION_FORM_URLENCODED;
019import static javax.ws.rs.core.MediaType.APPLICATION_JSON;
020import static org.apache.http.client.params.ClientPNames.HANDLE_REDIRECTS;
021
022import java.io.IOException;
023import java.net.URI;
024
025import org.apache.http.HttpResponse;
026import org.apache.http.client.ClientProtocolException;
027import org.apache.http.client.methods.HttpGet;
028import org.apache.http.client.methods.HttpPost;
029import org.apache.http.util.EntityUtils;
030import org.junit.Assert;
031import org.junit.Test;
032
033public class AuthzEndpointIT extends AbstractOAuthResourceIT {
034
035    @Test
036    public void testGetAuthzCode() throws ClientProtocolException, IOException {
037        // we want to grab the auth code, not get redirected
038        client.getParams().setBooleanParameter(HANDLE_REDIRECTS, false);
039        logger.debug("Entering testGetAuthzCode()...");
040        final HttpResponse response =
041                client.execute(new HttpGet(authzEndpoint +
042                        "?client_id=CLIENT-ID&redirect_uri=http://example.com&response_type=code"));
043        logger.debug("Retrieved authorization endpoint response.");
044        final String redirectHeader =
045                response.getFirstHeader("Location").getValue();
046        final String authCode =
047                URI.create(redirectHeader).getQuery().split("&")[0].split("=")[1];
048        Assert.assertNotNull("Didn't find authorization code!", authCode);
049        logger.debug("with authorization code: {}", authCode);
050
051    }
052
053    @Test
054    public void testUseAuthCode() throws ClientProtocolException, IOException {
055        // we want to grab the auth code, not get redirected
056        client.getParams().setBooleanParameter(HANDLE_REDIRECTS, false);
057        logger.debug("Entering testUseAuthCode()...");
058        final HttpResponse response =
059                client.execute(new HttpGet(authzEndpoint +
060                        "?client_id=CLIENT-ID&redirect_uri=http://example.com&response_type=code"));
061        logger.debug("Retrieved authorization endpoint response.");
062        final String redirectHeader =
063                response.getFirstHeader("Location").getValue();
064        logger.debug("Redirect header '{}'", redirectHeader);
065        final String authCode =
066                URI.create(redirectHeader).getQuery().split("&")[4].split("=")[1];
067        Assert.assertNotNull("Didn't find authorization code!", authCode);
068        logger.debug("with authorization code: {}", authCode);
069        final HttpPost post =
070                new HttpPost(tokenEndpoint +
071                        "?grant_type=authorization_code&code=" + authCode +
072                        "&client_secret=foo&client_id=CLIENT-ID&redirect_uri=http://example.com");
073        post.addHeader("Accept", APPLICATION_JSON);
074        post.addHeader("Content-type", APPLICATION_FORM_URLENCODED);
075        final HttpResponse tokenResponse = client.execute(post);
076        logger.debug("Got a token response: \n{}", EntityUtils
077                .toString(tokenResponse.getEntity()));
078        Assert.assertEquals("Couldn't retrieve a token from token endpoint!",
079                            200,
080                            tokenResponse.getStatusLine().getStatusCode());
081    }
082
083}
