145    @Test
146    public void testCreateObject() throws RepositoryException, IOException,
147            InvalidChecksumException, URISyntaxException {
148        final String pid = "testObject";
149        final String path = "/" + pid;
150        when(mockObjects.createObject(mockSession, path)).thenReturn(mockObject);
151        when(mockObject.getNode()).thenReturn(mockNode);
152        when(mockNode.getPath()).thenReturn(path);
153        final Response actual =
154                testObj.createObject(createPathList(pid), FEDORA_OBJECT, null,
155                        null, null, getUriInfoImpl(), null);
156        assertNotNull(actual);
157        assertEquals(CREATED.getStatusCode(), actual.getStatus());
158        assertTrue(actual.getEntity().toString().endsWith(pid));
159        verify(mockObjects).createObject(mockSession, path);
160        verify(mockSession).save();
161    }

206    @Test
207    public void testCreateDatastream() throws RepositoryException, IOException,
208            InvalidChecksumException, URISyntaxException {
209        final String pid = "FedoraDatastreamsTest1";
210        final String dsId = "testDS";
211        final String dsContent = "asdf";
212        final String dsPath = "/" + pid + "/" + dsId;
213        final InputStream dsContentStream = IOUtils.toInputStream(dsContent);
214        when(mockNode.getSession()).thenReturn(mockSession);
215
216
217        when(
218                mockDatastreams.createDatastreamNode(any(Session.class),
219                        eq(dsPath), anyString(), eq(dsContentStream),
220                        any(URI.class))).thenReturn(mockNode);
221        Datastream mockDatastream = mock(Datastream.class);
222        when(mockDatastream.getNode()).thenReturn(mockNode);
223        when(mockDatastreams.createDatastream(mockSession, dsPath)).thenReturn(mockDatastream);
224        when(mockNode.getPath()).thenReturn(dsPath);
225        final Response actual =
226                testObj.createObject(createPathList(pid, dsId),
227                        FEDORA_DATASTREAM, null, MediaType.APPLICATION_OCTET_STREAM_TYPE, null, getUriInfoImpl(),
228                        dsContentStream);
229        assertEquals(CREATED.getStatusCode(), actual.getStatus());
230        verify(mockDatastreams)
231                .createDatastreamNode(any(Session.class), eq(dsPath),
232                        anyString(), any(InputStream.class), any(URI.class));
233        verify(mockSession).save();
234    }

292    @Test
293    public void testSparqlUpdate() throws RepositoryException, IOException {
294        final String pid = "FedoraObjectsRdfTest1";
295        final String path = "/" + pid;
296        final InputStream mockStream =
297                new ByteArrayInputStream("my-sparql-statement".getBytes());
298        when(mockNodes.getObject(mockSession, path)).thenReturn(mockObject);
299        when(mockObject.updatePropertiesDataset(any(GraphSubjects.class), any(String.class)))
300            .thenReturn(mockDataset);
301        when(mockObject.getEtagValue()).thenReturn("");
302
303        when(mockObject.getLastModifiedDate()).thenReturn(Calendar.getInstance().getTime());
304        when(mockDataset.getNamedModel(GraphProperties.PROBLEMS_MODEL_NAME))
305        .thenReturn(mockModel);
306        testObj.updateSparql(createPathList(pid), getUriInfoImpl(), mockStream, mockRequest);
307
308        verify(mockObject).updatePropertiesDataset(any(GraphSubjects.class),
309                eq("my-sparql-statement"));
310        verify(mockSession).save();
311        verify(mockSession).logout();
312    }
