31  /**
32   * This is a convenience class carrying the various RDF content-type values as
33   * Strings and MediaTypes, after the fashion of the constants available on
34   * javax.ws.rs.core.MediaType
35   *
36   * @author ba2213
37   */
38  public abstract class RDFMediaType extends MediaType {
39  
40      public static final String N3 = contentTypeN3;
41  
42      public static final MediaType N3_TYPE = typeFromString(N3);
43  
44      public static final String N3_ALT2 = contentTypeN3Alt2;
45  
46      public static final MediaType N3_ALT2_TYPE = typeFromString(N3_ALT2);
47  
48      public static final String TURTLE = contentTypeTurtle;
49  
50      public static final MediaType TURTLE_TYPE = typeFromString(TURTLE);
51  
52      public static final String TURTLE_X = contentTypeTurtleAlt2;
53  
54      public static final MediaType TURTLE_X_TYPE = typeFromString(TURTLE_X);
55  
56      public static final String RDF_XML = contentTypeRDFXML;
57  
58      public static final MediaType RDF_XML_TYPE = typeFromString(RDF_XML);
59  
60      public static final String NTRIPLES = contentTypeNTriples;
61  
62      public static final MediaType NTRIPLES_TYPE = typeFromString(NTRIPLES);
63  
64      public final static String JSON_LD = "application/ld+json";
65  
66      public final static MediaType JSON_LD_TYPE = typeFromString(JSON_LD);
67  
68      public static final List<Variant> POSSIBLE_RDF_VARIANTS = mediaTypes(
69              RDF_XML_TYPE, TURTLE_TYPE, N3_TYPE, N3_ALT2_TYPE, NTRIPLES_TYPE, APPLICATION_XML_TYPE,
70              TEXT_PLAIN_TYPE, TURTLE_X_TYPE, JSON_LD_TYPE).add().build();
71  
72      public static final String POSSIBLE_RDF_RESPONSE_VARIANTS_STRING[] = {
73          TURTLE, N3, N3_ALT2, RDF_XML, NTRIPLES, TEXT_PLAIN, APPLICATION_XML, TURTLE_X, JSON_LD };
74  
75      private static MediaType typeFromString(final String type) {
76          return new MediaType(type.split("/")[0], type.split("/")[1]);
77      }
