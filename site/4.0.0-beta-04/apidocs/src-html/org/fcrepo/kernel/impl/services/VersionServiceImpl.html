016package org.fcrepo.kernel.impl.services;
017
018import org.fcrepo.kernel.impl.FedoraBinaryImpl;
019import org.fcrepo.kernel.services.VersionService;
020import org.slf4j.Logger;
021import org.springframework.stereotype.Component;
022
023import javax.jcr.Node;
024import javax.jcr.PathNotFoundException;
025import javax.jcr.RepositoryException;
026import javax.jcr.Session;
027import javax.jcr.Workspace;
028import javax.jcr.version.Version;
029import javax.jcr.version.VersionException;
030import javax.jcr.version.VersionHistory;
031import javax.jcr.version.VersionIterator;
032import javax.jcr.version.VersionManager;
033
034import static org.fcrepo.jcr.FedoraJcrTypes.VERSIONABLE;
035import static org.slf4j.LoggerFactory.getLogger;
036
037/**
038 * This service exposes management of node versioning.  Instead of invoking
039 * the JCR VersionManager methods, this provides a level of indirection that
040 * allows for special handling of features built on top of JCR such as user
041 * transactions.
042 * @author Mike Durbin
043 */
044
045@Component
046public class VersionServiceImpl extends AbstractService implements VersionService {
047
048    private static final Logger LOGGER = getLogger(VersionService.class);
049
050    @Override
051    public String createVersion(final Session session,
052                              final String absPath) throws RepositoryException {
053        final Node node = session.getNode(absPath);
054        if (!isVersioningEnabled(node)) {
055            enableVersioning(node);
056        }
057        return checkpoint(session, absPath);
058    }
059
060    @Override
061    public void revertToVersion(final Session session, final String absPath,
062                                final String label) throws RepositoryException {
063        final Workspace workspace = session.getWorkspace();
064        final Version v = getVersionForLabel(workspace, absPath, label);
065        if (v == null) {
066            throw new PathNotFoundException("Unknown version \"" + label + "\"!");
067        }
068        final VersionManager versionManager = workspace.getVersionManager();
069        versionManager.checkin(absPath);
070        versionManager.restore(v, true);
071        versionManager.checkout(absPath);
072    }
073
074    @Override
075    public void removeVersion(final Session session, final String absPath,
076                              final String label) throws RepositoryException {
077        final Workspace workspace = session.getWorkspace();
078        final Version v = getVersionForLabel(workspace, absPath, label);
079
080        if (v == null) {
081            throw new PathNotFoundException("Unknown version \"" + label + "\"!");
082        } else if (workspace.getVersionManager().getBaseVersion(absPath).equals(v) ) {
083            throw new VersionException("Cannot remove current version");
084        } else {
085            // remove labels
086            final VersionHistory history = v.getContainingHistory();
087            final String[] versionLabels = history.getVersionLabels(v);
088            for ( final String versionLabel : versionLabels ) {
089                LOGGER.debug("Removing label: {}", versionLabel);
090                history.removeVersionLabel( versionLabel );
091            }
092            history.removeVersion( v.getName() );
093        }
094    }
095
096
097    private static Version getVersionForLabel(final Workspace workspace, final String absPath,
098                                       final String label) throws RepositoryException {
099        // first see if there's a version label
100        final VersionHistory history = workspace.getVersionManager().getVersionHistory(absPath);
101
102        if (history.hasVersionLabel(label)) {
103            return history.getVersionByLabel(label);
104        }
105        // there was no version with the given JCR Version Label, check to see if
106        // there's a version whose UUID is equal to the label
107        final VersionIterator versionIt = history.getAllVersions();
108        if (versionIt == null) {
109            return null;
110        }
111        while (versionIt.hasNext()) {
112            final Version v = versionIt.nextVersion();
113            if (v.getFrozenNode().getIdentifier().equals(label)) {
114                return v;
115            }
116        }
117        return null;
118    }
119
120    private static boolean isVersioningEnabled(final Node n) throws RepositoryException {
121        return n.isNodeType(VERSIONABLE) || (FedoraBinaryImpl.hasMixin(n) && isVersioningEnabled(n.getParent()));
122    }
123
124    private static void enableVersioning(final Node node) throws RepositoryException {
125        node.addMixin(VERSIONABLE);
126
127        if (FedoraBinaryImpl.hasMixin(node)) {
128            node.getParent().addMixin(VERSIONABLE);
129        }
130        node.getSession().save();
131    }
132
133    private static String checkpoint(final Session session, final String absPath) throws RepositoryException {
134        LOGGER.trace("Setting implicit version checkpoint set for {}", absPath);
135        final Workspace workspace = session.getWorkspace();
136        final Version v = workspace.getVersionManager().checkpoint(absPath);
137        return ( v == null ) ? null : v.getFrozenNode().getIdentifier();
138    }
139
140}
