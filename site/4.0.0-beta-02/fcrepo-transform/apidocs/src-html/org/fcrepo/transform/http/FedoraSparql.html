103@Component

106public class FedoraSparql extends AbstractResource {
107
108    @InjectedSession
109    protected Session session;
110
111    private static final Logger LOGGER = getLogger(FedoraSparql.class);
112
113    /**
114     * A stub method to return sparql service description for fcr:sparql GET
115     *
116     * @return
117     * @throws IOException
118     */
119    @GET
120    @Timed
121    @Produces({RDF_XML, TURTLE, N3, N3_ALT2, NTRIPLES, TEXT_PLAIN, APPLICATION_XML, TURTLE_X, JSON_LD})
122    public Response sparqlServiceDescription(@Context final Request request,
123                                             @Context final UriInfo uriInfo)
124            throws IOException {
125
126        final SparqlServiceDescription sd = new SparqlServiceDescription(session, uriInfo);
127        final Variant bestPossibleResponse = request.selectVariant(POSSIBLE_RDF_VARIANTS);
128
129        LOGGER.debug("Getting sparql service description with media type {} ...", bestPossibleResponse);
130
131        Lang tmpLang;
132        if (bestPossibleResponse == null ||
133                (tmpLang = RDFLanguages.contentTypeToLang(bestPossibleResponse.getMediaType().toString())) == null) {
134            // set default format to rdf/xml
135            tmpLang = RDFLanguages.RDFXML;
136        }
137        final Lang rdfLang = tmpLang;
138        final StreamingOutput stream = new StreamingOutput() {
139            @Override
140            public void write(final OutputStream output) throws IOException {
141
142                LOGGER.debug("Writting sparql service description with jena RdfLanguages name {}.", rdfLang.getName());
143                final Writer outWriter = new OutputStreamWriter(output);
144                sd.createServiceDescription().asModel().write(outWriter, rdfLang.getName());
145            }
146        };
147        return ok(stream).header("Content-Type", rdfLang.getContentType().getContentType()).build();
148    }
149
150    /**
151     * A stub method so we can return a text/html representation using
152     * the right template.
153     *
154     * @return html representation of the query form
155     * @throws RepositoryException
156     */
157    @GET
158    @Timed
159    @Produces({TEXT_HTML})
160    public Response sparqlQueryForm() throws IOException, RepositoryException {
161
162        final Properties properties = new Properties();
163        final URL propertiesUrl =
164            getClass().getResource(velocityPropertiesLocation);
165        final VelocityEngine velocity = new VelocityEngine();
166        LOGGER.debug("Using Velocity configuration from {}", propertiesUrl);
167        try (final InputStream propertiesStream = propertiesUrl.openStream()) {
168            properties.load(propertiesStream);
169        }
170        velocity.init(properties);
171        final Template template =
172            velocity.getTemplate(templatesLocation + "/search-sparql"
173                    + templateFilenameExtension);
174        final org.apache.velocity.context.Context context =
175            new VelocityContext();
176        context.put("uriInfo", uriInfo);
177        context.put("model", new NamespaceRdfContext(session).asModel());
178        context.put("helpers", ViewHelpers.getInstance());
179
180        final StreamingOutput stream = new StreamingOutput() {
181            @Override
182            public void write(final OutputStream output) throws IOException {
183
184                final Writer outWriter = new OutputStreamWriter(output);
185                template.merge(context, outWriter);
186                outWriter.flush();
187            }
188        };
189
190        return ok(stream).build();
191    }

193    /**
194     * Execute a SPARQL query against the JCR index
195     * @param requestBodyStream
196     * @param uriInfo
197     * @return SPARQL query results
198     * @throws IOException
199     * @throws RepositoryException
200     */
201    @POST
202    @Produces({contentTypeTextTSV, contentTypeTextCSV, contentTypeSSE,
203            contentTypeTextPlain, contentTypeResultsJSON,
204            contentTypeResultsXML, contentTypeResultsBIO, contentTypeTurtle,
205            contentTypeN3, contentTypeNTriples, contentTypeRDFXML})
206    public Response runSparqlQuery(final InputStream requestBodyStream,
207        @Context final Request request, @Context final UriInfo uriInfo)
208        throws IOException, RepositoryException {
209
210        final IdentifierTranslator graphSubjects = new HttpIdentifierTranslator(session, FedoraNodes.class, uriInfo);
211
212        final Variant bestPossibleResponse =
213            request.selectVariant(POSSIBLE_SPARQL_RDF_VARIANTS);
214
215        final String sparqlQuery = IOUtils.toString(requestBodyStream);
216
217        return rexecSparql(sparqlQuery, bestPossibleResponse, graphSubjects);
218    }
219
220    /**
221     * Execute a SPARQL query against the JCR index
222     * @param query
223     * @param request
224     * @param uriInfo
225     * @return SPARQL query results
226     * @throws IOException
227     * @throws RepositoryException
228     */
229    @POST
230    @Consumes({contentTypeHTMLForm})
231    @Produces({contentTypeTextTSV, contentTypeTextCSV, contentTypeSSE,
232            contentTypeTextPlain, contentTypeResultsJSON,
233            contentTypeResultsXML, contentTypeResultsBIO, contentTypeTurtle,
234            contentTypeN3, contentTypeNTriples, contentTypeRDFXML})
235    public Response runSparqlQuery(@FormParam("query") final String query,
236                                   @Context final Request request,
237                                   @Context final UriInfo uriInfo)
238            throws IOException, RepositoryException {
239
240        LOGGER.trace("POST SPARQL query with {}: {}", contentTypeHTMLForm, query);
241        if (Strings.isNullOrEmpty(query)) {
242            return status(BAD_REQUEST)
243                    .entity("SPARQL must not be null. Please submit a query with parameter 'query'.")
244                    .build();
245        }
246        return rexecSparql(query,
247                           request.selectVariant(POSSIBLE_SPARQL_RDF_VARIANTS),
248                           new HttpIdentifierTranslator(session, FedoraNodes.class, uriInfo));
249    }
250
251    private Response rexecSparql(final String sparql,
252                                 final Variant bestPossibleResponse,
253                                 final IdentifierTranslator graphSubjects)
254            throws RepositoryException {
255        LOGGER.trace("Running SPARQL query: {}", sparql);
256
257        final JQLConverter jqlConverter = new JQLConverter(session, graphSubjects, sparql);
258
259        LOGGER.trace("Converted to JQL query: {}", jqlConverter.getStatement());
260
261        final ResultSet resultSet = jqlConverter.execute();
262
263        final ResultSetStreamingOutput streamingOutput =
264            new ResultSetStreamingOutput(resultSet, bestPossibleResponse.getMediaType());
265
266        addCallback(streamingOutput, new LogoutCallback(session));
267
268        return ok(streamingOutput).build();
269    }
