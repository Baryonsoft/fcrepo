016package org.fcrepo.kernel.impl.identifiers;
017
018import static com.google.common.base.Joiner.on;
019import static com.google.common.base.Splitter.fixedLength;
020import static com.google.common.collect.Iterables.concat;
021import static com.google.common.collect.Iterables.getLast;
022import static com.google.common.collect.Iterables.transform;
023import static com.google.common.collect.Lists.transform;
024import static java.util.Arrays.asList;
025import static java.util.Collections.emptyList;
026import static java.util.Collections.singletonList;
027import static java.util.UUID.randomUUID;
028import static org.fcrepo.jcr.FedoraJcrTypes.FCR_CONTENT;
029import static org.modeshape.jcr.api.JcrConstants.JCR_CONTENT;
030import static org.slf4j.LoggerFactory.getLogger;
031
032import java.util.List;
033
034import org.fcrepo.kernel.identifiers.InternalIdentifierConverter;
035import org.slf4j.Logger;
036
037import com.google.common.base.Function;
038
039/**
040 * Injects and extracts segments of hierarchy before the last segment of a
041 * multi-part identifier to ensure efficient performance of the JCR.
042 *
043 * @author ajs6f
044 * @since Mar 26, 2014
045 */
046public class HierarchyConverter extends InternalIdentifierConverter {
047
048    public static final String DEFAULT_SEPARATOR = "/";
049
050    private String separator = DEFAULT_SEPARATOR;
051
052    private String prefix = "";
053
054    private final Function<String, String> addPrefix = new Function<String, String>() {
055
056        @Override
057        public String apply(final String input) {
058            return prefix + input;
059        }
060    };
061
062    private static final int DEFAULT_LENGTH = 2;
063
064    private static final int DEFAULT_COUNT = 4;
065
066    private int length = DEFAULT_LENGTH;
067
068    private int levels = DEFAULT_COUNT;
069
070    private static final Logger log = getLogger(HierarchyConverter.class);
071
072    /*
073     * (non-Javadoc)
074     * @see com.google.common.base.Converter#doBackward(java.lang.Object)
075     */
076    @Override
077    protected String doBackward(final String flat) {
078        log.debug("Converting incoming identifier: {}", flat);
079        String nonContentSuffixed = flat;
080        // strip content-indicating suffix
081        if (flat.endsWith(FCR_CONTENT)) {
082            nonContentSuffixed = flat.substring(0, flat.length() - (FCR_CONTENT.length()));
083        }
084        final List<String> hierarchySegments = createHierarchySegments();
085        final List<String> flatSegments = asList(nonContentSuffixed.split(separator));
086        List<String> firstSegments = emptyList();
087        List<String> lastSegment = emptyList();
088        if (flatSegments.size() == 0) {
089            // either empty identifier or separator identifier
090            return on(separator).join(hierarchySegments);
091        }
092        if (flatSegments.size() > 1) {
093            lastSegment = singletonList(getLast(flatSegments));
094            firstSegments = flatSegments.subList(0, flatSegments.size() - 1);
095        } else {
096            // just one segment
097            lastSegment = singletonList(flatSegments.get(0));
098        }
099        final Iterable<String> allSegments = concat(firstSegments, hierarchySegments, lastSegment);
100        if (flat.endsWith(FCR_CONTENT)) {
101            return on(separator).join(concat(allSegments, singletonList(JCR_CONTENT)));
102        }
103        return on(separator).join(allSegments);
104    }
105
106    /*
107     * (non-Javadoc)
108     * @see com.google.common.base.Converter#doForward(java.lang.Object)
109     */
110    @Override
111    protected String doForward(final String hierarchical) {
112        log.debug("Converting outgoing identifier: {}", hierarchical);
113        String nonContentSuffixed = hierarchical;
114        // strip content-indicating suffix
115        if (hierarchical.endsWith(JCR_CONTENT)) {
116            nonContentSuffixed = hierarchical.substring(0, hierarchical.length() - (JCR_CONTENT.length()));
117        }
118        final List<String> segments = asList(nonContentSuffixed.split(separator));
119        if (segments.size() <= levels) {
120            // must be a root identifier
121            return "";
122        }
123        List<String> firstSegments = emptyList();
124        List<String> lastSegment = emptyList();
125        if (segments.size() > levels + 1) {
126            // we subtract one for the final segment, then levels for the
127            // inserted hierarchy segments we want to remove
128            firstSegments = segments.subList(0, segments.size() - 1 - levels);
129            lastSegment = singletonList(getLast(segments));
130        } else {
131            // just the trailing non-hierarchical segment
132            lastSegment = singletonList(getLast(segments));
133        }
134        if (hierarchical.endsWith(JCR_CONTENT)) {
135            return on(separator).join(concat(firstSegments, lastSegment, singletonList(FCR_CONTENT)));
136        }
137        return on(separator).join(concat(firstSegments, lastSegment));
138    }
139
140    private List<String> createHierarchySegments() {
141        // offers a list of segments sliced out of a UUID, each prefixed
142        if (levels == 0) {
143            return emptyList();
144        }
145        return transform(fixedLength(length).splitToList(createHierarchyCharacterBlock()), addPrefix);
146    }
147
148    private CharSequence createHierarchyCharacterBlock() {
149        return randomUUID().toString().replace("-", "").substring(0, length * levels);
150    }
151
152    /**
153     * @param sep the separator to use
154     */
155    public void setSeparator(final String sep) {
156        this.separator = sep;
157    }
158
159    /**
160     * @param l the length to set
161     */
162    public void setLength(final int l) {
163        if (l < 1) {
164            throw new IllegalArgumentException("Segment length must be at least one!");
165        }
166        this.length = l;
167    }
168
169    /**
170     * @param l the levels to set
171     */
172    public void setLevels(final int l) {
173        this.levels = l;
174    }
175
176    /**
177     * @param p the prefix to set
178     */
179    public void setPrefix(final String p) {
180        this.prefix = p;
181    }
182
183}
