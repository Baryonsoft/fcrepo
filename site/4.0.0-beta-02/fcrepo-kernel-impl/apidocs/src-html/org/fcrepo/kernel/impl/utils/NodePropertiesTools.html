042/**
043 * Tools for replacing, appending and deleting JCR node properties
044 * @author Chris Beer
045 * @since May 10, 2013
046 */
047public class NodePropertiesTools {
048
049    private static final Logger LOGGER = getLogger(NodePropertiesTools.class);
050    public static final String REFERENCE_PROPERTY_SUFFIX = "_ref";
051
052    /**
053     * Given a JCR node, property and value, either:
054     *  - if the property is single-valued, replace the existing property with
055     *    the new value
056     *  - if the property is multivalued, append the new value to the property
057     * @param node the JCR node
058     * @param propertyName a name of a JCR property (either pre-existing or
059     *   otherwise)
060     * @param newValue the JCR value to insert
061     * @throws RepositoryException
062     */
063    public void appendOrReplaceNodeProperty(final IdentifierTranslator subjects,
064                                                   final Node node,
065                                                   final String propertyName,
066                                                   final Value newValue)
067        throws RepositoryException {
068
069        // if it already exists, we can take some shortcuts
070        if (node.hasProperty(propertyName)) {
071
072            final Property property = node.getProperty(propertyName);
073
074            if (property.isMultiple()) {
075                LOGGER.debug("Appending value {} to {} property {}", newValue,
076                             PropertyType.nameFromValue(property.getType()),
077                             propertyName);
078
079                // if the property is multi-valued, go ahead and append to it.
080                final List<Value> newValues = new ArrayList<>();
081                Collections.addAll(newValues,
082                                   node.getProperty(propertyName).getValues());
083
084                if (!newValues.contains(newValue)) {
085                    newValues.add(newValue);
086                    property.setValue(toArray(newValues, Value.class));
087                }
088
089                addReferencePlaceholders(subjects, node, property, newValue);
090            } else {
091                // or we'll just overwrite it
092                LOGGER.debug("Overwriting {} property {} with new value {}",
093                             PropertyType.nameFromValue(property.getType()),
094                             propertyName, newValue);
095                property.setValue(newValue);
096                addReferencePlaceholders(subjects, node, property, newValue);
097            }
098        } else {
099            boolean isMultiple = true;
100            try {
101                isMultiple = isMultivaluedProperty(node, propertyName);
102
103            } catch (final NoSuchPropertyDefinitionException e) {
104                // simply represents a new kind of property on this node
105            }
106            if (isMultiple) {
107                LOGGER.debug("Creating new multivalued {} property {} with " +
108                             "initial value [{}]",
109                             PropertyType.nameFromValue(newValue.getType()),
110                             propertyName, newValue);
111                final Property property = node.setProperty(propertyName, new Value[]{newValue}, newValue.getType());
112                addReferencePlaceholders(subjects, node, property, newValue);
113            } else {
114                LOGGER.debug("Creating new single-valued {} property {} with " +
115                             "initial value {}",
116                             PropertyType.nameFromValue(newValue.getType()),
117                             propertyName, newValue);
118                final Property property = node.setProperty(propertyName, newValue, newValue.getType());
119                addReferencePlaceholders(subjects, node, property, newValue);
120            }
121        }
122
123    }
124
125    private void addReferencePlaceholders(final IdentifierTranslator subjects,
126                                          final Node node,
127                                          final Property property,
128                                          final Value newValue) throws RepositoryException {
129        if (property.getType() == URI) {
130            final Resource resource = ResourceFactory.createResource(newValue.getString());
131
132            if (subjects.isFedoraGraphSubject(resource)) {
133                final Node refNode = node.getSession().getNode(subjects.getPathFromSubject(resource));
134                final String referencePropertyName = getReferencePropertyName(property);
135
136                if (!property.isMultiple() && node.hasProperty(referencePropertyName)) {
137                    node.setProperty(referencePropertyName, (Value[])null);
138                }
139
140                final Value v = node.getSession().getValueFactory().createValue(refNode, true);
141                appendOrReplaceNodeProperty(subjects, node, referencePropertyName, v);
142            }
143        }
144    }
145
146    private void removeReferencePlaceholders(final IdentifierTranslator subjects,
147                                             final Node node,
148                                             final Property property,
149                                             final Value newValue) throws RepositoryException {
150        if (property.getType() == URI) {
151            final Resource resource = ResourceFactory.createResource(newValue.getString());
152
153            if (subjects.isFedoraGraphSubject(resource)) {
154                final String referencePropertyName = getReferencePropertyName(property);
155
156                if (!property.isMultiple() && node.hasProperty(referencePropertyName)) {
157                    node.setProperty(referencePropertyName, (Value[])null);
158                } else {
159                    final Node refNode = node.getSession().getNode(subjects.getPathFromSubject(resource));
160                    final Value v = node.getSession().getValueFactory().createValue(refNode, true);
161                    removeNodeProperty(subjects, node, referencePropertyName, v);
162                }
163            }
164        }
165    }
166    /**
167     * Given a JCR node, property and value, remove the value (if it exists)
168     * from the property, and remove the
169     * property if no values remove
170     *
171     * @param node the JCR node
172     * @param propertyName a name of a JCR property (either pre-existing or
173     *   otherwise)
174     * @param valueToRemove the JCR value to remove
175     * @throws RepositoryException
176     */
177    public void removeNodeProperty(final IdentifierTranslator subjects,
178                                          final Node node,
179                                          final String propertyName,
180                                          final Value valueToRemove)
181        throws RepositoryException {
182        // if the property doesn't exist, we don't need to worry about it.
183        if (node.hasProperty(propertyName)) {
184
185            final Property property = node.getProperty(propertyName);
186
187            if (FedoraTypesUtils.isMultipleValuedProperty.apply(property)) {
188
189                final List<Value> newValues = new ArrayList<>();
190
191                boolean remove = false;
192
193                for (final Value v : node.getProperty(propertyName).getValues()) {
194                    if (v.equals(valueToRemove)) {
195                        remove = true;
196                    } else {
197                        newValues.add(v);
198                    }
199                }
200
201                // we only need to update the property if we did anything.
202                if (remove) {
203                    if (newValues.isEmpty()) {
204                        LOGGER.debug("Removing property {}", propertyName);
205                        property.setValue((Value[])null);
206                    } else {
207                        LOGGER.debug("Removing value {} from property {}",
208                                     valueToRemove, propertyName);
209                        property
210                            .setValue(toArray(newValues, Value.class));
211                    }
212                    removeReferencePlaceholders(subjects, node, property, valueToRemove);
213                }
214
215            } else {
216                if (property.getValue().equals(valueToRemove)) {
217                    LOGGER.debug("Removing value from property {}", propertyName);
218                    property.setValue((Value)null);
219
220                    if (property.getType() == URI && node.hasProperty(getReferencePropertyName(propertyName))) {
221                        removeReferencePlaceholders(subjects, node, property, valueToRemove);
222                    }
223                }
224            }
225        }
226    }

228    /**
229     * When we add certain URI properties, we also want to leave a reference node
230     * @param propertyName
231     * @return property name as a reference
232     */
233    public static String getReferencePropertyName(final String propertyName) {
234        return propertyName + REFERENCE_PROPERTY_SUFFIX;
235    }
236
237    /**
238     * Given an internal reference node property, get the original name
239     * @param refPropertyName
240     * @return original property name of the reference property
241     */
242    public static String getReferencePropertyOriginalName(final String refPropertyName) {
243        final int i = refPropertyName.lastIndexOf(REFERENCE_PROPERTY_SUFFIX);
244
245        if (i < 0) {
246            return refPropertyName;
247        } else {
248            return refPropertyName.substring(0, i);
249        }
250    }
251
252    private static String getReferencePropertyName(final Property property) throws RepositoryException {
253        return getReferencePropertyName(property.getName());
254    }
255    /**
256     * Get the JCR property type ID for a given property name. If unsure, mark
257     * it as UNDEFINED.
258     *
259     * @param node the JCR node to add the property on
260     * @param propertyName the property name
261     * @return a PropertyType value
262     * @throws RepositoryException
263     */
264    public int getPropertyType(final Node node, final String propertyName)
265        throws RepositoryException {
266        LOGGER.debug("Getting type of property: {} from node: {}",
267                propertyName, node);
268        final PropertyDefinition def =
269            getDefinitionForPropertyName(node, propertyName);
270
271        if (def == null) {
272            return UNDEFINED;
273        }
274
275        return def.getRequiredType();
276    }
277
278    /**
279     * Determine if a given JCR property name is single- or multi- valued.
280     * If unsure, choose the least restrictive
281     * option (multivalued)
282     *
283     * @param node the JCR node to check
284     * @param propertyName the property name
285     *   (which may or may not already exist)
286     * @return true if the property is (or could be) multivalued
287     * @throws RepositoryException
288     */
289    public static boolean isMultivaluedProperty(final Node node,
290                                                final String propertyName)
291        throws RepositoryException {
292        final PropertyDefinition def =
293            getDefinitionForPropertyName(node, propertyName);
294
295        if (def == null) {
296            throw new NoSuchPropertyDefinitionException();
297        }
298
299        return def.isMultiple();
300    }
