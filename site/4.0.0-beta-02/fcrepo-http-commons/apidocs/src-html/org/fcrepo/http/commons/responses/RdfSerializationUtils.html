016package org.fcrepo.http.commons.responses;

018import static com.google.common.collect.ImmutableList.of;
019import static com.hp.hpl.jena.graph.Node.ANY;
020import static com.hp.hpl.jena.graph.NodeFactory.createURI;
021import static javax.ws.rs.core.HttpHeaders.CACHE_CONTROL;
022import static javax.ws.rs.core.HttpHeaders.LAST_MODIFIED;
023import static org.fcrepo.kernel.RdfLexicon.JCR_NAMESPACE;
024import static org.fcrepo.kernel.impl.rdf.JcrRdfTools.getRDFNamespaceForJcrNamespace;
025import static org.joda.time.format.DateTimeFormat.forPattern;
026import static org.slf4j.LoggerFactory.getLogger;
027
028import java.util.Iterator;
029import java.util.List;
030import java.util.Locale;
031
032import javax.ws.rs.core.MultivaluedMap;
033import javax.ws.rs.core.PathSegment;
034import javax.ws.rs.core.UriInfo;
035
036import org.fcrepo.kernel.rdf.GraphProperties;
037import org.joda.time.DateTime;
038import org.joda.time.DateTimeZone;
039import org.joda.time.format.DateTimeFormatter;
040import org.slf4j.Logger;
041
042import com.hp.hpl.jena.datatypes.xsd.XSDDateTime;
043import com.hp.hpl.jena.graph.Node;
044import com.hp.hpl.jena.query.Dataset;
045import com.hp.hpl.jena.sparql.core.Quad;
046import com.hp.hpl.jena.sparql.util.Context;
047
048/**
049 * Utilities to help with serializing a graph to an HTTP resource
050 *
051 * @author awoods
052 */
053public class RdfSerializationUtils {
054
055    private static final Logger LOGGER = getLogger(RdfSerializationUtils.class);
056
057    /**
058     * No public constructor on utility class
059     */
060    private RdfSerializationUtils() {
061    }
062
063    /**
064     * The RDF predicate that will indicate the primary node type.
065     */
066    public static Node primaryTypePredicate =
067            createURI(getRDFNamespaceForJcrNamespace(JCR_NAMESPACE) +
068                    "primaryType");
069
070    /**
071     * The RDF predicate that will indicate the last-modified date of the node.
072     */
073    public static Node lastModifiedPredicate =
074            createURI(getRDFNamespaceForJcrNamespace(JCR_NAMESPACE) +
075                    "lastModified");
076
077    /**
078     * DateTimeFormatter for RFC2822 (used in HTTP headers), e.g.:
079     *    Mon, 01 Jul 2013 07:51:23Z
080     */
081    protected static DateTimeFormatter RFC2822DATEFORMAT =
082            forPattern("EEE, dd MMM yyyy HH:mm:ss Z").withLocale(Locale.US)
083                    .withZone(DateTimeZone.forID("GMT"));
084
085    /**
086     * Get the very first value for a predicate as a string, or null if the
087     * predicate is not used
088     *
089     * @param rdf
090     * @param subject
091     * @param predicate
092     * @return first value for the given predicate or null if not found
093     */
094    public static String getFirstValueForPredicate(final Dataset rdf,
095            final Node subject, final Node predicate) {
096        final Iterator<Quad> statements =
097                rdf.asDatasetGraph().find(ANY, subject, predicate, ANY);
098        // we'll take the first one we get
099        if (statements.hasNext()) {
100            final Quad statement = statements.next();
101            LOGGER.trace("Checking statement: {}", statement);
102            return statement.asTriple().getObject().getLiteral()
103                    .getLexicalForm();
104        }
105        LOGGER.trace("No value found for predicate: {}", predicate);
106        return null;
107    }
108
109    /**
110     * Get the subject of the dataset, given by the context's "uri"
111     *
112     * @param rdf
113     * @return subject of the dataset
114     */
115    static Node getDatasetSubject(final Dataset rdf) {
116        final Context context = rdf.getContext();
117        final String uri = context.getAsString(GraphProperties.URI_SYMBOL);
118        LOGGER.debug("uri from context: {}", uri);
119        if (uri != null) {
120            return createURI(uri);
121        }
122        return null;
123    }
124
125    /**
126     * Set the cache control and last modified HTTP headers from data in the
127     * graph
128     *
129     * @param httpHeaders
130     * @param rdf
131     */
132    public static void setCachingHeaders(final MultivaluedMap<String, Object> httpHeaders,
133                                         final Dataset rdf,
134                                         final UriInfo uriInfo) {
135
136        final List<PathSegment> segments = uriInfo.getPathSegments();
137        if (!segments.isEmpty() && segments.get(0).getPath().startsWith("tx:")) {
138            // Do not set caching headers if we are in a transaction
139            return;
140        }
141
142        httpHeaders.put(CACHE_CONTROL, of((Object)"max-age=0", (Object)"must-revalidate"));
143
144        LOGGER.trace("Attempting to discover the last-modified date of the node for the resource in question...");
145        final Iterator<Quad> iterator =
146            rdf.asDatasetGraph().find(ANY, getDatasetSubject(rdf),
147                    lastModifiedPredicate, ANY);
148
149        if (!iterator.hasNext()) {
150            return;
151        }
152
153        final Object dateObject = iterator.next().getObject().getLiteralValue();
154
155        if (!(dateObject instanceof XSDDateTime)) {
156            LOGGER.debug("Found last-modified date, but it was not an XSDDateTime: {}", dateObject);
157
158            return;
159        }
160
161        final XSDDateTime lastModified = (XSDDateTime) dateObject;
162        LOGGER.debug("Found last-modified date: {}", lastModified);
163        final String lastModifiedAsRdf2822 =
164                RFC2822DATEFORMAT
165                        .print(new DateTime(lastModified.asCalendar()));
166        httpHeaders.put(LAST_MODIFIED, of((Object) lastModifiedAsRdf2822));
167    }
