016package org.fcrepo.kernel.services;
017
018import java.io.InputStream;
019import java.net.URI;
020import java.util.Collection;
021
022import javax.jcr.Node;
023import javax.jcr.RepositoryException;
024import javax.jcr.Session;
025
026import org.fcrepo.kernel.Datastream;
027import org.fcrepo.kernel.exception.InvalidChecksumException;
028import org.fcrepo.kernel.exception.ResourceTypeException;
029import org.fcrepo.kernel.rdf.IdentifierTranslator;
030import org.fcrepo.kernel.utils.FixityResult;
031import org.fcrepo.kernel.utils.iterators.RdfStream;
032
033/**
034 * @author bbpennel
035 * @since Feb 20, 2014
036 */
037public interface DatastreamService extends Service {
038
039    /**
040     * Create a stub datastream without content
041     * @param session
042     * @param dsPath
043     * @return created datastream
044     * @throws RepositoryException
045     */
046    Datastream createDatastream(Session session, String dsPath) throws RepositoryException;
047
048    /**
049     * Create a new Datastream node in the repository
050     * 
051     * @param session
052     * @param dsPath the absolute path to put the datastream
053     * @param contentType the mime-type for the requestBodyStream
054     * @param originalFileName the original file name for the input stream
055     * @param requestBodyStream binary payload for the datastream
056     * @return created datastream
057     * @throws RepositoryException
058     * @throws InvalidChecksumException
059     */
060    Datastream createDatastream(Session session, String dsPath, String contentType, String originalFileName,
061            InputStream requestBodyStream) throws RepositoryException, InvalidChecksumException;
062
063    /**
064     * Create a new Datastream node in the repository
065     *
066     * @param session
067     * @param dsPath the absolute path to put the datastream
068     * @param contentType the mime-type for the requestBodyStream
069     * @param originalFileName the original file name for the input stream
070     * @param requestBodyStream binary payload for the datastream
071     * @param checksum the digest for the binary payload (as urn:sha1:xyz) @return
072     * @throws RepositoryException
073     * @throws InvalidChecksumException
074     */
075    Datastream createDatastream(Session session, String dsPath, String contentType, String originalFileName,
076            InputStream requestBodyStream, URI checksum) throws RepositoryException, InvalidChecksumException;
077
078    /**
079     * Retrieve the JCR node for a Datastream by pid and dsid
080     *
081     * @param path
082     * @return retrieved jcr node
083     * @throws RepositoryException
084     */
085    Node getDatastreamNode(Session session, String path) throws RepositoryException;
086
087    /**
088     * Retrieve a Datastream instance by pid and dsid
089     *
090     * @param path jcr path to the datastream
091     * @return retrieved Datastream
092     * @throws RepositoryException
093     */
094    Datastream getDatastream(Session session, String path) throws RepositoryException;
095
096    /**
097     * Retrieve a Datastream instance by pid and dsid
098     *
099     * @param node datastream node
100     * @return node as a Datastream
101     */
102    Datastream asDatastream(Node node) throws ResourceTypeException, RepositoryException;
103
104    /**
105     * Get the fixity results for the datastream as a RDF Dataset
106     *
107     * @param subjects
108     * @param datastream
109     * @return fixity results for datastream
110     * @throws RepositoryException
111     */
112    RdfStream getFixityResultsModel(IdentifierTranslator subjects, Datastream datastream) throws RepositoryException;
113
114    /**
115     * Run the fixity check on the datastream and attempt to automatically
116     * correct failures if additional copies of the bitstream are available
117     *
118     * @param datastream
119     * @return results
120     * @throws RepositoryException
121     */
122    Collection<FixityResult> runFixityAndFixProblems(Datastream datastream) throws RepositoryException;
123
124    /**
125     * Get the fixity results for this datastream's bitstream, and compare it
126     * against the given checksum and size.
127     *
128     * @param resource
129     * @param dsChecksum -the checksum and algorithm represented as a URI
130     * @param dsSize
131     * @return fixity results for datastream's bitstream
132     * @throws RepositoryException
133     */
134    Collection<FixityResult> getFixity(Node resource, URI dsChecksum, long dsSize) throws RepositoryException;
135
136}
