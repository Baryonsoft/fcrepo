048/**
049 * Listen to Jena statement events, and when the statement is changed in the
050 * graph store, make the change within JCR as well.
051 *
052 * @author awoods
053 */
054public class JcrPropertyStatementListener extends StatementListener {
055
056    private static final Logger LOGGER = getLogger(JcrPropertyStatementListener.class);
057
058    private final JcrRdfTools jcrRdfTools;
059
060    private final HashMap<AnonId, Node> skolemizedBnodeMap;
061
062    private NodePropertiesTools propertiesTools = new NodePropertiesTools();
063
064    private Model problems;
065
066    private final IdentifierTranslator subjects;
067
068    private final Session session;
069
070    private static final JcrTools jcrTools = new JcrTools();
071
072    /**
073     * Return a Listener given the subject factory and JcrSession.
074     * @param subjects
075     * @param session
076     * @param problemModel
077     * @return JcrPropertyStatementListener for the given subject factory and JcrSession
078     */
079    public static JcrPropertyStatementListener getListener(
080            final IdentifierTranslator subjects, final Session session, final Model problemModel) {
081        return new JcrPropertyStatementListener(subjects, session, problemModel);
082    }
083
084    /**
085     * Construct a statement listener within the given session
086     *
087     * @param subjects
088     * @param session
089     */
090    private JcrPropertyStatementListener(final IdentifierTranslator subjects,
091            final Session session, final Model problems) {
092        super();
093        this.session = session;
094        this.subjects = subjects;
095        this.problems = problems;
096        this.jcrRdfTools = JcrRdfTools.withContext(subjects, session);
097        this.skolemizedBnodeMap = new HashMap<>();
098    }
099
100    /**
101     * When a statement is added to the graph, serialize it to a JCR property
102     *
103     * @param s
104     */
105    @Override
106    public void addedStatement(final Statement s) {
107        LOGGER.debug(">> added statement {}", s);
108
109        try {
110            final Resource subject = s.getSubject();
111
112            // if it's not about a node, ignore it.
113            if (!subjects.isFedoraGraphSubject(subject) && !subject.isAnon()) {
114                return;
115            }
116
117            final Node subjectNode;
118
119            if (subject.isAnon()) {
120                if (skolemizedBnodeMap.containsKey(subject.getId())) {
121                    subjectNode = skolemizedBnodeMap.get(subject.getId());
122                } else {
123                    subjectNode = jcrTools.findOrCreateNode(session, "/.well-known/genid/" + randomUUID().toString());
124                    subjectNode.addMixin("fedora:blanknode");
125                    skolemizedBnodeMap.put(subject.getId(), subjectNode);
126                }
127            } else {
128                String path = null;
129                try {
130                    path = decode(subjects.getPathFromSubject(subject), "UTF-8");
131                } catch ( UnsupportedEncodingException ex ) {
132                    LOGGER.warn("Required encoding (UTF-8) not supported, trying undecoded path",ex);
133                    path = subjects.getPathFromSubject(subject);
134                }
135                subjectNode = session.getNode(path);
136            }
137
138            // special logic for handling rdf:type updates.
139            // if the object is an already-existing mixin, update
140            // the node's mixins. If it isn't, just treat it normally.
141            if (s.getPredicate().equals(RDF.type)
142                    && s.getObject().isResource()) {
143                final Resource mixinResource = s.getObject().asResource();
144
145                if (mixinResource.getNameSpace().equals(LDP_NAMESPACE)) {
146                    return;
147                }
148
149                try {
150                    final String namespacePrefix = session.getNamespacePrefix(mixinResource.getNameSpace());
151                    final String mixinName = namespacePrefix + ":" + mixinResource.getLocalName();
152
153                    if (FedoraTypesUtils.getNodeTypeManager(subjectNode).hasNodeType(mixinName)) {
154
155                        if (subjectNode.canAddMixin(mixinName)) {
156                            subjectNode.addMixin(mixinName);
157                        } else {
158                            problems.add(subject, COULD_NOT_STORE_PROPERTY, s.getPredicate().getURI());
159                        }
160
161                        return;
162
163                    }
164                } catch (final NamespaceException e) {
165                    LOGGER.trace("Unable to resolve registered namespace for resource {}: {}", mixinResource, e);
166                }
167            }
168
169            // extract the JCR propertyName from the predicate
170            final String propertyName =
171                    jcrRdfTools.getPropertyNameFromPredicate(subjectNode,
172                                                             s.getPredicate(),
173                                                             s.getModel()
174                                                                     .getNsPrefixMap());
175
176            if (validateModificationsForPropertyName(
177                    subject, subjectNode, s.getPredicate())) {
178                final Value v =
179                    jcrRdfTools.createValue(subjectNode, s.getObject(),
180                            propertiesTools.getPropertyType(subjectNode,
181                                    propertyName));
182                propertiesTools.appendOrReplaceNodeProperty(subjects,
183                        subjectNode, propertyName, v);
184            }
185        } catch (final RepositoryException e) {
186            throw propagate(e);
187        }
188
189    }
190
191    /**
192     * When a statement is removed, remove it from the JCR properties
193     *
194     * @param s
195     */
196    @Override
197    public void removedStatement(final Statement s) {
198        LOGGER.trace(">> removed statement {}", s);
199
200        try {
201            final Resource subject = s.getSubject();
202
203            // if it's not about a node, we don't care.
204            if (!subjects.isFedoraGraphSubject(subject)) {
205                return;
206            }
207
208            final Node subjectNode = session.getNode(subjects.getPathFromSubject(subject));
209
210            // special logic for handling rdf:type updates.
211            // if the object is an already-existing mixin, update
212            // the node's mixins. If it isn't, just treat it normally.
213            if (s.getPredicate().equals(RDF.type)
214                    && s.getObject().isResource()) {
215                final Resource mixinResource = s.getObject().asResource();
216                final String errorPrefix = "Error removing node type";
217                try {
218                    final String namespacePrefix = session.getNamespacePrefix(mixinResource.getNameSpace());
219                    final String mixinName = namespacePrefix + ":" + mixinResource.getLocalName();
220
221                    if (FedoraTypesUtils.getNodeTypeManager(subjectNode).hasNodeType(mixinName)) {
222                        try {
223                            subjectNode.removeMixin(mixinName);
224                        } catch (final RepositoryException e) {
225                            LOGGER.info(
226                                    "problem with removing <{}> <{}> <{}>: {}",
227                                    subject.getURI(),
228                                    RdfLexicon.COULD_NOT_STORE_PROPERTY,
229                                    s.getPredicate().getURI(),
230                                    e);
231                            String errorMessage = e.getMessage();
232                            if (StringUtils.isNotBlank(errorMessage)) {
233                                errorMessage = errorPrefix + " '" + mixinName +
234                                        "': \n" + e.getClass().getName() + ": " + errorMessage;
235                            } else {
236                                errorMessage = errorPrefix + " '" + mixinName +
237                                        "': \n" + e.getClass().getName();
238                            }
239                            problems.add(subject, RdfLexicon.COULD_NOT_STORE_PROPERTY, errorMessage);
240                        }
241                        return;
242                    }
243
244                } catch (final NamespaceException e) {
245                    LOGGER.trace("Unable to resolve registered namespace for resource {}: {}", mixinResource, e);
246
247                    String errorMessage = e.getMessage();
248                    if (StringUtils.isNotBlank(errorMessage)) {
249                        errorMessage = errorPrefix + " " +
250                                e.getClass().getName() + ": "  + errorMessage;
251                    } else {
252                        errorMessage = errorPrefix + ": " + e.getClass().getName();
253                    }
254                    problems.add(subject, RdfLexicon.COULD_NOT_STORE_PROPERTY, errorMessage);
255                }
256
257            }
258
259            final String propertyName =
260                jcrRdfTools.getPropertyNameFromPredicate(subjectNode, s.getPredicate());
261
262
263            // if the property doesn't exist, we don't need to worry about it.
264            if (subjectNode.hasProperty(propertyName) &&
265                    validateModificationsForPropertyName(subject, subjectNode,
266                                                            s.getPredicate())) {
267                final Value v =
268                    jcrRdfTools.createValue(subjectNode, s.getObject(),
269                            propertiesTools.getPropertyType(subjectNode,
270                                    propertyName));
271                propertiesTools.removeNodeProperty(subjects, subjectNode,
272                        propertyName, v);
273            }
274
275        } catch (final RepositoryException e) {
276            throw propagate(e);
277        }
278
279    }
280
281    private boolean validateModificationsForPropertyName(
282            final Resource subject, final Node subjectNode, final Resource predicate) {
283        if (jcrRdfTools.isInternalProperty(subjectNode, predicate)) {
284            LOGGER.debug("problem with <{}> <{}> <{}>",
285                         subject.getURI(),
286                         RdfLexicon.COULD_NOT_STORE_PROPERTY,
287                         predicate.getURI());
288            problems.add(subject, RdfLexicon.COULD_NOT_STORE_PROPERTY, predicate.getURI());
289            return false;
290        }
291
292        return true;
293    }
294
295    /**
296     * Get a list of any problems from trying to apply the statement changes to
297     * the node's properties
298     *
299     * @return model containing any problems from applying the statement changes
300     */
301    public Model getProblems() {
302        return problems;
303    }
