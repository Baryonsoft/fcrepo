077    private final Timer objectSizeCalculationTimer = getMetrics().timer(
078            name(RepositoryService.class, "objectSizeCalculation"));
079
080    /**
081     * Calculate the total size of all the binary properties in the repository
082     *
083     * @return size in bytes
084     */
085    @Override
086    public Long getRepositorySize() {
087        try {
088
089            final Timer.Context context = objectSizeCalculationTimer.time();
090            LOGGER.debug("Calculating repository size from index");
091
092            try {
093                // Differentiating between the local getRepositorySize and
094                // ServiceHelpers
095                return ServiceHelpers.getRepositorySize(repo);
096
097            } finally {
098                context.stop();
099            }
100        } catch (final RepositoryException e) {
101            throw propagate(e);
102        }
103    }

119    /*
120     * (non-Javadoc)
121     * @see
122     * org.fcrepo.kernel.services.RepositoryService#getRepositoryNamespaces(
123     * javax.jcr.Session)
124     */
125    @Override
126    public Map<String, String> getRepositoryNamespaces(final Session session) throws RepositoryException {
127
128        final NamespaceRegistry reg = session.getWorkspace().getNamespaceRegistry();
129        return asMap(newHashSet(reg.getPrefixes()), new Function<String, String>() {
130
131            @Override
132            public String apply(final String p) {
133                try {
134                    return reg.getURI(p);
135                } catch (final RepositoryException e) {
136                    throw propagate(e);
137                }
138            }
139        });
140    }

150        throws RepositoryException {
151
152        final Model model =
153            JcrRdfTools.withContext(idTranslator, session).getNamespaceTriples().asModel();

157        final Dataset dataset = create(model);

177    /*
178     * (non-Javadoc)
179     * @see
180     * org.fcrepo.kernel.services.RepositoryService#searchRepository(org.fcrepo
181     * .kernel.rdf.GraphSubjects, com.hp.hpl.jena.rdf.model.Resource,
182     * javax.jcr.Session, java.lang.String, int, long)
183     */
184    @Override
185    public Dataset searchRepository(final IdentifierTranslator subjectFactory,
186            final Resource searchSubject, final Session session,
187            final String terms, final int limit, final long offset)
188        throws RepositoryException {
189
190        final Model model;
191
192        if (terms != null && terms.trim().length() != 0) {
193            final QueryManager queryManager =
194                    session.getWorkspace().getQueryManager();
195
196            final QueryObjectModelFactory factory =
197                    queryManager.getQOMFactory();
198
199            final Source selector =
200                    factory.selector(FEDORA_RESOURCE, "resourcesSelector");
201            final Constraint constraints =
202                    factory.fullTextSearch("resourcesSelector", null, factory
203                            .literal(session.getValueFactory().createValue(
204                                    terms)));
205
206            final Query query =
207                    factory.createQuery(selector, constraints, null, null);
208
209            // include an extra document to determine if additional pagination
210            // is
211            // necessary
212            query.setLimit(limit + 1);
213            query.setOffset(offset);
214
215            final QueryResult queryResult = query.execute();
216
217            final NodeIterator nodeIterator = queryResult.getNodes();
218            final long size = nodeIterator.getSize();
219
220            // remove that extra document from the nodes we'll iterate over
221            final Iterator<Node> limitedIterator =
222                    limit(new org.fcrepo.kernel.utils.iterators.NodeIterator(nodeIterator),
223                            limit);
224
225            model =
226                JcrRdfTools.withContext(subjectFactory, session)
227                        .getJcrPropertiesModel(limitedIterator, searchSubject)
228                        .asModel();
229
230            /* add the result description to the RDF model */
231
232            model.add(searchSubject, SEARCH_HAS_TOTAL_RESULTS, model
233                    .createTypedLiteral(size));
234            model.add(searchSubject, SEARCH_HAS_MORE, model
235                    .createTypedLiteral(nodeIterator.hasNext()));
236        } else {
237            model = createDefaultModel();
238        }
239
240        final Dataset dataset = DatasetFactory.create(model);
241
242        final String uri = searchSubject.getURI();
243        final Context context =
244                dataset.getContext() != null ? dataset.getContext()
245                        : new Context();
246        context.set(GraphProperties.URI_SYMBOL, uri);
247
248        return dataset;
249
250    }

258    @Override
259    public Problems backupRepository(final Session session,
260                                     final File backupDirectory) throws RepositoryException {
261        final RepositoryManager repoMgr = ((org.modeshape.jcr.api.Session) session)
262                .getWorkspace()
263                .getRepositoryManager();
264
265        final Problems problems = repoMgr.backupRepository(backupDirectory);
266
267        return problems;
