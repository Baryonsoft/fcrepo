053/**
054 * @author ajs6f
055 * @since Oct 24, 2013
056 */
057public abstract class PersistingRdfStreamConsumer implements RdfStreamConsumer {
058
059    private final RdfStream stream;
060
061    private final IdentifierTranslator idTranslator;
062
063    private final Session session;
064
065    // if it's not about a Fedora resource, we don't care.
066    protected final Predicate<Triple> isFedoraSubjectTriple;
067
068    private final Map<AnonId, Resource> skolemizedBnodeMap;
069
070    private final JcrRdfTools jcrRdfTools;
071
072    private static final JcrTools jcrTools = new JcrTools();
073
074    private static final Model m = createDefaultModel();
075
076    private static final Logger LOGGER = getLogger(PersistingRdfStreamConsumer.class);
077
078    /**
079     * Ordinary constructor.
080     *
081     * @param graphSubjects
082     * @param session
083     * @param stream
084     */
085    public PersistingRdfStreamConsumer(final IdentifierTranslator graphSubjects,
086            final Session session, final RdfStream stream) {
087        this.idTranslator = graphSubjects;
088        this.jcrRdfTools = JcrRdfTools.withContext(graphSubjects, session);
089        this.isFedoraSubjectTriple = new Predicate<Triple>() {
090
091            @Override
092            public boolean apply(final Triple t) {
093
094                final boolean result =
095                        graphSubjects.isFedoraGraphSubject(m.asStatement(t).getSubject()) || t.getSubject().isBlank();
096                if (result) {
097                    LOGGER.debug(
098                            "Discovered a Fedora-relevant subject in triple: {}.",
099                            t);
100                } else {
101                    LOGGER.debug("Ignoring triple: {}.", t);
102                }
103                return result;
104            }
105
106        };
107        this.skolemizedBnodeMap = new HashMap<>();
108        // we knock out managed RDF and non-Fedora RDF
109        this.stream =
110            stream.withThisContext(stream.filter(and(not(isManagedTriple),
111                    isFedoraSubjectTriple)).iterator());
112        this.session = session;
113    }
114
115    @Override
116    public void consume() throws RepositoryException {
117        while (stream.hasNext()) {
118            Statement t = m.asStatement(stream.next());
119            LOGGER.debug("Operating triple {}.", t);
120
121            if (t.getObject().isAnon()) {
122                t = t.changeObject(getSkolemizedResource(t.getObject()));
123            }
124
125            if (t.getSubject().isAnon()) {
126                t = m.createStatement(getSkolemizedResource(t.getSubject()), t.getPredicate(), t.getObject());
127            }
128
129            LOGGER.trace("Operating on skolemized triple {}.", t);
130
131            operateOnTriple(t);
132        }
133
134    }
135
136    private Resource getSkolemizedResource(final RDFNode resource) throws RepositoryException {
137        final AnonId id = resource.asResource().getId();
138
139        if (!skolemizedBnodeMap.containsKey(id)) {
140            final Node orCreateNode =
141                    jcrTools.findOrCreateNode(session, "/.well-known/genid/" + randomUUID().toString());
142            orCreateNode.addMixin("fedora:blanknode");
143            final Resource skolemizedSubject = idTranslator().getSubject(orCreateNode.getPath());
144            skolemizedBnodeMap.put(id, skolemizedSubject);
145        }
146
147        return skolemizedBnodeMap.get(id);
148    }
149
150    protected void operateOnTriple(final Statement t)
151        throws RepositoryException {
152        final Resource subject = t.getSubject();
153        final Node subjectNode = session().getNode(idTranslator().getPathFromSubject(subject));
154
155        // if this is a user-managed RDF type assertion, update the node's
156        // mixins. If it isn't, treat it as a "data" property.
157        if (t.getPredicate().equals(type) && t.getObject().isResource()) {
158            final Resource mixinResource = t.getObject().asResource();
159            if (!isManagedMixin.apply(mixinResource)) {
160                LOGGER.debug("Operating on node: {} with mixin: {}.",
161                        subjectNode, mixinResource);
162                operateOnMixin(mixinResource, subjectNode);
163            } else {
164                LOGGER.debug("Found repository-managed mixin on which we will not operate.");
165            }
166        } else {
167            LOGGER.debug("Operating on node: {} from triple: {}.", subjectNode,
168                    t);
169            operateOnProperty(t, subjectNode);
170        }
171    }
172
173    protected String getPropertyNameFromPredicate(final Node subjectNode,
174        final Property predicate) throws RepositoryException {
175        return jcrRdfTools().getPropertyNameFromPredicate(subjectNode, predicate,
176                stream.namespaces());
177    }
178
179    protected String getPropertyNameFromPredicate(final Node subjectNode,
180                                                  final Resource predicate) throws RepositoryException {
181        return jcrRdfTools().getPropertyNameFromPredicate(subjectNode, predicate, stream.namespaces());
182    }
183
184    protected Value createValue(final Node subjectNode, final RDFNode object,
185        final Integer propertyType) throws RepositoryException {
186        return jcrRdfTools().createValue(subjectNode, object, propertyType);
187    }
188
189    protected abstract void operateOnProperty(final Statement t,
190        final Node subjectNode) throws RepositoryException;
191
192    protected abstract void operateOnMixin(final Resource mixinResource,
193        final Node subjectNode) throws RepositoryException;
194
195    @Override
196    public ListenableFuture<Boolean> consumeAsync() {
197        // TODO make this actually asynch
198        final SettableFuture<Boolean> result = SettableFuture.create();
199        try {
200            consume();
201            result.set(true);
202        } catch (final Exception e) {
203            LOGGER.warn("Got exception consuming RDF stream", e);
204            result.setException(e);
205            result.set(false);
206        }
207        return result;
208    }
209
210
211    /**
212     * @return the stream
213     */
214    public RdfStream stream() {
215        return stream;
216    }
217
218
219    /**
220     * @return the idTranslator
221     */
222    public IdentifierTranslator idTranslator() {
223        return idTranslator;
224    }
225
226
227    /**
228     * @return the session
229     */
230    public Session session() {
231        return session;
232    }
233
234
235    /**
236     * @return the jcrRdfTools
237     */
238    public JcrRdfTools jcrRdfTools() {
239        return jcrRdfTools;
240    }
