016package org.fcrepo.kernel.utils.iterators;
017
018import static com.google.common.collect.Sets.newHashSet;
019
020import java.util.Iterator;
021import java.util.Set;
022
023import com.google.common.collect.AbstractIterator;
024
025/**
026 * A wrapping {@link Iterator} that calculates two differences between a
027 * {@link Set} A and a source Iterator B. The differences are (A - (A ∩ B)) and
028 * (B - (A ∩ B)). The ordinary output of this iterator is (B - (A ∩ B)), and
029 * after exhaustion, sets containing (A - (A ∩ B)) and (A ∩ B) are available.
030 *
031 * @author ajs6f
032 * @since Oct 24, 2013
033 * @param <E> The type of element common to both source and set.
034 */
035public class DifferencingIterator<E> extends AbstractIterator<E> {
036
037    Set<? extends E> notCommon;
038
039    private Set<E> common;
040
041    private Iterator<E> source;
042
043    /**
044     * Ordinary constructor.
045     *
046     * @param original
047     * @param source
048     */
049    public DifferencingIterator(final Set<? extends E> original,
050            final Iterator<E> source) {
051        super();
052        this.notCommon = newHashSet(original);
053        this.common = newHashSet();
054        this.source = source;
055    }
056
057    @Override
058    protected E computeNext() {
059        if (source.hasNext()) {
060            E next = source.next();
061            // we only want to return this element if it is not common
062            // to the two inputs
063            while (common.contains(next) || notCommon.contains(next)) {
064                // it was common, so shift it to common
065                if (notCommon.remove(next)) {
066                    common.add(next);
067                }
068                // move onto the next candidate
069                if (!source.hasNext()) {
070                    return endOfData();
071                }
072                next = source.next();
073            }
074            // it was not common so return it
075            return next;
076        }
077        return endOfData();
078    }
079
080    /**
081     * This method will return null until the source iterator is exhausted.
082     *
083     * @return The elements that turned out to be common to the two inputs.
084     */
085    public Set<E> common() {
086        return source.hasNext() ? null : common;
087    }
088
089    /**
090     * This method will return null until the source iterator is exhausted.
091     *
092     * @return The elements that turned out not to be common to the two inputs.
093     */
094    public Set<? extends E> notCommon() {
095        return source.hasNext() ? null : notCommon;
096    }
097
098}
