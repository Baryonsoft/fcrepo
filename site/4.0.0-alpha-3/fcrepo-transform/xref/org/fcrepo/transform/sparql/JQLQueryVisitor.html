175     /**
176      * Get the raw JCR query
177      * @return
178      * @throws RepositoryException
179      */
180     public QueryObjectModel getQuery() throws RepositoryException {
181         final org.modeshape.jcr.api.query.qom.QueryObjectModelFactory modeQueryFactory =
182             (org.modeshape.jcr.api.query.qom.QueryObjectModelFactory)queryFactory;
183         final int actualLimit;
184 
185         if (this.hasLimit) {
186             actualLimit = checkedCast(this.limit);
187         } else {
188             actualLimit = MAX_VALUE;
189         }
190 
191         final Limit selectLimit = modeQueryFactory.limit(actualLimit, checkedCast(this.offset));
192         final SelectQuery query = modeQueryFactory.select(getSource(),
193                                                              getConstraint(),
194                                                              getOrderings(),
195                                                              getColumns(),
196                                                              selectLimit,
197                                                              distinct);
198 
199 
200         return modeQueryFactory.createQuery(query);
201     }

203     /**
204      * Get the JCR query source information
205      * @return
206      */
207     private Source getSource() {
208         final Sets.SetView<String> difference = difference(joins.keySet(), joinConditions.keySet());
209 
210         final Source parentSource;
211 
212         final Iterator<String> unmatchedJoins = difference.iterator();
213 
214         if (unmatchedJoins.hasNext()) {
215             parentSource = joins.get(unmatchedJoins.next());
216             this.source = parentSource;
217         } else {
218             throw new JQLParsingException("No primary source column found in query");
219         }
220 
221         try {
222             for (final Map.Entry<String, Source> entry : joins.entrySet()) {
223 
224                 if (entry.getValue() != parentSource) {
225                     final String joinType;
226 
227                     if (joinTypes.containsKey(entry.getKey())) {
228                         joinType = JCR_JOIN_TYPE_INNER;
229                     } else {
230                         joinType = JCR_JOIN_TYPE_LEFT_OUTER;
231                     }
232 
233                     this.source =
234                         queryFactory.join(this.source, entry.getValue(),
235                                 joinType, joinConditions.get(entry.getKey()));
236                 }
237             }
238 
239         } catch (final RepositoryException e) {
240             LOGGER.info("Repository exception building query source", e);
241         }
242         return this.source;
243     }

341     @Override
342     public void visitOrderBy(final Query query) {
343         if (query.hasOrderBy()) {
344             LOGGER.trace("VISIT ORDER BY: {}", query.getOrderBy());
345             try {
346                 for (final SortCondition sortCondition : query.getOrderBy()) {
347 
348                     final PropertyValue property;
349                     final Expr expression = sortCondition.getExpression();
350 
351                     if (expression.isConstant()) {
352                         property = queryFactory.propertyValue(FEDORA_RESOURCE, expression.getConstant().asString());
353                     } else if (expression.isVariable()) {
354                         final Column c = variables.get(expression.getVarName());
355 
356                         property = queryFactory.propertyValue(c.getSelectorName(), c.getPropertyName());
357                     } else {
358                         property = null;
359                     }
360 
361                     if (property != null) {
362                         final Ordering ordering;
363 
364                         if (sortCondition.getDirection() == ORDER_DESCENDING) {
365                             ordering = queryFactory.descending(property);
366                         } else {
367                             ordering = queryFactory.ascending(property);
368 
369                         }
370 
371                         this.orderings.add(ordering);
372                     } else {
373                         LOGGER.debug("IGNORING UNKNOWN ORDER CONDITION {}", sortCondition);
374                     }
375                 }
376 
377             } catch (final RepositoryException e) {
378                 throw propagate(e);
379             }
380         }
381     }

413     @Override
414     public void visit(final ElementTriplesBlock el) {
415         LOGGER.trace("VISIT TRIPLES: {}", el);
416         final Iterator<Triple> tripleIterator = el.patternElts();
417 
418         while (tripleIterator.hasNext()) {
419             final Triple next = tripleIterator.next();
420             next.getObject();
421         }
422     }

424     @Override
425     public void visit(final ElementPathBlock el) {
426         LOGGER.trace("VISIT PATH BLOCK: {}", el);
427         Iterator<TriplePath> triplePathIterator = el.patternElts();
428 
429 
430         try {
431             while (triplePathIterator.hasNext()) {
432                 final TriplePath next = triplePathIterator.next();
433                 final Node subject = next.getSubject();
434 
435                 if (subject.isVariable()) {
436                     final String selectorName =
437                         "fedoraResource_" + subject.getName();
438 
439                     this.joins.put(subject.getName(), queryFactory.selector(
440                             FEDORA_RESOURCE, selectorName));
441 
442                     final Column c =
443                         queryFactory.column(selectorName, JCR_PATH, subject
444                                 .getName());
445                     variables.put(subject.getName(), c);
446                 }
447             }
448 
449             triplePathIterator = el.patternElts();
450 
451             while (triplePathIterator.hasNext()) {
452                 final TriplePath next = triplePathIterator.next();
453                 LOGGER.trace(" - TRIPLE PATH: {}", next);
454 
455                 final Node subject = next.getSubject();
456                 final Node predicate = next.getPredicate();
457                 final Node object = next.getObject();
458                 final Model defaultModel = createDefaultModel();
459 
460                 if (subject.isVariable()) {
461                     final Column c = variables.get(subject.getName());
462 
463                     if (resultsVars.contains(subject.getName())) {
464                         columns.add(c);
465                     }
466 
467                     if (predicate.isVariable()) {
468                         throw new NotImplementedException(
469                                 "Element path may not contain a variable predicate");
470                     }
471 
472                     final String propertyName =
473                         jcrTools.getPropertyNameFromPredicate(defaultModel
474                                 .createProperty(predicate.getURI()));
475 
476                     if (propertyName.equals("rdf:type") && object.isURI()) {
477                         final String mixinName =
478                             jcrTools.getPropertyNameFromPredicate(defaultModel
479                                     .createProperty(object.getURI()));
480 
481                         if (session.getWorkspace().getNodeTypeManager()
482                                 .hasNodeType(mixinName)) {
483                             final String selectorName =
484                                 "ref_type_" + mixinName.replace(":", "_");
485 
486                             this.joins.put(selectorName, queryFactory.selector(
487                                     mixinName, selectorName));
488 
489                             joinTypes.put(selectorName, JCR_JOIN_TYPE_INNER);
490                             joinConditions.put(selectorName, queryFactory
491                                     .sameNodeJoinCondition(c.getSelectorName(),
492                                             selectorName, "."));
493                             continue;
494                         }
495                     }
496 
497                     final int propertyType = jcrTools.getPropertyType(FEDORA_RESOURCE, propertyName);
498 
499                     if (object.isVariable()) {
500 
501                         final Column objectColumn;
502 
503                         if ((propertyType == REFERENCE || propertyType == WEAKREFERENCE || propertyType == URI)
504                                 && variables.containsKey(object.getName()))  {
505 
506                             objectColumn = variables.get(object.getName());
507 
508                             final String joinPropertyName;
509 
510                             if (propertyType == URI) {
511                                 joinPropertyName =
512                                     propertiesTools
513                                             .getReferencePropertyName(propertyName);
514                             } else {
515                                 joinPropertyName = propertyName;
516                             }
517 
518                             joinConditions.put(object.getName(),
519                                                   queryFactory.equiJoinCondition(
520                                                       c.getSelectorName(), joinPropertyName,
521                                                       objectColumn.getSelectorName(), "jcr:uuid"));
522                         } else {
523                             objectColumn = queryFactory.column(c.getSelectorName(),
524                                                                   propertyName,
525                                                                   object.getName());
526 
527                             variables.put(object.getName(), objectColumn);
528 
529                             if (resultsVars.contains(object.getName())) {
530                                 columns.add(objectColumn);
531                             }
532                         }
533 
534                         if (!inOptional) {
535                             appendConstraint(queryFactory.propertyExistence(c.getSelectorName(), propertyName));
536                         }
537                     } else {
538 
539                         if (!inOptional) {
540                             final PropertyValue field = queryFactory.propertyValue(c.getSelectorName(), propertyName);
541                             final Value jcrValue = jcrTools.createValue(defaultModel.asRDFNode(object), propertyType);
542                             final Literal literal = queryFactory.literal(jcrValue);
543                             appendConstraint(queryFactory.comparison(field, JCR_OPERATOR_EQUAL_TO, literal));
544                         }
545 
546                     }
547 
548                 } else if (predicate.isVariable()) {
549                     throw new NotImplementedException("Element path with constant subject and variable predicate");
550 
551                 } else if (object.isVariable()) {
552                     throw new NotImplementedException(
553                             "Element path with constant subject and predicate, and a variable object");
554 
555                 } else {
556                     throw new NotImplementedException("Element path with constant subject/predicate/object");
557                 }
558 
559             }
560         } catch (final RepositoryException e) {
561             throw propagate(e);
562         }
563     }

663     @Override
664     public void visit(final ExprFunction1 func) {
665         LOGGER.trace("VISIT EXPRFUNCTION1: {}", func);
666         final String funcName = func.getFunctionSymbol().getSymbol().toLowerCase();
667 
668         try {
669             switch (funcName) {
670                 case "not":
671                     final JQLQueryVisitor subVisitor1 = new JQLQueryVisitor(this);
672                     func.getArg().visit(subVisitor1);
673                     appendConstraint(queryFactory.not(subVisitor1.getConstraint()));
674                     break;
675                 case "bound":
676                     final Column column = variables.get(func.getArg().getVarName());
677                     appendConstraint(queryFactory.propertyExistence(column.getSelectorName(),
678                                                                     column.getPropertyName()));
