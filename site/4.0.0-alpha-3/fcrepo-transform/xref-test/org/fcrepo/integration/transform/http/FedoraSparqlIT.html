79      @Test
80      public void itShouldHaveAnHtmlView() throws IOException {
81          HttpGet request = new HttpGet(serverAddress + "/fcr:sparql");
82          request.addHeader("Accept", "text/html");
83          HttpResponse response = client.execute(request);

190     private String getResponseContent(final String sparql) throws IOException {
191         HttpPost sparqlRequest = new HttpPost(serverAddress + "/fcr:sparql");
192         BasicHttpEntity entity =  new BasicHttpEntity();
193         entity.setContent(IOUtils.toInputStream(sparql));
194         sparqlRequest.setEntity(entity);
195         HttpResponse response = client.execute(sparqlRequest);
196         assertEquals(200, response.getStatusLine().getStatusCode());
197 
198         String content = EntityUtils.toString(response.getEntity());
199         logger.trace("Retrieved sparql feed:\n" + content);
200         return content;
201     }
