104    /**
105     * Calculate the total size of all the binary properties in the repository
106     *
107     * @return size in bytes
108     */
109    public Long getRepositorySize() {
110        try {
111
112            final Timer.Context context = objectSizeCalculationTimer.time();
113            LOGGER.debug("Calculating repository size from index");
114
115            try {
116                return FedoraTypesUtils.getRepositorySize(repo);
117
118            } finally {
119                context.stop();
120            }
121        } catch (final RepositoryException e) {
122            throw propagate(e);
123        }
124    }

153    /**
154     * Get a map of JCR prefixes to their URI namespaces
155     *
156     * @param session
157     * @return
158     * @throws RepositoryException
159     */
160    public static Map<String, String> getRepositoryNamespaces(
161            final Session session) throws RepositoryException {
162
163        final NamespaceRegistry reg =
164            session.getWorkspace().getNamespaceRegistry();
165        return asMap(newHashSet(reg.getPrefixes()),
166                new Function<String, String>() {
167
168                    @Override
169                    public String apply(final String p) {
170                        try {
171                            return reg.getURI(p);
172                        } catch (final RepositoryException e) {
173                            throw propagate(e);
174                        }
175                    }
176                });
177    }

216    /**
217     * Perform a full-text search on the whole repository and return the
218     * information as an RDF Dataset
219     *
220     * @param subjectFactory
221     * @param searchSubject RDF resource to use as the subject of the search
222     * @param session
223     * @param terms
224     * @param limit
225     * @param offset
226     * @return
227     * @throws RepositoryException
228     */
229    public Dataset searchRepository(final GraphSubjects subjectFactory,
230            final Resource searchSubject, final Session session,
231            final String terms, final int limit, final long offset)
232        throws RepositoryException {
233
234        final Model model;
235
236        if (terms != null) {
237            final QueryManager queryManager =
238                    session.getWorkspace().getQueryManager();
239
240            final QueryObjectModelFactory factory =
241                    queryManager.getQOMFactory();
242
243            final Source selector =
244                    factory.selector(FEDORA_RESOURCE, "resourcesSelector");
245            final Constraint constraints =
246                    factory.fullTextSearch("resourcesSelector", null, factory
247                            .literal(session.getValueFactory().createValue(
248                                    terms)));
249
250            final Query query =
251                    factory.createQuery(selector, constraints, null, null);
252
253            // include an extra document to determine if additional pagination
254            // is
255            // necessary
256            query.setLimit(limit + 1);
257            query.setOffset(offset);
258
259            final QueryResult queryResult = query.execute();
260
261            final NodeIterator nodeIterator = queryResult.getNodes();
262            final long size = nodeIterator.getSize();
263
264            // remove that extra document from the nodes we'll iterate over
265            final Iterator<Node> limitedIterator =
266                    limit(new org.fcrepo.kernel.utils.iterators.NodeIterator(nodeIterator),
267                            limit);
268
269            model =
270                JcrRdfTools.withContext(subjectFactory, session)
271                        .getJcrPropertiesModel(limitedIterator, searchSubject)
272                        .asModel();
273
274            /* add the result description to the RDF model */
275
276            model.add(searchSubject, SEARCH_HAS_TOTAL_RESULTS, model
277                    .createTypedLiteral(size));
278            model.add(searchSubject, SEARCH_HAS_MORE, model
279                    .createTypedLiteral(nodeIterator.hasNext()));
280        } else {
281            model = createDefaultModel();
282        }
283
284        final Dataset dataset = DatasetFactory.create(model);
285
286        final String uri = searchSubject.getURI();
287        final Context context =
288                dataset.getContext() != null ? dataset.getContext()
289                        : new Context();
290        context.set(GraphProperties.URI_SYMBOL, uri);
291
292        return dataset;
293
294    }

296    /**
297     * This method backups up a running repository
298     *
299     * @param session
300     * @param backupDirectory
301     * @return
302     * @throws RepositoryException
303     */
304    public Problems backupRepository(final Session session,
305                                     final File backupDirectory) throws RepositoryException {
306        final RepositoryManager repoMgr = ((org.modeshape.jcr.api.Session) session)
307                .getWorkspace()
308                .getRepositoryManager();
309
310        final Problems problems = repoMgr.backupRepository(backupDirectory);
311
312        return problems;
313    }

320     * @return

322     */
323    public Problems restoreRepository(final Session session,
324                                      final File backupDirectory) throws RepositoryException {

326                .getWorkspace()

329        final Problems problems = repoMgr.restoreRepository(backupDirectory);

331        return problems;
