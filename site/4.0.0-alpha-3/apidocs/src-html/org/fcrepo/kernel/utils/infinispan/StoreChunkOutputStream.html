016package org.fcrepo.kernel.utils.infinispan;
017
018import java.io.ByteArrayOutputStream;
019import java.io.IOException;
020import java.io.OutputStream;
021
022import org.infinispan.container.InternalEntryFactory;
023import org.infinispan.container.InternalEntryFactoryImpl;
024import org.infinispan.container.entries.InternalCacheEntry;
025import org.infinispan.loaders.CacheLoaderException;
026import org.infinispan.loaders.CacheStore;
027import org.infinispan.metadata.EmbeddedMetadata;
028import org.slf4j.Logger;
029
030import static org.slf4j.LoggerFactory.getLogger;
031
032/**
033 * A near-copy of a Modeshape class (of the same name, but is
034 * unfortunately hidden from us) that takes a single OutputStream
035 * and chunks it into 1MB chunks for Infinispan
036 *
037 * @author Chris Beer
038 * @date Mar 14, 2013
039 */
040public class StoreChunkOutputStream extends OutputStream {
041
042    private static final Logger LOGGER = getLogger(StoreChunkOutputStream.class);
043
044    // 1 MB
045    public static final int CHUNKSIZE = 1024 * 1024 * 1;
046
047    private static final int CHUNK_BUFFER_SIZE = 1024;
048
049    protected final CacheStore blobCache;
050
051    protected final String keyPrefix;
052
053    private final ByteArrayOutputStream chunkBuffer;
054
055    private boolean closed;
056
057    protected int chunkIndex;
058
059    private final InternalEntryFactory entryFactory =
060        new InternalEntryFactoryImpl();
061
062    /**
063     * Prepare to store the OutputStream in the given CacheStore
064     * with the given prefix
065     * @param blobCache
066     * @param keyPrefix
067     */
068    public StoreChunkOutputStream(final CacheStore blobCache,
069                                  final String keyPrefix) {
070        super();
071        this.blobCache = blobCache;
072        this.keyPrefix = keyPrefix;
073        chunkBuffer = new ByteArrayOutputStream(CHUNK_BUFFER_SIZE);
074    }
075
076    /**
077     * @return Number of chunks stored.
078     */
079    public int getNumberChunks() {
080        return chunkIndex;
081    }
082
083    @Override
084    public void write(final int b) throws IOException {
085        if (chunkBuffer.size() == CHUNKSIZE) {
086            storeBufferInBLOBCache();
087        }
088        chunkBuffer.write(b);
089    }
090
091    @Override
092    public void write(final byte[] b, final int off, final int len)
093        throws IOException {
094        if (len + chunkBuffer.size() <= CHUNKSIZE) {
095            chunkBuffer.write(b, off, len);
096        } else {
097            final int storeLength = CHUNKSIZE - chunkBuffer.size();
098            write(b, off, storeLength);
099            storeBufferInBLOBCache();
100            write(b, off + storeLength, len - storeLength);
101        }
102    }
103
104    @Override
105    public void close() throws IOException {
106        LOGGER.debug("Close. Buffer size at close: {0}", chunkBuffer.size());
107        if (closed) {
108            LOGGER.debug("Stream already closed.");
109            return;
110        }
111        closed = true;
112        // store last chunk
113        if (chunkBuffer.size() > 0) {
114            storeBufferInBLOBCache();
115        }
116    }
117
118    private void storeBufferInBLOBCache() throws IOException {
119        final byte[] chunk = chunkBuffer.toByteArray();
120        try {
121            final String chunkKey = keyPrefix + "-" + chunkIndex;
122            final InternalCacheEntry c = blobCache.load(chunkKey);
123            final InternalCacheEntry cacheEntry;
124            if (c == null) {
125                cacheEntry =
126                    entryFactory.create(chunkKey,
127                                        chunk,
128                                        new EmbeddedMetadata
129                                        .Builder().build());
130            } else {
131                cacheEntry = entryFactory.create(chunkKey, chunk, c);
132            }
133
134            LOGGER.debug("Store chunk {0}", chunkKey);
135            blobCache.store(cacheEntry);
136            chunkIndex++;
137            chunkBuffer.reset();
138        } catch (final CacheLoaderException e) {
139            throw new IOException(e);
140        }
141    }
142
143}
