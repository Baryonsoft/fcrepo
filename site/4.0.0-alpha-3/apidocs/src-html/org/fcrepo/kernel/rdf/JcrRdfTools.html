177     * Convert a Fedora RDF Namespace into its JCR equivalent
178     *
179     * @param rdfNamespaceUri a namespace from an RDF document
180     * @return the JCR namespace, or the RDF namespace if no matching JCR
181     *         namespace is found
182     */
183    public static String getJcrNamespaceForRDFNamespace(
184            final String rdfNamespaceUri) {
185        if (rdfNamespacesToJcrNamespaces.containsKey(rdfNamespaceUri)) {
186            return rdfNamespacesToJcrNamespaces.get(rdfNamespaceUri);

228    /**
229     * Get an {@link RdfStream} for the given JCR NodeIterator
230     *
231     * @param nodeIterator
232     * @param iteratorSubject
233     * @return
234     * @throws RepositoryException
235     */
236    public RdfStream getJcrPropertiesModel(final Iterator<Node> nodeIterator,
237            final Resource iteratorSubject) throws RepositoryException {
238
239        final RdfStream results = new RdfStream();
240        while (nodeIterator.hasNext()) {
241            final Node node = nodeIterator.next();
242            results.concat(new PropertiesRdfContext(node, graphSubjects,
243                    llstore));
244            if (iteratorSubject != null) {
245                results.concat(singleton(create(iteratorSubject.asNode(),
246                        HAS_MEMBER_OF_RESULT.asNode(), graphSubjects
247                                .getGraphSubject(node).asNode())));
248            }
249        }
250        return results;
251    }

322    /**
323     * Decides whether the RDF representation of this {@link Node} will receive LDP Container status.
324     *
325     * @param node
326     * @return
327     * @throws RepositoryException
328     */
329    public static boolean isContainer(final Node node) throws RepositoryException {
330        return HAS_CHILD_NODE_DEFINITIONS.apply(node.getPrimaryNodeType())
331                || any(ImmutableList.copyOf(node.getMixinNodeTypes()),
332                        HAS_CHILD_NODE_DEFINITIONS);
333    }
334
335    static Predicate<NodeType> HAS_CHILD_NODE_DEFINITIONS =
336        new Predicate<NodeType>() {
337
338            @Override
339            public boolean apply(final NodeType input) {
340                return input.getChildNodeDefinitions().length > 0;
341            }
342        };

346     * not be modified from external sources)
347     *

352    public boolean isInternalProperty(final Node subjectNode,
353            final Resource predicate) {
354        switch (predicate.getNameSpace()) {
355            case REPOSITORY_NAMESPACE:
356            case JCR_NAMESPACE:
357            case LDP_NAMESPACE:

401    /**
402     * Create a JCR value from an RDF node with the given JCR type
403     * @param valueFactory
404     * @param data
405     * @param type
406     * @return
407     * @throws RepositoryException
408     */
409    public Value createValue(final ValueFactory valueFactory, final RDFNode data, final int type)
410        throws RepositoryException {
411        assert (valueFactory != null);
412
413        if (data.isURIResource()
414                && (type == REFERENCE || type == WEAKREFERENCE)) {
415            // reference to another node (by path)
416            final Node nodeFromGraphSubject =
417                graphSubjects.getNodeFromGraphSubject(data.asResource());
418            return valueFactory.createValue(nodeFromGraphSubject,
419                    type == WEAKREFERENCE);
420        } else if (data.isURIResource() || type == URI) {
421            // some random opaque URI
422            return valueFactory.createValue(data.toString(), PropertyType.URI);
423        } else if (data.isResource()) {
424            // a non-URI resource (e.g. a blank node)
425            return valueFactory.createValue(data.toString(), UNDEFINED);
426        } else if (data.isLiteral() && type == UNDEFINED) {
427            // the JCR schema doesn't know what this should be; so introspect
428            // the RDF and try to figure it out
429            final Literal literal = data.asLiteral();
430            final RDFDatatype dataType = literal.getDatatype();
431            final Object rdfValue = literal.getValue();
432
433            if (rdfValue instanceof Boolean) {
434                return valueFactory.createValue((Boolean) rdfValue);
435            } else if (rdfValue instanceof Byte
436                    || (dataType != null && dataType.getJavaClass() == Byte.class)) {
437                return valueFactory.createValue(literal.getByte());
438            } else if (rdfValue instanceof Double) {
439                return valueFactory.createValue((Double) rdfValue);
440            } else if (rdfValue instanceof Float) {
441                return valueFactory.createValue((Float) rdfValue);
442            } else if (rdfValue instanceof Long
443                    || (dataType != null && dataType.getJavaClass() == Long.class)) {
444                return valueFactory.createValue(literal.getLong());
445            } else if (rdfValue instanceof Short
446                    || (dataType != null && dataType.getJavaClass() == Short.class)) {
447                return valueFactory.createValue(literal.getShort());
448            } else if (rdfValue instanceof Integer) {
449                return valueFactory.createValue((Integer) rdfValue);
450            } else if (rdfValue instanceof XSDDateTime) {
451                return valueFactory.createValue(((XSDDateTime) rdfValue)
452                        .asCalendar());
453            } else {
454                return valueFactory.createValue(literal.getString(), STRING);
455            }
456
457        } else {
458            LOGGER.debug("Using default JCR value creation for RDF literal: {}",
459                    data);
460            return valueFactory.createValue(data.asLiteral().getString(), type);
461        }
462    }

532    /**
533     * Get the JCR property name for an RDF predicate
534     * @param namespaceRegistry
535     * @param predicate
536     * @param namespaceMapping
537     * @return
538     * @throws RepositoryException
539     */
540    public String getPropertyNameFromPredicate(final NamespaceRegistry namespaceRegistry,
541                                               final com.hp.hpl.jena.rdf.model.Property predicate,
542                                               final Map<String, String> namespaceMapping) throws RepositoryException {
543
544        final String prefix;
545
546        final String namespace =
547            getJcrNamespaceForRDFNamespace(predicate.getNameSpace());
548
549        assert (namespaceRegistry != null);
550
551        if (namespaceRegistry.isRegisteredUri(namespace)) {
552            LOGGER.debug("Discovered namespace: {} in namespace registry.",namespace);
553            prefix = namespaceRegistry.getPrefix(namespace);
554        } else {
555            LOGGER.debug("Didn't discover namespace: {} in namespace registry.",namespace);
556            final ImmutableBiMap<String, String> nsMap =
557                ImmutableBiMap.copyOf(namespaceMapping);
558            if (nsMap.containsValue(namespace)) {
559                LOGGER.debug("Discovered namespace: {} in namespace map: {}.", namespace,
560                        nsMap);
561                prefix = nsMap.inverse().get(namespace);
562                namespaceRegistry.registerNamespace(prefix, namespace);
563            } else {
564                prefix = namespaceRegistry.registerNamespace(namespace);
565            }
566        }
567
568        final String localName = predicate.getLocalName();
569
570        final String propertyName = prefix + ":" + localName;
571
572        LOGGER.debug("Took RDF predicate {} and translated it to JCR property {}", predicate, propertyName);
573
574        return propertyName;
575
576    }

597    /**
598     * Given a node type and a property name, figure out an appropraite jcr value type
599     * @param nodeType
600     * @param propertyName
601     * @return
602     * @throws RepositoryException
603     */
604    public int getPropertyType(final NodeType nodeType, final String propertyName) throws RepositoryException {
605        final PropertyDefinition[] propertyDefinitions = nodeType.getPropertyDefinitions();
606        int type = UNDEFINED;
607        for (final PropertyDefinition propertyDefinition : propertyDefinitions) {
608            if (propertyDefinition.getName().equals(propertyName)) {
609                if (type != UNDEFINED) {
610                    return UNDEFINED;
611                }
612
613                type = propertyDefinition.getRequiredType();
614            }
615        }
616
617        return type;
618    }
