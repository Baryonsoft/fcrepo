042/**
043 * {@link RdfStream} that holds contexts related to a specific {@link Node}.
044 *
045 * @author ajs6f
046 * @date Oct 10, 2013
047 */
048public class NodeRdfContext extends RdfStream {
049
050    private final Node node;
051
052    private final GraphSubjects graphSubjects;
053
054    private final com.hp.hpl.jena.graph.Node subject;
055
056    private final LowLevelStorageService lowLevelStorageService;
057
058    private static final Logger LOGGER = getLogger(NodeRdfContext.class);
059
060    /**
061     * Default constructor.
062     *
063     * @param node
064     * @param graphSubjects
065     * @throws RepositoryException
066     */
067    public NodeRdfContext(final Node node, final GraphSubjects graphSubjects,
068        final LowLevelStorageService lowLevelStorageService)
069        throws RepositoryException {
070        super();
071        this.node = node;
072        this.graphSubjects = graphSubjects;
073        this.subject = graphSubjects.getGraphSubject(node).asNode();
074
075        // TODO fix GraphProperties to allow for LowLevelStorageServices to pass
076        // through it
077        // this is horribly ugly. LowLevelStorageServices are supposed to be
078        // managed beans.
079        // but the contract of GraphProperties isn't wide enough to pass one in,
080        // so rather than
081        // alter GraphProperties right now, I'm just spinning one on the fly.
082        if (lowLevelStorageService == null) {
083            this.lowLevelStorageService = new LowLevelStorageService();
084            this.lowLevelStorageService.setRepository(node.getSession()
085                    .getRepository());
086        } else {
087            this.lowLevelStorageService = lowLevelStorageService;
088        }
089
090        // add JCR mixins as rdf:type triples
091        final Iterator<NodeType> nodeTypes = forArray(node.getMixinNodeTypes());
092        concat(Iterators.transform(nodeTypes, nodetype2triple()));
093    }
094
095    /**
096     * @return The {@link Node} in question
097     */
098    public Node node() {
099        return node;
100    }
101
102    /**
103     * @return local {@link GraphSubjects}
104     */
105    public GraphSubjects graphSubjects() {
106        return graphSubjects;
107    }
108
109    /**
110     * @return the RDF subject at the center of this context
111     */
112    public com.hp.hpl.jena.graph.Node subject() {
113        return subject;
114    }
115
116    /**
117     * @return the {@link LowLevelStorageService} in scope
118     */
119    public LowLevelStorageService lowLevelStorageService() {
120        return lowLevelStorageService;
121    }
122
123    private Function<NodeType, Triple> nodetype2triple() {
124        return new Function<NodeType, Triple>() {
125
126            @Override
127            public Triple apply(final NodeType nodeType) {
128                try {
129                    final String fullTypeName = nodeType.getName();
130                    LOGGER.trace("Translating JCR mixin name: {}", fullTypeName);
131                    final String prefix = fullTypeName.split(":")[0];
132                    final String typeName = fullTypeName.split(":")[1];
133                    final String namespace = getJcrUri(prefix);
134                    LOGGER.trace("with JCR namespace: {}", namespace);
135                    final com.hp.hpl.jena.graph.Node rdfType =
136                        createURI(getRDFNamespaceForJcrNamespace(namespace)
137                                + typeName);
138                    LOGGER.trace("into RDF resource: {}", rdfType);
139                    return create(subject(), type.asNode(), rdfType);
140                } catch (final RepositoryException e) {
141                    throw propagate(e);
142                }
143            }
144
145        };
146    }
147
148    private String getJcrUri(final String prefix) throws RepositoryException {
149        return node().getSession().getWorkspace().getNamespaceRegistry()
150                .getURI(prefix);
151    }
152
153}
