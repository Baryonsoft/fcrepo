050/**
051 * An {@link RdfStream} containing information about the fixity of a
052 * {@link Node}.
053 *
054 * @author ajs6f
055 * @date Oct 15, 2013
056 */
057public class FixityRdfContext extends NodeRdfContext {
058
059    /**
060     * Ordinary constructor.
061     *
062     * @param node
063     * @param graphSubjects
064     * @param lowLevelStorageService
065     * @param blobs
066     * @throws RepositoryException
067     */
068    public FixityRdfContext(final Node node, final GraphSubjects graphSubjects,
069            final LowLevelStorageService lowLevelStorageService,
070            final Iterable<FixityResult> blobs) throws RepositoryException {
071        super(node, graphSubjects, lowLevelStorageService);
072
073        concat(Iterators.concat(Iterators.transform(blobs.iterator(),
074                new Function<FixityResult, Iterator<Triple>>() {
075
076                    @Override
077                    public Iterator<Triple> apply(final FixityResult blob) {
078                        final com.hp.hpl.jena.graph.Node resultSubject = getTransientFixitySubject();
079                        final ImmutableSet.Builder<Triple> b = builder();
080                        try {
081                            b.add(create(resultSubject, RDF.type.asNode(), FIXITY_TYPE.asNode()));
082                            b.add(create(graphSubjects.getGraphSubject(node)
083                                    .asNode(), HAS_FIXITY_RESULT.asNode(),
084                                    resultSubject));
085                            final String storeIdentifier = blob.getStoreIdentifier();
086                            final com.hp.hpl.jena.graph.Node contentLocation = createResource(storeIdentifier)
087                                                                     .asNode();
088                            b.add(create(resultSubject, HAS_CONTENT_LOCATION.asNode(),
089                                            contentLocation));
090                            b.add(create(contentLocation,
091                                            RDF.type.asNode(),
092                                            CONTENT_LOCATION_TYPE.asNode()));
093                            b.add(create(contentLocation,
094                                            HAS_CONTENT_LOCATION_VALUE.asNode(),
095                                            createLiteral(storeIdentifier)));
096
097                            for (final FixityResult.FixityState state : blob.status) {
098                                b.add(create(resultSubject, HAS_FIXITY_STATE
099                                        .asNode(), createLiteral(state
100                                        .toString())));
101                            }
102                            final String checksum =
103                                blob.computedChecksum.toString();
104                            b.add(create(resultSubject, HAS_MESSAGE_DIGEST
105                                    .asNode(), createURI(checksum)));
106                            b.add(create(resultSubject, HAS_SIZE
107                                    .asNode(), createTypedLiteral(blob.computedSize)
108                                    .asNode()));
109                            return b.build().iterator();
110                        } catch (final RepositoryException e) {
111                            throw propagate(e);
112                        }
113                    }
114                })));
115    }
116
117    private com.hp.hpl.jena.graph.Node getTransientFixitySubject() {
118        return createURI(subject().getURI() + "#fixity/" + Calendar.getInstance().getTimeInMillis());
119    }
120}
