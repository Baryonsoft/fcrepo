017package org.fcrepo.http.api;
018
019import static java.util.Collections.singletonMap;
020import static javax.ws.rs.core.Response.created;
021import static javax.ws.rs.core.Response.noContent;
022import static javax.ws.rs.core.Response.status;
023import static javax.ws.rs.core.Response.Status.BAD_REQUEST;
024import static org.slf4j.LoggerFactory.getLogger;
025
026import java.util.List;
027
028import javax.jcr.RepositoryException;
029import javax.jcr.Session;
030import javax.servlet.http.HttpServletRequest;
031import javax.servlet.http.HttpSession;
032import javax.ws.rs.POST;
033import javax.ws.rs.Path;
034import javax.ws.rs.PathParam;
035import javax.ws.rs.core.Context;
036import javax.ws.rs.core.PathSegment;
037import javax.ws.rs.core.Response;
038
039import org.fcrepo.http.commons.AbstractResource;
040import org.fcrepo.http.commons.session.InjectedSession;
041import org.fcrepo.kernel.Transaction;
042import org.fcrepo.kernel.TxSession;
043import org.fcrepo.kernel.services.TransactionService;
044import org.slf4j.Logger;
045import org.springframework.beans.factory.annotation.Autowired;
046import org.springframework.context.annotation.Scope;
047import org.springframework.stereotype.Component;
048
049/**
050 * Transactions over REST
051 */
052@Component
053@Scope("prototype")
054@Path("/{path: .*}/fcr:tx")
055public class FedoraTransactions extends AbstractResource {
056
057    private static final Logger LOGGER = getLogger(FedoraTransactions.class);
058
059    @Autowired
060    private TransactionService txService;
061
062    @InjectedSession
063    protected Session session;
064
065    /**
066     * Create a new transaction resource and add it to the registry
067     *
068     * @param pathList
069     * @return
070     * @throws RepositoryException
071     */
072    @POST
073    public Response createTransaction(@PathParam("path")
074        final List<PathSegment> pathList, @Context
075        final HttpServletRequest req) throws RepositoryException {
076
077        LOGGER.debug("creating transaction at path {}", pathList);
078
079        if (!pathList.isEmpty()) {
080            return status(BAD_REQUEST).build();
081        }
082
083        if (session instanceof TxSession) {
084            final Transaction t =
085                    txService.getTransaction(((TxSession) session)
086                            .getTxId());
087            t.updateExpiryDate();
088            return noContent().expires(t.getExpires()).build();
089
090        } else {
091            final Transaction t = txService.beginTransaction(session);
092            final HttpSession httpSession = req.getSession(true);
093            if (httpSession != null) {
094                httpSession.setAttribute("currentTx", t.getId());
095            }
096            return created(
097                    uriInfo.getBaseUriBuilder().path(FedoraNodes.class)
098                            .buildFromMap(
099                                    singletonMap("path", "tx:" +
100                                            t.getId()))).expires(
101                    t.getExpires()).build();
102        }
103    }
104
105    /**
106     * Commit a transaction resource
107     *
108     * @param pathList
109     * @return
110     * @throws RepositoryException
111     */
112    @POST
113    @Path("fcr:commit")
114    public Response commit(@PathParam("path")
115        final List<PathSegment> pathList) throws RepositoryException {
116
117        return finalizeTransaction(pathList, true);
118
119    }
120
121    /**
122     * Rollback a transaction
123     */
124    @POST
125    @Path("fcr:rollback")
126    public Response rollback(@PathParam("path")
127        final List<PathSegment> pathList) throws RepositoryException {
128
129        return finalizeTransaction(pathList, false);
130    }
131
132    private Response finalizeTransaction(@PathParam("path")
133        final List<PathSegment> pathList, final boolean commit)
134        throws RepositoryException {
135
136        final String path = toPath(pathList);
137        if (!path.equals("/")) {
138            return status(BAD_REQUEST).build();
139        }
140
141        final String txId;
142        if (session instanceof TxSession) {
143            txId = ((TxSession) session).getTxId();
144        } else {
145            txId = "";
146        }
147
148        if (txId.isEmpty()) {
149            LOGGER.debug("cannot finalize an empty tx id {} at path {}",
150                    txId, path);
151            return status(BAD_REQUEST).build();
152        }
153
154        if (commit) {
155            LOGGER.debug("commiting transaction {} at path {}", txId, path);
156            txService.commit(txId);
157
158        } else {
159            LOGGER.debug("rolling back transaction {} at path {}", txId,
160                    path);
161            txService.rollback(txId);
162        }
163        return noContent().build();
164    }
165}
