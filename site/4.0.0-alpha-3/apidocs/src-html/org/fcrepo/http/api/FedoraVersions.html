022import static javax.ws.rs.core.Response.Status.NOT_FOUND;
023import static org.fcrepo.http.commons.domain.RDFMediaType.N3;
024import static org.fcrepo.http.commons.domain.RDFMediaType.N3_ALT1;
025import static org.fcrepo.http.commons.domain.RDFMediaType.N3_ALT2;
026import static org.fcrepo.http.commons.domain.RDFMediaType.NTRIPLES;
027import static org.fcrepo.http.commons.domain.RDFMediaType.RDF_JSON;
028import static org.fcrepo.http.commons.domain.RDFMediaType.RDF_XML;
029import static org.fcrepo.http.commons.domain.RDFMediaType.TURTLE;
030import static org.slf4j.LoggerFactory.getLogger;
031
032import java.io.IOException;
033import java.util.Collections;
034import java.util.List;
035
036import javax.jcr.RepositoryException;
037import javax.jcr.Session;
038import javax.ws.rs.GET;
039import javax.ws.rs.POST;
040import javax.ws.rs.Path;
041import javax.ws.rs.PathParam;

052import org.fcrepo.http.commons.session.InjectedSession;
053import org.fcrepo.kernel.FedoraResource;
054import org.fcrepo.kernel.rdf.GraphSubjects;

056import org.slf4j.Logger;
057import org.springframework.context.annotation.Scope;
058import org.springframework.stereotype.Component;
059
060/**
061 * Endpoint for managing versions of nodes
062 */
063@Component
064@Scope("prototype")
065@Path("/{path: .*}/fcr:versions")
066public class FedoraVersions extends AbstractResource {
067
068    @InjectedSession
069    protected Session session;
070
071    private static final Logger LOGGER = getLogger(FedoraNodes.class);
072
073    /**
074     * Get the list of versions for the object
075     *
076     * @param pathList
077     * @param request
078     * @param uriInfo
079     * @return
080     * @throws RepositoryException
081     */
082    @GET
083    @HtmlTemplate(value = "fcr:versions")
084    @Produces({TURTLE, N3, N3_ALT1, N3_ALT2, RDF_XML, RDF_JSON, NTRIPLES,
085            TEXT_HTML})
086    public RdfStream getVersionList(@PathParam("path")
087            final List<PathSegment> pathList,
088            @Context
089            final Request request,
090            @Context
091            final UriInfo uriInfo) throws RepositoryException {
092        final String path = toPath(pathList);
093
094        LOGGER.trace("Getting versions list for: {}", path);
095
096        final FedoraResource resource = nodeService.getObject(session, path);
097
098        return resource.getVersionTriples(nodeTranslator()).session(session).topic(
099                nodeTranslator().getGraphSubject(resource.getNode()).asNode());
100    }
101
102    /**
103     * Create a new version checkpoint and tag it with the given label.  If
104     * that label already describes another version it will silently be
105     * reassigned to describe this version.
106     *
107     * @param pathList
108     * @param versionLabel
109     * @return
110     * @throws RepositoryException
111     */
112    @POST
113    @Path("/{versionLabel}")
114    public Response addVersion(@PathParam("path")
115            final List<PathSegment> pathList,
116            @PathParam("versionLabel")
117            final String versionLabel) throws RepositoryException {
118        return addVersion(toPath(pathList), versionLabel);
119    }
120
121    /**
122     * Create a new version checkpoint with no label.
123     */
124    @POST
125    public Response addVersion(@PathParam("path")
126            final List<PathSegment> pathList) throws RepositoryException {
127        return addVersion(toPath(pathList), null);
128    }
129
130    private Response addVersion(String path, String label) throws RepositoryException {
131        try {
132            final FedoraResource resource =
133                    nodeService.getObject(session, path);
134            versionService.createVersion(session.getWorkspace(),
135                    Collections.singleton(path));
136            if (label != null) {
137                resource.addVersionLabel(label);
138            }
139            return noContent().build();
140        } finally {
141            session.logout();
142        }
143    }
144
145    /**
146     * Retrieve the tagged version of an object.
147     * @param pathList
148     * @param versionLabel
149     * @param uriInfo
150     * @return
151     * @throws RepositoryException
152     * @throws IOException
153     */
154    @Path("/{versionLabel}")
155    @GET
156    @Produces({TURTLE, N3, N3_ALT1, N3_ALT2, RDF_XML, RDF_JSON, NTRIPLES})
157    public RdfStream getVersion(@PathParam("path")
158            final List<PathSegment> pathList,
159            @PathParam("versionLabel")
160            final String versionLabel,
161            @Context
162            final Request request,
163            @Context
164            final UriInfo uriInfo) throws RepositoryException, IOException {
165        final String path = toPath(pathList);
166        LOGGER.trace("Getting version profile for: {} at version: {}", path,
167                versionLabel);
168
169        final FedoraResource resource =
170            nodeService.getObject(session, path, versionLabel);
171
172        if (resource == null) {
173            throw new WebApplicationException(status(NOT_FOUND).build());
174        } else {
175            return resource.getTriples(nodeTranslator()).session(session).topic(
176                    nodeTranslator().getGraphSubject(resource.getNode()).asNode());
177        }
178    }
179
180    /**
181     * A translator suitable for subjects that represent nodes.
182     */
183    protected GraphSubjects nodeTranslator() {
184        return new HttpGraphSubjects(session, FedoraNodes.class, uriInfo);
185    }
186
187}
