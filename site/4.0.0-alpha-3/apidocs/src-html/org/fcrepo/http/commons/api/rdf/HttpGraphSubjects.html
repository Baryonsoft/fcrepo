017package org.fcrepo.http.commons.api.rdf;
018
019import static com.hp.hpl.jena.rdf.model.ResourceFactory.createResource;
020import static java.util.Collections.singletonMap;
021import static javax.jcr.PropertyType.PATH;
022import static org.fcrepo.jcr.FedoraJcrTypes.FCR_CONTENT;
023import static org.fcrepo.kernel.services.TransactionService.getCurrentTransactionId;
024import static org.modeshape.jcr.api.JcrConstants.JCR_CONTENT;
025import static org.slf4j.LoggerFactory.getLogger;
026
027import java.net.URI;
028import java.util.Map;
029
030import javax.jcr.Node;
031import javax.jcr.Repository;
032import javax.jcr.RepositoryException;
033import javax.jcr.Session;
034import javax.jcr.ValueFormatException;
035import javax.jcr.Workspace;
036import javax.ws.rs.core.UriBuilder;
037import javax.ws.rs.core.UriInfo;
038
039import com.google.common.base.Function;
040import org.fcrepo.kernel.exception.RepositoryRuntimeException;
041import org.fcrepo.kernel.rdf.GraphSubjects;
042import org.fcrepo.kernel.services.functions.GetDefaultWorkspace;
043import org.slf4j.Logger;
044import com.hp.hpl.jena.rdf.model.Resource;
045
046/**
047 * Translate JCR paths to URLs to the given class
048 */
049public class HttpGraphSubjects implements GraphSubjects {
050
051    private static final Logger LOGGER = getLogger(HttpGraphSubjects.class);
052    public static final String WORKSPACE_PREFIX = "workspace:";
053    public static final String TX_PREFIX = "tx:";
054
055    private final UriBuilder nodesBuilder;
056
057    private final String basePath;
058
059    private final int pathIx;
060
061    private final URI context;
062
063    private final Session session;
064
065    private final String defaultWorkspace;
066
067    private Function<Repository, String> getDefaultWorkspace = new GetDefaultWorkspace();
068
069    /**
070     * Build HTTP graph subjects relative to the given JAX-RS resource, using the UriInfo provided.
071     *
072     * The session may provide additional information (e.g. workspaces) that need to be
073     * taken into account.
074     *
075     * @param relativeTo
076     * @param uris
077     */
078    public HttpGraphSubjects(final Session session, final Class<?> relativeTo, final UriInfo uris) {
079        this.context = uris.getRequestUri();
080        this.nodesBuilder = uris.getBaseUriBuilder().path(relativeTo);
081        String normalizedBasePath = nodesBuilder.build("").toString();
082        if (!normalizedBasePath.endsWith("/")) {
083            normalizedBasePath = normalizedBasePath + "/";
084        }
085        this.basePath = normalizedBasePath;
086        this.pathIx = normalizedBasePath.length() - 1;
087        this.session = session;
088        this.defaultWorkspace = getDefaultWorkspace.apply(session.getRepository());
089        LOGGER.debug("Resolving graph subjects to a base URI of \"{}\"",
090                normalizedBasePath);
091    }
092
093    @Override
094    public Resource getGraphSubject(final String absPath)
095        throws RepositoryException {
096        final URI result =
097                nodesBuilder.buildFromMap(getPathMap(absPath));
098        LOGGER.debug("Translated path {} into RDF subject {}", absPath, result);
099        return createResource(result.toString());
100    }
101
102    @Override
103    public Resource getContext() {
104        return createResource(context.toString());
105    }
106
107    @Override
108    public Resource getGraphSubject(final Node node) throws RepositoryException {
109        final URI result = nodesBuilder.buildFromMap(getPathMap(node));
110        LOGGER.debug("Translated node {} into RDF subject {}", node, result);
111        return createResource(result.toString());
112    }
113
114    @Override
115    public Node getNodeFromGraphSubject(final Resource subject) throws RepositoryException {
116
117        final String absPath = getPathFromGraphSubject(subject);
118
119        if (absPath == null) {
120            return null;
121        }
122
123        final Node node;
124
125
126        if (session.nodeExists(absPath)) {
127            node = session.getNode(absPath);
128            LOGGER.trace("RDF resource {} maps to JCR node {}", subject, node);
129        } else {
130            node = null;
131            LOGGER.debug("RDF resource {} looks like a Fedora node, but when we checked was not in the repository",
132                            subject);
133        }
134
135        return node;
136
137    }
138
139    @Override
140    public String getPathFromGraphSubject(final Resource subject) throws RepositoryException {
141        if (!isFedoraGraphSubject(subject)) {
142            LOGGER.debug(
143                    "RDF resource {} was not a URI resource with our expected basePath {}, aborting.",
144                    subject, basePath);
145            return null;
146        }
147
148        final StringBuilder pathBuilder = new StringBuilder();
149        final String absPath;
150        final String[] pathSegments =
151                subject.getURI().substring(pathIx).split("/");
152
153        for (final String segment : pathSegments) {
154            if (segment.startsWith(TX_PREFIX)) {
155                final String tx = segment.substring(TX_PREFIX.length());
156
157                final String currentTxId = getCurrentTransactionId(session);
158
159                if (currentTxId == null || !tx.equals(currentTxId)) {
160                    throw new RepositoryException(
161                            "Subject is not in this transaction");
162                }
163
164            } else if (segment.startsWith(WORKSPACE_PREFIX)) {
165                final String workspace = segment.substring(WORKSPACE_PREFIX.length());
166                if (!session.getWorkspace().getName().equals(workspace)) {
167                    throw new RepositoryException(
168                            "Subject is not in this workspace");
169                }
170            } else if (segment.equals(FCR_CONTENT)) {
171                pathBuilder.append("/");
172                pathBuilder.append(JCR_CONTENT);
173            } else {
174                if (!segment.isEmpty()) {
175                    pathBuilder.append("/");
176                    pathBuilder.append(segment);
177                }
178            }
179        }
180
181        if (pathBuilder.length() == 0) {
182            absPath = "/";
183        } else {
184            absPath = pathBuilder.toString();
185        }
186
187        if (isValidJcrPath(absPath)) {
188            return absPath;
189        } else {
190            return null;
191        }
192
193    }
194
195    private boolean isValidJcrPath(final String absPath) {
196        try {
197            session.getValueFactory().createValue(absPath, PATH);
198            return true;
199        } catch (final ValueFormatException e) {
200            LOGGER.trace("Unable to validate JCR path", e);
201            return false;
202        } catch (final RepositoryException e) {
203            throw new RepositoryRuntimeException("Unable to validate JCR path", e);
204        }
205    }
206
207    @Override
208    public boolean isFedoraGraphSubject(final Resource subject) {
209        return subject.isURIResource() && subject.getURI().startsWith(basePath)
210                && isValidJcrPath(subject.getURI().substring(pathIx));
211    }
212
213    private Map<String, String> getPathMap(final Node node)
214        throws RepositoryException {
215        return getPathMap(node.getPath());
216    }
217
218    private Map<String, String> getPathMap(final String absPath) throws RepositoryException {
219        // the path param value doesn't start with a slash
220        String path = absPath.substring(1);
221        if (path.endsWith(JCR_CONTENT)) {
222            path = path.replace(JCR_CONTENT, FCR_CONTENT);
223        }
224
225        if (session != null) {
226            final Workspace workspace = session.getWorkspace();
227
228            final String txId = getCurrentTransactionId(session);
229
230            if (txId != null) {
231                path = TX_PREFIX + txId + "/" + path;
232            } else if (workspace != null &&
233                    !workspace.getName().equals(defaultWorkspace)) {
234                path = WORKSPACE_PREFIX + workspace.getName() + "/" + path;
235            }
236        }
237
238        return singletonMap("path", path);
239    }
240}
