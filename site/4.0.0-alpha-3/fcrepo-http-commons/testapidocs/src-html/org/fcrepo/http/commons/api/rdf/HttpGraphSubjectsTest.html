074    @Before
075    public void setUp() throws RepositoryException {
076        initMocks(this);
077        uriInfo = getUriInfoImpl(testPath);
078        when(mockSession.getValueFactory()).thenReturn(mockValueFactory);
079        when(mockSession.getRepository()).thenReturn(mockRepository);
080        testObj =
081            new HttpGraphSubjects(mockSession, MockNodeController.class,
082                    uriInfo);
083    }
084
085    @Test
086    public void testGetGraphSubject() throws RepositoryException {
087        final String expected = "http://localhost:8080/fcrepo/rest" + testPath;
088        when(mockNode.getPath()).thenReturn(testPath);
089        when(mockWorkspace.getName()).thenReturn("default");
090        when(mockSession.getWorkspace()).thenReturn(mockWorkspace);
091        when(mockNode.getSession()).thenReturn(mockSession);
092        Resource actual = testObj.getGraphSubject(mockNode);
093        assertEquals(expected, actual.getURI());
094        when(mockNode.getPath()).thenReturn(testPath + "/jcr:content");
095        actual = testObj.getGraphSubject(mockNode);
096        assertEquals(expected + "/fcr:content", actual.getURI());
097    }
098
099    @Test
100    public void testGetNodeFromGraphSubject() throws PathNotFoundException,
101            RepositoryException {
102
103        when(mockSession.nodeExists(testPath)).thenReturn(true);
104        when(mockSession.getNode(testPath)).thenReturn(mockNode);
105        when(mockWorkspace.getName()).thenReturn("default");
106        when(mockSession.getWorkspace()).thenReturn(mockWorkspace);
107        // test a good subject
108        when(mockSubject.getURI()).thenReturn(
109                "http://localhost:8080/fcrepo/rest" + testPath);
110        when(mockSubject.isURIResource()).thenReturn(true);
111        Node actual = testObj.getNodeFromGraphSubject(mockSubject);
112        Mockito.verify(mockSession).getNode(testPath);
113        assertEquals(mockNode, actual);
114        // test a bad subject
115        when(mockSubject.getURI()).thenReturn(
116                "http://localhost:8080/fcrepo/rest2" + testPath + "/bad");
117        actual = testObj.getNodeFromGraphSubject(mockSubject);
118        assertEquals(null, actual);
119        // test a non-existent path
120        when(mockSubject.getURI()).thenReturn(
121                "http://localhost:8080/fcrepo/rest" + testPath + "/bad");
122        actual = testObj.getNodeFromGraphSubject(mockSubject);
123        assertEquals(null, actual);
124        // test a fcr:content path
125        when(mockSession.nodeExists(testPath + "/jcr:content")).thenReturn(true);
126        when(mockSubject.getURI())
127                .thenReturn(
128                        "http://localhost:8080/fcrepo/rest" + testPath +
129                                "/fcr:content");
130        actual = testObj.getNodeFromGraphSubject(mockSubject);
131        verify(mockSession).getNode(testPath + "/jcr:content");
132    }

143    @Test
144    public void testIsFedoraGraphSubject() {
145        when(mockSubject.getURI()).thenReturn(
146                "http://localhost:8080/fcrepo/rest/foo");
147        when(mockSubject.isURIResource()).thenReturn(true);
148        boolean actual = testObj.isFedoraGraphSubject(mockSubject);

170    private static UriInfo getUriInfoImpl(final String path) {
171        // UriInfo ui = mock(UriInfo.class,withSettings().verboseLogging());
172        final UriInfo ui = mock(UriInfo.class);
173        final UriBuilder ub = new UriBuilderImpl();
174        ub.scheme("http");
175        ub.host("localhost");
176        ub.port(8080);
177        ub.path("/fcrepo");
178
179        final UriBuilder rb = new UriBuilderImpl();
180        rb.scheme("http");
181        rb.host("localhost");
182        rb.port(8080);
183        rb.path("/fcrepo/rest" + path);
184
185        when(ui.getRequestUri()).thenReturn(
186                URI.create("http://localhost:8080/fcrepo/rest" + path));
187        when(ui.getBaseUri()).thenReturn(
188                URI.create("http://localhost:8080/fcrepo"));
189        when(ui.getBaseUriBuilder()).thenReturn(ub);
190        when(ui.getAbsolutePathBuilder()).thenReturn(rb);
191
192        return ui;
193    }
