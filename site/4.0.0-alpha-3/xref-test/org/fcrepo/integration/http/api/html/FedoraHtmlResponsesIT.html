185     @Ignore
186     @Test
187     public void testVersionCreationAndNavigation() throws IOException {
188         final String pid = randomUUID().toString();
189         createAndVerifyObjectWithIdFromRootPage(pid);
190 
191         final String updateSparql = "PREFIX dc: <http://purl.org/dc/elements/1.1/>\n" +
192                 "PREFIX fedora: <http://fedora.info/definitions/v4/rest-api#>\n" +
193                 "\n" +
194                 "INSERT DATA { <> fedoraconfig:versioningPolicy \"auto-version\" ; dc:title \"Object Title\". }";
195         postSparqlUpdateUsingHttpClient(updateSparql, pid);
196 
197         final HtmlPage objectPage = webClient.getPage(serverAddress + pid);
198         assertEquals("Auto versioning should be set.", "auto-version", objectPage.getFirstByXPath("//span[@property='http://fedora.info/definitions/v4/config#versioningPolicy']/text()").toString());
199         assertEquals("Title should be set.", "Object Title", objectPage.getFirstByXPath("//span[@property='http://purl.org/dc/elements/1.1/title']/text()").toString());
200 
201         final String updateSparql2 = "PREFIX dc: <http://purl.org/dc/elements/1.1/>\n" +
202                 "\n" +
203                 "DELETE { <> dc:title ?t }\n" +
204                 "INSERT { <> dc:title \"Updated Title\" }" +
205                 "WHERE { <> dc:title ?t }";
206         postSparqlUpdateUsingHttpClient(updateSparql2, pid);
207 
208         final HtmlPage versions =
209             webClient.getPage(serverAddress + pid + "/fcr:versions");
210         final List<DomAttr> versionLinks =
211             castList(versions.getByXPath("//a[@class='version_link']/@href"));
212         assertEquals("There should be two revisions.", 2, versionLinks.size());
213 
214         // get the labels
215         // will look like "Version from 2013-00-0T00:00:00.000Z"
216         // and will sort chronologically based on a String comparison
217         final List<DomText> labels =
218             castList(versions
219                     .getByXPath("//a[@class='version_link']/text()"));
220         final boolean chronological = labels.get(0).asText().compareTo(labels.get(1).toString()) < 0;
221         logger.debug("Versions {} in chronological order: {}, {}", chronological ? "are" : "are not", labels.get(0).asText(), labels.get(1).asText());
222 
223         final HtmlPage firstRevision =
224             webClient.getPage(versionLinks.get(chronological ? 0 : 1)
225                     .getNodeValue());
226         final List<DomText> v1Titles =
227             castList(firstRevision
228                     .getByXPath("//span[@property='http://purl.org/dc/elements/1.1/title']/text()"));
229         final HtmlPage secondRevision =
230             webClient.getPage(versionLinks.get(chronological ? 1 : 0)
231                     .getNodeValue());
232         final List<DomText> v2Titles =
233             castList(secondRevision
234                     .getByXPath("//span[@property='http://purl.org/dc/elements/1.1/title']/text()"));
235 
236         assertEquals("Version one should have one title.", 1, v1Titles.size());
237         assertEquals("Version two should have one title.", 1, v2Titles.size());
238         assertNotEquals("Each version should have a different title.", v1Titles.get(0), v2Titles.get(0));
239         assertEquals("First version should be preserved.", "Object Title", v1Titles.get(0).getWholeText());
240         assertEquals("Second version should be preserved.", "Updated Title", v2Titles.get(0).getWholeText());
241     }

243     private void postSparqlUpdateUsingHttpClient(final String sparql, final String pid) throws IOException {
244         final HttpPatch method = new HttpPatch(serverAddress + pid);
245         method.addHeader("Content-Type", "application/sparql-update");
246         final BasicHttpEntity entity = new BasicHttpEntity();
247         entity.setContent(new ByteArrayInputStream(sparql.getBytes()));
248         method.setEntity(entity);
249         final HttpResponse response = client.execute(method);
250         assertEquals("Expected successful response.", 204,
251                 response.getStatusLine().getStatusCode());
252     }
