27  import javax.jcr.RepositoryException;
28  import javax.jcr.Session;

30  import java.util.Collection;

32  import static com.google.common.base.Preconditions.checkNotNull;

36  /**
37   * This service exposes management of node versioning.  Instead of invoking
38   * the JCR VersionManager methods, this provides a level of indirection that
39   * allows for special handling of features built on top of JCR such as user
40   * transactions.
41   * @author Mike Durbin
42   */
43  
44  @Component
45  public class VersionService extends RepositoryService {
46  
47      private static final Logger LOGGER = getLogger(VersionService.class);
48  
49      protected static final String VERSIONABLE = "mix:versionable";
50  
51      protected static final String VERSION_POLICY = "fedoraconfig:versioningPolicy";
52  
53      protected static final String AUTO_VERSION = "auto-version";
54  
55      @Autowired
56      TransactionService txService;
57  
58      /**
59       * Notifies the version manager that the node at a given path was updated
60       * so that if automatic versioning is set for that node, a version
61       * checkpoint will be made.  When a node object is available, use of
62       * {@link #nodeUpdated(javax.jcr.Node)} will save the overhead of a
63       * redundant node lookup.
64       * @param session
65       * @param absPath the absolute path to the node that was created or
66       *                modified
67       * @throws RepositoryException
68       */
69      public void nodeUpdated(final Session session, String absPath) throws RepositoryException {
70          nodeUpdated(session.getNode(absPath));
71      }
72  
73      /**
74       * Notifies the version manager that the given node was updated
75       * so that if automatic versioning is set for that node, a version
76       * checkpoint will be made.
77       * @param n the node that was updated
78       * @throws RepositoryException
79       */
80      public void nodeUpdated(Node n) throws RepositoryException {
81          if (isVersioningEnabled(n)
82                  && isImplicitVersioningEnabled(n)) {
83              queueOrCommitCheckpoint(n.getSession(), n.getPath());
84          } else {
85              LOGGER.trace("No implicit version checkpoint set for {}", n.getPath());
86          }
87      }
88  
89      /**
90       * Explicitly creates a version for the nodes at each path provided.
91       * @param workspace the workspace in which the node resides
92       * @param paths absolute paths to the nodes within the workspace
93       * @throws RepositoryException
94       */
95      public void createVersion(final Workspace workspace,
96              final Collection<String> paths) throws RepositoryException {
97          for (final String absPath : paths) {
98              checkpoint(workspace, absPath);
99          }
100     }
101 
102     private boolean isVersioningEnabled(final Node n) throws RepositoryException {
103         return n.isNodeType(VERSIONABLE);
104     }
105 
106     private boolean isImplicitVersioningEnabled(final Node n) throws RepositoryException {
107         if (!n.hasProperty(VERSION_POLICY)) {
108             return false;
109         } else {
110             final Property p = n.getProperty(VERSION_POLICY);
111 
112             return propertyContains(p, AUTO_VERSION);
113         }
114     }
115 
116     private void queueOrCommitCheckpoint(Session session, String absPath) throws RepositoryException {
117         final String txId = TransactionService.getCurrentTransactionId(session);
118 
119         if (txId == null) {
120             checkpoint(session.getWorkspace(), absPath);
121         } else {
122             queueCheckpoint(txId, absPath);
123         }
124     }
125 
126     private void checkpoint(Workspace workspace, String absPath) throws RepositoryException {
127         LOGGER.trace("Setting implicit version checkpoint set for {}", absPath);
128         workspace.getVersionManager().checkpoint(absPath);
129     }
130 
131     private void queueCheckpoint(String txId, String absPath) throws TransactionMissingException {
132         final Transaction tx = txService.getTransaction(txId);
133         LOGGER.trace("Queuing implicit version checkpoint set for {}", absPath);
134         tx.addPathToVersion(absPath);
135     }
136     /**
137      * Creates a version checkpoint for the given node if versioning is enabled
138      * for that node type.  When versioning is enabled this is the equivalent of
139      * VersionManager#checkpoint(node.getPath()), except that it is aware of
140      * TxSessions and queues these operations accordingly.
141      *
142      * @param node the node for whom a new version is
143      *                to be minted
144      * @throws RepositoryException
145      */
146     public void checkpoint(Node node) throws RepositoryException {
147 
148         checkNotNull(node, "Cannot checkpoint null nodes");
149 
150         final Session session = node.getSession();
151         final String absPath = node.getPath();
152         if (node.isNodeType(VERSIONABLE)) {
153             LOGGER.trace("Setting checkpoint for {}", absPath);
154 
155             final String txId = TransactionService.getCurrentTransactionId(session);
156             if (txId != null) {
157                 final Transaction tx = txService.getTransaction(txId);
158                 tx.addPathToVersion(absPath);
159             } else {
160                 session.getWorkspace().getVersionManager().checkpoint(absPath);
161             }
162         } else {
163             LOGGER.trace("No checkpoint set for unversionable {}", absPath);
164         }
165     }
