017package org.fcrepo.kernel.services;
018
019import org.fcrepo.kernel.Transaction;
020import org.fcrepo.kernel.exception.TransactionMissingException;
021import org.slf4j.Logger;
022import org.springframework.beans.factory.annotation.Autowired;
023import org.springframework.stereotype.Component;
024
025import javax.jcr.Node;
026import javax.jcr.Property;
027import javax.jcr.RepositoryException;
028import javax.jcr.Session;
029import javax.jcr.Workspace;
030import java.util.Collection;
031
032import static com.google.common.base.Preconditions.checkNotNull;
033import static org.fcrepo.kernel.utils.FedoraTypesUtils.propertyContains;
034import static org.slf4j.LoggerFactory.getLogger;
035
036/**
037 * This service exposes management of node versioning.  Instead of invoking
038 * the JCR VersionManager methods, this provides a level of indirection that
039 * allows for special handling of features built on top of JCR such as user
040 * transactions.
041 * @author Mike Durbin
042 */
043
044@Component
045public class VersionService extends RepositoryService {
046
047    private static final Logger LOGGER = getLogger(VersionService.class);
048
049    protected static final String VERSIONABLE = "mix:versionable";
050
051    protected static final String VERSION_POLICY = "fedoraconfig:versioningPolicy";
052
053    protected static final String AUTO_VERSION = "auto-version";
054
055    @Autowired
056    TransactionService txService;
057
058    /**
059     * Notifies the version manager that the node at a given path was updated
060     * so that if automatic versioning is set for that node, a version
061     * checkpoint will be made.  When a node object is available, use of
062     * {@link #nodeUpdated(javax.jcr.Node)} will save the overhead of a
063     * redundant node lookup.
064     * @param session
065     * @param absPath the absolute path to the node that was created or
066     *                modified
067     * @throws RepositoryException
068     */
069    public void nodeUpdated(final Session session, String absPath) throws RepositoryException {
070        nodeUpdated(session.getNode(absPath));
071    }
072
073    /**
074     * Notifies the version manager that the given node was updated
075     * so that if automatic versioning is set for that node, a version
076     * checkpoint will be made.
077     * @param n the node that was updated
078     * @throws RepositoryException
079     */
080    public void nodeUpdated(Node n) throws RepositoryException {
081        if (isVersioningEnabled(n)
082                && isImplicitVersioningEnabled(n)) {
083            queueOrCommitCheckpoint(n.getSession(), n.getPath());
084        } else {
085            LOGGER.trace("No implicit version checkpoint set for {}", n.getPath());
086        }
087    }
088
089    /**
090     * Explicitly creates a version for the nodes at each path provided.
091     * @param workspace the workspace in which the node resides
092     * @param paths absolute paths to the nodes within the workspace
093     * @throws RepositoryException
094     */
095    public void createVersion(final Workspace workspace,
096            final Collection<String> paths) throws RepositoryException {
097        for (final String absPath : paths) {
098            checkpoint(workspace, absPath);
099        }
100    }
101
102    private boolean isVersioningEnabled(final Node n) throws RepositoryException {
103        return n.isNodeType(VERSIONABLE);
104    }
105
106    private boolean isImplicitVersioningEnabled(final Node n) throws RepositoryException {
107        if (!n.hasProperty(VERSION_POLICY)) {
108            return false;
109        } else {
110            final Property p = n.getProperty(VERSION_POLICY);
111
112            return propertyContains(p, AUTO_VERSION);
113        }
114    }
115
116    private void queueOrCommitCheckpoint(Session session, String absPath) throws RepositoryException {
117        final String txId = TransactionService.getCurrentTransactionId(session);
118
119        if (txId == null) {
120            checkpoint(session.getWorkspace(), absPath);
121        } else {
122            queueCheckpoint(txId, absPath);
123        }
124    }
125
126    private void checkpoint(Workspace workspace, String absPath) throws RepositoryException {
127        LOGGER.trace("Setting implicit version checkpoint set for {}", absPath);
128        workspace.getVersionManager().checkpoint(absPath);
129    }
130
131    private void queueCheckpoint(String txId, String absPath) throws TransactionMissingException {
132        final Transaction tx = txService.getTransaction(txId);
133        LOGGER.trace("Queuing implicit version checkpoint set for {}", absPath);
134        tx.addPathToVersion(absPath);
135    }
136    /**
137     * Creates a version checkpoint for the given node if versioning is enabled
138     * for that node type.  When versioning is enabled this is the equivalent of
139     * VersionManager#checkpoint(node.getPath()), except that it is aware of
140     * TxSessions and queues these operations accordingly.
141     *
142     * @param node the node for whom a new version is
143     *                to be minted
144     * @throws RepositoryException
145     */
146    public void checkpoint(Node node) throws RepositoryException {
147
148        checkNotNull(node, "Cannot checkpoint null nodes");
149
150        final Session session = node.getSession();
151        final String absPath = node.getPath();
152        if (node.isNodeType(VERSIONABLE)) {
153            LOGGER.trace("Setting checkpoint for {}", absPath);
154
155            final String txId = TransactionService.getCurrentTransactionId(session);
156            if (txId != null) {
157                final Transaction tx = txService.getTransaction(txId);
158                tx.addPathToVersion(absPath);
159            } else {
160                session.getWorkspace().getVersionManager().checkpoint(absPath);
161            }
162        } else {
163            LOGGER.trace("No checkpoint set for unversionable {}", absPath);
164        }
165    }
166}
