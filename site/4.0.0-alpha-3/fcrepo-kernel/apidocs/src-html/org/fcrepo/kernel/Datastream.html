048/**
049 * Abstraction for a Fedora datastream backed by a JCR node.
050 *
051 * @author ajs6f
052 * @date Feb 21, 2013
053 */
054public class Datastream extends FedoraResource implements FedoraJcrTypes {
055
056    private static final Logger LOGGER = getLogger(Datastream.class);
057
058    static final Histogram contentSizeHistogram =
059            getMetrics().histogram(name(Datastream.class, "content-size"));
060
061    /**
062     * The JCR node for this datastream
063     * @param n an existing {@link Node}
064     */
065    public Datastream(final Node n) {
066        super(n);
067
068        if (node.isNew()) {
069            initializeNewDatastreamProperties();
070        }
071    }
072
073    /**
074     * Create or find a FedoraDatastream at the given path
075     * @param session the JCR session to use to retrieve the object
076     * @param path the absolute path to the object
077     * @param nodeType primary type to assign to node
078     * @throws RepositoryException
079     */
080    public Datastream(final Session session, final String path,
081                      final String nodeType) throws RepositoryException {
082        super(session, path, nodeType);
083
084        if (node.isNew()) {
085            initializeNewDatastreamProperties();
086        }
087    }

089    /**
090     * Create or find a FedoraDatastream at the given path
091     * @param session the JCR session to use to retrieve the object
092     * @param path the absolute path to the object
093     * @throws RepositoryException
094     */
095    public Datastream(final Session session, final String path)
096        throws RepositoryException {
097        this(session, path, JcrConstants.NT_FILE);
098    }
099
100    private void initializeNewDatastreamProperties() {
101        try {
102            if (node.isNew() || !hasMixin(node)) {
103                LOGGER.debug("Setting {} properties on a {} node...",
104                             FEDORA_DATASTREAM, JcrConstants.NT_FILE);
105                node.addMixin(FEDORA_DATASTREAM);
106
107                if (node.hasNode(JCR_CONTENT)) {
108                    final Node contentNode = node.getNode(JCR_CONTENT);
109                    decorateContentNode(contentNode);
110                }
111            }
112        } catch (final RepositoryException ex) {
113            LOGGER.warn("Could not decorate {} with {} properties: {}" ,
114                        JCR_CONTENT, FEDORA_DATASTREAM, ex);
115        }
116    }

118    /**
119     * @return The InputStream of content associated with this datastream.
120     * @throws RepositoryException
121     */
122    public InputStream getContent() throws RepositoryException {
123        final Node contentNode = node.getNode(JCR_CONTENT);
124        LOGGER.trace("Retrieved datastream content node.");
125        return contentNode.getProperty(JCR_DATA).getBinary().getStream();
126    }
127
128    /**
129     * Sets the content of this Datastream.
130     *
131     * @param content  InputStream of binary content to be stored
132     * @param contentType MIME type of content (optional)
133     * @param checksum Checksum URI of the content (optional)
134     * @param originalFileName Original file name of the content (optional)
135     * @param storagePolicyDecisionPoint Policy decision point for storing the content (optional)
136     * @throws RepositoryException
137     * @throws InvalidChecksumException
138     */
139    public void setContent(final InputStream content, final String contentType,
140                           final URI checksum, final String originalFileName,
141                           final StoragePolicyDecisionPoint storagePolicyDecisionPoint)
142        throws RepositoryException, InvalidChecksumException {
143
144        final Node contentNode =
145            findOrCreateChild(node, JCR_CONTENT, NT_RESOURCE);
146
147        if (contentNode.canAddMixin(FEDORA_BINARY)) {
148            contentNode.addMixin(FEDORA_BINARY);
149        }
150
151        if (contentType != null) {
152            contentNode.setProperty(JCR_MIME_TYPE, contentType);
153        }
154
155        if (originalFileName != null) {
156            contentNode.setProperty(PREMIS_FILE_NAME, originalFileName);
157        }
158
159        LOGGER.debug("Created content node at path: {}", contentNode.getPath());
160
161        String hint = null;
162
163
164        if (storagePolicyDecisionPoint != null) {
165            hint = storagePolicyDecisionPoint.evaluatePolicies(node);
166        }
167        final ValueFactory modevf =
168            (ValueFactory) node.getSession().getValueFactory();
169        final Binary binary = modevf.createBinary(content, hint);
170
171        /*
172         * This next line of code deserves explanation. If we chose for the
173         * simpler line:
174         * Property dataProperty = contentNode.setProperty(JCR_DATA,
175         * requestBodyStream);
176         * then the JCR would not block on the stream's completion, and we would
177         * return to the requester before the mutation to the repo had actually
178         * completed. So instead we use createBinary(requestBodyStream), because
179         * its contract specifies:
180         * "The passed InputStream is closed before this method returns either
181         * normally or because of an exception."
182         * which lets us block and not return until the job is done! The simpler
183         * code may still be useful to us for an asynchronous method that we
184         * develop later.
185         */
186        final Property dataProperty = contentNode.setProperty(JCR_DATA, binary);
187
188        final String dsChecksum = binary.getHexHash();
189        final URI uriChecksumString = ContentDigest.asURI("SHA-1", dsChecksum);
190        if (checksum != null &&
191                !checksum.equals(uriChecksumString)) {
192            LOGGER.debug("Failed checksum test");
193            throw new InvalidChecksumException("Checksum Mismatch of " +
194                 uriChecksumString + " and " + checksum);
195        }
196
197        decorateContentNode(contentNode);
198
199        LOGGER.debug("Created data property at path: {}",
200                     dataProperty.getPath());
201
202    }
203
204    /**
205     * Set the content of this datastream
206     * @param content Binary content to be stored
207     * @throws InvalidChecksumException
208     * @throws RepositoryException
209     */
210    public void setContent(final InputStream content) throws InvalidChecksumException,
211                                                       RepositoryException {
212        setContent(content, null, null, null, null);
213    }
214
215    /**
216     * @return The size in bytes of content associated with this datastream.
217     */
218    public long getContentSize() {
219        try {
220            return node.getNode(JCR_CONTENT).getProperty(CONTENT_SIZE)
221                .getLong();
222        } catch (final RepositoryException e) {
223            LOGGER.info("Could not get contentSize()", e);
224        }
225
226        return -1L;
227    }
228
229    /**
230     * Get the pre-calculated content digest for the binary payload
231     * @return a URI with the format algorithm:value
232     * @throws RepositoryException
233     */
234    public URI getContentDigest() throws RepositoryException {
235        final Node contentNode = node.getNode(JCR_CONTENT);
236        try {
237            return new URI(contentNode.getProperty(CONTENT_DIGEST).getString());
238        } catch (final RepositoryException | URISyntaxException e) {
239            LOGGER.info("Could not get content digest: ", e);
240        }
241
242        return ContentDigest.missingChecksum();
243    }

245    /**
246     * @return The ID of this datastream, unique within an object. Normally just
247     *         the name of the backing JCR node.
248     * @throws RepositoryException
249     */
250    public String getDsId() throws RepositoryException {
251        return node.getName();
252    }

258    public FedoraObject getObject() throws RepositoryException {
259        return new FedoraObject(node.getParent());

262    /**
263     * @return The MimeType of content associated with this datastream.
264     * @throws RepositoryException
265     */
266    public String getMimeType() throws RepositoryException {
267        return node.hasNode(JCR_CONTENT) &&
268            node.getNode(JCR_CONTENT).hasProperty(JCR_MIME_TYPE) ?
269            node.getNode(JCR_CONTENT).getProperty(JCR_MIME_TYPE).getString() :
270            "application/octet-stream";
271    }
272
273    /**
274     * Return the calculated size of the DS node
275     * @return combined size of the properties and binary content
276     * @throws RepositoryException
277     */
278    @Override
279    public long getSize() throws RepositoryException {
280        return getNodePropertySize(node) + getContentSize();
281
282    }
283
284    /**
285     * Return the file name for the binary content
286     * @return original file name for the binary content, or the object's id.
287     * @throws RepositoryException
288     */
289    public String getFilename() throws RepositoryException {
290        if (node.hasNode(JCR_CONTENT) && node.getNode(JCR_CONTENT).hasProperty(PREMIS_FILE_NAME)) {
291            return node.getNode(JCR_CONTENT).getProperty(PREMIS_FILE_NAME).getString();
292        } else {
293            return getDsId();
294        }
295    }
296
297    private void decorateContentNode(final Node contentNode)
298        throws RepositoryException {
299        if (contentNode == null) {
300            LOGGER.warn("{} node appears to be null!", JCR_CONTENT);
301            return;
302        }
303        if (contentNode.canAddMixin(FEDORA_BINARY)) {
304            contentNode.addMixin(FEDORA_BINARY);
305        }
306
307        final Property dataProperty = contentNode.getProperty(JCR_DATA);
308        final Binary binary = (Binary) dataProperty.getBinary();
309        final String dsChecksum = binary.getHexHash();
310
311        contentSizeHistogram.update(dataProperty.getLength());
312
313        contentNode.setProperty(CONTENT_SIZE, dataProperty.getLength());
314        contentNode.setProperty(CONTENT_DIGEST, ContentDigest.
315                                asURI("SHA-1", dsChecksum).toString());
316
317        LOGGER.debug("Decorated data property at path: {}", dataProperty.getPath());
318    }
319
320    /**
321     * Check if the node has a fedora:datastream mixin
322     * @param node node to check
323     * @throws RepositoryException
324     */
325    public static boolean hasMixin(final Node node) throws RepositoryException {
326        return isFedoraDatastream.apply(node);
327    }
