017package org.fcrepo.integration.kernel;
018
019import static org.junit.Assert.assertEquals;
020import static org.junit.Assert.assertNotNull;
021
022import java.io.ByteArrayInputStream;
023import java.io.IOException;
024
025import javax.inject.Inject;
026import javax.jcr.Repository;
027import javax.jcr.RepositoryException;
028import javax.jcr.Session;
029
030import org.apache.commons.io.IOUtils;
031import org.fcrepo.kernel.Datastream;
032import org.fcrepo.kernel.exception.InvalidChecksumException;
033import org.fcrepo.kernel.services.DatastreamService;
034import org.fcrepo.kernel.services.ObjectService;
035import org.fcrepo.kernel.utils.ContentDigest;
036import org.junit.Test;
037import org.springframework.test.context.ContextConfiguration;
038
039@ContextConfiguration({"/spring-test/repo.xml"})
040public class DatastreamIT extends AbstractIT {
041
042    @Inject
043    Repository repo;
044
045    @Inject
046    DatastreamService datastreamService;
047
048    @Inject
049    ObjectService objectService;
050
051    @Test
052    public void testCreatedDate() throws RepositoryException, IOException,
053                                 InvalidChecksumException {
054        Session session = repo.login();
055        objectService.createObject(session, "/testDatastreamObject");
056        datastreamService.createDatastreamNode(session,
057                "/testDatastreamObject/testDatastreamNode1",
058                "application/octet-stream", null, new ByteArrayInputStream("asdf"
059                        .getBytes()));
060        session.save();
061        session.logout();
062        session = repo.login();
063        final Datastream ds =
064            datastreamService.getDatastream(session,
065                    "/testDatastreamObject/testDatastreamNode1");
066        assertNotNull("Couldn't find created date on datastream!", ds
067                .getCreatedDate());
068    }
069
070    @Test
071    public void testDatastreamContent() throws IOException,
072                                       RepositoryException,
073                                       InvalidChecksumException {
074        final Session session = repo.login();
075        objectService.createObject(session, "/testDatastreamObject");
076        datastreamService.createDatastreamNode(session,
077                "/testDatastreamObject/testDatastreamNode1",
078                "application/octet-stream", null, new ByteArrayInputStream("asdf"
079                        .getBytes()));
080
081        session.save();
082
083        final Datastream ds =
084            datastreamService.getDatastream(session,
085                    "/testDatastreamObject/testDatastreamNode1");
086        final String contentString = IOUtils.toString(ds.getContent(), "ASCII");
087
088        assertEquals("asdf", contentString);
089
090    }
091
092    @Test
093    public void testDatastreamContentDigestAndLength() throws IOException,
094                                                      RepositoryException,
095                                                      InvalidChecksumException {
096        final Session session = repo.login();
097        objectService.createObject(session, "/testDatastreamObject");
098        datastreamService.createDatastreamNode(session,
099                "/testDatastreamObject/testDatastreamNode2",
100                "application/octet-stream", null, new ByteArrayInputStream("asdf"
101                        .getBytes()));
102
103        session.save();
104
105        final Datastream ds =
106            datastreamService.getDatastream(session,
107                    "/testDatastreamObject/testDatastreamNode2");
108        assertEquals("urn:sha1:3da541559918a808c2402bba5012f6c60b27661c", ds
109                .getContentDigest().toString());
110        assertEquals(4L, ds.getContentSize());
111
112        final String contentString = IOUtils.toString(ds.getContent(), "ASCII");
113
114        assertEquals("asdf", contentString);
115    }
116
117    @Test
118    public void
119            testModifyDatastreamContentDigestAndLength() throws IOException,
120                                                        RepositoryException,
121                                                        InvalidChecksumException {
122        final Session session = repo.login();
123        objectService.createObject(session, "/testDatastreamObject");
124        datastreamService.createDatastreamNode(session,
125                "/testDatastreamObject/testDatastreamNode3",
126                "application/octet-stream", null, new ByteArrayInputStream("asdf"
127                        .getBytes()));
128
129        session.save();
130
131        final Datastream ds =
132            datastreamService.getDatastream(session,
133                    "/testDatastreamObject/testDatastreamNode3");
134
135        ds.setContent(new ByteArrayInputStream("0123456789".getBytes()));
136
137        assertEquals("urn:sha1:87acec17cd9dcd20a716cc2cf67417b71c8a7016", ds
138                .getContentDigest().toString());
139        assertEquals(10L, ds.getContentSize());
140
141        final String contentString = IOUtils.toString(ds.getContent(), "ASCII");
142
143        assertEquals("0123456789", contentString);
144    }
145
146    @Test
147    public void testDatastreamContentWithChecksum() throws IOException,
148                                                   RepositoryException,
149                                                   InvalidChecksumException {
150        final Session session = repo.login();
151        objectService.createObject(session, "/testDatastreamObject");
152        datastreamService.createDatastreamNode(session,
153                "/testDatastreamObject/testDatastreamNode4",
154                "application/octet-stream", null, new ByteArrayInputStream("asdf"
155                        .getBytes()), ContentDigest.asURI("SHA-1",
156                        "3da541559918a808c2402bba5012f6c60b27661c"));
157
158        session.save();
159
160        final Datastream ds =
161            datastreamService.getDatastream(session,
162                    "/testDatastreamObject/testDatastreamNode4");
163        assertEquals("urn:sha1:3da541559918a808c2402bba5012f6c60b27661c", ds
164                .getContentDigest().toString());
165
166        final String contentString = IOUtils.toString(ds.getContent(), "ASCII");
167
168        assertEquals("asdf", contentString);
169    }

171    @Test
172    public void testDatastreamFileName() throws IOException,
173                                                    RepositoryException,
174                                                    InvalidChecksumException {
175        final Session session = repo.login();
176        objectService.createObject(session, "/testDatastreamObject");
177        datastreamService.createDatastreamNode(session,
178                                                  "/testDatastreamObject/testDatastreamNode5",
179                                                  "application/octet-stream",
180                                                  "xyz.jpg",
181                                                  new ByteArrayInputStream("asdf".getBytes()));
182
183        session.save();
184
185        final Datastream ds =
186            datastreamService.getDatastream(session,
187                                               "/testDatastreamObject/testDatastreamNode5");
188        final String filename = ds.getFilename();
189
190        assertEquals("xyz.jpg", filename);
191
192    }
