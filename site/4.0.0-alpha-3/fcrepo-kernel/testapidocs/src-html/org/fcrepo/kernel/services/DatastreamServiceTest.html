256    @Test
257    public void testRunFixityAndFixProblems() throws RepositoryException,
258            IOException, CacheLoaderException, URISyntaxException {
259
260        final Datastream mockDatastream = mock(Datastream.class);
261
262        final InputStream mockIS = mock(InputStream.class);
263
264        final LowLevelCacheEntry mockGoodEntry = mock(LowLevelCacheEntry.class);
265        when(mockGoodEntry.getInputStream()).thenReturn(mockIS);
266        final LowLevelCacheEntry mockBadEntry = mock(LowLevelCacheEntry.class);
267        final FixityResult mockGoodResult = mock(FixityResult.class);
268        when(mockGoodResult.matches(any(Long.class), any(URI.class)))
269                .thenReturn(true);
270        when(mockGoodResult.getEntry()).thenReturn(mockGoodEntry);
271        final FixityResult mockBadResult = mock(FixityResult.class);
272        when(mockBadResult.matches(any(Long.class), any(URI.class)))
273                .thenReturn(false);
274        when(mockBadResult.getEntry()).thenReturn(mockBadEntry);
275
276        final FixityResult mockRepairedResult = mock(FixityResult.class);
277        when(mockRepairedResult.matches(any(Long.class), any(URI.class)))
278                .thenReturn(true);
279        when(mockRepairedResult.getEntry()).thenReturn(mockGoodEntry);
280        when(mockRepairedResult.isSuccess()).thenReturn(true);
281
282        final Collection<FixityResult> mockFixityResults =
283                asList(mockGoodResult, mockBadResult);
284
285        when(mockNode.getNode(JCR_CONTENT)).thenReturn(mockContent);
286        when(mockContent.getProperty(JCR_DATA)).thenReturn(mockData);
287        when(mockData.getBinary()).thenReturn(mockBinaryValue);
288
289        when(mockDatastream.getNode()).thenReturn(mockNode);
290        when(mockDatastream.getContentDigest()).thenReturn(
291                URI.create("urn:sha1:abc"));
292
293        when(
294                llStore.transformLowLevelCacheEntries(eq(mockContent),
295                        any(CheckCacheEntryFixity.class))).thenReturn(
296                mockFixityResults);
297
298        when(mockBadEntry.checkFixity(any(URI.class), any(Long.class)))
299                .thenReturn(mockRepairedResult);
300        final Collection<FixityResult> fixityResults =
301                testObj.runFixityAndFixProblems(mockDatastream);
302
303        verify(mockBadResult).setRepaired();
304        verify(mockBadEntry).storeValue(mockIS);
305
306        assertTrue("expected to find good result", fixityResults
307                .contains(mockGoodResult));
308        assertTrue("expected to find repaired result", fixityResults
309                .contains(mockBadResult));
310    }
