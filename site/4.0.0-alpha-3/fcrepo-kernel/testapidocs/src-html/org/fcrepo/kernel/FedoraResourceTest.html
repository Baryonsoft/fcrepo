314    @Test
315    public void testReplacePropertiesDataset() throws Exception {
316
317        mockStatic(JcrRdfTools.class);
318        final DefaultGraphSubjects defaultGraphSubjects = new DefaultGraphSubjects(mockSession);
319        when(JcrRdfTools.withContext(defaultGraphSubjects, mockSession)).thenReturn(mockJcrRdfTools);
320
321        when(mockNode.getPath()).thenReturn("/xyz");
322
323        final Model propertiesModel = createDefaultModel();
324        propertiesModel.add(propertiesModel.createResource("a"),
325                               propertiesModel.createProperty("b"),
326                               "c");
327
328
329        propertiesModel.add(propertiesModel.createResource("i"),
330                               propertiesModel.createProperty("j"),
331                               "k");
332
333        propertiesModel.add(propertiesModel.createResource("x"),
334                               propertiesModel.createProperty("y"),
335                               "z");
336        final RdfStream propertiesStream = RdfStream.fromModel(propertiesModel);
337        when(mockJcrRdfTools.getJcrTriples(mockNode)).thenReturn(propertiesStream);
338
339        final RdfStream treeStream = new RdfStream();
340        when(mockJcrRdfTools.getTreeTriples(mockNode)).thenReturn(treeStream);
341        final Model problemsModel = createDefaultModel();
342        when(getProblemsModel()).thenReturn(problemsModel);
343
344        final Model replacementModel = createDefaultModel();
345
346        replacementModel.add(replacementModel.createResource("a"),
347                                replacementModel.createProperty("b"),
348                               "n");
349
350
351        replacementModel.add(replacementModel.createResource("i"),
352                                replacementModel.createProperty("j"),
353                               "k");
354
355        final Model replacements = testObj.replaceProperties(defaultGraphSubjects, replacementModel).asModel();
356
357        assertTrue(replacements.containsAll(replacementModel));
358
359        assertFalse(problemsModel.contains(propertiesModel.createResource("x"),
360                                              propertiesModel.createProperty("y"),
361                                              "z"));
362    }
363    @Test
364    public void shouldGetEtagForAnObject() throws RepositoryException {
365        final Property mockMod = mock(Property.class);
366        final Calendar modDate = Calendar.getInstance();
367        modDate.set(2013, JULY, 30, 0, 0, 0);
368        when(mockNode.getPath()).thenReturn("some-path");
369        when(mockNode.hasProperty(JCR_LASTMODIFIED)).thenReturn(true);
370        when(mockNode.getProperty(JCR_LASTMODIFIED)).thenReturn(mockMod);
371        when(mockMod.getDate()).thenReturn(modDate);
372
373        assertEquals(shaHex("some-path"
374                + testObj.getLastModifiedDate().toString()), testObj
375                .getEtagValue());
376    }
