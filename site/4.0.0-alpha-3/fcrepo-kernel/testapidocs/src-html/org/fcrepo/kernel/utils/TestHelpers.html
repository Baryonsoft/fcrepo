061    public static Node getContentNodeMock(final byte[] content) {
062        final Node mock = mock(Node.class);
063        final long size = content.length;
064        final String digest = checksumString(content);
065        final String digestType = "SHA-1";
066        final Property mockFedoraSize = mock(Property.class);
067        final Property mockProp = mock(Property.class);
068        final Property mockDigest = mock(Property.class);
069        final Property mockDigestType = mock(Property.class);
070        final Property mockCreated = mock(Property.class);
071        final Binary mockBin = mock(Binary.class);
072        try {
073            when(mockFedoraSize.getLong()).thenReturn(size);
074            when(mockBin.getSize()).thenReturn(size);
075            when(mockBin.getStream()).thenAnswer(new Answer<InputStream>() {
076
077                @Override
078                public InputStream answer(final InvocationOnMock inv) {
079                    return new ByteArrayInputStream(content);
080                }
081            });
082            when(mockProp.getBinary()).thenReturn(mockBin);
083            when(mockDigest.getString()).thenReturn(digest);
084            when(mockDigestType.getString()).thenReturn(digestType);
085            when(mock.getProperty(JCR_DATA)).thenReturn(mockProp);
086            when(mock.getProperty(CONTENT_SIZE)).thenReturn(mockFedoraSize);
087            when(mock.getProperty(CONTENT_DIGEST)).thenReturn(mockDigest);
088            when(mock.getProperty(JCR_CREATEDBY)).thenReturn(mockCreated);
089        } catch (final RepositoryException e) {
090        } // shhh
091        return mock;
092    }

094    public static String checksumString(final String content) {
095        return checksumString(content.getBytes());
096    }
097
098    public static String checksumString(final byte[] content) {
099        try {
100            final MessageDigest d = MessageDigest.getInstance("SHA-1");
101            final byte[] digest = d.digest(content);
102            return ContentDigest.asURI("SHA-1", digest).toString();
103        } catch (final NoSuchAlgorithmException e) {
104            e.printStackTrace();
105        }
106        return "";
107    }

109    @SuppressWarnings("unchecked")
110    public static PropertyIterator getPropertyIterator(final int numValues,
111            final long size) {
112        final PropertyIterator mock = mock(PropertyIterator.class);
113        final Property mockProp = mock(Property.class);
114        try {
115            when(mockProp.isMultiple()).thenReturn(numValues > 1);
116            if (numValues != 1) {
117                final Value[] values = new Value[numValues];
118                for (int i = 0; i < numValues; i++) {
119                    final Value mockVal = mock(Value.class);
120                    final Binary mockBin = mock(Binary.class);
121                    when(mockVal.getBinary()).thenReturn(mockBin);
122                    when(mockBin.getSize()).thenReturn(size);
123                    values[i] = mockVal;
124                }
125                when(mockProp.getValues()).thenReturn(values);
126            } else {
127                final Binary mockBin = mock(Binary.class);
128                when(mockBin.getSize()).thenReturn(size);
129                when(mockProp.getBinary()).thenReturn(mockBin);
130            }
131        } catch (final RepositoryException e) {
132        } // shhh
133        when(mock.getSize()).thenReturn(1L);
134        when(mock.hasNext()).thenReturn(true, false);
135        when(mock.nextProperty()).thenReturn(mockProp).thenThrow(
136                IndexOutOfBoundsException.class);
137        return mock;
138    }

140    public static byte[] randomData(final int byteLength) {
141        final byte[] bytes = new byte[byteLength];
142        GARBAGE_GENERATOR.nextBytes(bytes);
143        return bytes;
144    }

147    /**
148     * Set a field via reflection
149     *
150     * @param parent the owner object of the field
151     * @param name the name of the field
152     * @param obj the value to set
153     * @throws NoSuchFieldException
154     */
155    public static void setField(final Object parent, final String name,
156                                final Object obj) throws NoSuchFieldException {
157        /* check the parent class too if the field could not be found */
158        try {
159            final Field f = findField(parent.getClass(), name);
160            f.setAccessible(true);
161            f.set(parent, obj);
162        } catch (final Exception e) {
163            e.printStackTrace();
164        }
165    }
166
167    private static Field findField(final Class<?> clazz, final String name)
168            throws NoSuchFieldException {
169        for (final Field f : clazz.getDeclaredFields()) {
170            if (f.getName().equals(name)) {
171                return f;
172            }
173        }
174        if (clazz.getSuperclass() == null) {
175            throw new NoSuchFieldException("Field " + name +
176                                                   " could not be found");
177        } else {
178            return findField(clazz.getSuperclass(), name);
179        }
180    }
