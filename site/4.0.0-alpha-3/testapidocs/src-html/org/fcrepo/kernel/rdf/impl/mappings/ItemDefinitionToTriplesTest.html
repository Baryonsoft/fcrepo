025import static com.hp.hpl.jena.vocabulary.RDFS.domain;
026import static com.hp.hpl.jena.vocabulary.RDFS.label;
027import static org.fcrepo.kernel.rdf.impl.mappings.ItemDefinitionToTriples.getResource;
028import static org.junit.Assert.assertEquals;
029import static org.junit.Assert.assertTrue;
030import static org.mockito.Mockito.when;
031import static org.mockito.MockitoAnnotations.initMocks;
032import static org.slf4j.LoggerFactory.getLogger;
033
034import java.io.IOException;
035import java.util.Set;
036
037import javax.jcr.RepositoryException;
038import javax.jcr.nodetype.ItemDefinition;
039import javax.jcr.nodetype.NodeType;
040
041import org.junit.Before;
042import org.junit.Test;
043import org.mockito.InjectMocks;
044import org.mockito.Mock;
045import org.modeshape.jcr.api.Namespaced;
046import org.slf4j.Logger;
047
048import com.hp.hpl.jena.graph.Node;
049import com.hp.hpl.jena.graph.Triple;
050import com.hp.hpl.jena.rdf.model.Resource;
051
052public class ItemDefinitionToTriplesTest {
053
054    @InjectMocks
055    private ItemDefinitionToTriples<ItemDefinition> testMapper;
056
057    @Mock
058    private NamespacedItemDefinition mockItemDefinition;
059
060    @Mock
061    private NamespacedNodeType mockNodeType;
062
063    @Mock
064    private Node mockRDFNode = createURI("mock:domain");
065
066    @Before
067    public void setUp() throws RepositoryException {
068        initMocks(this);
069        when(mockItemDefinition.getName()).thenReturn("mock:ItemDefinition");
070        when(mockItemDefinition.getLocalName()).thenReturn("ItemDefinition");
071        when(mockItemDefinition.getNamespaceURI()).thenReturn("mock");
072    }
073
074    @Test
075    public void testGoodDefinition() throws IOException, RepositoryException {
076        final Set<Triple> results =
077            copyOf(testMapper.apply(mockItemDefinition));
078        LOGGER.debug("Created RDF: ");
079        for (final Triple t : results) {
080            LOGGER.debug("{}", t);
081        }
082
083        final Node subject =
084            getResource((ItemDefinition) mockItemDefinition).asNode();
085        assertTrue(results.contains(create(subject, type.asNode(), Property
086                .asNode())));
087        assertTrue(results.contains(create(subject, label.asNode(),
088                createLiteral(mockItemDefinition.getName()))));
089        assertTrue(results.contains(create(subject, domain.asNode(),
090                mockRDFNode)));
091    }
092
093    @Test(expected = RuntimeException.class)
094    public void testBadDefinition() throws RepositoryException {
095        when(mockItemDefinition.getNamespaceURI()).thenThrow(
096                new RepositoryException("Expected."));
097        testMapper.apply(mockItemDefinition);
098    }
099
100    @Test
101    public void testGetResourceFromNodeType() throws RepositoryException {
102        when(mockNodeType.getNamespaceURI()).thenReturn("namespace#");
103        when(mockNodeType.getLocalName()).thenReturn("localname");
104        final Resource answer = getResource((NodeType) mockNodeType);
105        assertEquals("namespace#", answer.getNameSpace());
106        assertEquals("localname", answer.getLocalName());
107    }
108
109    private static interface NamespacedItemDefinition extends Namespaced,
110            ItemDefinition {
111    }
112
113    private static interface NamespacedNodeType extends Namespaced, NodeType {
114    }
115
116    private static final Logger LOGGER =
117        getLogger(ItemDefinitionToTriplesTest.class);
118
119}
