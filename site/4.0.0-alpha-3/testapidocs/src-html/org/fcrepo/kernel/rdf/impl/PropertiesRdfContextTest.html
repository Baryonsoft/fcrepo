018
019import static com.hp.hpl.jena.rdf.model.ResourceFactory.createResource;
020import static org.fcrepo.kernel.RdfLexicon.HAS_CONTENT;
021import static org.fcrepo.kernel.RdfLexicon.HAS_CONTENT_LOCATION;
022import static org.fcrepo.kernel.RdfLexicon.IS_CONTENT_OF;
023import static org.junit.Assert.assertTrue;
024import static org.mockito.Mockito.when;
025import static org.mockito.MockitoAnnotations.initMocks;
026import static org.modeshape.jcr.api.JcrConstants.JCR_CONTENT;
027import static org.slf4j.LoggerFactory.getLogger;
028
029import java.io.IOException;
030import java.io.StringWriter;
031import java.io.Writer;
032
033import javax.jcr.Node;
034import javax.jcr.Repository;
035import javax.jcr.RepositoryException;
036import javax.jcr.Session;
037import javax.jcr.nodetype.NodeType;
038
039import org.fcrepo.kernel.rdf.GraphSubjects;
040import org.fcrepo.kernel.services.LowLevelStorageService;
041import org.fcrepo.kernel.utils.LowLevelCacheEntry;
042import org.junit.Before;
043import org.junit.Test;
044import org.mockito.Mock;
045import org.slf4j.Logger;
046
047import com.google.common.collect.ImmutableSet;
048import com.hp.hpl.jena.rdf.model.Model;
049import com.hp.hpl.jena.rdf.model.Resource;
050
051public class PropertiesRdfContextTest {
052
053    @Test
054    public void testForLowLevelStorageTriples() throws RepositoryException,
055                                               IOException {
056        final Model results =
057            new PropertiesRdfContext(mockNode, mockGraphSubjects,
058                    mockLowLevelStorageService).asModel();
059        logRdf("Retrieved RDF for testForLowLevelStorageTriples():", results);
060        assertTrue("Didn't find triple showing node has content!", results
061                .contains(mockSubject, HAS_CONTENT, mockContentSubject));
062        assertTrue("Didn't find triple showing content has node!", results
063                .contains(mockContentSubject, IS_CONTENT_OF, mockSubject));
064        assertTrue("Didn't find triple showing content has location!", results
065                .contains(mockContentSubject, HAS_CONTENT_LOCATION,
066                        createResource(MOCK_EXTERNAL_IDENTIFIER)));
067    }
068
069    @Before
070    public void setUp() throws RepositoryException {
071        initMocks(this);
072        when(mockNode.getSession()).thenReturn(mockSession);
073        when(mockSession.getRepository()).thenReturn(mockRepository);
074        when(mockNode.hasNode(JCR_CONTENT)).thenReturn(true);
075        when(mockNode.getNode(JCR_CONTENT)).thenReturn(mockContentNode);
076        when(mockNode.hasProperties()).thenReturn(false);
077        when(mockNode.getMixinNodeTypes()).thenReturn(new NodeType[] {});
078        when(mockContentNode.getMixinNodeTypes()).thenReturn(new NodeType[] {});
079        when(mockContentNode.hasProperties()).thenReturn(false);
080        when(
081                mockLowLevelStorageService
082                        .getLowLevelCacheEntries(mockContentNode)).thenReturn(
083                ImmutableSet.of(mockLowLevelCacheEntry));
084        when(mockLowLevelCacheEntry.getExternalIdentifier()).thenReturn(
085                MOCK_EXTERNAL_IDENTIFIER);
086        when(mockGraphSubjects.getGraphSubject(mockNode)).thenReturn(
087                mockSubject);
088        when(mockGraphSubjects.getGraphSubject(mockContentNode)).thenReturn(
089                mockContentSubject);
090        when(mockNode.getPrimaryNodeType()).thenReturn(mockNodeType);
091        when(mockContentNode.getPrimaryNodeType()).thenReturn(mockNodeType);
092        when(mockNodeType.getName()).thenReturn("not:root");
093    }
094
095    private static final String MOCK_EXTERNAL_IDENTIFIER =
096        "info:external-identifier";
097
098    private static final Resource mockContentSubject =
099        createResource("http://example.com/node/jcr:content");;
100
101    private static final Resource mockSubject =
102        createResource("http://example.com/node");
103
104    @Mock
105    private Node mockNode, mockContentNode;
106
107    @Mock
108    private NodeType mockNodeType;
109
110    @Mock
111    private GraphSubjects mockGraphSubjects;
112
113    @Mock
114    private LowLevelStorageService mockLowLevelStorageService;
115
116    @Mock
117    private Session mockSession;
118
119    @Mock
120    private Repository mockRepository;
121
122    @Mock
123    private LowLevelCacheEntry mockLowLevelCacheEntry;
124
125    private void
126            logRdf(final String message, final Model model) throws IOException {
127        LOGGER.debug(message);
128        try (Writer w = new StringWriter()) {
129            model.write(w);
130            LOGGER.debug("\n" + w.toString());
131        }
132    }
133
134    private static final Logger LOGGER =
135        getLogger(PropertiesRdfContextTest.class);
136}
