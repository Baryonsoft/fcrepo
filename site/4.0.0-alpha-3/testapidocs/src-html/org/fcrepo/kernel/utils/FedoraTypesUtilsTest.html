160    @Test
161    public void testIsMultipleValuedProperty() throws RepositoryException {
162        final Property mockYes = mock(Property.class);
163        when(mockYes.isMultiple()).thenReturn(true);
164        final Property mockNo = mock(Property.class);
165        final Predicate<Property> test = isMultipleValuedProperty;
166        try {
167            test.apply(null);
168            fail("Null values should throw a NullPointerException");
169        } catch (final NullPointerException e) {
170        }
171        boolean actual = test.apply(mockYes);
172        assertEquals(true, actual);
173        actual = test.apply(mockNo);
174        assertEquals(false, actual);
175        when(mockYes.isMultiple()).thenThrow(new RepositoryException());
176        try {
177            test.apply(mockYes);
178            fail("Unexpected completion after RepositoryException!");
179        } catch (final RuntimeException e) {} // expected
180    }

314        when(mockResults.getRows()).thenReturn(mockIterator);

322        final long count = getRepositorySize(mockRepository);
323        assertEquals("Got wrong count!", 16L, count);
324        verify(mockSession).logout();
