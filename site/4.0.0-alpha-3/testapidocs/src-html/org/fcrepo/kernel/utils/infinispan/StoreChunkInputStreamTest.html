017package org.fcrepo.kernel.utils.infinispan;
018
019import static org.fcrepo.kernel.utils.TestHelpers.randomData;
020import static org.junit.Assert.assertEquals;
021import static org.junit.Assert.assertTrue;
022import static org.mockito.Matchers.anyString;
023import static org.mockito.Mockito.times;
024import static org.mockito.Mockito.verify;
025import static org.mockito.Mockito.when;
026import static org.mockito.MockitoAnnotations.initMocks;
027
028import java.io.IOException;
029
030import org.fcrepo.kernel.utils.TestHelpers;
031import org.infinispan.container.entries.InternalCacheEntry;
032import org.infinispan.loaders.CacheLoaderException;
033import org.infinispan.loaders.CacheStore;
034import org.junit.Before;
035import org.junit.Test;
036import org.mockito.Mock;
037
038public class StoreChunkInputStreamTest {
039
040    private static final int DATA_SIZE = 1024;
041
042    private StoreChunkInputStream testObj;
043
044    @Mock
045    private CacheStore mockStore;
046
047    @Mock
048    private InternalCacheEntry mockEntry;
049
050    private static final String MOCK_KEY = "key-to-a-mock-blob";
051
052    private static final String MOCK_FIRST_CHUNK = MOCK_KEY + "-0";
053
054    @Before
055    public void setUp() throws CacheLoaderException {
056        initMocks(this);
057        when(mockStore.containsKey(MOCK_FIRST_CHUNK)).thenReturn(true);
058        testObj = new StoreChunkInputStream(mockStore, MOCK_KEY);
059    }
060
061    @Test
062    public void testRead() throws IOException {
063        testObj.read();
064    }
065
066    @Test
067    public void testBufferedRead() throws IOException, CacheLoaderException {
068        final byte[] data = randomData(DATA_SIZE);
069        when(mockEntry.getValue()).thenReturn(data);
070        when(mockStore.load(anyString())).thenReturn(mockEntry).thenReturn(
071                mockEntry).thenReturn(null);
072        final int partition = 234;
073        final int expected = (DATA_SIZE - partition);
074        final byte[] buffer = new byte[DATA_SIZE];
075        long actual = testObj.read(buffer, 0, expected);
076        // can read less than a block of data
077        assertEquals(expected, actual);
078        // will not load the next chunk if more data is available
079        actual = testObj.read(buffer, 0, DATA_SIZE);
080        assertEquals(partition, actual);
081        actual = testObj.read(buffer, 0, DATA_SIZE);
082        // will load the next chunk if no data is available
083        assertEquals(DATA_SIZE, actual);
084        // and will report the end of the data accurately
085        actual = testObj.read(buffer, 0, DATA_SIZE);
086        assertEquals(-1, actual);
087
088    }
089
090    @Test
091    public void testAvailable() throws IOException, CacheLoaderException {
092        final byte[] data = TestHelpers.randomData(DATA_SIZE);
093        when(mockEntry.getValue()).thenReturn(data);
094        when(mockStore.load(MOCK_FIRST_CHUNK)).thenReturn(mockEntry);
095        assertEquals(0, testObj.available());
096        final int partition = 435;
097        testObj.skip(partition);
098        // part of the first buffer remains
099        assertEquals(DATA_SIZE - partition, testObj.available());
100        testObj.skip(DATA_SIZE - partition);
101        // none of the first buffer remains
102        assertEquals(0, testObj.available());
103        testObj.skip(1);
104        // no buffers remain
105        assertEquals(-1, testObj.available());
106    }
107
108    @Test
109    public void testSkip() throws IOException, CacheLoaderException {
110        final long expected = (DATA_SIZE - 1);
111        final byte[] data = randomData(DATA_SIZE);
112        when(mockEntry.getValue()).thenReturn(data);
113        when(mockStore.load(MOCK_FIRST_CHUNK)).thenReturn(mockEntry);
114        final long actual = testObj.skip(expected);
115        assertEquals(expected, actual);
116        verify(mockStore).load(anyString());
117        verify(mockStore).load(MOCK_FIRST_CHUNK);
118        verify(mockEntry).getValue();
119        assertTrue(testObj.read() > -1);
120        assertEquals(-1, testObj.read());
121    }
122
123    @Test
124    public void testSkipMultipleBuffers() throws IOException,
125            CacheLoaderException {
126        final byte[] data = randomData(DATA_SIZE);
127        when(mockEntry.getValue()).thenReturn(data);
128        when(mockStore.load(anyString())).thenReturn(mockEntry).thenReturn(
129                mockEntry).thenReturn(null);
130        long expected = (DATA_SIZE);
131        // ask for more than the buffer
132        long actual = testObj.skip(DATA_SIZE + 1);
133        // we should skip only one complete buffer
134        assertEquals(expected, actual);
135        // ok, skip all but the last byte remaining
136        expected = (DATA_SIZE - 1);
137        actual = testObj.skip(expected);
138        // new buffer, mostly skipped
139        assertEquals(expected, actual);
140        // we should still have 1 more byte
141        assertTrue(testObj.read() > -1);
142        // but only the one
143        assertEquals(-1, testObj.read());
144        // and we only had two cacheEntries
145        verify(mockEntry, times(2)).getValue();
146    }
147
148    @Test
149    public void testNextChunk() throws IOException {
150        testObj.nextChunk();
151    }
152
153}
