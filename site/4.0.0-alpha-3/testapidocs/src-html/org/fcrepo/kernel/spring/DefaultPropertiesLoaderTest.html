016package org.fcrepo.kernel.spring;
017
018import org.junit.Assert;
019import org.junit.After;
020import org.junit.Before;
021import org.junit.Test;
022
023import java.io.File;
024
025/**
026 * @author Andrew Woods
027 *         Date: 10/20/13
028 */
029public class DefaultPropertiesLoaderTest {
030
031    private DefaultPropertiesLoader loader;
032
033    private static final String PROP_FLAG = "integration-test";
034    private static final String PROP_TEST = "fcrepo.ispn.repo.CacheDirPath";
035    private static final String HOME_PROP = "fcrepo.home";
036
037    @Before
038    public void setUp() throws Exception {
039        loader = new DefaultPropertiesLoader();
040    }
041
042    @After
043    public void tearDown() throws Exception {
044        clearProps();
045    }
046
047    private void clearProps() {
048        System.clearProperty(PROP_FLAG);
049        System.clearProperty(PROP_TEST);
050        System.clearProperty(HOME_PROP);
051    }
052
053    @Test
054    public void testLoadSystemProperties() throws Exception {
055        System.setProperty(PROP_FLAG, "true");
056
057        loader.loadSystemProperties();
058        Assert.assertNotNull(System.getProperty(PROP_FLAG));
059        Assert.assertNull(System.getProperty(PROP_TEST));
060    }
061
062    @Test
063    public void testLoadSystemPropertiesProduction() throws Exception {
064        loader.loadSystemProperties();
065        Assert.assertNull(System.getProperty(PROP_FLAG));
066        Assert.assertNotNull(System.getProperty(PROP_TEST));
067    }
068
069    @Test
070    public void testContentIsInWorkingDir() {
071       loader.loadSystemProperties();
072       Assert.assertTrue("Default directories are within working directory.",
073               containsPath(System.getProperty(PROP_TEST),
074                       System.getProperty("user.dir")));
075    }
076
077    @Test
078    public void testCustomHomeDirWithRelativeSubdirs() {
079        System.setProperty(HOME_PROP, "/test");
080        System.setProperty(PROP_TEST, "sub");
081
082        loader.loadSystemProperties();
083
084        File home = new File("/test");
085        Assert.assertTrue("Relative subdirs are within fcrepo.home directory.",
086                containsPath(System.getProperty(PROP_TEST),
087                        home.getAbsolutePath()));
088        clearProps();
089    }
090
091    @Test
092    public void testCustomHomeDirWithAbsoluteSubdirs() {
093        System.setProperty(HOME_PROP, "/tmp/test");
094        System.setProperty(PROP_TEST, "/tmp/sub");
095
096        loader.loadSystemProperties();
097
098        File home = new File("/test");
099        Assert.assertFalse("Absolute subdirs are idependent of fcrepo.home.",
100                containsPath(System.getProperty(PROP_TEST),
101                        home.getAbsolutePath()));
102        clearProps();
103    }
104
105    @Test
106    public void relativePathsWorkForFedoraHome() {
107        System.setProperty(HOME_PROP, "test");
108        System.setProperty(PROP_TEST, "sub");
109
110        loader.loadSystemProperties();
111
112        Assert.assertEquals(new File(new File("test"), "sub"),
113                new File(System.getProperty(PROP_TEST)));
114        clearProps();
115    }
116
117    private boolean containsPath(String path, String parentPath) {
118        final File parent = new File(parentPath);
119        for (File f = new File(path) ; f.getParentFile() != null ; f = f.getParentFile()) {
120            if (f.getParentFile().equals(parent)) {
121                return true;
122            }
123        }
124        return false;
125    }
126}
