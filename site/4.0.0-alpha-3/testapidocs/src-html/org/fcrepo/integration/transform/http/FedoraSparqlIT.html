054    private Session session;

056    @Before
057    public void setUpTestData() throws RepositoryException {
058        session = repo.login();
059        session.setNamespacePrefix("zz", "http://zz.com/");
060        final ValueFactory valueFactory = session.getValueFactory();
061        final FedoraResource fedoraResource = new FedoraResource(session, "/abc", JcrConstants.NT_FOLDER);
062        final FedoraResource fedoraResource2 = new FedoraResource(session, "/xyz", JcrConstants.NT_FOLDER);
063        final FedoraResource fedoraResource3 = new FedoraResource(session, "/anobject", JcrConstants.NT_FOLDER);
064
065        fedoraResource.getNode().setProperty("dc:title", new Value[] { valueFactory.createValue("xyz") });
066        fedoraResource.getNode().setProperty("fedorarelsext:hasPart", new Value[] { valueFactory.createValue(fedoraResource2.getNode()) });
067        fedoraResource2.getNode().setProperty("fedorarelsext:isPartOf", new Value[] { valueFactory.createValue(fedoraResource.getNode()) });
068        fedoraResource3.getNode().setProperty("zz:name", new Value[] { valueFactory.createValue("junk") });
069        fedoraResource3.getNode().setProperty("rdf:type", new Value[] { valueFactory.createValue("info:some-type") });
070        session.save();
071
072    }
073
074    @After
075    public void destroy() {
076        session.logout();
077    }
078    
079    @Test
080    public void itShouldHaveAnHtmlView() throws IOException {
081        HttpGet request = new HttpGet(serverAddress + "/fcr:sparql");
082        request.addHeader("Accept", "text/html");
083        HttpResponse response = client.execute(request);
084        assertEquals(200, response.getStatusLine().getStatusCode());
085        String content = EntityUtils.toString(response.getEntity());
086        assertTrue(content.contains("SPARQL"));
087        assertTrue(content.contains("PREFIX dc: <http://purl.org/dc/elements/1.1/>"));
088    }
089
090    @Test
091    public void itShouldWorkWithSimpleProperties() throws IOException, RepositoryException {
092
093        final String sparql = "PREFIX  dc:  <http://purl.org/dc/elements/1.1/> SELECT ?subject WHERE { ?subject dc:title \"xyz\"}";
094
095        String content = getResponseContent(sparql);
096        final ResultSet resultSet = ResultSetFactory.fromTSV(IOUtils.toInputStream(content));
097
098
099        assertTrue(resultSet.hasNext());
100
101        assertEquals("subject", resultSet.getResultVars().get(0));
102
103        assertEquals(serverAddress + "/abc", resultSet.next().get("subject").toString());
104    }

139    @Test
140    public void itShouldWorkWithReferenceProperties() throws IOException, RepositoryException {
141
142        final String sparql = "PREFIX  fedorarelsext:  <http://fedora.info/definitions/v4/rels-ext#> SELECT ?subject ?part WHERE { ?subject fedorarelsext:hasPart ?part }";
143
144        String content = getResponseContent(sparql);
145        final ResultSet resultSet = ResultSetFactory.fromTSV(IOUtils.toInputStream(content));
146
147
148        assertTrue(resultSet.hasNext());
149
150        assertEquals("subject", resultSet.getResultVars().get(0));
151
152        final QuerySolution row = resultSet.next();
153        assertEquals(serverAddress + "/abc", row.get("subject").toString());
154        assertEquals(serverAddress + "/xyz", row.get("part").toString());
155    }
156
157    @Test
158    public void itShouldWorkWithJoinedQueries() throws IOException, RepositoryException {
159
160        final String sparql = "PREFIX  fedorarelsext:  <http://fedora.info/definitions/v4/rels-ext#>\n" +
161                              "PREFIX  dc:  <http://purl.org/dc/elements/1.1/>\n" +
162                              "SELECT ?part ?collectionTitle WHERE { ?part fedorarelsext:isPartOf ?collection .\n" +
163                                  "  ?collection dc:title ?collectionTitle }";
164
165        String content = getResponseContent(sparql);
166        final ResultSet resultSet = ResultSetFactory.fromTSV(IOUtils.toInputStream(content));
167
168
169        assertTrue(resultSet.hasNext());
170
171        assertEquals("part", resultSet.getResultVars().get(0));
172
173        final QuerySolution row = resultSet.next();
174        assertEquals(serverAddress + "/xyz", row.get("part").toString());
175        assertEquals("xyz", row.get("collectionTitle").asLiteral().getLexicalForm());
176    }
177
178    @Test
179    public void itShouldIndexCustomProperties() throws Exception {
180        final String sparql = "PREFIX zz: <http://zz.com/> SELECT ?subject WHERE { ?subject zz:name \"junk\"}";
181
182        String content = getResponseContent(sparql);
183
184        final ResultSet resultSet = ResultSetFactory.fromTSV(IOUtils.toInputStream(content));
185        assertTrue(resultSet.hasNext());
186        assertEquals("subject", resultSet.getResultVars().get(0));
187        assertEquals(serverAddress + "/anobject", resultSet.next().get("subject").toString());
188    }
189
190    private String getResponseContent(final String sparql) throws IOException {
191        HttpPost sparqlRequest = new HttpPost(serverAddress + "/fcr:sparql");
192        BasicHttpEntity entity =  new BasicHttpEntity();
193        entity.setContent(IOUtils.toInputStream(sparql));
194        sparqlRequest.setEntity(entity);
195        HttpResponse response = client.execute(sparqlRequest);
196        assertEquals(200, response.getStatusLine().getStatusCode());
197
198        String content = EntityUtils.toString(response.getEntity());
199        logger.trace("Retrieved sparql feed:\n" + content);
200        return content;
201    }
