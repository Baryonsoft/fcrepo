074    public void testAddAndRetrieveVersion() throws Exception {
075        final String pid = UUID.randomUUID().toString();
076        createObject(pid);
077
078        logger.info("Setting a title");
079        patchLiteralProperty(serverAddress + pid, "http://purl.org/dc/elements/1.1/title", "First Title");
080
081        final GraphStore nodeResults = getContent(serverAddress + pid);
082        assertTrue("Should find original title", nodeResults.contains(Node.ANY, Node.ANY, DC_TITLE.asNode(), NodeFactory.createLiteral("First Title")));
083
084        logger.info("Posting version v0.0.1");
085        postObjectVersion(pid, "v0.0.1");
086
087        logger.info("Replacing the title");
088        patchLiteralProperty(serverAddress + pid, "http://purl.org/dc/elements/1.1/title", "Second Title");
089
090        final GraphStore versionResults = getContent(serverAddress + pid + "/fcr:versions/v0.0.1");
091        logger.info("Got version profile:");
092
093        assertTrue("Didn't find a version triple!",
094                      versionResults.contains(Node.ANY, Node.ANY, HAS_PRIMARY_TYPE.asNode(), NodeFactory.createLiteral("nt:frozenNode")));
095
096        assertTrue("Should find a title in historic version", versionResults.contains(Node.ANY, Node.ANY, DC_TITLE.asNode(), Node.ANY));
097        assertTrue("Should find original title in historic version", versionResults.contains(Node.ANY, Node.ANY, DC_TITLE.asNode(), NodeFactory.createLiteral("First Title")));
098        assertFalse("Should not find the updated title in historic version", versionResults.contains(Node.ANY, Node.ANY, DC_TITLE.asNode(), NodeFactory.createLiteral("Second Title")));
099    }

198    public void isAutoVersionedContentStillAccessible() throws Exception {
199        final String objName = UUID.randomUUID().toString();
200        final String dsName = UUID.randomUUID().toString();
201        final String firstVersionText = "foo";
202        final String secondVersionText = "bar";
203
204        createObject(objName);
205        createDatastream(objName, dsName, firstVersionText);
206
207        setAutoVersioning(serverAddress + objName + "/" + dsName);
208
209        mutateDatastream(objName, dsName, secondVersionText);
210
211        final HttpGet retrieveMutatedDataStreamMethod =
212                new HttpGet(serverAddress +
213                        objName + "/" + dsName + "/fcr:content");
214        assertEquals("Datastream didn't accept mutation!", secondVersionText,
215                EntityUtils.toString(
216                        client.execute(
217                                retrieveMutatedDataStreamMethod).getEntity()));
218
219        final HttpGet getVersion =
220                new HttpGet(serverAddress + objName + "/" + dsName + "/fcr:versions");
221        logger.debug("Retrieved version profile:");
222        final GraphStore results = getGraphStore(getVersion);
223        final Resource subject =
224                createResource(serverAddress + objName + "/" + dsName);
225        assertTrue("Didn't find a version triple!",
226                results.contains(Node.ANY, subject.asNode(), HAS_VERSION.asNode(), Node.ANY));
227
228        final List<String> versionsInOrder = getChronologicalyOrderedVersionUrls(results, subject.asNode());
229        assertEquals("There must be a root version with two revisions.", 3, versionsInOrder.size());
230
231        final HttpGet retrieveFirstVersion = new HttpGet(versionsInOrder.get(1) + "/fcr:content");
232        assertEquals("First version wasn't preserved as expected!",
233                firstVersionText,
234                EntityUtils.toString(client.execute(
235                        retrieveFirstVersion).getEntity()));
236    }

278    private void testDatastreamContentUpdatesCreateNewVersions(final String objName, final String dsName) throws IOException {
279        final String firstVersionText = "foo";
280        final String secondVersionText = "bar";
281
282        createDatastream(objName, dsName, firstVersionText);
283
284        final GraphStore dsInitialVersion = getContent(serverAddress + objName);
285        assertTrue("Should find auto-created versoning policy",
286                dsInitialVersion.contains(Node.ANY, createResource(serverAddress + objName+ "/" + dsName).asNode(), VERSIONING_POLICY.asNode(), NodeFactory.createLiteral("auto-version")));
287
288        mutateDatastream(objName, dsName, secondVersionText);
289
290        final HttpGet retrieveMutatedDataStreamMethod =
291                new HttpGet(serverAddress +
292                        objName + "/" + dsName + "/fcr:content");
293        assertEquals("Datastream didn't accept mutation!", secondVersionText,
294                EntityUtils.toString(
295                        client.execute(
296                                retrieveMutatedDataStreamMethod).getEntity()));
297
298        final HttpGet getVersion =
299                new HttpGet(serverAddress + objName + "/" + dsName + "/fcr:versions");
300        logger.debug("Retrieved version profile:");
301        final GraphStore results = getGraphStore(getVersion);
302        final Resource subject =
303                createResource(serverAddress + objName + "/" + dsName);
304        assertTrue("Didn't find a version triple!",
305                results.contains(Node.ANY, subject.asNode(), HAS_VERSION.asNode(), Node.ANY));
306
307
308        final List<String> versionsInOrder = getChronologicalyOrderedVersionUrls(results, subject.asNode());
309        assertEquals("There must be a root version with two revisions.", 3, versionsInOrder.size());
310
311        final HttpGet retrieveFirstVersion = new HttpGet(versionsInOrder.get(1) + "/fcr:content");
312        assertEquals("First version wasn't preserved as expected!",
313                firstVersionText,
314                EntityUtils.toString(client.execute(
315                        retrieveFirstVersion).getEntity()));
316
317        final HttpGet retrieveSecondVersion = new HttpGet(versionsInOrder.get(2) + "/fcr:content");
318        assertEquals("Second version wasn't preserved as expected!",
319                secondVersionText,
320                EntityUtils.toString(client.execute(
321                        retrieveSecondVersion).getEntity()));
322    }

324    private List<String> getChronologicalyOrderedVersionUrls(final GraphStore graph, final Node subject) {
325        final List<String> versionUrls = new ArrayList<String>();
326        final Iterator<Quad> versionIt = graph.find(Node.ANY, subject, HAS_VERSION.asNode(), Node.ANY);
327        while (versionIt.hasNext()) {
328            versionUrls.add(versionIt.next().getObject().getURI());
329        }
330        Collections.sort(versionUrls, new Comparator<String>() {
331            @Override
332            public int compare(final String o1, final String o2) {
333                String d1 = getFirstLiteralIfExists(graph, createResource(o1).asNode(), LAST_MODIFIED_DATE.asNode(), "");
334                String d2 = getFirstLiteralIfExists(graph, createResource(o2).asNode(), LAST_MODIFIED_DATE.asNode(), "");
335                if (d1.equals(d2)) {
336                    d1 += o1;
337                    d2 += o2;
338                }
339                System.out.println("COMPARING: " + d1 + " compare with " + d2);
340                return d1.compareTo(d2);
341            }
342        });
343        for (final String url : versionUrls) {
344            System.out.println(url + " " + getFirstLiteralIfExists(graph, createResource(url).asNode(), LAST_MODIFIED_DATE.asNode(), "(none)"));
345        }
346        return versionUrls;
347    }

349    private String getFirstLiteralIfExists(final GraphStore graph, final Node subject, final Node predicate, final String defaultValue) {
350        final Iterator<Quad> quads = graph.find(Node.ANY, subject, predicate, Node.ANY);
351        if (quads.hasNext()) {
352            return quads.next().getObject().getLiteralValue().toString();
353        }
354        return defaultValue;
355    }

373    private void patchLiteralProperty(final String url, final String predicate, final String literal) throws IOException {
374        final HttpPatch updateObjectGraphMethod =
375                new HttpPatch(url);
376        updateObjectGraphMethod.addHeader("Content-Type",
377                "application/sparql-update");
378        final BasicHttpEntity e = new BasicHttpEntity();
379        e.setContent(new ByteArrayInputStream(
380                ("INSERT DATA { <> <" + predicate + "> \"" + literal + "\" } ")
381                        .getBytes()));
382        updateObjectGraphMethod.setEntity(e);
383        final HttpResponse response = client.execute(updateObjectGraphMethod);
384        assertEquals(NO_CONTENT.getStatusCode(), response.getStatusLine()
385                .getStatusCode());
386    }

388    private void addMixin(final String url, final String mixinUrl) throws IOException {
389        final HttpPatch updateObjectGraphMethod =
390                new HttpPatch(url);
391        updateObjectGraphMethod.addHeader("Content-Type",
392                "application/sparql-update");
393        final BasicHttpEntity e = new BasicHttpEntity();
394
395        e.setContent(new ByteArrayInputStream(
396                ("INSERT DATA { <> <http://www.w3.org/1999/02/22-rdf-syntax-ns#type> <" + mixinUrl + "> . } ")
397                        .getBytes()));
398        updateObjectGraphMethod.setEntity(e);
399        final HttpResponse response = client.execute(updateObjectGraphMethod);
400        assertEquals(Response.Status.NO_CONTENT.getStatusCode(), response.getStatusLine()
401                .getStatusCode());
402    }
