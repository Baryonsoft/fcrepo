017package org.fcrepo.integration.http.api;
018
019import static java.lang.Integer.MAX_VALUE;
020import static java.lang.Integer.parseInt;
021import static javax.ws.rs.core.Response.Status.CREATED;
022import static javax.ws.rs.core.Response.Status.OK;
023import static org.fcrepo.http.commons.test.util.TestHelpers.parseTriples;
024import static org.junit.Assert.assertEquals;
025import static org.slf4j.LoggerFactory.getLogger;
026
027import java.io.IOException;
028import java.io.UnsupportedEncodingException;
029
030import com.hp.hpl.jena.update.GraphStore;
031import org.apache.http.HttpResponse;
032import org.apache.http.client.ClientProtocolException;
033import org.apache.http.client.HttpClient;
034import org.apache.http.client.methods.HttpPost;
035import org.apache.http.client.methods.HttpPut;
036import org.apache.http.client.methods.HttpUriRequest;
037import org.apache.http.entity.StringEntity;
038import org.apache.http.impl.client.HttpClientBuilder;
039import org.apache.http.util.EntityUtils;
040import org.junit.Before;
041import org.junit.runner.RunWith;
042import org.slf4j.Logger;
043import org.springframework.test.context.ContextConfiguration;
044import org.springframework.test.context.junit4.SpringJUnit4ClassRunner;
045
046@RunWith(SpringJUnit4ClassRunner.class)
047@ContextConfiguration("/spring-test/test-container.xml")
048public abstract class AbstractResourceIT {
049
050    protected Logger logger;
051
052    @Before
053    public void setLogger() {
054        logger = getLogger(this.getClass());
055    }
056
057    protected static final int SERVER_PORT = parseInt(System.getProperty(
058            "test.port", "8080"));
059
060    protected static final String HOSTNAME = "localhost";
061
062    protected static final String serverAddress = "http://" + HOSTNAME + ":" +
063            SERVER_PORT + "/";
064
065    protected static HttpClient client = createClient();
066
067    protected static HttpClient createClient() {
068        return HttpClientBuilder.create().setMaxConnPerRoute(MAX_VALUE)
069                .setMaxConnTotal(MAX_VALUE).build();
070    }
071
072    protected static HttpPost postObjMethod(final String pid) {
073        return new HttpPost(serverAddress + pid);
074    }
075
076    protected static HttpPost postObjMethod(final String pid, final String query) {
077        if (query.equals("")) {
078            return new HttpPost(serverAddress + pid);
079        } else {
080            return new HttpPost(serverAddress + pid + "?" + query);
081        }
082    }
083
084    protected static HttpPost postDSMethod(final String pid, final String ds,
085        final String content) throws UnsupportedEncodingException {
086        final HttpPost post =
087                new HttpPost(serverAddress + pid + "/" + ds +
088                        "/fcr:content");
089        post.setEntity(new StringEntity(content));
090        return post;
091    }
092
093    protected static HttpPut putDSMethod(final String pid, final String ds,
094        final String content) throws UnsupportedEncodingException {
095        final HttpPut put =
096                new HttpPut(serverAddress + pid + "/" + ds +
097                        "/fcr:content");
098
099        put.setEntity(new StringEntity(content));
100        return put;
101    }
102
103    protected HttpResponse execute(final HttpUriRequest method)
104        throws ClientProtocolException, IOException {
105        logger.debug("Executing: " + method.getMethod() + " to " +
106                         method.getURI());
107        return client.execute(method);
108    }
109
110    protected int getStatus(final HttpUriRequest method)
111        throws ClientProtocolException, IOException {
112        final HttpResponse response = execute(method);
113        final int result = response.getStatusLine().getStatusCode();
114        if (!(result > 199) || !(result < 400)) {
115            logger.warn(EntityUtils.toString(response.getEntity()));
116        }
117        return result;
118    }
119
120
121    protected GraphStore getGraphStore(final HttpClient client, final HttpUriRequest method) throws IOException {
122
123        if (method.getFirstHeader("Accept") == null) {
124            method.addHeader("Accept", "application/n-triples");
125        } else {
126            logger.debug("Retrieving RDF in mimeType: {}", method
127                    .getFirstHeader("Accept"));
128        }
129
130        final HttpResponse response = client.execute(method);
131        assertEquals(OK.getStatusCode(), response.getStatusLine()
132                                             .getStatusCode());
133        final GraphStore result = parseTriples(response.getEntity());
134        logger.debug("Retrieved RDF: {}", result);
135        return result;
136
137    }
138
139    protected GraphStore getGraphStore(final HttpUriRequest method) throws IOException {
140        return getGraphStore(client, method);
141    }
142
143    protected HttpResponse createObject(final String pid) throws IOException {
144        final HttpResponse response = client.execute(postObjMethod(pid));
145        assertEquals(CREATED.getStatusCode(), response.getStatusLine().getStatusCode());
146        return response;
147    }
148
149    protected HttpResponse createDatastream(final String pid, final String dsid, final String content) throws IOException {
150        logger.trace(
151                "Attempting to create datastream for object: {} at datastream ID: {}",
152                pid, dsid);
153        final HttpResponse response =
154            client.execute(postDSMethod(pid, dsid, content));
155        assertEquals(CREATED.getStatusCode(), response.getStatusLine().getStatusCode());
156        return response;
157    }
158
159}
