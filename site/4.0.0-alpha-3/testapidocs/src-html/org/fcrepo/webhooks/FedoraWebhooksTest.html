017package org.fcrepo.webhooks;
018
019import static org.fcrepo.webhooks.FedoraWebhooks.WEBHOOK_CALLBACK_PROPERTY;
020import static org.fcrepo.webhooks.FedoraWebhooks.WEBHOOK_JCR_TYPE;
021import static org.fcrepo.webhooks.FedoraWebhooks.WEBHOOK_PREFIX;
022import static org.mockito.Mockito.mock;
023import static org.mockito.Mockito.verify;
024import static org.mockito.Mockito.when;
025
026import javax.jcr.Node;
027import javax.jcr.NodeIterator;
028import javax.jcr.Property;
029import javax.jcr.Session;
030import javax.jcr.Workspace;
031import javax.jcr.nodetype.NodeType;
032
033import org.fcrepo.kernel.observer.FedoraEvent;
034import org.fcrepo.http.commons.session.SessionFactory;
035import org.fcrepo.http.commons.test.util.TestHelpers;
036import org.junit.Before;
037import org.junit.Test;
038import org.modeshape.jcr.api.Repository;
039import org.modeshape.jcr.api.nodetype.NodeTypeManager;
040
041import com.google.common.eventbus.EventBus;
042
043public class FedoraWebhooksTest {
044
045    private FedoraWebhooks testObj;
046
047    private Session mockSession;
048
049    private Node mockRoot;
050
051    @Before
052    public void setUp() throws Exception {
053        testObj = new FedoraWebhooks();
054        mockSession = TestHelpers.mockSession(testObj);
055        mockRoot = mock(Node.class);
056        when(mockSession.getRootNode()).thenReturn(mockRoot);
057        final Repository mockRepo = mock(Repository.class);
058        when(mockSession.getRepository()).thenReturn(mockRepo);
059        when(mockRepo.login()).thenReturn(mockSession);
060        final Workspace mockWS = mock(Workspace.class);
061        final NodeTypeManager mockNT = mock(NodeTypeManager.class);
062        when(mockWS.getNodeTypeManager()).thenReturn(mockNT);
063        when(mockSession.getWorkspace()).thenReturn(mockWS);
064        TestHelpers.setField(testObj, "session", mockSession);
065        TestHelpers.setField(testObj, "readOnlySession", mockSession);
066        final SessionFactory mockSessions = mock(SessionFactory.class);
067        when(mockSessions.getInternalSession()).thenReturn(mockSession);
068        TestHelpers.setField(testObj, "sessions", mockSessions);
069    }
070
071    @Test
072    public void testInitialize() throws Exception {
073        final EventBus mockBus = mock(EventBus.class);
074        TestHelpers.setField(testObj, "eventBus", mockBus);
075        testObj.initialize();
076        verify(mockBus).register(testObj);
077    }
078
079    @Test
080    public void testShowWebhooks() throws Exception {
081
082        final NodeIterator mockNodes = mock(NodeIterator.class);
083
084        final Node mockHook = mock(Node.class);
085        final Property mockProp = mock(Property.class);
086        when(mockHook.getProperty(WEBHOOK_CALLBACK_PROPERTY))
087                .thenReturn(mockProp);
088
089        when(mockNodes.hasNext()).thenReturn(true, false);
090        when(mockNodes.nextNode()).thenReturn(mockHook).thenThrow(
091               new IndexOutOfBoundsException());
092        when(mockRoot.getNodes(WEBHOOK_PREFIX + "*")).thenReturn(
093                mockNodes);
094
095        testObj.showWebhooks();
096    }
097
098    @Test
099    public void testRegisterWebhook() throws Exception {
100        final String mockPath = "/webhook:foo";
101        final Node mockNode = mock(Node.class);
102        final NodeType mockType = mock(NodeType.class);
103        when(mockType.getName()).thenReturn(WEBHOOK_JCR_TYPE);
104        final NodeType[] mockTypes = new NodeType[] {mockType};
105        when(mockNode.getMixinNodeTypes()).thenReturn(mockTypes);
106        when(mockSession.getNode(mockPath)).thenReturn(mockNode);
107        when(mockRoot.getNode(mockPath.substring(1))).thenReturn(mockNode);
108        when(mockNode.getSession()).thenReturn(mockSession);
109        testObj.registerWebhook("foo");
110    }
111
112    @SuppressWarnings("unchecked")
113    @Test
114    public void testOnEvent() throws Exception {
115        final FedoraEvent mockEvent = mock(FedoraEvent.class);
116        final String mockPath = "/mock/path/to/node";
117        when(mockEvent.getPath()).thenReturn(mockPath);
118        final Node mockNode = mock(Node.class);
119        final NodeType mockType = mock(NodeType.class);
120        when(mockType.getName()).thenReturn(WEBHOOK_JCR_TYPE);
121        final NodeType[] mockTypes = new NodeType[] {mockType};
122        when(mockNode.getMixinNodeTypes()).thenReturn(mockTypes);
123        when(mockSession.getNode(mockPath)).thenReturn(mockNode);
124        when(mockRoot.getNode(mockPath.substring(1))).thenReturn(mockNode);
125        when(mockNode.getSession()).thenReturn(mockSession);
126        final NodeIterator mockNodes = mock(NodeIterator.class);
127        // let's say we have one webhook node
128        final Node mockHook = mock(Node.class);
129        final Property mockProp = mock(Property.class);
130        when(mockHook.getProperty(WEBHOOK_CALLBACK_PROPERTY))
131                .thenReturn(mockProp);
132        when(mockProp.getString()).thenReturn(
133                "http://fedora.gov/secrets/plans/takeovers/global");
134        when(mockNodes.hasNext()).thenReturn(true, false);
135        when(mockNodes.nextNode()).thenReturn(mockHook).thenThrow(
136                IndexOutOfBoundsException.class);
137        when(mockRoot.getNodes(WEBHOOK_PREFIX + "*")).thenReturn(
138                mockNodes);
139        testObj.onEvent(mockEvent);
140    }
141
142    @Test
143    public void testSessions() throws NoSuchFieldException {
144
145        final EventBus mockBus = mock(EventBus.class);
146        TestHelpers.setField(testObj, "eventBus", mockBus);
147
148        testObj.logoutSession();
149        verify(mockSession).logout();
150        verify(mockBus).unregister(testObj);
151    }
152
153}
