055/**
056 * The purpose of this class is to allow clients to manipulate the JCR
057 * namespaces in play in a repository. This is necessary to allow the use of
058 * traditional Fedora namespaced PIDs. Unlike Fedora Classic, a JCR requires
059 * that namespaces be registered before use. The catalog of namespaces is very
060 * simple, just a set of prefix-URI pairs.
061 *
062 * @author ajs6f
063 */
064@Component
065@Scope("prototype")
066@Path("/fcr:namespaces")
067public class FedoraRepositoryNamespaces extends AbstractResource {
068
069    @InjectedSession
070    protected Session session;
071
072    /**
073     * Register multiple object namespaces.
074     *
075     * @return 204
076     * @throws RepositoryException
077     */
078    @POST
079    @Timed
080    @Consumes({contentTypeSPARQLUpdate})
081    public Response updateNamespaces(final InputStream requestBodyStream)
082        throws RepositoryException, IOException {
083
084        try {
085            final Dataset dataset =
086                nodeService.getNamespaceRegistryDataset(session);
087            parseExecute(IOUtils.toString(requestBodyStream), dataset);
088            session.save();
089            return status(SC_NO_CONTENT).build();
090        } finally {
091            session.logout();
092        }
093    }
094
095    /**
096     * @return
097     * @throws RepositoryException
098     * @throws IOException
099     */
100    @GET
101    @Timed
102    @Produces({TURTLE, N3, N3_ALT1, N3_ALT2, RDF_XML, RDF_JSON, NTRIPLES,
103            TEXT_HTML})
104    @HtmlTemplate("jcr:namespaces")
105    public RdfStream getNamespaces() throws RepositoryException, IOException {
106        return nodeService.getNamespaceRegistryStream(session).session(session);
107    }
108}
