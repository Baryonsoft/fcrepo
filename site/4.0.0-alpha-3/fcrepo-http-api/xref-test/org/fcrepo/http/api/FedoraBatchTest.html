310         verify(mockSession, never()).save();

314         assertEquals("/FedoraDatastreamsTest1/testDS", multipart.getBodyParts()
315                 .get(0).getContentDisposition().getFileName());
316         assertEquals("asdf", IOUtils.toString(actualContent, "UTF-8"));

347     public static MultiPart getStringsAsMultipart(
348             final Map<String, String> contents) {
349         final MultiPart multipart = new MultiPart();
350         for (final Map.Entry<String, String> e : contents.entrySet()) {
351             final String id = e.getKey();
352             final String content = e.getValue();
353             final InputStream src = IOUtils.toInputStream(content);
354             final StreamDataBodyPart part = new StreamDataBodyPart(id, src);
355             try {
356                 final FormDataContentDisposition cd =
357                     new FormDataContentDisposition("form-data;name=" + id
358                             + ";filename=" + id + ".txt");
359                 part.contentDisposition(cd);
360             } catch (final ParseException ex) {
361                 ex.printStackTrace();
362             }
363             multipart.bodyPart(part);
364         }
365         return multipart;
366     }
