785    private void validateHTML(final String path) throws Exception {
786        final HttpGet getMethod = new HttpGet(serverAddress + path);
787        getMethod.addHeader("Accept", "text/html");
788        final HttpResponse response = client.execute(getMethod);
789        assertEquals(OK.getStatusCode(), response.getStatusLine()
790                .getStatusCode());
791        final String content = EntityUtils.toString(response.getEntity());
792        logger.debug("Retrieved HTML view:\n" + content);
793
794        final HtmlParser htmlParser = new HtmlParser(ALLOW);
795        htmlParser.setDoctypeExpectation(NO_DOCTYPE_ERRORS);
796        htmlParser.setErrorHandler(new HTMLErrorHandler());
797        htmlParser.setContentHandler(new TreeBuilder());
798        try (
799            final InputStream htmlStream =
800                new ByteArrayInputStream(content.getBytes())) {
801            htmlParser.parse(new InputSource(htmlStream));
802        }
803        logger.info("HTML found to be valid.");
804    }
