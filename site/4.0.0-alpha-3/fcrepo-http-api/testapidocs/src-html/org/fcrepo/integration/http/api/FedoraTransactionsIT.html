069    @Test
070    public void testCreateAndTimeoutTransaction() throws Exception {
071
072        /* create a short-lived tx */
073        final long testTimeout = min(500, REAP_INTERVAL / 2);
074        System.setProperty(TIMEOUT_SYSTEM_PROPERTY, Long.toString(testTimeout));
075
076        /* create a tx */
077        final HttpPost createTx = new HttpPost(serverAddress + "fcr:tx");
078        final HttpResponse response = execute(createTx);
079        assertEquals(201, response.getStatusLine().getStatusCode());
080
081        final String location = response.getFirstHeader("Location").getValue();
082
083        final HttpGet getWithinTx = new HttpGet(location);
084        HttpResponse resp = execute(getWithinTx);
085        IOUtils.toString(resp.getEntity().getContent());
086        assertEquals(200, resp.getStatusLine().getStatusCode());
087
088        int statusCode = 0;
089
090        sleep(REAP_INTERVAL * 2);
091        final HttpGet getWithExpiredTx = new HttpGet(location);
092        resp = execute(getWithExpiredTx);
093        IOUtils.toString(resp.getEntity().getContent());
094        statusCode = resp.getStatusLine().getStatusCode();
095
096        try {
097            assertEquals("Transaction did not expire", 410, statusCode);
098        } finally {
099            System.setProperty(TIMEOUT_SYSTEM_PROPERTY, Long
100                    .toString(DEFAULT_TIMEOUT));
101            System.clearProperty("fcrepo4.tx.timeout");
102        }
103    }

105    @Test
106    public void testCreateDoStuffAndRollbackTransaction() throws Exception {
107        /* create a tx */
108        final HttpPost createTx = new HttpPost(serverAddress + "fcr:tx");
109
110        final HttpResponse response = execute(createTx);
111        assertEquals(201, response.getStatusLine().getStatusCode());
112
113        final String txLocation =
114            response.getFirstHeader("Location").getValue();
115
116        /* create a new object inside the tx */
117        final HttpPost postNew =
118            new HttpPost(txLocation + "/object-in-tx-rollback");
119        HttpResponse resp = execute(postNew);
120        assertEquals(201, resp.getStatusLine().getStatusCode());
121
122        /* fetch the created tx from the endpoint */
123        final HttpGet getTx =
124            new HttpGet(txLocation + "/object-in-tx-rollback");
125        final GraphStore graphStore = getGraphStore(getTx);
126        logger.debug(graphStore.toString());
127
128        assertTrue(graphStore.toDataset().asDatasetGraph().contains(Node.ANY,
129                createURI(txLocation + "/object-in-tx-rollback"), ANY, ANY));
130
131        /* fetch the created tx from the endpoint */
132        final HttpGet getObj =
133            new HttpGet(serverAddress + "/object-in-tx-rollback");
134        resp = execute(getObj);
135        assertEquals(
136                "Expected to not find our object within the scope of the transaction",
137                404, resp.getStatusLine().getStatusCode());
138
139        /* and rollback */
140        final HttpPost rollbackTx =
141            new HttpPost(txLocation + "/fcr:tx/fcr:rollback");
142        resp = execute(rollbackTx);
143
144        assertEquals(204, resp.getStatusLine().getStatusCode());
145
146    }

148    @Test
149    public void testCreateDoStuffAndCommitTransaction() throws Exception {
150        /* create a tx */
151        final String objectInTxCommit = randomUUID().toString();
152
153        final HttpPost createTx = new HttpPost(serverAddress + "fcr:tx");
154
155        final HttpResponse response = execute(createTx);
156        assertEquals(201, response.getStatusLine().getStatusCode());
157
158        final String txLocation =
159            response.getFirstHeader("Location").getValue();
160
161        /* create a new object inside the tx */
162        final HttpPost postNew =
163            new HttpPost(txLocation + "/" + objectInTxCommit);
164        HttpResponse resp = execute(postNew);
165        assertEquals(201, resp.getStatusLine().getStatusCode());
166
167        /* fetch the created tx from the endpoint */
168        final HttpGet getTx = new HttpGet(txLocation + "/" + objectInTxCommit);
169        GraphStore graphStore = getGraphStore(getTx);
170
171        logger.debug(graphStore.toString());
172
173        assertTrue(graphStore.toDataset().asDatasetGraph().contains(ANY,
174                createURI(txLocation + "/" + objectInTxCommit), ANY, ANY));
175
176        /* fetch the object-in-tx outside of the tx */
177        final HttpGet getObj =
178                new HttpGet(serverAddress + objectInTxCommit);
179        resp = execute(getObj);
180        assertEquals(
181                "Expected to not find our object within the scope of the transaction",
182                404, resp.getStatusLine().getStatusCode());
183
184        /* and commit */
185        final HttpPost commitTx =
186            new HttpPost(txLocation + "/fcr:tx/fcr:commit");
187        resp = execute(commitTx);
188
189        assertEquals(204, resp.getStatusLine().getStatusCode());
190
191        /* fetch the object-in-tx outside of the tx after it has been committed */
192        final HttpGet getObjCommitted =
193            new HttpGet(serverAddress + objectInTxCommit);
194        graphStore = getGraphStore(getObjCommitted);
195
196        assertTrue("Expected to  find our object after the transaction was committed",
197                      graphStore.toDataset().asDatasetGraph()
198                          .contains(ANY, createURI(serverAddress + objectInTxCommit), ANY, ANY));
199
200    }
