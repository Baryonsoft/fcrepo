158    /**
159     * This method returns an HTTP response with content body appropriate to the following arguments.
160     *
161     * @param rangeValue starting and ending byte offsets, see {@link Range}
162     * @param limit is the number of child resources returned in the response, -1 for all
163     * @param rdfStream to which response RDF will be concatenated
164     * @return HTTP response
165     * @throws IOException
166     */
167    protected Response getContent(final String rangeValue,
168                                  final int limit,
169                                  final RdfStream rdfStream) throws IOException {
170        if (resource() instanceof FedoraBinary) {
171
172            final String contentTypeString = ((FedoraBinary) resource()).getMimeType();
173
174            final Lang lang = contentTypeToLang(contentTypeString);
175
176            if (!contentTypeString.equals("text/plain") && lang != null) {
177
178                final String format = lang.getName().toUpperCase();
179
180                final InputStream content = ((FedoraBinary) resource()).getContent();
181
182                final Model inputModel = createDefaultModel()
183                        .read(content,  (resource()).toString(), format);
184
185                rdfStream.concat(Iterators.transform(inputModel.listStatements(), Statement::asTriple));
186            } else {
187
188                final MediaType mediaType = MediaType.valueOf(contentTypeString);
189                if (MESSAGE_EXTERNAL_BODY.isCompatible(mediaType)
190                        && mediaType.getParameters().containsKey("access-type")
191                        && mediaType.getParameters().get("access-type").equals("URL")
192                        && mediaType.getParameters().containsKey("URL") ) {
193                    try {
194                        return temporaryRedirect(new URI(mediaType.getParameters().get("URL"))).build();
195                    } catch (final URISyntaxException e) {
196                        throw new RepositoryRuntimeException(e);
197                    }
198                }
199                return getBinaryContent(rangeValue);
200            }
201
202        } else {
203            rdfStream.concat(getResourceTriples(limit));
204            if (prefer != null) {
205                prefer.getReturn().addResponseHeaders(servletResponse);
206            }
207        }
208        servletResponse.addHeader("Vary", "Accept, Range, Accept-Encoding, Accept-Language");
209
210        return ok(rdfStream).build();
211    }

217    /**
218     * This method returns a stream of RDF triples associated with this target resource
219     *
220     * @param limit is the number of child resources returned in the response, -1 for all
221     * @return {@link RdfStream}
222     */
223    protected RdfStream getResourceTriples(final int limit) {
224        // use the thing described, not the description, for the subject of descriptive triples
225        if (resource() instanceof NonRdfSourceDescription) {
226            resource = ((NonRdfSourceDescription) resource()).getDescribedResource();
227        }
228        final PreferTag returnPreference;
229
230        if (prefer != null && prefer.hasReturn()) {
231            returnPreference = prefer.getReturn();
232        } else if (prefer != null && prefer.hasHandling()) {
233            returnPreference = prefer.getHandling();
234        } else {
235            returnPreference = PreferTag.emptyTag();
236        }
237
238        final LdpPreferTag ldpPreferences = new LdpPreferTag(returnPreference);
239
240        final RdfStream rdfStream = new RdfStream();
241
242        final Predicate<Triple> tripleFilter = ldpPreferences.prefersServerManaged() ? x -> true :
243            IS_MANAGED_TRIPLE.negate();
244
245        if (ldpPreferences.prefersServerManaged()) {
246            rdfStream.concat(getTriples(LdpRdfContext.class));
247        }
248
249        rdfStream.concat(filter(getTriples(TypeRdfContext.class), tripleFilter::test));
250
251        rdfStream.concat(filter(getTriples(PropertiesRdfContext.class), tripleFilter::test));
252
253        if (!returnPreference.getValue().equals("minimal")) {
254
255            // Additional server-managed triples about this resource
256            if (ldpPreferences.prefersServerManaged()) {
257                rdfStream.concat(getTriples(AclRdfContext.class));
258                rdfStream.concat(getTriples(RootRdfContext.class));
259                rdfStream.concat(getTriples(ContentRdfContext.class));
260                rdfStream.concat(getTriples(ParentRdfContext.class));
261            }
262
263            // containment triples about this resource
264            if (ldpPreferences.prefersContainment()) {
265                rdfStream.concat(getTriples(ChildrenRdfContext.class).limit(limit));
266            }
267
268            // LDP container membership triples for this resource
269            if (ldpPreferences.prefersMembership()) {
270                rdfStream.concat(getTriples(LdpContainerRdfContext.class));
271                rdfStream.concat(getTriples(LdpIsMemberOfRdfContext.class));
272            }
273
274            // Embed all hash and blank nodes
275            // using IS_MANAGED_TRIPLE directly to avoid Prefer header logic (we never want them for hash fragments)
276            rdfStream.concat(filter(getTriples(HashRdfContext.class), IS_MANAGED_TRIPLE.negate()::test));
277            rdfStream.concat(filter(getTriples(SkolemNodeRdfContext.class), tripleFilter::test));
278
279            // Include inbound references to this object
280            if (ldpPreferences.prefersReferences()) {
281                rdfStream.concat(getTriples(ReferencesRdfContext.class));
282            }
283
284            // Embed the children of this object
285            if (ldpPreferences.prefersEmbed()) {
286
287                final Iterator<FedoraResource> children = resource().getChildren();
288
289                rdfStream.concat(filter(concat(transform(children, child ->
290                child.getTriples(translator(),
291                        ImmutableList.of(
292                                TypeRdfContext.class, PropertiesRdfContext.class, SkolemNodeRdfContext.class)))),
293                        tripleFilter::test));
294
295            }
296        }
297
298        if (httpTripleUtil != null && ldpPreferences.prefersServerManaged()) {
299            httpTripleUtil.addHttpComponentModelsForResourceToStream(rdfStream, resource(), uriInfo, translator());
300        }
301
302
303        return rdfStream;
304    }

306    /**
307     * Get the binary content of a datastream
308     *
309     * @param rangeValue the range value
310     * @return Binary blob
311     * @throws IOException if io exception occurred
312     */
313    protected Response getBinaryContent(final String rangeValue)
314            throws IOException {
315            final FedoraBinary binary = (FedoraBinary)resource();
316
317            // we include an explicit etag, because the default behavior is to use the JCR node's etag, not
318            // the jcr:content node digest. The etag is only included if we are not within a transaction.
319            final String txId = TransactionServiceImpl.getCurrentTransactionId(session());
320            if (txId == null) {
321                checkCacheControlHeaders(request, servletResponse, binary, session());
322            }
323            final CacheControl cc = new CacheControl();
324            cc.setMaxAge(0);
325            cc.setMustRevalidate(true);
326            Response.ResponseBuilder builder;
327
328            if (rangeValue != null && rangeValue.startsWith("bytes")) {
329
330                final Range range = Range.convert(rangeValue);
331
332                final long contentSize = binary.getContentSize();
333
334                final String endAsString;
335
336                if (range.end() == -1) {
337                    endAsString = Long.toString(contentSize - 1);
338                } else {
339                    endAsString = Long.toString(range.end());
340                }
341
342                final String contentRangeValue =
343                        String.format("bytes %s-%s/%s", range.start(),
344                                endAsString, contentSize);
345
346                if (range.end() > contentSize ||
347                        (range.end() == -1 && range.start() > contentSize)) {
348
349                    builder = status(REQUESTED_RANGE_NOT_SATISFIABLE)
350                            .header("Content-Range", contentRangeValue);
351                } else {
352                    final long rangeStart = range.start();
353                    final long rangeSize = range.size() == -1 ? contentSize - rangeStart : range.size();
354                    final long remainingBytes = contentSize - rangeStart;
355                    final long bufSize = rangeSize < remainingBytes ? rangeSize : remainingBytes;
356
357                    if (bufSize < MAX_BUFFER_SIZE) {
358                        // Small size range content retrieval use javax.jcr.Binary to improve performance
359                        final byte[] buf = new byte[(int) bufSize];
360
361                        final Binary binaryContent = binary.getBinaryContent();
362                        try {
363                            binaryContent.read(buf, rangeStart);
364                        } catch (final RepositoryException e1) {
365                            throw new RepositoryRuntimeException(e1);
366                        }
367                        binaryContent.dispose();
368
369                        builder = status(PARTIAL_CONTENT).entity(buf)
370                                .header("Content-Range", contentRangeValue);
371                    } else {
372                        // For large range content retrieval, go with the InputStream class to balance
373                        // the memory usage, though this is a rare case in range content retrieval.
374                        final InputStream content = binary.getContent();
375                        final RangeRequestInputStream rangeInputStream =
376                                new RangeRequestInputStream(content, range.start(), range.size());
377
378                        builder = status(PARTIAL_CONTENT).entity(rangeInputStream)
379                                .header("Content-Range", contentRangeValue);
380                    }
381                }
382
383            } else {
384                final InputStream content = binary.getContent();
385                builder = ok(content);
386            }
387
388
389            // we set the content-type explicitly to avoid content-negotiation from getting in the way
390            return builder.type(binary.getMimeType())
391                    .cacheControl(cc)
392                    .build();
393
394        }

421    /**
422     * Add any resource-specific headers to the response
423     * @param resource the resource
424     */
425    protected void addResourceHttpHeaders(final FedoraResource resource) {
426        if (resource instanceof FedoraBinary) {
427
428            final FedoraBinary binary = (FedoraBinary)resource;
429            final ContentDisposition contentDisposition = ContentDisposition.type("attachment")
430                    .fileName(binary.getFilename())
431                    .creationDate(binary.getCreatedDate())
432                    .modificationDate(binary.getLastModifiedDate())
433                    .size(binary.getContentSize())
434                    .build();
435
436            servletResponse.addHeader("Content-Type", binary.getMimeType());
437            servletResponse.addHeader("Content-Length", String.valueOf(binary.getContentSize()));
438            servletResponse.addHeader("Accept-Ranges", "bytes");
439            servletResponse.addHeader("Content-Disposition", contentDisposition.toString());
440        }
441
442        servletResponse.addHeader("Link", "<" + LDP_NAMESPACE + "Resource>;rel=\"type\"");
443
444        if (resource instanceof NonRdfSource) {
445            servletResponse.addHeader("Link", "<" + LDP_NAMESPACE + "NonRDFSource>;rel=\"type\"");
446        } else if (resource instanceof Container) {
447            servletResponse.addHeader("Link", "<" + CONTAINER.getURI() + ">;rel=\"type\"");
448            if (resource.hasType(LDP_BASIC_CONTAINER)) {
449                servletResponse.addHeader("Link", "<" + BASIC_CONTAINER.getURI() + ">;rel=\"type\"");
450            } else if (resource.hasType(LDP_DIRECT_CONTAINER)) {
451                servletResponse.addHeader("Link", "<" + DIRECT_CONTAINER.getURI() + ">;rel=\"type\"");
452            } else if (resource.hasType(LDP_INDIRECT_CONTAINER)) {
453                servletResponse.addHeader("Link", "<" + INDIRECT_CONTAINER.getURI() + ">;rel=\"type\"");
454            } else {
455                servletResponse.addHeader("Link", "<" + BASIC_CONTAINER.getURI() + ">;rel=\"type\"");
456            }
457        } else {
458            servletResponse.addHeader("Link", "<" + LDP_NAMESPACE + "RDFSource>;rel=\"type\"");
459        }
460        if (httpHeaderInject != null) {
461            httpHeaderInject.addHttpHeaderToResponseStream(servletResponse, uriInfo, resource());
462        }
463
464    }

483    /**
484     * Add ETag and Last-Modified cache control headers to the response
485     * @param servletResponse the servlet response
486     * @param resource the fedora resource
487     * @param session the session
488     */
489    protected static void addCacheControlHeaders(final HttpServletResponse servletResponse,
490                                                 final FedoraResource resource,
491                                                 final Session session) {
492
493        final String txId = TransactionServiceImpl.getCurrentTransactionId(session);
494        if (txId != null) {
495            // Do not add caching headers if in a transaction
496            return;
497        }
498
499        final FedoraResource mutableResource = resource instanceof NonRdfSourceDescription
500                ? ((NonRdfSourceDescription) resource).getDescribedResource() : resource;
501        final EntityTag etag = new EntityTag(mutableResource.getEtagValue());
502        final Date date = mutableResource.getLastModifiedDate();
503
504        if (!etag.getValue().isEmpty()) {
505            servletResponse.addHeader("ETag", etag.toString());
506        }
507
508        if (date != null) {
509            servletResponse.addDateHeader("Last-Modified", date.getTime());
510        }
511    }

527    private static void evaluateRequestPreconditions(final Request request,
528                                                     final HttpServletResponse servletResponse,
529                                                     final FedoraResource resource,
530                                                     final Session session,
531                                                     final boolean cacheControl) {
532
533        final String txId = TransactionServiceImpl.getCurrentTransactionId(session);
534        if (txId != null) {
535            // Force cache revalidation if in a transaction
536            servletResponse.addHeader(CACHE_CONTROL, "must-revalidate");
537            servletResponse.addHeader(CACHE_CONTROL, "max-age=0");
538            return;
539        }
540
541        final EntityTag etag = new EntityTag(resource.getEtagValue());
542        final Date date = resource.getLastModifiedDate();
543        final Date roundedDate = new Date();
544
545        if (date != null) {
546            roundedDate.setTime(date.getTime() - date.getTime() % 1000);
547        }
548
549        Response.ResponseBuilder builder = request.evaluatePreconditions(etag);
550        if ( builder != null ) {
551            builder = builder.entity("ETag mismatch");
552        } else {
553            builder = request.evaluatePreconditions(roundedDate);
554            if ( builder != null ) {
555                builder = builder.entity("Date mismatch");
556            }
557        }
558
559        if (builder != null && cacheControl ) {
560            final CacheControl cc = new CacheControl();
561            cc.setMaxAge(0);
562            cc.setMustRevalidate(true);
563            // here we are implicitly emitting a 304
564            // the exception is not an error, it's genuinely
565            // an exceptional condition
566            builder = builder.cacheControl(cc).lastModified(date).tag(etag);
567        }
568        if (builder != null) {
569            throw new WebApplicationException(builder.build());
570        }
571    }

598    protected void replaceResourceWithStream(final FedoraResource resource,
599                                             final InputStream requestBodyStream,
600                                             final MediaType contentType,
601                                             final RdfStream resourceTriples) throws MalformedRdfException {
602        final Lang format = contentTypeToLang(contentType.toString());
603
604        final Model inputModel = createDefaultModel();
605        try {
606            inputModel.read(requestBodyStream, getUri(resource).toString(), format.getName().toUpperCase());
607
608        } catch (final RiotException e) {
609            throw new BadRequestException("RDF was not parsable: " + e.getMessage(), e);
610
611        } catch (final RuntimeIOException e) {
612            if (e.getCause() instanceof JsonParseException) {
613                throw new MalformedRdfException(e.getCause());
614            }
615            throw new RepositoryRuntimeException(e);
616        }
617
618        resource.replaceProperties(translator(), inputModel, resourceTriples);
619    }

621    protected void patchResourcewithSparql(final FedoraResource resource,
622            final String requestBody,
623            final RdfStream resourceTriples)
624                    throws MalformedRdfException, AccessDeniedException {
625        if (resource instanceof NonRdfSourceDescription) {
626            // update the described resource instead
627            ((NonRdfSourceDescription) resource).getDescribedResource()
628                    .updateProperties(translator(), requestBody, resourceTriples);
629        } else {
630            resource.updateProperties(translator(), requestBody, resourceTriples);
631        }
632    }
633
634    /**
635     * Create a checksum URI object.
636     **/
637    private static URI checksumURI( final String checksum ) {
638        if (!isBlank(checksum)) {
639            return URI.create(checksum);
640        }
641        return null;
642    }
