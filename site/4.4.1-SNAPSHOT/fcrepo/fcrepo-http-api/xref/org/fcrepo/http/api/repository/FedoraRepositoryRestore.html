16  package org.fcrepo.http.api.repository;
17  
18  import static java.util.stream.Collectors.joining;
19  import static javax.ws.rs.core.Response.noContent;
20  import static javax.ws.rs.core.Response.serverError;
21  import static org.slf4j.LoggerFactory.getLogger;
22  
23  import java.io.File;
24  import java.io.IOException;
25  import java.io.InputStream;
26  import java.util.Collection;
27  
28  import javax.inject.Inject;
29  import javax.jcr.Session;
30  import javax.ws.rs.POST;
31  import javax.ws.rs.Path;
32  import javax.ws.rs.WebApplicationException;
33  import javax.ws.rs.core.Response;
34  
35  import org.apache.commons.io.IOUtils;
36  import org.fcrepo.http.commons.AbstractResource;
37  import org.fcrepo.kernel.api.services.RepositoryService;
38  import org.slf4j.Logger;
39  import org.springframework.context.annotation.Scope;
40  
41  /**
42   * Restore a backup of the repository
43   *
44   * @author cbeer
45   */
46  @Scope("prototype")
47  @Path("/fcr:restore")
48  public class FedoraRepositoryRestore extends AbstractResource {
49  
50      private static final Logger LOGGER = getLogger(FedoraRepositoryRestore.class);
51  
52      @Inject
53      protected Session session;
54  
55      /**
56       * The fcrepo repository service
57       */
58      @Inject
59      protected RepositoryService repositoryService;
60  
61      /**
62       * This method runs a repository restore.
63       *
64       * @param bodyStream the body stream
65       * @return response
66       * @throws IOException if IO exception occurred
67       */
68      @POST
69      public Response runRestore(final InputStream bodyStream) throws IOException {
70  
71          if (null == bodyStream) {
72              throw new WebApplicationException(serverError().entity(
73                      "Request body must not be null").build());
74          }
75  
76          final String body = IOUtils.toString(bodyStream);
77          final File backupDirectory = new File(body.trim());
78          if (!backupDirectory.exists()) {
79              throw new WebApplicationException(serverError().entity(
80                      "Backup directory does not exist: "
81                              + backupDirectory.getAbsolutePath()).build());
82          }
83  
84          final Collection<Throwable> problems = repositoryService.restoreRepository(session, backupDirectory);
85          if (!problems.isEmpty()) {
86              LOGGER.error("Problems restoring up the repository:");
87  
88              // Report the problems (we'll just print them out) ...
89              final String problemsOutput = problems.stream().map(Throwable::getMessage).peek(LOGGER::error)
90                      .collect(joining("\n"));
91  
92              throw new WebApplicationException(serverError()
93                      .entity(problemsOutput).build());
94  
95          }
96          return noContent().build();
97  
98      }
99  }
