089    /**
090     * Initialize
091     *
092     * @throws RepositoryException
093     */
094    @PostConstruct
095    public void setUpRepositoryConfiguration() throws RepositoryException {
096        Session internalSession = null;
097        try {
098            internalSession = sessions.getInternalSession();
099            new FedoraResourceImpl(internalSession, FEDORA_STORAGE_POLICY_PATH, NT_FOLDER);
100            internalSession.save();

109    /**
110     * POST to store nodeType and hint
111     *
112     * @param request For now, follows pattern: mix:mimeType image/tiff
113     *        store-hint
114     * @return status
115     */
116
117    @POST
118    @Consumes(APPLICATION_FORM_URLENCODED)
119    @Timed
120    public Response post(final @PathParam("path") String path,
121                         final String request) throws RepositoryException {
122        LOGGER.debug("POST Received request param: {}", request);
123        final Response.ResponseBuilder response;
124
125        if (!path.equalsIgnoreCase(POLICY_RESOURCE)) {
126            return methodNotAllowed().entity(
127                    "POST method not allowed on " + getUriInfo().getAbsolutePath() +
128                            ", try /policies/fcr:storagepolicy")
129                    .build();
130        }
131
132        final String[] str = StringUtils.split(request); // simple split for now
133        validateArgs(str.length);
134        try {
135            final Node node =
136                getJcrTools().findOrCreateNode(session,
137                                             FEDORA_STORAGE_POLICY_PATH, "test");
138            if (isValidNodeTypeProperty(session, str[0]) ||
139                isValidConfigurationProperty(str[0])) {
140                node.setProperty(str[0], new String[] {str[1] + ":" + str[2]});
141
142                // TODO (for now) instantiate PolicyType based on mix:mimeType
143                final StoragePolicy policy = newPolicyInstance(str[0], str[1], str[2]);
144                // TODO (for now) based on object comparison using equals()
145                if (storagePolicyDecisionPoint.contains(policy)) {
146                    throw new StoragePolicyTypeException("Property already exists!");
147                }
148                storagePolicyDecisionPoint.addPolicy(policy);
149                session.save();
150                LOGGER.debug("Saved PDS hint {}", request);
151
152                response = Response.created(getUriInfo().getBaseUriBuilder()
153                                                .path(FedoraStoragePolicy.class)
154                                                .buildFromMap(singletonMap("path", str[0])));
155            } else {
156                throw new StoragePolicyTypeException(
157                        "Invalid property type specified: " + str[0]);
158            }
159        } finally {
160            session.logout();
161        }
162
163        return response.build();
164    }

166    /**
167     * For nodeType n or runtime property p get org.fcrepo.binary.StoragePolicy
168     * implementation. Note: Signature might need to change, or a more
169     * sophisticated method used, as implementation evolves.
170     *
171     * @param propertyType
172     * @param itemType
173     * @param value
174     * @return
175     * @throws StoragePolicyTypeException
176     */
177    protected StoragePolicy newPolicyInstance(final String propertyType,
178        final String itemType, final String value) {
179
180        switch (propertyType) {
181            case NodeType.MIX_MIMETYPE:
182            case "mix:mimeType":
183                return new MimeTypeStoragePolicy(itemType, value);
184            default:
185                throw new StoragePolicyTypeException("Mapping not found");
186        }
187    }

189    /**
190     * Delete NodeType. TODO for deleting multiple values with in a NodeType,
191     * the design of how things are stored will need to change.
192     */
193    @DELETE
194    @Timed
195    public Response deleteNodeType(@PathParam("path") final String nodeType)
196        throws RepositoryException {
197        try {
198            LOGGER.debug("Deleting node property{}", nodeType);
199            final Node node =
200                getJcrTools().findOrCreateNode(session,
201                                             FEDORA_STORAGE_POLICY_PATH, "test");
202            if (isValidNodeTypeProperty(session, nodeType)) {
203                node.getProperty(nodeType).remove();
204                session.save();
205
206                // remove all MimeType intances (since thats only the stored
207                // StoragePolicy for now.
208                // TODO Once StoragePolicy is updated to display StoragePolicy type, this
209                // would change
210                storagePolicyDecisionPoint.removeAll();
211                return Response.noContent().build();
212            }
213            throw new RepositoryException(
214                "Invalid property type specified.");
215        } finally {
216            session.logout();
217        }
218    }

233        return getStoragePolicy(path);

235
236    private Response getAllStoragePolicies() {
237        if (storagePolicyDecisionPoint == null ||
238            storagePolicyDecisionPoint.isEmpty()) {
239            return Response.ok("No Policies Found").build();
240        }
241        return Response.ok(storagePolicyDecisionPoint.toString()).build();
242    }
243
244    private Response getStoragePolicy(final String nodeType) throws RepositoryException {
245        LOGGER.debug("Get storage policy for: {}", nodeType);
246        Response.ResponseBuilder response;
247        try {
248            final Node node =
249                getJcrTools().findOrCreateNode(session,
250                                                 FEDORA_STORAGE_POLICY_PATH,
251                                              "test");
252
253            final Property prop = node.getProperty(nodeType);
254            if (null == prop) {
255                throw new PathNotFoundException("StoragePolicy not found: " + nodeType);
256            }
257
258            final Value[] values = prop.getValues();
259            if (values != null && values.length > 0) {
260                response = Response.ok(values[0].getString());
261            } else {
262                throw new PathNotFoundException("StoragePolicy not found: " + nodeType);
263            }
264
265        } finally {
266            session.logout();
267        }
268        return response.build();
269    }

271    /**
272     * Verifies whether node type is valid
273     *
274     * @param session
275     * @param type
276     * @return
277     * @throws RepositoryException
278     */
279    private boolean isValidNodeTypeProperty(final Session session,
280        final String type) throws RepositoryException {
281        try {
282            return session.getWorkspace().getNodeTypeManager()
283                .getNodeType(type).getName().equals(type);
284        } catch (final NoSuchNodeTypeException e) {
285            LOGGER.debug("No corresponding Node type found for: {}", type, e);
286            return false;
287        }
288    }
