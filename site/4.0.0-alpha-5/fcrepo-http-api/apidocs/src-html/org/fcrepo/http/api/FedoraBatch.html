088/**
089 * Controller for manipulating binary streams in larger batches
090 * by using multipart requests and responses
091 */
092@Component
093@Scope("prototype")
094@Path("/{path: .*}/fcr:batch")
095public class FedoraBatch extends AbstractResource {
096
097    public static final String ATTACHMENT = "attachment";
098    public static final String INLINE = "inline";
099    public static final String DELETE = "delete";
100    public static final String FORM_DATA_DELETE_PART_NAME = "delete[]";
101    @InjectedSession
102    protected Session session;
103
104    private static final Logger LOGGER = getLogger(FedoraBatch.class);
105
106    /**
107     * Apply batch modifications relative to the node.
108     *
109     * This endpoint supports two types of multipart requests:
110     *  - mixed (preferred)
111     *  - form-data (fallback for "dumb" clients)
112     *
113     *  The name-part of the multipart request is relative to the node
114     *  this operation was called on.
115     *
116     *  multipart/mixed:
117     *
118     *  mixed mode supports three content-disposition types:
119     *  - inline
120     *      Add or update an objects triples
121     *  - attachment
122     *      Add or update binary content
123     *  - delete
124     *      Delete an object
125     *
126     *  multipart/form-data:
127     *
128     *  form-data is a fallback for dumb clients (e.g. HttpClient, curl, etc).
129     *  Instead of using the Content-Disposition to determine what operation
130     *  to perform, form-data uses heuristics to figure out what to do.
131     *
132     *  - if the entity has a filename, always treat it as binary content
133     *  - if the content is RDF or SPARQL-Update, add or update triples
134     *  - if the entity has the name "delete[]", the body is a single path
135     *  to delete.
136     *  - otherwise, treat the entity as binary content.
137     *
138     *
139     * @param pathList
140     * @param multipart
141     * @return
142     * @throws RepositoryException
143     * @throws IOException
144     * @throws InvalidChecksumException
145     */
146    @POST
147    @Timed
148    public Response batchModify(@PathParam("path") final List<PathSegment> pathList,
149                                final MultiPart multipart)
150        throws RepositoryException, InvalidChecksumException, IOException, URISyntaxException {
151
152        final String path = toPath(pathList);
153
154        // TODO: this is ugly, but it works.
155        final PathFactory pathFactory = new ExecutionContext().getValueFactories().getPathFactory();
156        final org.modeshape.jcr.value.Path jcrPath = pathFactory.create(path);
157
158        try {
159
160            final Set<FedoraResource> resourcesChanged = new HashSet<>();
161
162            // iterate through the multipart entities
163            for (final BodyPart part : multipart.getBodyParts()) {
164                final ContentDisposition contentDisposition = part.getContentDisposition();
165
166
167                // a relative path (probably.)
168
169                final String contentDispositionType = contentDisposition.getType();
170
171                final String partName = contentDisposition.getParameters().get("name");
172
173                final String contentTypeString = part.getMediaType().toString();
174
175                LOGGER.trace("Processing {} part {} with media type {}",
176                                contentDispositionType, partName, contentTypeString);
177
178                final String realContentDisposition;
179
180                // we need to apply some heuristics for "dumb" clients that
181                // can only send form-data content
182                if (contentDispositionType.equals("form-data")) {
183
184                    if (contentDisposition.getFileName() != null) {
185                        realContentDisposition = ATTACHMENT;
186                    } else if (contentTypeString.equals(contentTypeSPARQLUpdate)
187                        || contentTypeToLang(contentTypeString) != null) {
188                        realContentDisposition = INLINE;
189                    } else if (partName.equals(FORM_DATA_DELETE_PART_NAME)) {
190                        realContentDisposition = DELETE;
191                    } else {
192                        realContentDisposition = ATTACHMENT;
193                    }
194
195                    LOGGER.trace("Converted form-data to content disposition {}", realContentDisposition);
196                } else {
197                    realContentDisposition = contentDispositionType;
198                }
199
200                // convert the entity to an InputStream
201                final Object entityBody = part.getEntity();
202
203                final InputStream src;
204                if (entityBody instanceof BodyPartEntity) {
205                    final BodyPartEntity entity =
206                        (BodyPartEntity) part.getEntity();
207                    src = entity.getInputStream();
208                } else if (entityBody instanceof InputStream) {
209                    src = (InputStream) entityBody;
210                } else {
211                    LOGGER.debug("Got unknown multipart entity for {}; ignoring it", partName);
212                    src = IOUtils.toInputStream("");
213                }
214
215                // convert the entity name to a node path
216                final String pathName;
217
218                if (partName.equals(FORM_DATA_DELETE_PART_NAME)) {
219                    pathName = IOUtils.toString(src);
220                } else {
221                    pathName = partName;
222                }
223
224                final String objPath = pathFactory.create(jcrPath, pathName).getCanonicalPath().getString();
225
226                switch (realContentDisposition) {
227                    case INLINE:
228
229                        final HttpIdentifierTranslator subjects =
230                            new HttpIdentifierTranslator(session, FedoraNodes.class, uriInfo);
231
232                        final FedoraResource resource;
233
234                        if (nodeService.exists(session, objPath)) {
235                            resource = nodeService.findOrCreateObject(session, objPath);
236                        } else {
237                            resource = objectService.createObject(session, objPath);
238                        }
239
240                        if (contentTypeString.equals(contentTypeSPARQLUpdate)) {
241                            resource.updatePropertiesDataset(subjects, IOUtils.toString(src));
242                        } else if (contentTypeToLang(contentTypeString) != null) {
243                            final Lang lang = contentTypeToLang(contentTypeString);
244
245                            final String format = lang.getName().toUpperCase();
246
247                            final Model inputModel =
248                                createDefaultModel().read(src,
249                                        subjects.getSubject(resource.getNode().getPath()).toString(), format);
250
251                            resource.replaceProperties(subjects, inputModel);
252                        } else {
253                            throw new WebApplicationException(notAcceptable()
254                                .entity("Invalid Content Type " + contentTypeString).build());
255                        }
256
257                        resourcesChanged.add(resource);
258
259                        break;
260
261                    case ATTACHMENT:
262
263                        final URI checksumURI;
264
265                        final String checksum = contentDisposition.getParameters().get("checksum");
266
267                        if (checksum != null && !checksum.equals("")) {
268                            checksumURI = new URI(checksum);
269                        } else {
270                            checksumURI = null;
271                        }
272
273                        resourcesChanged.add(datastreamService.createDatastream(session, objPath,
274                                                                  part.getMediaType().toString(),
275                                                                  contentDisposition.getFileName(),
276                                                                  src, checksumURI));
277                        break;
278
279                    case DELETE:
280                        nodeService.deleteObject(session, objPath);
281                        break;
282
283                    default:
284                        return status(Status.BAD_REQUEST)
285                                   .entity("Unknown Content-Disposition: " + realContentDisposition).build();
286                }
287            }
288
289            session.save();
290            versionService.nodeUpdated(session, path);
291            for (final FedoraResource resource : resourcesChanged) {
292                versionService.nodeUpdated(resource.getNode());
293            }
294
295            final HttpIdentifierTranslator subjects =
296                    new HttpIdentifierTranslator(session, FedoraNodes.class, uriInfo);
297
298            return created(new URI(subjects.getSubject(path).getURI())).build();
299
300        } finally {
301            session.logout();
302        }
303    }

305    /**
306     * Delete multiple child objects given by the child query parameter
307     *
308     * @param pathList
309     * @param childList
310     * @return
311     * @throws RepositoryException
312     */
313    @DELETE
314    @Timed
315    public Response batchDelete(@PathParam("path") final List<PathSegment> pathList,
316                                @QueryParam("child") final List<String> childList) throws RepositoryException {
317        try {
318            final String path = toPath(pathList);
319            for (final String dsid : childList) {
320                final String dsPath = path + "/" + dsid;
321                LOGGER.debug("purging node {}", dsPath);
322                nodeService.deleteObject(session, dsPath);
323            }
324            session.save();
325            versionService.nodeUpdated(session, path);
326            return noContent().build();
327        } finally {
328            session.logout();
329        }
330    }

332    /**
333     * Retrieve multiple datastream bitstreams in a single request as a
334     * multipart/mixed response.
335     *
336     * @param pathList
337     * @param requestedChildren
338     * @param request
339     * @return
340     * @throws RepositoryException
341     * @throws NoSuchAlgorithmException
342     */
343    @GET
344    @Produces("multipart/mixed")
345    @Timed
346    public Response getBinaryContents(
347        @PathParam("path") final List<PathSegment> pathList,
348        @QueryParam("child") final List<String> requestedChildren,
349        @Context final Request request) throws RepositoryException, NoSuchAlgorithmException {
350
351        final List<Datastream> datastreams = new ArrayList<>();
352
353        try {
354            final String path = toPath(pathList);
355            // TODO: wrap some of this JCR logic in an fcrepo abstraction;
356
357            final Node node = nodeService.getObject(session, path).getNode();
358
359            Date date = new Date();
360
361            final MessageDigest digest = MessageDigest.getInstance("SHA-1");
362
363            final NodeIterator ni;
364
365            if (requestedChildren.isEmpty()) {
366                ni = node.getNodes();
367            } else {
368                ni =
369                        node.getNodes(requestedChildren
370                                .toArray(new String[requestedChildren.size()]));
371            }
372
373            // complain if no children found
374            if ( ni.getSize() == 0 ) {
375                return status(Status.BAD_REQUEST).build();
376            }
377
378            // transform the nodes into datastreams, and calculate cache header
379            // data
380            while (ni.hasNext()) {
381
382                final Node dsNode = ni.nextNode();
383                final Datastream ds = datastreamService.asDatastream(dsNode);
384
385                if (!ds.hasContent()) {
386                    continue;
387                }
388
389                digest.update(ds.getContentDigest().toString().getBytes(
390                        UTF_8));
391
392                if (ds.getLastModifiedDate().after(date)) {
393                    date = ds.getLastModifiedDate();
394                }
395
396                datastreams.add(ds);
397            }
398
399            final URI digestURI =
400                    ContentDigest.asURI(digest.getAlgorithm(), digest.digest());
401            final EntityTag etag = new EntityTag(digestURI.toString());
402
403            final Date roundedDate = new Date();
404            roundedDate.setTime(date.getTime() - date.getTime() % 1000);
405
406            ResponseBuilder builder =
407                    request.evaluatePreconditions(roundedDate, etag);
408
409            final CacheControl cc = new CacheControl();
410            cc.setMaxAge(0);
411            cc.setMustRevalidate(true);
412
413            if (builder == null) {
414                final MultiPart multipart = new MultiPart();
415
416                for (final Datastream ds : datastreams) {
417                    final BodyPart bodyPart =
418                            new BodyPart(ds.getContent(), MediaType.valueOf(ds
419                                    .getMimeType()));
420                    bodyPart.setContentDisposition(ContentDisposition.type(
421                            ATTACHMENT).fileName(ds.getPath()).creationDate(
422                            ds.getCreatedDate()).modificationDate(
423                            ds.getLastModifiedDate()).size(ds.getContentSize())
424                            .build());
425                    multipart.bodyPart(bodyPart);
426                }
427
428                builder = ok(multipart, MULTIPART_FORM_DATA);
429            }
430
431            return builder.cacheControl(cc).lastModified(date).tag(etag)
432                    .build();
433
434        } finally {
435            session.logout();
436        }
437    }
