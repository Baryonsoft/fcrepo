086    /**
087     * Test whether a created lock can be viewed by both
088     * the creator (supplying the token) and by a request
089     * without the token and that the token is only visible
090     * when it was supplied to the session.
091     */
092    @Test
093    public void testLockMetadataAvailability() throws IOException {
094        final String pid = getRandomUniquePid();
095        createObject(pid);
096
097        final String lockToken = getLockToken(lockObject(pid));
098
099        final Node lockURI = createURI(serverAddress + pid + "/" + FCR_LOCK);
100        final Node nodeURI = createURI(serverAddress + pid);
101
102        GraphStore lockTriples = getLockProperties(pid, lockToken);
103
104        Assert.assertTrue("Lock relationship must be present!",
105                lockTriples.contains(ANY, lockURI, createURI(LOCKS.getURI()), nodeURI));
106
107        Assert.assertTrue("Lock token must be present!",
108                lockTriples.contains(ANY, lockURI, createURI(HAS_LOCK_TOKEN.getURI()), createLiteral(lockToken)));
109
110        lockTriples = getLockProperties(pid, null);
111
112        Assert.assertTrue("Lock relationship must be present!",
113                lockTriples.contains(ANY, lockURI, createURI(LOCKS.getURI()), nodeURI));
114
115        Assert.assertFalse("Lock token must be hidden!",
116                lockTriples.contains(ANY, lockURI, createURI(HAS_LOCK_TOKEN.getURI()), createLiteral(lockToken)));
117
118        assertUnlockWithToken(pid, lockToken);
119    }

121    /**
122     * A basic test to ensure that deep locking prevents updates to child nodes
123     * while a shallow lock does not.
124     */
125    @Test
126    public void testDeepVsShallowLocks() throws IOException {
127        final String pid = getRandomUniquePid();
128        createObject(pid);
129        final String childPid = pid + "/" + getRandomUniquePid();
130        createObject(childPid);
131
132        // Test shallow lock
133        final String shallowLockToken = getLockToken(lockObject(pid));
134
135        assertCannotSetPropertyWithoutLockToken(pid);
136        assertCanSetProperty(pid, shallowLockToken);
137        assertCanSetProperty("Properties of a child of a shallow locked node"
138                + " should be updatable without the lock token!", childPid, null);
139        assertUnlockWithToken(pid, shallowLockToken);
140
141        // Test deep lock
142        final String deepLockToken = getLockToken(lockObject(pid, true));
143
144        assertCannotSetPropertyWithoutLockToken(pid);
145        assertCannotSetPropertyWithoutLockToken("Deep lock must prevent property updates on child nodes!", childPid);
146        assertCanSetProperty(pid, deepLockToken);
147        assertCanSetProperty("Properties of a child of a shallow locked node"
148                + " should be updatable without the lock token!", childPid, deepLockToken);
149        assertUnlockWithToken(pid, deepLockToken);
150        assertCanSetProperty("Child of unlocked object must be able to be updated now!", childPid, null);
151    }

240     * Test that a transaction will fail on commit if a lock has been taken
241     * out on any touched resources between the completion of that operation
242     * within the transaction and the commit.
243     *
244     * This test won't work until https://www.pivotaltracker.com/story/show/69734118
245     * has been resolved.
246     */
247    @Test
248    @Ignore
249    public void testTransactionFailureOnCommit() throws IOException {
250        final String rootPid = getRandomUniquePid();
251        final String childPid = rootPid + "/" + getRandomUniquePid();
252        createObject(rootPid);
253        createObject(childPid);
254
255        final String inTxPid = getRandomUniquePid();
256        final String txId = createTransaction().replace(serverAddress, "");
257        LOGGER.info("Created transaction: " + txId);
258
259        // perform non-conflicting operation in transaction
260        createObject(txId + "/" + inTxPid);
261
262        // perform conflicting operation in transaction
263        assertCanSetProperty("Must be able to update properties on unlocked node within transaction!",

267        final String lockToken = getLockToken(lockObject(rootPid, true));

269        // commit the transaction (which should fail with CONFLICT)
270        Assert.assertEquals(CONFLICT.getStatusCode(), commitTransaction(txId).getStatusLine().getStatusCode());

273        Assert.assertEquals(NOT_FOUND.getStatusCode(), getObjectProperties(inTxPid).getStatusLine().getStatusCode());

276    /**
277     * Unlocks a lock with no token, asserts that it is successful and further
278     * asserts that property updates can again be made without the token.
279     */
280    private void assertUnlockWithoutToken(String pid) throws IOException {
281        Assert.assertEquals(NO_CONTENT.getStatusCode(),
282                unlockObject(pid, null).getStatusLine().getStatusCode());
283
284        assertCanSetProperty("Unlocked object must be able to be updated now!", pid, null);
285    }

287    /**
288     * Unlocks a lock with its token, asserts that it is successful and further
289     * asserts that property updates can again be made without the token.
290     */
291    private void assertUnlockWithToken(String pid, String lockToken) throws IOException {
292        Assert.assertEquals(NO_CONTENT.getStatusCode(),
293                unlockObject(pid, lockToken).getStatusLine().getStatusCode());
294
295        assertCanSetProperty("Unlocked object must be able to be updated now!", pid, null);
296    }

298    /**
299     * Assumes (and asserts) that a request to lock a resource was
300     * successful and returns the lock token from the response.
301     */
302    private String getLockToken(HttpResponse response) {
303        final StatusLine status = response.getStatusLine();
304        Assert.assertEquals(CREATED.getStatusCode(),
305                response.getStatusLine().getStatusCode());
306        final Header lockToken = response.getFirstHeader("Lock-Token");
307        Assert.assertNotNull("Lock-Token was not provided in response!", lockToken);
308        return lockToken.getValue();
309    }

318    /**
319     * Attempts to lock an object with the given deep locking status.
320     */
321    private HttpResponse lockObject(String pid, boolean deep) throws IOException {
322        final HttpPost post = new HttpPost(serverAddress + pid + "/" + FCR_LOCK
323                + (deep  ? "?deep=true" : "?deep=false"));
324        return client.execute(post);
325    }

327    /**
328     * Attempts to unlock lock an object.
329     */
330    private HttpResponse unlockObject(String pid, String lockToken) throws IOException {
331        final HttpDelete delete = new HttpDelete(serverAddress + pid + "/" + FCR_LOCK);
332        addLockToken(delete, lockToken);
333        return client.execute(delete);
334    }

346    private void assertCanSetProperty(final String message, final String pid, final String lockToken)
347            throws IOException {
348        final String propertyName = getRandomPropertyName();
349        final String propertyValue = getRandomPropertyValue();
350        Assert.assertEquals(message == null
351                    ? "Properties of a locked node must be updatable with the token!"
352                    : message, NO_CONTENT.getStatusCode(),
353                setProperty(pid, null, lockToken, propertyName, propertyValue).getStatusLine().getStatusCode());
354    }

356    private void assertCannotSetPropertyWithoutLockToken(final String pid) throws IOException {
357        assertCannotSetPropertyWithoutLockToken(null, pid);
358    }
359
360    private void assertCannotSetPropertyWithoutLockToken(final String message, final String pid) throws IOException {
361        final String propertyName = getRandomPropertyName();
362        final String propertyValue = getRandomPropertyValue();
363        Assert.assertEquals(message == null ? "Lock must prevent property updates!" : message,
364                CONFLICT.getStatusCode(),
365                setProperty(pid, null, null, propertyName, propertyValue).getStatusLine().getStatusCode());
366    }

368    /**
369     * Attempts to set a property on an object.
370     * @param pid the pid
371     * @param txId a transaction ID (may be null)
372     * @param lockToken a lock token (may be null)
373     * @param propertyUri the property to set
374     * @param value the value to set
375     * @return the HttpResponse
376     */
377    private HttpResponse setProperty(final String pid,
378                                       final String txId,
379                                       final String lockToken,
380                                       final String propertyUri,
381                                       final String value) throws IOException {
382        final HttpPatch postProp = new HttpPatch(serverAddress
383                + (txId != null ? txId + "/" : "") + pid);
384        addLockToken(postProp, lockToken);
385        postProp.setHeader("Content-Type", "application/sparql-update");
386        final String updateString =
387                "INSERT { <"
388                        + serverAddress + pid
389                        + "> <" + propertyUri + "> \"" + value + "\" } WHERE { }";
390        postProp.setEntity(new StringEntity(updateString));
391        return execute(postProp);
392    }

394    /**
395     * Adds a "Lock-Token" header to the request if token is not null.
396     */
397    private void addLockToken(HttpRequestBase method, String token) {
398        if (token != null) {
399            method.addHeader("Lock-Token", token);
400        }
401    }

403    /**
404     * Creates a transaction, asserts that it's successful and
405     * returns the transaction location.
406     * @return
407     * @throws IOException
408     */
409    private String createTransaction() throws IOException {
410        final HttpPost createTx = new HttpPost(serverAddress + "fcr:tx");
411        final HttpResponse response = execute(createTx);
412        Assert.assertEquals(CREATED.getStatusCode(), response.getStatusLine().getStatusCode());
413        return response.getFirstHeader("Location").getValue();
414    }
415
416    private HttpResponse commitTransaction(String txId) throws IOException {
417        final HttpPost commitTx = new HttpPost(serverAddress + txId + "/fcr:tx/fcr:commit");
418        return execute(commitTx);
419    }

421    private HttpResponse getObjectProperties(String pid) throws IOException {
422        return execute(new HttpGet(serverAddress + pid));
423    }
