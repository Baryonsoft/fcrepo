018

021import static org.fcrepo.http.commons.test.util.TestHelpers.mockRepository;
022import static org.fcrepo.http.commons.test.util.TestHelpers.setField;
023import static org.junit.Assert.assertEquals;
024import static org.junit.Assert.assertTrue;
025import static org.mockito.Mockito.verify;
026import static org.mockito.Mockito.when;
027import static org.mockito.MockitoAnnotations.initMocks;
028
029import javax.jcr.Session;
030import javax.jcr.Workspace;
031import javax.ws.rs.core.Response;
032import javax.ws.rs.core.UriBuilder;
033import javax.ws.rs.core.UriInfo;
034
035import org.fcrepo.kernel.RdfLexicon;
036import org.junit.Before;
037import org.junit.Test;
038import org.mockito.Mock;
039import org.modeshape.jcr.api.NamespaceRegistry;
040import org.modeshape.jcr.api.Repository;
041
042import com.hp.hpl.jena.rdf.model.Model;
043import com.sun.jersey.api.NotFoundException;
044
045/**
046 * @author Andrew Woods Date: 8/7/13
047 */
048public class FedoraRepositoryWorkspacesTest {
049
050    private FedoraRepositoryWorkspaces workspaces;
051
052    @Mock
053    private Workspace mockWorkspace;
054
055    @Mock
056    private NamespaceRegistry mockNamespaceRegistry;
057
058    @Mock
059    private UriInfo mockUriInfo;
060
061    @Mock
062    private UriBuilder mockUriBuilder;
063
064    @Mock
065    private Session mockSession;
066
067    @Mock
068    private Session mockWorkspaceSession;
069
070    @Mock
071    private Workspace mockOtherWorkspace;
072
073    @Mock
074    private Repository mockRepository;
075
076    @Before
077    public void setUp() throws Exception {
078        initMocks(this);
079        mockUriInfo = getUriInfoImpl();
080        workspaces = new FedoraRepositoryWorkspaces();
081        setField(workspaces, "session", mockSession);
082        setField(workspaces, "uriInfo", mockUriInfo);
083        when(mockSession.getWorkspace()).thenReturn(mockWorkspace);
084        when(mockWorkspace.getName()).thenReturn("default");
085        when(mockSession.getRepository()).thenReturn(mockRepository);
086    }
087
088    @Test
089    public void testGetWorkspaces() throws Exception {
090        when(mockWorkspace.getAccessibleWorkspaceNames()).thenReturn(
091                new String[] {"xxx"});
092
093
094        // Do the test.
095        final Model result = workspaces.getWorkspaces().asModel();
096
097        assertTrue(result.contains(createResource("http://localhost/fcrepo/"),
098                                      RdfLexicon.HAS_WORKSPACE,
099                                      createResource("http://localhost/fcrepo/workspace:xxx" )));
100
101    }
102
103    @Test
104    public void testCreateWorkspace() throws Exception {
105        final Repository mockRepository = mockRepository();
106        when(mockSession.getRepository()).thenReturn(mockRepository);
107        when(mockWorkspaceSession.getRepository()).thenReturn(mockRepository);
108        when(mockRepository.login("xxx")).thenReturn(mockWorkspaceSession);
109        when(mockWorkspaceSession.getWorkspace()).thenReturn(mockOtherWorkspace);
110        when(mockOtherWorkspace.getName()).thenReturn("xxx");
111        final Response response = workspaces.createWorkspace("xxx", mockUriInfo);
112        verify(mockWorkspace).createWorkspace("xxx");
113        assertEquals(201, response.getStatus());
114        final String location = response.getMetadata().getFirst("Location").toString();
115        assertEquals("http://localhost/fcrepo/workspace:xxx/", location);
116    }
117
118    @Test
119    public void testDeleteWorkspace() throws Exception {
120        when(mockWorkspace.getAccessibleWorkspaceNames()).thenReturn(new String[] { "xxx" });
121        final Response response = workspaces.deleteWorkspace("xxx");
122        verify(mockWorkspace).deleteWorkspace("xxx");
123        assertEquals(204, response.getStatus());
124    }
125
126    @Test(expected = NotFoundException.class)
127    public void testDeleteMissingWorkspace() throws Exception {
128        when(mockWorkspace.getAccessibleWorkspaceNames()).thenReturn(new String[] { "yyy" });
129        final Response response = workspaces.deleteWorkspace("xxx");
130        assertEquals(404, response.getStatus());
131    }
132}
