306    @Test
307    public void testGetDatastreamsContents() throws RepositoryException,
308                                            IOException,
309                                            NoSuchAlgorithmException {
310        final String pid = "FedoraDatastreamsTest1";
311        final String dsId = "testDS";
312        final String dsContent = "asdf";
313        final Datastream mockDs = mockDatastream(pid, dsId, dsContent);
314        when(mockDs.hasContent()).thenReturn(true);
315        when(mockIterator.hasNext()).thenReturn(true, false);
316        when(mockIterator.nextNode()).thenReturn(mockDsNode);
317        when(mockIterator.getSize()).thenReturn(1L);
318        when(mockObject.getNode()).thenReturn(mockNode);
319        when(mockNode.getNodes(new String[] {dsId})).thenReturn(mockIterator);
320        when(mockNodes.getObject(mockSession, "/FedoraDatastreamsTest1"))
321                .thenReturn(mockObject);
322        when(mockDatastreams.asDatastream(mockDsNode)).thenReturn(mockDs);
323
324        final Response resp =
325            testObj.getBinaryContents(createPathList(pid), asList(dsId),
326                                         mockRequest);
327        final MultiPart multipart = (MultiPart) resp.getEntity();
328
329        verify(mockDs).getContent();
330        verify(mockSession, never()).save();
331        assertEquals(1, multipart.getBodyParts().size());
332        final InputStream actualContent =
333            (InputStream) multipart.getBodyParts().get(0).getEntity();
334        assertEquals("/FedoraDatastreamsTest1/testDS", multipart.getBodyParts()
335                .get(0).getContentDisposition().getFileName());
336        assertEquals("asdf", IOUtils.toString(actualContent, "UTF-8"));
337    }

366    public static MultiPart getStringsAsMultipart(
367            final Map<String, String> contents) {
368        final MultiPart multipart = new MultiPart();
369        for (final Map.Entry<String, String> e : contents.entrySet()) {
370            final String id = e.getKey();
371            final String content = e.getValue();
372            final InputStream src = IOUtils.toInputStream(content);
373            final StreamDataBodyPart part = new StreamDataBodyPart(id, src);
374            try {
375                final FormDataContentDisposition cd =
376                    new FormDataContentDisposition("form-data;name=" + id
377                            + ";filename=" + id + ".txt");
378                part.contentDisposition(cd);
379            } catch (final ParseException ex) {
380                ex.printStackTrace();
381            }
382            multipart.bodyPart(part);
383        }
384        return multipart;
385    }
