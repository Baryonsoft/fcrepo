328    static {
329        final ImmutableSet.Builder<Property> b = ImmutableSet.builder();
330        b.addAll(membershipProperties).addAll(fixityProperties).addAll(
331                searchProperties).addAll(ldpProperties).addAll(
332                repositoryProperties).addAll(namespaceProperties).addAll(
333                otherServiceProperties).addAll(contentProperties).addAll(
334                versioningProperties).addAll(jcrProperties);

338    private static Predicate<Property> hasJcrNamespace =
339        new Predicate<Property>() {
340
341            @Override
342            public boolean apply(final Property p) {
343                return !p.isAnon() && p.getNameSpace().equals(JCR_NAMESPACE);
344
345            }
346        };
347
348    private static Predicate<Property> hasFedoraNamespace =
349        new Predicate<Property>() {
350
351            @Override
352            public boolean apply(final Property p) {
353                return !p.isAnon()
354                        && p.getNameSpace().startsWith(REPOSITORY_NAMESPACE);
355
356            }
357        };
358
359    /**
360     * Detects whether an RDF property is managed by the repository.
361     */
362    @SuppressWarnings("unchecked")
363    public static final Predicate<Property> isManagedPredicate = or(
364            in(managedProperties), hasJcrNamespace, hasFedoraNamespace);
365
366    /**
367     * Detects whether an RDF predicate URI is managed by the repository.
368    **/
369    public static Predicate<String> isManagedPredicateURI =
370        new Predicate<String>() {
371            @Override
372            public boolean apply(final String uri) {
373                return isManagedPredicate.apply( createProperty(uri) );
374            }
375        };
