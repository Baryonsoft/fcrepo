17  package org.fcrepo.kernel.services;
18  
19  import java.util.Collection;
20  
21  import javax.jcr.Node;
22  import javax.jcr.RepositoryException;
23  import javax.jcr.Session;
24  import javax.jcr.Workspace;
25  
26  /**
27   * @author bbpennel
28   * @date Feb 19, 2014
29   */
30  public interface VersionService extends Service {
31  
32      /**
33       * Notifies the version manager that the node at a given path was updated so
34       * that if automatic versioning is set for that node, a version checkpoint
35       * will be made. When a node object is available, use of
36       * {@link #nodeUpdated(javax.jcr.Node)} will save the overhead of a
37       * redundant node lookup.
38       *
39       * @param session
40       * @param absPath the absolute path to the node that was created or modified
41       * @throws RepositoryException
42       */
43      void nodeUpdated(Session session, String absPath)
44          throws RepositoryException;
45  
46      /**
47       * Notifies the version manager that the given node was updated so that if
48       * automatic versioning is set for that node, a version checkpoint will be
49       * made.
50       *
51       * @param n the node that was updated
52       * @throws RepositoryException
53       */
54      void nodeUpdated(Node n) throws RepositoryException;
55  
56      /**
57       * Explicitly creates a version for the nodes at each path provided.
58       *
59       * @param workspace the workspace in which the node resides
60       * @param paths absolute paths to the nodes within the workspace
61       * @throws RepositoryException
62       */
63      void createVersion(Workspace workspace, Collection<String> paths)
64          throws RepositoryException;
65  
66      /**
67       * Reverts the node to the version identified by the label.  This method
68       * will throw a PathNotFoundException if no version with the given label is
69       * found.
70       *
71       * @param workspace the workspace in which the node resides
72       * @param absPath the path to the node whose version is to be reverted
73       * @param label identifies the historic version
74       * @throws RepositoryException
75       */
76      void revertToVersion(Workspace workspace, String absPath, String label)
77          throws RepositoryException;
78  
79      /**
80       * Remove a version of a node.  This method will throw a PathNotFoundException
81       * if no version with the given label is found.
82       *
83       * @param workspace the workspace in which the node resides
84       * @param absPath the path to the node whose version is to be removed
85       * @param label identifies the historic version by label or id
86       * @throws RepositoryException
87       */
88      void removeVersion(Workspace workspace, String absPath, String label)
89          throws RepositoryException;
90  
91      /**
92       * Creates a version checkpoint for the given node if versioning is enabled
93       * for that node type. When versioning is enabled this is the equivalent of
94       * VersionManager#checkpoint(node.getPath()), except that it is aware of
95       * TxSessions and queues these operations accordingly.
96       *
97       * @param node the node for whom a new version is to be minted
98       * @throws RepositoryException
99       */
100     void checkpoint(Node node) throws RepositoryException;
101 
102     /**
103      * @param txService the txService to set
104      */
105     void setTxService(final TransactionService txService);
106 }
