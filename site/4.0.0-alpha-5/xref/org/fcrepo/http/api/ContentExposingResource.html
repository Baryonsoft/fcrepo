40  /**
41   * An abstract class that sits between AbstractResource and any resource that
42   * wishes to share the routines for building responses containing binary
43   * content.
44   *
45   * @author Mike Durbin
46   */
47  public abstract class ContentExposingResource extends AbstractResource {
48  
49      protected static final int REQUESTED_RANGE_NOT_SATISFIABLE = 416;
50  
51      protected static final int PARTIAL_CONTENT = 206;
52  
53      /**
54       * A helper method that does most of the work associated with processing a request
55       * for content (or a range of the content) into a Response
56       */
57      protected Response getDatastreamContentResponse(final Datastream ds, final String rangeValue, final Request request,
58                                                      final HttpServletResponse servletResponse,
59                                                      final HttpIdentifierTranslator subjects) throws
60              RepositoryException, IOException {
61  
62          // we include an explicit etag, because the default behavior is to use the JCR node's etag, not
63          // the jcr:content node digest.
64          checkCacheControlHeaders(request, servletResponse, ds);
65  
66          final CacheControl cc = new CacheControl();
67          cc.setMaxAge(0);
68          cc.setMustRevalidate(true);
69          Response.ResponseBuilder builder;
70  
71          final InputStream content = ds.getContent();
72  
73          if (rangeValue != null && rangeValue.startsWith("bytes")) {
74  
75              final Range range = Range.convert(rangeValue);
76  
77              final long contentSize = ds.getContentSize();
78  
79              final String endAsString;
80  
81              if (range.end() == -1) {
82                  endAsString = Long.toString(contentSize - 1);
83              } else {
84                  endAsString = Long.toString(range.end());
85              }
86  
87              final String contentRangeValue =
88                  String.format("bytes %s-%s/%s", range.start(),
89                                   endAsString, contentSize);
90  
91              if (range.end() > contentSize ||
92                      (range.end() == -1 && range.start() > contentSize)) {
93                  builder = status(REQUESTED_RANGE_NOT_SATISFIABLE)
94                          .header("Content-Range", contentRangeValue);
95              } else {
96                  final RangeRequestInputStream rangeInputStream =
97                      new RangeRequestInputStream(content, range.start(), range.size());
98  
99                  builder = status(PARTIAL_CONTENT).entity(rangeInputStream)
100                         .header("Content-Range", contentRangeValue);
101             }
102 
103         } else {
104             builder = ok(content);
105         }
106 
107         final ContentDisposition contentDisposition = ContentDisposition.type("attachment")
108                 .fileName(ds.getFilename())
109                 .creationDate(ds.getCreatedDate())
110                 .modificationDate(ds.getLastModifiedDate())
111                 .size(ds.getContentSize())
112                 .build();
113 
114         return builder.type(ds.getMimeType()).header(
115                 "Link",
116                 subjects.getSubject(ds.getNode().getPath()) +
117                         ";rel=\"describedby\"").header("Accept-Ranges",
118                 "bytes").cacheControl(cc)
119                 .header("Content-Disposition", contentDisposition)
120                 .build();
121     }

123     /**
124      * Evaluate the cache control headers for the request to see if it can be served from
125      * the cache.
126      *
127      * @param request
128      * @param servletResponse
129      * @param resource
130      * @throws javax.jcr.RepositoryException
131      */
132     protected static void checkCacheControlHeaders(final Request request,
133                                                    final HttpServletResponse servletResponse,
134                                                    final Datastream resource) throws RepositoryException {
135 
136         final EntityTag etag = new EntityTag(resource.getContentDigest().toString());
137         final Date date = resource.getLastModifiedDate();
138 
139         final Date roundedDate = new Date();
140         if (date != null) {
141             roundedDate.setTime(date.getTime() - date.getTime() % 1000);
142         }
143         final Response.ResponseBuilder builder =
144             request.evaluatePreconditions(roundedDate, etag);
145 
146         if (builder != null) {
147             final CacheControl cc = new CacheControl();
148             cc.setMaxAge(0);
149             cc.setMustRevalidate(true);
150             // here we are implicitly emitting a 304
151             // the exception is not an error, it's genuinely
152             // an exceptional condition
153             throw new WebApplicationException(builder.cacheControl(cc)
154                                                   .lastModified(date).tag(etag).build());
155         }
156 
157         addCacheControlHeaders(servletResponse, resource);
158     }
159 
160     /**
161      * Add ETag and Last-Modified cache control headers to the response
162      * @param servletResponse
163      * @param resource
164      * @throws RepositoryException
165      */
166     protected static void addCacheControlHeaders(final HttpServletResponse servletResponse,
167                                                  final Datastream resource) throws RepositoryException {
168 
169         final EntityTag etag = new EntityTag(resource.getContentDigest().toString());
170         final Date date = resource.getLastModifiedDate();
171 
172         if (!etag.getValue().isEmpty()) {
173             servletResponse.addHeader("ETag", etag.toString());
174         }
175 
176         if (date != null) {
177             servletResponse.addDateHeader("Last-Modified", date.getTime());
178         }
179     }
