136     @Test
137     public void testAuthenticateUserRole() {
138         final ServletContainerAuthenticationProvider provider =
139                 (ServletContainerAuthenticationProvider) ServletContainerAuthenticationProvider
140                 .getInstance();
141 
142         provider.setFad(fad);
143 
144         when(
145                 request.isUserInRole(ServletContainerAuthenticationProvider.FEDORA_USER_ROLE))
146                 .thenReturn(true);
147 
148         when(principal.getName()).thenReturn("userName");
149 
150         final ExecutionContext result =
151                 provider.authenticate(creds, "repo", "workspace", context,
152                         sessionAttributes);
153 
154         assertNotNull(result);
155 
156         final AdvancedAuthorizationProvider authProvider =
157                 (AdvancedAuthorizationProvider) result.getSecurityContext();
158         final AdvancedAuthorizationProvider.Context authContext =
159                 mock(AdvancedAuthorizationProvider.Context.class);
160 
161         // Perform hasPermission on auth provider to capture result principals
162         // from authenticate
163         authProvider.hasPermission(authContext, mock(Path.class),
164                 new String[] {"read"});
165 
166         verify(fad).hasPermission(any(Session.class), any(Path.class), any(String[].class));
167 
168         @SuppressWarnings("unchecked")
169         final Set<Principal> resultPrincipals = (Set<Principal>) sessionAttributes.get(FEDORA_ALL_PRINCIPALS);
170 
171         assertEquals(2, resultPrincipals.size());
172         assertTrue("EVERYONE principal must be present", resultPrincipals
173                 .contains(ServletContainerAuthenticationProvider.EVERYONE));
174         assertTrue("User principal must be present", resultPrincipals
175                 .contains(principal));
176 
177         assertEquals(
178                 "Resulting security context must exist and belong to userName",
179                 "userName", result.getSecurityContext().getUserName());
180 
181         assertEquals(fad, provider.getFad());
182     }

184     @Test
185     public void testAuthenticateWithPrincipalFactory() {
186         final ServletContainerAuthenticationProvider provider =
187                 (ServletContainerAuthenticationProvider) ServletContainerAuthenticationProvider.getInstance();
188 
189         provider.setFad(fad);
190 
191         when(request.isUserInRole(ServletContainerAuthenticationProvider.FEDORA_USER_ROLE)).thenReturn(true);
192 
193         final Set<Principal> groupPrincipals = new HashSet<>();
194         final Principal groupPrincipal = mock(Principal.class);
195         groupPrincipals.add(groupPrincipal);
196         final HttpHeaderPrincipalProvider principalProvider = mock(HttpHeaderPrincipalProvider.class);
197         when(principalProvider.getPrincipals(any(Credentials.class))).thenReturn(groupPrincipals);
198 
199         final Set<PrincipalProvider> providers = new HashSet<>();
200         providers.add(principalProvider);
201 
202         provider.setPrincipalProviders(providers);
203 
204         final ExecutionContext result = provider.authenticate(creds, "repo", "workspace", context, sessionAttributes);
205 
206         final AdvancedAuthorizationProvider authProvider = (AdvancedAuthorizationProvider) result.getSecurityContext();
207         final AdvancedAuthorizationProvider.Context authContext = mock(AdvancedAuthorizationProvider.Context.class);
208 
209         // Perform hasPermission on auth provider to capture result principals
210         // from authenticate
211         authProvider.hasPermission(authContext, mock(Path.class), new String[] {"read"});
212 
213         verify(fad).hasPermission(any(Session.class), any(Path.class), any(String[].class));
214 
215         @SuppressWarnings("unchecked")
216         final Set<Principal> resultPrincipals = (Set<Principal>) sessionAttributes.get(FEDORA_ALL_PRINCIPALS);
217 
218         assertEquals(3, resultPrincipals.size());
219         assertTrue("EVERYONE principal must be present", resultPrincipals
220                 .contains(ServletContainerAuthenticationProvider.EVERYONE));
221         assertTrue("User principal must be present", resultPrincipals.contains(principal));
222         assertTrue("Group Principal from factory must be present", resultPrincipals.contains(groupPrincipal));
