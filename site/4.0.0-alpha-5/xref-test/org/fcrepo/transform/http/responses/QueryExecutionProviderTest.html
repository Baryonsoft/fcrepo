48  public class QueryExecutionProviderTest {
49  
50      final QueryExecutionProvider testObj = new QueryExecutionProvider();
51  
52      Dataset testData = new DatasetImpl(createDefaultModel());
53  
54      {
55          testData.asDatasetGraph().getDefaultGraph().add(
56                  new Triple(createURI("test:subject"),
57                          createURI("test:predicate"),
58                          createLiteral("test:object")));
59          testData.asDatasetGraph().getDefaultGraph().add(
60                  new Triple(createURI("test:subject"), primaryTypePredicate,
61                          createLiteral("nt:file")));
62  
63      }
64  
65      @SuppressWarnings("unchecked")
66      @Test
67      public void testWriteTo() throws WebApplicationException,
68                               IllegalArgumentException, IOException {
69  
70          final Query sparqlQuery =
71              QueryFactory.create("SELECT ?x WHERE { ?x ?y ?z }");
72  
73          final QueryExecution testResult =
74              QueryExecutionFactory.create(sparqlQuery, testData);
75  
76          final ByteArrayOutputStream outStream = new ByteArrayOutputStream();
77  
78          testObj.writeTo(testResult, QueryExecution.class, mock(Type.class),
79                  null, valueOf(contentTypeResultsXML),
80                  mock(MultivaluedMap.class), outStream);
81          final byte[] results = outStream.toByteArray();
82          assertTrue("Got no output from serialization!", results.length > 0);
83          assertTrue("Couldn't find test RDF-object mentioned!", new String(
84                  results).contains("test:subject"));
85      }
86  
87      @Test
88      public void testGetSize() {
89          assertEquals("Returned wrong size from QueryExecutionProvider!",
90                  testObj.getSize(null, null, null, null, null), -1);
91  
92      }
93  
94      @Test
95      public void testIsWritable() throws Exception {
96          assertTrue(
97                  "Gave false response to QueryExecutionProvider.isWriteable() that contained a legitimate combination of parameters!",
98                  testObj.isWriteable(QueryExecution.class, QueryExecution.class,
99                          null, valueOf(contentTypeResultsXML)));
100         assertFalse(
101                 "RdfProvider.isWriteable() should return false if asked to serialize anything other than QueryExecution!",
102                 testObj.isWriteable(QueryExecutionProvider.class,
103                         QueryExecutionProvider.class, null,
104                         valueOf(contentTypeResultsXML)));
105         assertFalse(
106                 "RdfProvider.isWriteable() should return false to text/html!",
107                 testObj.isWriteable(QueryExecution.class, QueryExecution.class,
108                         null, TEXT_HTML_TYPE));
