482    /* Deletions */
483    @Test
484    public void testAdminCanDeleteOpenObjAndItsDescendants()
485            throws ClientProtocolException, IOException {
486        assertEquals("Admin cannot delete object testparent3!", NO_CONTENT
487                .getStatusCode(),
488                canDelete("exampleadmin", "testparent3", true));
489
490        assertEquals(
491                "Admin should not have permission to try to read deleted datastream testparent3/tsp1_data!",
492                FORBIDDEN.getStatusCode(), canDelete("exampleadmin",
493                        "testparent3/tsp1_data", true));
494
495        assertEquals(
496                "Admin should not have permission to try to read deleted datastream testparent3/tsp2_data!",
497                FORBIDDEN.getStatusCode(), canDelete("exampleadmin",
498                        "testparent3/tsp2_data", true));
499
500        assertEquals(
501                "Admin should not have permission to try to read deleted object testparent3/testchild3a!",
502                FORBIDDEN.getStatusCode(), canDelete("exampleadmin",
503                        "testparent3/testchild3a", true));
504
505        assertEquals(
506                "Admin should not have permission to try to read deleted object testparent3/testchild3b!",
507                FORBIDDEN.getStatusCode(), canDelete("exampleadmin",
508                        "testparent3/testchild3b", true));
509
510        assertEquals(
511                "Fedora Admin should not be able to read deleted datastream testparent3/tsp1_data!",
512                NOT_FOUND.getStatusCode(), canDelete("fedoraAdmin",
513                        "testparent3/tsp1_data", true));
514
515        assertEquals(
516                "Fedora Admin should not be able to read deleted datastream testparent3/tsp2_data!",
517                NOT_FOUND.getStatusCode(), canDelete("fedoraAdmin",
518                        "testparent3/tsp2_data", true));
519
520        assertEquals(
521                "Fedora Admin should not be able to read deleted object testparent3/testchild3a!",
522                NOT_FOUND.getStatusCode(), canDelete("fedoraAdmin",
523                        "testparent3/testchild3a", true));
524
525        assertEquals(
526                "Fedora Admin should not be able to read deleted object testparent3/testchild3b!",
527                NOT_FOUND.getStatusCode(), canDelete("fedoraAdmin",
528                        "testparent3/testchild3b", true));
529    }
