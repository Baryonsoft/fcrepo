065    @Test
066    public void testMultiValuedLiteralTriple() throws RepositoryException {
067
068        when(mockProperty.isMultiple()).thenReturn(true);
069        when(mockProperty.getType()).thenReturn(STRING);
070        when(mockProperty.getValues()).thenReturn(
071                new Value[] {mockValue, mockValue2});
072        when(mockValue.getString()).thenReturn(TEST_VALUE);
073        when(mockValue.getType()).thenReturn(STRING);
074        when(mockValue2.getString()).thenReturn(TEST_VALUE);
075        when(mockValue2.getType()).thenReturn(STRING);
076        final Function<Iterator<Value>, Iterator<Triple>> mapping =
077            testPropertyToTriple.apply(mockProperty);
078        final Iterator<Triple> ts =
079            mapping.apply(twoValueIterator(mockValue, mockValue2));
080        final Triple t1 = ts.next();
081        LOGGER.debug("Constructed triple: {}", t1);
082        final Triple t2 = ts.next();
083        LOGGER.debug("Constructed triple: {}", t2);
084
085        assertEquals("Got wrong RDF object!", TEST_VALUE, t1.getObject()
086                .getLiteralValue());
087        assertEquals("Got wrong RDF predicate!", createProperty(
088                TEST_PROPERTY_NAME).asNode(), t1.getPredicate());
089        assertEquals("Got wrong RDF subject!", TEST_NODE_SUBJECT.asNode(), t1
090                .getSubject());
091
092        assertEquals("Got wrong RDF object!", TEST_VALUE, t2.getObject()
093                .getLiteralValue());
094        assertEquals("Got wrong RDF predicate!", createProperty(
095                TEST_PROPERTY_NAME).asNode(), t2.getPredicate());
096        assertEquals("Got wrong RDF subject!", TEST_NODE_SUBJECT.asNode(), t2
097                .getSubject());
098    }

368    private static <T> Iterator<T> twoValueIterator(final T t, final T t2) {
369        return ImmutableList.of(t, t2).iterator();
