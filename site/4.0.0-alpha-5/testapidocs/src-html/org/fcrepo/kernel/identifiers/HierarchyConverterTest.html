017package org.fcrepo.kernel.identifiers;
018
019import static com.google.common.base.Joiner.on;
020import static com.google.common.base.Strings.repeat;
021import static java.util.Collections.nCopies;
022import static java.util.regex.Pattern.compile;
023import static org.fcrepo.jcr.FedoraJcrTypes.FCR_CONTENT;
024import static org.junit.Assert.assertEquals;
025import static org.junit.Assert.assertNull;
026import static org.junit.Assert.assertTrue;
027import static org.modeshape.jcr.api.JcrConstants.JCR_CONTENT;
028import static org.slf4j.LoggerFactory.getLogger;
029
030import java.util.regex.Matcher;
031
032import org.junit.Before;
033import org.junit.Test;
034import org.slf4j.Logger;
035
036public class HierarchyConverterTest {
037
038    private static final String separator = "/";
039
040    private HierarchyConverter testTranslator;
041
042    private static final String[] testIdSegments = {"test1", "test2", "test3"};
043
044    private static final String startingSegments = testIdSegments[0] + separator + testIdSegments[1];
045
046    private static final String endingSegment = testIdSegments[2];
047
048    private final String testId = on(separator).join(testIdSegments);
049
050    private static final Logger log = getLogger(HierarchyConverterTest.class);
051
052    @Before
053    public void setUp() {
054        testTranslator = new HierarchyConverter();
055        testTranslator.setPrefix("");
056        testTranslator.setLevels(0);
057        testTranslator.setLength(1);
058        testTranslator.setSeparator(separator);
059    }
060
061    @Test
062    public void testNullForward() {
063        assertNull("Should get null forward for null input!", testTranslator.convert(null));
064    }
065
066    @Test(expected = IllegalArgumentException.class)
067    public void testBadSegmentLength() {
068        testTranslator.setLength(0);
069    }
070
071    @Test
072    public void testNoOpForward() {
073        assertEquals("Should not have altered input identifier!", testId, testTranslator.convert(testId));
074    }
075
076    @Test
077    public void testVaryingSegments() {
078        for (byte length = 1; length < 5; length++) {
079            for (byte levels = 1; levels < 5; levels++) {
080                testRoundTrip(levels, length);
081            }
082        }
083    }
084
085    public void testRoundTrip(final byte levels, final byte length) {
086        testTranslator.setLevels(levels);
087        testTranslator.setLength(length);
088        final String result = testTranslator.reverse().convert(testId);
089        final String testRegexp =
090            startingSegments + separator + hierarchyRegexpSection(levels, length) + separator + endingSegment;
091        final Matcher matches = compile(testRegexp).matcher(result);
092        log.debug("Got result of translation: {}", result);
093        log.debug("Matching against test pattern: {}", testRegexp);
094        assertTrue("Did not find the appropriate modification to the input identifier!", matches.matches());
095        final String shouldBeOriginal = testTranslator.convert(result);
096        assertEquals("Didn't get original back!", testId, shouldBeOriginal);
097    }
098
099    private static String hierarchyRegexpSection(final byte levels, final byte length) {
100        return on(separator).join(nCopies(levels, repeat("\\w", length)));
101    }
102
103    @Test
104    public void testRecurse() {
105        testTranslator.setLevels(3);
106        testTranslator.setLength(3);
107        final String firstPass = testTranslator.reverse().convert(testId);
108        final String secondPass = testTranslator.reverse().convert(firstPass);
109        assertEquals("Failed to retrieve original after two stages of translation!", testId, testTranslator
110                .convert(testTranslator.convert(secondPass)));
111    }
112
113    @Test
114    public void testWeirdIds() {
115        testTranslator.setLevels(3);
116        testTranslator.setLength(3);
117        String result;
118
119        result = testTranslator.reverse().convert("");
120        log.debug("Empty identifier translated into {}.", "", result);
121        assertEquals("Should not have altered empty identifier!", "", testTranslator.convert(result));
122
123        result = testTranslator.reverse().convert(separator);
124        log.debug("Separator identifier translated into {}.", separator, result);
125        assertEquals("Should have altered separator identifier to empty identifier!", "", testTranslator
126                .convert(result));
127    }
128
129    @Test
130    public void testContentPaths() {
131        testTranslator.setLevels(0);
132        testTranslator.setLength(1);
133        String result;
134
135        final String externalTestId = on(separator).join(testId, FCR_CONTENT);
136        final String internalTestId = on(separator).join(testId, JCR_CONTENT);
137
138        result = testTranslator.reverse().convert(externalTestId);
139        assertEquals("Should have swapped content suffix to JCR!", internalTestId, result);
140
141        result = testTranslator.convert(internalTestId);
142        assertEquals("Should have swapped content suffix to FCR!", externalTestId, result);
143    }
144}
