053    @Test
054    public void testConsumeAsync() throws Exception {
055
056        final RdfStream testStream = new RdfStream(profferedStatements);
057
058        final Set<Statement> rejectedStatements =
059            newHashSet(profferedStatements);
060        final Set<Statement> acceptedStatements = newHashSet();
061
062        final Set<Resource> rejectedMixins = newHashSet(profferedMixins);
063        final Set<Resource> acceptedMixins = newHashSet();
064
065        testPersister =
066            new PersistingRdfStreamConsumer(mockGraphSubjects, mockSession, testStream) {
067
068                @Override
069                protected void operateOnProperty(final Statement s,
070                    final Node subjectNode) throws RepositoryException {
071                    rejectedStatements.remove(s);
072                    acceptedStatements.add(s);
073                }
074
075                @Override
076                protected void operateOnMixin(final Resource mixinResource,
077                        final Node subjectNode) throws RepositoryException {
078                    rejectedMixins.remove(mixinResource);
079                    acceptedMixins.add(mixinResource);
080                }
081            };
082
083        testPersister.consumeAsync();
084
085        assertTrue("Failed to operate on ordinary property!",
086                acceptedStatements.contains(propertyStatement)
087                        && !rejectedStatements.contains(propertyStatement));
088
089        assertTrue("Wrongly operated on LDP managed property!",
090                !acceptedStatements.contains(ldpManagedPropertyStatement)
091                        && rejectedStatements.contains(ldpManagedPropertyStatement));
092
093        assertTrue("Wrongly operated on JCR managed property!",
094                !acceptedStatements.contains(jcrManagedPropertyStatement)
095                        && rejectedStatements.contains(jcrManagedPropertyStatement));
096
097        assertTrue("Wrongly operated on Fedora managed property!",
098                !acceptedStatements.contains(fedoraManagedPropertyStatement)
099                        && rejectedStatements.contains(fedoraManagedPropertyStatement));
100
101        assertTrue("Wrongly operated on foreign property!",
102                !acceptedStatements.contains(foreignStatement)
103                        && rejectedStatements.contains(foreignStatement));
104
105        assertTrue("Wrongly operated on managed mixin!", !acceptedMixins
106                .contains(managedMixinStatement.getObject().asResource())
107                && rejectedMixins.contains(managedMixinStatement.getObject()
108                        .asResource()));
109
110        assertTrue("Failed to operate on ordinary mixin!", acceptedMixins
111                .contains(mixinStatement.getObject().asResource())
112                && !rejectedMixins.contains(mixinStatement.getObject()
113                        .asResource()));
114
115    }

118    public void testBadStream() throws Exception {
119        when(mockTriples.hasNext())
120                .thenThrow(new RuntimeException("Expected."));
121        testPersister =
122            new PersistingRdfStreamConsumer(mockGraphSubjects, mockSession,
123                    new RdfStream(mockTriples)) {
124
125                    @Override
126                    protected void operateOnProperty(final Statement s,
127                        final Node subjectNode) throws RepositoryException {
128                    }
129
130                    @Override
131                    protected void operateOnMixin(final Resource mixinResource,
132                            final Node subjectNode) throws RepositoryException {
133                    }
134                };
135        // this should blow out when we try to retrieve the result
136        testPersister.consumeAsync().get();
137    }

139    @Before
140    public void setUp() throws RepositoryException {
141        initMocks(this);
142
143        for (final Statement fedoraStatement : fedoraStatements) {
144            when(mockSession.getNode(mockGraphSubjects.getPathFromSubject(fedoraStatement.getSubject())))
145                    .thenReturn(mockNode);
146            when(mockGraphSubjects.isFedoraGraphSubject(fedoraStatement.getSubject())).thenReturn(true);
147        }
148        when(mockSession.getNode(mockGraphSubjects.getPathFromSubject(foreignStatement.getSubject()))).thenReturn(
149                mockNode);
150        when(mockGraphSubjects.isFedoraGraphSubject(foreignStatement.getSubject())).thenReturn(false);
151    }
