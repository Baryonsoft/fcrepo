017package org.fcrepo.http.commons.responses;
018
019import static com.hp.hpl.jena.graph.Node.ANY;
020import static com.hp.hpl.jena.graph.NodeFactory.createLiteral;
021import static com.hp.hpl.jena.graph.NodeFactory.createURI;
022import static com.hp.hpl.jena.rdf.model.ModelFactory.createDefaultModel;
023import static org.fcrepo.http.commons.responses.RdfSerializationUtils.RFC2822DATEFORMAT;
024import static org.fcrepo.http.commons.responses.RdfSerializationUtils.getFirstValueForPredicate;
025import static org.fcrepo.http.commons.responses.RdfSerializationUtils.lastModifiedPredicate;
026import static org.fcrepo.http.commons.responses.RdfSerializationUtils.setCachingHeaders;
027import static org.junit.Assert.assertEquals;
028import static org.junit.Assert.assertTrue;
029
030import java.util.Calendar;
031
032import javax.ws.rs.core.MultivaluedMap;
033
034import org.joda.time.DateTime;
035import org.junit.Test;
036
037import com.hp.hpl.jena.graph.Triple;
038import com.hp.hpl.jena.query.Dataset;
039import com.hp.hpl.jena.query.DatasetFactory;
040import com.hp.hpl.jena.rdf.model.Model;
041import com.hp.hpl.jena.sparql.core.DatasetImpl;
042import com.hp.hpl.jena.sparql.util.Context;
043import com.hp.hpl.jena.sparql.util.Symbol;
044import com.sun.jersey.core.util.MultivaluedMapImpl;
045
046public class RdfSerializationUtilsTest {
047
048    Dataset testData = new DatasetImpl(createDefaultModel());
049
050    {
051        testData.asDatasetGraph().getDefaultGraph().add(
052                new Triple(createURI("test:subject"),
053                        createURI("test:predicate"),
054                        createLiteral("test:object")));
055
056    }
057
058    @Test
059    public void testGetFirstValueForPredicate() {
060        final String foundValue =
061            getFirstValueForPredicate(testData, ANY,
062                    createURI("test:predicate"));
063        assertEquals("Didn't find correct value for predicate!", foundValue,
064                "test:object");
065    }
066
067    @SuppressWarnings("unchecked")
068    @Test
069    public void testSetCachingHeaders() {
070        final MultivaluedMap<?, ?> headers = new MultivaluedMapImpl();
071        setCachingHeaders((MultivaluedMap<String, Object>) headers, testData);
072        assertTrue(headers.get("Cache-Control").size() > 0);
073    }
074
075    @SuppressWarnings("unchecked")
076    @Test
077    public void testSetCachingHeadersWithLastModified() {
078        final MultivaluedMap<?, ?> headers = new MultivaluedMapImpl();
079
080        final Model m = createDefaultModel();
081
082        final Calendar c = Calendar.getInstance();
083        m.add(m.createResource("test:subject"), m
084                .createProperty(lastModifiedPredicate.getURI()), m
085                .createTypedLiteral(c));
086        final Dataset testDatasetWithLastModified = DatasetFactory.create(m);
087        final Context context = testDatasetWithLastModified.getContext();
088        context.set(Symbol.create("uri"), "test:subject");
089
090        setCachingHeaders((MultivaluedMap<String, Object>) headers,
091                testDatasetWithLastModified);
092        assertTrue(new DateTime(c).withMillisOfSecond(0).isEqual(
093                RFC2822DATEFORMAT.parseDateTime((String) headers.get(
094                        "Last-Modified").get(0))));
095    }
096
097}
