022import static org.fcrepo.http.commons.test.util.TestHelpers.mockSession;
023import static org.fcrepo.http.commons.test.util.TestHelpers.setField;
024import static org.mockito.Matchers.any;
025import static org.mockito.Matchers.anyListOf;
026import static org.mockito.Matchers.eq;
027import static org.mockito.Mockito.verify;
028import static org.mockito.Mockito.when;
029import static org.mockito.MockitoAnnotations.initMocks;
030
031import javax.jcr.RepositoryException;
032import javax.jcr.Session;
033import javax.servlet.http.HttpServletResponse;
034import javax.ws.rs.core.MediaType;
035import javax.ws.rs.core.Request;
036import javax.ws.rs.core.UriBuilder;
037import javax.ws.rs.core.UriInfo;
038import javax.ws.rs.core.Variant;
039
040import org.fcrepo.kernel.rdf.IdentifierTranslator;
041import org.fcrepo.kernel.services.RepositoryService;
042import org.junit.Before;
043import org.junit.Ignore;
044import org.junit.Test;
045import org.mockito.Mock;
046
047@Ignore("Destined for death.")
048public class FedoraFieldSearchTest {
049
050    private FedoraFieldSearch testObj;
051
052    private Session mockSession;
053
054    @Mock
055    private Request mockRequest;
056
057    @Mock
058    private RepositoryService mockService;
059
060    private UriInfo uriInfo;
061
062    @Mock
063    private UriBuilder mockUriBuilder;
064
065    @Mock
066    private HttpServletResponse mockResponse;
067
068    @Before
069    public void setUp() throws Exception {
070        initMocks(this);
071        testObj = new FedoraFieldSearch();
072        mockSession = mockSession(testObj);
073        this.uriInfo = getUriInfoImpl();
074        setField(testObj, "uriInfo", uriInfo);
075        setField(testObj, "repositoryService", mockService);
076        setField(testObj, "session", mockSession);
077    }
078
079    @Test
080    public void testFieldSearch() throws RepositoryException {
081        when(mockRequest.selectVariant(anyListOf(Variant.class))).thenReturn(
082                new Variant(MediaType.valueOf("application/n-triples"), null,
083                        null));
084        when(
085                mockService
086                        .searchRepository(
087                                any(IdentifierTranslator.class),
088                                eq(createResource("http://localhost/fcrepo/fcr:search?q=ZZZ")),
089                                eq(mockSession), eq("ZZZ"), eq(0), eq(0L)))
090                .thenReturn(createMem());
091        when(uriInfo.getRequestUriBuilder()).thenReturn(mockUriBuilder);
092
093        testObj.searchSubmitRdf("ZZZ", 0, 0, mockRequest, mockResponse, uriInfo);
094
095        verify(mockService)
096                .searchRepository(
097                        any(IdentifierTranslator.class),
098                eq(createResource("http://localhost/fcrepo/fcr:search?q=ZZZ")),
099                        eq(mockSession), eq("ZZZ"), eq(0), eq(0L));
100    }
101
102}
