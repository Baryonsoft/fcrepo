058@RunWith(SpringJUnit4ClassRunner.class)
059@ContextConfiguration({"/spring-test/headers-jms.xml", "/spring-test/repo.xml",
060        "/spring-test/eventing.xml"})
061@DirtiesContext
062public class HeadersJMSIT implements MessageListener {
063
064    @Inject
065    private Repository repository;
066
067    @Inject
068    private ActiveMQConnectionFactory connectionFactory;
069
070    private Connection connection;
071
072    private javax.jms.Session session;
073
074    private MessageConsumer consumer;
075
076    private volatile Set<Message> messages;
077
078    private JcrTools jcrTools = new JcrTools(true);
079
080    private static final Logger LOGGER = getLogger(HeadersJMSIT.class);
081
082    /**
083     * Time to wait for a set of test messages.
084     */
085    private static final long TIMEOUT = 2000;
086
087    @Test
088    public void testIngestion() throws RepositoryException,
089                               InterruptedException, JMSException {
090
091        final String pid = "/testIngestion";
092        final String expectedEventType =
093            REPOSITORY_NAMESPACE + EventType.valueOf(NODE_ADDED).toString();
094        LOGGER.debug("Expecting a {} event", expectedEventType);
095        Boolean success = false;
096
097        final Session session = repository.login();
098        try {
099            final Node node = jcrTools.findOrCreateNode(session, pid);
100            node.addMixin(FEDORA_OBJECT);
101            session.save();
102
103            final Long start = currentTimeMillis();
104            synchronized (this) {
105                while ((currentTimeMillis() - start < TIMEOUT) && (!success)) {
106                    for (final Message message : messages) {
107                        if (getIdentifier(message).equals(pid)) {
108                            if (getEventTypes(message).contains(expectedEventType)) {
109                                success = true;
110                            }
111                        }
112                    }
113                    LOGGER.debug("Waiting for next message...");
114                    wait(1000);
115                }
116            }
117        } finally {
118            session.logout();
119        }
120        assertTrue(
121                "Found no message with correct identifer and correct event type!",
122                success);
123    }
124
125    @Test
126    public void testRemoval() throws RepositoryException, InterruptedException,
127                             JMSException {
128
129        final String pid = "/testRemoval";
130        final String expectedEventType =
131            REPOSITORY_NAMESPACE + EventType.valueOf(NODE_REMOVED).toString();
132        LOGGER.debug("Expecting a {} event", expectedEventType);
133        Boolean success = false;
134
135        final Session session = repository.login();
136        try {
137            final Node node = jcrTools.findOrCreateNode(session, pid);
138            node.addMixin(FEDORA_OBJECT);
139            session.save();
140            node.remove();
141            session.save();
142
143            final Long start = currentTimeMillis();
144            synchronized (this) {
145                while ((currentTimeMillis() - start < TIMEOUT) && (!success)) {
146                    for (final Message message : messages) {
147                        if (getIdentifier(message).equals(pid)) {
148                            if (getEventTypes(message).contains(expectedEventType)) {
149                                success = true;
150                            }
151                        }
152                    }
153                    LOGGER.debug("Waiting for next message...");
154                    wait(1000);
155                }
156            }
157        } finally {
158            session.logout();
159        }
160        assertTrue(
161                "Found no message with correct identifer and correct event type!",
162                success);
163    }
164
165    @Override
166    public void onMessage(final Message message) {
167        try {
168            LOGGER.debug(
169                    "Received JMS message: {} with identifier: {}, timestamp: {}, and event type: {}",
170                    message.getJMSMessageID(), getIdentifier(message),
171                    getTimestamp(message), getEventTypes(message));
172        } catch (final JMSException e) {
173            propagate(e);
174        }
175        messages.add(message);
176        synchronized (this) {
177            this.notifyAll();
178        }
179    }
180
181    @Before
182    public void acquireConnection() throws JMSException {
183        LOGGER.debug(this.getClass().getName() + " acquiring JMS connection.");
184        connection = connectionFactory.createConnection();
185        connection.start();
186        session = connection.createSession(false, AUTO_ACKNOWLEDGE);
187        consumer = session.createConsumer(session.createTopic("fedora"));
188        messages = new HashSet<>();
189        consumer.setMessageListener(this);
190    }
191
192    @After
193    public void releaseConnection() throws JMSException {
194        // ignore any remaining or queued messages
195        consumer.setMessageListener(new NoopListener());
196        // and shut the listening machinery down
197        LOGGER.debug(this.getClass().getName() + " releasing JMS connection.");
198        consumer.close();
199        session.close();
200        connection.close();
201    }
202
203    private static String getIdentifier(final Message msg) throws JMSException {
204        final String id = msg.getStringProperty(IDENTIFIER_HEADER_NAME);
205        LOGGER.debug("Processing an event with identifier: {}", id);
206        return id;
207    }
208
209    private static String getEventTypes(final Message msg) throws JMSException {
210        final String type = msg.getStringProperty(EVENT_TYPE_HEADER_NAME);
211        LOGGER.debug("Processing an event with type: {}", type);
212        return type;
213    }
214
215    private static Long getTimestamp(final Message msg) throws JMSException {
216        return msg.getLongProperty(TIMESTAMP_HEADER_NAME);
217    }
