079    @Test
080    public void testObjectGraph() throws Exception {
081        final Session session = repo.login();
082        final FedoraObject object =
083            objectService.createObject(session, "/graphObject");
084        final IdentifierTranslator subjects = new DefaultIdentifierTranslator();
085        final Dataset graphStore = object.getPropertiesDataset(subjects);
086
087        final String graphSubject = subjects.getSubject("/graphObject").getURI();
088
089        assertFalse("Graph store should not contain JCR prefixes",
090                    compile("jcr").matcher(graphStore.toString()).find());
091        assertFalse("Graph store should contain our fcrepo prefix",
092                    compile("fcrepo")
093                    .matcher(graphStore.toString()).find());
094
095        parseExecute("PREFIX dc: <http://purl.org/dc/elements/1.1/>\n" +
096                          "INSERT { <http://example/egbook> dc:title " +
097                          "\"This is an example of an update that will be " +
098                          "ignored\" } WHERE {}", graphStore);
099
100        parseExecute("PREFIX dc: <http://purl.org/dc/elements/1.1/>\n" +
101                          "INSERT { <" + graphSubject + "> dc:title " +
102                          "\"This is an example title\" } WHERE {}",
103                          graphStore);
104
105
106        final Value[] values = object.getNode().getProperty("dc:title").getValues();
107        assertTrue(values.length > 0);
108
109        assertTrue(values[0]
110                       .getString(),
111                      values[0]
112                          .getString().equals("This is an example title"));
113
114
115        parseExecute("PREFIX myurn: <info:myurn/>\n" +
116                          "INSERT { <" + graphSubject + "> myurn:info " +
117                          "\"This is some example data\"} WHERE {}",
118                          graphStore);
119
120        final Value value =
121            object.getNode().getProperty(object.getNode().getSession()
122                                         .getNamespacePrefix("info:myurn/") +
123                                         ":info").getValues()[0];
124
125        assertEquals("This is some example data", value.getString());
126
127        parseExecute("PREFIX fedora-rels-ext: <"
128                + RELATIONS_NAMESPACE + ">\n" +
129                "INSERT { <" + graphSubject + "> fedora-rels-ext:" +
130                "isPartOf <" + graphSubject + "> } WHERE {}", graphStore);
131
132        assertTrue(object.getNode().getProperty("fedorarelsext:isPartOf")
133                   .getValues()[0].getString(),
134                   object.getNode().getProperty("fedorarelsext:isPartOf")
135                   .getValues()[0].getString()
136                   .equals(object.getNode().getIdentifier()));
137
138
139        parseExecute("PREFIX dc: <http://purl.org/dc/elements/1.1/>\n" +
140                          "DELETE { <" + graphSubject + "> dc:title " +
141                          "\"This is an example title\" } WHERE {}",
142                          graphStore);
143
144        assertFalse("Found unexpected dc:title",
145                    object.getNode().hasProperty("dc:title"));
146
147        parseExecute("PREFIX fedora-rels-ext: <" +
148                    RELATIONS_NAMESPACE + ">\n" +
149                    "DELETE { <" + graphSubject + "> " +
150                    "fedora-rels-ext:isPartOf <" + graphSubject + "> " +
151                    "} WHERE {}", graphStore);
152        assertFalse("found unexpected reference",
153                    object.getNode().hasProperty("fedorarelsext:isPartOf"));
154
155        session.save();
156
157    }

159    @Test
160    public void testObjectGraphWithUriProperty() throws RepositoryException {
161        final Session session = repo.login();
162        final FedoraObject object =
163            objectService.createObject(session, "/graphObject");
164        final IdentifierTranslator subjects = new DefaultIdentifierTranslator();
165        final String graphSubject = subjects.getSubject("/graphObject").getURI();
166        final Dataset graphStore = object.getPropertiesDataset(subjects);
167
168        parseExecute("PREFIX some: <info:some#>\n" +
169                         "INSERT { <" + graphSubject + "> some:urlProperty " +
170                         "<" + graphSubject + "> } WHERE {}", graphStore);
171
172        final String prefix = session.getWorkspace().getNamespaceRegistry().getPrefix("info:some#");
173
174        assertNotNull(object.getNode().getProperty(prefix + ":urlProperty"));
175
176        assertEquals(object.getNode(), session.getNodeByIdentifier(object.getNode().getProperty(prefix + ":urlProperty_ref").getValues()[0].getString()));
177
178        parseExecute("PREFIX some: <info:some#>\n" +
179                         "DELETE { <" + graphSubject + "> some:urlProperty " +
180                         "<" + graphSubject + "> } WHERE {}", graphStore);
181
182        assertFalse(object.getNode().hasProperty(prefix + ":urlProperty_ref"));
183
184
185        parseExecute("PREFIX some: <info:some#>\n" +
186                         "INSERT DATA { <" + graphSubject + "> some:urlProperty <" + graphSubject + ">;\n" +
187                         "       some:urlProperty <info:somewhere/else> . }", graphStore);
188
189        assertEquals(1, object.getNode().getProperty(prefix + ":urlProperty_ref").getValues().length);
190        assertEquals(object.getNode(), session.getNodeByIdentifier(object.getNode().getProperty(prefix + ":urlProperty_ref").getValues()[0].getString()));
191
192    }
