065/**
066 * JAX-RS Resource offering identifier creation.
067 *
068 * @author ajs6f
069 * @author cbeer
070 */
071@Component
072@Scope("prototype")
073@Path("/{path: .*}/fcr:identifier")
074public class FedoraIdentifiers extends AbstractResource {
075
076    @InjectedSession
077    protected Session session;
078
079    /**
080     * Mint identifiers (without creating the objects)
081     *
082     * POST /path/to/mint/from/fcr:identifier?count=15
083     *
084     * @param count number of PIDs to return
085     * @return HTTP 200 with block of PIDs
086     */
087    @POST
088    @Timed
089    @Produces({TURTLE, N3, N3_ALT2, RDF_XML, NTRIPLES, APPLICATION_XML, TEXT_PLAIN, TURTLE_X,
090                      TEXT_HTML, APPLICATION_XHTML_XML})
091    public RdfStream getNextPid(@PathParam("path")
092            final List<PathSegment> pathList,
093            @QueryParam("count")
094            @DefaultValue("1")
095            final Integer count,
096            @Context
097            final UriInfo uriInfo) {
098
099
100        final String path = toPath(pathList);
101
102        final Node pidsResult = createURI(uriInfo.getAbsolutePath().toASCIIString());
103
104        final Collection<String> identifiers = transform(create(closed(1, count), integers()), pidMinter.makePid());
105
106        final HttpIdentifierTranslator subjects =
107                new HttpIdentifierTranslator(session, FedoraNodes.class, uriInfo);
108
109        return new RdfStream(transform(transform(identifiers, absolutize(path)),
110                identifier2triple(subjects, pidsResult))).topic(pidsResult).session(session);
111
112    }
113
114    private static Function<String, String> absolutize(final String path) {
115        return new Function<String, String>() {
116
117            @Override
118            public String apply(final String identifier) {
119                return path.equals("/") ? "/" + identifier : path + "/"
120                        + identifier;
121            }
122        };
123    }
124
125    private static Function<String, Triple> identifier2triple(
126        final IdentifierTranslator subjects, final Node pidsResult) {
127        return new Function<String, Triple>() {
128
129            @Override
130            public Triple apply(final String identifier) {
131
132                try {
133                    final Node s =
134                        subjects.getSubject(identifier).asNode();
135                    return Triple.create(pidsResult, HAS_MEMBER_OF_RESULT
136                            .asNode(), s);
137                } catch (final RepositoryException e) {
138                    throw propagate(e);
139                }
140            }
141        };
142    }
143}
