068/**
069 * @author Mike Durbin
070 */
071@Component
072@Scope("prototype")
073@Path("/{path: .*}/fcr:lock")
074public class FedoraLocks extends AbstractResource implements FedoraJcrTypes {
075
076    private static final Logger LOGGER = getLogger(FedoraLocks.class);
077
078    @InjectedSession
079    protected Session session;
080
081    /**
082     * Gets a description of the lock resource.
083     */
084    @GET
085    @Produces({TURTLE, N3, N3_ALT2, RDF_XML, NTRIPLES, APPLICATION_XML, TEXT_PLAIN, TURTLE_X,
086            TEXT_HTML, APPLICATION_XHTML_XML})
087    public RdfStream getLock(@PathParam("path") final List<PathSegment> pathList) throws RepositoryException {
088
089        final String path = toPath(pathList);
090        final Node node = session.getNode(path);
091        final Lock lock = lockService.getLock(session, path);
092        return getLockRdfStream(node, lock);
093    }
094
095    /**
096     * Creates a lock at the given path that is tied to the current session.
097     * Only the current session may make changes to the current path while
098     * the lock is held.
099     * @param isDeep if true the created lock will affect all nodes in the subgraph
100     *               below
101     */
102    @POST
103    @Timed
104    public Response createLock(@PathParam("path") final List<PathSegment> pathList,
105                               @QueryParam("deep") @DefaultValue("false") final boolean isDeep)
106        throws RepositoryException, URISyntaxException {
107        try {
108            final String path = toPath(pathList);
109            final Node node = session.getNode(path);
110            final Lock lock = lockService.acquireLock(session, path, isDeep);
111            session.save();
112            final String location = getTranslator().getSubject(node.getPath()).getURI();
113            LOGGER.debug("Locked {} with lock token {}.", path, lock.getLockToken());
114            return created(new URI(location)).entity(location).header("Lock-Token", lock.getLockToken()).build();
115        } finally {
116            session.logout();
117        }
118    }
119
120    /**
121     * Deletes a lock at the given path, freeing up other sessions to make
122     * changes to it or its descendants.
123     * @return
124     */
125    @DELETE
126    @Timed
127    public Response deleteLock(@PathParam("path") final List<PathSegment> pathList)
128        throws RepositoryException, URISyntaxException {
129        try {
130            final String path = toPath(pathList);
131            lockService.releaseLock(session, path);
132            session.save();
133            LOGGER.debug("Unlocked {}.", path);
134            return noContent().build();
135        } finally {
136            session.logout();
137        }
138    }
139
140    private HttpIdentifierTranslator getTranslator() {
141        return new HttpIdentifierTranslator(session, FedoraNodes.class, uriInfo);
142    }
143
144    private RdfStream getLockRdfStream(Node node, Lock lock) throws RepositoryException {
145        HttpIdentifierTranslator translator = getTranslator();
146        final com.hp.hpl.jena.graph.Node nodeSubject = translator.getSubject(node.getPath()).asNode();
147        final com.hp.hpl.jena.graph.Node lockSubject = createURI(nodeSubject.getURI() + "/" + FCR_LOCK);
148
149        final Triple[] lockTriples;
150        final Triple locksT = create(lockSubject, LOCKS.asNode(), nodeSubject);
151        final Triple isDeepT = create(lockSubject, IS_DEEP.asNode(),
152                createLiteral("false", "", XSDDatatype.XSDboolean));
153        if (lock.getLockToken() != null) {
154            lockTriples = new Triple[] {
155                locksT,
156                isDeepT,
157                create(lockSubject, HAS_LOCK_TOKEN.asNode(), createLiteral(lock.getLockToken()))};
158        } else {
159            lockTriples = new Triple[] {
160                locksT, isDeepT };
161        }
162        return new RdfStream(lockTriples).topic(lockSubject).session(session);
163    }
