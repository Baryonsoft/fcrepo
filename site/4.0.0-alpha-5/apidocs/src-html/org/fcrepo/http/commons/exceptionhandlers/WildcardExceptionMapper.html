017package org.fcrepo.http.commons.exceptionhandlers;
018
019import static com.google.common.base.Throwables.getStackTraceAsString;
020import static javax.ws.rs.core.Response.serverError;
021import static org.slf4j.LoggerFactory.getLogger;
022
023import javax.jcr.lock.LockException;
024import javax.jcr.security.AccessControlException;
025import javax.ws.rs.WebApplicationException;
026import javax.ws.rs.core.Response;
027import javax.ws.rs.ext.ExceptionMapper;
028import javax.ws.rs.ext.Provider;
029
030import org.fcrepo.kernel.exception.TransactionMissingException;
031import org.slf4j.Logger;
032
033/**
034 * Catch all the exceptions!
035 */
036@Provider
037public class WildcardExceptionMapper implements ExceptionMapper<Exception> {
038
039    Boolean showStackTrace = true;
040
041    private static final Logger LOGGER =
042        getLogger(WildcardExceptionMapper.class);
043
044    @Override
045    public Response toResponse(final Exception e) {
046
047        if (WebApplicationException.class.isAssignableFrom(e.getClass())) {
048            LOGGER.debug(
049                            "WebApplicationException intercepted by WildcardExceptionMapper: \n",
050                            e);
051            return ((WebApplicationException) e).getResponse();
052        }
053
054        if (java.security.AccessControlException.class.isAssignableFrom(e
055                .getClass())) {
056            return new AccessControlExceptionMapper()
057                    .toResponse((java.security.AccessControlException) e);
058        }
059
060        if (javax.jcr.AccessDeniedException.class.isAssignableFrom(e.getClass())) {
061            return new AccessDeniedExceptionMapper()
062                    .toResponse((javax.jcr.AccessDeniedException) e);
063        }
064        if (AccessControlException.class.isAssignableFrom(e.getClass())) {
065            return new AccessControlExceptionMapper()
066                    .toResponse((AccessControlException) e);
067        }
068
069        if (e.getCause() instanceof LockException) {
070            return new LockExceptionMapper()
071                    .toResponse((LockException) e.getCause());
072        }
073
074        if (e.getCause() instanceof TransactionMissingException) {
075            return new TransactionMissingExceptionMapper()
076                    .toResponse((TransactionMissingException) e.getCause());
077        }
078
079        LOGGER.info("Exception intercepted by WildcardExceptionMapper: \n", e);
080        return serverError().entity(
081                showStackTrace ? getStackTraceAsString(e) : null).build();
082    }
083
084    /**
085     * Set whether the full stack trace should be returned as part of the
086     * error response. This may be a bad idea if the stack trace is exposed
087     * to the public.
088     * @param showStackTrace
089     */
090    public void setShowStackTrace(final Boolean showStackTrace) {
091        this.showStackTrace = showStackTrace;
092    }
093}
