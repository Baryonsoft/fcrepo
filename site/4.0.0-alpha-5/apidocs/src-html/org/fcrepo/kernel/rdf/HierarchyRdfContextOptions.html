017package org.fcrepo.kernel.rdf;
018
019/**
020 * Options for the HierarchyRdfContext serialization
021 * @author cabeer
022 */
023public class HierarchyRdfContextOptions {
024    public static final HierarchyRdfContextOptions DEFAULT = HierarchyRdfContextOptions.defaultOptions();
025    public static final int DEFAULT_LIMIT = -1;
026    public static final int DEFAULT_OFFSET = -1;
027    public static final boolean DEFAULT_MEMBERSHIP = true;
028    public static final boolean DEFAULT_CONTAINMENT = true;
029
030    final private int limit;
031    final private int offset;
032    final private boolean membership;
033    final private boolean containment;
034
035    /**
036     * Options with the default values
037     */
038    public HierarchyRdfContextOptions() {
039        this(DEFAULT_LIMIT, DEFAULT_OFFSET);
040    }
041
042    /**
043     * Options with a limit and offset
044     * @param limit
045     * @param offset
046     */
047    public HierarchyRdfContextOptions(final int limit, final int offset) {
048        this(limit, offset, DEFAULT_MEMBERSHIP, DEFAULT_CONTAINMENT);
049    }
050
051    /**
052     * Set the full range of hierarchy options
053     * @param limit number of hierarchy nodes to display
054     * @param offset pagination offset
055     * @param membership include membership triples
056     * @param containment include containment triples
057     */
058    public HierarchyRdfContextOptions(final int limit,
059                                      final int offset,
060                                      final boolean membership,
061                                      final boolean containment) {
062        this.limit = limit;
063        this.offset = offset;
064        this.membership = membership;
065        this.containment = containment;
066    }
067
068
069    /**
070     * Should the serialization include membership triples
071     * @return
072     */
073    public boolean membershipEnabled() {
074        return membership || containment;
075    }
076
077    /**
078     * Should the serialziation include containment triples
079     * @return
080     */
081    public boolean containmentEnabled() {
082        return containment;
083    }
084
085    /**
086     * Is there a pagination offset that needs to be applied>?
087     * @return
088     */
089    public boolean hasOffset() {
090        return getOffset() >= 0;
091    }
092
093    /**
094     * Is there a pagination limit that needs to be applied?
095     * @return
096     */
097    public boolean hasLimit() {
098        return getLimit() >= 0;
099    }
100
101    /**
102     * Get the pagination limit value
103     * @return
104     */
105    public int getLimit() {
106        return limit;
107    }
108
109    /**
110     * Get the pagination offset value
111     * @return
112     */
113    public int getOffset() {
114        return offset;
115    }
116
117    private static HierarchyRdfContextOptions defaultOptions() {
118        return new HierarchyRdfContextOptions();
119    }
120
121}
