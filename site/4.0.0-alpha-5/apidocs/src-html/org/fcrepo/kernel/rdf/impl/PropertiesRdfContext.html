051/**
052 * {@link NodeRdfContext} for RDF that derives from JCR properties on a
053 * {@link Node}.
054 *
055 * @author ajs6f
056 * @date Oct 10, 2013
057 */
058public class PropertiesRdfContext extends NodeRdfContext {
059
060    private PropertyToTriple property2triple;
061
062    private static final Logger LOGGER = getLogger(PropertiesRdfContext.class);
063
064    /**
065     * Default constructor.
066     *
067     * @param node
068     * @throws RepositoryException
069     */
070
071    public PropertiesRdfContext(final javax.jcr.Node node, final IdentifierTranslator graphSubjects,
072        final LowLevelStorageService lowLevelStorageService) throws RepositoryException {
073        super(node, graphSubjects, lowLevelStorageService);
074        property2triple = new PropertyToTriple(graphSubjects);
075        putPropertiesIntoContext();
076    }
077
078    private void putPropertiesIntoContext() throws RepositoryException {
079
080        LOGGER.trace(
081                "Pushing RDF triples into context for properties of node: {}",
082                node());
083
084        // this node's own properties
085        if (node().hasProperties()) {
086            concat(triplesFromProperties(node()));
087        }
088
089        // if there's an accessible jcr:content node, include information about
090        // it
091        javax.jcr.Node contentNode = null;
092        try {
093            if (node().hasNode(JCR_CONTENT)) {
094                contentNode = node().getNode(JCR_CONTENT);
095            }
096        } catch (final AccessDeniedException e) {
097            LOGGER.trace("Access denied to content node", e);
098        }
099        if (contentNode != null) {
100            final Node contentSubject = graphSubjects().getSubject(contentNode.getPath()).asNode();
101            final Node subject = graphSubjects().getSubject(node().getPath()).asNode();
102            // add triples representing parent-to-content-child relationship
103            concat(new Triple[] {
104                    create(subject, HAS_CONTENT.asNode(), contentSubject),
105                    create(contentSubject, IS_CONTENT_OF.asNode(), subject)});
106            // add properties from content child
107            concat(new PropertiesRdfContext(node().getNode(JCR_CONTENT),
108                    graphSubjects(), lowLevelStorageService()));
109
110            // add triples describing storage of content child
111            lowLevelStorageService().setRepository(
112                    node().getSession().getRepository());
113            concat(Iterators.transform(lowLevelStorageService().getLowLevelCacheEntries(
114                    contentNode).iterator(),
115                    new Function<LowLevelCacheEntry, Triple>() {
116
117                        @Override
118                        public Triple apply(final LowLevelCacheEntry llce) {
119                            return create(contentSubject,
120                                    HAS_CONTENT_LOCATION.asNode(),
121                                    createResource(llce.getExternalIdentifier()).asNode());
122                        }
123                    }));
124
125        }
126
127        if (node().getPrimaryNodeType().getName().equals(ROOT)) {
128            concat(new RootRdfContext(node(), graphSubjects(), lowLevelStorageService()));
129        }
130
131    }
132
133    private Iterator<Triple> triplesFromProperties(final javax.jcr.Node n)
134        throws RepositoryException {
135        LOGGER.trace("Creating triples for node: {}", n);
136        final UnmodifiableIterator<Property> nonBinaryProperties =
137            Iterators.filter(new PropertyIterator(n.getProperties()),
138                    not(isBinaryContentProperty));
139
140        final UnmodifiableIterator<Property> nonBinaryPropertiesCopy =
141            Iterators.filter(new PropertyIterator(n.getProperties()),
142                    not(isBinaryContentProperty));
143
144        return Iterators.concat(new ZippingIterator<>(
145                Iterators.transform(
146                    nonBinaryProperties, property2values),
147                Iterators.transform(
148                    nonBinaryPropertiesCopy, property2triple)));
149
150    }
