056/**
057 * Assemble {@link Triple}s derived from the root {@link Node} of a repository.
058 *
059 * @author ajs6f
060 * @date Oct 18, 2013
061 */
062public class RootRdfContext extends NodeRdfContext {
063
064    private static final Logger LOGGER = getLogger(RootRdfContext.class);
065
066    /**
067     * Ordinary constructor.
068     *
069     * @param node
070     * @param graphSubjects
071     * @param llsService
072     * @throws RepositoryException
073     */
074    public RootRdfContext(final Node node, final IdentifierTranslator graphSubjects,
075            final LowLevelStorageService llsService) throws RepositoryException {
076
077        super(node, graphSubjects, llsService);
078
079        LOGGER.trace("Creating RDF triples for repository description");
080        final Repository repository = node().getSession().getRepository();
081
082        final ImmutableSet.Builder<Triple> b = builder();
083
084        for (final String key : repository.getDescriptorKeys()) {
085            final String descriptor = repository.getDescriptor(key);
086            if (descriptor != null) {
087                final String uri = REPOSITORY_NAMESPACE + "repository/" + key;
088                b.add(create(subject(), createURI(uri),
089                        createLiteral(descriptor)));
090            }
091        }
092        final NodeTypeManager nodeTypeManager =
093            node().getSession().getWorkspace().getNodeTypeManager();
094
095        final NodeTypeIterator nodeTypes = nodeTypeManager.getAllNodeTypes();
096        while (nodeTypes.hasNext()) {
097            final NodeType nodeType = nodeTypes.nextNodeType();
098            b.add(create(subject(), HAS_NODE_TYPE.asNode(),
099                    createLiteral(nodeType.getName())));
100        }
101
102        b.add(create(subject(), HAS_OBJECT_COUNT.asNode(), createLiteral(String
103                .valueOf(getRepositoryCount(repository)))));
104        b.add(create(subject(), HAS_OBJECT_SIZE.asNode(), createLiteral(String
105                .valueOf(getRepositorySize(repository)))));
106
107        // Get the cluster configuration, if available
108        // this ugly test checks to see whether this is an ordinary JCR
109        // repository or a ModeShape repo, which will possess the extra info
110        if (JcrRepository.class.isAssignableFrom(repository.getClass())) {
111            final Map<String, String> config =
112                new GetClusterConfiguration().apply(repository);
113            assert (config != null);
114
115            for (final Map.Entry<String, String> entry : config.entrySet()) {
116                b.add(create(subject(), createURI(REPOSITORY_NAMESPACE + entry.getKey()),
117                        createLiteral(entry.getValue())));
118            }
119        }
120
121        // retrieve the metrics from the service
122        final SortedMap<String, Counter> counters = getMetrics().getCounters();
123        // and add the repository metrics to the RDF model
124        if (counters.containsKey("LowLevelStorageService.fixity-check-counter")) {
125            b.add(create(subject(), HAS_FIXITY_CHECK_COUNT.asNode(),
126                    createTypedLiteral(
127                            counters.get(
128                                    "org.fcrepo.services."
129                                            + "LowLevelStorageService."
130                                            + "fixity-check-counter")
131                                    .getCount()).asNode()));
132        }
133
134        if (counters.containsKey("LowLevelStorageService.fixity-error-counter")) {
135            b.add(create(subject(), HAS_FIXITY_ERROR_COUNT.asNode(),
136                    createTypedLiteral(
137                            counters.get(
138                                    "org.fcrepo.services."
139                                            + "LowLevelStorageService."
140                                            + "fixity-error-counter")
141                                    .getCount()).asNode()));
142        }
143
144        if (counters
145                .containsKey("LowLevelStorageService.fixity-repaired-counter")) {
146            b.add(create(subject(), HAS_FIXITY_REPAIRED_COUNT.asNode(),
147                    createTypedLiteral(
148                            counters.get(
149                                    "org.fcrepo.services."
150                                            + "LowLevelStorageService."
151                                            + "fixity-repaired-counter")
152                                    .getCount()).asNode()));
153        }
154
155        // offer all these accumulated triples
156        concat(b.build());
157    }
