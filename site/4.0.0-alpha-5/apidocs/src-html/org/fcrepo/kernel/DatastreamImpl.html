048/**
049 * Abstraction for a Fedora datastream backed by a JCR node.
050 *
051 * @author ajs6f
052 * @date Feb 21, 2013
053 */
054public class DatastreamImpl extends FedoraResourceImpl implements Datastream {
055
056    private static final Logger LOGGER = getLogger(DatastreamImpl.class);
057
058    static final Histogram contentSizeHistogram =
059            getMetrics().histogram(name(DatastreamImpl.class, "content-size"));
060
061    /**
062     * The JCR node for this datastream
063     *
064     * @param n an existing {@link Node}
065     */
066    public DatastreamImpl(final Node n) {
067        super(n);
068
069        if (node.isNew()) {
070            initializeNewDatastreamProperties();
071        }
072    }
073
074    /**
075     * Create or find a FedoraDatastream at the given path
076     *
077     * @param session the JCR session to use to retrieve the object
078     * @param path the absolute path to the object
079     * @param nodeType primary type to assign to node
080     * @throws RepositoryException
081     */
082    public DatastreamImpl(final Session session, final String path, final String nodeType) throws RepositoryException {
083        super(session, path, nodeType);
084
085        if (node.isNew()) {
086            initializeNewDatastreamProperties();
087        }
088    }

090    /**
091     * Create or find a FedoraDatastream at the given path
092     *
093     * @param session the JCR session to use to retrieve the object
094     * @param path the absolute path to the object
095     * @throws RepositoryException
096     */
097    public DatastreamImpl(final Session session, final String path) throws RepositoryException {
098        this(session, path, JcrConstants.NT_FILE);
099    }
100
101    private void initializeNewDatastreamProperties() {
102        try {
103            if (node.isNew() || !hasMixin(node)) {
104                LOGGER.debug("Setting {} properties on a {} node...",
105                        FEDORA_DATASTREAM, JcrConstants.NT_FILE);
106                node.addMixin(FEDORA_DATASTREAM);
107
108                if (hasContent()) {
109                    decorateContentNode(getContentNode());
110                }
111            }
112        } catch (final RepositoryException ex) {
113            LOGGER.warn("Could not decorate {} with {} properties: {}",
114                    JCR_CONTENT, FEDORA_DATASTREAM, ex);
115        }
116    }

147    /*
148     * (non-Javadoc)
149     * @see org.fcrepo.kernel.Datastream#setContent(java.io.InputStream,
150     * java.lang.String, java.net.URI, java.lang.String,
151     * org.fcrepo.kernel.services.policy.StoragePolicyDecisionPoint)
152     */
153    @Override
154    public void setContent(final InputStream content, final String contentType,
155            final URI checksum, final String originalFileName,
156            final StoragePolicyDecisionPoint storagePolicyDecisionPoint)
157        throws RepositoryException, InvalidChecksumException {
158
159        final Node contentNode =
160                findOrCreateChild(node, JCR_CONTENT, NT_RESOURCE);
161
162        if (contentNode.canAddMixin(FEDORA_BINARY)) {
163            contentNode.addMixin(FEDORA_BINARY);
164        }
165
166        if (contentType != null) {
167            contentNode.setProperty(JCR_MIME_TYPE, contentType);
168        }
169
170        if (originalFileName != null) {
171            contentNode.setProperty(PREMIS_FILE_NAME, originalFileName);
172        }
173
174        LOGGER.debug("Created content node at path: {}", contentNode.getPath());
175
176        String hint = null;
177
178        if (storagePolicyDecisionPoint != null) {
179            hint = storagePolicyDecisionPoint.evaluatePolicies(node);
180        }
181        final ValueFactory modevf =
182                (ValueFactory) node.getSession().getValueFactory();
183        final Binary binary = modevf.createBinary(content, hint);
184
185        /*
186         * This next line of code deserves explanation. If we chose for the
187         * simpler line: Property dataProperty =
188         * contentNode.setProperty(JCR_DATA, requestBodyStream); then the JCR
189         * would not block on the stream's completion, and we would return to
190         * the requester before the mutation to the repo had actually completed.
191         * So instead we use createBinary(requestBodyStream), because its
192         * contract specifies: "The passed InputStream is closed before this
193         * method returns either normally or because of an exception." which
194         * lets us block and not return until the job is done! The simpler code
195         * may still be useful to us for an asynchronous method that we develop
196         * later.
197         */
198        final Property dataProperty = contentNode.setProperty(JCR_DATA, binary);
199
200        final String dsChecksum = binary.getHexHash();
201        final URI uriChecksumString = ContentDigest.asURI("SHA-1", dsChecksum);
202        if (checksum != null &&
203                !checksum.equals(uriChecksumString)) {
204            LOGGER.debug("Failed checksum test");
205            throw new InvalidChecksumException("Checksum Mismatch of " +
206                    uriChecksumString + " and " + checksum);
207        }
208
209        decorateContentNode(contentNode);
210
211        LOGGER.debug("Created data property at path: {}",
212                dataProperty.getPath());
213
214    }

216    /*
217     * (non-Javadoc)
218     * @see org.fcrepo.kernel.Datastream#setContent(java.io.InputStream)
219     */
220    @Override
221    public void setContent(final InputStream content) throws InvalidChecksumException, RepositoryException {
222        setContent(content, null, null, null, null);
223    }

225    /*
226     * (non-Javadoc)
227     * @see org.fcrepo.kernel.Datastream#getContentSize()
228     */
229    @Override
230    public long getContentSize() {
231        try {
232            return getContentNode().getProperty(CONTENT_SIZE)
233                    .getLong();
234        } catch (final RepositoryException e) {
235            LOGGER.info("Could not get contentSize()", e);
236        }
237
238        return -1L;
239    }
240
241    /*
242     * (non-Javadoc)
243     * @see org.fcrepo.kernel.Datastream#getContentDigest()
244     */
245    @Override
246    public URI getContentDigest() throws RepositoryException {
247        try {
248            return new URI(getContentNode().getProperty(CONTENT_DIGEST).getString());
249        } catch (final RepositoryException | URISyntaxException e) {
250            LOGGER.info("Could not get content digest: ", e);
251        }
252
253        return ContentDigest.missingChecksum();
254    }

295     * (non-Javadoc)
296     * @see org.fcrepo.kernel.Datastream#getFilename()

298    @Override
299    public String getFilename() throws RepositoryException {
300        if (hasContent() && getContentNode().hasProperty(PREMIS_FILE_NAME)) {
301            return getContentNode().getProperty(PREMIS_FILE_NAME).getString();
302        }
303        return getDsId();
304    }
305
306    private static void decorateContentNode(final Node contentNode) throws RepositoryException {
307        if (contentNode == null) {
308            LOGGER.warn("{} node appears to be null!", JCR_CONTENT);
309            return;
310        }
311        if (contentNode.canAddMixin(FEDORA_BINARY)) {
312            contentNode.addMixin(FEDORA_BINARY);
313        }
314
315        final Property dataProperty = contentNode.getProperty(JCR_DATA);
316        final Binary binary = (Binary) dataProperty.getBinary();
317        final String dsChecksum = binary.getHexHash();
318
319        contentSizeHistogram.update(dataProperty.getLength());
320
321        contentNode.setProperty(CONTENT_SIZE, dataProperty.getLength());
322        contentNode.setProperty(CONTENT_DIGEST, ContentDigest.asURI("SHA-1", dsChecksum).toString());
323
324        LOGGER.debug("Decorated data property at path: {}", dataProperty.getPath());
325    }

327    /**
328     * Check if the node has a fedora:datastream mixin
329     *
330     * @param node node to check
331     */
332    public static boolean hasMixin(final Node node) {
333        return isFedoraDatastream.apply(node);
334    }
