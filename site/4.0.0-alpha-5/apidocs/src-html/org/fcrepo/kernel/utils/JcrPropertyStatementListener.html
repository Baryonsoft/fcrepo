019import static com.google.common.base.Throwables.propagate;
020import static org.fcrepo.kernel.RdfLexicon.COULD_NOT_STORE_PROPERTY;
021import static org.fcrepo.kernel.RdfLexicon.LDP_NAMESPACE;
022import static org.slf4j.LoggerFactory.getLogger;
023
024import javax.jcr.NamespaceException;
025import javax.jcr.Node;
026import javax.jcr.RepositoryException;
027import javax.jcr.Session;
028import javax.jcr.Value;
029
030import org.fcrepo.kernel.RdfLexicon;
031import org.fcrepo.kernel.rdf.IdentifierTranslator;
032import org.fcrepo.kernel.rdf.JcrRdfTools;
033import org.slf4j.Logger;
034
035import com.hp.hpl.jena.rdf.listeners.StatementListener;
036import com.hp.hpl.jena.rdf.model.Model;
037import com.hp.hpl.jena.rdf.model.Resource;
038import com.hp.hpl.jena.rdf.model.Statement;
039import com.hp.hpl.jena.vocabulary.RDF;

041/**
042 * Listen to Jena statement events, and when the statement is changed in the
043 * graph store, make the change within JCR as well.
044 */
045public class JcrPropertyStatementListener extends StatementListener {
046
047    private final JcrRdfTools jcrRdfTools;
048
049    private NodePropertiesTools propertiesTools = new NodePropertiesTools();
050
051    private Model problems;
052
053    private static final Logger LOGGER =
054            getLogger(JcrPropertyStatementListener.class);
055
056    private final IdentifierTranslator subjects;
057
058    private final Session session;
059
060    /**
061     * Return a Listener given the subject factory and JcrSession.
062     * @param subjects
063     * @param session
064     * @param problemModel
065     * @return
066     */
067    public static JcrPropertyStatementListener getListener(
068            final IdentifierTranslator subjects, final Session session, final Model problemModel) {
069        return new JcrPropertyStatementListener(subjects, session, problemModel);
070    }
071
072    /**
073     * Construct a statement listener within the given session
074     *
075     * @param subjects
076     * @param session
077     */
078    private JcrPropertyStatementListener(final IdentifierTranslator subjects,
079            final Session session, final Model problems) {
080        super();
081        this.session = session;
082        this.subjects = subjects;
083        this.problems = problems;
084        this.jcrRdfTools = JcrRdfTools.withContext(subjects, session);
085    }
086
087    /**
088     * When a statement is added to the graph, serialize it to a JCR property
089     *
090     * @param s
091     */
092    @Override
093    public void addedStatement(final Statement s) {
094        LOGGER.debug(">> added statement {}", s);
095
096        try {
097            final Resource subject = s.getSubject();
098
099            // if it's not about a node, ignore it.
100            if (!subjects.isFedoraGraphSubject(subject)) {
101                return;
102            }
103
104            final Node subjectNode = session.getNode(subjects.getPathFromSubject(subject));
105
106            // special logic for handling rdf:type updates.
107            // if the object is an already-existing mixin, update
108            // the node's mixins. If it isn't, just treat it normally.
109            if (s.getPredicate().equals(RDF.type)
110                    && s.getObject().isResource()) {
111                final Resource mixinResource = s.getObject().asResource();
112
113                if (mixinResource.getNameSpace().equals(LDP_NAMESPACE)) {
114                    return;
115                }
116
117                try {
118                    final String namespacePrefix = session.getNamespacePrefix(mixinResource.getNameSpace());
119                    final String mixinName = namespacePrefix + ":" + mixinResource.getLocalName();
120
121                    if (FedoraTypesUtils.getNodeTypeManager(subjectNode).hasNodeType(mixinName)) {
122
123                        if (subjectNode.canAddMixin(mixinName)) {
124                            subjectNode.addMixin(mixinName);
125                        } else {
126                            problems.add(subject, COULD_NOT_STORE_PROPERTY, s.getPredicate().getURI());
127                        }
128
129                        return;
130
131                    }
132                } catch (final NamespaceException e) {
133                    LOGGER.trace("Unable to resolve registered namespace for resource {}: {}", mixinResource, e);
134                }
135            }
136
137            // extract the JCR propertyName from the predicate
138            final String propertyName =
139                    jcrRdfTools.getPropertyNameFromPredicate(subjectNode,
140                                                             s.getPredicate(),
141                                                             s.getModel()
142                                                                     .getNsPrefixMap());
143
144            if (validateModificationsForPropertyName(
145                    subject, subjectNode, s.getPredicate())) {
146                final Value v =
147                    jcrRdfTools.createValue(subjectNode, s.getObject(),
148                            propertiesTools.getPropertyType(subjectNode,
149                                    propertyName));
150                propertiesTools.appendOrReplaceNodeProperty(subjects,
151                        subjectNode, propertyName, v);
152            }
153        } catch (final RepositoryException e) {
154            throw propagate(e);
155        }
156
157    }
158
159    /**
160     * When a statement is removed, remove it from the JCR properties
161     *
162     * @param s
163     */
164    @Override
165    public void removedStatement(final Statement s) {
166        LOGGER.trace(">> removed statement {}", s);
167
168        try {
169            final Resource subject = s.getSubject();
170
171            // if it's not about a node, we don't care.
172            if (!subjects.isFedoraGraphSubject(subject)) {
173                return;
174            }
175
176            final Node subjectNode = session.getNode(subjects.getPathFromSubject(subject));
177
178            // special logic for handling rdf:type updates.
179            // if the object is an already-existing mixin, update
180            // the node's mixins. If it isn't, just treat it normally.
181            if (s.getPredicate().equals(RDF.type)
182                    && s.getObject().isResource()) {
183                final Resource mixinResource = s.getObject().asResource();
184                try {
185                    final String namespacePrefix = session.getNamespacePrefix(mixinResource.getNameSpace());
186                    final String mixinName = namespacePrefix + ":" + mixinResource.getLocalName();
187
188
189                    if (FedoraTypesUtils.getNodeTypeManager(subjectNode).hasNodeType(mixinName)) {
190                        try {
191                            subjectNode.removeMixin(mixinName);
192                        } catch (final RepositoryException e) {
193                            LOGGER.info(
194                                    "problem with removing <{}> <{}> <{}>: {}",
195                                    subject.getURI(),
196                                    RdfLexicon.COULD_NOT_STORE_PROPERTY,
197                                    s.getPredicate().getURI(),
198                                    e);
199                            problems.add(subject, RdfLexicon.COULD_NOT_STORE_PROPERTY, s.getPredicate().getURI());
200                        }
201                        return;
202                    }
203
204                } catch (final NamespaceException e) {
205                    LOGGER.trace("Unable to resolve registered namespace for resource {}: {}", mixinResource, e);
206                }
207
208            }
209
210            final String propertyName =
211                jcrRdfTools.getPropertyNameFromPredicate(subjectNode, s.getPredicate());
212
213
214            // if the property doesn't exist, we don't need to worry about it.
215            if (subjectNode.hasProperty(propertyName) &&
216                    validateModificationsForPropertyName(subject, subjectNode,
217                                                            s.getPredicate())) {
218                final Value v =
219                    jcrRdfTools.createValue(subjectNode, s.getObject(),
220                            propertiesTools.getPropertyType(subjectNode,
221                                    propertyName));
222                propertiesTools.removeNodeProperty(subjects, subjectNode,
223                        propertyName, v);
224            }
225
226        } catch (final RepositoryException e) {
227            throw propagate(e);
228        }
229
230    }
231
232    private boolean validateModificationsForPropertyName(
233            final Resource subject, final Node subjectNode, final Resource predicate) {
234        if (jcrRdfTools.isInternalProperty(subjectNode, predicate)) {
235            LOGGER.debug("problem with <{}> <{}> <{}>",
236                         subject.getURI(),
237                         RdfLexicon.COULD_NOT_STORE_PROPERTY,
238                         predicate.getURI());
239            problems.add(subject, RdfLexicon.COULD_NOT_STORE_PROPERTY, predicate.getURI());
240            return false;
241        }
242
243        return true;
244    }
245
246    /**
247     * Get a list of any problems from trying to apply the statement changes to
248     * the node's properties
249     *
250     * @return
251     */
252    public Model getProblems() {
253        return problems;
254    }
