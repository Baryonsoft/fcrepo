072/**
073 * Endpoint for transforming object properties using stored
074 * or POSTed transformations.
075 */
076@Component
077@Scope("prototype")
078@Path("/{path: .*}/fcr:transform")
079public class FedoraTransform extends AbstractResource {
080
081    @InjectedSession
082    protected Session session;
083
084    private static final Logger LOGGER = getLogger(FedoraTransform.class);
085
086    @Autowired(required = false)
087    private TransformationFactory transformationFactory;
088
089    /**
090     * Register the LDPath configuration tree in JCR
091     *
092     * @throws RepositoryException
093     * @throws java.io.IOException
094     * @throws SecurityException
095     */
096    @PostConstruct
097    public void setUpRepositoryConfiguration() throws RepositoryException, IOException {
098
099        if (transformationFactory == null) {
100            transformationFactory = new TransformationFactory();
101        }
102
103        final Session internalSession = sessions.getInternalSession();
104        try {
105            // register our CND
106            jcrTools.registerNodeTypes(internalSession, "ldpath.cnd");
107
108            // create the configuration base path
109            jcrTools.findOrCreateNode(internalSession, "/fedora:system/fedora:transform", "fedora:configuration",
110                    "fedora:node_type_configuration");
111            final Node node =
112                jcrTools.findOrCreateNode(internalSession, CONFIGURATION_FOLDER + "default", NT_FOLDER, NT_FOLDER);
113            LOGGER.debug("Transforming node: {}", node.getPath());
114
115            // register an initial default program
116            if (!node.hasNode(NT_BASE)) {
117                final Node baseConfig = node.addNode(NT_BASE, NT_FILE);
118                jcrTools.uploadFile(internalSession, baseConfig.getPath(), getClass().getResourceAsStream(
119                        "/ldpath/default/nt_base_ldpath_program.txt"));
120            }
121            internalSession.save();
122        } finally {
123            internalSession.logout();
124        }
125    }
126
127    /**
128     * Execute an LDpath program transform
129     *
130     * @param pathList
131     * @return Binary blob
132     * @throws RepositoryException
133     */
134    @GET
135    @Path("{program}")
136    @Produces({APPLICATION_JSON})
137    @Timed
138    public Object evaluateLdpathProgram(@PathParam("path")
139        final List<PathSegment> pathList, @PathParam("program")
140        final String program) throws RepositoryException {
141
142        try {
143            final String path = toPath(pathList);
144            final FedoraResource object = nodeService.getObject(session, path);
145
146            final Dataset propertiesDataset =
147                object.getPropertiesDataset(new HttpIdentifierTranslator(session, FedoraNodes.class, uriInfo));
148
149            return getNodeTypeTransform(object.getNode(), program).apply(propertiesDataset);
150
151        } finally {
152            session.logout();
153        }
154    }
155
156    /**
157     * Get the LDPath output as a JSON stream appropriate for e.g. Solr
158     *
159     * @param pathList
160     * @param requestBodyStream
161     * @return
162     * @throws RepositoryException
163     */
164    @POST
165    @Consumes({APPLICATION_RDF_LDPATH, contentTypeSPARQLQuery})
166    @Produces({APPLICATION_JSON, contentTypeTextTSV, contentTypeTextCSV,
167            contentTypeSSE, contentTypeTextPlain, contentTypeResultsJSON,
168            contentTypeResultsXML, contentTypeResultsBIO, contentTypeTurtle,
169            contentTypeN3, contentTypeNTriples, contentTypeRDFXML})
170    @Timed
171    public Object evaluateTransform(@PathParam("path")
172        final List<PathSegment> pathList, @HeaderParam("Content-Type")
173        final MediaType contentType, final InputStream requestBodyStream)
174        throws RepositoryException {
175
176        try {
177            final String path = toPath(pathList);
178            final FedoraResource object = nodeService.getObject(session, path);
179            final Dataset propertiesDataset =
180                object.getPropertiesDataset(new HttpIdentifierTranslator(session, FedoraNodes.class, uriInfo));
181
182            return transformationFactory.getTransform(contentType, requestBodyStream).apply(propertiesDataset);
183
184        } finally {
185            session.logout();
186        }
187    }
188}
