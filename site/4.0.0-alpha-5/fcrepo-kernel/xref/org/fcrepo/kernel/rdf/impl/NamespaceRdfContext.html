43  /**
44   * An {@link RdfStream} that holds the namespace mappings for serializations,
45   * as well as {@link Triple}s describing those namespaces.
46   *
47   * @author ajs6f
48   * @date Oct 9, 2013
49   */
50  public class NamespaceRdfContext extends RdfStream {
51  
52      private static final Logger LOGGER = getLogger(NamespaceRdfContext.class);
53  
54      /**
55       * Default constructor. Loads context with RDF describing namespaces in
56       * scope in the repository.
57       *
58       * @param session
59       * @throws RepositoryException
60       */
61      public NamespaceRdfContext(final Session session) throws RepositoryException {
62          super();
63          final NamespaceRegistry namespaceRegistry =
64              session.getWorkspace().getNamespaceRegistry();
65          checkNotNull(namespaceRegistry,
66                  "Couldn't find namespace registry in repository!");
67  
68          final ImmutableMap.Builder<String, String> namespaces =
69              ImmutableMap.builder();
70          final ImmutableCollection.Builder<Triple> nsTriples =
71              ImmutableSet.builder();
72          for (String prefix : namespaceRegistry.getPrefixes()) {
73              if (!prefix.isEmpty()) {
74                  final String nsURI = namespaceRegistry.getURI(prefix);
75                  if (prefix.equals("jcr")) {
76                      prefix = "fcrepo";
77                  }
78                  LOGGER.trace(
79                          "Discovered namespace prefix \"{}\" with URI \"{}\"",
80                          prefix, nsURI);
81                  final String rdfNsUri = getRDFNamespaceForJcrNamespace(nsURI);
82                  // first, let's put the namespace in context
83                  namespaces.put(prefix, rdfNsUri);
84                  LOGGER.trace("Added namespace prefix \"{}\" with URI \"{}\"",
85                          prefix, rdfNsUri);
86                  final Node nsSubject = createURI(rdfNsUri);
87                  // now, some triples describing this namespace
88                  nsTriples.add(create(nsSubject, type.asNode(), VOAF_VOCABULARY
89                          .asNode()));
90                  nsTriples.add(create(nsSubject, HAS_NAMESPACE_PREFIX.asNode(),
91                          createLiteral(prefix)));
92                  nsTriples.add(create(nsSubject, HAS_NAMESPACE_URI.asNode(),
93                          createLiteral(rdfNsUri)));
94              }
95          }
96          concat(nsTriples.build()).namespaces(namespaces.build());
97      }
98  }
