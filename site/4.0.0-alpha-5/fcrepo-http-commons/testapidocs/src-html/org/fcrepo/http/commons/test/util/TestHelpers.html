213    public static Datastream mockDatastream(final String pid,
214            final String dsId, final String content) {
215        final Datastream mockDs = mock(Datastream.class);
216        final FedoraObject mockObj = mock(FedoraObject.class);
217        try {
218            when(mockObj.getName()).thenReturn(pid);
219            when(mockDs.getPath()).thenReturn("/" + pid + "/" + dsId);
220            when(mockDs.getObject()).thenReturn(mockObj);
221            when(mockDs.getDsId()).thenReturn(dsId);
222            when(mockDs.getMimeType()).thenReturn("application/octet-stream");
223            when(mockDs.getCreatedDate()).thenReturn(new Date());
224            when(mockDs.getLastModifiedDate()).thenReturn(new Date());
225            if (content != null) {
226                final MessageDigest md = MessageDigest.getInstance("SHA-1");
227                final byte[] digest = md.digest(content.getBytes());
228                final URI cd = asURI("SHA-1", digest);
229                when(mockDs.getContent()).thenReturn(
230                        IOUtils.toInputStream(content));
231                when(mockDs.getContentDigest()).thenReturn(cd);
232                when(mockDs.getEtagValue()).thenReturn(cd.toString());
233            }
234        } catch (final Throwable t) {
235        }
236        return mockDs;
237    }

263    /**
264     * Set a field via reflection
265     *
266     * @param parent the owner object of the field
267     * @param name the name of the field
268     * @param obj the value to set
269     */
270    public static void setField(final Object parent, final String name,
271            final Object obj) {
272        /* check the parent class too if the field could not be found */
273        try {
274            final Field f = findField(parent.getClass(), name);
275            f.setAccessible(true);
276            f.set(parent, obj);
277        } catch (final Exception e) {
278            e.printStackTrace();
279        }
280    }
281
282    private static Field findField(final Class<?> clazz, final String name)
283        throws NoSuchFieldException {
284        for (final Field f : clazz.getDeclaredFields()) {
285            if (f.getName().equals(name)) {
286                return f;
287            }
288        }
289        if (clazz.getSuperclass() == null) {
290            throw new NoSuchFieldException("Field " + name
291                    + " could not be found");
292        }
293        return findField(clazz.getSuperclass(), name);
294    }
