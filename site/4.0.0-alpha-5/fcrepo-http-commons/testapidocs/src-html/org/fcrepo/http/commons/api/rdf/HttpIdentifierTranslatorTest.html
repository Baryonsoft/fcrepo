065    @Test
066    public void testGetGraphSubject() throws RepositoryException {
067        final String expected = "http://localhost:8080/fcrepo/rest" + testPath;
068        when(mockNode.getPath()).thenReturn(testPath);
069        when(mockWorkspace.getName()).thenReturn("default");
070        when(mockSession.getWorkspace()).thenReturn(mockWorkspace);
071        when(mockNode.getSession()).thenReturn(mockSession);
072        Resource actual = testObj.getSubject(mockNode.getPath());
073        assertEquals(expected, actual.getURI());
074        when(mockNode.getPath()).thenReturn(testPath + "/jcr:content");
075        actual = testObj.getSubject(mockNode.getPath());
076        assertEquals(expected + "/fcr:content", actual.getURI());
077    }

079    @Test
080    public void testGetNodeFromGraphSubject() throws PathNotFoundException,
081            RepositoryException {
082
083        when(mockSession.nodeExists(testPath)).thenReturn(true);
084        when(mockSession.getNode(testPath)).thenReturn(mockNode);
085        when(mockWorkspace.getName()).thenReturn("default");
086        when(mockSession.getWorkspace()).thenReturn(mockWorkspace);
087        // test a good subject
088        when(mockSubject.getURI()).thenReturn(
089                "http://localhost:8080/fcrepo/rest" + testPath);
090        when(mockSubject.isURIResource()).thenReturn(true);
091        Node actual = mockSession.getNode(testObj.getPathFromSubject(mockSubject));
092        verify(mockSession).getNode(testPath);
093        assertEquals(mockNode, actual);
094        // test a bad subject
095        when(mockSubject.getURI()).thenReturn(
096                "http://localhost:8080/fcrepo/rest2" + testPath + "/bad");
097        actual = mockSession.getNode(testObj.getPathFromSubject(mockSubject));
098        assertEquals(null, actual);
099        // test a non-existent path
100        when(mockSubject.getURI()).thenReturn(
101                "http://localhost:8080/fcrepo/rest" + testPath + "/bad");
102        actual = mockSession.getNode(testObj.getPathFromSubject(mockSubject));
103        assertEquals(null, actual);
104        // test a fcr:content path
105        when(mockSession.nodeExists(testPath + "/jcr:content")).thenReturn(true);
106        when(mockSubject.getURI())
107                .thenReturn(
108                        "http://localhost:8080/fcrepo/rest" + testPath +
109                                "/fcr:content");
110        actual = mockSession.getNode(testObj.getPathFromSubject(mockSubject));
111        verify(mockSession).getNode(testPath + "/jcr:content");
112    }

166    protected static UriInfo getUriInfoImpl(final String path) {
167        // UriInfo ui = mock(UriInfo.class,withSettings().verboseLogging());
168        final UriInfo ui = mock(UriInfo.class);
169        final UriBuilder ub = new UriBuilderImpl();
170        ub.scheme("http");
171        ub.host("localhost");
172        ub.port(8080);
173        ub.path("/fcrepo");
174
175        final UriBuilder rb = new UriBuilderImpl();
176        rb.scheme("http");
177        rb.host("localhost");
178        rb.port(8080);
179        rb.path("/fcrepo/rest" + path);
180
181        when(ui.getRequestUri()).thenReturn(
182                URI.create("http://localhost:8080/fcrepo/rest" + path));
183        when(ui.getBaseUri()).thenReturn(
184                URI.create("http://localhost:8080/fcrepo"));
185        when(ui.getBaseUriBuilder()).thenReturn(ub);
186        when(ui.getAbsolutePathBuilder()).thenReturn(rb);
187
188        return ui;
189    }
