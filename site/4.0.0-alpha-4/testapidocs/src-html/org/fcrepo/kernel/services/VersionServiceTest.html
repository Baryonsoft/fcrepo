017package org.fcrepo.kernel.services;
018
019import org.fcrepo.kernel.Transaction;
020import org.junit.Before;
021import org.junit.Test;
022import org.mockito.Mock;
023
024import javax.jcr.Node;
025import javax.jcr.Property;
026import javax.jcr.Session;
027import javax.jcr.Value;
028import javax.jcr.Workspace;
029import javax.jcr.version.VersionManager;
030
031import static org.junit.Assert.assertNotNull;
032import static org.mockito.Mockito.mock;
033import static org.mockito.Mockito.never;
034import static org.mockito.Mockito.only;
035import static org.mockito.Mockito.verify;
036import static org.mockito.Mockito.when;
037import static org.mockito.MockitoAnnotations.initMocks;
038
039/**
040 * @author Mike Durbin
041 */
042public class VersionServiceTest {
043
044    private static final String USER_NAME = "test";
045
046    private VersionService testObj;
047
048    private TransactionService txService;
049
050    @Mock
051    private Session s;
052
053    @Mock
054    private Workspace mockWorkspace;
055
056    @Mock
057    private VersionManager mockVM;
058
059
060    @Before
061    public void setup() throws Exception {
062        txService = new TransactionServiceImpl();
063        initMocks(this);
064        testObj = new VersionServiceImpl();
065
066        testObj.setTxService(txService);
067        txService.setVersionService(testObj);
068
069        s = mock(Session.class);
070        mockWorkspace = mock(Workspace.class);
071        when(mockWorkspace.getName()).thenReturn("default");
072        when(s.getWorkspace()).thenReturn(mockWorkspace);
073        mockVM = mock(VersionManager.class);
074        when(mockWorkspace.getVersionManager()).thenReturn(mockVM);
075
076        // add a node that's versioned (but not auto-versioned)
077        final Node versionedNode = mock(Node.class);
078        when(versionedNode.getPath()).thenReturn("/example-versioned");
079        when(versionedNode.getSession()).thenReturn(s);
080        when(versionedNode.isNodeType(VersionServiceImpl.VERSIONABLE))
081                .thenReturn(true);
082        when(s.getNode("/example-versioned")).thenReturn(versionedNode);
083
084        // add a node that's autoversioned
085        final Node autoversionedNode = mock(Node.class);
086        when(autoversionedNode.getPath()).thenReturn("/example-auto-versioned");
087        when(autoversionedNode.getSession()).thenReturn(s);
088        when(autoversionedNode.isNodeType(VersionServiceImpl.VERSIONABLE))
089                .thenReturn(true);
090        when(s.getNode("/example-auto-versioned")).thenReturn(autoversionedNode);
091        final Property autoVersionProperty = mock(Property.class);
092        final Value autoVersionValue = mock(Value.class);
093        when(autoVersionValue.getString()).thenReturn(
094                VersionServiceImpl.AUTO_VERSION);
095        when(autoVersionProperty.isMultiple()).thenReturn(true);
096        when(autoVersionProperty.getValues()).thenReturn(new Value[] { autoVersionValue });
097        when(autoversionedNode.hasProperty(VersionServiceImpl.VERSION_POLICY))
098                .thenReturn(true);
099        when(autoversionedNode.getProperty(VersionServiceImpl.VERSION_POLICY))
100                .thenReturn(autoVersionProperty);
101
102
103        // add a node that's unversioned
104        final Node unversionedNode = mock(Node.class);
105        when(unversionedNode.getPath()).thenReturn("/example-unversioned");
106        when(unversionedNode.getSession()).thenReturn(s);
107        when(unversionedNode.isNodeType(VersionServiceImpl.VERSIONABLE))
108                .thenReturn(false);
109        when(s.getNode("/example-unversioned")).thenReturn(unversionedNode);
110    }
111
112    @Test
113    public void testCheckpoint() throws Exception {
114        // request a version be created
115        testObj.nodeUpdated(s, "/example-versioned");
116
117        // ensure that it was
118        verify(mockVM, never()).checkpoint("/example-versioned");
119    }
120
121    @Test
122    public void testCheckpointUnversioned() throws Exception {
123        // request a version be created
124        testObj.nodeUpdated(s, "/example-unversioned");
125
126        // ensure that it was
127        verify(mockVM, never()).checkpoint("/example-unversioned");
128    }
129
130    @Test
131    public void testCheckpointAutoVersioned() throws Exception {
132        // request a version be created
133        testObj.nodeUpdated(s, "/example-auto-versioned");
134
135        // ensure that it was
136        verify(mockVM, only()).checkpoint("/example-auto-versioned");
137    }
138
139    @Test
140    public void testDeferredCheckpointVersioned() throws Exception {
141        // start a transaction
142        final Transaction t = txService.beginTransaction(s, USER_NAME);
143        s = t.getSession();
144        when(s.getNamespaceURI(TransactionServiceImpl.FCREPO4_TX_ID))
145                .thenReturn(t.getId());
146
147        assertNotNull("Transaction must have started!",
148                txService.getTransaction(
149                        s.getNode("/example-auto-versioned").getSession()));
150
151        // request a version be created
152        testObj.nodeUpdated(s, "/example-versioned");
153
154        // ensure that no version was created (because the transaction is still open)
155        verify(mockVM, never()).checkpoint("/example-versioned");
156
157        // close the transaction
158        txService.commit(t.getId());
159
160        // ensure that no version was made because none was explicitly requested
161        verify(mockVM, never()).checkpoint("/example-versioned");
162    }
163
