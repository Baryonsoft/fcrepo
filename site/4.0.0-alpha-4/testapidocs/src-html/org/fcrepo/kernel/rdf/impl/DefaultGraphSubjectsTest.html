017package org.fcrepo.kernel.rdf.impl;
018
019import static com.hp.hpl.jena.rdf.model.ResourceFactory.createResource;
020import static org.fcrepo.kernel.rdf.impl.DefaultGraphSubjects.RESOURCE_NAMESPACE;
021import static org.junit.Assert.assertEquals;
022import static org.junit.Assert.assertFalse;
023import static org.junit.Assert.assertTrue;
024import static org.junit.Assert.assertNull;
025import static org.mockito.Mockito.verify;
026import static org.mockito.Mockito.when;
027import static org.mockito.MockitoAnnotations.initMocks;
028
029import javax.jcr.Node;
030import javax.jcr.RepositoryException;
031import javax.jcr.Session;
032
033import org.junit.Before;
034import org.junit.Test;
035import org.mockito.Mock;
036
037import com.hp.hpl.jena.rdf.model.Resource;
038
039public class DefaultGraphSubjectsTest {
040
041    private DefaultGraphSubjects testObj;
042
043    @Mock
044    Node mockNode;
045
046    @Mock
047    Resource mockSubject;
048
049    @Mock
050    Session mockSession;
051
052    @Before
053    public void setUp() {
054        initMocks(this);
055        testObj = new DefaultGraphSubjects(mockSession);
056    }
057
058    @Test
059    public void testGetGraphSubject() throws RepositoryException {
060        final String testPath = "/foo/bar";
061        final String expected = RESOURCE_NAMESPACE + testPath.substring(1);
062        when(mockNode.getPath()).thenReturn(testPath);
063        Resource actual = testObj.getGraphSubject(mockNode);
064        assertEquals(expected, actual.getURI());
065        when(mockNode.getPath()).thenReturn(testPath + "/jcr:content");
066        actual = testObj.getGraphSubject(mockNode);
067        assertEquals(expected + "/fcr:content", actual.getURI());
068    }
069
070    @Test
071    public void testGetNodeFromGraphSubject() throws RepositoryException {
072        final String expected = "/foo/bar";
073        when(mockSession.nodeExists(expected)).thenReturn(true);
074        when(mockSession.nodeExists(expected + "/bad")).thenReturn(false);
075        when(mockSession.getNode(expected)).thenReturn(mockNode);
076        // test a good subject
077        when(mockSubject.getURI()).thenReturn(RESOURCE_NAMESPACE + expected.substring(1));
078        when(mockSubject.isURIResource()).thenReturn(true);
079        Node actual = testObj.getNodeFromGraphSubject(mockSubject);
080        assertEquals(mockNode, actual);
081        // test a bad subject
082        when(mockSubject.getURI()).thenReturn(
083                "info:fedora2" + expected + "/bad");
084        actual = testObj.getNodeFromGraphSubject(mockSubject);
085        assertEquals("Somehow got a Node from a bad RDF subject!", null, actual);
086        // test a non-existent path
087        when(mockSubject.getURI()).thenReturn(RESOURCE_NAMESPACE + expected.substring(1) + "/bad");
088        actual = testObj.getNodeFromGraphSubject(mockSubject);
089        assertEquals("Somehow got a Node from a non-existent RDF subject!",
090                null, actual);
091        // test a fcr:content path
092        when(mockSubject.getURI()).thenReturn(RESOURCE_NAMESPACE + expected.substring(1) + "/fcr:content");
093        when(mockSession.nodeExists(expected + "/jcr:content")).thenReturn(true);
094        actual = testObj.getNodeFromGraphSubject(mockSubject);
095        verify(mockSession).getNode(expected + "/jcr:content");
096    }
097
098    @Test
099    public void testGetPathFromGraphSubject() throws RepositoryException {
100        final String expected = "/foo/bar";
101        // test a good subject
102        when(mockSubject.getURI())
103            .thenReturn(RESOURCE_NAMESPACE + expected.substring(1));
104        when(mockSubject.isURIResource()).thenReturn(true);
105        String actual = testObj.getPathFromGraphSubject(mockSubject);
106        assertEquals(expected, actual);
107        // test a bad subject
108        when(mockSubject.getURI()).thenReturn(
109                                                 "info:fedora2/" + expected.substring(1) + "/bad");
110        actual = testObj.getPathFromGraphSubject(mockSubject);
111        assertNull(actual);
112        // test a fcr:content path
113        when(mockSubject.getURI()).thenReturn(RESOURCE_NAMESPACE + expected.substring(1) + "/fcr:content");
114        actual = testObj.getPathFromGraphSubject(mockSubject);
115        assertEquals(expected + "/jcr:content", actual);
116    }
117
118    @Test
119    public void testIsFedoraGraphSubject() {
120        when(mockSubject.getURI()).thenReturn(RESOURCE_NAMESPACE + "foo");
121        when(mockSubject.isURIResource()).thenReturn(true);
122        boolean actual = testObj.isFedoraGraphSubject(mockSubject);
123        assertEquals(true, actual);
124        when(mockSubject.getURI()).thenReturn("http://fedora/foo");
125        actual = testObj.isFedoraGraphSubject(mockSubject);
126        assertEquals(false, actual);
127    }
128
129    @Test(expected = NullPointerException.class)
130    public void testIsFedoraGraphSubjectWithNull() {
131        testObj.isFedoraGraphSubject(null);
132    }
133
134    @Test
135    public void testIsFedoraGraphSubjectWithBlankNode() {
136        final Boolean actual = testObj.isFedoraGraphSubject(createResource());
137        assertFalse("Misrecognized an RDF blank node as a Fedora resource!",
138                actual);
139    }
140
141    @Test
142    public void testGetContext() {
143        final Resource context = testObj.getContext();
144        assertTrue("Context was returned other than as an RDF blank node!",
145                context.isAnon());
146    }
147
148}
