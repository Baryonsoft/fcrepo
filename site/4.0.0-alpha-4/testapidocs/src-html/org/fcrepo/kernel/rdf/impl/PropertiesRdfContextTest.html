054public class PropertiesRdfContextTest {
055
056    @Test
057    public void testForLowLevelStorageTriples() throws RepositoryException,
058                                               IOException {
059        final Model results =
060            new PropertiesRdfContext(mockNode, mockGraphSubjects,
061                    mockLowLevelStorageService).asModel();
062        logRdf("Retrieved RDF for testForLowLevelStorageTriples():", results);
063        assertTrue("Didn't find triple showing node has content!", results
064                .contains(mockSubject, HAS_CONTENT, mockContentSubject));
065        assertTrue("Didn't find triple showing content has node!", results
066                .contains(mockContentSubject, IS_CONTENT_OF, mockSubject));
067        assertTrue("Didn't find triple showing content has location!", results
068                .contains(mockContentSubject, HAS_CONTENT_LOCATION,
069                        createResource(MOCK_EXTERNAL_IDENTIFIER)));
070    }
071
072    @Before
073    public void setUp() throws RepositoryException {
074        initMocks(this);
075        when(mockNode.getSession()).thenReturn(mockSession);
076        when(mockContentNode.getSession()).thenReturn(mockSession);
077        when(mockSession.getRepository()).thenReturn(mockRepository);
078        when(mockSession.getWorkspace()).thenReturn(mockWorkspace);
079        when(mockWorkspace.getNamespaceRegistry()).thenReturn(mockNamespaceRegistry);
080        when(mockNamespaceRegistry.getURI("jcr")).thenReturn(JCR_NAMESPACE);
081        when(mockNode.hasNode(JCR_CONTENT)).thenReturn(true);
082        when(mockNode.getNode(JCR_CONTENT)).thenReturn(mockContentNode);
083        when(mockNode.hasProperties()).thenReturn(false);
084        when(mockNode.getMixinNodeTypes()).thenReturn(new NodeType[] {});
085        when(mockContentNode.getMixinNodeTypes()).thenReturn(new NodeType[] {});
086        when(mockContentNode.hasProperties()).thenReturn(false);
087        when(
088                mockLowLevelStorageService
089                        .getLowLevelCacheEntries(mockContentNode)).thenReturn(
090                ImmutableSet.of(mockLowLevelCacheEntry));
091        when(mockLowLevelCacheEntry.getExternalIdentifier()).thenReturn(
092                MOCK_EXTERNAL_IDENTIFIER);
093        when(mockGraphSubjects.getGraphSubject(mockNode)).thenReturn(
094                mockSubject);
095        when(mockGraphSubjects.getGraphSubject(mockContentNode)).thenReturn(
096                mockContentSubject);
097        when(mockNode.getPrimaryNodeType()).thenReturn(mockNodeType);
098        when(mockContentNode.getPrimaryNodeType()).thenReturn(mockNodeType);
099        when(mockNodeType.getSupertypes()).thenReturn(new NodeType[] {mockNodeType});
100        when(mockNodeType.getName()).thenReturn(
101                 mockNodeTypePrefix + ":" + mockNodeName);
102
103        //when(mockNodeType.getName()).thenReturn("not:root");
104    }

111

145
146    private static void
147            logRdf(final String message, final Model model) throws IOException {
148        LOGGER.debug(message);
149        try (Writer w = new StringWriter()) {
150            model.write(w);
151            LOGGER.debug("\n" + w.toString());
152        }
153    }
154
155    private static final Logger LOGGER =
156        getLogger(PropertiesRdfContextTest.class);
