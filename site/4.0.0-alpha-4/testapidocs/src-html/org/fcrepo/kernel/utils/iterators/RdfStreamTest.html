255    @Test
256    public void testTransform() {
257
258        final String oneResult = "One result";
259
260        final String otherResult = "The other result";
261
262        final Function<Triple, String> f = new Function<Triple, String>() {
263
264            @Override
265            public String apply(final Triple t) {
266                return t.equals(triple) ? oneResult : otherResult;
267            }
268
269        };
270        when(mockIterator.hasNext()).thenReturn(true, true, true, false);
271        when(mockIterator.next()).thenReturn(triple1, triple2, triple);
272        final Iterator<String> testIterator = testStream.transform(f);
273        assertEquals(otherResult, testIterator.next());
274        assertEquals(otherResult, testIterator.next());
275        assertEquals(oneResult, testIterator.next());
276    }

281                new RuntimeException("Expected.")).thenReturn(true);
