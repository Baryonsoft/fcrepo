065    @Test
066    public void testRemovingNonExistentProperty() throws Exception {
067
068        when(mockNode.hasProperty(propertyShortName)).thenReturn(false);
069        testRemover = new RdfRemover(mockGraphSubjects, mockSession, testStream);
070        testRemover.operateOnProperty(descriptiveStmnt, mockNode);
071        verifyZeroInteractions(mockProperty);
072    }
073
074    @Test
075    public void testRemovingExistentProperty() throws Exception {
076
077        when(mockNode.hasProperty(propertyShortName)).thenReturn(true);
078        when(mockNode.getProperty(propertyShortName)).thenReturn(mockProperty);
079        when(mockProperty.isMultiple()).thenReturn(false);
080        when(mockProperty.getValue()).thenReturn(mockValue);
081        testRemover =
082            new RdfRemover(mockGraphSubjects, mockSession, testStream);
083        testRemover.operateOnProperty(descriptiveStmnt, mockNode);
084        verify(mockProperty).setValue((Value) null);
085    }
086
087    @Test
088    public void testRemovingExistentMixin() throws Exception {
089        testRemover =
090            new RdfRemover(mockGraphSubjects, mockSession, testStream);
091        testRemover.operateOnMixin(mixinStmnt.getObject().asResource(),
092                mockNode);
093        verify(mockNode).removeMixin(mixinShortName);
094    }
095
096    @Test
097    public void testRemovingNonExistentMixin() throws Exception {
098        doThrow(new NoSuchNodeTypeException("Expected.")).when(mockNode)
099                .removeMixin(mixinShortName);
100        testRemover =
101            new RdfRemover(mockGraphSubjects, mockSession, testStream);
102        testRemover.operateOnMixin(mixinStmnt.getObject().asResource(),
103                mockNode);
104        verify(mockNode).removeMixin(mixinShortName);
105    }
106
107    @Test
108    public void testRemovingMixinThatCannotExistInRepo() throws Exception {
109        when(mockNodeTypeManager.hasNodeType(mixinShortName)).thenReturn(false);
110        testRemover =
111            new RdfRemover(mockGraphSubjects, mockSession, testStream);
112        testRemover.operateOnMixin(mixinStmnt.getObject().asResource(),
113                mockNode);
114        verify(mockNode, never()).removeMixin(mixinShortName);
115    }

117    @Before
118    public void setUp() throws RepositoryException {
119        initMocks(this);
120        when(mockNode.getSession()).thenReturn(mockSession);
121        when(mockNode.getName()).thenReturn("mockNode");
122        when(mockNode.getPath()).thenReturn("/mockNode");
123        when(mockSession.getWorkspace()).thenReturn(mockWorkspace);
124        when(mockSession.getValueFactory()).thenReturn(mockValueFactory);
125        when(
126                mockSession
127                        .getNamespacePrefix(getJcrNamespaceForRDFNamespace(type
128                                .getNameSpace()))).thenReturn("rdf");
129        when(mockValueFactory.createValue(description, STRING)).thenReturn(mockValue);
130        when(mockWorkspace.getNamespaceRegistry()).thenReturn(
131                mockNamespaceRegistry);
132        when(mockNamespaceRegistry.getURI(propertyNamespacePrefix)).thenReturn(
133                propertyNamespaceUri);
134        when(mockNamespaceRegistry.getURI("rdf")).thenReturn(
135                type.getNameSpace());
136        when(mockNamespaceRegistry.isRegisteredUri(propertyNamespaceUri))
137                .thenReturn(true);
138        when(mockNamespaceRegistry.isRegisteredUri(type.getNameSpace()))
139        .thenReturn(true);
140        when(mockNamespaceRegistry.getPrefix(propertyNamespaceUri)).thenReturn(
141                propertyNamespacePrefix);
142        when(mockNamespaceRegistry.getPrefix(type.getNameSpace())).thenReturn(
143                "rdf");
144        when(mockWorkspace.getNodeTypeManager())
145                .thenReturn(mockNodeTypeManager);
146        when(mockNodeTypeManager.getNodeType(FEDORA_RESOURCE)).thenReturn(
147                mockNodeType);
148        when(mockNodeTypeManager.hasNodeType(mixinShortName)).thenReturn(true);
149        when(mockNode.getPrimaryNodeType()).thenReturn(mockNodeType);
150        when(mockNode.getMixinNodeTypes()).thenReturn(new NodeType[] {});
151        when(mockNode.canAddMixin(mixinShortName)).thenReturn(true);
152        when(mockNodeType.getPropertyDefinitions()).thenReturn(
153                new PropertyDefinition[] {mockPropertyDefinition});
154        when(mockPropertyDefinition.isMultiple()).thenReturn(false);
155        when(mockPropertyDefinition.getName()).thenReturn(propertyShortName);
156        when(mockPropertyDefinition.getRequiredType()).thenReturn(STRING);
157        when(mockGraphSubjects.getGraphSubject(mockNode)).thenReturn(
158                mockNodeSubject);
159        when(mockTriples.hasNext()).thenReturn(true, true, false);
160        when(mockTriples.next()).thenReturn(descriptiveTriple, mixinTriple);
161        testStream = new RdfStream(mockTriples);
162        testStream.namespaces(mockNamespaceMap);
163    }
