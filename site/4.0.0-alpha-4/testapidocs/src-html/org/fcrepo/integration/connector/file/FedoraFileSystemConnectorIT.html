102    @Test
103    public void testGetFederatedObject() throws RepositoryException {
104        final Session session = repo.login();
105
106        final FedoraObject object = objectService.getObject(session, testFile);
107        assertNotNull(object);
108
109        final Node node = object.getNode();
110        final NodeType[] mixins = node.getMixinNodeTypes();
111        assertEquals(2, mixins.length);
112
113        boolean found = false;
114        for (final NodeType nodeType : mixins) {
115            if (nodeType.getName().equals(FEDORA_DATASTREAM)) {
116                found = true;
117            }
118        }
119        assertTrue("Mixin not found: " + FEDORA_DATASTREAM, found);
120
121        session.save();
122        session.logout();
123    }

125    @Test
126    public void testGetFederatedContent() throws RepositoryException {
127        final Session session = repo.login();
128
129        final Node node = datastreamService.getDatastreamNode(session, testFile + "/jcr:content");
130        assertNotNull(node);
131
132        final NodeType[] mixins = node.getMixinNodeTypes();
133        assertEquals(2, mixins.length);
134
135        boolean found = false;
136        for (final NodeType nodeType : mixins) {
137            if (nodeType.getName().equals(FEDORA_BINARY)) {
138                found = true;
139            }
140        }
141        assertTrue("Mixin not found: " + FEDORA_BINARY, found);
142
143        final Property size = node.getProperty(CONTENT_SIZE);
144
145        final File file = new File(testFile.replace("/federated", "target/test-classes"));
146        assertTrue(file.getAbsolutePath(), file.exists());
147        assertEquals(file.length(), size.getLong());
148
149        session.save();
150        session.logout();
151    }

153    @Test
154    public void testWriteProperty() throws RepositoryException {
155        final Session session = repo.login();
156
157        final FedoraResource object = nodeService.getObject(session, testFile);
158        assertNotNull(object);
159
160        final String sparql = "PREFIX fedora: <http://fedora.info/definitions/v4/rest-api#> " +
161                "INSERT DATA { " +
162                "<info:fedora" + testFile + "> " +
163                "fedora:name " +
164                "'some-test-name' }";
165
166        // Write the properties
167        object.updatePropertiesDataset(new DefaultGraphSubjects(session),
168                sparql);
169
170        // Verify
171        final Property property = object.getNode().getProperty("fedora:name");
172        assertNotNull(property);
173        assertEquals("some-test-name", property.getValues()[0].toString());
174
175        session.save();
176        session.logout();
177    }

179    @Test
180    public void testRemoveProperty() throws RepositoryException {
181        Session session = repo.login();
182
183        final FedoraResource object = nodeService.getObject(session, testFile);
184        assertNotNull(object);
185
186        final String sparql = "PREFIX fedora: <http://fedora.info/definitions/v4/rest-api#> " +
187                "INSERT DATA { " +
188                "<info:fedora" + testFile + "> " +
189                "fedora:remove " +
190                "'some-property-to-remove' }";
191
192        // Write the properties
193        final GraphSubjects graphSubjects = new DefaultGraphSubjects(session);
194        object.updatePropertiesDataset(graphSubjects, sparql);
195
196        // Verify property exists
197        final Property property = object.getNode().getProperty("fedora:remove");
198        assertNotNull(property);
199        assertEquals("some-property-to-remove", property.getValues()[0].getString());
200
201        final String sparqlRemove = "PREFIX fedora: <http://fedora.info/definitions/v4/rest-api#> " +
202                "DELETE {" +
203                "  <info:fedora" + testFile + "> fedora:remove ?s " +
204                "} WHERE { " +
205                "  <info:fedora" + testFile + "> fedora:remove ?s" +
206                "}";
207
208        // Remove the properties
209        final GraphSubjects graphSubjectsRemove = new DefaultGraphSubjects(session);
210        object.updatePropertiesDataset(graphSubjectsRemove, sparqlRemove);
211
212        // Persist the object (although the propery will be removed from memory without this.)
213        session.save();
214
215        // Verify
216        boolean thrown = false;
217        try{
218            object.getNode().getProperty("fedora:remove");
219        } catch (PathNotFoundException e){
220            thrown = true;
221        }
222        assertTrue("Exception expected - property should be missing", thrown);
223
224        session.logout();
225    }

227    @Test
228    public void testFixity() throws RepositoryException, IOException, NoSuchAlgorithmException {
229        final Session session = repo.login();
230
231        final Node node = datastreamService.getDatastreamNode(session, testFile + "/jcr:content");
232        assertNotNull(node);
233
234        final File file =
235            new File(getProperty(PROP_TEST_DIR), testFile.replace("federated",
236                    ""));
237        final byte[] hash = getHash(SHA_1, file);
238
239        final URI calculatedChecksum = asURI(SHA_1.toString(), hash);
240
241        final Collection<FixityResult> results =
242            datastreamService
243                    .getFixity(node, calculatedChecksum, file.length());
244        assertNotNull(results);
245
246        assertFalse("Found no results!", results.isEmpty());
247
248        final Iterator<FixityResult> resultIterator = results.iterator();
249        while (resultIterator.hasNext()) {
250            final FixityResult result = resultIterator.next();
251            assertTrue(result.isSuccess());
252        }
253
254        session.save();
255        session.logout();
256    }
