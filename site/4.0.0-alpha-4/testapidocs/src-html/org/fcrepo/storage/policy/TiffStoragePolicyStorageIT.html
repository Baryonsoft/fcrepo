050public class TiffStoragePolicyStorageIT {
051
052    protected Logger logger;
053
054    static private Repository repo;
055
056    private DatastreamService datastreamService;
057
058    private ObjectService objectService;
059
060    private LowLevelStorageService lowLevelService;
061
062    private StoragePolicyDecisionPointImpl pdp;
063
064    GetBinaryKey getBinaryKey = new GetBinaryKey();
065
066    @Before
067    public void setLogger() {
068        logger = LoggerFactory.getLogger(this.getClass());
069    }
070
071    @Before
072    public void setRepository() throws RepositoryException {
073
074        final URL config =
075                this.getClass().getClassLoader().getResource(
076                    "repository.json");
077
078        final Map<String, String> params = singletonMap(JcrRepositoryFactory.URL,
079                                              config.toString());
080        repo = new JcrRepositoryFactory().getRepository(params);
081
082        pdp = new StoragePolicyDecisionPointImpl();
083        pdp.addPolicy(new MimeTypeStoragePolicy("image/tiff", "tiff-store"));
084
085        datastreamService = new DatastreamServiceImpl();
086        datastreamService.setRepository(repo);
087        ((DatastreamServiceImpl) datastreamService).setStoragePolicyDecisionPoint(pdp);
088        objectService = new ObjectServiceImpl();
089        objectService.setRepository(repo);
090        lowLevelService = new LowLevelStorageService();
091        lowLevelService.setRepository(repo);
092    }
093
094    @Test
095    //@Ignore
096    public void testPolicyDrivenStorage() throws Exception {
097        ByteArrayInputStream data;
098        final Session session = repo.login();
099
100        objectService.createObject(session, "/testCompositeObject");
101
102        data = new ByteArrayInputStream("9876543219876543210987654321098765432109876543210987654321098765432109876543210987654321009876543210".getBytes());
103        datastreamService.createDatastreamNode(session,
104                                               "/testCompositeObject/content",
105                                               "application/octet-stream",
106                                                null,
107                                               data);
108        data = new ByteArrayInputStream("87acec17cd9dcd20a716cc2cf67417b71c8a701687acec17cd9dcd20a716cc2cf67417b71c8a701687acec17cd9dcd20a716cc2cf67417b71c8a701687acec17cd9dcd20a716cc2cf67417b71c8a701687acec17cd9dcd20a716cc2cf67417b71c8a701687acec17cd9dcd20a716cc2cf67417b71c8a701687acec17cd9dcd20a716cc2cf67417b71c8a7016".getBytes());
109        datastreamService
110            .createDatastreamNode(session,
111                                  "/testCompositeObject/tiffContent",
112                                  "image/tiff",
113                                  null,
114                                  data);
115
116        session.save();
117
118        final Node node = session.getNode("/testCompositeObject/content");
119
120        final BinaryKey key =
121            getBinaryKey.apply(node.getNode(JcrConstants.JCR_CONTENT)
122                               .getProperty(JcrConstants.JCR_DATA));
123
124        logger.info("content key: {}", key);
125
126        final Node tiffNode =
127            session.getNode("/testCompositeObject/tiffContent");
128
129        final BinaryKey tiffKey =
130            getBinaryKey.apply(tiffNode.getNode(JcrConstants.JCR_CONTENT)
131                               .getProperty(JcrConstants.JCR_DATA));
132
133        logger.info("tiff key: {}", tiffKey);
134
135        final Set<LowLevelCacheEntry> lowLevelContentEntries =
136            lowLevelService.getLowLevelCacheEntries(key);
137
138        final Iterator<LowLevelCacheEntry> iterator =
139            lowLevelContentEntries.iterator();
140
141        assertEquals(1, lowLevelContentEntries.size());
142
143        final Set<LowLevelCacheEntry> lowLevelTiffEntries =
144            lowLevelService.getLowLevelCacheEntries(tiffKey);
145
146        final Iterator<LowLevelCacheEntry> tiffIterator =
147            lowLevelTiffEntries.iterator();
148
149        assertEquals(1, lowLevelTiffEntries.size());
150
151        final LowLevelCacheEntry e = iterator.next();
152
153        assertThat(e.getExternalIdentifier(),
154                   containsString("TransientBinaryStore"));
155
156        final LowLevelCacheEntry tiffEntry = tiffIterator.next();
157        assertThat(tiffEntry.getExternalIdentifier(),
158                   containsString("FileSystemBinaryStore"));
159    }
