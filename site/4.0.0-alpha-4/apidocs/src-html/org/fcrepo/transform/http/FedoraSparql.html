082/**
083 * Primitive SPARQL JAX-RS endpoint
084 *
085 * @author cabeer
086 */
087@Component
088@Scope("prototype")
089@Path("/fcr:sparql")
090public class FedoraSparql extends AbstractResource {
091
092    @InjectedSession
093    protected Session session;
094
095    private static final Logger LOGGER = getLogger(FedoraSparql.class);
096
097
098    /**
099     * A stub method so we can return a text/html representation using
100     * the right template.
101     *
102     * @return
103     * @throws RepositoryException
104     */
105    @GET
106    @Timed
107    @Produces({TEXT_HTML})
108    public Response sparqlQueryForm() throws IOException, RepositoryException {
109
110        final Properties properties = new Properties();
111        final URL propertiesUrl =
112            getClass().getResource(velocityPropertiesLocation);
113        final VelocityEngine velocity = new VelocityEngine();
114        LOGGER.debug("Using Velocity configuration from {}", propertiesUrl);
115        try (final InputStream propertiesStream = propertiesUrl.openStream()) {
116            properties.load(propertiesStream);
117        }
118        velocity.init(properties);
119        final Template template =
120            velocity.getTemplate(templatesLocation + "/search-sparql"
121                    + templateFilenameExtension);
122        final org.apache.velocity.context.Context context =
123            new VelocityContext();
124        context.put("uriInfo", uriInfo);
125        context.put("model", new NamespaceRdfContext(session).asModel());
126        context.put("helpers", ViewHelpers.getInstance());
127
128        final StreamingOutput stream = new StreamingOutput() {
129            @Override
130            public void write(final OutputStream output) throws IOException {
131
132                final Writer outWriter = new OutputStreamWriter(output);
133                template.merge(context, outWriter);
134                outWriter.flush();
135            }
136        };
137
138        return ok(stream).build();
139    }
140
141    /**
142     * Execute a SPARQL query against the JCR index
143     * @param requestBodyStream
144     * @param uriInfo
145     * @return
146     * @throws IOException
147     * @throws RepositoryException
148     */
149    @POST
150    @Consumes({contentTypeSPARQLQuery})
151    @Produces({contentTypeTextTSV, contentTypeTextCSV, contentTypeSSE,
152            contentTypeTextPlain, contentTypeResultsJSON,
153            contentTypeResultsXML, contentTypeResultsBIO, contentTypeTurtle,
154            contentTypeN3, contentTypeNTriples, contentTypeRDFXML})
155    public Response runSparqlQuery(final InputStream requestBodyStream,
156        @Context final Request request, @Context final UriInfo uriInfo)
157        throws IOException, RepositoryException {
158
159        final GraphSubjects graphSubjects = new HttpGraphSubjects(session, FedoraNodes.class, uriInfo);
160
161        final Variant bestPossibleResponse =
162            request.selectVariant(POSSIBLE_SPARQL_RDF_VARIANTS);
163
164        final String sparqlQuery = IOUtils.toString(requestBodyStream);
165
166        LOGGER.trace("Running SPARQL query: {}", sparqlQuery);
167
168        final JQLConverter jqlConverter = new JQLConverter(session, graphSubjects, sparqlQuery);
169
170        LOGGER.trace("Converted to JQL query: {}", jqlConverter.getStatement());
171
172        final ResultSet resultSet = jqlConverter.execute();
173
174        final ResultSetStreamingOutput streamingOutput =
175            new ResultSetStreamingOutput(resultSet, bestPossibleResponse
176                    .getMediaType());
177
178        addCallback(streamingOutput, new LogoutCallback(session));
179
180        return ok(streamingOutput).build();
181    }
