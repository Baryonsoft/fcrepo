029import javax.ws.rs.core.UriInfo;
030import java.net.URI;
031import java.util.Map;

033import static com.hp.hpl.jena.rdf.model.ResourceFactory.createResource;
034import static java.util.Collections.singletonMap;
035import static org.fcrepo.jcr.FedoraJcrTypes.FCR_CONTENT;
036import static org.fcrepo.jcr.FedoraJcrTypes.FCR_VERSIONS;
037import static org.modeshape.jcr.api.JcrConstants.JCR_CONTENT;
038import static org.slf4j.LoggerFactory.getLogger;

040/**
041 * An extension of HttpGraphSubjects that includes mapping from JCR frozen nodes
042 * that represent versions of versionable nodes to a URI pattern that presents
043 * them as children of the original node.  The URL patterns are as follows:
044 *
045 * If node "node" is versionable, and has the unversionable subgraph
046 * "path/to/versioned/part" and a version snapshot exists with the label
047 * "label", the path to access the frozen nodes
048 * is:
049 * " ../node/fcr:versions/label/path/to/versioned/part".
050 *
051 * The "label" referred to here is either the cannonical identifier assigned
052 * to the version or the user label optionally assinged at version creation
053 * time.  When this class translates JCR nodes to subjects, it will always
054 * return that cannonical identifier, but when mapping from graph subjects to
055 * JCR nodes, either the user label or cannonical ID are supported.
056 *
057 * @author Mike Durbin
058 */
059public class VersionAwareHttpGraphSubjects extends HttpGraphSubjects {
060
061    private static final Logger LOGGER = getLogger(VersionAwareHttpGraphSubjects.class);
062
063    private Session internalSession;
064
065    /**
066     * Creates a GraphSubject implementation that has all of the functionality
067     * of HttpGraphSubjects with the additional mapping versioned frozen nodes
068     * to URIs that appear in the hierarchy of the original node.
069     *
070     * This mapping can be heavyweight in that it has to do version lookups and
071     * other expensive operations for nodes that appear to be historic versions.
072     *
073     * @param session the current session (may contain transaction information)
074     * @param internalSession a session with read access to all content
075     * @param relativeTo a class whose path annotation will be used as the
076     *                   base for all relative paths
077     * @param uris a UriInfo implementation with information about the request
078     *             URI.
079     */
080    public VersionAwareHttpGraphSubjects(final Session session,
081        final Session internalSession, final Class<?> relativeTo, final UriInfo uris) {
082        super(session, relativeTo, uris);
083        this.internalSession = internalSession;
084    }
085
086    @Override
087    public String getPathFromGraphSubject(@NotNull final String subjectUri) throws RepositoryException {
088        if (subjectUri.matches("^.*\\Q" + FCR_VERSIONS + "/\\E.+$")) {
089            final Node node = getNodeFromGraphSubjectForVersionNode(subjectUri);
090            if (node == null) {
091                return null;
092            }
093            return node.getPath();
094        }
095        return super.getPathFromGraphSubject(subjectUri);
096    }
097
098    @Override
099    public Resource getGraphSubject(final String absPath) throws RepositoryException {
100        if (absPath.contains("jcr:versionStorage")) {
101            final Node probableFrozenNode = internalSession.getNode(absPath);
102            if (probableFrozenNode.getPrimaryNodeType().getName().equals("nt:frozenNode")) {
103                final URI result = nodesBuilder.buildFromMap(getPathMapForVersionNode(probableFrozenNode));
104                LOGGER.debug("Translated path {} into RDF subject {}", absPath, result);
105                return createResource(result.toString());
106            }
107            LOGGER.debug("{} was not a frozen node... no version-specific translation required", absPath);
108        }
109        return super.getGraphSubject(absPath);
110    }
111
112    @Override
113    public Resource getGraphSubject(final Node node) throws RepositoryException {
114        if (node.getPrimaryNodeType().getName().equals("nt:frozenNode")) {
115            final URI result = nodesBuilder.buildFromMap(getPathMapForVersionNode(node));
116            LOGGER.debug("Translated node {} into RDF subject {}", node, result);
117            return createResource(result.toString());
118        }
119        return super.getGraphSubject(node);
120    }
121
122    /**
123     * For frozen nodes (which represent version snapshots) we translate the
124     * paths from their frozen storage URI to
125     * "../node/fcr:versions/label/path/to/versioned/part".
126     * @see #getNodeFromGraphSubjectForVersionNode
127     * @param frozenNode an nt:frozenNode whose fedora-specific URI path is to be
128     *             retrieved
129     * @return a map with a single entry with the key "path" and the value
130     *         equal to the fedora path for the version node.
131     * @throws NullPointerException if session is null
132     */
133    private Map<String,String> getPathMapForVersionNode(final Node frozenNode)
134        throws RepositoryException {
135        Node versionableFrozenNode = frozenNode;
136        Node versionableNode = internalSession.getNodeByIdentifier(
137                versionableFrozenNode.getProperty("jcr:frozenUuid").getString());
138        String pathWithinVersionable = "";
139        while (!versionableNode.isNodeType("mix:versionable")) {
140            LOGGER.debug("node {} is not versionable, checking parent...", versionableNode.getIdentifier());
141            pathWithinVersionable = "/" + getRelativePath(versionableNode,
142                    versionableNode.getParent()) + pathWithinVersionable;
143            versionableFrozenNode = versionableFrozenNode.getParent();
144            versionableNode = internalSession.getNodeByIdentifier(
145                    versionableFrozenNode.getProperty("jcr:frozenUuid").getString());
146        }
147
148        pathWithinVersionable = versionableFrozenNode.getIdentifier()
149                + (pathWithinVersionable.length() > 0 ? pathWithinVersionable : "");
150        final String pathToVersionable = versionableNode.getPath();
151        LOGGER.debug("frozen node found with id {}.", versionableFrozenNode.getIdentifier());
152        String path =  pathToVersionable + "/" + FCR_VERSIONS + "/" + pathWithinVersionable;
153        if (path.endsWith(JCR_CONTENT)) {
154            path = path.replace(JCR_CONTENT, FCR_CONTENT);
155        }
156        return singletonMap("path", path.startsWith("/") ? path.substring(1) : path);
157    }
158
159    private static String getRelativePath(final Node node, final Node ancestor) throws RepositoryException {
160        return node.getPath().substring(ancestor.getPath().length() + 1);
161    }
162
163    /**
164     * Gets the frozen node in the system for the version described as the
165     * passed subject.
166     * @see #getPathMapForVersionNode
167     * @param subjectUri a subject containing the fedora identifier as a URI
168     * @return the node that represents the version snapshot of the subject.
169     * @throws RepositoryException
170     * @throws NullPointerException if session is null
171     */
172    public Node getNodeFromGraphSubjectForVersionNode(final String subjectUri)
173        throws RepositoryException {
174
175        /*
176         * Get the part of the URI after fcr:versions/ that represents the label
177         * optionally followed by a path into the versioned subgraph).
178         */
179        final String labelAndPath
180            = subjectUri.substring(subjectUri.indexOf(FCR_VERSIONS + "/") + FCR_VERSIONS.length() + 1);
181
182        if (labelAndPath.contains("/")) {
183            /*
184             * The subjectUri references a node within a version snapshot
185             * (identified by label) of a subgraph.
186             */
187            final int firstSlash = labelAndPath.indexOf('/');
188            final String label = labelAndPath.substring(0, firstSlash);
189            final String pathIntoVersionedGraph = labelAndPath.substring(firstSlash + 1);
190            LOGGER.trace("subjectUri={}, label={}, pathIntoVersionedGraph={}", subjectUri, label,
191                    pathIntoVersionedGraph);
192            final Node versionedFrozenNode = getFrozenNodeByLabel(subjectUri, label);
193            if (versionedFrozenNode == null) {
194                // there is no version with the given label!
195                return null;
196            }
197            return internalSession.getNode(versionedFrozenNode.getPath() + "/"
198                    + pathIntoVersionedGraph.replace(FCR_CONTENT, JCR_CONTENT));
199        }
200        /**
201         * The subjectUri references a version of a verisonable node identified
202         * by label (the root of the versioned subgraph).
203         */
204        final String label = labelAndPath;
205        LOGGER.trace("subjectUri={}, label={}", subjectUri, label);
206        return getFrozenNodeByLabel(subjectUri, label);
207    }
208
209    /**
210     * A private helper method that tries to look up frozen node for the given subject
211     * by a label.  That label may either be one that was assigned at creation time
212     * (and is a version label in the JCR sense) or a system assigned identifier that
213     * was used for versions created without a label.  The current implementation
214     * uses the JCR UUID for the frozen node as the system-assigned label.
215     */
216    private Node getFrozenNodeByLabel(final String subjectUri, final String label) throws RepositoryException {
217        final String baseResourcePath
218            = getPathFromGraphSubject(subjectUri.substring(0, subjectUri.indexOf("/" + FCR_VERSIONS)));
219        try {
220            final Node frozenNode = internalSession.getNodeByIdentifier(label);
221
222            /*
223             * We found a node whose identifier is the "label" for the version.  Now
224             * we must do due dilligence to make sure it's a frozen node representing
225             * a version of the subject node.
226             */
227            final Property p = frozenNode.getProperty("jcr:frozenUuid");
228            if (p != null) {
229                final Node subjectNode = internalSession.getNode(baseResourcePath);
230                if (p.getString().equals(subjectNode.getIdentifier())) {
231                    return frozenNode;
232                }
233            }
234            /*
235             * Though a node with an id of the label was found, it wasn't the
236             * node we were looking for, so fall through and look for a labeled
237             * node.
238             */
239        } catch (final ItemNotFoundException ex) {
240            /*
241             * the label wasn't a uuid of a frozen node but
242             * instead possibly a version label.
243             */
244        }
245
246        final VersionHistory hist =
247            internalSession.getWorkspace().getVersionManager().getVersionHistory(baseResourcePath);
248        if (hist.hasVersionLabel(label)) {
249            LOGGER.debug("Found version for {} by label {}.",  subjectUri, label);
250            return hist.getVersionByLabel(label).getFrozenNode();
251        }
252        LOGGER.warn("Unknown version {} with label or uuid {}!", subjectUri, label);
253        return null;
254    }
