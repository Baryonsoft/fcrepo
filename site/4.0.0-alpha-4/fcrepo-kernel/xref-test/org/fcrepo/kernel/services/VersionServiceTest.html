60      @Before
61      public void setup() throws Exception {
62          txService = new TransactionServiceImpl();
63          initMocks(this);
64          testObj = new VersionServiceImpl();
65  
66          testObj.setTxService(txService);
67          txService.setVersionService(testObj);
68  
69          s = mock(Session.class);
70          mockWorkspace = mock(Workspace.class);
71          when(mockWorkspace.getName()).thenReturn("default");
72          when(s.getWorkspace()).thenReturn(mockWorkspace);
73          mockVM = mock(VersionManager.class);
74          when(mockWorkspace.getVersionManager()).thenReturn(mockVM);
75  
76          // add a node that's versioned (but not auto-versioned)
77          final Node versionedNode = mock(Node.class);
78          when(versionedNode.getPath()).thenReturn("/example-versioned");
79          when(versionedNode.getSession()).thenReturn(s);
80          when(versionedNode.isNodeType(VersionServiceImpl.VERSIONABLE))
81                  .thenReturn(true);
82          when(s.getNode("/example-versioned")).thenReturn(versionedNode);
83  
84          // add a node that's autoversioned
85          final Node autoversionedNode = mock(Node.class);
86          when(autoversionedNode.getPath()).thenReturn("/example-auto-versioned");
87          when(autoversionedNode.getSession()).thenReturn(s);
88          when(autoversionedNode.isNodeType(VersionServiceImpl.VERSIONABLE))
89                  .thenReturn(true);
90          when(s.getNode("/example-auto-versioned")).thenReturn(autoversionedNode);
91          final Property autoVersionProperty = mock(Property.class);
92          final Value autoVersionValue = mock(Value.class);
93          when(autoVersionValue.getString()).thenReturn(
94                  VersionServiceImpl.AUTO_VERSION);
95          when(autoVersionProperty.isMultiple()).thenReturn(true);
96          when(autoVersionProperty.getValues()).thenReturn(new Value[] { autoVersionValue });
97          when(autoversionedNode.hasProperty(VersionServiceImpl.VERSION_POLICY))
98                  .thenReturn(true);
99          when(autoversionedNode.getProperty(VersionServiceImpl.VERSION_POLICY))
100                 .thenReturn(autoVersionProperty);
101 
102 
103         // add a node that's unversioned
104         final Node unversionedNode = mock(Node.class);
105         when(unversionedNode.getPath()).thenReturn("/example-unversioned");
106         when(unversionedNode.getSession()).thenReturn(s);
107         when(unversionedNode.isNodeType(VersionServiceImpl.VERSIONABLE))
108                 .thenReturn(false);
109         when(s.getNode("/example-unversioned")).thenReturn(unversionedNode);
110     }

139     @Test
140     public void testDeferredCheckpointVersioned() throws Exception {
141         // start a transaction
142         final Transaction t = txService.beginTransaction(s, USER_NAME);
143         s = t.getSession();
144         when(s.getNamespaceURI(TransactionServiceImpl.FCREPO4_TX_ID))
145                 .thenReturn(t.getId());
146 
147         assertNotNull("Transaction must have started!",
148                 txService.getTransaction(
149                         s.getNode("/example-auto-versioned").getSession()));
150 
151         // request a version be created
152         testObj.nodeUpdated(s, "/example-versioned");
153 
154         // ensure that no version was created (because the transaction is still open)
155         verify(mockVM, never()).checkpoint("/example-versioned");
156 
157         // close the transaction
158         txService.commit(t.getId());
159 
160         // ensure that no version was made because none was explicitly requested
161         verify(mockVM, never()).checkpoint("/example-versioned");
162     }
