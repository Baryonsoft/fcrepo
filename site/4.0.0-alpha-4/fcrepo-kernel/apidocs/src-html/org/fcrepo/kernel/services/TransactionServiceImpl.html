020package org.fcrepo.kernel.services;
021
022import static java.lang.System.currentTimeMillis;
023import static org.slf4j.LoggerFactory.getLogger;
024
025import java.util.Iterator;
026import java.util.Map;
027import java.util.Map.Entry;
028import java.util.concurrent.ConcurrentHashMap;
029
030import javax.jcr.RepositoryException;
031import javax.jcr.Session;
032
033import org.fcrepo.kernel.Transaction;
034import org.fcrepo.kernel.TransactionImpl;
035import org.fcrepo.kernel.TxSession;
036import org.fcrepo.kernel.exception.TransactionMissingException;
037import org.slf4j.Logger;
038import org.springframework.beans.factory.annotation.Autowired;
039import org.springframework.scheduling.annotation.Scheduled;
040import org.springframework.stereotype.Component;
041
042/**
043 * This is part of the strawman implementation for Fedora transactions This
044 * service implements a simple {@link Transaction} service which is able to
045 * create/commit/rollback {@link Transaction} objects A {@link Scheduled}
046 * annotation is used for removing timed out Transactions
047 *
048 * @author frank asseg
049 */
050@Component
051public class TransactionServiceImpl extends AbstractService implements TransactionService {
052
053    private static final Logger LOGGER = getLogger(TransactionService.class);
054
055    /**
056     * A key for looking up the transaction id in a session key-value pair
057     */
058    static final String FCREPO4_TX_ID = "fcrepo4.tx.id";
059
060    @Autowired
061    private VersionService versionService;
062
063    /**
064     * TODO since transactions have to be available on all nodes, they have to
065     * be either persisted or written to a distributed map or sth, not just this
066     * plain hashmap that follows
067     */
068    private static Map<String, Transaction> transactions = new ConcurrentHashMap<>();
069
070    public static final long REAP_INTERVAL = 1000;
071
072    /**
073     * Every REAP_INTERVAL milliseconds, check for expired transactions. If the
074     * tx is expired, roll it back and remove it from the registry.
075     */
076    /*
077     * (non-Javadoc)
078     * @see
079     * org.fcrepo.kernel.services.TransactionService#removeAndRollbackExpired()
080     */
081    @Override
082    @Scheduled(fixedRate = REAP_INTERVAL)
083    public void removeAndRollbackExpired() {
084        synchronized (transactions) {
085            final Iterator<Entry<String, Transaction>> txs =
086                    transactions.entrySet().iterator();
087            while (txs.hasNext()) {
088                final Transaction tx = txs.next().getValue();
089                if (tx.getExpires().getTime() <= currentTimeMillis()) {
090                    try {
091                        tx.rollback();
092                    } catch (final RepositoryException e) {
093                        LOGGER.error(
094                                "Got exception rolling back expired" +
095                                        " transaction {}: {}",
096                                        tx, e);
097                    }
098                    txs.remove();
099                }
100            }
101        }
102    }
103
104    /**
105     * Create a new Transaction and add it to the currently open ones
106     *
107     * @param sess The session to use for this Transaction
108     * @return the {@link Transaction}
109     */
110    @Override
111    public Transaction beginTransaction(final Session sess, final String userName)
112        throws RepositoryException {
113        final Transaction tx = new TransactionImpl(sess, userName);
114        final String txId = tx.getId();
115        transactions.put(txId, tx);
116        sess.setNamespacePrefix(FCREPO4_TX_ID, txId);
117        return tx;
118    }
119
120    /**
121     * Retrieve an open {@link Transaction}
122     *
123     * @param txid the Id of the {@link Transaction}
124     * @return the {@link Transaction}
125     */
126    @Override
127    public Transaction getTransaction(final String txid)
128        throws TransactionMissingException {
129
130        final Transaction tx = transactions.get(txid);
131
132        if (tx == null) {
133            throw new TransactionMissingException(
134                    "Transaction is not available");
135        }
136
137        return tx;
138    }
139
140    /**
141     * Get the current Transaction for a session
142     *
143     * @param session
144     * @return
145     * @throws TransactionMissingException
146     */
147    @Override
148    public Transaction getTransaction(final Session session)
149        throws TransactionMissingException {
150
151        final String txId = getCurrentTransactionId(session);
152
153        if (txId == null) {
154            throw new TransactionMissingException(
155                    "Transaction is not available");
156        }
157
158        return getTransaction(txId);
159    }
160
161    /**
162     * Get the current Transaction ID for a session
163     *
164     * @param session
165     * @return
166     */
167    public static String getCurrentTransactionId(final Session session) {
168        try {
169            if (session instanceof TxSession) {
170                return ((TxSession) session).getTxId();
171            }
172            return session.getNamespaceURI(FCREPO4_TX_ID);
173        } catch (final RepositoryException e) {
174            LOGGER.trace("Unable to retrieve current transaction ID from session", e);
175            return null;
176        }
177    }
178
179    /**
180     * Check if a Transaction exists
181     *
182     * @param txid the Id of the {@link Transaction}
183     * @return the {@link Transaction}
184     */
185    @Override
186    public boolean exists(final String txid) {
187        return transactions.containsKey(txid);
188    }
189
190    /**
191     * Commit a {@link Transaction} with the given id
192     *
193     * @param txid the id of the {@link Transaction}
194     * @throws RepositoryException
195     */
196    @Override
197    public Transaction commit(final String txid) throws RepositoryException {
198        final Transaction tx = transactions.remove(txid);
199        if (tx == null) {
200            throw new RepositoryException("Transaction with id " + txid +
201                    " is not available");
202        }
203        tx.commit(versionService);
204        return tx;
205    }
206
207    /**
208     * Roll a {@link Transaction} back
209     *
210     * @param txid the id of the {@link Transaction}
211     * @return the {@link Transaction} object
212     * @throws RepositoryException if the {@link Transaction} could not be found
213     */
214    @Override
215    public Transaction rollback(final String txid) throws RepositoryException {
216        final Transaction tx = transactions.remove(txid);
217        if (tx == null) {
218            throw new RepositoryException("Transaction with id " + txid +
219                    " is not available");
220        }
221        tx.rollback();
222        return tx;
223    }
224
225    /**
226     * Checks if a user is bound to a {@link Transaction}
227     *
228     * @param txId the id of the {@link Transaction}
229     * @param userName the name  of the {@link java.security.Principal}
230     * @return if the transaction was created by this user
231    */
232    @Override
233    public boolean isAssociatedWithUser(final String txId, final String userName) {
234        boolean existsForUser = false;
235        if (exists(txId)) {
236            final Transaction transaction = transactions.get(txId);
237            if (transaction.isAssociatedWithUser(userName)) {
238                existsForUser = true;
239            }
240        }
241        return existsForUser;
242    }
243
244    /**
245     * @param versionService the versionService to set
246     */
247    @Override
248    public void setVersionService(final VersionService versionService) {
249        this.versionService = versionService;
250    }
251
252}
