052/**
053 * Uncategorized helper methods
054 */
055@Component
056public abstract class ServiceHelpers {
057
058    /**
059     * Get the total size of a Node's properties
060     * 
061     * @param node
062     * @return size in bytes
063     * @throws RepositoryException
064     */
065    public static Long getNodePropertySize(final Node node)
066        throws RepositoryException {
067        Long size = 0L;
068        for (final PropertyIterator i = node.getProperties(); i.hasNext();) {
069            final Property p = i.nextProperty();
070            if (p.isMultiple()) {
071                for (final Value v : p.getValues()) {
072                    size += v.getBinary().getSize();
073                }
074            } else {
075                size += p.getBinary().getSize();
076            }
077        }
078        return size;
079    }
080
081    /**
082     * @param obj
083     * @return object size in bytes
084     * @throws RepositoryException
085     */
086    public static Long getObjectSize(final Node obj) throws RepositoryException {
087        return getNodePropertySize(obj) + getObjectDSSize(obj);
088    }
089
090    /**
091     * @param obj
092     * @return object's datastreams' total size in bytes
093     * @throws RepositoryException
094     */
095    private static Long getObjectDSSize(final Node obj)
096        throws RepositoryException {
097        Long size = 0L;
098        for (final NodeIterator i = obj.getNodes(); i.hasNext();) {
099            final Node node = i.nextNode();
100            if (node.isNodeType(NT_FILE)) {
101                size += getDatastreamSize(node);
102            }
103        }
104        return size;
105    }
106
107    /**
108     * Get the size of a datastream by calculating the size of the properties
109     * and the binary properties
110     * 
111     * @param ds
112     * @return
113     * @throws RepositoryException
114     */
115    public static Long getDatastreamSize(final Node ds)
116        throws RepositoryException {
117        return getNodePropertySize(ds) + getContentSize(ds);
118    }
119
120    /**
121     * Get the size of the JCR content binary property
122     * 
123     * @param ds
124     * @return
125     * @throws RepositoryException
126     */
127    public static Long getContentSize(final Node ds) throws RepositoryException {
128        long size = 0L;
129        if (ds.hasNode(JCR_CONTENT)) {
130            final Node contentNode = ds.getNode(JCR_CONTENT);
131
132            if (contentNode.hasProperty(JCR_DATA)) {
133                size =
134                        ds.getNode(JCR_CONTENT).getProperty(JCR_DATA)
135                                .getBinary().getSize();
136            }
137        }
138
139        return size;
140    }
141
142    /**
143     * A static factory function to insulate services from the details of
144     * building a DistributedExecutorService
145     * 
146     * @param cache
147     * @return
148     */
149    public static DistributedExecutorService getClusterExecutor(
150            final InfinispanBinaryStore cacheStore) {
151        // Watch out! This is trying to pluck out the blob cache store. This
152        // works as long as
153        // modeshape continues to be ordered..
154        return new DefaultExecutorService(cacheStore.getCaches().get(1));
155    }
156
157    /**
158     * Get the fixity function to map a low-level cache entry to its fixity
159     * result
160     * 
161     * @param dsChecksum
162     * @param dsSize
163     * @return
164     */
165    public static
166    Function<LowLevelCacheEntry, FixityResult>
167    getCheckCacheFixityFunction(final URI dsChecksum, final long dsSize) {
168        return new CheckCacheEntryFixity(dsChecksum, dsSize);
169    }
170
171    /**
172     * @return a double of the size of the fedora:datastream binary content
173     * @throws RepositoryException
174     */
175    public static long getRepositoryCount(final Repository repository)
176        throws RepositoryException {
177        final Session session = repository.login();
178        try {
179            final QueryManager queryManager =
180                session.getWorkspace().getQueryManager();
181
182            final String querystring =
183                "SELECT [" + JCR_PATH + "] FROM ["
184                        + FEDORA_OBJECT + "]";
185
186            final QueryResult queryResults =
187                queryManager.createQuery(querystring, JCR_SQL2).execute();
188
189            return queryResults.getRows().getSize();
190        } finally {
191
192            session.logout();
193        }
194    }
195
196    /**
197     * @return a double of the size of the fedora:datastream binary content
198     * @throws RepositoryException
199     */
200    public static long getRepositorySize(final Repository repository)
201        throws RepositoryException {
202        final Session session = repository.login();
203        long sum = 0;
204        final QueryManager queryManager =
205            session.getWorkspace().getQueryManager();
206
207        final String querystring =
208                "SELECT [" + CONTENT_SIZE + "] FROM [" +
209                        FEDORA_BINARY + "]";
210
211        final QueryResult queryResults =
212            queryManager.createQuery(querystring, JCR_SQL2).execute();
213
214        for (final RowIterator rows = queryResults.getRows(); rows.hasNext();) {
215            final Value value =
216                    rows.nextRow().getValue(CONTENT_SIZE);
217            sum += value.getLong();
218        }
219
220        session.logout();
221
222        return sum;
223    }
