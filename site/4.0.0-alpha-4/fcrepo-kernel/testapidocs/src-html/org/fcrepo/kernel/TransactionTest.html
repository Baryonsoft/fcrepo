017package org.fcrepo.kernel;
018
019import static java.lang.System.currentTimeMillis;
020import static org.fcrepo.kernel.Transaction.State.COMMITED;
021import static org.fcrepo.kernel.Transaction.State.DIRTY;
022import static org.fcrepo.kernel.Transaction.State.NEW;
023import static org.fcrepo.kernel.Transaction.State.ROLLED_BACK;
024import static org.junit.Assert.assertEquals;
025import static org.junit.Assert.assertFalse;
026import static org.junit.Assert.assertTrue;
027import static org.mockito.Mockito.never;
028import static org.mockito.Mockito.verify;
029import static org.mockito.Mockito.when;
030import static org.mockito.MockitoAnnotations.initMocks;
031
032import javax.jcr.RepositoryException;
033import javax.jcr.Session;
034
035import org.junit.Before;
036import org.junit.Test;
037import org.mockito.Mock;
038
039public class TransactionTest {
040
041    private Transaction testObj;
042
043    private static final String USER_NAME = "test";
044
045    @Mock
046    private Session mockSession;
047
048    @Before
049    public void setUp() {
050        initMocks(this);
051        testObj = new TransactionImpl(mockSession, USER_NAME);
052    }
053
054    @Test
055    public void testRollback() throws RepositoryException {
056        testObj.rollback();
057        verify(mockSession).refresh(false);
058        verify(mockSession).logout();
059        assertEquals(ROLLED_BACK, testObj.getState());
060        final long update = testObj.getExpires().getTime();
061        assertTrue(update <= currentTimeMillis());
062    }
063
064    @Test
065    public void testCommit() throws RepositoryException {
066        testObj.commit(null);
067        verify(mockSession).save();
068        verify(mockSession).logout();
069        assertEquals(COMMITED, testObj.getState());
070        final long update = testObj.getExpires().getTime();
071        assertTrue(update <= currentTimeMillis());
072    }
073
074    @Test
075    public void testExpire() throws RepositoryException {
076        final long orig = testObj.getExpires().getTime();
077        testObj.expire();
078        verify(mockSession, never()).save();
079        verify(mockSession).logout();
080        final long update = testObj.getExpires().getTime();
081        assertTrue(update < orig);
082        assertTrue(update <= currentTimeMillis());
083        assertTrue(update < orig);
084    }
085
086    @Test
087    public void testExpiryUpdate() {
088        final long orig = testObj.getExpires().getTime();
089        testObj.updateExpiryDate();
090        final long update = testObj.getExpires().getTime();
091        assertTrue("Unexpected negative expiry delta: " + (update - orig),
092                update - orig >= 0);
093    }
094
095    @Test
096    public void testState() throws RepositoryException {
097        assertEquals(NEW, testObj.getState());
098        when(mockSession.hasPendingChanges()).thenReturn(true, false);
099        assertEquals(DIRTY, testObj.getState());
100        testObj.commit(null);
101    }
102
103    @Test
104    public void testUserAssociation() {
105        final String otherUser = "dummy";
106        assertTrue("Transaction expected to be associated with user " + USER_NAME,
107                testObj.isAssociatedWithUser(USER_NAME));
108        assertFalse("Transaction should not be associated with the user" + otherUser,
109                testObj.isAssociatedWithUser(otherUser));
110        assertFalse("Transaction should not be associated with an empty user",
111                testObj.isAssociatedWithUser(null));
112
113        testObj = new TransactionImpl(mockSession, null);
114        assertTrue("Transaction should not be associated with a user",
115                testObj.isAssociatedWithUser(null));
116        assertFalse("Transaction should not be associated with a user",
117                testObj.isAssociatedWithUser(USER_NAME));
118    }
119}
