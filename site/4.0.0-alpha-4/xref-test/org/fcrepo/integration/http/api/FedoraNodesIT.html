372         logger.debug("Entering verifyFullSetOfRdfTypes()...");
373         final String pid = "FedoraGraphWithRdfTypes";

397                 http://www.jcp.org/jcr/mix/1.0simpleVersionable

400                 http://www.jcp.org/jcr/nt/1.0folder

413         verifyResource(model, nodeUri, rdfType, MIX_NAMESPACE, "lastModified");
414         verifyResource(model, nodeUri, rdfType, MIX_NAMESPACE, "lockable");
415         verifyResource(model, nodeUri, rdfType, MIX_NAMESPACE, "referenceable");
416         verifyResource(model, nodeUri, rdfType, MIX_NAMESPACE, "simpleVersionable");
417         verifyResource(model, nodeUri, rdfType, MIX_NAMESPACE, "versionable");

555                 + subjectURI + "> <info:rubydora#label> ?p}").getBytes()));
556 
557         updateObjectGraphMethod.setEntity(e);
558 
559         final HttpResponse response = client.execute(updateObjectGraphMethod);
560         assertEquals(NO_CONTENT.getStatusCode(), response.getStatusLine()
561                 .getStatusCode());
562 
563         final HttpGet getObjMethod = new HttpGet(subjectURI);
564 
565         getObjMethod.addHeader("Accept", "application/rdf+xml");
566         final HttpResponse getResponse = client.execute(getObjMethod);
567         assertEquals(OK.getStatusCode(), getResponse.getStatusLine()

573                 "<" + subjectURI + "> <info:rubydora#label> \"asdfg\" \\.",
574                 DOTALL).matcher(content).find());
