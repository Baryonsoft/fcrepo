246         final NodeTypeDefinition[] nodeTypes2 = new NodeTypeDefinition[]{type2};
247         mgr.registerNodeTypes(nodeTypes2, true);
248 
249         //create object with inheriting type
250         FedoraResource object = objectService.createObject(session, "/testNTTnheritanceObject");
251         final javax.jcr.Node node = object.getNode();
252         node.addMixin("test:testInher");
253 
254         session.save();
255         session.logout();
256         session = repo.login();
257 
258         object = objectService.createObject(session, "/testNTTnheritanceObject");
259 
260         //test that supertype has been inherited as rdf:type
261         final Node s = createGraphSubjectNode("/testNTTnheritanceObject");
