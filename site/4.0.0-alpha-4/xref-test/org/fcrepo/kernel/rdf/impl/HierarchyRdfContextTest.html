148     @Test
149     public void testForLDPTriples() throws RepositoryException, IOException {
150         nodeIsContainer();
151         when(mockNode.hasNodes()).thenReturn(false);
152         final Model results = getResults();
153 
154         logRdf("Created RDF for testForLDPTriples()", results);
155 
156         // check for LDP-specified Page information
157         assertTrue("Didn't find page described as LDP Page!", results.contains(
158                 testPage, type, PAGE));
159         assertTrue("Didn't find page described as LDP Page of correct node!",
160                 results.contains(testPage, PAGE_OF, testSubject));
161         assertTrue("Didn't find page described as having inlined members!",
162                 results.contains(testPage, MEMBERS_INLINED,
163                         createPlainLiteral("true")));
164 
165         // check for LDP-specified node information
166         assertTrue(
167                 "Didn't find node described as being the subject of membership!",
168                 results.contains(testSubject, MEMBERSHIP_SUBJECT, testSubject));
169         assertTrue("Didn't find node described as being an LDP Container!",
170                 results.contains(testSubject, type, CONTAINER));
171         assertTrue(
172                 "Didn't find node described as have an LDP membership object!",
173                 results.contains(testSubject, MEMBERSHIP_OBJECT, MEMBER_SUBJECT));
174         assertTrue(
175                 "Didn't find node described as using the correct LDP membership predicate!",
176                 results.contains(testSubject, MEMBERSHIP_PREDICATE, HAS_CHILD));
177     }
