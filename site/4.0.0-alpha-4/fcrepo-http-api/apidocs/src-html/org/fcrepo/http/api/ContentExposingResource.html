017package org.fcrepo.http.api;
018
019import com.sun.jersey.core.header.ContentDisposition;
020import org.fcrepo.http.commons.AbstractResource;
021import org.fcrepo.http.commons.api.rdf.HttpGraphSubjects;
022import org.fcrepo.http.commons.domain.Range;
023import org.fcrepo.http.commons.responses.RangeRequestInputStream;
024import org.fcrepo.kernel.Datastream;
025
026import javax.jcr.RepositoryException;
027import javax.ws.rs.core.CacheControl;
028import javax.ws.rs.core.EntityTag;
029import javax.ws.rs.core.Request;
030import javax.ws.rs.core.Response;
031import java.io.IOException;
032import java.io.InputStream;
033import java.util.Date;
034
035import static javax.ws.rs.core.Response.ok;
036import static javax.ws.rs.core.Response.status;
037
038/**
039 * An abstract class that sits between AbstractResource and any resource that
040 * wishes to share the routines for building responses containing binary
041 * content.
042 *
043 * @author Mike Durbin
044 */
045public abstract class ContentExposingResource extends AbstractResource {
046
047    protected static final int REQUESTED_RANGE_NOT_SATISFIABLE = 416;
048
049    protected static final int PARTIAL_CONTENT = 206;
050
051    /**
052     * A helper method that does most of the work associated with processing a request
053     * for content (or a range of the content) into a Response
054     */
055    protected Response getDatastreamContentResponse(Datastream ds, String rangeValue, Request request,
056                                                    HttpGraphSubjects subjects) throws
057            RepositoryException, IOException {
058        final EntityTag etag =
059                new EntityTag(ds.getContentDigest().toString());
060        final Date date = ds.getLastModifiedDate();
061        final Date roundedDate = new Date();
062        roundedDate.setTime(date.getTime() - date.getTime() % 1000);
063        Response.ResponseBuilder builder =
064                request.evaluatePreconditions(roundedDate, etag);
065
066        final CacheControl cc = new CacheControl();
067        cc.setMaxAge(0);
068        cc.setMustRevalidate(true);
069
070        if (builder == null) {
071
072            final InputStream content = ds.getContent();
073
074            if (rangeValue != null && rangeValue.startsWith("bytes")) {
075
076                final Range range = Range.convert(rangeValue);
077
078                final long contentSize = ds.getContentSize();
079
080                final String endAsString;
081
082                if (range.end() == -1) {
083                    endAsString = Long.toString(contentSize - 1);
084                } else {
085                    endAsString = Long.toString(range.end());
086                }
087
088                final String contentRangeValue =
089                        String.format("bytes %s-%s/%s", range.start(),
090                                endAsString, contentSize);
091
092                if (range.end() > contentSize ||
093                        (range.end() == -1 && range.start() > contentSize)) {
094                    builder =
095                            status(
096                                    REQUESTED_RANGE_NOT_SATISFIABLE)
097                                    .header("Content-Range",
098                                            contentRangeValue);
099                } else {
100                    final RangeRequestInputStream rangeInputStream =
101                            new RangeRequestInputStream(content, range
102                                    .start(), range.size());
103
104                    builder =
105                            status(PARTIAL_CONTENT).entity(
106                                    rangeInputStream)
107                                    .header("Content-Range",
108                                            contentRangeValue);
109                }
110
111            } else {
112                builder = ok(content);
113            }
114        }
115
116        final ContentDisposition contentDisposition = ContentDisposition.type("attachment")
117                .fileName(ds.getFilename())
118                .creationDate(ds.getCreatedDate())
119                .modificationDate(ds.getLastModifiedDate())
120                .size(ds.getContentSize())
121                .build();
122
123        return builder.type(ds.getMimeType()).header(
124                "Link",
125                subjects.getGraphSubject(ds.getNode()) +
126                        ";rel=\"meta\"").header("Accept-Ranges",
127                "bytes").cacheControl(cc).lastModified(date).tag(etag)
128                .header("Content-Disposition", contentDisposition)
129                .build();
130    }
131
132}
