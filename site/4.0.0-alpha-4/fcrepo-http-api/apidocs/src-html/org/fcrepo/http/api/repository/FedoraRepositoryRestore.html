017package org.fcrepo.http.api.repository;
018
019import static javax.ws.rs.core.Response.noContent;
020import static javax.ws.rs.core.Response.serverError;
021import static org.slf4j.LoggerFactory.getLogger;
022
023import java.io.File;
024import java.io.IOException;
025import java.io.InputStream;
026import java.util.ArrayList;
027import java.util.List;
028
029import javax.jcr.RepositoryException;
030import javax.jcr.Session;
031import javax.ws.rs.POST;
032import javax.ws.rs.Path;
033import javax.ws.rs.WebApplicationException;
034import javax.ws.rs.core.Response;
035
036import org.apache.commons.io.IOUtils;
037import org.fcrepo.http.commons.AbstractResource;
038import org.fcrepo.http.commons.session.InjectedSession;
039import org.modeshape.jcr.api.Problem;
040import org.modeshape.jcr.api.Problems;
041import org.slf4j.Logger;
042import org.springframework.context.annotation.Scope;
043import org.springframework.stereotype.Component;
044
045/**
046 * Restore a backup of the repository
047 */
048@Component
049@Scope("prototype")
050@Path("/fcr:restore")
051public class FedoraRepositoryRestore extends AbstractResource {
052
053    private static final Logger LOGGER = getLogger(FedoraRepositoryRestore.class);
054
055    @InjectedSession
056    protected Session session;
057
058    /**
059     * This method runs a repository restore.
060     *
061     * @return
062     * @throws RepositoryException
063     * @throws IOException
064     */
065    @POST
066    public Response runRestore(final InputStream bodyStream) throws RepositoryException,
067        IOException {
068
069        if (null == bodyStream) {
070            throw new WebApplicationException(serverError().entity(
071                    "Request body must not be null").build());
072        }
073
074        final String body = IOUtils.toString(bodyStream);
075        final File backupDirectory = new File(body.trim());
076        if (!backupDirectory.exists()) {
077            throw new WebApplicationException(serverError().entity(
078                    "Backup directory does not exist: "
079                            + backupDirectory.getAbsolutePath()).build());
080        }
081
082        try {
083            final Problems problems = repositoryService.restoreRepository(session, backupDirectory);
084            if (problems.hasProblems()) {
085                LOGGER.error("Problems restoring up the repository:");
086
087                final List<String> problemsOutput = new ArrayList<>();
088
089                // Report the problems (we'll just print them out) ...
090                for (final Problem problem : problems) {
091                    LOGGER.error("{}", problem.getMessage());
092                    problemsOutput.add(problem.getMessage());
093                }
094
095                throw new WebApplicationException(serverError()
096                        .entity(problemsOutput).build());
097
098            }
099            return noContent().build();
100        } finally {
101            session.logout();
102        }
103    }
104}
