118    @Test
119    public void testPutContent() throws RepositoryException, InvalidChecksumException, URISyntaxException, ParseException {
120        final String pid = "FedoraDatastreamsTest1";
121        final String dsId = "testDS";
122        final String dsContent = "asdf";
123        final String dsPath = "/" + pid + "/" + dsId;
124        final InputStream dsContentStream = IOUtils.toInputStream(dsContent);
125        when(mockNode.isNew()).thenReturn(true);
126        when(mockNode.getNode(JCR_CONTENT)).thenReturn(mockContentNode);
127        when(mockContentNode.getPath()).thenReturn(dsPath + "/jcr:content");
128        when(mockNodeService.exists(mockSession, dsPath)).thenReturn(false);
129        when(
130                mockDatastreams.createDatastreamNode(any(Session.class),
131                        eq(dsPath), anyString(), eq("xyz"), any(InputStream.class), eq((URI)null)))
132                .thenReturn(mockNode);
133        when(mockDatastreams.exists(mockSession, dsPath)).thenReturn(true);
134        final Response actual =
135            testObj.modifyContent(createPathList(pid, dsId), null, "inline; filename=\"xyz\"", null,
136                    dsContentStream, null);
137        assertEquals(CREATED.getStatusCode(), actual.getStatus());
138        verify(mockDatastreams).createDatastreamNode(any(Session.class),
139                eq(dsPath), anyString(), eq("xyz"), any(InputStream.class), eq((URI)null));
140        verify(mockSession).save();
141    }

225    @Test
226    public void testModifyContent() throws RepositoryException, InvalidChecksumException, URISyntaxException, ParseException {
227        final String pid = "FedoraDatastreamsTest1";
228        final String dsId = "testDS";
229        final String dsContent = "asdf";
230        final String dsPath = "/" + pid + "/" + dsId;
231        final URI checksum = new URI("urn:sha1:some-checksum");
232        final InputStream dsContentStream = IOUtils.toInputStream(dsContent);
233        when(mockNode.isNew()).thenReturn(false);
234        final Datastream mockDs = mockDatastream(pid, dsId, dsContent);
235        when(mockDatastreams.getDatastream(mockSession, dsPath)).thenReturn(
236                mockDs);
237        final Request mockRequest = mock(Request.class);
238        when(
239                mockRequest.evaluatePreconditions(any(Date.class),
240                        any(EntityTag.class))).thenReturn(null);
241        when(
242                mockDatastreams.createDatastreamNode(any(Session.class),
243                        eq(dsPath), anyString(), eq((String)null), any(InputStream.class), eq(checksum)))
244                .thenReturn(mockNode);
245        when(mockDatastreams.exists(mockSession, dsPath)).thenReturn(true);
246        final Response actual =
247            testObj.modifyContent(createPathList(pid, dsId), "urn:sha1:some-checksum", null, null,
248                    dsContentStream, mockRequest);
249        assertEquals(NO_CONTENT.getStatusCode(), actual.getStatus());
250        verify(mockDatastreams).createDatastreamNode(any(Session.class),
251                eq(dsPath), anyString(), eq((String)null), any(InputStream.class), eq(checksum));
252        verify(mockSession).save();
253    }

255    @Test
256    public void testGetContent() throws RepositoryException, IOException {
257        final String pid = "FedoraDatastreamsTest1";
258        final String dsId = "testDS";
259        final String path = "/" + pid + "/" + dsId;
260        final String dsContent = "asdf";
261        final Datastream mockDs = mockDatastream(pid, dsId, dsContent);
262        when(mockDatastreams.getDatastream(mockSession, path)).thenReturn(
263                mockDs);
264        when(mockDs.getNode()).thenReturn(mockNode);
265        when(mockNode.getPath()).thenReturn(path);
266        final Request mockRequest = mock(Request.class);
267        final Response actual =
268            testObj.getContent(createPathList(pid, dsId), null, mockRequest);
269        verify(mockDs).getContent();
270        verify(mockSession, never()).save();
271        final String actualContent =
272            IOUtils.toString((InputStream) actual.getEntity());
273        assertEquals("http://localhost/fcrepo" + path + ";rel=\"meta\"", actual
274                .getMetadata().getFirst("Link"));
275        assertEquals("asdf", actualContent);
276    }
