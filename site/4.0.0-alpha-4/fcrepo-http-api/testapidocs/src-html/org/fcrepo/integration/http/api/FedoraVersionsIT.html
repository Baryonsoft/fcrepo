075    @Test
076    public void testAddAndRetrieveVersion() throws Exception {
077        final String pid = UUID.randomUUID().toString();
078        createObject(pid);
079        addMixin( pid, MIX_NAMESPACE + "versionable" );
080
081        logger.info("Setting a title");
082        patchLiteralProperty(serverAddress + pid, "http://purl.org/dc/elements/1.1/title", "First Title");
083
084        final GraphStore nodeResults = getContent(serverAddress + pid);
085        assertTrue("Should find original title", nodeResults.contains(Node.ANY, Node.ANY, DC_TITLE.asNode(), NodeFactory.createLiteral("First Title")));
086
087        logger.info("Posting version v0.0.1");
088        postObjectVersion(pid, "v0.0.1");
089
090        logger.info("Replacing the title");
091        patchLiteralProperty(serverAddress + pid, "http://purl.org/dc/elements/1.1/title", "Second Title");
092
093        final GraphStore versionResults = getContent(serverAddress + pid + "/fcr:versions/v0.0.1");
094        logger.info("Got version profile:");
095
096        assertTrue("Didn't find a version triple!",
097                      versionResults.contains(Node.ANY, Node.ANY, HAS_PRIMARY_TYPE.asNode(), NodeFactory.createLiteral("nt:frozenNode")));
098
099        assertTrue("Should find a title in historic version", versionResults.contains(Node.ANY, Node.ANY, DC_TITLE.asNode(), Node.ANY));
100        assertTrue("Should find original title in historic version", versionResults.contains(Node.ANY, Node.ANY, DC_TITLE.asNode(), NodeFactory.createLiteral("First Title")));
101        assertFalse("Should not find the updated title in historic version", versionResults.contains(Node.ANY, Node.ANY, DC_TITLE.asNode(), NodeFactory.createLiteral("Second Title")));
102    }

207    @Test
208    public void isAutoVersionedContentStillAccessible() throws Exception {
209        final String objName = UUID.randomUUID().toString();
210        final String dsName = UUID.randomUUID().toString();
211        final String firstVersionText = "foo";
212        final String secondVersionText = "bar";
213
214        createObject(objName);
215        createDatastream(objName, dsName, firstVersionText);
216        addMixin( objName + "/" + dsName, MIX_NAMESPACE + "versionable" );
217
218        setAutoVersioning(serverAddress + objName + "/" + dsName);
219
220        mutateDatastream(objName, dsName, secondVersionText);
221
222        final HttpGet retrieveMutatedDataStreamMethod =
223                new HttpGet(serverAddress +
224                        objName + "/" + dsName + "/fcr:content");
225        assertEquals("Datastream didn't accept mutation!", secondVersionText,
226                EntityUtils.toString(
227                        client.execute(
228                                retrieveMutatedDataStreamMethod).getEntity()));
229
230        final HttpGet getVersion =
231                new HttpGet(serverAddress + objName + "/" + dsName + "/fcr:versions");
232        logger.debug("Retrieved version profile:");
233        final GraphStore results = getGraphStore(getVersion);
234        final Resource subject =
235                createResource(serverAddress + objName + "/" + dsName);
236        assertTrue("Didn't find a version triple!",
237                results.contains(Node.ANY, subject.asNode(), HAS_VERSION.asNode(), Node.ANY));
238
239        verifyVersions(results, subject.asNode(), firstVersionText);
240    }

281    private void testDatastreamContentUpdatesCreateNewVersions(final String objName, final String dsName) throws IOException {
282        final String firstVersionText = "foo";
283        final String secondVersionText = "bar";
284        createDatastream(objName, dsName, firstVersionText);
285        final GraphStore dsInitialVersion = getContent(serverAddress + objName);
286        assertTrue("Should find auto-created versoning policy",
287                dsInitialVersion.contains(Node.ANY, createResource(serverAddress + objName+ "/" + dsName).asNode(), VERSIONING_POLICY.asNode(), NodeFactory.createLiteral("auto-version")));
288
289        mutateDatastream(objName, dsName, secondVersionText);
290        final HttpGet retrieveMutatedDataStreamMethod =
291                new HttpGet(serverAddress +
292                        objName + "/" + dsName + "/fcr:content");
293        assertEquals("Datastream didn't accept mutation!", secondVersionText,
294                EntityUtils.toString(
295                        client.execute(
296                                retrieveMutatedDataStreamMethod).getEntity()));
297
298        final HttpGet getVersion =
299                new HttpGet(serverAddress + objName + "/" + dsName + "/fcr:versions");
300        logger.debug("Retrieved version profile:");
301        final GraphStore results = getGraphStore(getVersion);
302        final Resource subject =
303                createResource(serverAddress + objName + "/" + dsName);
304        assertTrue("Didn't find a version triple!",
305                results.contains(Node.ANY, subject.asNode(), HAS_VERSION.asNode(), Node.ANY));
306
307        verifyVersions(results, subject.asNode(), firstVersionText, secondVersionText);
308    }

310    /**
311     * Verifies that one version exists with each supplied value.  This method
312     * makes assertions that each of the provided values is the content of a
313     * version node and nothing else.  Order isn't important, and no assumption
314     * is made about whether extra versions exist.
315     */
316    private void verifyVersions(final GraphStore graph, final Node subject, final String ... values) throws IOException {
317        final ArrayList<String> remainingValues = newArrayList(values);
318        final Iterator<Quad> versionIt = graph.find(ANY, subject, HAS_VERSION.asNode(), ANY);
319        while (versionIt.hasNext() && !remainingValues.isEmpty()) {
320            final String value =
321                EntityUtils.toString(client
322                        .execute(new HttpGet(versionIt.next().getObject().getURI() + "/fcr:content")).getEntity());
323            remainingValues.remove(value);
324        }
325        if (!remainingValues.isEmpty()) {
326            fail(remainingValues.get(0) + " was not preserved in the version history!");
327        }
328    }

331        final HttpPost httpPost = new HttpPost(serverAddress + "/fcr:nodetypes");
332        final BasicHttpEntity entity = new BasicHttpEntity();
333        entity.setContent(new ByteArrayInputStream(snippet.getBytes()));

337        assertEquals(NO_CONTENT.getStatusCode(), response.getStatusLine().getStatusCode());

340    private static void setAutoVersioning(final String url) throws IOException {
341        patchLiteralProperty(url,
342                "http://fedora.info/definitions/v4/config#versioningPolicy",
343                "auto-version");
344    }
345
346    private static void patchLiteralProperty(final String url, final String predicate, final String literal) throws IOException {
347        final HttpPatch updateObjectGraphMethod =
348                new HttpPatch(url);
349        updateObjectGraphMethod.addHeader("Content-Type",
350                "application/sparql-update");
351        final BasicHttpEntity e = new BasicHttpEntity();
352        e.setContent(new ByteArrayInputStream(
353                ("INSERT DATA { <> <" + predicate + "> \"" + literal + "\" } ")
354                        .getBytes()));
355        updateObjectGraphMethod.setEntity(e);
356        final HttpResponse response = client.execute(updateObjectGraphMethod);
357        assertEquals(NO_CONTENT.getStatusCode(), response.getStatusLine()
358                .getStatusCode());
359    }
