047/**
048 * Translate JCR paths to URLs.  There are a few types of translations
049 * that occur as part of this implementation:
050 * <ul>
051 *     <li>jcr:content is replaced with fcr:content</li>
052 *     <li>information about the transaction is added</li>
053 * </ul>
054 *
055 */
056public class HttpGraphSubjects implements GraphSubjects {
057
058    private static final Logger LOGGER = getLogger(HttpGraphSubjects.class);
059    public static final String WORKSPACE_PREFIX = "workspace:";
060    public static final String TX_PREFIX = "tx:";
061
062    protected final UriBuilder nodesBuilder;
063
064    private final String basePath;
065
066    private final int pathIx;
067
068    private final URI context;
069
070    private final Session session;
071
072    private final String defaultWorkspace;
073
074    private final Function<Repository, String> getDefaultWorkspace = new GetDefaultWorkspace();
075
076    /**
077     * Build HTTP graph subjects relative to the given JAX-RS resource, using the UriInfo provided.
078     *
079     * The session may provide additional information (e.g. workspaces) that need to be
080     * taken into account.
081     *
082     * @param relativeTo
083     * @param uris
084     */
085    public HttpGraphSubjects(final Session session, final Class<?> relativeTo, final UriInfo uris) {
086        this.context = uris.getRequestUri();
087        this.nodesBuilder = uris.getBaseUriBuilder().path(relativeTo);
088        String normalizedBasePath = nodesBuilder.build("").toString();
089        if (!normalizedBasePath.endsWith("/")) {
090            normalizedBasePath = normalizedBasePath + "/";
091        }
092        this.basePath = normalizedBasePath;
093        this.pathIx = normalizedBasePath.length() - 1;
094        this.session = session;
095        this.defaultWorkspace = getDefaultWorkspace.apply(session.getRepository());
096        LOGGER.debug("Resolving graph subjects to a base URI of \"{}\"",
097                normalizedBasePath);
098    }
099
100    @Override
101    public Resource getGraphSubject(final String absPath)
102        throws RepositoryException {
103        final URI result =
104                nodesBuilder.buildFromMap(getPathMap(absPath));
105        LOGGER.debug("Translated path {} into RDF subject {}", absPath, result);
106        return createResource(result.toString());
107    }
108
109    @Override
110    public Resource getContext() {
111        return createResource(context.toString());
112    }
113
114    @Override
115    public Resource getGraphSubject(final Node node) throws RepositoryException {
116        final URI result = nodesBuilder.buildFromMap(getPathMap(node));
117        LOGGER.debug("Translated node {} into RDF subject {}", node, result);
118        return createResource(result.toString());
119    }
120
121    @Override
122    public Node getNodeFromGraphSubject(final Resource subject) throws RepositoryException {
123        final String subjectUri = getResourceURI(subject);
124        if (subjectUri == null) {
125            return null;
126        }
127        final String absPath = getPathFromGraphSubject(subjectUri);
128
129        if (absPath == null) {
130            return null;
131        }
132
133        final Node node;
134
135        if (session.nodeExists(absPath)) {
136            node = session.getNode(absPath);
137            LOGGER.trace("RDF resource {} maps to JCR node {}", subjectUri, node);
138        } else {
139            node = null;
140            LOGGER.debug("RDF resource {} looks like a Fedora node, but when we checked was not in the repository",
141                    subjectUri);
142        }
143
144        return node;
145    }
146
147    private String getResourceURI(final Resource subject) {
148        if (!subject.isURIResource()) {
149            LOGGER.debug("RDF resource {} was not a URI resource: returning null.",
150                    subject);
151            return null;
152        }
153        return subject.getURI();
154    }
155
156    @Override
157    public String getPathFromGraphSubject(final Resource subject) throws RepositoryException {
158        final String subjectUri = getResourceURI(subject);
159        if (subjectUri == null) {
160            return null;
161        }
162        return getPathFromGraphSubject(subjectUri);
163    }
164
165    /**
166     * Gets a path from the graph subject's URI.  This method does the heavy
167     * lifting for getNodeFromGraphSubject and getPathFromGraphSubject in a way
168     * that's tied to a more generic URI rather than a rdf Resource.
169     */
170    protected String getPathFromGraphSubject(@NotNull final String subjectUri) throws RepositoryException {
171
172        if (!isFedoraGraphSubject(subjectUri)) {
173            LOGGER.debug(
174                    "RDF resource {} was not a URI resource with our expected basePath {}, returning null.",
175                    subjectUri, basePath);
176            return null;
177        }
178
179        final StringBuilder pathBuilder = new StringBuilder();
180        final String absPath;
181        final String[] pathSegments =
182                subjectUri.substring(pathIx).split("/");
183
184        for (final String segment : pathSegments) {
185            if (segment.startsWith(TX_PREFIX)) {
186                final String tx = segment.substring(TX_PREFIX.length());
187
188                final String currentTxId = getCurrentTransactionId(session);
189
190                if (currentTxId == null || !tx.equals(currentTxId)) {
191                    throw new RepositoryException(
192                            "Subject is not in this transaction");
193                }
194
195            } else if (segment.startsWith(WORKSPACE_PREFIX)) {
196                final String workspace = segment.substring(WORKSPACE_PREFIX.length());
197                if (!session.getWorkspace().getName().equals(workspace)) {
198                    throw new RepositoryException(
199                            "Subject is not in this workspace");
200                }
201            } else if (segment.equals(FCR_CONTENT)) {
202                pathBuilder.append("/");
203                pathBuilder.append(JCR_CONTENT);
204            } else {
205                if (!segment.isEmpty()) {
206                    pathBuilder.append("/");
207                    pathBuilder.append(segment);
208                }
209            }
210        }
211
212        if (pathBuilder.length() == 0) {
213            absPath = "/";
214        } else {
215            absPath = pathBuilder.toString();
216        }
217
218        if (isValidJcrPath(absPath)) {
219            return absPath;
220        }
221        return null;
222
223    }
224
225    private boolean isValidJcrPath(final String absPath) {
226        try {
227            String pathToValidate = absPath;
228            String txId = getCurrentTransactionId(session);
229            if (txId != null) {
230                String txIdWithSlash = "/" + TX_PREFIX + txId;
231                /* replace the first occurrence of tx within the path */
232                pathToValidate = StringUtils.replaceOnce(absPath, txIdWithSlash, StringUtils.EMPTY);
233                LOGGER.debug("removed {} from URI {}. Path for JCR validation is now: {}",
234                        txIdWithSlash, absPath, pathToValidate);
235            }
236            session.getValueFactory().createValue(pathToValidate, PATH);
237            return true;
238        } catch (final ValueFormatException e) {
239            LOGGER.trace("Unable to validate JCR path", e);
240            return false;
241        } catch (final RepositoryException e) {
242            throw new RepositoryRuntimeException("Unable to validate JCR path", e);
243        }
244    }
245
246    @Override
247    public boolean isFedoraGraphSubject(final Resource subject) {
248        return subject.isURIResource()
249                && isFedoraGraphSubject(subject.getURI());
250    }
251
252    private boolean isFedoraGraphSubject(final String subjectUri) {
253        return subjectUri.startsWith(basePath)
254                && isValidJcrPath(subjectUri.substring(pathIx));
255    }
256
257    protected Map<String, String> getPathMap(final Node node)
258        throws RepositoryException {
259        return getPathMap(node.getPath());
260    }
261
262    private Map<String, String> getPathMap(final String absPath) {
263        // the path param value doesn't start with a slash
264        String path = absPath.substring(1);
265        if (path.endsWith(JCR_CONTENT)) {
266            path = path.replace(JCR_CONTENT, FCR_CONTENT);
267        }
268
269        if (session != null) {
270            final Workspace workspace = session.getWorkspace();
271
272            final String txId = getCurrentTransactionId(session);
273
274            if (txId != null) {
275                path = TX_PREFIX + txId + "/" + path;
276            } else if (workspace != null &&
277                    !workspace.getName().equals(defaultWorkspace)) {
278                path = WORKSPACE_PREFIX + workspace.getName() + "/" + path;
279            }
280        }
281
282        return singletonMap("path", path);
283    }
