055    @Test
056    public void testGetGraphSubject() throws RepositoryException {
057        final String expected = "http://localhost:8080/fcrepo/rest" + testPath;
058        when(mockNode.getPath()).thenReturn(testPath);
059        when(mockWorkspace.getName()).thenReturn("default");
060        when(mockSession.getWorkspace()).thenReturn(mockWorkspace);
061        when(mockNode.getSession()).thenReturn(mockSession);
062        Resource actual = testObj.getGraphSubject(mockNode);
063        assertEquals(expected, actual.getURI());
064        when(mockNode.getPath()).thenReturn(testPath + "/jcr:content");
065        actual = testObj.getGraphSubject(mockNode);
066        assertEquals(expected + "/fcr:content", actual.getURI());
067    }
068
069    @Test
070    public void testGetNodeFromGraphSubject() throws PathNotFoundException,
071            RepositoryException {
072
073        when(mockSession.nodeExists(testPath)).thenReturn(true);
074        when(mockSession.getNode(testPath)).thenReturn(mockNode);
075        when(mockWorkspace.getName()).thenReturn("default");
076        when(mockSession.getWorkspace()).thenReturn(mockWorkspace);
077        // test a good subject
078        when(mockSubject.getURI()).thenReturn(
079                "http://localhost:8080/fcrepo/rest" + testPath);
080        when(mockSubject.isURIResource()).thenReturn(true);
081        Node actual = testObj.getNodeFromGraphSubject(mockSubject);
082        Mockito.verify(mockSession).getNode(testPath);
083        assertEquals(mockNode, actual);
084        // test a bad subject
085        when(mockSubject.getURI()).thenReturn(
086                "http://localhost:8080/fcrepo/rest2" + testPath + "/bad");
087        actual = testObj.getNodeFromGraphSubject(mockSubject);
088        assertEquals(null, actual);
089        // test a non-existent path
090        when(mockSubject.getURI()).thenReturn(
091                "http://localhost:8080/fcrepo/rest" + testPath + "/bad");
092        actual = testObj.getNodeFromGraphSubject(mockSubject);
093        assertEquals(null, actual);
094        // test a fcr:content path
095        when(mockSession.nodeExists(testPath + "/jcr:content")).thenReturn(true);
096        when(mockSubject.getURI())
097                .thenReturn(
098                        "http://localhost:8080/fcrepo/rest" + testPath +
099                                "/fcr:content");
100        actual = testObj.getNodeFromGraphSubject(mockSubject);
101        verify(mockSession).getNode(testPath + "/jcr:content");
102    }

156    protected static UriInfo getUriInfoImpl(final String path) {
157        // UriInfo ui = mock(UriInfo.class,withSettings().verboseLogging());
158        final UriInfo ui = mock(UriInfo.class);
159        final UriBuilder ub = new UriBuilderImpl();
160        ub.scheme("http");
161        ub.host("localhost");
162        ub.port(8080);
163        ub.path("/fcrepo");
164
165        final UriBuilder rb = new UriBuilderImpl();
166        rb.scheme("http");
167        rb.host("localhost");
168        rb.port(8080);
169        rb.path("/fcrepo/rest" + path);
170
171        when(ui.getRequestUri()).thenReturn(
172                URI.create("http://localhost:8080/fcrepo/rest" + path));
173        when(ui.getBaseUri()).thenReturn(
174                URI.create("http://localhost:8080/fcrepo"));
175        when(ui.getBaseUriBuilder()).thenReturn(ub);
176        when(ui.getAbsolutePathBuilder()).thenReturn(rb);
177
178        return ui;
179    }
