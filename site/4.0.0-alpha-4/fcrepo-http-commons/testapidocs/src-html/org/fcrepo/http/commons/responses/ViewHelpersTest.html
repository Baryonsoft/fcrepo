182    @Test
183    public void shouldConvertRdfObjectsToStrings() {
184
185        final DatasetGraph mem = createMem();
186        mem.add(createAnon(), createURI("subject"), createURI("a/b/c"),
187                NodeFactory.createLiteral("abc"));
188        mem.add(createAnon(), createURI("subject"),
189                createURI("a-numeric-type"), createTypedLiteral(0).asNode());
190        mem.add(createAnon(), createURI("subject"),
191                createURI("an-empty-string"), createLiteral(""));
192        mem.add(createAnon(), createURI("subject"), createURI("a-uri"),
193                createURI("some-uri"));
194
195        assertEquals("abc", testObj.getObjectsAsString(mem,
196                createURI("subject"), createResource("a/b/c"), true));
197        assertEquals("0", testObj.getObjectsAsString(mem, createURI("subject"),
198                createResource("a-numeric-type"), true));
199        assertEquals("<empty>", testObj.getObjectsAsString(mem,
200                createURI("subject"), createResource("an-empty-string"), true));
201        assertEquals("&lt;<a href=\"some-uri\">some-uri</a>&gt;", testObj
202                .getObjectsAsString(mem, createURI("subject"),
203                        createResource("a-uri"), true));
204
205        assertEquals("some-uri", testObj
206                .getObjectsAsString(mem, createURI("subject"),
207                        createResource("a-uri"), false));
208        assertEquals("", testObj.getObjectsAsString(mem, createURI("subject"),
209                createResource("a-nonexistent-uri"), true));
210
211    }

223    @Test
224    public void shouldSortTriplesForDisplay() {
225        final Model model = createDefaultModel();
226
227        model.setNsPrefix("prefix", "namespace");
228        final Property propertyA = model.createProperty("namespace", "a");
229        final Property propertyB = model.createProperty("namespace", "b");
230        final Property propertyC = model.createProperty("c");
231        final Literal literalA = model.createLiteral("a");
232        final Literal literalB = model.createLiteral("b");
233        final Resource resourceB = model.createResource("b");
234        model.add(resourceB, propertyA, literalA);
235
236        final Resource a = model.createResource("a");
237        model.add(a, propertyC, literalA);
238
239        model.add(a, propertyB, literalA);
240
241        model.add(a, propertyA, literalA);
242        model.add(a, propertyA, literalB);
243
244        final Iterator<Quad> iterator =
245            DatasetFactory.create(model).asDatasetGraph().find();
246
247        final List<Quad> sortedTriples =
248            testObj.getSortedTriples(model, iterator);
249
250        sortedTriples.get(0).matches(ANY, a.asNode(), propertyA.asNode(),
251                literalA.asNode());
252        sortedTriples.get(1).matches(ANY, a.asNode(), propertyA.asNode(),
253                literalB.asNode());
254        sortedTriples.get(2).matches(ANY, a.asNode(), propertyB.asNode(),
255                literalA.asNode());
256        sortedTriples.get(3).matches(ANY, a.asNode(), propertyC.asNode(),
257                literalA.asNode());
258        sortedTriples.get(4).matches(ANY, resourceB.asNode(),
259                propertyC.asNode(), literalA.asNode());
260
261    }
