016package org.fcrepo.auth.roles.basic.integration;
017
018import static org.fcrepo.auth.common.ServletContainerAuthenticationProvider.EVERYONE_NAME;
019import java.util.ArrayList;
020import java.util.List;
021
022import org.fcrepo.auth.roles.common.integration.AbstractRolesIT;
023import org.fcrepo.auth.roles.common.integration.RolesFadTestObjectBean;
024
025/**
026 * @author Scott Prater
027 */
028public abstract class AbstractBasicRolesIT extends AbstractRolesIT {
029
030    protected final static List<RolesFadTestObjectBean> test_objs =
031            defineTestObjects();
032
033    private static List<RolesFadTestObjectBean> defineTestObjects() {
034        final List<RolesFadTestObjectBean> test_objs = new ArrayList<>();
035        /* public object with public datastream */
036        final RolesFadTestObjectBean objA = new RolesFadTestObjectBean();
037        objA.setPath("testparent1");
038        objA.addACL(EVERYONE_NAME, "reader");
039        objA.addACL("examplereader", "reader");
040        objA.addACL("examplewriter", "writer");
041        objA.addACL("exampleadmin", "admin");
042        objA.addDatastream("tsp1_data", "Test Parent 1, datastream 1,  Hello!");
043        test_objs.add(objA);
044
045        /* public object with one public datastream, one restricted datastream */
046        final RolesFadTestObjectBean objB = new RolesFadTestObjectBean();
047        objB.setPath("testparent2");
048        objB.addACL(EVERYONE_NAME, "reader");
049        objB.addACL("examplereader", "reader");
050        objB.addACL("examplewriter", "writer");
051        objB.addACL("exampleadmin", "admin");
052        objB.addDatastream("tsp1_data", "Test Parent 2, datastream 1,  Hello!");
053        objB.addDatastream("tsp2_data",
054                "Test Parent 2, datastream 2,  secret stuff");
055        objB.addDatastreamACL("tsp2_data", "examplereader", "reader");
056        objB.addDatastreamACL("tsp2_data", "examplewriter", "writer");
057        objB.addDatastreamACL("tsp2_data", "exampleadmin", "admin");
058        test_objs.add(objB);
059
060        /* public child object with datastream, no ACLs */
061        final RolesFadTestObjectBean objC = new RolesFadTestObjectBean();
062        objC.setPath("testparent1/testchild1NoACL");
063        objC.addDatastream("tsc1_data", "Test Child 1, datastream 1,  Hello!");
064        test_objs.add(objC);
065
066        /* restricted child object with restricted datastreams */
067        final RolesFadTestObjectBean objD = new RolesFadTestObjectBean();
068        objD.setPath("testparent1/testchild2WithACL");
069        objD.addACL("examplereader", "reader");
070        objD.addACL("examplewriter", "writer");
071        objD.addACL("exampleadmin", "admin");
072        objD.addDatastream("tsc1_data",
073                "Test Child 2, datastream 1,  really secret stuff");
074        objD.addDatastream("tsc2_data",
075                "Test Child 2, datastream 2,  really really secret stuff");
076        objD.addDatastreamACL("tsc2_data", "examplewriter", "writer");
077        objD.addDatastreamACL("tsc2_data", "exampleadmin", "admin");
078        test_objs.add(objD);
079
080        /*
081         * even more restricted child object, with even more restricted
082         * datastreams
083         */
084        final RolesFadTestObjectBean objE = new RolesFadTestObjectBean();
085        objE.setPath("testparent1/testchild4WithACL");
086        objE.addACL("examplewriter", "writer");
087        objE.addACL("exampleadmin", "admin");
088        objE.addDatastream("tsc1_data",
089                "Test Child 3, datastream 1,  really secret stuff");
090        objE.addDatastream("tsc2_data",
091                "Test Child 3, datastream 2,  really really secret stuff");
092        objE.addDatastreamACL("tsc2_data", "exampleadmin", "admin");
093        test_objs.add(objE);
094
095        /* private child object with 1 private datastream, 1 public datastream */
096        final RolesFadTestObjectBean objF = new RolesFadTestObjectBean();
097        objF.setPath("testparent2/testchild5WithACL");
098        objF.addACL("exampleadmin", "admin");
099        objF.addDatastream("tsc1_data",
100                "Test Child 5, datastream 1, burn before reading");
101        objF.addDatastream("tsc2_data", "Test Child 5, datastream 2, Hello!");
102        objF.addDatastreamACL("tsc2_data", EVERYONE_NAME, "reader");
103        test_objs.add(objF);
104
105        /* Public object, restricted datastream */
106        final RolesFadTestObjectBean objG = new RolesFadTestObjectBean();
107        objG.setPath("testparent3");
108        objG.addACL(EVERYONE_NAME, "reader");
109        objG.addACL("examplereader", "reader");
110        objG.addACL("examplewriter", "writer");
111        objG.addACL("exampleadmin", "admin");
112        objG.addDatastream("tsp1_data", "Test Parent 3, datastream 1, hello!");
113        objG.addDatastream("tsp2_data",
114                "Test Parent 3, datastream 2, private stuff");
115        objG.addDatastreamACL("tsp2_data", "exampleadmin", "admin");
116        test_objs.add(objG);
117
118        final RolesFadTestObjectBean objH = new RolesFadTestObjectBean();
119        objH.setPath("testparent3/testchild3a");
120        objH.addACL("exampleadmin", "admin");
121        test_objs.add(objH);
122
123        final RolesFadTestObjectBean objI = new RolesFadTestObjectBean();
124        objI.setPath("testparent3/testchild3b");
125        test_objs.add(objI);
126
127        return test_objs;
128
129    }
130}
