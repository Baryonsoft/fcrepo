52  /**
53   * A very simple abstraction to prevent event-driven machinery downstream from the repository from relying directly
54   * on a JCR interface {@link Event}. Can represent either a single JCR event or several.
55   *
56   * @author ajs6f
57   * @since Feb 19, 2013
58   */
59  public class FedoraEventImpl implements FedoraEvent {
60  
61      private final String path;
62      private final String userID;
63      private final String userData;
64      private final long date;
65      private final Map<String, String> info;
66      private final String eventID;
67  
68      private final Set<EventType> eventTypes = new HashSet<>();
69      private final Set<String> eventProperties = new HashSet<>();
70  
71      private static final List<Integer> PROPERTY_TYPES = asList(Event.PROPERTY_ADDED,
72              Event.PROPERTY_CHANGED, Event.PROPERTY_REMOVED);
73  
74      private static final UniqueValueSupplier pidMinter = new DefaultPathMinter();
75  
76      /**
77       * Create a new FedoraEvent
78       * @param type the Fedora EventType
79       * @param path the node path corresponding to this event
80       * @param userID the acting user for this event
81       * @param userData any user data for this event
82       * @param date the timestamp for this event
83       * @param info supplementary information
84       */
85      public FedoraEventImpl(final EventType type, final String path, final String userID,
86              final String userData, final long date, final Map<String, String> info) {
87          this(singleton(type), path, userID, userData, date, info);
88      }
89  
90     /**
91       * Create a new FedoraEvent
92       * @param types a collection of Fedora EventTypes
93       * @param path the node path corresponding to this event
94       * @param userID the acting user for this event
95       * @param userData any user data for this event
96       * @param date the timestamp for this event
97       * @param info supplementary information
98       */
99      public FedoraEventImpl(final Collection<EventType> types, final String path, final String userID,
100             final String userData, final long date, final Map<String, String> info) {
101         requireNonNull(types, "FedoraEvent requires a non-null event type");
102         requireNonNull(path, "FedoraEvent requires a non-null path");
103 
104         this.eventTypes.addAll(types);
105         this.path = path;
106         this.userID = userID;
107         this.userData = userData;
108         this.date = date;
109         this.info = isNull(info) ? emptyMap() : info;
110         this.eventID = pidMinter.get();
111     }
112 
113 
114     /**
115      * @return the event types of the underlying JCR {@link Event}s
116      */
117     @Override
118     public Set<EventType> getTypes() {
119         return eventTypes;
120     }

122     /**
123      * @param type the type
124      * @return this object for continued use
125      */
126     @Override
127     public FedoraEvent addType(final EventType type) {
128         eventTypes.add(type);
129         return this;
130     }

132     /**
133      * @return the property names of the underlying JCR property {@link Event}s
134     **/
135     @Override
136     public Set<String> getProperties() {
137         return eventProperties;
138     }
139

141      * Add a property name to this event
142      * @param property property name
143      * @return this object for continued use
144     **/
145     @Override
146     public FedoraEvent addProperty( final String property ) {
147         eventProperties.add(property);
148         return this;
149     }
150

181     }

183     /**
184      * Get the event ID.
185      * @return Event identifier to use for building event URIs (e.g., in an external triplestore).

192     /**
193      * Return a Map with any additional information about the event.
194      * @return a Map of additional information.
195      */
196     @Override
197     public Map<String, String> getInfo() {
198         return info;
199     }
200 
201     @Override
202     public String toString() {
203 
204         return toStringHelper(this)
205             .add("Event types:", getTypes().stream()
206                             .map(EventType::getName)
207                             .collect(joining(", ")))
208             .add("Event properties:", String.join(",", eventProperties))
209             .add("Path:", getPath())
210             .add("Date: ", getDate()).toString();
211     }
212 
213     private static final Map<Integer, EventType> translation = ImmutableMap.<Integer, EventType>builder()
214             .put(Event.NODE_ADDED, NODE_ADDED)
215             .put(Event.NODE_REMOVED, NODE_REMOVED)
216             .put(Event.PROPERTY_ADDED, PROPERTY_ADDED)
217             .put(Event.PROPERTY_REMOVED, PROPERTY_REMOVED)
218             .put(Event.PROPERTY_CHANGED, PROPERTY_CHANGED)
219             .put(Event.NODE_MOVED, NODE_MOVED)
220             .put(Event.PERSIST, PERSIST).build();
221 
222     /**
223      * Get the Fedora event type for a JCR type
224      *
225      * @param i the integer value of a JCR type
226      * @return EventType
227      */
228     public static EventType valueOf(final Integer i) {
229         final EventType type = translation.get(i);
230         if (isNull(type)) {
231             throw new IllegalArgumentException("Invalid event type: " + i);
232         }
233         return type;
234     }

237     /**
238      * Convert a JCR Event to a FedoraEvent
239      * @param event the JCR Event
240      * @return a FedoraEvent
241      */
242     public static FedoraEvent from(final Event event) {
243         requireNonNull(event);
244         try {
245             final Map<String, String> info = event.getInfo();
246 
247             return new FedoraEventImpl(valueOf(event.getType()), cleanPath(event),
248                     event.getUserID(), event.getUserData(), event.getDate(), info);
249 
250         } catch (final RepositoryException ex) {
251             throw new RepositoryRuntimeException("Error converting JCR Event to FedoraEvent", ex);
252         }
253     }
254 
255     /**
256      * The JCR-based Event::getPath contains some Modeshape artifacts that must be removed or modified in
257      * order to correspond to the public resource path. For example, JCR Events will contain a trailing
258      * /jcr:content for Binaries, a trailing /propName for properties, and /#/ notation for URI fragments.
259      */
260     private static String cleanPath(final Event event) throws RepositoryException {
261         // remove any trailing data for property changes
262         final String path = PROPERTY_TYPES.contains(event.getType()) ?
263             event.getPath().substring(0, event.getPath().lastIndexOf("/")) : event.getPath();
264 
265         // reformat any hash URIs and remove any trailing /jcr:content
266         final HashConverter converter = new HashConverter();
267         return converter.reverse().convert(path.replaceAll("/" + JCR_CONTENT, ""));
268     }
