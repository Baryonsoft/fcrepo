16  package org.fcrepo.kernel.services;
17  
18  import javax.jcr.RepositoryException;
19  import javax.jcr.Session;
20  
21  /**
22   * @author bbpennel
23   * @since Feb 19, 2014
24   */
25  public interface VersionService {
26  
27      /**
28       * Explicitly creates a version for the nodes at each path provided.
29       *
30       * @param session the session in which the node resides
31       * @param absPath absolute paths to the node
32       * @param label a label to be applied to the new version
33       * @throws RepositoryException if repository exception occurred
34       * @return the identifier
35       */
36      String createVersion(Session session, String absPath, String label) throws RepositoryException;
37  
38      /**
39       * Reverts the node to the version identified by the label.  This method
40       * will throw a PathNotFoundException if no version with the given label is
41       * found.
42       *
43       * @param session the session in which the node resides
44       * @param absPath the path to the node whose version is to be reverted
45       * @param label identifies the historic version
46       * @throws RepositoryException if repository exception occurred
47       */
48      void revertToVersion(Session session, String absPath, String label)
49          throws RepositoryException;
50  
51      /**
52       * Remove a version of a node.  This method will throw a PathNotFoundException
53       * if no version with the given label is found.
54       *
55       * @param session the session in which the node resides
56       * @param absPath the path to the node whose version is to be removed
57       * @param label identifies the historic version by label or id
58       * @throws RepositoryException if repository exception occurred
59       */
60      void removeVersion(Session session, String absPath, String label)
61          throws RepositoryException;
62  
63  }
