16  package org.fcrepo.http.commons.responses;
17  
18  import java.io.FilterInputStream;
19  import java.io.IOException;
20  import java.io.InputStream;
21  
22  import org.apache.commons.io.IOUtils;
23  import org.apache.commons.io.input.BoundedInputStream;
24  import org.apache.commons.io.input.ProxyInputStream;
25  
26  /**
27   * An InputStream wrapper that skips N bytes and only returns
28   * the data up to a certain length
29   *
30   * @author awoods
31   */
32  public class RangeRequestInputStream extends FilterInputStream {
33  
34      /**
35       * Creates a <code>FilterInputStream</code>
36       * by assigning the  argument <code>in</code>
37       * to the field <code>this.in</code> so as
38       * to remember it for later use.
39       *
40       * @param in the underlying input stream, or <code>null</code> if
41       *           this instance is to be created without an underlying stream.
42       * @param skip the number of bytes to skip at the beginning of the stream
43       * @param length the number of bytes from the inputstream to read
44       * @throws IOException if IO exception occurred
45       */
46      public RangeRequestInputStream(final InputStream in,
47                                     final long skip,
48                                     final long length) throws IOException {
49          super(new BoundedInputStream(new SkipInputStream(in, skip), length));
50      }
51  
52  
53      /**
54       * An InputStream wrapper that skips bytes
55       * @param in
56       * @param skip
57       * @throws IOException
58       */
59      private static class SkipInputStream extends ProxyInputStream {
60  
61          /**
62           * An InputStream wrapper that always skips the first N bytes
63           * @param in
64           * @param skip
65           * @throws IOException
66           */
67          public SkipInputStream(final InputStream in,
68                                 final long skip) throws IOException {
69              super(in);
70              IOUtils.skip(in, skip);
71          }
72      }
73  }
