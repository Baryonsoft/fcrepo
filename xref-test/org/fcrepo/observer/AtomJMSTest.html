59      @Test
60      public void testAtomStream() throws LoginException, RepositoryException {
61          Session session = repository.login();
62          session.getRootNode().addNode("test1").addMixin("fedora:object");
63          session.save();
64          session.logout();
65          while (entry == null)
66              try {
67                  Thread.sleep(1000);
68              } catch (InterruptedException e) {
69                  e.printStackTrace();
70              }
71          List<Category> categories = entry.getCategories("xsd:string");
72          String path = null;
73          for (Category cat : categories) {
74              if (cat.getLabel().equals("fedora-types:pid")) {
75                  logger.debug("Found Category with term: " + cat.getTerm());
76                  path = cat.getTerm();
77              }
78          }
79          assertEquals("Got wrong pid!", "test1", path);
80          assertEquals("Got wrong method!", "ingest", entry.getTitle());
81      }

83      @Override
84      public void onMessage(Message message) {
85          logger.debug("Received JMS message: " + message.toString());
86          TextMessage tMessage = (TextMessage) message;
87          try {
88              entry =
89                      (Entry) parser.parse(
90                              new ByteArrayInputStream(tMessage.getText()
91                                      .getBytes("UTF-8"))).getRoot();
92              logger.debug("Parsed Entry: " + entry.toString());
93          } catch (Exception e) {
94              e.printStackTrace();
95              fail(e.getMessage());
96          }

98      }
